static const char *latm_dmx_probe_data(const u8 *data, u32 size, GF_FilterProbeScore *score)
{
	u32 nb_frames=0;
	u32 nb_skip=0;
	GF_M4ADecSpecInfo acfg;
	GF_BitStream *bs = gf_bs_new(data, size, GF_BITSTREAM_READ);
	while (1) {
		// Attempt to synchronize and decode LATM frames from the bitstream using the latm_dmx_sync_frame_bs function.
		// For each successful synchronization and decoding, check if the sample rate index is valid.
		// If an invalid sample rate index is encountered, reset the frame count and exit the loop.
		// Track the number of skipped frames; if more than one frame is skipped, reset the frame count and exit the loop.
		// Increment the frame count for each successfully processed frame.
		// This process continues until an error is detected or the end of the bitstream is reached.
		// <MASK>
	}
	gf_bs_del(bs);
	if (nb_frames>=2) {
		*score = nb_skip ? GF_FPROBE_MAYBE_NOT_SUPPORTED : GF_FPROBE_SUPPORTED;
		return "audio/aac+latm";
	}
	return NULL;
}
