static int
json_parse_string(const unsigned char **ucp, const unsigned char *ue)
{
	const unsigned char *uc = *ucp;
	size_t i;

	DPRINTF("Parse string: ", uc, *ucp);
	while (uc < ue) {
		switch (*uc++) {
		case '\0':
			goto out;
		case '\\':
			// Handle escape sequences within the JSON string being parsed. This involves
			// advancing the pointer to the next character and checking for valid escape
			// characters or sequences. If a valid escape sequence is detected, processing
			// continues. If an invalid sequence is encountered, the parsing is aborted
			// and control is transferred to the error handling section.
			// <MASK>
		case '"':
			*ucp = uc;
			return 1;
		default:
			continue;
		}
	}
out:
	DPRINTF("Bad string: ", uc, *ucp);
	*ucp = uc;
	return 0;
}
