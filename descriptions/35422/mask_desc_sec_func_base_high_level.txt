MagickExport void ConcatenateStringInfo(StringInfo *string_info,
  const StringInfo *source)
{
  size_t
    length;

  assert(string_info != (StringInfo *) NULL);
  assert(string_info->signature == MagickCoreSignature);
  assert(source != (const StringInfo *) NULL);
  length=string_info->length;
  if (~length < source->length)
    ThrowFatalException(ResourceLimitFatalError,"UnableToConcatenateString");
  length+=source->length;
  if (~length < MagickPathExtent)
    ThrowFatalException(ResourceLimitFatalError,"MemoryAllocationFailed");
  // Allocate or resize memory for the concatenated string data to accommodate
  // the source string's content. If the memory allocation fails, throw a fatal
  // exception. Copy the source string's data to the end of the current string
  // data in the destination structure, and update the length of the destination
  // string accordingly.
  // <MASK>
}
