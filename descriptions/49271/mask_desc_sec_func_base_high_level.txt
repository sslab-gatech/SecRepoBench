static void copy_CTB(uint8_t *dst, const uint8_t *src, int width, int height,
                     ptrdiff_t stride_dst, ptrdiff_t stride_src)
{
    int i, j;

    if (((intptr_t)dst | (intptr_t)src | stride_dst | stride_src) & 15) {
        // Copy a rectangular block of pixels from the source buffer to the destination buffer.
        // The dimensions of the block are specified by 'width' and 'height', and the memory
        // layout is determined by 'stride_src' and 'stride_dst'. This implementation is used
        // when the memory addresses or strides are not aligned to a 16-byte boundary.
        // The function iterates over each row and copies segments of 8 bytes at a time.
        // If there are any remaining bytes at the end of a row that do not form a complete
        // 8-byte segment, they are copied individually.
        // <MASK>
    } else {
        for (i = 0; i < height; i++) {
            for (j = 0; j < width; j+=16)
                AV_COPY128(dst+j, src+j);
            dst += stride_dst;
            src += stride_src;
        }
    }
}
