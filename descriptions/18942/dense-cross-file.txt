  {
    yr_object_destroy(*obj_ptr);

    obj_ptr = (YR_OBJECT**) yr_arena_next_address(
        obj_arena, obj_ptr, sizeof(YR_OBJECT*));
  }

  yr_arena_destroy(obj_arena);
  yr_arena_destroy(it_arena);
  yr_modules_unload_all(context);
  yr_free(stack.items);

  return result;
}
