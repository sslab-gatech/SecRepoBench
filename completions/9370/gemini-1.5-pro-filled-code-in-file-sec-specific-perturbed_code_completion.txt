case TIFF_SHORT:
      {
        uint16
          shorty;

        shorty=0;
        if (TIFFGetField(tiff,exif_info[i].tag,&shorty,sans) == 1)
          {
            if (exif_info[i].variable_length > 0)
              {
                size_t num=(exif_info[i].variable_length < MagickPathExtent) ? exif_info[i].variable_length : MagickPathExtent-1;
                if (num > 1) /* If the EXIF tag has multiple values of type TIFF_SHORT, format them as a comma-separated string. */
                  {
                    ssize_t j;
                    uint16 *fields;
                    char tmp[32];
                    *exif_value_str='\0';
                    if (TIFFGetField(tiff,exif_info[i].tag,&fields,sans) == 1)
                      {
                        for (j=0; j < num; j++)
                          {
                            (void) FormatLocaleString(tmp,sizeof(tmp),"%u",fields[j]);
                            (void) ConcatenateMagickString(exif_value_str,tmp,MagickPathExtent);
                            if (j < num-1)
                              (void) ConcatenateMagickString(exif_value_str,",",MagickPathExtent);
                          }
                      }
                  }
                else
                  (void) FormatLocaleString(exif_value_str,MagickPathExtent,"%u",shorty);
              }
            else
              (void) FormatLocaleString(exif_value_str,MagickPathExtent,"%u",shorty);
          }
        break;
