// <MASK>
    if (phantom_only) {
      return true; // Skip reading actual points if phantom_only is true
    }

    // Read x & y coordinates
    return read_points(p, contour_points, end, &contour_point_t::x, FLAG_X_SHORT, FLAG_X_SAME)
        && read_points(p, contour_points, end, &contour_point_t::y, FLAG_Y_SHORT, FLAG_Y_SAME);
