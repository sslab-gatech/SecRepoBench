size_t num_contours = endPtsOfContours.size() - 1; // Initialize num_contours
    contour_points.clear();
    contour_points.reserve (num_contours + phantom_points_.size());

    for (size_t i = 0; i < num_contours; ++i)
      {
	contour_points.emplace_back (endPtsOfContours[i], endPtsOfContours[i + 1]);
      }

    for (const auto &phantom_point : phantom_points_)
      {
	contour_points.push_back (phantom_point);
      }

    if (phantom_only) return true;
