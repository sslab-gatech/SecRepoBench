  {
    contour_points.resize (4);
    return phantom_only;
  }

  for (unsigned i = 1; i < num_contours; i++)
    if (unlikely (endPtsOfContours[i] <= endPtsOfContours[i-1]))
      return false;

  contour_points.resize (num_points + 4);

  if (phantom_only)
    return true;
