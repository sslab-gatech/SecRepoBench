    i = 0;
  }
  
  // Store the converted value
  *dest = (unsigned short)i;
  
  // Issue warning if the flag is 0
  if (*dest == 0)
    HUNSPELL_WARNING(stderr, "error: line %d: 0 is wrong flag id\n",
                     af->getlinenum());

  dest++;
