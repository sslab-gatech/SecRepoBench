{
          size_t token_start = src - flags.c_str();
          size_t token_length = index - token_start;
          std::string token = flags.substr(token_start, token_length);
          int i = atoi(token.c_str());
          if (i >= DEFAULTFLAGS || i < 0) {
            HUNSPELL_WARNING(stderr,
                             "error: line %d: flag id %d is too large (max: %d)\n",
                             af->getlinenum(), i, DEFAULTFLAGS - 1);
            i = 0;
          }
          *dest = (unsigned short)i;
          if (*dest == 0)
            HUNSPELL_WARNING(stderr, "error: line %d: 0 is wrong flag id\n",
                             af->getlinenum());
          dest++;
          src = index + 1;  // Move past the comma for next iteration
