{
  // Parse the segment as an integer
  int i = atoi(src);
  
  // Check if the number is within valid range
  if (i >= DEFAULTFLAGS || i < 0) {
    if (af != NULL)
      HUNSPELL_WARNING(stderr,
                     "error: line %d: flag id %d is too large (max: %d)\n",
                     af->getlinenum(), i, DEFAULTFLAGS - 1);
    i = 0;  // Set to 0 if out of range
  }
  
  // Store the parsed number
  *dest = (unsigned short)i;
  
  // Warn if the number is 0
  if (*dest == 0 && af != NULL)
    HUNSPELL_WARNING(stderr, "error: line %d: 0 is wrong flag id\n",
                     af->getlinenum());
  
  // Move pointers for next iteration
  src = flags.c_str() + index + 1;  // Point to character after comma
  dest++;  // Move to next position in result array
}
