} else {
          std::string number_str(src, flags.begin() + index);
          int parsed_number = atoi(number_str.c_str());
          if (parsed_number >= DEFAULTFLAGS || parsed_number < 0) {
            HUNSPELL_WARNING(stderr,
                             "error: line %d: flag id %d is too large (max: %d)\n",
                             af->getlinenum(), parsed_number, DEFAULTFLAGS - 1);
            parsed_number = 0;
          }
          *dest = (unsigned short)parsed_number;
          if (*dest == 0) {
            HUNSPELL_WARNING(stderr, "error: line %d: 0 is wrong flag id\n",
                             af->getlinenum());
          }
          dest++;
          src = flags.begin() + index + 1; // Move past the comma
        }
