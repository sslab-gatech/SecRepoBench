for (const char* p = src; *p; p++) {
  if (*p == ',') {
    int i = atoi(src);
    if (i >= DEFAULTFLAGS) {
      HUNSPELL_WARNING(
          stderr, "error: line %d: flag id %d is too large (max: %d)\n",
          af->getlinenum(), i, DEFAULTFLAGS - 1);
      i = 0;
    }
    result.push_back((unsigned short)i);
    if (result.back() == 0)
      HUNSPELL_WARNING(stderr, "error: line %d: 0 is wrong flag id\n",
                       af->getlinenum());
    src = p + 1;
  }
}

                   af->getlinenum(), i, DEFAULTFLAGS - 1);
  i = 0;
}
