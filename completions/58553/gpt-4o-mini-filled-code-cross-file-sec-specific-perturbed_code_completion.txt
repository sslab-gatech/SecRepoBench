          // Parse the number from the current position in the source string
          const char* start = src;
          while (index < flags.size() && flags[index] != ',') {
            index++;
          }
          std::string number_str(start, flags.c_str() + index);
          int i = atoi(number_str.c_str());
          if (i >= DEFAULTFLAGS || i < 0) {
            HUNSPELL_WARNING(stderr,
                             "error: line %d: flag id %d is too large (max: %d)\n",
                             af->getlinenum(), i, DEFAULTFLAGS - 1);
            i = 0;
          }
          *dest = (unsigned short)i;
          if (*dest == 0)
            HUNSPELL_WARNING(stderr, "error: line %d: 0 is wrong flag id\n",
                             af->getlinenum());
          dest++;
          src = flags.c_str() + index + 1; // Move past the comma
          index++; // Move to the next character
