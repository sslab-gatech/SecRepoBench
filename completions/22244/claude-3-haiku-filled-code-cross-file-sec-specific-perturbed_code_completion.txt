// <MASK>
if(packet->payload_packet_len >= 8) {
    if(packet->tcp != NULL) {
        ovpn_payload = packet->payload + 4;
    }

    operationcode = ovpn_payload[0];
    alen = ovpn_payload[1];

    if(packet->udp != NULL) {
        if(flow->l4.udp.openvpn_counter == 0 && operationcode == 0x38 && alen == 0x10) {
            NDPI_LOG_INFO(ndpi_struct, "found openvpn over udp");
            ndpi_set_detected_protocol(ndpi_struct, flow, NDPI_PROTOCOL_OPENVPN, NDPI_PROTOCOL_UNKNOWN);
        } else if(flow->l4.udp.openvpn_counter < 5 && operationcode == 0x17) {
            session_remote = &ovpn_payload[2];
            memcpy(flow->l4.udp.openvpn_session, session_remote, alen);
            flow->l4.udp.openvpn_session[alen] = '\0';
        } else if(flow->l4.udp.openvpn_counter >= 5 && flow->l4.udp.openvpn_counter < 10 && operationcode == 0x18) {
            hmac_size = ovpn_payload[2];
            if(memcmp(&ovpn_payload[3], flow->l4.udp.openvpn_session, alen) == 0) {
                NDPI_LOG_INFO(ndpi_struct, "found openvpn over udp");
                ndpi_set_detected_protocol(ndpi_struct, flow, NDPI_PROTOCOL_OPENVPN, NDPI_PROTOCOL_UNKNOWN);
            } else {
                failed = 1;
            }
        }
        flow->l4.udp.openvpn_counter++;
    }

    if(failed) {
        NDPI_ADD_PROTOCOL_TO_BITMASK(flow->excluded_protocol_bitmask, NDPI_PROTOCOL_OPENVPN);
    }
}
