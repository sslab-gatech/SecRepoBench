if (packet->payload_packet_len >= 14) {
  if (packet->tcp != NULL)
    ovpn_payload = &packet->payload[1];

  operationcode = ovpn_payload[0] >> 3;

  if (packet->udp != NULL) {
    if ((packet->payload_packet_len == 14) && (flow->packet_counter == 1) && (operationcode == 7)) {
      NDPI_LOG_INFO(ndpi_struct, "found openvpn\n");
      ndpi_set_detected_protocol(ndpi_struct, flow, NDPI_PROTOCOL_OPENVPN, NDPI_PROTOCOL_UNKNOWN);
      return;
    }

    if ((flow->openvpn_counter < 5) && (operationcode == 1)) {
      alen = ovpn_payload[1];
      if ((alen > 0) && ((alen * 4) <= (packet->payload_packet_len - 2))) {
        session_remote = &ovpn_payload[2 + (alen * 4)];
        memcpy(flow->openvpn_session_id, session_remote, 8);
      }
    }

    if ((flow->openvpn_counter > 3) && (flow->openvpn_counter < 10) && (operationcode == 8)) {
      if (memcmp(flow->openvpn_session_id, &ovpn_payload[1], 8) == 0) {
        NDPI_LOG_INFO(ndpi_struct, "found openvpn\n");
        ndpi_set_detected_protocol(ndpi_struct, flow, NDPI_PROTOCOL_OPENVPN, NDPI_PROTOCOL_UNKNOWN);
        return;
      } else {
        failed = 1;
      }
    }
  }

  flow->openvpn_counter++;
}

if (failed)
  NDPI_ADD_PROTOCOL_TO_BITMASK(flow->excluded_protocol_bitmask, NDPI_PROTOCOL_OPENVPN);
