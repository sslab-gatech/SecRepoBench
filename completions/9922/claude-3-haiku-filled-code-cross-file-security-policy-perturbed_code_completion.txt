case '\\':
    uc++;
    if (uc >= endptr) {
        DPRINTF("Bad string: ", uc, *ucp);
        *ucp = uc;
        return 0;
    }
    switch (*uc++) {
    case '\\':
    case '"':
    case '/':
        break;
    case 'b':
        *uc++ = '\b';
        break;
    case 'f':
        *uc++ = '\f';
        break;
    case 'n':
        *uc++ = '\n';
        break;
    case 'r':
        *uc++ = '\r';
        break;
    case 't':
        *uc++ = '\t';
        break;
    case 'u':
        if (uc + 4 >= endptr) {
            DPRINTF("Bad string: ", uc, *ucp);
            *ucp = uc;
            return 0;
        }
        // Parse the 4 hexadecimal digits following \u
        // and add the corresponding Unicode character to the buffer
        break;
    default:
        DPRINTF("Bad string: ", uc, *ucp);
        *ucp = uc;
        return 0;
    }
    break;
