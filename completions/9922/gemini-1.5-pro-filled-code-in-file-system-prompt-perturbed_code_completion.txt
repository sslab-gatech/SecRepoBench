if (uc == endptr)
				goto out;
			switch (*uc++) {
			case '\\': case '"': case '/': case 'b': case 'f':
			case 'n': case 'r': case 't':
				continue;
			case 'u':
				for (i = 0; i < 4 && uc < endptr; i++) {
					if (!json_isxdigit(*uc++))
						goto out;
				}
				if (i != 4)
					goto out;
				continue;
			default:
				goto out;
			}
