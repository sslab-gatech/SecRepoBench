if (uc == endptr) {
				goto out; // Unexpected end of data
			}
			switch (*uc++) {
			case '"':
			case '\\':
			case '/':
			case 'b':
			case 'f':
			case 'n':
			case 'r':
			case 't':
				// Valid escape sequences, continue parsing
				break;
			case 'u': {
				// Unicode escape sequence, ensure 4 hexadecimal digits follow
				if (endptr - uc < 4) {
					goto out; // Not enough data for a valid Unicode escape
				}
				for (i = 0; i < 4; i++) {
					if (!isxdigit(uc[i])) {
						goto out; // Invalid hexadecimal digit
					}
				}
				uc += 4; // Advance past the 4 hexadecimal digits
				break;
			}
			default:
				goto out; // Invalid escape sequence
			}
