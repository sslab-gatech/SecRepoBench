uc++; // Move past the escape character
			if (uc >= endptr) {
				DPRINTF("Unexpected end of string: ", uc, *ucp);
				goto out; // Exit if we reach the end
			}
			switch (*uc++) {
				case '\\': // Handle backslash
				case '\"': // Handle double quote
				case '/':  // Handle forward slash
				case 'b':  // Handle backspace
				case 'f':  // Handle form feed
				case 'n':  // Handle newline
				case 'r':  // Handle carriage return
				case 't':  // Handle tab
					break; // Valid escape sequence
				case 'u': { // Handle unicode escape
					for (i = 0; i < 4; i++) {
						if (uc >= endptr || !json_isxdigit(*uc)) {
							DPRINTF("Invalid unicode escape: ", uc, *ucp);
							goto out; // Exit if invalid
						}
						uc++;
					}
					break;
				}
				default:
					DPRINTF("Invalid escape sequence: ", uc, *ucp);
					goto out; // Exit if invalid
			}
