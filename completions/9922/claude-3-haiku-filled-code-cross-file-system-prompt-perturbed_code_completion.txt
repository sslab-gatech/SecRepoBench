switch (*uc++) {
case '\\':
    switch (*uc++) {
    case '\\':
    case '"':
    case '/':
        break;
    case 'b':
        *uc++ = '\b';
        break;
    case 'f':
        *uc++ = '\f';
        break;
    case 'n':
        *uc++ = '\n';
        break;
    case 'r':
        *uc++ = '\r';
        break;
    case 't':
        *uc++ = '\t';
        break;
    case 'u':
        // Parse a 4-digit hexadecimal Unicode code point
        if (uc + 4 > endptr) {
            DPRINTF("Bad string: ", uc, *ucp);
            return 0;
        }
        unsigned int codepoint = 0;
        for (i = 0; i < 4; i++) {
            int digit = json_ishex(*uc);
            if (digit < 0) {
                DPRINTF("Bad string: ", uc, *ucp);
                return 0;
            }
            codepoint = (codepoint << 4) + digit;
            uc++;
        }
        // TODO: Convert the Unicode code point to a UTF-8 sequence and append it to the output
        break;
    default:
        DPRINTF("Bad string: ", uc, *ucp);
        return 0;
    }
    break;
default:
    uc--;
    break;
}
