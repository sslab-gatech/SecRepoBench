switch (*uc++) {
	case '\\':
		uc++;
		switch (*uc++) {
		case 'u':
			uc += 4; /* skip 4 hexadecimal digits */
			continue;
		case '\\': case '"': case '/': case 'b': case 'f': case 'n': case 'r': case 't':
			continue;
		default:
			DPRINTF("Bad escape: ", uc, *ucp);
			return 0;
		}
	default:
		continue;
	}
