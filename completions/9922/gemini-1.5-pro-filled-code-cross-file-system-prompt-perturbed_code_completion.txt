if (uc == endptr)
				goto out;
			switch (*uc) {
			case '\\':
			case '"':
			case '/':
				uc++;
				break;
			case 'b':
				uc++;
				break;
			case 'f':
				uc++;
				break;
			case 'n':
				uc++;
				break;
			case 'r':
				uc++;
				break;
			case 't':
				uc++;
				break;
			case 'u':
				for (i = 0; i < 4; i++) {
					if (++uc == endptr)
						goto out;
					if (json_ishex(*uc))
						continue;
					goto out;
				}
				uc++;
				break;

			default:
				goto out;
			}
			break;
