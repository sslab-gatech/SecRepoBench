{
    if (uc >= endptr) {
        goto out;
    }
    switch (*uc++) {
    case '"':
    case '\\':
    case '/':
    case 'b':
    case 'f':
    case 'n':
    case 'r':
    case 't':
        break;
    case 'u':
        for (i = 0; i < 4; i++) {
            if (uc >= endptr || !isxdigit(*uc++)) {
                goto out;
            }
        }
        break;
    default:
        goto out;
    }
    continue;
}
