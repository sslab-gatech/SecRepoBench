			case '\\':
			case '"':
			case '/':
				break;
			case 'b':
				*ucp = uc;

			case 'f':
				*ucp = uc;
				return 1;
			case 'n':
				*ucp = uc;
				return 1;
			case 'r':
				*ucp = uc;
				return 1;
			case 't':
				*ucp = uc;
				return 1;
			case 'u':
				for (i = 0; i < 4; i++) {
					if (uc == endptr || !isxdigit(*uc))
						goto out;
					uc++;
				}
				break;
