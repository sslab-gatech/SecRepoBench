switch (*uc++) {
			case '\\':
				if (uc < endptr) {
					switch (*uc++) {
					case '"': case '\\': case '/':
					case 'b': case 'f': case 'n':
					case 'r': case 't':
						break;
					case 'u':
						for (i = 0; i < 4; i++) {
							if (uc < endptr && json_isxdigit(*uc)) {
								uc++;
							} else {
								DPRINTF("Bad unicode escape: ", uc, *ucp);
								goto out;
							}
						}
						break;
					default:
						DPRINTF("Bad escape: ", uc, *ucp);
						goto out;
					}
				} else {
					DPRINTF("Unexpected end of string: ", uc, *ucp);
					goto out;
				}
				break;
			default:
				continue;
			}
