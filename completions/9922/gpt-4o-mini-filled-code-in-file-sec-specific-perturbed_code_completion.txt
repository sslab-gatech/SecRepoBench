			case 'n': case 'r': case 't':

				uc++;

						DPRINTF("Invalid unicode escape: ", uc, *ucp);
						goto out;
					}
					uc++;
				}
				break;

				DPRINTF("Unknown escape sequence: ", uc, *ucp);
