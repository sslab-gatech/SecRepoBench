if (uc == endptr)
				goto out;
			switch (*uc) {
			case '\\':
			case '"':
			case '/':
				uc++;
				break;
			case 'b':
				uc++;
				break;
			case 'f':
				uc++;
				break;
			case 'n':
				uc++;
				break;
			case 'r':
				uc++;
				break;
			case 't':
				uc++;
				break;
			case 'u':
				if (endptr - uc < 5) {
					uc = endptr;
					goto out;
				}
				for (i = 0; i < 4; i++) {
					if (!json_ishex(uc[i+1])) {
						uc += i + 1;
						goto out;
					}
				}
				uc += 5;
				break;

			default:
				goto out;
			}
