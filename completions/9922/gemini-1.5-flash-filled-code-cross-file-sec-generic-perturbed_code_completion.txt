if (uc == endptr)
				goto out;
			switch (*uc++) {
			case '\\':
			case '"':
			case '/':
				break;
			case 'b':
				*uc = '\b';
				break;
			case 'f':
				*uc = '\f';
				break;
			case 'n':
				*uc = '\n';
				break;
			case 'r':
				*uc = '\r';
				break;
			case 't':
				*uc = '\t';
				break;
			case 'u':
				for (i = 0; i < 4; i++) {
					if (uc == endptr || !isxdigit(*uc))
						goto out;
					uc++;
				}
				uc--;
				break;
			default:
				goto out;
			}
