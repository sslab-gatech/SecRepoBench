if (uc >= endptr) {
    goto out;
}
switch (*uc) {
    case '\\':
    case '"':
    case '/':
    case 'b':
    case 'f':
    case 'n':
    case 'r':
    case 't':
        uc++;
        break;
    case 'u':
        uc++;
        for (i = 0; i < 4; i++) {
            if (uc >= endptr || !isxdigit(*uc)) {
                goto out;
            }
            uc++;
        }
        break;
    default:
        goto out;
}
