if (uc >= endptr)
				goto out;
			switch (*uc++) {
			case '\\':
				break;
			case '"':
				break;
			case '/':
				break;
			case 'b':
				break;
			case 'f':
				break;
			case 'n':
				break;
			case 'r':
				break;
			case 't':
				break;
			case 'u':
				/* Validate the presence of four hex characters. */
				for (i = 0; i < 4; i++) {
					if (uc >= endptr)
						goto out;
					if (!json_isxdigit(*uc++))
						goto out;
				}
				uc--; /* Step back to allow the loop to increment it. */
				break;
			default:
				goto out;
			}
			break;
