{
    if (uc >= endptr)
        goto out;
    switch (*uc++) {
    case '"':
    case '\\':
    case '/':
    case 'b':
    case 'f':
    case 'n':
    case 'r':
    case 't':
        /* Valid single-character escape */
        break;
    case 'u':
        /* Unicode escape: expecting 4 hex digits */
        if (uc + 4 > endptr)
            goto out;
        for (i = 0; i < 4; i++) {
            if (!json_isxdigit(uc[i]))
                goto out;
        }
        uc += 4;
        break;
    default:
        /* Invalid escape sequence */
        goto out;
    }
}
