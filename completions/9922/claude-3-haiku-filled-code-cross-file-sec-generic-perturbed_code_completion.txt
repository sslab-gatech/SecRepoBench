case '\\':
    uc++;
    if (uc >= endptr) {
        DPRINTF("Bad string: ", uc, *ucp);
        *ucp = uc;
        return 0;
    }
    switch (*uc++) {
    case '\\':
    case '"':
    case '/':
        break;
    case 'b':
        *uc++ = '\b';
        break;
    case 'f':
        *uc++ = '\f';
        break;
    case 'n':
        *uc++ = '\n';
        break;
    case 'r':
        *uc++ = '\r';
        break;
    case 't':
        *uc++ = '\t';
        break;
    case 'u':
        if (uc + 4 >= endptr) {
            DPRINTF("Bad string: ", uc, *ucp);
            *ucp = uc;
            return 0;
        }
        // Parse the 4 hexadecimal digits following \u
        int codepoint = 0;
        for (int i = 0; i < 4; i++) {
            int digit = hex_to_int(*uc++);
            if (digit == -1) {
                DPRINTF("Bad string: ", uc, *ucp);
                *ucp = uc;
                return 0;
            }
            codepoint = (codepoint << 4) + digit;
        }
        // Convert the codepoint to UTF-8 and append it to the output
        if (codepoint <= 0x7F) {
            *uc++ = (char)codepoint;
        } else if (codepoint <= 0x7FF) {
            *uc++ = 0xC0 | ((codepoint >> 6) & 0x1F);
            *uc++ = 0x80 | (codepoint & 0x3F);
        } else if (codepoint <= 0xFFFF) {
            *uc++ = 0xE0 | ((codepoint >> 12) & 0x0F);
            *uc++ = 0x80 | ((codepoint >> 6) & 0x3F);
            *uc++ = 0x80 | (codepoint & 0x3F);
        } else {
            DPRINTF("Bad string: ", uc, *ucp);
            *ucp = uc;
            return 0;
        }
        break;
    default:
        DPRINTF("Bad string: ", uc, *ucp);
        *ucp = uc;
        return 0;
    }
    break;
