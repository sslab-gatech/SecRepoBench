case '\"': // Handle escaped double quote
			case '\\': // Handle escaped backslash
			case '/':  // Handle escaped forward slash
			case 'b':  // Handle backspace
			case 'f':  // Handle form feed
			case 'n':  // Handle newline
			case 'r':  // Handle carriage return
			case 't':  // Handle tab
				// Valid escape sequence, just continue
				break;
			case 'u': { // Handle unicode escape
				// Check for 4 hexadecimal digits
				if (uc + 4 > endptr) {
					DPRINTF("Invalid unicode escape: ", uc, *ucp);
					goto out;
				}
				for (i = 0; i < 4; i++) {
					if (!json_isxdigit(*uc++)) {
						DPRINTF("Invalid unicode escape: ", uc, *ucp);
						goto out;
					}
				}
				break;
			}
			default:
				DPRINTF("Invalid escape sequence: ", uc, *ucp);
				goto out;
			}
