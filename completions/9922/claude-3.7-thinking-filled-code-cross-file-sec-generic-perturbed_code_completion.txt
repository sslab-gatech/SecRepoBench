if (uc >= endptr) {
    goto out;  // Unexpected end of data
}

switch (*uc) {
case '\\':
case '"':
case '/':
    uc++;  // Skip these valid escape characters
    break;
case 'b':
case 'f':
case 'n':
case 'r':
case 't':
    uc++;  // Skip these valid escape characters
    break;
case 'u':
    uc++;  // Skip 'u'
    // Ensure we have 4 hex digits
    if (uc + 4 > endptr) {
        goto out;  // Not enough characters for \uXXXX
    }
    // Verify all 4 characters are hex digits
    for (i = 0; i < 4; i++) {
        char c = uc[i];
        if (!((c >= '0' && c <= '9') || 
              (c >= 'a' && c <= 'f') || 
              (c >= 'A' && c <= 'F'))) {
            goto out;  // Not a hex digit
        }
    }
    uc += 4;  // Skip the 4 hex digits
    break;
default:
    goto out;  // Invalid escape sequence
}
