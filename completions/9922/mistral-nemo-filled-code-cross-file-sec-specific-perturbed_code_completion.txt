case '\\':
	uc++;
	if (uc >= endptr)
		goto out;
	switch (*uc++) {
	case 'u':
		if (uc + 4 >= endptr)
			goto out;
		if (!isxdigit(uc[0]) || !isxdigit(uc[1]) || !isxdigit(uc[2]) || !isxdigit(uc[3]))
			goto out;
		uc += 4;
		break;
	case '\\':
	case '"':
	case '/':
	case 'b':
	case 'f':
	case 'n':
	case 'r':
	case 't':
		break;
	default:
		goto out;
	}
	break;
