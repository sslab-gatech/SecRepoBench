if (uc == endptr)
				goto out;
			switch (*uc++) {
			case '\\':
				break;
			case '"':
				break;
			case '/':
				break;
			case 'b':
				break;
			case 'f':
				break;
			case 'n':
				break;
			case 'r':
				break;
			case 't':
				break;
			case 'u':
				/* Validate \uXXXX sequence (4 hex digits) */
				for (i = 0; i < 4; i++, uc++) {
					if (uc == endptr)
						goto out;
					if (!json_isxdigit(*uc))
						goto out;
				}
				uc--; /* Step back to last hex digit */
				break;
			default:
				goto out;
			}
			break;
