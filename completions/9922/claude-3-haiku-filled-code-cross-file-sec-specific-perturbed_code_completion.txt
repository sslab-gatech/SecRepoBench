case '\\':
    uc++;
    if (uc >= endptr) {
        DPRINTF("Unexpected end of data", uc, *ucp);
        return 0;
    }
    switch (*uc++) {
    case '\\':
    case '"':
    case '/':
    case 'b':
    case 'f':
    case 'n':
    case 'r':
    case 't':
        break;
    case 'u':
        if (uc + 4 >= endptr) {
            DPRINTF("Unexpected end of data", uc, *ucp);
            return 0;
        }
        for (i = 0; i < 4; i++) {
            if (!json_isxdigit(*uc++)) {
                DPRINTF("Invalid escape sequence", uc, *ucp);
                return 0;
            }
        }
        break;
    default:
        DPRINTF("Invalid escape sequence", uc, *ucp);
        return 0;
    }
    break;
