// Handle escape sequences inside a JSON string. Advance the pointer `uc` for
// valid escape sequences: `\\`, `\"`, `\/`, `\b`, `\f`, `\n`, `\r`, `\t`, and
// `\u` followed by 4 hexadecimal digits. If an invalid escape sequence or 
// unexpected end of data is encountered, exit the parsing loop.
switch (*uc++) {
case '\\':
    if (uc >= endptr) {
        return 0; // Unexpected end of data
    }
    switch (*uc++) {
    case '\\':
    case '"':
    case '/':
    case 'b':
    case 'f':
    case 'n':
    case 'r':
    case 't':
        break;
    case 'u':
        for (int i = 0; i < 4; i++) {
            if (uc >= endptr || !json_isxdigit(*uc)) {
                return 0; // Invalid Unicode escape sequence
            }
            uc++;
        }
        break;
    default:
        return 0; // Invalid escape sequence
    }
    break;
default:
    break;
}
