if (uc == endptr)
				goto out;
			switch (*uc++) {
			case '\\':
				if (uc == endptr)
					goto out;
				switch (*uc++) {
				case '\\':
				case '"':
				case '/':
					break;
				case 'b':
					*uc = '\b';
					break;
				case 'f':
					*uc = '\f';
					break;
				case 'n':
					*uc = '\n';
					break;
				case 'r':
					*uc = '\r';
					break;
				case 't':
					*uc = '\t';
					break;
				case 'u':
					i = 0;
					for (; i < 4 && uc < endptr; uc++) {
						if (!json_isxdigit(*uc))
							goto out;
						i++;
					}
					if (i != 4)
						goto out;
					break;
				default:
					goto out;
				}
				break;
			default:
				continue;
			}
