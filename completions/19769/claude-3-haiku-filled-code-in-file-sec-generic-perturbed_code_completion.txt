MagickPrivate MagickBooleanType IsFxFunction(const char *expressionstr,
  const char *name,const size_t length)
{
  // Check if the initial portion of the expression matches the function name
  if (LocaleNCompare(expressionstr, name, length) != 0)
    return MagickFalse;

  // Check if the character following the matched portion is either a space or an opening parenthesis
  if (expressionstr[length] == ' ' || expressionstr[length] == '(')
    return MagickTrue;

  return MagickFalse;
}
