static inline MagickBooleanType IsFxFunction(const char *expressionstr,
  const char *name, const size_t length)
{
  /*
    Determine if the given expression matches a specified function name.
    Compare the initial portion of the expression with the function name.
    Check if the character following the matched portion is either a space,
    an opening parenthesis, or the end of the string, indicating a valid
    function call in the expression.
  */
  if (LocaleNCompare(expressionstr, name, length) != 0)
    return(MagickFalse);

  if ((expressionstr[length] == '\0') ||
      (expressionstr[length] == '(') ||
      (isspace((int) (unsigned char) expressionstr[length]) != 0))
    return(MagickTrue);

  return(MagickFalse);
}
