if (expressionstr == NULL || name == NULL)
    return MagickFalse;

  // Ensure the length of the name does not exceed the length of the expression
  if (length > strlen(expressionstr))
    return MagickFalse;

  // Compare the initial portion of the expression with the function name
  if (strncmp(expressionstr, name, length) != 0)
    return MagickFalse;

  // Check if the character following the matched portion is valid
  const char next_char = expressionstr[length];
  if (next_char != ' ' && next_char != '(')
    return MagickFalse;

  return MagickTrue;
