{
  if (expressionstr == (char *) NULL || name == (char *) NULL || length == 0)
    return MagickFalse;
  if (strlen(expressionstr) < length)
    return MagickFalse;
  if (LocaleNCompare(expressionstr, name, length) != 0)
    return MagickFalse;
  {
    char c = expressionstr[length];
    if ((c == '\0') || (c == '(') || (isspace((int) (unsigned char) c) != 0))
      return MagickTrue;
  }
  return MagickFalse;
}
