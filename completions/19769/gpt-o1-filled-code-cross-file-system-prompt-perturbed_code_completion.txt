if ((expressionstr == (char *) NULL) || (name == (char *) NULL) ||
      (length == 0) || (strlen(expressionstr) < length))
    return(MagickFalse);

  if (LocaleNCompare(expressionstr,name,length) != 0)
    return(MagickFalse);

  {
    int c=(int) (unsigned char) expressionstr[length];
    if ((c == '(') || (c == '\0') || (isspace(c) != 0))
      return(MagickTrue);
  }

  return(MagickFalse);
