// Parse the bone name
    const char* szStart = szCurrent;
    while (*szCurrent != '\"' && !IsSpaceOrNewLine(*szCurrent) && *szCurrent != '\0') {
        ++szCurrent;
    }
    const char* szEnd = szCurrent;

    if (*szCurrent == '\"') {
        ++szCurrent;
        boneNameLength = (unsigned int)(szEnd - szStart);
        bone.mName = std::string(szStart + 1, boneNameLength - 1); // Exclude the surrounding quotes
    } else {
        LogWarning("Bone name is expected to be enclosed in double quotation marks. Assuming no name.");
        bone.mName = "";
    }

    // the only negative bone parent index that could occur is -1 AFAIK
    if(!ParseSignedInt(szCurrent,&szCurrent,(int&)bone.iParent))  {
        LogErrorNoThrow("Unexpected EOF/EOL while parsing bone parent index. Assuming -1");
        bone.iParent = -1;
    }

    // go to the beginning of the next line
    SMDI_PARSE_RETURN;
