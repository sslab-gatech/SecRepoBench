int boneIndex = 0;

    LogErrorNoThrow("Unexpected EOF/EOL while parsing bone index");
    SMDI_PARSE_RETURN;
}
if (!ParseUnsignedInt(szCurrent, szCurrentOut, (unsigned int&)boneIndex)) {
    LogErrorNoThrow("Unexpected EOF/EOL while parsing bone index");
    SMDI_PARSE_RETURN;
}
if (boneIndex >= (int)m_Bones.size()) {
    LogErrorNoThrow("Invalid bone index");
    SMDI_PARSE_RETURN;
}
