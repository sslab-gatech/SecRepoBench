const char* szStart = szCurrent;
    while (*szCurrent && *szCurrent != '\"' && !IsSpaceOrNewLine(*szCurrent)) {
        ++szCurrent;
    }

    if (szCurrent == szStart) {
        LogErrorNoThrow("Bone name cannot be empty");
        SMDI_PARSE_RETURN;
    }

    boneNameLength = static_cast<unsigned int>(szCurrent - szStart);
    bone.mName = std::string(szStart, boneNameLength);
