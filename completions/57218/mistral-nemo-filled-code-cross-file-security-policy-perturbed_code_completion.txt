        LogWarning("Bone name is expected to be enclosed in double quotation marks. Found unexpected double quote.");
        bQuota = true;
    }

    if (!bQuota && *szCurrent == '\n') {
        LogErrorNoThrow("Unexpected EOL while parsing bone name");
        SMDI_PARSE_RETURN;
    }

    if (!bQuota && *szCurrent == '\0') {
        LogErrorNoThrow("Unexpected EOF while parsing bone name");
        SMDI_PARSE_RETURN;
    }

        boneNameLength = (unsigned int)(szEnd - szCurrent);
        ++szCurrent;
        break;
