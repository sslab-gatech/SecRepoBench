const char* szBoneNameStart = szCurrent;
while (*szCurrent && *szCurrent != '\n' && *szCurrent != '\r') {
    ++szCurrent;
}
if (szCurrent == szBoneNameStart) {
    LogErrorNoThrow("Unexpected EOF/EOL while parsing bone name");
    SMDI_PARSE_RETURN;
}
boneNameLength = (unsigned int)(szCurrent - szBoneNameStart);
bone.mName = std::string(szBoneNameStart, boneNameLength);
