if(!SkipSpaces(szCurrent,&szCurrent)) {
    LogErrorNoThrow("Unexpected EOF/EOL while parsing bone index");
    SMDI_PARSE_RETURN;
}
if(!ParseUnsignedInt(szCurrent,&szCurrent,bone.iIndex)) {
    LogErrorNoThrow("Unexpected EOF/EOL while parsing bone index");
    SMDI_PARSE_RETURN;
}
if (bone.iIndex >= mBones.size()) {
    mBones.resize(bone.iIndex + 1);
}
bone = mBones[bone.iIndex];
