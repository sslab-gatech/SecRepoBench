    if (!SkipSpaces(&szCurrent)) {
        LogErrorNoThrow("Unexpected EOF/EOL while parsing bone name");
        SMDI_PARSE_RETURN;
    }

    // Check if the next character is a double quote
    bool bQuota = true;
    if ('\"' != *szCurrent) {
        LogWarning("Bone name is expected to be enclosed in double quotation marks.");
        bQuota = false;
    } else {
        ++szCurrent; // Move past the opening quote
    }

    // Find the end of the bone name
    const char* szEnd = szCurrent;
    while (true) {
        if (bQuota && '\"' == *szEnd) {
            boneNameLength = static_cast<unsigned int>(szEnd - szCurrent);
            ++szEnd; // Move past the closing quote
            break;
        } else if (!bQuota && IsSpaceOrNewLine(*szEnd)) {
            boneNameLength = static_cast<unsigned int>(szEnd - szCurrent);
            break;
        } else if (*szEnd == '\0') {
            LogErrorNoThrow("Unexpected EOF/EOL while parsing bone name");
            SMDI_PARSE_RETURN;
        }
        ++szEnd;
