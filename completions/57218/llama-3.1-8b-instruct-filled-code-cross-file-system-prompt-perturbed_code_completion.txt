if (bQuota && '\"' == *currentChar) {
    ++currentChar;
    const char* szEnd = currentChar;
    for ( ;; ) {
        if ('\"' == *szEnd) {
            iBone = (unsigned int)(szEnd - currentChar);
            ++szEnd;
            break;
        } else if (!(*szEnd)) {
            LogErrorNoThrow("Unexpected EOF/EOL while parsing bone name");
            SMDI_PARSE_RETURN;
        }
        ++szEnd;
    }
    bone.mName = std::string(currentChar,iBone);
    currentChar = szEnd;
} else {
    const char* szEnd = currentChar;
    for ( ;; ) {
        if (IsSpaceOrNewLine(*szEnd)) {
            iBone = (unsigned int)(szEnd - currentChar);
            break;
        } else if (!(*szEnd)) {
            LogErrorNoThrow("Unexpected EOF/EOL while parsing bone name");
            SMDI_PARSE_RETURN;
        }
        ++szEnd;
    }
    bone.mName = std::string(currentChar,iBone);
    currentChar = szEnd;
}
