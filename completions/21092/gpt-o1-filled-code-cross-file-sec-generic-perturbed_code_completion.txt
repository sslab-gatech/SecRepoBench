{
  int i;

  // If inputvalue is less than or equal to the first fromCoord, map it directly.
  if (inputvalue <= arrayZ[0].fromCoord)
    return inputvalue - arrayZ[0].fromCoord + arrayZ[0].toCoord;

  // Find the segment in which inputvalue fits and do linear interpolation.
  for (i = 1; i < len; i++)
  {
    if (inputvalue <= arrayZ[i].fromCoord)
    {
      int denom = arrayZ[i].fromCoord - arrayZ[i - 1].fromCoord;
      // If two fromCoords are equal, use the previous toCoord.
      if (denom == 0)
        return arrayZ[i - 1].toCoord;

      return arrayZ[i - 1].toCoord +
        ((arrayZ[i].toCoord - arrayZ[i - 1].toCoord)
         * (inputvalue - arrayZ[i - 1].fromCoord) + denom / 2)
        / denom;
    }
  }

  // If inputvalue is larger than all fromCoords, map beyond the last segment.
  return inputvalue - arrayZ[len - 1].fromCoord + arrayZ[len - 1].toCoord;
}
