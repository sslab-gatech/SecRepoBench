{
    unsigned int glyph_id = 0;
    for (unsigned int range_index = 0; range_index < glyph_count; range_index++)
    {
        hb_codepoint_t sid = get_glyph(range_index, glyph_count);
        if (sid != 0)
        {
            mapping->set(glyph_id, sid);
            glyph_id++;
            if (glyph_id >= glyph_count)
                break;
        }
    }
}
