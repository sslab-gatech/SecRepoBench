{
  unsigned int glyph = 0;
  for (unsigned int i = 0; i < ranges.length && glyph < glyph_count; i++)
  {
    unsigned int sid = ranges[i].code;
    unsigned int count = ranges[i].get_nominal_glyph_count ();
    for (unsigned int j = 0; j < count && glyph < glyph_count; j++, glyph++)
    {
      mapping->set (glyph, sid);
      sid++;
    }
  }
