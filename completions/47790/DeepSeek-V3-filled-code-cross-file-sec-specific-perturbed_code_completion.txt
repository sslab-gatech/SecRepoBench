{
  unsigned int glyph = 0;
  for (const auto &_ : get_ranges ())
  {
    unsigned count = _.count;
    unsigned sid = _.code;
    for (unsigned int i = 0; i < count && glyph < glyph_count; i++, glyph++)
    {
      mapping->set (glyph, sid + i);
    }
    if (glyph >= glyph_count)
      break;
  }
