{
    unsigned int current_glyph_id = 0;

    for (unsigned int sid = 0; sid < glyph_count; ++sid) {
        if (current_glyph_id >= glyph_count) {
            break; // Exit if we reach the maximum number of glyphs
        }

        // Map the current glyph identifier to the SID
        mapping->insert(std::make_pair(current_glyph_id, sid));

        // Increment the glyph identifier for the next mapping
        current_glyph_id++;
    }
}
