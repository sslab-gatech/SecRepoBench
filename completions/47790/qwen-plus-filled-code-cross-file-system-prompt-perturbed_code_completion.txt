{
    hb_codepoint_t sid = 0;
    unsigned int glyph_id = 0;

    for (const auto& range : ranges) // Assuming 'ranges' is a collection of SID ranges
    {
        hb_codepoint_t start_sid = range.start; // Starting SID of the range
        unsigned int num_sids = range.count;   // Number of consecutive SIDs in the range

        for (unsigned int i = 0; i < num_sids && glyph_id < glyph_count; ++i, ++glyph_id)
        {
            sid = start_sid + i;
            mapping->set (glyph_id, sid); // Map current glyph identifier to the SID
        }

        if (glyph_id >= glyph_count)
            break; // Exit loop when the maximum number of glyphs is reached
    }
}
