{
  hb_codepoint_t glyph = 1;
  for (unsigned int i = 0; glyph < glyph_count; i++)
  {
    hb_codepoint_t sid_start = ranges[i].first;
    hb_codepoint_t sid_end   = sid_start + ranges[i].nLeft;
    for (hb_codepoint_t sid = sid_start; sid <= sid_end && glyph < glyph_count; sid++)
    {
      mapping->set(glyph, sid);
      glyph++;
    }
  }
}
