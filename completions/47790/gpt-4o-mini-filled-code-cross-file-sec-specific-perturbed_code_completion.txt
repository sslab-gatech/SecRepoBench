{
    unsigned int current_glyph = 0;
    for (unsigned int sid = 0; sid < glyph_count; sid++)
    {
        hb_codepoint_t glyph = get_glyph(sid, glyph_count);
        if (glyph != 0)
        {
            mapping->insert(std::make_pair(glyph, sid));
            current_glyph++;
            if (current_glyph >= glyph_count)
                break;
        }
    }
}
