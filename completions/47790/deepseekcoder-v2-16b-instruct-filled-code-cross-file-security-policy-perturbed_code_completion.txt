for (unsigned int i = 0; i < glyph_count; ++i) {
    // Retrieve the starting SID and the number of consecutive SIDs to map
    unsigned int start_sid = get_start_sid_for_glyph(i);
    unsigned int sid_count = get_sid_count_for_glyph(i);

    // Map the current glyph identifier to the SIDs
    for (unsigned int j = 0; j < sid_count; ++j) {
        mapping[i] = start_sid + j;
    }
}
