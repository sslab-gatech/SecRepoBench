{
    unsigned int glyph_id = 0;
    for (const auto& range : ranges) // Assuming 'ranges' is a collection of SID ranges
    {
        hb_codepoint_t start_sid = range.start_sid;
        unsigned int count = range.count;

        for (unsigned int i = 0; i < count && glyph_id < glyph_count; ++i, ++glyph_id)
        {
            hb_codepoint_t sid = start_sid + i;
            mapping->set(glyph_id, sid); // Map current glyph ID to SID
        }

        if (glyph_id >= glyph_count)
            break; // Exit loop when maximum glyphs are reached
    }
}
