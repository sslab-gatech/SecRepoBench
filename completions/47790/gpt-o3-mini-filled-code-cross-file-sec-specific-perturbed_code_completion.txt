  unsigned int glyph = 0;

  {
    hb_codepoint_t start_sid = this->ranges[range_index].start_sid;
    unsigned int range_length = this->ranges[range_index].count;

    for (unsigned int i = 0; i < range_length && glyph < glyph_count; i++)

    }
  }
