void cff1::accelerator_t::get_glyph_from_name(const char *name, int len, hb_codepoint_t *glyph) const
{
  if (unlikely (!is_valid())) return false;
  if (is_CID()) return false;
  if (len < 0) len = strlen(name);
  if (unlikely (!len)) return false;

  hb_codepoint_t sid = lookup_name_for_glyph(name, len);
  if (sid == CFF_UNDEF_SID) return false;

  *glyph = sid_to_glyph(sid);
  return true;
}
