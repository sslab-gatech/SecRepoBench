{
  hb_codepoint_t gid = 1;
  for (unsigned int i = 0; i < sid_ranges.length; i++)
  {
    hb_codepoint_t sid = sid_ranges[i].code;
    for (int left = (int)sid_ranges[i].glyph; left >= 0; left--)
    {
      if (gid >= glyph_count)
        return;
      mapping->set (gid, sid);
      gid++;
      sid++;
    }
  }
}
