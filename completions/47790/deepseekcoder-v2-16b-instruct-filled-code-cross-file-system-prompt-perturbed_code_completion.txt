for (unsigned int range = 0; range < num_ranges; range++)
    {
      unsigned int start_sid = sids[range * 2];
      unsigned int num_sids = sids[range * 2 + 1];

      for (unsigned int sid = start_sid; sid < start_sid + num_sids; sid++)
      {
        for (unsigned int glyph = 0; glyph < glyph_count; glyph++)
        {
          mapping->set (glyph, get_glyph (sid, glyph_count));
          glyph++;
          if (glyph >= glyph_count)
            break;
        }
      }
    }
