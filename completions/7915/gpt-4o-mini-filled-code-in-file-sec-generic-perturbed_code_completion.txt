path_info=(PathInfo *) AcquireQuantumMemory(2UL,sizeof(*path_info));
  if (path_info == (PathInfo *) NULL)
    return((PathInfo *) NULL);
  path_info[0].code=MoveToCode;
  path_info[0].point=primitive_info[0].point;
  n=1;
  for (i=0; primitive_info[i].primitive != UndefinedPrimitive; i++)
  {
    if (primitive_info[i].primitive == MoveToPrimitive)
      {
        path_info[n].code=MoveToCode;
        path_info[n].point=primitive_info[i].point;
        n++;
      }
    else if (primitive_info[i].primitive == LinePrimitive)
      {
        path_info[n].code=LineToCode;
        path_info[n].point=primitive_info[i].point;
        n++;
      }
    else if (primitive_info[i].primitive == ClosePathPrimitive)
      {
        path_info[n].code=EndCode;
        n++;
        break;
      }
  }
  path_info[n].code=EndCode;
  path_info[n].point.x=0.0;
  path_info[n].point.y=0.0;
  return(path_info);
