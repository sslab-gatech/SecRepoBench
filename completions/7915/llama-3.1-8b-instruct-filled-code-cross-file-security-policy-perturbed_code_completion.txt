ssize_t
GetPathInfoSize(const PrimitiveInfo *primitive_info)
{
  ssize_t
    size;

  size=0;
  for (i=0; primitive_info[i].primitive!= UndefinedPrimitive; i++)
    {
      size++;
      if (primitive_info[i].coordinates > 0)
        size+=2;
    }
  size+=2;
  return(size);
}

PathInfo *
AllocatePathInfo(const PrimitiveInfo *primitive_info)
{
  ssize_t
    size;

  size=GetPathInfoSize(primitive_info);
  path_info=(PathInfo *) AcquireMagickMemory((size_t) (size* sizeof(PathInfo)));
  if (path_info == (PathInfo *) NULL)
    return((PathInfo *) NULL);
  memset(path_info,0,sizeof(PathInfo));
  return(path_info);
}
