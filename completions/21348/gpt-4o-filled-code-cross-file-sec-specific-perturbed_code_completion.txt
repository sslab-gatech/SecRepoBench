u_int offset = ip_offset + ip_len + sizeof(struct ndpi_udphdr);
      u_int8_t version = packet[offset];
      u_int8_t type = packet[offset + 1];

      if(version == 1 && type == 0) { // Check for TZSP version 1 and Ethernet encapsulation
        offset += 2; // Skip version and type

        // Process TZSP tags
        while(offset < header->caplen) {
          u_int8_t tag_type = packet[offset];
          u_int8_t tag_length = packet[offset + 1];

          if(tag_type == 1) { // END tag
            offset += 2;
            break;
          } else if(tag_type == 0) { // PADDING tag
            offset += 1;
          } else {
            offset += 2 + tag_length; // Skip tag type, length, and data
          }
        }

        if(offset < header->caplen) {
          eth_offset = offset;
          goto datalink_check;
        }
      }
