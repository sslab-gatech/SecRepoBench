u_int offset = ip_offset + ip_len + sizeof(struct ndpi_udphdr);
      u_int8_t version = packet[offset];
      u_int8_t type = packet[offset + 1];

      if (version == TZSP_VERSION && type == TZSP_TYPE_ETHERNET) {
        tunnel_type = ndpi_tzsp_tunnel;

        u_int tag_offset = offset + 2;
        while (tag_offset < header->caplen) {
          u_int8_t tag_type = packet[tag_offset];
          u_int16_t tag_len = ntohs(*((u_int16_t *)(packet + tag_offset + 1)));

          if (tag_type == TZSP_TAG_END) {
            break;
          } else if (tag_type == TZSP_TAG_PADDING) {
            tag_offset += tag_len + 3;
            continue;
          }

          tag_offset += tag_len + 3;
        }

        if (tag_offset < header->caplen) {
          eth_offset = tag_offset;
          datalink_type = DLT_EN10MB;
          goto datalink_check;
        }
      }
