        if (mrb_float_is_nan(fval) || mrb_float_is_infinite(fval)) {
          mrb_value str;
          if (mrb_float_is_nan(fval)) {
            str = mrb_str_new_cstr(mrb, "NaN");
          } else {
            str = mrb_str_new_cstr(mrb, "Inf");
          }
          if (flags & FPLUS) {
            str = mrb_str_plus(mrb, str, mrb_str_new_cstr(mrb, "+"));
          } else if (flags & FSPACE) {
            str = mrb_str_plus(mrb, str, mrb_str_new_cstr(mrb, " "));
          }
          PUSH(RSTRING_PTR(str), RSTRING_LEN(str));
        } else {

          int need;
          fmt_setup(fbuf, sizeof(fbuf), *p, flags, width, prec);
          need = 0;
          if (*p!= 'e' && *p!= 'E') {
            int i;
            frexp(fval, &i);
            if (i > 0)
              need = BIT_DIGITS(i);
          }
          if (need > MRB_INT_MAX - ((flags&FPREC)? prec : 6)) {
          too_big_width:
            mrb_raise(mrb, E_ARGUMENT_ERROR,

          }

            need = width;
          if (need > MRB_INT_MAX - 20) {

          }

          CHECK(need);
          n = mrb_float_to_cstr(mrb, &buf[blen], need, fbuf, fval);
          if (n < 0 || n >= need) {
            mrb_raise(mrb, E_RUNTIME_ERROR, "formatting error");
          }
          blen += n;
        }
