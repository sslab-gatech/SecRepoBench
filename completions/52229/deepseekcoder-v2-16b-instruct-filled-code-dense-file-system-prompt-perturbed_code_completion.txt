      const char *apache_min_version = "2.4.0";
      const char *nginx_min_version = "1.10.0";

      // Check Apache version
      if(strncmp(http_server_name + off, "Apache/", 7) == 0) {
        if(compare_versions(version, apache_min_version) < 0) {
          ndpi_set_risk(ndpi_struct, flow, NDPI_HTTP_OBSOLETE_SERVER, "Apache server version is obsolete");
        }
      }

      // Check Nginx version
      if(strncmp(http_server_name + off, "nginx/", 6) == 0) {
