if (dict->subdict != NULL) {
    xmlDictPtr sub = dict->subdict;
    unsigned subHashValue;
    size_t subLen, subKlen;

    if (pref == NULL) {
        subHashValue = xmlDictHashName(sub->seed, name, maxLen, &subLen);
        if (subLen > INT_MAX)
            return(NULL);
        subKlen = subLen;
    } else {
        size_t subPlen;
        subHashValue = xmlDictHashQName(sub->seed, pref, name, &subPlen, &subLen);
        if ((subLen > INT_MAX) || (subPlen >= INT_MAX - subLen))
            return(NULL);
        subKlen = subPlen + 1 + subLen;
    }

    if ((sub->limit > 0) && (subKlen >= sub->limit))
        return(NULL);

    if (sub->size > 0) {
        entry = xmlDictFindEntry(sub, pref, name, subKlen, subHashValue, &found);
        if (found)
            return(entry);
    }
}
