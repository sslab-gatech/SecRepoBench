if (dict->subdict != NULL) {
        unsigned subhashValue;
        size_t subklen;

        if (pref == NULL) {
            subhashValue = xmlDictHashName(dict->seed, name, maxLen, &len);
            if (len > INT_MAX)
                return(NULL);
            subklen = len;
        } else {
            subhashValue = xmlDictHashQName(dict->seed, pref, name, &plen, &len);
            if ((len > INT_MAX) || (plen >= INT_MAX - len))
                return(NULL);
            subklen = plen + 1 + len;
        }

        if ((dict->subdict->limit > 0) && (subklen >= dict->subdict->limit))
            return(NULL);

        entry = xmlDictFindEntry(dict->subdict, pref, name, subklen, subhashValue, &found);
        if (found)
            return(entry);
    }
