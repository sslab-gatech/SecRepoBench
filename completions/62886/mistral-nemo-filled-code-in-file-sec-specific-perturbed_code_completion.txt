if (dict->subdict != NULL) {
    unsigned hashValue;
    size_t maxLen, len, plen, klen;
    int found = 0;

    if (pref == NULL) {
        hashValue = xmlDictHashName(dict->subdict->seed, name, maxLen, &len);
        if (len > INT_MAX)
            return(NULL);
        klen = len;
    } else {
        hashValue = xmlDictHashQName(dict->subdict->seed, pref, name, &plen, &len);
        if ((len > INT_MAX) || (plen >= INT_MAX - len))
            return(NULL);
        klen = plen + 1 + len;
    }

    entry = xmlDictFindEntry(dict->subdict, pref, name, klen, hashValue, &found);
    if (found)
        return(entry);
}
