if (dict->subdict != NULL) {
        const xmlDictEntry *subEntry;
        if (pref == NULL) {
            unsigned subHashValue = xmlDictHashName(dict->subdict->seed, name, maxLen, &len);
            subEntry = xmlDictFindEntry(dict->subdict, NULL, name, len, subHashValue, &found);
        } else {
            unsigned subHashValue = xmlDictHashQName(dict->subdict->seed, pref, name, &plen, &len);
            subEntry = xmlDictFindEntry(dict->subdict, pref, name, plen + 1 + len, subHashValue, &found);
        }
        if (found)
            return subEntry;
    }
