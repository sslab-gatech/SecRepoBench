// Check if the main dictionary has a subdictionary.
if (dict->subdict != NULL) {
    unsigned subhashValue;
    int subfound;
    if (pref == NULL) {
        subhashValue = xmlDictHashName(dict->subdict->seed, name, maxLen, &len);
    } else {
        subhashValue = xmlDictHashQName(dict->subdict->seed, pref, name, &plen, &len);
    }
    const xmlDictEntry *subentry = xmlDictFindEntry(dict->subdict, pref, name, len, subhashValue, &subfound);
    if (subfound) {
        return subentry;
    }
}
