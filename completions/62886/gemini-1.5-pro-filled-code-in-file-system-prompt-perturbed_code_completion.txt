if (dict->subdict != NULL) {
        size_t sublen;
        unsigned subhash;

        if (pref == NULL) {
            subhash = xmlDictHashName(dict->subdict->seed, name, maxLen, &sublen);
        } else {
            subhash = xmlDictHashQName(dict->subdict->seed, pref, name, &plen, &sublen);
        }

        entry = xmlDictFindEntry(dict->subdict, pref, name,
                                (pref == NULL) ? sublen : plen + 1 + sublen,
                                subhash, &found);
        if (found)
            return(entry);
    }
