if (dict->subDict != NULL) {
        unsigned subHashValue;
        size_t subLen, subPlen;

        if (pref == NULL) {
            subHashValue = xmlDictHashName(dict->subDict->seed, name, maxLen, &subLen);
            if (subLen > INT_MAX)
                return(NULL);
        } else {
            subHashValue = xmlDictHashQName(dict->subDict->seed, pref, name, &subPlen, &subLen);
            if ((subLen > INT_MAX) || (subPlen >= INT_MAX - subLen))
                return(NULL);
        }

        if (dict->subDict->size > 0)
            entry = xmlDictFindEntry(dict->subDict, pref, name, subLen, subHashValue, &found);
        if (found)
            return(entry);
    }
