if (dict->subdict != NULL) {
        unsigned subHashValue;
        xmlDictEntry *subEntry = NULL;

        if (pref == NULL) {
            subHashValue = xmlDictHashName(dict->seed, name, maxLen, &len);
        } else {
            subHashValue = xmlDictHashQName(dict->seed, pref, name, &plen, &len);
        }

        if (dict->subdict->size > 0) {
            subEntry = xmlDictFindEntry(dict->subdict, pref, name, klen, subHashValue, &found);
        }

        if (found && subEntry != NULL) {
            return subEntry;
        }
