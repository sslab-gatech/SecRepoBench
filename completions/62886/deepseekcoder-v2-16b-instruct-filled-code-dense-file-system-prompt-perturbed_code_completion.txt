if (dict->subdict != NULL) {
        unsigned subhashValue;
        size_t subklen;

        if (pref == NULL) {
            subhashValue = xmlDictHashName(dict->seed, name, maxLen, &subklen);
            if (subklen > INT_MAX)
                return(NULL);
        } else {
            subhashValue = xmlDictHashQName(dict->seed, pref, name, &subklen, &len);
            if ((subklen > INT_MAX) || (len >= INT_MAX - subklen))
                return(NULL);
        }

        if ((dict->subdict->limit > 0) && (subklen >= dict->subdict->limit))
            return(NULL);

        entry = xmlDictFindEntry(dict->subdict, pref, name, subklen, subhashValue, &found);
        if (found)
            return(entry);
    }
