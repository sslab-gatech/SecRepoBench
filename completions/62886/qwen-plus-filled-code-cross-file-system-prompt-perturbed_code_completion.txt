if (dict->subdict != NULL) {
    unsigned subHashValue;
    const xmlDictEntry *subEntry;

    if (pref == NULL) {
        subHashValue = xmlDictHashName(dict->subdict->seed, name, maxLen, &len);
    } else {
        subHashValue = xmlDictHashQName(dict->subdict->seed, pref, name, &plen, &len);
    }

    if (len > INT_MAX || (pref != NULL && plen >= INT_MAX - len))
        return(NULL);

    subEntry = xmlDictFindEntry(dict->subdict, pref, name, klen, subHashValue, &found);
    if (found)
        return(subEntry);
}
