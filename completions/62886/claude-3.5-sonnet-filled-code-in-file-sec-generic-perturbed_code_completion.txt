if (dict->subdict != NULL) {
    unsigned subHashValue;
    int subFound = 0;
    xmlDictEntry *subEntry;

    if (pref == NULL) {
        subHashValue = xmlDictHashName(dict->subdict->seed, name, maxLen, &len);
    } else {
        subHashValue = xmlDictHashQName(dict->subdict->seed, pref, name, &plen, &len);
    }

    if (dict->subdict->size > 0) {
        subEntry = xmlDictFindEntry(dict->subdict, pref, name, klen, subHashValue, &subFound);
        if (subFound) {
            return subEntry;
        }
    }
}
