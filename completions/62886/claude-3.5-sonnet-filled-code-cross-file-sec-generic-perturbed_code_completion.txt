if (dict->subdict != NULL) {
    xmlDictPtr subdict = dict->subdict;
    unsigned subHashValue;
    
    if (pref == NULL) {
        subHashValue = xmlDictHashName(subdict->seed, name, maxLen, NULL);
    } else {
        subHashValue = xmlDictHashQName(subdict->seed, pref, name, NULL, NULL);
    }

    if (found) {
        return entry;
    }
