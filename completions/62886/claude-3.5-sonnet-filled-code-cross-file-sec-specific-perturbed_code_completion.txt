if (dict->subdict != NULL) {
    const xmlDictEntry *subEntry;
    unsigned subHashValue;

    if (pref == NULL) {
        subHashValue = xmlDictHashName(dict->subdict->seed, name, maxLen, NULL);
    } else {
        subHashValue = xmlDictHashQName(dict->subdict->seed, pref, name, NULL, NULL);
    }

    subEntry = xmlDictFindEntry(dict->subdict, pref, name, klen, subHashValue, &found);
    if (found) {
        return subEntry;
    }
}
