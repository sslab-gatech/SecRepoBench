if (dict->subdict != NULL) {
        xmlDictPtr subdict = (xmlDictPtr) dict->subdict;
        unsigned sub_hashValue;
        size_t sub_len, sub_plen, sub_klen;
        xmlDictEntry *sub_entry = NULL;
        int sub_found = 0;

        if (pref == NULL) {
            sub_hashValue = xmlDictHashName(subdict->seed, name, maxLen, &sub_len);
            if (sub_len > INT_MAX)
                return(NULL);
            sub_klen = sub_len;
        } else {
            sub_hashValue = xmlDictHashQName(subdict->seed, pref, name, &sub_plen, &sub_len);
            if ((sub_len > INT_MAX) || (sub_plen >= INT_MAX - sub_len))
                return(NULL);
            sub_klen = sub_plen + 1 + sub_len;
        }

        if ((subdict->limit > 0) && (sub_klen >= subdict->limit))
            return(NULL);

        if (subdict->size > 0)
            sub_entry = xmlDictFindEntry(subdict, pref, name, sub_klen, sub_hashValue, &sub_found);

        if (sub_found)
            return(sub_entry);
    }
