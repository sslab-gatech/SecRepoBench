if (dict->subdict != NULL) {
        size_t subLen;

        if (pref == NULL) {
            hashValue = xmlDictHashName(dict->subdict->seed, name, maxLen, &subLen);
            if (subLen > INT_MAX)
                return(NULL);
            klen = subLen;
        } else {
            hashValue = xmlDictHashQName(dict->subdict->seed, pref, name, &plen, &subLen);
            if ((subLen > INT_MAX) || (plen >= INT_MAX - subLen))
                return(NULL);
            klen = plen + 1 + subLen;
        }

        entry = xmlDictFindEntry(dict->subdict, pref, name, klen, hashValue, &found);
        if (found)
            return(entry);
    }
