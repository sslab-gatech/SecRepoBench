// Check if the main dictionary has a subdictionary.
if (dict->subdict != NULL) {
    unsigned subHashValue;
    if (pref == NULL) {
        subHashValue = xmlDictHashName(dict->seed, name, maxLen, &len);
    } else {
        subHashValue = xmlDictHashQName(dict->seed, pref, name, &plen, &len);
    }
    xmlDictEntry *subEntry = xmlDictFindEntry(dict->subdict, pref, name, klen, subHashValue, &found);
    if (found)
        return subEntry;
}
