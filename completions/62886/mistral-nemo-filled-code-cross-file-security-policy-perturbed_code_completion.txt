if (dict->subDict != NULL && dict->subDict->size > 0) {
    unsigned subHashValue;
    size_t subLen, subPlen, subKlen;

    if (pref == NULL) {
        subHashValue = xmlDictHashName(dict->subDict->seed, name, maxLen, &subLen);
        if (subLen > INT_MAX)
            return(NULL);
        subKlen = subLen;
    } else {
        subHashValue = xmlDictHashQName(dict->subDict->seed, pref, name, &subPlen, &subLen);
        if ((subLen > INT_MAX) || (subPlen >= INT_MAX - subLen))
            return(NULL);
        subKlen = subPlen + 1 + subLen;
    }

    if ((dict->subDict->limit > 0) && (subKlen >= dict->subDict->limit))
        return(NULL);

    xmlDictEntry *subEntry = xmlDictFindEntry(dict->subDict, pref, name, subKlen, subHashValue, &found);
    if (found)
        return(subEntry);
}
