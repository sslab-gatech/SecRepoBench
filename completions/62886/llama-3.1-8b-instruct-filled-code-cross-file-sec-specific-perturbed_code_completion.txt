// Check if the main dictionary has a subdictionary.
// If a subdictionary exists, compute the hash value for the provided name,
// using either the name alone or the combination of prefix and name,
// depending on whether a prefix is given.
// Attempt to find an existing entry in the subdictionary using the computed hash value.
// If an entry is found in the subdictionary, return that entry.
if (dict->nbSubdicts > 0) {
    xmlDictPtr subdict;
    unsigned hashValue;
    size_t len, plen, klen;

    for (subdict = dict->subdicts; subdict!= NULL; subdict = subdict->next) {
        if (pref == NULL) {
            hashValue = xmlDictHashName(subdict->seed, name, maxLen, &len);
            if (len > INT_MAX)
                continue;
            klen = len;
        } else {
            hashValue = xmlDictHashQName(subdict->seed, pref, name, &plen, &len);
            if ((len > INT_MAX) || (plen >= INT_MAX - len))
                continue;
            klen = plen + 1 + len;
        }

        if ((subdict->limit > 0) && (klen >= subdict->limit))
            continue;

        if (subdict->size > 0) {
            entry = xmlDictFindEntry(subdict, pref, name, klen, hashValue, &found);
            if (found)
                return(entry);
        }
    }
}
