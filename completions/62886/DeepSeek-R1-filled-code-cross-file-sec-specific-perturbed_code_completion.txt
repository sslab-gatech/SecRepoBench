if (dict->subdict != NULL) {
    xmlDictPtr sub = dict->subdict;
    unsigned subHashValue;
    size_t sub_len, sub_plen, sub_klen;

    if (pref == NULL) {
        subHashValue = xmlDictHashName(sub->seed, name, maxLen, &sub_len);
        if (sub_len > INT_MAX)
            return(NULL);
        sub_klen = sub_len;
    } else {
        subHashValue = xmlDictHashQName(sub->seed, pref, name, &sub_plen, &sub_len);
        if ((sub_len > INT_MAX) || (sub_plen >= INT_MAX - sub_len))
            return(NULL);
        sub_klen = sub_plen + 1 + sub_len;
    }

    if ((sub->limit > 0) && (sub_klen >= sub->limit))
        return(NULL);

    if (sub->size > 0) {
        entry = xmlDictFindEntry(sub, pref, name, sub_klen, subHashValue, &found);
        if (found)
            return(entry);
    }
}
