realm_offset = name_offset + 1 + cname_len;

/* Check if cname ends with '$', indicating it's a hostname/machine name */
if(cname_len && cname_str[cname_len-1] == '$') {
  cname_str[cname_len-1] = '\0'; /* Strip the trailing '$' */
  snprintf(flow->protos.kerberos.hostname, sizeof(flow->protos.kerberos.hostname), "%s", cname_str);
} else {
  snprintf(flow->protos.kerberos.username, sizeof(flow->protos.kerberos.username), "%s", cname_str);
}

/* Find ASN.1 identifier for realm data */
for(i=0; i<10; i++) {
  if(realm_offset < flow_packet->payload_packet_len && flow_packet->payload[realm_offset] != 0x1b)
    realm_offset++;
