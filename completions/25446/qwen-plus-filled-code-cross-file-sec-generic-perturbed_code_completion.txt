u_int16_t realm_offset = body_offset + 2;

if(flow_packet->payload[realm_offset] == 0x1b) {
  // Ensure the cname string is within bounds before processing
  if((name_offset + cname_len + 2) < flow_packet->payload_packet_len) {
    if(flow_packet->payload[name_offset + cname_len] == '$') {
      // Strip the '$' and store as hostname
      cname_str[cname_len - 1] = '\0';
      snprintf(flow->protos.kerberos.hostname, sizeof(flow->protos.kerberos.hostname), "%s", cname_str);
    } else {
      // Store as username
      snprintf(flow->protos.kerberos.username, sizeof(flow->protos.kerberos.username), "%s", cname_str);
    }
  }
}

// Increment realm offset by scanning through the payload for a specific ASN.1 identifier
for(i = 0; i < 20 && realm_offset < flow_packet->payload_packet_len; i++) {
  if(flow_packet->payload[realm_offset] != 0x1b)
    realm_offset++;
  else
    break;
}
