if (image->magick_columns == 0 || image->magick_rows == 0)
            {
              (void) ThrowMagickException(exception,GetMagickModule(),
                OptionWarning,"ImageDimensionsInvalid","`%s'","Image has zero width or height");
              continue; /* skip processing if dimensions are invalid */
            }

          // Assuming the pattern is valid and we have a valid image
          fx_info=AcquireFxInfo(image,lookup_pattern+4,exception); // Start after "hex:"
          status=FxEvaluateChannelExpression(fx_info,RedPixelChannel,0,0,&value,exception);
          if (status != MagickFalse)
            {
              unsigned char red=(unsigned char) ClampToQuantum(value);
              status=FxEvaluateChannelExpression(fx_info,GreenPixelChannel,0,0,&value,exception);
              unsigned char green=(unsigned char) ClampToQuantum(value);
              status=FxEvaluateChannelExpression(fx_info,BluePixelChannel,0,0,&value,exception);
              unsigned char blue=(unsigned char) ClampToQuantum(value);
              PixelInfo pixel;
              pixel.red=(double) QuantumRange*red;
              pixel.green=(double) QuantumRange*green;
              pixel.blue=(double) QuantumRange*blue;
              pixel.alpha=(double) QuantumRange; // Assuming full opacity
              char name[MagickPathExtent];
              (void) QueryColorname(image,&pixel,SVGCompliance,name,exception);
              AppendString2Text(name);
            }
