if (LocaleNCompare("hex:",lookup_pattern,4) == 0)
        {
          // Retrieve pixel information from the image.
          // Acquire a new FxInfo instance using the specified pattern starting after the prefix "hex:".
          // Evaluate the red, green, and blue channel expressions, storing the results in respective pixel channels.
          // Scale the evaluated channel values to the QuantumRange and assign them to the pixel structure.
          FxInfo
            *fx_info;

          MagickStatusType
            status;

          PixelInfo
            pixel;

          double
            red_value,
            green_value,
            blue_value;

          fx_info=AcquireFxInfo(image,lookup_pattern+4,exception);
          status=FxEvaluateChannelExpression(fx_info,RedPixelChannel,0,0,&red_value,exception);
          status&=FxEvaluateChannelExpression(fx_info,GreenPixelChannel,0,0,&green_value,exception);
          status&=FxEvaluateChannelExpression(fx_info,BluePixelChannel,0,0,&blue_value,exception);

          if (status != MagickFalse)
            {
              GetPixelInfo(image,&pixel);
              pixel.red=(Quantum) (red_value * QuantumRange);
              pixel.green=(Quantum) (green_value * QuantumRange);
              pixel.blue=(Quantum) (blue_value * QuantumRange);
              SetPixelInfo(image,&pixel);

              char
                name[MagickPathExtent];

              (void) QueryColorname(image,&pixel,SVGCompliance,name,exception);
              AppendString2Text(name);
            }

          fx_info=DestroyFxInfo(fx_info);
        }
