if (tlvBufferPtr == NULL || tlvDataLen == 0)
			{
				resRec.assign(NULL);
				return resRec;
			}

			uint8_t* nextRecordPtr = record.getRecordBasePtr() + record.getTotalSize();
			resRec.assign(nextRecordPtr);

			if (nextRecordPtr + resRec.getTotalSize() > tlvBufferPtr + tlvDataLen)
			{
				resRec.assign(NULL);
			}

			if (!resRec.isNull() && (tlvDataLen == 0 || resRec.getTotalSize() == 0))
			{
				resRec.assign(NULL);
			}

			return resRec;
