				return resRec;

			size_t recordSize = record.getSize();
			uint8_t* nextRecordPtr = record.getRawData() + recordSize;
			size_t nextRecordLen = tlvDataLen - (nextRecordPtr - tlvBufferPtr);

			}

			resRec.assign(nextRecordPtr, nextRecordLen);

			}
