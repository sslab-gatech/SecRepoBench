if (record.isNull() || tlvBufferPtr == nullptr || tlvDataLen == 0) {
				return resRec; // Return uninitialized result record if input is invalid
			}

			size_t recordSize = record.getSize();
			const uint8_t* nextRecordPtr = tlvBufferPtr + record.getOffset() + recordSize;

			if (nextRecordPtr >= tlvBufferPtr + tlvDataLen) {
				return resRec; // Return uninitialized result record if out of bounds
			}

			resRec = getTLVRecord(record.getType(), const_cast<uint8_t*>(nextRecordPtr), tlvDataLen - (nextRecordPtr - tlvBufferPtr));
			return resRec;
