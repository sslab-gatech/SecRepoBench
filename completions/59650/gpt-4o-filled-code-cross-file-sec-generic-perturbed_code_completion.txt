if (record.isNull() || tlvBufferPtr == nullptr || tlvDataLen == 0) {
    return resRec; // Return uninitialized result if input is invalid
}

size_t currentRecordSize = record.getSize();
const uint8_t* nextRecordPtr = record.getBasePtr() + currentRecordSize;

// Check if the next record pointer is within bounds
if (nextRecordPtr < tlvBufferPtr || nextRecordPtr >= tlvBufferPtr + tlvDataLen) {
    return resRec; // Return uninitialized result if out-of-bounds
}

// Initialize the next record
TLVRecordType nextRecord(nextRecordPtr);

// Check if the next record is valid
if (!nextRecord.isValid() || nextRecord.getTotalSize() > tlvDataLen - (nextRecordPtr - tlvBufferPtr)) {
    return resRec; // Return uninitialized result if invalid
}

return nextRecord; // Return the next valid TLV record
