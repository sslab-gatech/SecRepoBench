if (record.isNull())
    return resRec;

// Calculate the next TLV record by advancing the current record's base pointer by its total size
const uint8_t* nextRecordBasePtr = record.getRecordBasePtr() + record.getTotalSize();
resRec.assign(const_cast<uint8_t*>(nextRecordBasePtr));

// Check if the calculated record is out-of-bounds of the TLV memory range
if (resRec.getRecordBasePtr() + resRec.getTotalSize() > tlvBufferPtr + tlvDataLen)
    resRec.assign(NULL);

// Check if the calculated record's total size is valid
if (!resRec.isNull() && resRec.getTotalSize() == 0)
    resRec.assign(NULL);

return resRec;
