if (record.isNull())
				return resRec;

			uint8_t* nextRecordPtr = record.getRecordBasePtr() + record.getTotalSize();
			if (nextRecordPtr + sizeof(TLVRecordType) > (uint8_t*)tlvBufferPtr + tlvDataLen)
				return resRec;

			resRec.assign(nextRecordPtr);
			if (resRec.getRecordBasePtr() + resRec.getTotalSize() > (uint8_t*)tlvBufferPtr + tlvDataLen)
				resRec.assign(NULL);

			if (!resRec.isNull() && (tlvDataLen == 0 || resRec.getTotalSize() == 0))
				resRec.assign(NULL);

			return resRec;
