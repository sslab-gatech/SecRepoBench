			if (record.isNull())
			{
				return resRec;
			}

			size_t nextRecordOffset = record.getOffset() + record.getSize();

			// If the calculated record is invalid or out-of-bounds of the TLV memory range, reset it to null.
			if (nextRecordOffset >= tlvDataLen)
			{
				resRec.assign(NULL);
				return resRec;
			}

			// Return the next TLV record or null if any invalid conditions are detected.
			resRec = getTLVRecord(NULL, tlvBufferPtr + nextRecordOffset, tlvDataLen - nextRecordOffset);
