// Check if the input record is null; if so, return an uninitialized result record.
if (record.isNull())
{
    resRec.assign(NULL);
    return resRec;
}

// Calculate the next TLV record by advancing the current record's base pointer by its total size.
size_t recordSize = record.getSize();
uint8_t* nextRecordPtr = record.getRawData() + recordSize;

// If the calculated record is invalid or out-of-bounds of the TLV memory range, reset it to null.
if (nextRecordPtr < tlvBufferPtr || nextRecordPtr >= tlvBufferPtr + tlvDataLen)
{
    resRec.assign(NULL);
    return resRec;
}

// Create a new TLVRecordType with the nextRecordPtr and assign it to resRec.
resRec = TLVRecordType(nextRecordPtr);

// Return the next TLV record or null if any invalid conditions are detected.
if (resRec.isNull())
{
    return resRec;
}

return resRec;
