if (record.isNull())
			{
				return resRec;
			}

			uint8_t* nextRecordPtr = record.getBasePtr() + record.getTotalSize();

			if (nextRecordPtr < tlvBufferPtr || nextRecordPtr >= tlvBufferPtr + tlvDataLen)
			{
				resRec.assign(NULL);
				return resRec;
			}

			resRec.assign(nextRecordPtr);

			if (!resRec.isValid())
			{
				resRec.assign(NULL);
			}

			return resRec;
