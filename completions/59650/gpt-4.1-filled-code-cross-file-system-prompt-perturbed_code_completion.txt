if (record.isNull())
				return resRec;

			const uint8_t* curRecBase = record.getRecordBasePtr();
			size_t curRecSize = record.getTotalSize();

			if (curRecBase == NULL || curRecSize == 0)
			{
				resRec.assign(NULL);
				return resRec;
			}

			const uint8_t* nextRecBase = curRecBase + curRecSize;
			size_t offset = nextRecBase - tlvBufferPtr;

			if (offset >= tlvDataLen)
			{
				resRec.assign(NULL);
				return resRec;
			}

			TLVRecordType nextRec(nextRecBase, tlvDataLen - offset);
			if (nextRec.isNull() || (offset + nextRec.getTotalSize() > tlvDataLen))
			{
				resRec.assign(NULL);
				return resRec;
			}

			return nextRec;
