if (record.isNull())
	{
		resRec.assign(NULL);
	}
	else
	{
		size_t recordTotalSize = record.getTotalSize();
		const uint8_t* newRecBasePtr = record.getBasePtr() + recordTotalSize;
		if (newRecBasePtr >= tlvBufferPtr + tlvDataLen || *newRecBasePtr == 0)
		{
			resRec.assign(NULL);
		}
		else
		{
			resRec = TLVRecordType(newRecBasePtr);
		}
	}
return resRec;
