if (record.getRecordBasePtr() + record.getTotalSize() + sizeof(TLVRecordType::TLVRawData) > tlvBufferPtr + tlvDataLen)
		resRec.assign(NULL);
else
{
	resRec.assign((TLVRecordType::TLVRawData*)(record.getRecordBasePtr() + record.getTotalSize()));
	if (resRec.isNull() || resRec.getTotalSize() == 0)
		resRec.assign(NULL);
}
return resRec;
