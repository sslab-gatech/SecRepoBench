if (record.isNull())
			{
				return resRec;
			}

			size_t recordSize = record.getSize();
			uint8_t* nextRecordPtr = record.getRawData() + recordSize;

			if (nextRecordPtr > tlvBufferPtr + tlvDataLen || nextRecordPtr < tlvBufferPtr)
			{
				return resRec;
			}

			resRec.assign(nextRecordPtr, tlvBufferPtr, tlvDataLen);
			return resRec;
