if (record.isNull() || tlvBufferPtr == nullptr || tlvDataLen == 0) {
				return resRec; // Return uninitialized result record
			}

			const uint8_t* nextRecordPtr = record.getBasePtr() + record.getSize();
			if (nextRecordPtr >= tlvBufferPtr + tlvDataLen) {
				return resRec; // Out of bounds, return uninitialized result record
			}

			resRec = getTLVRecord(nextRecordPtr, tlvBufferPtr, tlvDataLen);
			return resRec;
