if (record.isNull())
				return resRec;

			const uint8_t* nextRecordPtr = record.getRecordBasePtr() + record.getRecordTotalSize();
			size_t nextRecordOffset = nextRecordPtr - tlvBufferPtr;

			if (nextRecordOffset >= tlvDataLen)
				return resRec;
			
			size_t recordDataLen = tlvDataLen - nextRecordOffset;
			if (recordDataLen < sizeof(uint32_t)*2) // Type and Length fields
				return resRec;

			resRec.assign(nextRecordPtr, recordDataLen);
			if (!resRec.isValid())
				resRec.assign(NULL);

			return resRec; // for NRVO optimization
