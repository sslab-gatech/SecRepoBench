// Check if the input and output formats are uninitialized (set to zero).
if (*InputFmt == 0 && *OutputFormat == 0) {

    // Set input and output processing functions to default
    p->FromInput = DefaultInput;
    p->ToOutput = DefaultOutput;

    // Update flags to allow formatter change
    *dwFlags |= cmsFLAGS_CAN_CHANGE_FORMATTER;
}
else {

    // Obtain the appropriate formatters for both input and output based on current formats
    p->FromInput = _cmsGetFormatter(ContextID, *InputFmt, cmsFormatterInput, CMS_PACK_FLAGS_16BITS).Fmt16;
    p->ToOutput = _cmsGetFormatter(ContextID, *OutputFormat, cmsFormatterOutput, CMS_PACK_FLAGS_16BITS).Fmt16;

    // If the formatters are unavailable, signal an error, delete the transformation, and terminate the function
    if (p->FromInput == NULL || p->ToOutput == NULL) {

        cmsSignalError(ContextID, cmsERROR_UNKNOWN_EXTENSION, "Unsupported raster format");
        cmsDeleteTransform(p);
        return NULL;
    }

    // Determine bytes per pixel for the input format
    cmsUInt32Number BytesPerPix = _cmsGetBytesPerPix(*InputFmt);

    // If conditions are met, adjust flags to permit formatter changes
    if (BytesPerPix >= 2) {
        *dwFlags |= cmsFLAGS_CAN_CHANGE_FORMATTER;
    }
}
