static
void _cmsHandleExtraChannels(_cmsTRANSFORM* p,
                           const void* in,
                           void* out,
                           cmsUInt32Number PixelsPerLine,
                           cmsUInt32Number LineCount,
                           const cmsStride* Stride)
{
    cmsUInt32Number i, j, strideIn, strideOut;

    strideIn = 0;
    strideOut = 0;

    for (i = 0; i < LineCount; i++) {

           const cmsUInt8Number* accum = (cmsUInt8Number*) in + strideIn;
           cmsUInt8Number* output = (cmsUInt8Number*) out + strideOut;

           for (j = 0; j < PixelsPerLine; j++) {

                  accum = p->FromInput(p, accum, Stride->BytesPerPlaneIn);
                  output = p->ToOutput(p, output, Stride->BytesPerPlaneOut);
           }

           strideIn += Stride->BytesPerLineIn;
           strideOut += Stride->BytesPerLineOut;
    }
}
