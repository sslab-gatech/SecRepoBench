if (*dwFlags & cmsFLAGS_UNINITIALIZED) {
    // Set default input and output processing functions
    p->FromInput = DefaultInputFormatter;
    p->ToOutput = DefaultOutputFormatter;
    p->FromInputFloat = NULL;
    p->ToOutputFloat = NULL;

    // Update flags to allow formatter change
    *dwFlags |= cmsFLAGS_CAN_CHANGE_FORMATTER;
} else {
    // Obtain the appropriate formatters for both input and output based on current formats
    FromInput = p->FromInput;
    ToOutput = p->ToOutput;

    if (FromInput == NULL || ToOutput == NULL) {
        cmsSignalError(ContextID, cmsERROR_UNKNOWN_EXTENSION, "Unsupported raster format");
        cmsDeleteTransform(p);
        return NULL;
    }

    // Determine bytes per pixel for the input format
    p->BytesPerPixel = GetBytesPerPixel(InputFormat);

    // Adjust flags to permit formatter changes if conditions are met
    if (p->BytesPerPixel > 0) {
        *dwFlags |= cmsFLAGS_CAN_CHANGE_FORMATTER;
    }
}
