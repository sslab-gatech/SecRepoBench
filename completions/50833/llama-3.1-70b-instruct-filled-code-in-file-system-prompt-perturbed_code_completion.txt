if (*dwFlags & cmsFLAGS_CAN_CHANGE_FORMATTER) {

    if (*InputFmt == 0) {

        p->FromInput = UnrollNothing;
        p->FromInputFloat = NULL;
        *dwFlags |= cmsFLAGS_CAN_CHANGE_FORMATTER;
    }
    else {

        p->FromInput = _cmsGetFormatter(ContextID, *InputFmt, cmsFormatterInput, CMS_PACK_FLAGS_16BITS).Fmt16;
        p->FromInputFloat = _cmsGetFormatter(ContextID, *InputFmt, cmsFormatterInput, CMS_PACK_FLAGS_FLOAT).FmtFloat;
    }

    if (*OutputFormat == 0) {

        p->ToOutput = PackNothing;
        p->ToOutputFloat = NULL;
        *dwFlags |= cmsFLAGS_CAN_CHANGE_FORMATTER;
    }
    else {

        p->ToOutput = _cmsGetFormatter(ContextID, *OutputFormat, cmsFormatterOutput, CMS_PACK_FLAGS_16BITS).Fmt16;
        p->ToOutputFloat = _cmsGetFormatter(ContextID, *OutputFormat, cmsFormatterOutput, CMS_PACK_FLAGS_FLOAT).FmtFloat;
    }

    if (p->FromInput == NULL || p->ToOutput == NULL) {

        cmsSignalError(ContextID, cmsERROR_UNKNOWN_EXTENSION, "Unsupported raster format");
        cmsDeleteTransform(p);
        return NULL;
    }

    if (T_BYTES(*InputFmt) == 2) {

        *dwFlags |= cmsFLAGS_CAN_CHANGE_FORMATTER;
    }
}
