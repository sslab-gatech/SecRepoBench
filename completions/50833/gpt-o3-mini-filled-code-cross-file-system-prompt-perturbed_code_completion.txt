{
    if (*InputFmt == 0 || *OutputFormat == 0) {

        // Uninitialized: assign default 16-bit formats (constants are defined elsewhere)
        *InputFmt = DEFAULT_16_INPUT;
        *OutputFormat = DEFAULT_16_OUTPUT;

        p->FromInput = _cmsGetFormatter(ContextID, *InputFmt, cmsFormatterInput, CMS_PACK_FLAGS_16BITS).Fmt16;
        p->ToOutput  = _cmsGetFormatter(ContextID, *OutputFormat, cmsFormatterOutput, CMS_PACK_FLAGS_16BITS).Fmt16;

        *dwFlags |= cmsFLAGS_CAN_CHANGE_FORMATTER;
    }
    else {

        p->FromInput = _cmsGetFormatter(ContextID, *InputFmt, cmsFormatterInput, CMS_PACK_FLAGS_16BITS).Fmt16;
        p->ToOutput  = _cmsGetFormatter(ContextID, *OutputFormat, cmsFormatterOutput, CMS_PACK_FLAGS_16BITS).Fmt16;

        if (p->FromInput == NULL || p->ToOutput == NULL) {
            cmsSignalError(ContextID, cmsERROR_UNKNOWN_EXTENSION, "Unsupported raster format");
            cmsDeleteTransform(p);
            return NULL;
        }

        {
            // Determine bytes per pixel for the input format.
            // Assumption: the top byte encodes bytes per channel and the next byte encodes number of channels.
            cmsUInt32Number bytesPerChannel = ((*InputFmt) >> 24) & 0xFF;
            cmsUInt32Number nChannels       = ((*InputFmt) >> 16) & 0xFF;
            cmsUInt32Number bpp             = bytesPerChannel * nChannels;

            // If the total bytes per pixel is small (e.g. 2 bytes or less),
            // allow the formatter to be changed.
            if (bpp <= 2)
                *dwFlags |= cmsFLAGS_CAN_CHANGE_FORMATTER;
        }
    }
}
