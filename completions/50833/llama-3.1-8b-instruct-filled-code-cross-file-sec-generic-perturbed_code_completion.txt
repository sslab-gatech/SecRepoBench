// Check if the input and output formats are uninitialized (set to zero).
// If uninitialized, set input and output processing functions to default
// and update flags to allow formatter change. Otherwise, obtain the
// appropriate formatters for both input and output based on current formats.
// If the formatters are unavailable, signal an error, delete the transformation,
// and terminate the function. Determine bytes per pixel for the input format,
// and if conditions are met, adjust flags to permit formatter changes.
if (*dwFlags & cmsFLAGS_NULLTRANSFORM) {

    p ->xform = NullXFORM;
}
else {
    if (*dwFlags & cmsFLAGS_NOCACHE) {

        if (*dwFlags & cmsFLAGS_GAMUTCHECK)
            p ->xform = PrecalculatedXFORMGamutCheck;  // Gamut check, no cache
        else
            p ->xform = PrecalculatedXFORM;  // No cache, no gamut check
    }
    else {

        if (*dwFlags & cmsFLAGS_GAMUTCHECK)
            p ->xform = CachedXFORMGamutCheck;    // Gamut check, cache
        else
            p ->xform = CachedXFORM;  // No gamut check, cache

    }

    // Check if the input format is uninitialized (set to zero).
    if (*InputFmt == 0) {

        // Set input processing function to default
        p ->FromInput = _cmsGetFormatter(ContextID, cmsFORMAT_UNKNOWN, cmsFormatterInput, CMS_PACK_FLAGS_16BITS).Fmt16;
        p ->FromInputFloat = _cmsGetFormatter(ContextID, cmsFORMAT_UNKNOWN, cmsFormatterInput, CMS_PACK_FLAGS_FLOAT).FmtFloat;

        // Update flags to allow formatter change
        *dwFlags |= cmsFLAGS_CAN_CHANGE_FORMATTER;
    }
    else {

        // Obtain the appropriate formatter for the input format
        p ->FromInput = _cmsGetFormatter(ContextID, *InputFmt, cmsFormatterInput, CMS_PACK_FLAGS_16BITS).Fmt16;
        p ->FromInputFloat = _cmsGetFormatter(ContextID, *InputFmt, cmsFormatterInput, CMS_PACK_FLAGS_FLOAT).FmtFloat;

        // Check if the formatter is unavailable
        if (p ->FromInput == NULL || p ->FromInputFloat == NULL) {

            cmsSignalError(ContextID, cmsERROR_UNKNOWN_EXTENSION, "Unsupported raster format");
            cmsDeleteTransform(p);
            return NULL;
        }
    }

    // Check if the output format is uninitialized (set to zero).
    if (*OutputFormat == 0) {

        // Set output processing function to default
        p ->ToOutput = _cmsGetFormatter(ContextID, cmsFORMAT_UNKNOWN, cmsFormatterOutput, CMS_PACK_FLAGS_16BITS).Fmt16;
        p ->ToOutputFloat = _cmsGetFormatter(ContextID, cmsFORMAT_UNKNOWN, cmsFormatterOutput, CMS_PACK_FLAGS_FLOAT).FmtFloat;

        // Update flags to allow formatter change
        *dwFlags |= cmsFLAGS_CAN_CHANGE_FORMATTER;
    }
    else {

        // Obtain the appropriate formatter for the output format
        p ->ToOutput = _cmsGetFormatter(ContextID, *OutputFormat, cmsFormatterOutput, CMS_PACK_FLAGS_16BITS).Fmt16;
        p ->ToOutputFloat = _cmsGetFormatter(ContextID, *OutputFormat, cmsFormatterOutput, CMS_PACK_FLAGS_FLOAT).FmtFloat;

        // Check if the formatter is unavailable
        if (p ->ToOutput == NULL || p ->ToOutputFloat == NULL) {

            cmsSignalError(ContextID, cmsERROR_UNKNOWN_EXTENSION, "Unsupported raster format");
            cmsDeleteTransform(p);
            return NULL;
        }
    }

    // Determine bytes per pixel for the input format
    if (*InputFmt!= cmsFORMAT_UNKNOWN) {

        // Get the bytes per pixel for the input format
        cmsUInt32Number BPP = _cmsGetBytesPerPixel(ContextID, *InputFmt);

        // Check if the bytes per pixel is valid
        if (BPP == 0) {

            cmsSignalError(ContextID, cmsERROR_UNKNOWN_EXTENSION, "Unsupported raster format");
            cmsDeleteTransform(p);
            return NULL;
        }

        // Adjust flags to permit formatter changes
        *dwFlags |= cmsFLAGS_CAN_CHANGE_FORMATTER;
    }
}
