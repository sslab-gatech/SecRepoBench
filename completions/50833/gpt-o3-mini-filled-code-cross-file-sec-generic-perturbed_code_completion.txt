if (*InputFmt == 0 || *OutputFormat == 0) {
    // Uninitialized formats: assign a default 16-bit format and enable formatter change.
    *InputFmt  = TYPE_RGB_16;
    *OutputFormat = TYPE_RGB_16;
    *dwFlags   |= cmsFLAGS_CAN_CHANGE_FORMATTER;
}

p->FromInput = _cmsGetFormatter(ContextID, *InputFmt, cmsFormatterInput, CMS_PACK_FLAGS_16BITS).Fmt16;
p->ToOutput  = _cmsGetFormatter(ContextID, *OutputFormat, cmsFormatterOutput, CMS_PACK_FLAGS_16BITS).Fmt16;
if (p->FromInput == NULL || p->ToOutput == NULL) {
    cmsSignalError(ContextID, cmsERROR_UNKNOWN_EXTENSION, "Unsupported raster format");
    cmsDeleteTransform(p);
    return NULL;
}

/* Determine the bits per pixel of the input format.
   If it is lower than 16 bits, allow changes to the formatter later on. */
{
    cmsUInt32Number inBPP = _cmsFormatterBPP(*InputFmt);
    if (inBPP < 16)
         *dwFlags |= cmsFLAGS_CAN_CHANGE_FORMATTER;
}
