// <MASK>
if (input_string.empty()) {
    return;
}

// Remove leading spaces or new lines
input_string.erase(0, input_string.find_first_not_of(" \n"));

// Replace all forward slashes and backslashes with the operating system's directory separator
char dirSeparator = ASSIMP_PATHSEPARATOR;
for (size_t i = 0; i < input_string.length(); i++) {
    if (input_string[i] == '/' || input_string[i] == '\\') {
        input_string[i] = dirSeparator;
    }
}

// Reduce double delimiters to single delimiters
size_t pos = 0;
while ((pos = input_string.find(std::string(1, dirSeparator) + std::string(1, dirSeparator))) != std::string::npos) {
    input_string.erase(pos, 1);
}

// Handle special cases
pos = input_string.find("://");
if (pos != std::string::npos) {
    pos += 3;
    input_string = input_string.substr(0, pos) + input_string.substr(pos).replace(input_string.substr(pos).find(std::string(1, dirSeparator)), 1, std::string(1, dirSeparator));
}

pos = input_string.find("\\\\" + std::string(1, dirSeparator));
if (pos != std::string::npos) {
    input_string = input_string.substr(0, pos + 2) + input_string.substr(pos + 3);
}

// Decode URI-encoded characters
pos = 0;
while ((pos = input_string.find('%', pos)) != std::string::npos) {
    if (pos + 2 < input_string.length()) {
        std::string hexStr = input_string.substr(pos + 1, 2);
        char c = static_cast<char>(std::stoi(hexStr, nullptr, 16));
        input_string.replace(pos, 3, 1, c);
    }
    pos++;
}
