if (input_string.empty()) {
            return;
        }

        // Remove leading spaces or new lines
        size_t start = input_string.find_first_not_of(" \n\r\t");
        if (start != std::string::npos) {
            input_string = input_string.substr(start);
        }

        // Replace slashes and handle double delimiters
        std::string::iterator it = input_string.begin();
        while (it != input_string.end()) {
            if (*it == '/' || *it == '\\') {
                *it = getOsSeparator();
            }

            // Skip over "://" and "\\\\" sequences
            if ((it + 2) < input_string.end() && (*(it + 1) == ':' || *(it + 1) == '\\') && *(it + 2) == '/') {
                it += 3;
                continue;
            }

            // Reduce double delimiters to single
            if ((it + 1) != input_string.end() && *it == getOsSeparator() && *(it + 1) == getOsSeparator()) {
                it = input_string.erase(it);
                continue;
            }

            ++it;
        }

        // Decode hexadecimal sequences in URIs
        for (size_t i = 0; i < input_string.length(); ++i) {
            if (input_string[i] == '%' && i + 2 < input_string.length() && IsHex(input_string[i + 1]) && IsHex(input_string[i + 2])) {
                char hex[3] = { input_string[i + 1], input_string[i + 2], '\0' };
                input_string[i] = static_cast<char>(strtol(hex, nullptr, 16));
                input_string.erase(i + 1, 2);
            }
        }
