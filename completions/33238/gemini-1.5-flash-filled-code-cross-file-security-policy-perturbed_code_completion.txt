if (input_string.empty()) {
            return;
        }

        // Remove leading spaces and newlines
        size_t first_non_space = input_string.find_first_not_of(" \n\r");
        if (first_non_space != std::string::npos) {
            input_string = input_string.substr(first_non_space);
        } else {
            input_string = ""; // Handle case where string contains only spaces/newlines
            return;
        }

        // Replace slashes with OS-specific separator
        char separator = '/'; // Assuming forward slash as default
#ifdef _WIN32
        separator = '\\';
#endif
        for (size_t i = 0; i < input_string.length(); ++i) {
            if (input_string[i] == '/' || input_string[i] == '\\') {
                input_string[i] = separator;
            }
        }

        // Reduce double delimiters to single delimiters
        size_t pos = input_string.find(std::string(2, separator));
        while (pos != std::string::npos) {
            input_string.erase(pos, 1);
            pos = input_string.find(std::string(2, separator));
        }

        // Handle "://" and "\\\\" sequences
        if (input_string.size() >= 3 && input_string.substr(0, 3) == std::string("///")) {
            input_string.erase(0, 1);
        } else if (input_string.size() >= 3 && input_string.substr(0, 3) == std::string("///")) {
            input_string.erase(0, 1);
        }

        // Convert hexadecimal sequences in URIs
        size_t hex_pos = input_string.find('%');
        while (hex_pos != std::string::npos) {
            if (hex_pos + 2 < input_string.length()) {
                std::string hex_str = input_string.substr(hex_pos + 1, 2);
                try {
                    unsigned int hex_val = std::stoul(hex_str, nullptr, 16);
                    input_string.replace(hex_pos, 3, 1, static_cast<char>(hex_val));
                } catch (const std::invalid_argument& e) {
                    // Ignore invalid hex sequences
                }
            }
            hex_pos = input_string.find('%', hex_pos + 1);
