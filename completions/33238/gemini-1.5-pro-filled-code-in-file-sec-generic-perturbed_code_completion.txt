if (input_string.empty()) {
            return;
        }

        // remove leading spaces or new lines
        input_string = ai_str_ltrim_copy(input_string);

        // Replace slashes with the OS directory separator.
        for (std::string::iterator it = input_string.begin(); it != input_string.end(); ++it) {
            if ('/' == *it || '\\' == *it) {
                *it = mSep;
            }
        }

        // Ensure no double delimiters.
        std::string::size_type ss = input_string.find( mSep, 0 );
        while (std::string::npos != ss) {
            if (ss+1 < input_string.length() && input_string[ss+1] == mSep) {
                input_string.erase(ss,1);
                ss = input_string.find( mSep, ss );
            }
            else ++ss;
        }

        // skip over "://" and "\\\\"
        if (input_string.length() > 3 && input_string[1] == ':' && input_string[2] == '/' && input_string[3] == '/') {
            input_string.erase(0,2);
        }
        if (input_string.length() > 3 && input_string[0] == '\\' && input_string[1] == '\\' && input_string[2] == '\\' && input_string[3] == '\\') {
            input_string.erase(0,2);
        }

        // Convert hexadecimal sequences in URIs to their respective characters.
        for (size_t i = 0; i < input_string.length(); ++i) {
            if (input_string[i] == '%') {
                if (i + 2 < input_string.length() && IsHex(input_string[i+1]) && IsHex(input_string[i+2])) {
                    char hex[3] = {input_string[i+1], input_string[i+2], '\0'};
                    input_string.replace(i, 3, 1, (char)strtol(hex, nullptr, 16));
                }
            }
        }
