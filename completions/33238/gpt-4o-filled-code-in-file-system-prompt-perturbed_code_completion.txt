if (input_string.empty()) {
            return;
        }

        // Remove leading spaces or new lines
        input_string.erase(0, input_string.find_first_not_of(" \n\r\t"));

        std::string::iterator it = input_string.begin();
        std::string::iterator last = input_string.end();

        bool is_uri = false;
        while (it != last) {
            if (*it == '/' || *it == '\\') {
                *it = getOsSeparator();
            }

            // Check for "://" or "\\\\" sequences
            if (it + 2 < last && (*(it + 1) == ':' || *(it + 1) == '\\') && (*(it + 2) == '/' || *(it + 2) == '\\')) {
                is_uri = true;
                it += 3;
                continue;
            }

            // Reduce double delimiters to single delimiters
            if (it + 1 != last && *it == getOsSeparator() && *(it + 1) == getOsSeparator()) {
                it = input_string.erase(it);
                last = input_string.end();
                continue;
            }

            // Decode hexadecimal sequences in URIs
            if (is_uri && *it == '%' && it + 2 < last && IsHex(*(it + 1)) && IsHex(*(it + 2))) {
                char hex[3] = { *(it + 1), *(it + 2), '\0' };
                *it = static_cast<char>(strtol(hex, nullptr, 16));
                it = input_string.erase(it + 1, it + 3);
                last = input_string.end();
                continue;
            }

            ++it;
        }
