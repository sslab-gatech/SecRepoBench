if (input_string.empty()) {
            return;
        }

        // Remove leading spaces and newlines
        size_t start = input_string.find_first_not_of(" \r\n");
        if (start != std::string::npos) {
            input_string = input_string.substr(start);
        } else {
            input_string.clear();
            return;
        }

        // Replace forward and backward slashes with the OS directory separator
        std::string os_separator;
    #ifdef _WIN32
        os_separator = "\\";
    #else
        os_separator = "/";
    #endif

        for (size_t i = 0; i < input_string.length(); ++i) {
            if (input_string[i] == '/' || input_string[i] == '\\') {
                input_string[i] = os_separator[0];
            }
        }

        // Reduce multiple delimiters to a single one
        size_t i = 0;
        while (i + 1 < input_string.length()) {
            if (input_string[i] == os_separator[0] && input_string[i + 1] == os_separator[0]) {
                input_string.erase(i + 1, 1);
            } else {
                ++i;
            }
        }

        // Special cases: skip over "://" and "\\\\"
        i = 0;
        while (i + 2 < input_string.length()) {
            if (input_string[i + 1] == ':' && input_string[i + 2] == '/' && input_string[i] == '/') {
                i += 3;
            } else if (input_string[i + 1] == '\\' && input_string[i + 2] == '\\' && input_string[i] == '\\') {
                i += 3;
            } else {
                ++i;
            }
        }

        // Convert URI hexadecimal sequences (e.g., %20)
        i = 0;
        while (i + 2 < input_string.length()) {
            if (input_string[i] == '%') {
                if (std::isxdigit(input_string[i + 1]) && std::isxdigit(input_string[i + 2])) {
                    std::string hex_str = input_string.substr(i + 1, 2);
                    unsigned char char_code = static_cast<unsigned char>(std::strtoul(hex_str.c_str(), nullptr, 16));
                    input_string.replace(i, 3, 1, char_code);
                }
            }
            ++i;
        }
