if (input_string.empty()) {
        return;
    }

    // Remove leading spaces and new lines
    size_t start = input_string.find_first_not_of(" \n\r\t");
    if (start != std::string::npos) {
        input_string = input_string.substr(start);
    }

    // Replace slashes and handle special sequences
    std::string result;
    result.reserve(input_string.size());

    for (size_t i = 0; i < input_string.size(); ++i) {
        char c = input_string[i];

        // Handle "://" and "\\\\" sequences
        if ((i > 0 && input_string[i - 1] == ':' && c == '/') ||
            (i > 0 && input_string[i - 1] == '\\' && c == '\\')) {
            result += c;
            continue;
        }

        // Replace slashes with the system's directory separator
        if (c == '/' || c == '\\') {
            if (result.empty() || result.back() != '/') {
                result += '/';
            }
        } else if (c == '%' && i + 2 < input_string.size() &&
                   isxdigit(input_string[i + 1]) && isxdigit(input_string[i + 2])) {
            // Convert %XX to the corresponding character
            std::string hex = input_string.substr(i + 1, 2);
            char decoded_char = static_cast<char>(std::stoi(hex, nullptr, 16));
            result += decoded_char;
            i += 2; // Skip the next two hex digits
        } else {
            result += c;
        }
    }

    input_string = result;
