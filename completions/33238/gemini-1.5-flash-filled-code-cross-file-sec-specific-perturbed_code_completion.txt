if (input_string.empty()) {
            return;
        }
        // Remove leading spaces and newlines
        size_t first_char = input_string.find_first_not_of(" \n\r\t");
        if (first_char != std::string::npos) {
            input_string = input_string.substr(first_char);
        } else {
            input_string = ""; // Handle case where string contains only whitespace
            return;
        }

        std::string os_sep = "/"; // Use forward slash as OS separator for now.  Could be made more sophisticated.
        for (size_t i = 0; i < input_string.length(); ++i) {
            if (input_string[i] == '/' || input_string[i] == '\\') {
                input_string[i] = os_sep[0];
            }
        }

        // Reduce double delimiters to single delimiters
        size_t pos = input_string.find(os_sep + os_sep);
        while (pos != std::string::npos) {
            input_string.replace(pos, 2, os_sep);
            pos = input_string.find(os_sep + os_sep);
        }

        // Handle special cases like "://" and "\\\\"
        pos = input_string.find("://");
        if (pos != std::string::npos) {
            // Skip over "://" sequence
            size_t next_sep = input_string.find(os_sep, pos + 3);
            if (next_sep != std::string::npos) {
                //If there is another separator after ://, then we can skip over it.
                size_t len = next_sep - pos;
                input_string.replace(pos, len, "://");
            }
        }

        // Handle hexadecimal sequences in URIs
        pos = input_string.find('%');
        while (pos != std::string::npos) {
            if (pos + 2 < input_string.length()) {
                std::string hex_str = input_string.substr(pos + 1, 2);
                try {
                    unsigned int hex_val = std::stoul(hex_str, nullptr, 16);
                    input_string.replace(pos, 3, 1, static_cast<char>(hex_val));
                } catch (const std::invalid_argument& e) {
                    // Ignore invalid hex sequences
                }
            }
            pos = input_string.find('%', pos + 1);
