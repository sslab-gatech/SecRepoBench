void Cleanup(std::string& input_string) const {
    if (input_string.empty()) {
        return;
    }

    // Remove leading spaces or new lines
    input_string.erase(0, input_string.find_first_not_of(" \n\r\t"));

    // Replace forward slashes and backslashes with the OS's directory separator
    std::string::size_type pos = 0;
    while ((pos = input_string.find('/', pos)) != std::string::npos) {
        input_string[pos] = mSep;
        ++pos;
    }
    pos = 0;
    while ((pos = input_string.find('\\', pos)) != std::string::npos) {
        input_string[pos] = mSep;
        ++pos;
    }

    // Ensure double delimiters are reduced to single delimiters
    pos = 0;
    while ((pos = input_string.find(mSep + mSep, pos)) != std::string::npos) {
        input_string.erase(pos, 1);
    }

    // Detect and convert hexadecimal sequences in URIs to their respective characters
    pos = 0;
    while ((pos = input_string.find("%", pos)) != std::string::npos) {
        if (pos + 2 < input_string.size() && IsHex(input_string[pos + 1]) && IsHex(input_string[pos + 2])) {
            std::string hex_str = input_string.substr(pos + 1, 2);
            char c = static_cast<char>(std::strtol(hex_str.c_str(), nullptr, 16));
            input_string.replace(pos, 3, 1, c);
        }
        ++pos;
    }
}
