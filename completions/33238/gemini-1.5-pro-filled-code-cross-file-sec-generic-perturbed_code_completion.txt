if (input_string.empty()) {
            return;
        }

        // Remove leading spaces and newlines
        size_t first_char = input_string.find_first_not_of(" \r\n");
        if (first_char != std::string::npos) {
            input_string.erase(0, first_char);
        }

        size_t i = 0;
        while (i < input_string.length()) {
            if (input_string[i] == '/' || input_string[i] == '\\') {
                input_string[i] = directory_separator;
                // Skip over double delimiters
                while (i + 1 < input_string.length() && (input_string[i+1] == '/' || input_string[i+1] == '\\')) {
                    input_string.erase(i + 1, 1);
                }
            }
            // Handle URI and UNC path cases
            else if (input_string.substr(i, 3) == "://") {
                i += 2; // Skip over "://"
            }
            else if (input_string.substr(i, 4) == "\\\\") {
                i += 3; // Skip over "\\\\"
            }
            // Convert hexadecimal sequences in URIs to characters
            else if (input_string[i] == '%') {
                if (i + 2 < input_string.length()) {
                    char hex_str[3] = {input_string[i+1], input_string[i+2], '\0'};
                    char *endptr;
                    long int hex_value = strtol(hex_str, &endptr, 16);
                    if (*endptr == '\0' && hex_value >= 0 && hex_value <= 255) {
                        input_string.replace(i, 3, 1, static_cast<char>(hex_value));
                    }
                }
            }
            i++;
        }
