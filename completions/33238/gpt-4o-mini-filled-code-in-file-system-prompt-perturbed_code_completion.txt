if (input_string.empty()) {
            return;
        }

        // Remove leading spaces and new lines
        input_string.erase(0, input_string.find_first_not_of(" \n\r"));

        std::string::size_type pos = 0;
        while ((pos = input_string.find_first_of("/\\", pos)) != std::string::npos) {
            input_string.replace(pos, 1, std::string(1, mSep));
            ++pos;
        }

        // Remove duplicate separators
        while ((pos = input_string.find(mSep + std::string(1, mSep))) != std::string::npos) {
            input_string.erase(pos, 1);
        }

        // Handle special cases
        if (input_string.find("://") != std::string::npos) {
            // Skip over "://"
            pos = input_string.find("://") + 3;
        } else {
            pos = 0;
        }

        // Convert hexadecimal sequences
        while ((pos = input_string.find('%', pos)) != std::string::npos) {
            if (pos + 2 < input_string.length() && IsHex(input_string[pos + 1]) && IsHex(input_string[pos + 2])) {
                std::string hex = input_string.substr(pos + 1, 2);
                char ch = static_cast<char>(strtol(hex.c_str(), nullptr, 16));
                input_string.replace(pos, 3, 1, ch);
            } else {
                ++pos;
            }
        }
