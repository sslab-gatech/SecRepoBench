if (input_string.empty()) {
            return;
        }

        // Remove leading spaces and newlines
        size_t start = input_string.find_first_not_of(" \r\n");
        if (start != std::string::npos) {
            input_string = input_string.substr(start);
        } else {
            input_string.clear();
            return;
        }

        // Replace forward and backward slashes with the OS directory separator
        std::string delimiter(1, Assimp::IOStream::getOsSeparator());
        size_t pos = 0;
        while (pos < input_string.length()) {
            if (input_string[pos] == '/' || input_string[pos] == '\\') {
                input_string.replace(pos, 1, delimiter);
            }
            pos++;
        }

        // Reduce multiple delimiters to a single one
        pos = 0;
        while (pos < input_string.length() - 1) {
            if (input_string[pos] == Assimp::IOStream::getOsSeparator() && input_string[pos + 1] == Assimp::IOStream::getOsSeparator()) {
                input_string.replace(pos, 2, delimiter);
            } else if (pos > 0 && input_string[pos - 1] == ':' && input_string[pos] == Assimp::IOStream::getOsSeparator() && input_string[pos + 1] == Assimp::IOStream::getOsSeparator()) {
                pos += 2;
                continue;
            }
            else if (pos > 0 && input_string[pos - 1] == Assimp::IOStream::getOsSeparator() && input_string[pos] == Assimp::IOStream::getOsSeparator() && input_string[pos + 1] == Assimp::IOStream::getOsSeparator() && input_string[pos + 2] == Assimp::IOStream::getOsSeparator()) {
                pos += 4;
                continue;
            }
            else {
                pos++;
            }
        }

        // Decode URI-encoded characters (e.g., %20 for space)
        pos = 0;
        while ((pos = input_string.find('%', pos)) != std::string::npos) {
            if (pos + 2 < input_string.length() && isxdigit(input_string[pos + 1]) && isxdigit(input_string[pos + 2])) {
                std::string hex_str = input_string.substr(pos + 1, 2);
                unsigned char char_code = static_cast<unsigned char>(std::strtol(hex_str.c_str(), nullptr, 16));
                input_string.replace(pos, 3, 1, char_code);
            } else {
                pos++;
            }
        }
