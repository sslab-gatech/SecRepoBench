if (input_string.empty()) {
            return;
        }

        // Remove leading spaces and newlines
        size_t first_char = input_string.find_first_not_of(" \t\r\n");
        if (first_char != std::string::npos) {
            input_string = input_string.substr(first_char);
        }

        std::string os_sep = "/"; // Default to forward slash
#ifdef _WIN32
        os_sep = "\\"; // Use backslash on Windows
#endif

        // Replace slashes and handle double delimiters
        for (size_t i = 0; i < input_string.length(); ++i) {
            if (input_string[i] == '/' || input_string[i] == '\\') {
                input_string[i] = os_sep[0];
            }
            if (i > 0 && input_string[i] == os_sep[0] && input_string[i - 1] == os_sep[0]) {
                input_string.erase(i, 1);
                --i;
            }
            // Skip over "://" and "\\\\" sequences
            if (i > 1 && input_string[i] == os_sep[0] && input_string[i - 1] == os_sep[0] && input_string[i-2] == os_sep[0]) {
                input_string.erase(i, 1);
                --i;
            }
        }

        // Handle URI encoding (hexadecimal sequences)
        size_t pos = input_string.find('%');
        while (pos != std::string::npos) {
            if (pos + 2 < input_string.length()) {
                std::string hex_str = input_string.substr(pos + 1, 2);
                try {
                    unsigned int hex_val = std::stoul(hex_str, nullptr, 16);
                    input_string[pos] = static_cast<char>(hex_val);
                    input_string.erase(pos + 1, 2);
                } catch (const std::invalid_argument& e) {
                    // Ignore invalid hex sequences
                }
            }
            pos = input_string.find('%', pos + 1);
        }
