if (input_string.empty()) {
        return;
    }

    // Remove leading spaces or new lines from the path
    size_t start = input_string.find_first_not_of(" \n\r\t");
    if (start != std::string::npos) {
        input_string = input_string.substr(start);
    }

    // Replace both forward slashes ('/') and backslashes ('\\') with the operating system's directory separator
    std::string os_separator = (std::string(1, std::filesystem::path::preferred_separator));
    std::string::size_type pos = 0;
    while ((pos = input_string.find('/', pos)) != std::string::npos) {
        input_string.replace(pos, 1, os_separator);
        ++pos;
    }
    pos = 0;
    while ((pos = input_string.find('\\', pos)) != std::string::npos) {
        input_string.replace(pos, 1, os_separator);
        ++pos;
    }

    // Ensure double delimiters are reduced to single delimiters
    std::string double_delimiter = os_separator + os_separator;
    while ((pos = input_string.find(double_delimiter)) != std::string::npos) {
        input_string.replace(pos, double_delimiter.length(), os_separator);
    }

    // Handle special cases such as skipping over "://" and "\\\\" sequences
    std::string::size_type protocol_pos = input_string.find("://");
    if (protocol_pos != std::string::npos) {
        // Skip over "://"
        input_string = input_string.substr(protocol_pos + 3);
    }
    protocol_pos = input_string.find("\\\\");
    if (protocol_pos != std::string::npos) {
        // Skip over "\\\\"
        input_string = input_string.substr(protocol_pos + 2);
    }

    // Detect and convert hexadecimal sequences (e.g., %20) in URIs to their respective characters
    std::string::size_type hex_pos = 0;
    while ((hex_pos = input_string.find('%', hex_pos)) != std::string::npos) {
        if (hex_pos + 2 < input_string.size()) {
            std::string hex_code = input_string.substr(hex_pos + 1, 2);
            char ch = static_cast<char>(std::stoi(hex_code, nullptr, 16));
            input_string.replace(hex_pos, 3, 1, ch);
            hex_pos += 1; // Move past the converted character
        } else {
            break;
        }
    }
