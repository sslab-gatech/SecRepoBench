// Retrieve the 'type' key from the configuration section.
type = get_parser_key_type(config, cf, s, "type");
if (type == -1) {
    flb_error("[multiline_parser] no 'type' defined in file '%s'", cfg);
    goto fconf_error;
}

// Convert the type string to its corresponding integer value.
if (type == FLB_ML_REGEX) {
    type = FLB_ML_REGEX;
}
else if (type == FLB_ML_TIMEFLOW) {
    type = FLB_ML_TIMEFLOW;
}
else {
    flb_error("[multiline_parser] invalid 'type' defined in file '%s'", cfg);
    goto fconf_error;
}

// Proceed to handle the 'match_string' key retrieval.
match_string = get_parser_key(config, cf, s, "match_string");
if (!match_string) {
    flb_error("[multiline_parser] no 'match_string' defined in file '%s'", cfg);
    goto fconf_error;
}
