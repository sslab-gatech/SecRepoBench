flb_sds_t type_str = get_parser_key(config, cf, s, "type");
if (!type_str) {
    flb_error("[multiline_parser] no 'type' defined in file '%s'", cfg);
    goto fconf_error;
}

if (strcasecmp(type_str, "regex") == 0) {
    type = FLB_ML_REGEX;
}
else if (strcasecmp(type_str, "endswith") == 0) {
    type = FLB_ML_ENDSWITH;
}
else if (strcasecmp(type_str, "pattern") == 0) {
    type = FLB_ML_PATTERN;
}
else {
    flb_error("[multiline_parser] invalid 'type' in file '%s'", cfg);
    flb_sds_destroy(type_str);
    goto fconf_error;
}

flb_sds_destroy(type_str);
