bool_str = get_parser_key(config, cf, s, "type");
        if (!bool_str) {
            flb_error("[multiline_parser] no 'type' defined in file '%s'", cfg);
            goto fconf_error;
        }

        type = atoi(bool_str);
        flb_sds_destroy(bool_str);

        if (type < 0 || type >= FLB_ML_TYPE_MAX) { // Assuming FLB_ML_TYPE_MAX is defined
            flb_error("[multiline_parser] invalid 'type' value '%d' in file '%s'", type, cfg);
            goto fconf_error;
        }
