flb_sds_t type_str = get_parser_key(config, cf, s, "type");
    if (!type_str) {
        flb_error("[multiline_parser] no 'type' defined for parser '%s' in file '%s'", name, cfg);
        goto fconf_error;
    }
    char *endptr;
    long val = strtol(type_str, &endptr, 10);
    if (*endptr != '\0') {
        flb_error("[multiline_parser] invalid 'type' value '%s' for parser '%s' in file '%s'", type_str, name, cfg);
        flb_sds_destroy(type_str);
        goto fconf_error;
    }
    type = (int) val;
    flb_sds_destroy(type_str);
