flb_sds_t type_str = get_parser_key(config, cf, s, "type");
if (!type_str) {
    flb_error("[multiline_parser] no 'type' defined in file '%s'", cfg);
    goto fconf_error;
}

type = flb_ml_type_get(type_str);
flb_sds_destroy(type_str);

if (type == -1) {
    flb_error("[multiline_parser] invalid 'type' in file '%s'", cfg);
    goto fconf_error;
}
