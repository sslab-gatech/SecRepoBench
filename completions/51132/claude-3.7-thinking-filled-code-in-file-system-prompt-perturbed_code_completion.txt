/* type */
flb_sds_t type_str = get_parser_key(config, cf, s, "type");
if (!type_str) {
    flb_error("[multiline_parser] no 'type' defined for '%s' in file '%s'", name, cfg);
    goto fconf_error;
}

/* Convert type string to numeric value */
if (strcasecmp(type_str, "regex") == 0) {
    type = FLB_ML_REGEX;
}
else if (strcasecmp(type_str, "endswith") == 0) {
    type = FLB_ML_ENDSWITH;
}
else if (strcasecmp(type_str, "startswith") == 0) {
    type = FLB_ML_STARTSWITH;
}
else {
    flb_error("[multiline_parser] invalid type '%s' for '%s' in file '%s'", 
              type_str, name, cfg);
    flb_sds_destroy(type_str);
    goto fconf_error;
}

flb_sds_destroy(type_str);
