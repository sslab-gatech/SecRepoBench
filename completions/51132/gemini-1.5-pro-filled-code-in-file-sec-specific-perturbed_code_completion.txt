bool_str = get_parser_key(config, cf, s, "type");
        if (!bool_str) {
            flb_error("[multiline_parser] no 'type' defined for '%s' in file '%s'",
                      name, cfg);
            goto fconf_error;
        }
        if (strcasecmp(bool_str, "regex") == 0) {
            type = FLB_ML_REGEX;
        }
        else if (strcasecmp(bool_str, "endswith") == 0) {
            type = FLB_ML_ENDSWITH;
        }
        else {
            flb_error("[multiline_parser] invalid 'type=%s' for '%s' in file '%s'",
                      bool_str, name, cfg);
            flb_sds_destroy(bool_str);
            goto fconf_error;
        }
        flb_sds_destroy(bool_str);
