// Retrieve the 'type' key from the configuration section.
bool_str = get_parser_key(config, cf, s, "type");
if (!bool_str) {
    flb_error("[multiline_parser] no 'type' defined in file '%s'", cfg);
    goto fconf_error;
}

// Convert the type string to its corresponding integer value.
if (strcasecmp(bool_str, "regex") == 0) {
    type = FLB_ML_REGEX;
} else if (strcasecmp(bool_str, "wildcard") == 0) {
    type = FLB_ML_WILDCARD;
} else {
    flb_error("[multiline_parser] invalid 'type' value '%s' in file '%s'", bool_str, cfg);
    flb_sds_destroy(bool_str);
    goto fconf_error;
}

// Proceed to handle the'match_string' key retrieval.
flb_sds_destroy(bool_str);
