    type_str = get_parser_key(config, cf, s, "type");

        goto fconf_error;
    }
    type = atoi(type_str);
    flb_sds_destroy(type_str);
    if (type < 0 || type > FLB_ML_TYPE_MAX) {
        flb_error("[multiline_parser] invalid 'type' value '%d' in file '%s'",

        goto fconf_error;
    }
