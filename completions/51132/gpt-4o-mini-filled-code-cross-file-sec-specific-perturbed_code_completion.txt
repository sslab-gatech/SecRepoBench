bool_str = get_parser_key(config, cf, s, "type");
if (!bool_str) {
    flb_error("[multiline_parser] no 'type' defined in file '%s'", cfg);
    goto fconf_error;
}
type = atoi(bool_str);
flb_sds_destroy(bool_str);
if (type < 0) {
    flb_error("[multiline_parser] invalid 'type' value in file '%s'", cfg);
    goto fconf_error;
}
