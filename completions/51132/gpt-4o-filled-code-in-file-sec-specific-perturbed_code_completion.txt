/* type */
        tmp_str = get_parser_key(config, cf, s, "type");
        if (!tmp_str) {
            flb_error("[multiline_parser] no 'type' defined for '%s' in file '%s'", name, cfg);
            goto fconf_error;
        }

        if (strcasecmp(tmp_str, "regex") == 0) {
            type = FLB_ML_REGEX;
        }
        else if (strcasecmp(tmp_str, "multiline") == 0) {
            type = FLB_ML_MULTILINE;
        }
        else {
            flb_error("[multiline_parser] invalid 'type' for '%s' in file '%s'", name, cfg);
            flb_sds_destroy(tmp_str);
            goto fconf_error;
        }
        flb_sds_destroy(tmp_str);
