// Retrieve the 'type' key from the configuration section.
char *type_str = get_parser_key(config, cf, s, "type");
if (!type_str) {
    flb_error("[multiline_parser] no 'type' defined in file '%s'", cfg);
    goto fconf_error;
}

// Convert the type string to its corresponding integer value.
if (strcasecmp(type_str, "regex") == 0) {
    type = FLB_ML_REGEX;
} else if (strcasecmp(type_str, "docker") == 0) {
    type = FLB_ML_DOCKER;
} else {
    flb_error("[multiline_parser] invalid 'type' specified in file '%s'", cfg);
    flb_sds_destroy(type_str);
    goto fconf_error;
}

flb_sds_destroy(type_str);
type_str = NULL;
