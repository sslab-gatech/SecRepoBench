key_content = get_parser_key(config, cf, s, "type");
        if (!key_content) {
            flb_error("[multiline_parser] no 'type' defined for '%s' in file '%s'",
                      name, cfg);
            goto fconf_error;
        }

        if (strcasecmp(key_content, "regex") == 0) {
            type = FLB_ML_REGEX;
        }
        else if (strcasecmp(key_content, "pattern") == 0) {
            type = FLB_ML_PATTERN;
        }
        else {
            flb_error("[multiline_parser] invalid type '%s' for '%s' in file '%s'",
                      key_content, name, cfg);
            flb_sds_destroy(key_content);
            goto fconf_error;
        }
        flb_sds_destroy(key_content);
