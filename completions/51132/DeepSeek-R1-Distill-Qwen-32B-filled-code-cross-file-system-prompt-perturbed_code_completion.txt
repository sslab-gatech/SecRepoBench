flb_sds_t type_str;
    type_str = get_parser_key(config, cf, s, "type");
    if (!type_str) {
        flb_error("[multiline_parser] no 'type' defined in file '%s'", cfg);
        goto fconf_error;
    }

    if (strcasecmp(type_str, "regex") == 0) {
        type = FLB_ML_REGEX;
    } else {
        flb_error("[multiline_parser] invalid 'type' value '%s' in file '%s'", type_str, cfg);
        goto fconf_error;
    }

    flb_sds_destroy(type_str);
