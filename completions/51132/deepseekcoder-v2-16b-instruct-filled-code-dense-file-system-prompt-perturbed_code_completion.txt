        // Convert the type string to its corresponding integer value.

        type_str = get_parser_key(config, cf, s, "type");

            flb_error("[multiline_parser] no 'type' defined in file '%s'", cfg);

        }
        type = flb_ml_parser_type_from_string(type_str);
        if (type == FLB_ML_PARSER_TYPE_INVALID) {
