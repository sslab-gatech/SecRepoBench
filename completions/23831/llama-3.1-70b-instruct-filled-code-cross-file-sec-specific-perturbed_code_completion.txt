if (packet->payload_packet_len == 12) {
  ndpi_log_proto(ndpi_struct, flow, NDPI_PROTOCOL_IAX);
  ndpi_add_connection(ndpi_struct, flow, NDPI_PROTOCOL_IAX, NDPI_REAL_PROTOCOL);
} else {
  u_int16_t pos = 12;
  while (pos < packet->payload_packet_len) {
    u_int16_t element_length = ntohs(get_u_int16_t(packet->payload, pos + 2));
    if (pos + 4 + element_length > packet->payload_packet_len) {
      break;
    }
    pos += 4 + element_length;
    if (pos == packet->payload_packet_len) {
      ndpi_log_proto(ndpi_struct, flow, NDPI_PROTOCOL_IAX);
      ndpi_add_connection(ndpi_struct, flow, NDPI_PROTOCOL_IAX, NDPI_REAL_PROTOCOL);
    }
  }
}
