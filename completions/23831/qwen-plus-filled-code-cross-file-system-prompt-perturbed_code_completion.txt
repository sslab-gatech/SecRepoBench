if (packet->payload_packet_len == 12) {
    ndpi_int_iax_add_connection(ndpi_struct, flow, NDPI_PROTOCOL_IAX, NDPI_REAL_PROTOCOL);
} else {
    u_int16_t pos = 12;
    while (pos < packet->payload_packet_len) {
        u_int8_t info_element_length = packet->payload[pos++];
        if ((pos + info_element_length) > packet->payload_packet_len)
            break; // Invalid length, exit loop

        pos += info_element_length; // Move to the next information element
        if (pos == packet->payload_packet_len) {
            ndpi_int_iax_add_connection(ndpi_struct, flow, NDPI_PROTOCOL_IAX, NDPI_REAL_PROTOCOL);
            break;
        }
    }
}
