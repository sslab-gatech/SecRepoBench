  while (len < packet->payload_packet_len) {
    u_int8_t ie_type = packet->payload[len];
    u_int8_t ie_len = packet->payload[len + 1];
    len += 2 + ie_len;
    if (len == packet->payload_packet_len) {
      ndpi_set_detected_protocol(ndpi_struct, flow, NDPI_PROTOCOL_IAX, NDPI_PROTOCOL_UNKNOWN);
