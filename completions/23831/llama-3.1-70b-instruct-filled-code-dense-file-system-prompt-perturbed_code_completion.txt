if (packet->payload_packet_len == 12) {
  NDPI_LOG_INFO(ndpi_struct, "found IAX\n");
  ndpi_int_iax_add_connection(ndpi_struct, flow);
} else {
  u_int16_t pos = 12;
  while (pos < packet->payload_packet_len) {
    u_int16_t len = get_u_int16_t(packet->payload, pos + 1);
    if (pos + 3 + len == packet->payload_packet_len) {
      NDPI_LOG_INFO(ndpi_struct, "found IAX\n");
      ndpi_int_iax_add_connection(ndpi_struct, flow);
      break;
    }
    pos += 3 + len;
  }
}
