if (packet->payload_packet_len == 12) {
  ndpi_log_detection(ndpi_struct, flow, NDPI_PROTOCOL_IAX, NDPI_LOG_INFO);
  ndpi_add_connection(ndpi_struct, flow, NDPI_PROTOCOL_IAX, NDPI_CONNECTION_DIRECTION_BIDIRECTIONAL);
} else {
  u_int16_t pos = 12;
  while (pos < packet->payload_packet_len) {
    u_int8_t element_length = packet->payload[pos];
    pos += element_length + 1;
    if (pos == packet->payload_packet_len) {
      ndpi_log_detection(ndpi_struct, flow, NDPI_PROTOCOL_IAX, NDPI_LOG_INFO);
      ndpi_add_connection(ndpi_struct, flow, NDPI_PROTOCOL_IAX, NDPI_CONNECTION_DIRECTION_BIDIRECTIONAL);
      break;
    }
  }
}
