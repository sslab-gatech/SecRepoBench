if (packet->payload_packet_len == 12) {
  NDPI_LOG_INFO(ndpi_struct, "IAX detected\n");
  ndpi_int_iax_add_connection(ndpi_struct, flow);
} else {
  packetlength = 12;
  for (i = 0; i < NDPI_IAX_MAX_INFORMATION_ELEMENTS && packetlength < packet->payload_packet_len; i++) {
    if (packetlength + 2 > packet->payload_packet_len)
      break;
    packetlength += 2 + packet->payload[packetlength + 1];
    if (packetlength == packet->payload_packet_len) {
      NDPI_LOG_INFO(ndpi_struct, "IAX detected\n");
      ndpi_int_iax_add_connection(ndpi_struct, flow);
      return;
    }
  }
}
