if (packet->payload_packet_len == 12) {
  ndpi_log_proto(ndpi_struct, flow, NDPI_PROTOCOL_IAX, packet->payload_packet_len);
  ndpi_add_connection(ndpi_struct, flow, NDPI_PROTOCOL_IAX, packet->payload_packet_len);
} else {
  u_int16_t pos = 12;
  while (pos < packet->payload_packet_len) {
    u_int16_t len = ntohs(get_u_int16_t(packet->payload, pos + 2));
    if (pos + len + 4 == packet->payload_packet_len) {
      ndpi_log_proto(ndpi_struct, flow, NDPI_PROTOCOL_IAX, packet->payload_packet_len);
      ndpi_add_connection(ndpi_struct, flow, NDPI_PROTOCOL_IAX, packet->payload_packet_len);
      break;
    }
    pos += len + 4;
  }
}
