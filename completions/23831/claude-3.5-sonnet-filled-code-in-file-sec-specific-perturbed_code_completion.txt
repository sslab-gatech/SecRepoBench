if (packet->payload_packet_len == 12) {
  NDPI_LOG_INFO(ndpi_struct, "IAX detected\n");
  ndpi_int_iax_add_connection(ndpi_struct, flow);
} else {
  u_int16_t offset = 12;
  for (i = 0; i < NDPI_IAX_MAX_INFORMATION_ELEMENTS && offset < packet->payload_packet_len; i++) {
    u_int8_t element_type = packet->payload[offset];
    u_int16_t element_length = ntohs(get_u_int16_t(packet->payload, offset + 1));
    
    if (element_length == 0 || offset + 2 + element_length > packet->payload_packet_len) {
      break;
    }
    
    offset += 2 + element_length;
    
    if (offset == packet->payload_packet_len) {
      NDPI_LOG_INFO(ndpi_struct, "IAX detected\n");
      ndpi_int_iax_add_connection(ndpi_struct, flow);
      return;
    }
  }
}
