// Convert the `ckey` and given string to their UTF-16 representations if UTF-8 mode is enabled.
// Assign their lengths to `ckeyl` and `ctryl`. Assign the given string to `ctry`.
if (utf8) {
  u8_u16(ckey_utf16, ckey);
  u8_u16(ctry_utf16, ctry);
  ckeyl = ckey_utf16.size();
  ctryl = ctry_utf16.size();
} else {
  ckey_utf16.clear();
  ctry_utf16.clear();
  ckeyl = ckey.length();
  ctryl = ctry.length();
}
