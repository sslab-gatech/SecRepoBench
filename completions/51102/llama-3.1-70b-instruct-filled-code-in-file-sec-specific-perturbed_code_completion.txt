// Convert the `ckey` and given string to their UTF-16 representations if UTF-8 mode is enabled.
// Assign their lengths to `ckeyl` and `ctryl`. Assign the given string to `ctry`.
if (utf8) {
  ckey_utf = ckey;
  ckeyl = u8_u16(ckey_utf, ckey);
  ctry_utf = try_chars;
  ctryl = u8_u16(ctry_utf, try_chars);
  u8_u16(ctry_utf, try_chars);
} else {
  ckeyl = strlen(ckey);
  ctryl = strlen(try_chars);
  ctry = try_chars;
}
