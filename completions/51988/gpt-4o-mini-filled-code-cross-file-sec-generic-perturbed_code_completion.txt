if (word == nullptr || index < 0 || index >= strlen(word) - 1) {
    return 0; // Invalid input
  }

  char firstChar = word[index];
  char secondChar = word[index + 1];

  // Check if either character is a hyphen
  if (firstChar == '-' || secondChar == '-') {
    return 0; // Hyphen found, not a valid case pair
  }

  // Convert characters to uppercase
  char upperFirstChar = toupper(firstChar);
  char upperSecondChar = toupper(secondChar);

  // Check if the characters are unchanged when converted to uppercase
  if (firstChar == upperFirstChar || secondChar == upperSecondChar) {
    return 1; // Valid case pair found
  }

  return 0; // No valid case pair
