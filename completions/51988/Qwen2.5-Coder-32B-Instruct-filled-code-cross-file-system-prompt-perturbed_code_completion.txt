if (index >= 0 && index < static_cast<int>(strlen(word)) - 1) {
  char firstChar = word[index];
  char secondChar = word[index + 1];

  // Convert to uppercase and check if they remain unchanged
  bool isFirstUppercase = toupper(firstChar) == firstChar;
  bool isSecondUppercase = toupper(secondChar) == secondChar;

  // Check if neither character is a hyphen
  bool isNotHyphen = firstChar != '-' && secondChar != '-';

  if ((isFirstUppercase || isSecondUppercase) && isNotHyphen) {
    return 1;
  }
}
