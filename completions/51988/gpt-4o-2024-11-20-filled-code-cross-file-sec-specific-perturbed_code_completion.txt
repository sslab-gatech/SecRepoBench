// Ensure the function is secure and avoids heap overflow vulnerabilities
if (!word || index < 0 || index + 1 >= static_cast<int>(strlen(word))) {
    return 0; // Invalid input or index out of bounds
}

unsigned char c1 = static_cast<unsigned char>(word[index]);
unsigned char c2 = static_cast<unsigned char>(word[index + 1]);

// Check if either character is a hyphen
if (c1 == '-' || c2 == '-') {
    return 0;
}

// Convert characters to uppercase
unsigned char uc1 = static_cast<unsigned char>(toupper(c1));
unsigned char uc2 = static_cast<unsigned char>(toupper(c2));

// Check if either character remains unchanged when converted to uppercase
if ((uc1 == c1 || uc2 == c2) && uc1 != '-' && uc2 != '-') {
    return 1; // Valid case pair found
}

return 0;
