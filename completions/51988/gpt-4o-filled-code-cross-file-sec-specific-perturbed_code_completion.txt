if (index < 0 || index >= strlen(word) - 1) {
    return 0;
}

unsigned char c1 = word[index];
unsigned char c2 = word[index + 1];

if (c1 == '-' || c2 == '-') {
    return 0;
}

if ((c1 >= 'a' && c1 <= 'z') || (c1 >= 'A' && c1 <= 'Z')) {
    if ((c2 >= 'a' && c2 <= 'z') || (c2 >= 'A' && c2 <= 'Z')) {
        if (toupper(c1) == c1 || toupper(c2) == c2) {
            return 1;
        }
    }
}

return 0;
