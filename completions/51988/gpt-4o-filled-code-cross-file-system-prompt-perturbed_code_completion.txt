if (index < 0 || index >= strlen(word) - 1) {
    return 0;
  }

  unsigned char c1 = word[index];
  unsigned char c2 = word[index + 1];

  // Check if either character is a hyphen
  if (c1 == '-' || c2 == '-') {
    return 0;
  }

  // Convert to uppercase
  unsigned char uc1 = toupper(c1);
  unsigned char uc2 = toupper(c2);

  // Check if either character is already uppercase
  if (uc1 == c1 || uc2 == c2) {
    return 1;
  }

  return 0;
