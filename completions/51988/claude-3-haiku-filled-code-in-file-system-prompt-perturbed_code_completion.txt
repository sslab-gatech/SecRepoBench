int AffixMgr::cpdcase_check(const char* word, int index) {
    if (!utf8) {
        // Check if the two characters at the specified positions in the word form a valid case pair.
        // If the word is not in UTF-8 format, convert the characters to a Unicode representation.
        // Determine if either of the characters remains unchanged when converted to uppercase,
        // indicating they are already uppercase.
        // Ensure neither character is a hyphen ('-').
        // If the conditions are met, return 1 to indicate a match is found.
        char c1 = word[index - 1];
        char c2 = word[index];
        if (isupper(c1) && isupper(c2) && c1 != '-' && c2 != '-') {
            return 1;
        }
    } else {
        // Implement the UTF-8 version of the cpdcase_check logic
        std::vector<w_char> w;
        u8_u16(w, word);
        if (index > 0 && index < static_cast<int>(w.size())) {
            w_char c1 = w[index - 1];
            w_char c2 = w[index];
            if (isupper(c1.l) && isupper(c2.l) && c1.l != '-' && c2.l != '-') {
                return 1;
            }
        }
    }
    return 0;
}
