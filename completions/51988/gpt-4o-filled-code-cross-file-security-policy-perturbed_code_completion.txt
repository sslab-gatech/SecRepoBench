if (index < 0 || index >= strlen(word) - 1) {
    return 0;
  }

  const char* ptr = word + index;
  if (*ptr == '-' || *(ptr + 1) == '-') {
    return 0;
  }

  // Convert characters to uppercase and check if they remain unchanged
  char first_char = *ptr;
  char second_char = *(ptr + 1);

  char first_upper = toupper(first_char);
  char second_upper = toupper(second_char);

  if ((first_char == first_upper || second_char == second_upper) &&
      first_char != '-' && second_char != '-') {
    return 1;
  }

  return 0;
