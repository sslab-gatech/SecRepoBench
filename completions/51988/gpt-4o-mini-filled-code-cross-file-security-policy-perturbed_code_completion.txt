if (index < 0 || index >= strlen(word) - 1) {
    return 0; // Out of bounds
}

char first_char = word[index];
char second_char = word[index + 1];

if (first_char == '-' || second_char == '-') {
    return 0; // Invalid case: one of the characters is a hyphen
}

// Check if the characters are uppercase or unchanged when converted
if (first_char == toupper(first_char) && second_char == toupper(second_char)) {
    return 1; // Both characters are uppercase
}

return 0; // No match found
