// <MASK>
int casevalue = 0;
if (utf8) {
  w_char wc1, wc2;
  u8_u16(wc1, word, index);
  u8_u16(wc2, word, index + 1);
  if ((wc1[0] == wc1[1]) || (wc2[0] == wc2[1])) {
    casevalue = 1;
  }
} else {
  if ((word[index] == toupper(word[index])) || (word[index + 1] == toupper(word[index + 1]))) {
    casevalue = 1;
  }
}
if ((word[index] != '-') && (word[index + 1] != '-')) {
  return casevalue;
}
return 0;
