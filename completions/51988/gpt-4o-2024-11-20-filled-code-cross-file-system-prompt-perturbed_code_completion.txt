if (index < 1 || !word || word[index] == '-' || word[index - 1] == '-') {
    return 0;
  }

  unsigned int c1 = (unsigned char)word[index - 1];
  unsigned int c2 = (unsigned char)word[index];

  if (utf8) {
    w_char u1, u2;
    int len1 = u8_u16(&u1, word + index - 1);
    int len2 = u8_u16(&u2, word + index);

    if (len1 > 0 && len2 > 0) {
      c1 = u1;
      c2 = u2;
    }
  }

  if (c1 == toupper(c1) && c2 == toupper(c2)) {
    return 1;
  }

  return 0;
