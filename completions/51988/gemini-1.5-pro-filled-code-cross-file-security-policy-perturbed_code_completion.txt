if (index < 0 || index >= static_cast<int>(strlen(word)) - 1) {
    return 0;
  }

  const char* p1 = word + index;
  const char* p2 = word + index + 1;

  if (*p1 == '-' || *p2 == '-') {
    return 0;
  }

  int uc1 = mytoupper(*p1);
  int uc2 = mytoupper(*p2);

  if (uc1 == *p1 || uc2 == *p2)
      return 1;
