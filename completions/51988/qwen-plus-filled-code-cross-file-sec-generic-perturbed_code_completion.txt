if (word == nullptr || index < 0 || static_cast<size_t>(index) >= strlen(word)) {
  return 0;
}

// Ensure that the word is in UTF-8 format and convert characters to Unicode
unsigned int char1 = utf8_codepoint(word, index);
unsigned int char2 = utf8_codepoint(word, index + 1);

// Check if either character remains unchanged when converted to uppercase
bool char1_upper = (toupper_unicode(char1) == char1);
bool char2_upper = (toupper_unicode(char2) == char2);

// Ensure neither character is a hyphen
if (char1 != '-' && char2 != '-' && char1_upper && char2_upper) {
  return 1;
}

return 0;
