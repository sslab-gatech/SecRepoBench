// Convert the characters to Unicode representation
  w_char c1 = word[index];
  w_char c2 = word[index + 1];
  if (!utf8 || !isutf8(c1) || !isutf8(c2)) {
    return 0;
  }

  // Determine if either of the characters remains unchanged when converted to uppercase
  if (toupper(c1) == c1 || toupper(c2) == c2) {
    return 0;
  }

  if (c1 == '-' || c2 == '-') {

  }
