if (word[index] == '-' || word[index + 1] == '-') {
    return 0; // Invalid case: one of the characters is a hyphen
  }

  char firstChar = word[index];
  char secondChar = word[index + 1];

  // Convert to uppercase and check if they remain unchanged
  char upperFirstChar = toupper(firstChar);
  char upperSecondChar = toupper(secondChar);

  if (firstChar == upperFirstChar && secondChar == upperSecondChar) {
    return 1; // Both characters are already uppercase
  }

  return 0; // No match found
