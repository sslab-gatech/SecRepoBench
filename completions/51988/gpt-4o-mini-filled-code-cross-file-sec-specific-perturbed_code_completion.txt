if (word == nullptr || index < 0 || index >= strlen(word) - 1) {
    return 0; // Invalid input
  }

  char firstChar = word[index];
  char secondChar = word[index + 1];

  // Check if both characters are not hyphens
  if (firstChar == '-' || secondChar == '-') {
    return 0;
  }

  // Check if both characters are uppercase or if they remain unchanged when converted to uppercase
  if (isupper(firstChar) && isupper(secondChar)) {
    return 1;
  }

  // Check if they remain unchanged when converted to uppercase
  if (toupper(firstChar) == firstChar && toupper(secondChar) == secondChar) {
    return 1;
  }

  return 0;
