if (word == nullptr || index < 0 || index >= strlen(word)) {
  return 0;
}

char first_char = word[index];
char second_char = word[index + 1];

if (first_char == '-' || second_char == '-') {
  return 0;
}

unsigned char uc_first = static_cast<unsigned char>(first_char);
unsigned char uc_second = static_cast<unsigned char>(second_char);

if (toupper(uc_first) == uc_first && toupper(uc_second) == uc_second) {
  return 1;
}

return 0;
