if (method != NULL && methodlen > 0 && methodlen < sizeof(flow->thrift.method)) {
    strncpy(flow->thrift.method, method, methodlen);
    flow->thrift.method[methodlen] = '\0';
  } else {
    NDPI_LOG_INFO(ndpi_struct, "Invalid Thrift method, setting risk flag.\n");
    flow->risk |= NDPI_RISK_THRIFT_INVALID_METHOD;
    flow->thrift.method[0] = '\0';
  }
