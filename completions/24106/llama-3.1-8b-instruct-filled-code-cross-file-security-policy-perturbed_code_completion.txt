{
  /*
    Class context.
  */
  for (p=q; *q!= '\0'; )
  {
    if (GetNextToken(q,&q,extent,token) < 1)
      break;
    if (LocaleCompare(token,"pop")!= 0)
      continue;
    (void) GetNextToken(q,(const char **) NULL,extent,token);
    if (LocaleCompare(token,"class")!= 0)
      continue;
    break;
  }
  (void) GetNextToken(q,&q,extent,token);
  break;
}
