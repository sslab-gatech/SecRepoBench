import numpy as np
from PIL import Image

def fill_masked_region(image, mask, color):
    """
    Fill the masked region of an image with a given color.

    :param image: PIL Image object
    :param mask: PIL Image object containing the mask (1 for masked, 0 for not masked)
    :param color: Tuple (R, G, B) representing the color to fill the masked region
    :return: PIL Image object with the masked region filled with the given color
    """
    # Convert the image and mask to numpy arrays
    image_array = np.array(image)
    mask_array = np.array(mask)

    # Get the shape of the image and mask
    height, width, _ = image_array.shape

    # Create a new numpy array with the same shape as the image, filled with the given color
    filled_array = np.zeros((height, width, 3), dtype=np.uint8)
    filled_array[:] = color

    # Apply the mask to the filled array
    filled_array[mask_array == 0] = image_array[mask_array == 0]

    # Convert the filled array back to a PIL Image object
    filled_image = Image.fromarray(filled_array, mode='RGB')

    return filled_image
