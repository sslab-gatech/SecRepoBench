extent=GetQuantumExtent(image,quantumdata,quantumdata->format);
  if (extent == 0)
    return(MagickFalse);
  quantumdata->pixels=(MemoryInfo **) RelinquishMagickMemory(quantumdata->pixels);
  quantumdata->pixels=(MemoryInfo **) AcquireQuantumMemory(quantumdata->number_threads,sizeof(*quantumdata->pixels));
  if (quantumdata->pixels == (MemoryInfo **) NULL)
    return(MagickFalse);
  for (size_t i = 0; i < quantumdata->number_threads; i++)
  {
    quantumdata->pixels[i]=AcquireVirtualMemory(extent,sizeof(unsigned char));
    if (quantumdata->pixels[i] == (MemoryInfo *) NULL)
    {
      DestroyQuantumPixels(quantumdata);
      return(MagickFalse);
    }
  }
  return(MagickTrue);
