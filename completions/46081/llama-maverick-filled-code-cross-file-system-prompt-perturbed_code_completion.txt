quantum=GetQuantumRange(quantumdata->depth);
  extent=GetQuantumExtent(image,quantumdata,quantum);
  if (quantumdata->pixels == NULL)
    {
      quantumdata->pixels=AcquireAlignedMemory(1,(image->columns*
        image->rows+quantumdata->pad)*extent);
      if (quantumdata->pixels == NULL)
        return(MagickFalse);
    }
  else
    if (quantumdata->extent < extent)
      {
        quantumdata->pixels=ResizeQuantumMemory(quantumdata,
          quantumdata->pixels,1,(image->columns*image->rows+
          quantumdata->pad)*extent);
        if (quantumdata->pixels == NULL)
          return(MagickFalse);
      }
  quantumdata->extent=extent;
  return(MagickTrue);
