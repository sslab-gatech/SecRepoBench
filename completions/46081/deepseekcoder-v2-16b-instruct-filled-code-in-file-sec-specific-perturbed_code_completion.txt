extent=(size_t) (packet_size*image->columns*((quantum_info->depth+7)/8));
  if (extent > quantum_info->number_threads * quantum_info->extent)
    {
      quantum_info->pixels = (MemoryInfo **) RelinquishMagickMemory(quantum_info->pixels);
      quantum_info->pixels = (MemoryInfo **) AcquireQuantumMemory(quantum_info->number_threads, sizeof(*quantum_info->pixels));
      if (quantum_info->pixels == (MemoryInfo **) NULL)
        return(MagickFalse);
      for (i = 0; i < (ssize_t) quantum_info->number_threads; i++)
        {
          unsigned char *pixels;
          quantum_info->pixels[i] = AcquireVirtualMemory(extent + 1, sizeof(*pixels));
          if (quantum_info->pixels[i] == (MemoryInfo *) NULL)
            {
              DestroyQuantumPixels(quantum_info);
              return(MagickFalse);
            }
          pixels = (unsigned char *) GetVirtualMemoryBlob(quantum_info->pixels[i]);
          (void) memset(pixels, 0, (extent + 1) * sizeof(*pixels));
          pixels[extent] = QuantumSignature;
        }
    }
  return(MagickTrue);
}
