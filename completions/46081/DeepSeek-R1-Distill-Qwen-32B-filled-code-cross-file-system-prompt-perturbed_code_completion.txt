extent=image->channels * image->padding * quantumdata->depth;
  quantum=image->width * image->height * extent;
  if (quantumdata->extent < quantum)
    {
      void
        *pixels;

      pixels=MagickMalloc(quantum);
      if (pixels == (void *) NULL)
        return(MagickFalse);
      quantumdata->pixels=(unsigned char *) pixels;
      quantumdata->extent=quantum;
    }
  return(MagickTrue);
