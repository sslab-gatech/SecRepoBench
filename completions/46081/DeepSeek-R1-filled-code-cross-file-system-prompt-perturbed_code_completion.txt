if (quantumdata->format == FloatingPointQuantumFormat)
    quantumdata->depth=(quantumdata->depth <= 32) ? 32 : 64;
  quantum=image->number_channels*quantumdata->depth;
  quantumdata->pad=(size_t) ((quantum*image->columns % 32) != 0 ? 32-
    (quantum*image->columns % 32) : 0);
  extent=((quantum*image->columns+quantumdata->pad+7)/8)*image->rows;
  if (quantumdata->extent >= extent)
    return(MagickTrue);
  if (quantumdata->pixels != (unsigned char *) NULL)
    quantumdata->pixels=(unsigned char *) ResizeQuantumMemory(
      quantumdata->pixels,extent,sizeof(*quantumdata->pixels));
  else
    quantumdata->pixels=(unsigned char *) AcquireQuantumMemory(extent,
      sizeof(*quantumdata->pixels));
  if (quantumdata->pixels == (unsigned char *) NULL)
    return(MagickFalse);
  quantumdata->extent=extent;
  return(MagickTrue);
