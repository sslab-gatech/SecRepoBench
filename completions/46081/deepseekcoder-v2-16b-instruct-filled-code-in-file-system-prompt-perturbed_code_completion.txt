extent=(size_t) (packet_size*image->columns*((quantum_info->depth+7)/8));
  if (extent > quantum_info->extent)
    {
      quantum_info->extent=extent;
      DestroyQuantumPixels(quantum_info);
      if (AcquireQuantumPixels(quantum_info,extent) == MagickFalse)
        return(MagickFalse);
    }
  return(MagickTrue);
