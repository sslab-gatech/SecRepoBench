line.clear();
    if (m_cachePos >= m_cacheSize || m_filePos == 0) {
        if (!readNextBlock()) {
            return false;
        }
    }

    while (true) {
        if (IsLineEnd(m_cache[m_cachePos])) {
            ++m_cachePos;
            break;
        }

        line.push_back(m_cache[m_cachePos]);
        ++m_cachePos;

        if (m_cachePos >= m_cacheSize) {
            if (!readNextBlock()) {
                return false;
            }
        }
    }

    return true;
