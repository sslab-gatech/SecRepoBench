if (m_cachePos >= m_cache.size()) {
        if (!readNextBlock()) {
            return false;
        }
    }

    size_t i = m_cachePos;
    size_t j = 0;
    while (i < m_cache.size()) {
        if (m_cache[i] == '\n') {
            m_cachePos = i + 1;
            break;
        }
        line.push_back(m_cache[i]);
        ++i;
        ++j;
    }
    if (i == m_cache.size() && j > 0) {
        m_cachePos = i;
    }
    return j > 0;
