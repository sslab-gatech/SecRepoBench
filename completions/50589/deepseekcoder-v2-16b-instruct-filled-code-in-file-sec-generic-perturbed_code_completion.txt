template <class T>
AI_FORCE_INLINE bool IOStreamBuffer<T>::getNextLine(std::vector<T> &line) {
    line.clear();
    while (true) {
        if (isEndOfCache(m_cachePos, m_cacheSize)) {
            if (!readNextBlock()) {
                return false;
            }
        }

        size_t startPos = m_cachePos;
        while (m_cachePos < m_cacheSize && !IsLineEnd(m_cache[m_cachePos])) {
            ++m_cachePos;
        }

        if (m_cachePos < m_cacheSize) {
            line.insert(line.end(), m_cache.begin() + startPos, m_cache.begin() + m_cachePos);
            ++m_cachePos;
            return true;
        } else {
            line.insert(line.end(), m_cache.begin() + startPos, m_cache.end());
            if (!readNextBlock()) {
                return !line.empty();
            }
        }
    }
}
