template <class T>
AI_FORCE_INLINE bool IOStreamBuffer<T>::getNextLine(std::vector<T> &line) {
    line.clear();
    if (m_cachePos >= m_cacheSize || m_filePos == 0) {
        if (!readNextBlock()) {
            return false;
        }
    }

    size_t i = 0;
    while (true) {
        if (IsLineEnd(m_cache[m_cachePos])) {
            ++m_cachePos;
            break;
        }

        if (i >= line.size()) {
            line.resize(line.size() + 1);
        }

        line[i] = m_cache[m_cachePos];
        ++m_cachePos;
        ++i;

        if (m_cachePos >= m_cacheSize) {
            if (!readNextBlock()) {
                break;
            }
        }
    }

    if (i == 0 && m_cachePos >= m_cacheSize) {
        return false;
    }

    line.resize(i);
    return true;
}
