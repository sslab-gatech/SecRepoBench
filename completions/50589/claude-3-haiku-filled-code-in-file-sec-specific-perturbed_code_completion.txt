AI_FORCE_INLINE bool IOStreamBuffer<T>::getNextLine(std::vector<T> &line) {
    line.clear();
    while (true) {
        if (m_cachePos >= m_cacheSize) {
            if (!readNextBlock()) {
                return line.empty();
            }
        }

        size_t i = m_cachePos;
        while (i < m_cacheSize && !IsLineEnd(m_cache[i])) {
            ++i;
        }

        line.insert(line.end(), m_cache.begin() + m_cachePos, m_cache.begin() + i);
        m_cachePos = i;

        if (IsLineEnd(m_cache[m_cachePos])) {
            ++m_cachePos;
            return true;
        }
    }
}
