    while (true) {
        // Check if the cache is exhausted
        if (m_cachePos >= m_cache.size()) {
            if (!readNextBlock()) {
                // End of stream reached
                return !line.empty();
            }
        }

        // Read characters from the cache
        while (m_cachePos < m_cache.size()) {
            T c = m_cache[m_cachePos++];
            if (c == '\n') {
                // End of line reached
                return true;
            }
            if (c != '\r') {
                // Add character to the line, ignoring carriage returns
                line.push_back(c);
            }
        }
    }
