line.clear();
    if (m_cachePos >= m_cacheSize || 0 == m_filePos) {
        if (!readNextBlock()) {
            return false;
        }
    }

    size_t i = 0;
    while (true) {
        if (IsLineEnd(m_cache[m_cachePos])) {
            ++m_cachePos;
            break;
        }

        if (i >= line.size()) {
            line.resize(line.size() * 2 + 1);
        }

        line[i++] = m_cache[m_cachePos++];
        
        if (m_cachePos >= m_cacheSize) {
            if (!readNextBlock()) {
                break;
            }
        }
    }

    line.resize(i);
    return !line.empty();
