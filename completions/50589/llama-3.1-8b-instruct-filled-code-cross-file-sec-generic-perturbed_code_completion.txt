AI_FORCE_INLINE bool IOStreamBuffer<T>::getNextLine(std::vector<T> &line) {
    // Check if the cache is exhausted and requires refilling
    if (0 == m_cachePos) {
        if (!readNextBlock()) {
            return false;
        }
    }

    // Return the last block-value if getNextLine was used before
    if (0!= m_cachePos) {
        line = std::vector<T>(m_cache.begin() + m_cachePos, m_cache.end());
        m_cachePos = 0;
    } else {
        line = std::vector<T>(m_cache.begin(), m_cache.end());
    }

    // Skip end-of-line characters
    while (m_cachePos < line.size() && std::isspace(line[m_cachePos])) {
        ++m_cachePos;
    }

    // Resize the buffer dynamically if needed
    if (m_cachePos < line.size()) {
        line.resize(m_cachePos);
    }

    return true;
}
