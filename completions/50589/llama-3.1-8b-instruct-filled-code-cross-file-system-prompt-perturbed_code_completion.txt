bool IOStreamBuffer<T>::getNextLine(std::vector<T> &line) {
    // If the cache is not empty, return the last block-value
    if (0!= m_cachePos) {
        line = std::vector<T>(m_cache.begin() + m_cachePos, m_cache.end());
        m_cachePos = 0;
        return true;
    } else {
        // If the cache is empty, read the next block
        if (!readNextBlock()) {
            return false;
        }

        // If the block is empty, return false
        if (m_cache.empty()) {
            return false;
        }

        // Otherwise, return the block
        line = std::vector<T>(m_cache.begin(), m_cache.end());
        return true;
    }
}
