line.clear();
    if (m_cachePos >= m_cacheSize || 0 == m_filePos) {
        if (!readNextBlock()) {
            return false;
        }
    }

    while (!IsLineEnd(m_cache[m_cachePos]) && m_cachePos < m_cacheSize) {
        line.push_back(m_cache[m_cachePos]);
        ++m_cachePos;
        if (m_cachePos >= m_cacheSize) {
            if (!readNextBlock()) {
                break;
            }
        }
    }
    if (m_cachePos < m_cacheSize) {
        ++m_cachePos;
    }
    return line.size() > 0;
}
