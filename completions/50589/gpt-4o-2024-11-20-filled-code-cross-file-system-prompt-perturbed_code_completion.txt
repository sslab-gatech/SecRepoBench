line.clear();

    while (true) {
        // Check if the cache is exhausted
        if (m_cachePos >= m_cache.size()) {
            if (!readNextBlock()) {
                // End of stream, return false if no data was read
                return !line.empty();
            }
        }

        // Process the current character in the cache
        T currentChar = m_cache[m_cachePos++];
        if (currentChar == '\n') {
            // End of line reached, return true
            return true;
        } else if (currentChar != '\r') {
            // Add character to the line, skipping '\r'
            line.push_back(currentChar);
        }
    }
