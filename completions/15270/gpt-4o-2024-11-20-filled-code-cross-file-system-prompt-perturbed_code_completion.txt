if (codeccontext->codec_type == AVMEDIA_TYPE_SUBTITLE) {
            if (codeccontext->sub_charenc) {
                if (codeccontext->codec->caps_internal & FF_CODEC_CAP_BITMAP_SUB) {
                    av_log(codeccontext, AV_LOG_WARNING,
                           "Character encoding %s is ignored for bitmap-based subtitles\n",
                           codeccontext->sub_charenc);
                    codeccontext->sub_charenc = NULL;
                } else {
                    if (codeccontext->sub_charenc_mode == FF_SUB_CHARENC_MODE_AUTOMATIC)
                        codeccontext->sub_charenc_mode = FF_SUB_CHARENC_MODE_PRE_DECODER;

                    if (codeccontext->sub_charenc_mode != FF_SUB_CHARENC_MODE_DO_NOTHING) {
#if CONFIG_ICONV
                        codeccontext->internal->iconv_ctx = av_iconv_open("UTF-8", codeccontext->sub_charenc);
                        if (!codeccontext->internal->iconv_ctx) {
                            av_log(codeccontext, AV_LOG_ERROR,
                                   "Unable to open iconv context with character encoding %s\n",
                                   codeccontext->sub_charenc);
                            ret = AVERROR(EINVAL);
                            goto free_and_end;
                        }
#else
                        av_log(codeccontext, AV_LOG_ERROR,
                               "Character encoding %s requires a libavcodec build with iconv support\n",
                               codeccontext->sub_charenc);
                        ret = AVERROR(ENOSYS);
                        goto free_and_end;
#endif
                    }
                }
            }
        }
