if (codeccontext->codec_type == AVMEDIA_TYPE_SUBTITLE) {
    if (codeccontext->sub_charenc_mode == AV_SUB_CHARENC_MODE_AUTO) {
        codeccontext->sub_charenc_mode = AV_SUB_CHARENC_MODE_PRE_DECODER;
    }

    if (codeccontext->sub_charenc[0] && codeccontext->sub_charenc[0] != ' ') {
        if (codeccontext->codec_descriptor->props & AV_CODEC_PROP_BITMAP_SUB) {
            av_log(codeccontext, AV_LOG_WARNING,
                   "Codec '%s' is bitmap-based, ignoring character encoding '%s'\n",
                   codeccontext->codec_descriptor->name, codeccontext->sub_charenc);
        } else if (CONFIG_ICONV) {
            iconv_t cd = iconv_open("UTF-8", codeccontext->sub_charenc);
            if (cd == (iconv_t)-1) {
                av_log(codeccontext, AV_LOG_ERROR,
                       "Subtitle character encoding '%s' is not supported\n",
                       codeccontext->sub_charenc);
                ret = AVERROR(EINVAL);
                goto free_and_end;
            }
            iconv_close(cd);
        }
    }
}
