if (codeccontext->codec_type == AVMEDIA_TYPE_SUBTITLE) {
            if (codeccontext->subtitle_charenc) {
                if (codeccontext->codec->caps_internal & FF_CODEC_CAP_BITMAP_SUB) {
                    av_log(codeccontext, AV_LOG_WARNING,
                           "Character encoding is ignored for bitmap-based subtitle codec '%s'\n",
                           codeccontext->codec->name);
                } else {
                    if (!codeccontext->sub_charenc_mode)
                        codeccontext->sub_charenc_mode = FF_SUB_CHARENC_MODE_PRE_DECODER;

                    if (codeccontext->sub_charenc_mode == FF_SUB_CHARENC_MODE_PRE_DECODER) {
#if CONFIG_ICONV
                        codeccontext->sub_charenc = av_strdup(codeccontext->subtitle_charenc);
                        if (!codeccontext->sub_charenc) {
                            ret = AVERROR(ENOMEM);
                            goto free_and_end;
                        }
#else
                        av_log(codeccontext, AV_LOG_ERROR,
                               "Character encoding support requires iconv to be enabled in the build.\n");
                        ret = AVERROR(EINVAL);
                        goto free_and_end;
#endif
                    } else {
                        av_log(codeccontext, AV_LOG_ERROR,
                               "Invalid subtitle character encoding mode.\n");
                        ret = AVERROR(EINVAL);
                        goto free_and_end;
                    }
                }
            }
        }
