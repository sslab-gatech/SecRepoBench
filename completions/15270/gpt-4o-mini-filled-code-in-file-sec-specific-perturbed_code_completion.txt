if (codeccontext->subtitle_header) {
        av_freep(&codeccontext->subtitle_header);
    }

    if (codeccontext->extradata_size > 0) {
        if (codeccontext->extradata) {
            av_freep(&codeccontext->extradata);
        }
        codeccontext->extradata_size = 0;
    }

    if (codeccontext->codec_id == AV_CODEC_ID_SUBRIP || codeccontext->codec_id == AV_CODEC_ID_SRT) {
        // Handle character encoding for subtitle codecs
        if (codeccontext->subtitle_header) {
            // Set character encoding to UTF-8 for SRT
            codeccontext->subtitle_header = av_strdup("UTF-8");
            if (!codeccontext->subtitle_header) {
                ret = AVERROR(ENOMEM);
                goto free_and_end;
            }
        }
    }
