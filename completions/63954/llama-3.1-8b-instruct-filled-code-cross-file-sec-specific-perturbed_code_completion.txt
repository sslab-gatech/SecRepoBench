static
char* GetData(cmsIT8* it8, int setIndex, int nField)
{
    TABLE* t = GetTable(it8);
    // Retrieves data from the specified row and column in the table.
    // Checks if the specified row and column indices are valid.
    if (setIndex < 0 || setIndex >= t->nSets || nField < 0 || nField >= t->nSamples) {
        cmsSignalError(0, cmsERROR_INVALIDARG, "Invalid set or field index");
        return NULL;
    }
    return t->Data[setIndex * t->nSamples + nField];
}
