static
char* GetData(cmsIT8* it8, int setIndex, int nField)
{
    TABLE* t = GetTable(it8);

    // Validate setIndex and nField to prevent out-of-bounds access
    if (setIndex < 0 || setIndex >= t->nPatches) {
        cmsSignalError(0, cmsERROR_INDEX, "Invalid setIndex");
        return NULL;
    }

    if (nField < 0 || nField >= t->nSamples) {
        cmsSignalError(0, cmsERROR_INDEX, "Invalid nField");
        return NULL;
    }

    // Retrieves data from the specified row and column in the table.
    return t->Data [setIndex * t->nSamples + nField];
}
