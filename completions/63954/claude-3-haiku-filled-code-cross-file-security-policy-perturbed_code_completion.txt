if (setIndex < 0 || setIndex >= t->nPatches) {
    cmsSignalError(0, cmsERROR_RANGE, "Invalid setIndex value in GetData()");
    return NULL;
}

if (nField < 0 || nField >= t->nSamples) {
    cmsSignalError(0, cmsERROR_RANGE, "Invalid nField value in GetData()");
    return NULL;
}

return t->Data[setIndex * t->nSamples + nField];
