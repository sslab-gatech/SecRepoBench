if (!t->Data) return NULL;

if (setIndex > t->nPatches || setIndex < 0) {
    SynError(it8, "Patch %d out of range, there are %d patches", setIndex, t->nPatches);
    return NULL;
}

if (nField > t->nSamples || nField < 0) {
    SynError(it8, "Sample %d out of range, there are %d samples", nField, t->nSamples);
    return NULL;
}
