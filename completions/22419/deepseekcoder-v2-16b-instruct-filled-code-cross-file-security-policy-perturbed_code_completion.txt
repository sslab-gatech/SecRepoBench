// Extract and process the Kerberos realm from the packet payload.
// Locate the starting position of the realm data using ASN.1 encoding markers.
// Determine the length of the realm and ensure it is within the packet bounds.
// Copy the realm string from the packet payload, convert it to lowercase, and store it in the flow's Kerberos domain field.

u_int16_t realm_offset = 0;
u_int16_t realm_len = 0;
const u_int8_t *realm_ptr = NULL;

// Locate the starting position of the realm data using ASN.1 encoding markers.
for (u_int16_t i = 0; i < packet->payload_packet_len; i++) {
    if (packet->payload[i] == 0x30) { // SEQUENCE
        realm_offset = i;
        break;
    }
}

if (realm_offset == 0) {
    return; // No realm found
}

// Determine the length of the realm data.
realm_ptr = &packet->payload[realm_offset];
u_int16_t initial_offset = realm_offset;

while (realm_ptr[realm_len] != 0x00) {
    realm_len++;
}

// Ensure the realm data is within the packet bounds.
if (realm_offset + realm_len >= packet->payload_packet_len) {
    return; // Realm data out of bounds
}

// Copy the realm string from the packet payload, convert it to lowercase, and store it in the flow's Kerberos domain field.
char realm_str[48];
strncpy(realm_str, (char*)&packet->payload[realm_offset + 2], realm_len - 2); // Skip the length and the constructed bit
realm_str[realm_len - 2] = '\0';
for (u_int16_t i = 0; i < realm_len - 2; i++) {
    realm_str[i] = tolower(realm_str[i]);
}

snprintf(kerberos_flow->protos.kerberos.domain, sizeof(kerberos_flow->protos.kerberos.domain), "%s", realm_str);
