{
  if(realm_offset < packet->payload_packet_len) {
    u_int8_t realm_len = packet->payload[realm_offset];

    /* Ensure we don't read past the packet boundary */
    if((realm_offset + 1 + realm_len) <= packet->payload_packet_len) {
      char realm_str[48];

      /* Truncate to avoid buffer overflows */
      if(realm_len > (sizeof(realm_str) - 1))
        realm_len = sizeof(realm_str) - 1;

      /* Copy the realm string */
      memcpy(realm_str, &packet->payload[realm_offset + 1], realm_len);
      realm_str[realm_len] = '\0';

      /* Convert to lowercase */
      for(i = 0; i < realm_len; i++) {
        realm_str[i] = tolower((unsigned char) realm_str[i]);
      }

      /* Store in the Kerberos flow structure */
      snprintf(kerberos_flow->protos.kerberos.domain,
               sizeof(kerberos_flow->protos.kerberos.domain),
               "%s",
               realm_str);
    }
  }
