if (n < 0) {
    mrb_raisef(mrb, E_ARGUMENT_ERROR, "negative array size (or size too big)");
  }

  if (n > len) {
    n = len;
  }

  if (n == 0) {
    return mrb_ary_new(mrb);
  }

  val = ary_subseq(mrb, ary, 0, n);

L_SHIFT:
  if (ARY_SHARED_P(ary)) {
    ary->as.heap.ptr += n;
    ARY_SET_LEN(ary, len - n);
  }
