if (len > ARY_SHIFT_SHARED_MIN) {
    ary_make_shared(mrb, ary);
    goto L_SHIFT;
  }
  else if (len == n) {
    ARY_SET_LEN(ary, 0);
  }
  else {
    mrb_value *ptr = ARY_PTR(ary);
    mrb_int size = len-n;

    while (size--) {
      *ptr = *(ptr+n);
      ++ptr;
    }
    ARY_SET_LEN(ary, len-n);
  }
  return val;
}
