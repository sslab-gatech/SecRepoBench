if (n < 0) {
  mrb_raise(mrb, E_ARGUMENT_ERROR, "negative array size");
}

if (n == 0) {
  return mrb_ary_new(mrb);
}

  val = ary_subseq(mrb, ary, 0, n);
}
else {
  val = mrb_ary_new_from_values(mrb, n, ARY_PTR(ary));
}
if (n == len) {
  ARY_SET_LEN(ary, 0);
  return val;
