if (n < 0) {
    mrb_raise(mrb, E_ARGUMENT_ERROR, "negative number of elements to shift");
  }

  if (len == 0) {
    return mrb_nil_value();
  }

  if (n == 0) {
    n = 1;
  }

  if (len < n) {
    n = len;
  }

  mrb_value *ptr = ARY_PTR(ary);
  mrb_int size = len - n;

  while (size--) {
    *ptr = *(ptr + n);
    ++ptr;
  }

  ARY_SET_LEN(ary, len - n);

  return val;
}
