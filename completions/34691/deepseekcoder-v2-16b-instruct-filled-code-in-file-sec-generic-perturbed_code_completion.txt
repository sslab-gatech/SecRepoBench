if (n > ARY_SHIFT_SHARED_MIN) {
    ary_make_shared(mrb, ary);
    goto L_SHIFT;
  }
  else if (n == 0) {
    // If n is 0, we don't need to shift any elements, so we just return nil.
    return mrb_nil_value();
  }
  else {
    mrb_value *ptr = ARY_PTR(ary);
    mrb_int size = len - n;

    while (size--) {
      *ptr = *(ptr + n);
      ++ptr;
    }
    ARY_SET_LEN(ary, len - n);
  }
  return val;
}
