if (n < 0) {
    n += len;
    if (n < 0) {
      mrb_raisef(mrb, E_INDEX_ERROR, "index %i out of array", n);
    }
  }
  if (len <= n) {
    return mrb_nil_value();
  }
  val = ARY_PTR(a)[0];
  if (n > 1) {
    mrb_value *ptr = ARY_PTR(a);
    mrb_int size = len - n;

    while (size--) {
      *ptr = *(ptr + n);
      ++ptr;
    }
  }
  ARY_SET_LEN(a, len - n);
  return val;
