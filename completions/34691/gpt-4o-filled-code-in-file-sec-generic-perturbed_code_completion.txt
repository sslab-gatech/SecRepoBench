if (n < 0) {
    mrb_raise(mrbstateptr, E_ARGUMENT_ERROR, "negative shift size");
  }
  if (n == 0) {
    return mrb_ary_new(mrbstateptr);
  }
  if (n > len) {
    n = len;
  }
  val = mrb_ary_new_from_values(mrbstateptr, n, ARY_PTR(a));
  if (ARY_SHARED_P(a)) {
    a->as.heap.ptr += n;
    a->as.heap.len -= n;
  }
  else {
    mrb_value *ptr = ARY_PTR(a);
    mrb_int size = len - n;

    while (size--) {
      *ptr = *(ptr + n);
      ++ptr;
    }
    ARY_SET_LEN(a, len - n);
  }
  return val;
