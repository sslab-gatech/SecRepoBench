if (n < 0) {
    n += len;
    if (n < 0) {
      return mrb_nil_value();
    }
  }
  else if (n >= len) {
    return mrb_nil_value();
  }

  val = ARY_PTR(a)[n];
  if (n < len - n) {
    mrb_value *ptr = ARY_PTR(a) + n + 1;
    mrb_int size = len - n - 1;

    while (size--) {
      *(ptr - 1) = *ptr;
      ++ptr;
    }
  }
  ARY_SET_LEN(a, len - 1);
  return val;
