/* Shift multiple elements when an integer argument n is given. */
  if (n <= 0) {
    /* For non-positive n, return an empty array without modifying self. */
    return mrb_ary_new(mrb);
  }
  if (n > len) n = len; /* If n is larger than the array size, shift all elements. */

  /* Collect the first n elements to return. */
  val = mrb_ary_new_from_values(mrb, n, ARY_PTR(ary));

  if (n == 0) {
    /* If n was 0 after adjustments, just return an empty array. */
    return val;
  }

  if (len > ARY_SHIFT_SHARED_MIN) {
    /* For bigger arrays, switch to shared mode and adjust pointer/length. */
    ary_make_shared(mrb, ary);
    ary->as.heap.ptr += n;
    ary->as.heap.len -= n;
  }
  else if (len == n) {
    /* If we're shifting exactly all elements, just set array size to 0. */
    ARY_SET_LEN(ary, 0);
  }
  else {
    /* Otherwise, manually move remaining elements to the front. */
    mrb_value *ptr = ARY_PTR(ary);
    mrb_int size = len - n;
    while (size--) {
      *ptr = *(ptr + n);
      ptr++;
    }
    ARY_SET_LEN(ary, len - n);
  }

  return val;
