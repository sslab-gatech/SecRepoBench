if (n < 0) {
    n += len;
    if (n < 0) {
      mrb_raisef(mrb, E_INDEX_ERROR, "index %i out of array", n);
    }
  }
  if (n > len) {
    n = len;
  }
  val = mrb_nil_value();
  if (n > 0) {
    val = ARY_PTR(a)[0];
    if (n < len) {
      value_move(ARY_PTR(a), ARY_PTR(a) + n, len - n);
    }
  }
  ARY_SET_LEN(a, len - n);
  return val;
