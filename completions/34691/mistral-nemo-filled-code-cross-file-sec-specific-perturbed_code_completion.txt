if (n < 0) {
    n += len;
    if (n < 0) {
      return mrb_nil_value();
    }
  }
  else if (n >= len) {
    return mrb_nil_value();
  }

  val = ARY_PTR(ary)[n];
  if (n < len - n) {
    mrb_value *ptr = ARY_PTR(ary) + n + 1;
    mrb_int size = len - n - 1;

    while (size--) {
      *ptr = *(ptr - 1);
      ++ptr;
    }
  }
  ARY_SET_LEN(ary, len - n - 1);
  return val;
