/*
    Draw stroked polygon.
  */
  status=MagickTrue;
  if (draw_info->stroke.alpha != TransparentAlpha)
    {
      if ((draw_info->stroke_width <= 1.0) &&
          (draw_info->stroke_antialias != MagickFalse) &&
          (miterlimit <= 1.0) &&
          (draw_info->linejoin == MiterJoin))
        {
          status=DrawPolygonPrimitive(image,draw_info,primitive_info,
            exception);
          primitive_info=DestroyPrimitiveInfo(primitive_info,number_vertices);
          return(status);
        }
      stroke_polygon=TraceStrokePolygon(image,draw_info,primitive_info);
      if (stroke_polygon == (PrimitiveInfo *) NULL)
        return(MagickFalse);
      status=DrawPolygonPrimitive(image,draw_info,stroke_polygon,exception);
      stroke_polygon=DestroyPrimitiveInfo(stroke_polygon,0);
    }
  if (status != MagickFalse)
    status=DrawPolygonPrimitive(image,draw_info,primitive_info,exception);
  primitive_info=DestroyPrimitiveInfo(primitive_info,number_vertices);
  return(status);
