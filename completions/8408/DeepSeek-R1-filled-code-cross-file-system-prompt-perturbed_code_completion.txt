DrawInfo
    **graphic_context;

  GradientInfo
    *gradient;

  MagickBooleanType
    status;

  PointInfo
    point;

  PolygonInfo
    *polygon;

  PrimitiveInfo
    *primitive_info;

  register ssize_t
    i;

  ssize_t
    n;

  StopInfo
    *stops;

  status=MagickTrue;
  n=0;
  stops=(StopInfo *) NULL;
  primitive_info=(PrimitiveInfo *) NULL;
  graphic_context=(DrawInfo **) AcquireQuantumMemory(MagickPathExtent,
    sizeof(*graphic_context));
  if (graphic_context == (DrawInfo **) NULL)
    ThrowBinaryException(ResourceLimitError,"MemoryAllocationFailed",
      image->filename);
  graphic_context[n]=CloneDrawInfo((ImageInfo *) NULL,draw_info);
  GetAffineMatrix(&affine);
  GetAffineMatrix(&current);
  primitive=AcquireString(draw_info->primitive);
  token=primitive;
  for (q=(const char *) primitive; *q != '\0'; )
  {
    /*
      Interpret graphic primitive.
    */
    GetNextToken(q,&q,MagickPathExtent,primitive_name);
    if (*primitive_name == '\0')
      break;
    if (LocaleCompare(primitive_name,"push") == 0)
      {
        n++;
        graphic_context[n]=CloneDrawInfo((ImageInfo *) NULL,graphic_context[n-1]);
        continue;
      }
    if (LocaleCompare(primitive_name,"pop") == 0)
      {
        graphic_context[n]=DestroyDrawInfo(graphic_context[n]);
        n--;
        if (n < 0)
          ThrowBinaryException(DrawError,"UnbalancedGraphicContextPushPop",
            primitive_name);
        continue;
      }
    if (LocaleCompare(primitive_name,"affine") == 0)
      {
        GetNextToken(q,&q,MagickPathExtent,token);
        affine.sx=StringToDouble(token,&next_token);
        GetNextToken(q,&q,MagickPathExtent,token);
        affine.rx=StringToDouble(token,&next_token);
        GetNextToken(q,&q,MagickPathExtent,token);
        affine.ry=StringToDouble(token,&next_token);
        GetNextToken(q,&q,MagickPathExtent,token);
        affine.sy=StringToDouble(token,&next_token);
        GetNextToken(q,&q,MagickPathExtent,token);
        affine.tx=StringToDouble(token,&next_token);
        GetNextToken(q,&q,MagickPathExtent,token);
        affine.ty=StringToDouble(token,&next_token);
        continue;
      }
    if (LocaleCompare(primitive_name,"translate") == 0)
      {
        GetNextToken(q,&q,MagickPathExtent,token);
        affine.tx=StringToDouble(token,&next_token);
        GetNextToken(q,&q,MagickPathExtent,token);
        affine.ty=StringToDouble(token,&next_token);
        continue;
      }
    if (LocaleCompare(primitive_name,"scale") == 0)
      {
        GetNextToken(q,&q,MagickPathExtent,token);
        affine.sx=StringToDouble(token,&next_token);
        GetNextToken(q,&q,MagickPathExtent,token);
        affine.sy=StringToDouble(token,&next_token);
        continue;
      }
    if (LocaleCompare(primitive_name,"rotate") == 0)
      {
        GetNextToken(q,&q,MagickPathExtent,token);
        current.rx=StringToDouble(token,&next_token);
        current.ry=current.rx;
        break;
      }
    /*
      Parse and render drawing primitive.
    */
    primitive_info=ParsePrimitive(image,graphic_context[n],primitive_name,
      &q,exception);
    if (primitive_info == (PrimitiveInfo *) NULL)
      {
        status=MagickFalse;
        break;
      }
    status=RenderPrimitive(image,graphic_context[n],primitive_info,exception);
    primitive_info=(PrimitiveInfo *) RelinquishMagickMemory(primitive_info);
    if (status == MagickFalse)
      break;
  }
