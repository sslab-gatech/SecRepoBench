  DrawInfo **graphic_context = (DrawInfo **) AcquireQuantumMemory(10, sizeof(*graphic_context));

    return MagickFalse;

  for (size_t i = 0; i < 10; i++)
  {
    graphic_context[i] = CloneDrawInfo((ImageInfo *) NULL, draw_info);
    if (graphic_context[i] == (DrawInfo *) NULL)
    {
      for (size_t j = 0; j < i; j++)
        graphic_context[j] = DestroyDrawInfo(graphic_context[j]);
      graphic_context = (DrawInfo **) RelinquishMagickMemory(graphic_context);
      return MagickFalse;
    }
  }

  for (size_t i = 0; i < 10; i++)
  {
    // Execute the drawing command based on the current graphic context
    if (ExecuteDrawingCommand(image, graphic_context[i], exception) == MagickFalse)
    {
      for (size_t j = 0; j < 10; j++)
        graphic_context[j] = DestroyDrawInfo(graphic_context[j]);
      graphic_context = (DrawInfo **) RelinquishMagickMemory(graphic_context);
      return MagickFalse;
    }
  }

  for (size_t i = 0; i < 10; i++)
