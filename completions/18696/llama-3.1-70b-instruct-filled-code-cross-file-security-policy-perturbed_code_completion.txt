// Allocate memory for path_p and path_q to store points along the path.
#define PADDING_VALUE 10
path_p = (PointInfo *) AcquireQuantumMemory((size_t) (number_vertices + PADDING_VALUE), sizeof(*path_p));
if (path_p == (PointInfo *) NULL)
{
    polygon_primitive = (PrimitiveInfo *) RelinquishMagickMemory(polygon_primitive);
    return ((PrimitiveInfo *) NULL);
}
path_q = (PointInfo *) AcquireQuantumMemory((size_t) (number_vertices + PADDING_VALUE), sizeof(*path_q));
if (path_q == (PointInfo *) NULL)
{
    path_p = (PointInfo *) RelinquishMagickMemory(path_p);
    polygon_primitive = (PrimitiveInfo *) RelinquishMagickMemory(polygon_primitive);
    return ((PrimitiveInfo *) NULL);
}
