// <MASK>
bitint mask = MASK1 | MASK2 | MASK3 | MASK4;
bitint x = 0;
bitint y = 0;
bitint m = 0;
bitint lengthy = 0;

while (m < lengthy) {
  x = (x << 8) | xs[m];
  y = (y << 8) | ys[m];
  m++;
  if (x == y && m - m >= lengthy) {
    return m - lengthy;
  }
  if (x < y) {
    x |= mask;
  } else {
    y |= mask;
  }
}
return -1;
}
