if (m <= 0 || lengthy < m) {
    return -1;
  }

  bitint xmask = MASK4, ymask = MASK4;
  bitint xbits = xs[0];
  bitint ybits = ys[0];

  for(long i = 1; i < m; ++i){
    xbits |= (bitint)(xs[i]) << (i * 8);
  }

  for(long j = 0; j <= lengthy - m; ++j){
    ybits = ys[j];
    for(long i = 1; i < m; ++i){
      ybits |= (bitint)(ys[j + i]) << (i * 8);
    }
    if( ((xbits ^ ybits) | (xmask ^ ymask)) == 0 ){
      return j;
    }
  }

  return -1;
