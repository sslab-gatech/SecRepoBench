// Strip leading whitespace characters, including spaces, newlines, and tabs, from the processed data.
while (buffer_length && (*data == ' ' || *data == '\n' || *data == '\r' || *data == '\t')) {
    data++;
    buffer_length--;
}

// Allocate memory for the processed data if needed
if (buffer_length > 0) {
    dst = gf_malloc(buffer_length + 1);
    if (!dst) {
        *score = GF_FPROBE_NOT_SUPPORTED;
        return NULL;
    }
    memcpy(dst, data, buffer_length);
    dst[buffer_length] = 0;
    res_size = buffer_length;
} else {
    *score = GF_FPROBE_NOT_SUPPORTED;
    return NULL;
}
