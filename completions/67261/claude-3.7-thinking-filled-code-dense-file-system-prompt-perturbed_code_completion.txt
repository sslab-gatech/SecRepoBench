u32 res_size = buffer_length;
char *dst = NULL;
const char *res = NULL;
GF_Err e;

e = gf_utf_get_string_from_bom(data, buffer_length, &dst, (const char **)&res, &res_size);
if (e) return NULL;
data = (const u8 *)res;

// Strip leading whitespace
while (res_size && data[0] && strchr("\n\r\t ", (char)data[0])) {
    data++;
    res_size--;
}
