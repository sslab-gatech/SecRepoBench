encoding_error:
    /*
     * We encountered invalid UTF-8 data. Raise a fatal error with details
     * about the problematic byte and terminate further parsing.
     */
    xmlFatalErrMsgInt(ctxt, XML_ERR_INVALID_CHAR,
                      "Invalid UTF-8 character 0x%X\n",
                      (unsigned int) ctxt->input->cur[0]);
    xmlHaltParser(ctxt);
    goto done;
