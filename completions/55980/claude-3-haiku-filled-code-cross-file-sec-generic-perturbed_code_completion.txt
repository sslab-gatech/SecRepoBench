encoding_error:
    // Report the encoding error
    int len = parserContext->input->end - parserContext->input->cur;
    xmlChar *bytes = xmlStrndup(parserContext->input->cur, len);
    xmlFatalErrMsgStr(parserContext, XML_ERR_INVALID_ENCODING,
                     "Encoding error while parsing CDATA section: invalid bytes %s\n",
                     bytes);
    xmlFree(bytes);

    // Terminate the parsing process
    xmlHaltParser(parserContext);
    return;
