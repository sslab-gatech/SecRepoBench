encoding_error:
    if (xmlCheckUTF8(parserContext->input->cur) != 0) {
        xmlFatalErrMsgStr(parserContext, XML_ERR_INVALID_CHAR,
                          "Invalid UTF-8 sequence starting with '0x%02X'\n",
                          *parserContext->input->cur);
    } else {
        xmlFatalErrMsgStr(parserContext, XML_ERR_INVALID_ENCODING,
                          "Input is not proper UTF-8, indicate encoding !\n"
                          "Bytes: 0x%02X 0x%02X 0x%02X 0x%02X\n",
                          parserContext->input->cur[0],
                          parserContext->input->cur[1],
                          parserContext->input->cur[2],
                          parserContext->input->cur[3]);
    }
    xmlHaltParser(parserContext);
    return(0);
