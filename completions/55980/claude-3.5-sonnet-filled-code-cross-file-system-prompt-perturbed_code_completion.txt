encoding_error:
    if ((ctxt->input != NULL) && (ctxt->input->end - ctxt->input->cur >= 4)) {
        xmlChar *cur = ctxt->input->cur;
        xmlGenericErrorFunc channel = ctxt->genErr;
        void *data = ctxt->userData;
        
        if (channel != NULL) {
            channel(data, "Bytes: 0x%02X 0x%02X 0x%02X 0x%02X\n",
                    cur[0], cur[1], cur[2], cur[3]);
        }
    }
    xmlHaltParser(ctxt);
    if ((ctxt->sax) && (ctxt->sax->error != NULL))
        ctxt->sax->error(ctxt->userData, "Input is not proper UTF-8, indicate encoding !\n");
    ctxt->errNo = XML_ERR_INVALID_ENCODING;
    return(0);
