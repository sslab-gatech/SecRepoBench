encoding_error:
    if ((parserContext->sax != NULL) && (parserContext->sax->error != NULL)) {
        xmlChar buffer[150];

        snprintf((char *) buffer, 149, "Bytes: 0x%02X 0x%02X 0x%02X 0x%02X\n",
                 parserContext->input->cur[0], parserContext->input->cur[1],
                 parserContext->input->cur[2], parserContext->input->cur[3]);
        parserContext->sax->error(parserContext->userData, (char *) buffer);
    }
    xmlHaltParser(parserContext);
    return(0);
