encoding_error:
    xmlFatalErr(parserContext, XML_ERR_INVALID_ENCODING,
                "Input is not proper UTF-8, indicate encoding "
                "Bytes: 0x%02X 0x%02X",
                (unsigned char)parserContext->input->cur[0],
                (unsigned char)parserContext->input->cur[1]);
    xmlHaltParser(parserContext);
    return ret;
