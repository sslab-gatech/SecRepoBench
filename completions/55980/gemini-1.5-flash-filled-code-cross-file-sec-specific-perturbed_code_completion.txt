encoding_error:
    if (ctxt->errNo == XML_ERR_INVALID_ENCODING) {
        xmlChar *enc = xmlGetCharEncodingName(ctxt->input->encoding);
        xmlFatalErr(ctxt, XML_ERR_INVALID_ENCODING,
                    "Input is not proper UTF-8, indicate %s\n", enc);
        xmlFree(enc);
    }
    xmlHaltParser(ctxt);
    return(0);
