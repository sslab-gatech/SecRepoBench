encoding_error:
    // Handle encoding errors encountered during XML parsing
    if (ctxt->input != NULL && ctxt->input->cur != NULL) {
        // Report an encoding error with details about the problematic bytes
        xmlFatalErrMsgStr(ctxt, XML_ERR_INVALID_ENCODING,
                          "Encoding error detected at byte: %.20s\n",
                          ctxt->input->cur);
    } else {
        // Generic encoding error if no specific details are available
        xmlFatalErr(ctxt, XML_ERR_INVALID_ENCODING, "Encoding error detected\n");
    }

    // Terminate the parsing process upon encountering this error
    xmlHaltParser(ctxt);
    return 0;
