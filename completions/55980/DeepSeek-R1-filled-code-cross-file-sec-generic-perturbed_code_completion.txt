encoding_error:
    xmlFatalErr(parserContext, XML_ERR_INVALID_ENCODING, "Invalid UTF-8 bytes detected");
    xmlHaltParser(parserContext);
#ifdef DEBUG_PUSH
    xmlGenericError(xmlGenericErrorContext, "PP: entering EOF due to encoding error\n");
#endif
    if ((parserContext->sax) && (parserContext->sax->endDocument != NULL))
        parserContext->sax->endDocument(parserContext->userData);
    goto done;
