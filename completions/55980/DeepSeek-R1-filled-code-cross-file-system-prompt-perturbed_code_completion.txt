encoding_error:
    xmlFatalErrMsgStr(parserContext, XML_ERR_INVALID_ENCODING,
        "Bytes: 0x%02X 0x%02X 0x%02X 0x%02X do not conform to UTF-8 encoding\n",
        (unsigned char)parserContext->input->cur[0],
        (unsigned char)parserContext->input->cur[1],
        (unsigned char)parserContext->input->cur[2],
        (unsigned char)parserContext->input->cur[3]);
    xmlHaltParser(parserContext);
    return 0;
