encoding_error: {
        xmlChar buf[150];
        int i;
        char *cur = (char *)parserContext->input->cur;
        for (i=0;i<80;i++) {
            if (cur[i] == 0) break;
            if (cur[i] == '\n') {
                buf[i] = '\n';
                break;
            }
            buf[i] = cur[i];
        }
        buf[i] = 0;
        xmlFatalErrMsg(parserContext, XML_ERR_INVALID_ENCODING,
                       "Input is not proper UTF-8, indicate encoding !\n"
                       "Bytes: 0x%02X 0x%02X 0x%02X 0x%02X\n",
                       parserContext->input->cur[0],
                       parserContext->input->cur[1],
                       parserContext->input->cur[2],
                       parserContext->input->cur[3]);
        xmlHaltParser(parserContext);
        return(0);
    }
