encoding_error:
    if ((parserContext->sax != NULL) && (parserContext->sax->error != NULL)) {
        xmlErrorPtr err = xmlGetLastError();

        if (err == NULL) {
            xmlGenericError(xmlGenericErrorContext, "Encoding error\n");
            parserContext->sax->error(parserContext->userData,
                                      "Encoding error\n");
        } else {
            parserContext->sax->error(parserContext->userData,
                                      "%s\n", err->message);
        }
    }
    xmlHaltParser(parserContext);
    return(0);
