if ( nfields < 1 ) {
    return;
}
nbytes = nfields * sizeof(*data_variable->internal->fields);
data_variable->internal->fields = (matvar_t **)calloc(nfields, sizeof(*data_variable->internal->fields));
if ( NULL != data_variable->internal->fields ) {
    data_variable->data = data_variable->internal->fields;
} else {
    Mat_Critical("Error allocating memory for matvar->data");
    return;
}
