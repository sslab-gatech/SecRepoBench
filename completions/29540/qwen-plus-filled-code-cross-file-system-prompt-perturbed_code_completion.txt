if ( nfields < 1 || nelems < 1 ) {
        return;
    }

    size_t nbytes = nfields * sizeof(matvar_t *) * nelems;
    if ( fields_are_variables ) {
        nbytes = nfields * sizeof(matvar_t *);
    }

    data_variable->nbytes += nbytes;

    fields = (matvar_t**)malloc(nbytes);
    if ( NULL == fields ) {
        Mat_Critical("Error allocating memory for matvar->data");
        return;
    }

    memset(fields, 0, nbytes);
    data_variable->data = fields;
