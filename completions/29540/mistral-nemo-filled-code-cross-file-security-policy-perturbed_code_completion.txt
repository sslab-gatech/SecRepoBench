if ( NULL != data_variable->dims ) {
        nbytes = sizeof(matvar_t*) * nfields;
        fields = (matvar_t**)calloc(nbytes, sizeof(*fields));
        if ( NULL != fields ) {
            data_variable->data = fields;
            data_variable->internal->num_fields = nfields;
        } else {
            Mat_Critical("Error allocating memory for fields");
            return;
        }
    } else {
        return;
