    return;

size_t nbytes = 0;
err = Mat_MulDims(matvar, &nbytes);
if (err) {
    Mat_Critical("Integer multiplication overflow");
    return;
}
matvar->nbytes = (hsize_t)nbytes;
matvar->data_size = sizeof(matvar_t**);
err = Mul(&matvar->nbytes, matvar->nbytes, matvar->data_size);
if (err) {
    Mat_Critical("Integer multiplication overflow");
    return;
}
matvar->data = malloc(matvar->nbytes);
if (NULL == matvar->data) {
    Mat_Critical("Error allocating memory for matvar->data");
    return;
