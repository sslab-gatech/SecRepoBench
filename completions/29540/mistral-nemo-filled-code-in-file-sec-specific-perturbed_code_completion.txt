import matio

# Create a new MAT file with the given name and optional header string
mat = matio.Mat_Create73('example.mat', 'This is an example MAT file')

# Create a new MAT variable with the given name, class type, and data type
var = matio.Mat_VarCreate('example', matio.MAT_C_DOUBLE, matio.MAT_T_DOUBLE)

# Write data to the variable
data = [1, 2, 3, 4, 5]
matio.Mat_VarWriteData(var, data)

# Close the MAT file
matio.Mat_Close(mat)
