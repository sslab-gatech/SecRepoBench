if ( nelems < 1 || nfields < 1 ) {
        return;
    }

    data_variable->data_size = sizeof(matvar_t**);
    {
        int err = 0;
        size_t tmp;
        err |= Mul(&tmp, nelems, nfields);
        err |= Mul(&data_variable->nbytes, tmp, data_variable->data_size);
        if ( err ) {
            Mat_Critical("Integer multiplication overflow");
            return;
        }
    }

    data_variable->data = malloc(data_variable->nbytes);
    fields = (matvar_t**)data_variable->data;
    if ( NULL == fields ) {
        Mat_Critical("Couldn't allocate memory for the fields");
        return;
    }
