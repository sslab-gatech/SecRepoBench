if ( nelems < 1 || nfields < 1 ) {
    return;
}

data_variable->nbytes = nelems * nfields * sizeof(matvar_t*);
fields = (matvar_t**)malloc(data_variable->nbytes);
if ( NULL == fields ) {
    Mat_Critical("Error allocating memory for fields array");
    return;
}
memset(fields, 0, data_variable->nbytes);
data_variable->data = fields;
