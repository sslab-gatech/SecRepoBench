if ( nelems < 1 || nfields < 1 ) {
    return;
}
size_t total_fields = ( fields_are_variables ) ? nfields : nelems * nfields;
data_variable->nbytes = total_fields * sizeof(matvar_t*);
fields = (matvar_t**)malloc(data_variable->nbytes);
if ( NULL == fields ) {
    Mat_Critical("Error allocating memory for matvar->data fields");
    return;
}
data_variable->data = fields;
