if ( nelems < 1 || nfields < 1 ) {
        return;
    }

    data_variable->nbytes = nelems*nfields*sizeof(matvar_t *);
    fields = (matvar_t **)calloc((size_t)nelems*nfields,sizeof(matvar_t*));
    if ( NULL == fields ) {
        Mat_Critical("Error allocating memory for fields");
        return;
    }

    data_variable->data = fields;
    data_variable->class_type = MAT_C_STRUCT;
    data_variable->isComplex  = MAT_F_DONT_KNOW;
