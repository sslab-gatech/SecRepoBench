if (nfields < 1 || nelems < 1) {
    return;
}

data_variable->internal->nbytes = nfields * nelems * sizeof(matvar_t *);
fields = (matvar_t **)calloc((size_t)(nfields * nelems), sizeof(*fields));
if (fields == NULL) {
    Mat_Critical("Error allocating memory for matvar->data");
    return;
}
data_variable->data = fields;
