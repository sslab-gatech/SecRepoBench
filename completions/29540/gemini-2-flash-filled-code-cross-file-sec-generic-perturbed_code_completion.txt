if ( nelems < 1 || nfields < 1 )
        return;

    data_variable->nbytes = (size_t)(nelems*nfields*sizeof(matvar_t*));
    fields = (matvar_t**)malloc(data_variable->nbytes);
    if ( NULL == fields ) {
        Mat_Critical("Memory allocation failure");
        return;
    }
    memset(fields,0,(size_t)data_variable->nbytes);
    data_variable->data = fields;
