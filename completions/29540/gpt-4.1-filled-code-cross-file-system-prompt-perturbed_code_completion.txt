if ( nelems < 1 || nfields < 1 ) {
    return;
}
data_variable->nbytes = (size_t)(nelems * nfields * sizeof(matvar_t*));
fields = (matvar_t**)calloc((size_t)(nelems * nfields), sizeof(matvar_t*));
if ( NULL == fields ) {
    Mat_Critical("Error allocating memory for fields array");
    return;
}
data_variable->data = fields;
