if ( 0 == nelems || 0 == nfields ) {
        return;
    }

    data_variable->nbytes = nelems * nfields * sizeof(*fields);

    fields = (matvar_t**)calloc((size_t)(nelems * nfields), sizeof(*fields));
    if ( NULL == fields ) {
        Mat_Critical("Error allocating memory for fields array");
        return;
    }

    data_variable->data = fields;
