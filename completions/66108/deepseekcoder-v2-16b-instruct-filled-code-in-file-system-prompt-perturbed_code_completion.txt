if (sync[size] != 0xFF) {

    u32 frame_size = gf_mp3_frame_size(ctx->hdr);
    u32 bytes_to_skip = frame_size - offsetskipped;
    if (bytes_to_skip > remain) {
        bytes_to_skip = remain;
    }
    start += bytes_to_skip;
    remain -= bytes_to_skip;
    goto restart;
}
