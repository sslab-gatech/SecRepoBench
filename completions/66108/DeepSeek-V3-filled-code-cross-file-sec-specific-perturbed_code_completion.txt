if (sync + size < start + remain) {
    if (sync[size] == 0xFF) {
        // Valid frame, proceed
    } else if (remain - (sync - start) >= 3 && !memcmp(sync + size, "TAG", 3)) {
        skip_id3v1 = GF_TRUE;
    } else {
        GF_LOG(GF_LOG_WARNING, GF_LOG_MEDIA, ("[MP3Dmx] Invalid frame at offset %u, attempting resync\n", (u32)(sync - ctx->mp3_buffer)));
        bytes_to_drop = offsetskipped + 1;
