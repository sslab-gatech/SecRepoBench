size = gf_mp3_frame_size(ctx->hdr);

/* Confirm that the next byte is sync or check for ID3v1 tag. */
if ((remain >= offsetskipped + size + 3) && !memcmp(sync + size, "TAG", 3)) {
	skip_id3v1 = GF_TRUE;
} else if ((remain >= offsetskipped + size + 1) && (sync[size] == 0xFF)) {
	/* Next frame likely in sync. */
} else {
	GF_LOG(GF_LOG_WARNING, GF_LOG_MEDIA, ("[MP3Dmx] Next frame is not sync, possible misalignment at offset %u\n",
		(unsigned int)(sync - ctx->mp3_buffer)));
}
