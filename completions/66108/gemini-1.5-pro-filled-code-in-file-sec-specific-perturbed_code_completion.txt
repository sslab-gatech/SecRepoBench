size = gf_mp3_frame_size(ctx->hdr);
		if (offsetskipped + size > remain) {
			if (!is_eos) {
				GF_LOG(GF_LOG_WARNING, GF_LOG_MEDIA, ("[MP3Dmx] incomplete frame, waiting for more data\n"));
			}
			break;
		}

		if (sync[size]==0xFF) {
			if ((remain > size+3) && (sync[size+1]=='T') && (sync[size+2]=='A') && (sync[size+3]=='G')) {
				skip_id3v1 = GF_TRUE;
			}
		} else if (!is_eos) {
			GF_LOG(GF_LOG_DEBUG, GF_LOG_MEDIA, ("[MP3Dmx] invalid frame detected - trying to resync\n"));
			ctx->is_sync = GF_FALSE;
			bytes_to_drop = 1;
			goto drop_byte;
		}
