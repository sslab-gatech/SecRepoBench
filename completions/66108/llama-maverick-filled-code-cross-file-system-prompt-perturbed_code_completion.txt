if (remain > size + offsetskipped) {
	if (sync[size + offsetskipped] != 0xFF) {
		if ((sync[size + offsetskipped] == 'T') && (sync[size + offsetskipped + 1] == 'A') && (sync[size + offsetskipped + 2] == 'G')) {
			skip_id3v1 = GF_TRUE;
		} else {
			GF_LOG(GF_LOG_DEBUG, GF_LOG_MEDIA, ("[MP3Dmx] Invalid frame or garbage after frame, resyncing\n"));
			ctx->is_sync = GF_FALSE;
		}
	}
