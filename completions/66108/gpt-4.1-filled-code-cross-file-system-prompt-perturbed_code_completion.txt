size = gf_mp3_frame_size(ctx->hdr);
if (!size) {
	GF_LOG(GF_LOG_WARNING, GF_LOG_MEDIA, ("[MP3Dmx] Invalid MP3 frame header, skipping byte\n"));
	start = sync + 1;
	remain -= (u32)(sync - start + 1);
	continue;
}

// Check for ID3v1 tag at the end of the frame
if ((remain >= offsetskipped + size + 128) &&
    (sync[offsetskipped + size] == 'T') &&
    (sync[offsetskipped + size + 1] == 'A') &&
    (sync[offsetskipped + size + 2] == 'G')) {
	skip_id3v1 = GF_TRUE;
}
