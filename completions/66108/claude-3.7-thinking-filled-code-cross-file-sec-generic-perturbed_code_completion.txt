size = gf_mp3_frame_size(ctx->hdr);
if (!size || (size > remain - offsetskipped)) {
    // Invalid size or not enough data
    if (!size) {
        GF_LOG(GF_LOG_WARNING, GF_LOG_MEDIA, ("[MP3Dmx] Invalid frame size\n"));
        bytes_to_drop = 1;
        goto drop_byte;
    }
    break;
}

// Check for ID3v1 tag after the frame
if (remain >= offsetskipped + size + 3) {
    if ((sync[size] == 'T') && (sync[size+1] == 'A') && (sync[size+2] == 'G')) {
        skip_id3v1 = GF_TRUE;
        GF_LOG(GF_LOG_DEBUG, GF_LOG_MEDIA, ("[MP3Dmx] Found ID3v1 tag\n"));
    }
}
