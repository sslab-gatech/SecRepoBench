			break;
		}
		if (sync[size] != 0xFF) {
			GF_LOG(GF_LOG_WARNING, GF_LOG_MEDIA, ("[MP3Dmx] invalid frame sync byte 0x%X\n", sync[size]));

			if (offset == 0xFFFFFFFF) {
				break;
			}
			start += offset + 1;
			remain -= offset + 1;
			continue;
		}
