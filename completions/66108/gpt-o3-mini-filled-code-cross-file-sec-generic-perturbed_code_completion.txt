size = gf_mp3_frame_size(ctx->hdr);
    if (!size) {
        GF_LOG(GF_LOG_WARNING, GF_LOG_MEDIA, ("[MP3Dmx] zero frame size, dropping one byte for resync\n"));
        bytes_to_drop = 1;
        goto drop_byte;
    }
    if (remain > offsetskipped + size) {
        if (start[offsetskipped + size] == 0xFF) {
            /* valid frame, no action needed */
        } else if (remain >= offsetskipped + size + 3 &&
                   start[offsetskipped + size] == 'T' &&
                   start[offsetskipped + size + 1] == 'A' &&
                   start[offsetskipped + size + 2] == 'G') {
            skip_id3v1 = GF_TRUE;
        } else {
            GF_LOG(GF_LOG_WARNING, GF_LOG_MEDIA, ("[MP3Dmx] invalid sync byte 0x%02X after frame, dropping one byte for resync\n", start[offsetskipped + size]));
            bytes_to_drop = 1;
            goto drop_byte;
        }
    }
