size = gf_mp3_frame_size(ctx->hdr);
u8 *frame_end = sync + size;
if (frame_end < start + remain) {
    if (frame_end[0] != 0xFF) {
        if ((frame_end + 3 <= start + remain) && !memcmp(frame_end, "TAG", 3)) {
            skip_id3v1 = GF_TRUE;
        } else {
            GF_LOG(GF_LOG_WARNING, GF_LOG_MEDIA, ("[MP3Dmx] Invalid frame sync after frame at offset %u\n", (u32)(frame_end - ctx->mp3_buffer)));
            break;
        }
    }
