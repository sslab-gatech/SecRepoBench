//ready to send packet
size = gf_mp3_frame_size(ctx->hdr);
if (sync[size] == 0xFF) {
    //check for ID3v1 tag
    if (sync[size+1] == 'T' && sync[size+2] == 'A' && sync[size+3] == 'G') {
        skip_id3v1 = GF_TRUE;
    } else {
        GF_LOG(GF_LOG_WARNING, GF_LOG_MEDIA, ("[MP3Dmx] Invalid frame at offset %d, trying to resync\n", offsetskipped));
        ctx->is_sync = GF_FALSE;
    }
}
