// Check if the byte after the frame is 0xFF for synchronization
		if (offsetskipped + 1 < remain && sync[1] == 0xFF) {
			// Check for ID3v1 tag ("TAG") at the end of the file
			if (remain >= 128 && !memcmp(sync + remain - 128, "TAG", 3)) {
				skip_id3v1 = GF_TRUE;
			}
		} else {
			// Log a warning about the invalid frame and attempt to resync
			GF_LOG(GF_LOG_WARNING, GF_LOG_MEDIA, ("[MP3Dmx] Invalid frame detected, attempting to resync\n"));
			start += offsetskipped + 1;
			remain -= offsetskipped + 1;
			continue;
		}
