		if (!size) {
			GF_LOG(GF_LOG_WARNING, GF_LOG_MEDIA, ("[MP3Dmx] invalid frame, resyncing\n"));
			bytes_to_drop = 1;
			goto drop_byte;
		}

		if (sync + size < start + remain) {
			if (sync[size] == 0xFF) {
				// Good, we found the next sync word
			} else if (sync[size] == 'T' && sync[size + 1] == 'A' && sync[size + 2] == 'G') {
				skip_id3v1 = GF_TRUE;
			} else {
				GF_LOG(GF_LOG_WARNING, GF_LOG_MEDIA, ("[MP3Dmx] invalid frame, resyncing\n"));
				bytes_to_drop = 1;
				goto drop_byte;
			}
