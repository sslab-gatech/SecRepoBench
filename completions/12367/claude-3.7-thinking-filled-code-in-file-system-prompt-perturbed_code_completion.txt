if (avctx->codec_id == AV_CODEC_ID_HYMT) {
    // Check if there's enough data for slice info at the end
    if (buf_size < 8)
        return AVERROR_INVALIDDATA;
    
    // Last 4 bytes contain the offset where slice info starts
    slices_info_offset = AV_RL32(packet->data + buf_size - 4);
    
    // Validate the slices_info_offset
    if (slices_info_offset >= buf_size - 4)
        return AVERROR_INVALIDDATA;
    
    // Get the number of slices from the packet data
    nb_slices = AV_RL32(packet->data + slices_info_offset);
    
    // Validate number of slices
    if (nb_slices <= 0 || nb_slices > height)
        return AVERROR_INVALIDDATA;
    
    // Validate that we have enough data for all slice information
    if (slices_info_offset + 4 + nb_slices * 8 > buf_size - 4)
        return AVERROR_INVALIDDATA;
    
    // Calculate slice height
    slice_height = (height + nb_slices - 1) / nb_slices;
} else {
    // Not HYMT codec, default to one slice
    nb_slices = 1;
    slice_height = height;
    slices_info_offset = 0;
}
