if (avctx->codec_id == AV_CODEC_ID_HYMT) {
        if ((unsigned)buf_size < 16 || buf_size - AV_RL32(buf + buf_size - 12) < 0)
            return AVERROR_INVALIDDATA;

        nb_slices = AV_RL32(buf + buf_size - 8);
        if (nb_slices > (unsigned)buf_size / 8)
            return AVERROR_INVALIDDATA;

        slice_height = AV_RL32(buf + buf_size - 4);
        if (!slice_height || slice_height > height || height % slice_height)
            return AVERROR_INVALIDDATA;

        slices_info_offset = buf_size - 16 - nb_slices * 8;
        if (slices_info_offset < table_size)
            return AVERROR_INVALIDDATA;
    } else {
        nb_slices = 1;
        slice_height = height;
