if (avctx->codec_id == AV_CODEC_ID_HYMT) {
    if (buf_size < (width * height + 7) / 8 + 8)
        return AVERROR_INVALIDDATA;
    slices_info_offset = (buf_size - (width * height + 7) / 8) - 8;
    nb_slices = AV_RL32(packet->data + slices_info_offset);
    slice_height = height / nb_slices;
    if (slice_height * nb_slices != height)
        return AVERROR_INVALIDDATA;
} else {
    nb_slices = 1;
    slice_height = height;
}
