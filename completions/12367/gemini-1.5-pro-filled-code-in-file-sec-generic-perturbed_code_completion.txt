if (avctx->codec_id == AV_CODEC_ID_HYMT) {
        if (buf_size < 16 || buf_size < AV_RL32(buf + buf_size - 16) + 4 || AV_RL32(buf + buf_size - 12) > INT_MAX - 16 || AV_RL32(buf + buf_size - 8) > INT_MAX - 16 || AV_RL32(buf + buf_size - 4) > INT_MAX - 16) {
            return AVERROR_INVALIDDATA;
        }

        nb_slices = AV_RL32(buf + buf_size - 16);
        slice_height = AV_RL32(buf + buf_size - 12);
        slices_info_offset = buf_size - 16 - AV_RL32(buf + buf_size - 8) - AV_RL32(buf + buf_size - 4);

        if (nb_slices < 1 || slice_height < 1 || slices_info_offset < table_size || (unsigned)slices_info_offset > INT_MAX - table_size || nb_slices > (buf_size - slices_info_offset) / 8) {
            return AVERROR_INVALIDDATA;
        }

        if (height % slice_height || nb_slices * slice_height != height) {
            return AVERROR_INVALIDDATA;
        }
    } else {
        nb_slices = 1;
        slice_height = height;
    }
