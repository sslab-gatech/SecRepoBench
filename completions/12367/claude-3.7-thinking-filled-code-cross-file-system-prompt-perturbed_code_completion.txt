if (avctx->codec_id == AV_CODEC_ID_HYMT) {
    // Check if there's enough data for slice info
    if (packet->size <= 4)
        return AVERROR_INVALIDDATA;
    
    // Read number of slices from the end of the packet
    nb_slices = AV_RL32(packet->data + packet->size - 4);
    
    // Validate number of slices
    if (nb_slices <= 0 || nb_slices > height)
        return AVERROR_INVALIDDATA;
    
    // Each slice info takes 8 bytes (offset + size)
    // Slice info is stored before the nb_slices value
    slices_info_offset = packet->size - 4 - nb_slices * 8;
    
    // Make sure slices_info_offset is valid
    if (slices_info_offset < 0 || slices_info_offset >= packet->size - 4)
        return AVERROR_INVALIDDATA;
    
    // Calculate slice height (assuming equal-sized slices)
    slice_height = height / nb_slices;
} else {
    // Default to one slice for the entire frame
    nb_slices = 1;
    slice_height = height;
    slices_info_offset = 0;
}
