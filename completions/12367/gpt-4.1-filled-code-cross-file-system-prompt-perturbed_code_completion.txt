if (avctx->codec_id == AV_CODEC_ID_HYMT) {
    // The last 12 bytes of the packet should contain: 
    //   - 4 bytes: number of slices (nb_slices)
    //   - 4 bytes: slice height
    //   - 4 bytes: offset to slice info table
    if (buf_size < table_size + 12)
        return AVERROR_INVALIDDATA;

    nb_slices = AV_RL32(packet->data + buf_size - 12);
    slice_height = AV_RL32(packet->data + buf_size - 8);
    slices_info_offset = AV_RL32(packet->data + buf_size - 4);

    // Validate values
    if (nb_slices <= 0 || slice_height <= 0 || 
        slices_info_offset < table_size ||
        slices_info_offset + nb_slices * 8 > (unsigned)(buf_size - 12) ||
        nb_slices > (height + slice_height - 1) / slice_height ||
        slice_height > height)
        return AVERROR_INVALIDDATA;
} else {
    nb_slices = 1;
    slice_height = height;
    slices_info_offset = 0;
}
