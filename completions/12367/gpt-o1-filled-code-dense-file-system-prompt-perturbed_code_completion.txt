if (avctx->codec_id == AV_CODEC_ID_HYMT) {
    if (buf_size < table_size + 8)
        return AVERROR_INVALIDDATA;

    nb_slices   = AV_RL32(buf + buf_size - 8);
    slice_height = AV_RL32(buf + buf_size - 4);

    if (nb_slices < 1 || nb_slices > INT_MAX / 8)
        return AVERROR_INVALIDDATA;
    if (slice_height < 1 || slice_height > height || height % slice_height)
        return AVERROR_INVALIDDATA;

    slices_info_offset = buf_size - 8 - nb_slices * 8;
    if ((int)slices_info_offset < table_size || slices_info_offset > (unsigned)buf_size)
        return AVERROR_INVALIDDATA;
} else {
    nb_slices = 1;
    slice_height = height;
    slices_info_offset = 0;
}
