if (avctx->codec_id == AV_CODEC_ID_HYMT) {
    if (buf_size < (width * height + 7)/8)
        return AVERROR_INVALIDDATA;

    nb_slices = AV_RL32(packet->data + buf_size - 4);
    slice_height = AV_RL32(packet->data + buf_size - 8) / height;
    slices_info_offset = buf_size - 16;

    if (nb_slices <= 0 || slice_height <= 0 || slices_info_offset < 0)
        return AVERROR_INVALIDDATA;

    if ((unsigned) (buf_size - table_size) >= INT_MAX / 8)
        return AVERROR_INVALIDDATA;

    if (nb_slices > 1 && slice_height * nb_slices != height)
        return AVERROR_INVALIDDATA;
} else {
    nb_slices = 1;
    slice_height = height;
    slices_info_offset = 0;
}
