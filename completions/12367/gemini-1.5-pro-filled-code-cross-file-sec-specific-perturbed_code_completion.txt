if (avctx->codec_id == AV_CODEC_ID_HYMT) {
        if (buf_size < 12)
            return AVERROR_INVALIDDATA;

        nb_slices = AV_RL32(buf + buf_size - 4);
        if (nb_slices < 1 || nb_slices > (buf_size - 8) / 8) {
            av_log(avctx, AV_LOG_ERROR, "Invalid number of slices %d\n", nb_slices);
            return AVERROR_INVALIDDATA;
        }

        slice_height = AV_RL32(buf + buf_size - 8);

        if (slice_height < 1 || slice_height > height || height % slice_height != 0 || height / slice_height != nb_slices) {
            av_log(avctx, AV_LOG_ERROR, "Invalid slice height %d\n", slice_height);
            return AVERROR_INVALIDDATA;
        }

        slices_info_offset = buf_size - 4 - nb_slices * 8;
    } else {
        nb_slices = 1;
        slice_height = height;
    }
