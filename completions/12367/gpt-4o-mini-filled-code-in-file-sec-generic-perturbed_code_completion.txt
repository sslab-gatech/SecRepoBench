if (avctx->codec_id == AV_CODEC_ID_HYMT) {
        if (buf_size < 8) {
            return AVERROR_INVALIDDATA;
        }
        slices_info_offset = AV_RL32(buf);
        nb_slices = AV_RL32(buf + 4);
        if (slices_info_offset + nb_slices * 8 > buf_size) {
            return AVERROR_INVALIDDATA;
        }
        slice_height = height / nb_slices;
    } else {
        nb_slices = 1;
        slice_height = height;
