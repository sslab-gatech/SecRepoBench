        slices_info_offset = AV_RL32(packet->data + buf_size - 8);
        nb_slices = AV_RL32(packet->data + slices_info_offset);
        slice_height = AV_RL32(packet->data + slices_info_offset + 4);

        if (nb_slices <= 0 || slice_height <= 0 || slices_info_offset < 0 ||
            slices_info_offset + nb_slices * 8 > buf_size)
            return AVERROR_INVALIDDATA;

        if (slice_height > height)
            return AVERROR_INVALIDDATA;

    } else {
        nb_slices = 1;
