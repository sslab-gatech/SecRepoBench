if (avctx->codec_id == AV_CODEC_ID_HYMT &&
        buf_size > 8 &&
        !memcmp(packet->data + buf_size - 8, "endimage", 8)) {
        nb_slices = AV_RL32(packet->data + buf_size - 12);
        slice_height = AV_RL32(packet->data + buf_size - 16);
        slices_info_offset = buf_size - 8 - nb_slices * 8;

        if (nb_slices <= 0 || nb_slices > height || slice_height <= 0 ||
            slice_height > height || slices_info_offset > buf_size ||
            slices_info_offset < table_size ||
            (unsigned)nb_slices > (buf_size - table_size) / 8) {
            av_log(avctx, AV_LOG_ERROR, "Invalid slice parameters\n");
            return AVERROR_INVALIDDATA;
        }
    } else {
        nb_slices = 1;
        slice_height = height;
        slices_info_offset = 0;
    }
