if (avctx->codec_id == AV_CODEC_ID_HYMT) {
    if (buf_size < 20 || buf_size < AV_RL32(buf + 16) + 8) {
        av_log(avctx, AV_LOG_ERROR, "Invalid buffer size for codec HYMT.\n");
        return AVERROR_INVALIDDATA;
    }

    slices_info_offset = AV_RL32(buf + 16);
    if (slices_info_offset > buf_size - 8 * ((buf_size - slices_info_offset) / 8)) {
        av_log(avctx, AV_LOG_ERROR, "Invalid slice information offset.\n");
        return AVERROR_INVALIDDATA;
    }

    nb_slices = (buf_size - slices_info_offset) / 8;
    if (nb_slices <= 0 || nb_slices > height) {
        av_log(avctx, AV_LOG_ERROR, "Invalid number of slices.\n");
        return AVERROR_INVALIDDATA;
    }

    slice_height = height / nb_slices;
    if (slice_height * nb_slices != height) {
        av_log(avctx, AV_LOG_ERROR, "Slice height does not divide evenly into frame height.\n");
        return AVERROR_INVALIDDATA;
    }
} else {
    nb_slices = 1;
    slice_height = height;
}
