if (avctx->codec_id == AV_CODEC_ID_HYMT) {
    // Ensure packet is large enough for basic slice info
    if (buf_size <= 16) {
        return AVERROR_INVALIDDATA;
    }
    
    // Read number of slices and slice height from the end of the packet
    nb_slices = AV_RL32(packet->data + buf_size - 8);
    slice_height = AV_RL32(packet->data + buf_size - 4);
    
    // Validate the slice values
    if (nb_slices <= 0 || slice_height <= 0 || 
        nb_slices > height || slice_height * nb_slices > height) {
        return AVERROR_INVALIDDATA;
    }
    
    // Check for potential integer overflow and ensure all slice info fits
    if (nb_slices > (INT_MAX / 8) || buf_size < table_size + nb_slices * 8 + 8) {
        return AVERROR_INVALIDDATA;
    }
    
    // Calculate where the slice info begins
    slices_info_offset = buf_size - 8 - nb_slices * 8;
    
    // Make sure the offset is valid and doesn't overlap with huffman tables
    if (slices_info_offset < table_size) {
        return AVERROR_INVALIDDATA;
    }
} else {
    // Default to a single slice for the entire height
    nb_slices = 1;
    slice_height = height;
    slices_info_offset = 0;
}
