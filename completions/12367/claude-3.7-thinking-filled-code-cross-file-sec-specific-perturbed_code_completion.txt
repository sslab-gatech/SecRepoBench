if (avctx->codec_id == AV_CODEC_ID_HYMT) {
    // HYMT format has slice info at the end of the packet
    
    // Minimum size needed: 4 bytes for slice count + at least 8 bytes for one slice info
    if (buf_size < 12)
        return AVERROR_INVALIDDATA;
    
    // Last 4 bytes contain the number of slices
    nb_slices = AV_RL32(buf + buf_size - 4);
    
    // Validate number of slices
    if (nb_slices <= 0 || nb_slices > height)
        return AVERROR_INVALIDDATA;
    
    // Check for potential overflow in nb_slices * 8
    if (nb_slices > (INT_MAX - 4) / 8)
        return AVERROR_INVALIDDATA;
    
    // Each slice needs 8 bytes of info (offset + size)
    // Total space needed: nb_slices * 8 + 4 bytes for count
    if (buf_size < nb_slices * 8 + 4)
        return AVERROR_INVALIDDATA;
    
    slices_info_offset = buf_size - 4 - nb_slices * 8;
    
    // Validate we have enough data for the slice info
    if (slices_info_offset < table_size)
        return AVERROR_INVALIDDATA;
    
    // Calculate the height of each slice
    slice_height = (height + nb_slices - 1) / nb_slices;
} else {
    // Default to a single slice for other codecs
    nb_slices = 1;
    slice_height = height;
    slices_info_offset = 0;
}
