if (avctx->codec_id == AV_CODEC_ID_HYMT) {
        if (buf_size < 4)
            return AVERROR_INVALIDDATA;
        slices_info_offset = AV_RL32(buf + buf_size - 4);
        if (slices_info_offset < 0 || slices_info_offset >= buf_size - 4)
            return AVERROR_INVALIDDATA;
        nb_slices = AV_RL32(buf + slices_info_offset);
        if (nb_slices <= 0 || slices_info_offset + 4 + 8 * nb_slices > buf_size)
            return AVERROR_INVALIDDATA;
        slice_height = (height + nb_slices - 1) / nb_slices;
        if (slice_height <= 0 || slice_height * nb_slices < height)
            return AVERROR_INVALIDDATA;
        for (int i = 0; i < nb_slices; i++) {
            uint32_t offset = AV_RL32(buf + slices_info_offset + 4 + i * 8);
            uint32_t size = AV_RL32(buf + slices_info_offset + 4 + i * 8 + 4);
            if (offset >= buf_size || size == 0 || offset + size > buf_size)
                return AVERROR_INVALIDDATA;
        }
        if (slices_info_offset < table_size)
            return AVERROR_INVALIDDATA;
        slices_info_offset += 4;
    } else {
        nb_slices = 1;
        slice_height = height;
        slices_info_offset = 0;
    }
