if (avctx->codec_id == AV_CODEC_ID_HYMT) {
        if (buf_size < 8)
            return AVERROR_INVALIDDATA;

        slices_info_offset = AV_RL32(buf + buf_size - 8);
        nb_slices = AV_RL16(buf + buf_size - 4);
        slice_height = AV_RL16(buf + buf_size - 2);

        if (slices_info_offset >= buf_size || slices_info_offset + nb_slices * 8 > buf_size) {
            return AVERROR_INVALIDDATA;
        }

        if (nb_slices <= 0 || slice_height <= 0 || slice_height > height) {
            return AVERROR_INVALIDDATA;
        }
    } else {
        nb_slices = 1;
        slice_height = height;
        slices_info_offset = 0;
    }
