if (avctx->codec_id == AV_CODEC_ID_HYMT) {
    if (buf_size < 20) {
        av_log(avctx, AV_LOG_ERROR, "Invalid buffer size for HYMT codec\n");
        return AVERROR_INVALIDDATA;
    }

    slices_info_offset = AV_RL32(buf + 4);
    if (slices_info_offset >= buf_size || slices_info_offset < 8) {
        av_log(avctx, AV_LOG_ERROR, "Invalid slices info offset\n");
        return AVERROR_INVALIDDATA;
    }

    nb_slices = AV_RL32(buf + 8);
    if (nb_slices <= 0 || nb_slices > height) {
        av_log(avctx, AV_LOG_ERROR, "Invalid number of slices\n");
        return AVERROR_INVALIDDATA;
    }

    slice_height = height / nb_slices;
    if (slice_height * nb_slices != height) {
        av_log(avctx, AV_LOG_ERROR, "Slice height does not divide evenly into frame height\n");
        return AVERROR_INVALIDDATA;
    }
} else {
    nb_slices = 1;
    slice_height = height;
}
