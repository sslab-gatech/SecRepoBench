if (avctx->codec_id == AV_CODEC_ID_HYMT) {
        if (buf_size < 4)
            return AVERROR_INVALIDDATA;
        slices_info_offset = AV_RL32(buf + buf_size - 4);
        if (slices_info_offset > buf_size - 4)
            return AVERROR_INVALIDDATA;
        int remaining = buf_size - 4 - slices_info_offset;
        if (remaining < 0 || remaining % 8 != 0)
            return AVERROR_INVALIDDATA;
        nb_slices = remaining / 8;
        if (nb_slices <= 0)
            return AVERROR_INVALIDDATA;
        slice_height = height / nb_slices;
        if (slice_height <= 0 || slice_height * nb_slices != height)
            return AVERROR_INVALIDDATA;
    } else {
        nb_slices = 1;
        slice_height = height;
        slices_info_offset = 0;
    }
