if (avctx->codec_id == AV_CODEC_ID_HYMT && buf_size > 8) {
        nb_slices = AV_RL32(buf);
        slice_height = AV_RL32(buf + 4);
        slices_info_offset = 8;

        if (nb_slices <= 0 || nb_slices > height || slice_height <= 0 ||
            slice_height > height || nb_slices * slice_height != height ||
            (unsigned)slices_info_offset + (unsigned)nb_slices * 8 > buf_size) {
            av_log(avctx, AV_LOG_ERROR, "Invalid slice parameters.\n");
            return AVERROR_INVALIDDATA;
        }
    } else {
        nb_slices = 1;
        slice_height = height;
        slices_info_offset = 0;
    }
