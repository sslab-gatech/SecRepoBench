    // and specific conditions at the end of the packet data. Determine
    // the number of slices, slice height, and offset for slice information.
    // If any of these are invalid, return an error. 
    // If the codec ID does not match, default to a single slice covering the full height.
    if (avctx->codec_id == AV_CODEC_ID_HYMT) {

            return AVERROR_INVALIDDATA;

        // Validate specific conditions at the end of the packet data
        if (buf_size < table_size + 16) // Additional 16 bytes for header
            return AVERROR_INVALIDDATA;

        // Determine the number of slices, slice height, and offset for slice information
        nb_slices = AV_RL32(packet->data + buf_size - 16); // Assuming 4 bytes for nb_slices
        if (nb_slices <= 0 || nb_slices > 100) // Sanity check
            return AVERROR_INVALIDDATA;

        slice_height = height / nb_slices;
        if (slice_height * nb_slices != height)
            return AVERROR_INVALIDDATA;

        slices_info_offset = AV_RL32(packet->data + buf_size - 20); // Assuming 4 bytes for offset
        if (slices_info_offset < 0 || slices_info_offset >= buf_size - 16)
            return AVERROR_INVALIDDATA;

        slice_height = height;
