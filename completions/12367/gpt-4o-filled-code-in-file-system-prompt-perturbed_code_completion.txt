if (avctx->codec_id == AV_CODEC_ID_HYMT) {
        if (buf_size < 8 || AV_RL32(packet->data + buf_size - 8) != buf_size - 8) {
            av_log(avctx, AV_LOG_ERROR, "Invalid buffer size or missing slice information\n");
            return AVERROR_INVALIDDATA;
        }
        nb_slices = AV_RL32(packet->data + buf_size - 4);
        if (nb_slices <= 0 || nb_slices > height) {
            av_log(avctx, AV_LOG_ERROR, "Invalid number of slices\n");
            return AVERROR_INVALIDDATA;
        }
        slice_height = (height + nb_slices - 1) / nb_slices;
        slices_info_offset = buf_size - 8 - nb_slices * 8;
        if (slices_info_offset < table_size) {
            av_log(avctx, AV_LOG_ERROR, "Invalid slice information offset\n");
            return AVERROR_INVALIDDATA;
        }
    } else {
        nb_slices = 1;
        slice_height = height;
        slices_info_offset = 0;
    }
