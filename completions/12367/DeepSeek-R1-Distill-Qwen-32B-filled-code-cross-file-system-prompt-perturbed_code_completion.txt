if (avctx->codec_id == AV_CODEC_ID_HYMT) {
    if ((buf_size - table_size) < 16)
        return AVERROR_INVALIDDATA;

    nb_slices = AV_RL32(buf + table_size);
    slice_height = AV_RL32(buf + table_size + 4);

    if (nb_slices <= 0 || nb_slices > height || slice_height <= 0 ||
        nb_slices * slice_height > height)
        return AVERROR_INVALIDDATA;

    slices_info_offset = table_size + 8;
} else {
    nb_slices = 1;
    slice_height = height;
    slices_info_offset = 0;
}
