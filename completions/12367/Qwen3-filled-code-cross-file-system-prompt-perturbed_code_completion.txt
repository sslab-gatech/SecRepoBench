if (avctx->codec_id == AV_CODEC_ID_HYMT) {
    slices_info_offset = table_size;
    if (buf_size < slices_info_offset + 8) {
        return AVERROR_INVALIDDATA;
    }
    nb_slices = AV_RL32(packet->data + slices_info_offset);
    slice_height = AV_RL32(packet->data + slices_info_offset + 4);
    slices_info_offset += 8;

    if (nb_slices <= 0 || slice_height <= 0 ||
        (unsigned)slice_height > height ||
        height % slice_height != 0 ||
        (unsigned)(slices_info_offset + nb_slices * 8) > buf_size) {
        return AVERROR_INVALIDDATA;
    }
} else {
    nb_slices = 1;
    slice_height = height;
    slices_info_offset = 0;
}
