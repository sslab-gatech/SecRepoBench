if (avctx->codec_id == AV_CODEC_ID_HYMT) {
        if (buf_size < 16 || buf_size < AV_RL32(buf + buf_size - 12) + 4 || AV_RL32(buf + buf_size - 8) != 0 || AV_RL32(buf + buf_size - 4) != 0) {
            av_log(avctx, AV_LOG_ERROR, "Invalid buffer size or corrupted packet detected.\n");
            return AVERROR_INVALIDDATA;
        }

        nb_slices = AV_RL32(buf + buf_size - 16);
        if (nb_slices < 1 || nb_slices > (buf_size - 16) / 8) {
            av_log(avctx, AV_LOG_ERROR, "Invalid number of slices: %d\n", nb_slices);
            return AVERROR_INVALIDDATA;
        }

        slice_height = (height + nb_slices - 1) / nb_slices;
        slices_info_offset = buf_size - 16 - nb_slices * 8;
    } else {
        nb_slices = 1;
        slice_height = height;
    }
