if (avctx->codec_id == AV_CODEC_ID_HYMT && buf_size > 8) {
        if (AV_RL64(buf + buf_size - 8) != 0x0100000000000000ULL) {
            av_log(avctx, AV_LOG_ERROR, "Invalid end of packet data\n");
            return AVERROR_INVALIDDATA;
        }

        nb_slices = AV_RL32(buf + buf_size - 4);
        if (nb_slices <= 0 || nb_slices > height) {
            av_log(avctx, AV_LOG_ERROR, "Invalid number of slices: %d\n", nb_slices);
            return AVERROR_INVALIDDATA;
        }

        slice_height = (height + nb_slices - 1) / nb_slices;
        slices_info_offset = buf_size - 8 - nb_slices * 8;

        if (slices_info_offset > buf_size - 8 || slices_info_offset < table_size) {
            av_log(avctx, AV_LOG_ERROR, "Invalid slices info offset: %u\n", slices_info_offset);
            return AVERROR_INVALIDDATA;
        }
    } else {
        nb_slices = 1;
        slice_height = height;
    }
