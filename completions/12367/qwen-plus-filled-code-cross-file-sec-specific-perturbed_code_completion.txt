if (avctx->codec_id == AV_CODEC_ID_HYMT) {
    if (buf_size < 8 || packet->size < 8)
        return AVERROR_INVALIDDATA;

    slices_info_offset = AV_RL32(buf + buf_size - 8);
    nb_slices = AV_RL32(buf + buf_size - 4);

    if (slices_info_offset >= (unsigned)buf_size || nb_slices <= 0 || 
        slices_info_offset + nb_slices * 8 > (unsigned)buf_size)
        return AVERROR_INVALIDDATA;

    slice_height = height / nb_slices;
    if (height % nb_slices != 0)
        return AVERROR_INVALIDDATA;
} else {
    nb_slices = 1;
    slice_height = height;
}
