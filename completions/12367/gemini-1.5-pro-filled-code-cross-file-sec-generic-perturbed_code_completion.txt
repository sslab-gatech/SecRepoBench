if (avctx->codec_id == AV_CODEC_ID_HYMT) {
        if (buf_size < 12 || buf[buf_size - 12] != 1 || buf[buf_size - 8] != 1) {
            av_log(avctx, AV_LOG_ERROR, "Invalid HYMT packet format.\n");
            return AVERROR_INVALIDDATA;
        }

        nb_slices = AV_RL32(buf + buf_size - 4);
        if (nb_slices <= 0 || nb_slices > (buf_size - 12) / 8) {
            av_log(avctx, AV_LOG_ERROR, "Invalid number of slices: %d\n", nb_slices);
            return AVERROR_INVALIDDATA;
        }

        slice_height = height / nb_slices;
        if (height % nb_slices != 0) {
            av_log(avctx, AV_LOG_ERROR, "Uneven slice height not supported.\n");
            return AVERROR_INVALIDDATA;
        }

        slices_info_offset = buf_size - 12 - nb_slices * 8;
    } else {
        nb_slices = 1;
        slice_height = height;
    }
