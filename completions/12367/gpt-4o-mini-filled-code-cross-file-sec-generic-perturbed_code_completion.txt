if (avctx->codec_id == AV_CODEC_ID_HYMT) {
        // Validate the buffer size and ensure it meets the expected conditions
        if (buf_size < sizeof(uint32_t) * 2) {
            return AVERROR_INVALIDDATA;
        }

        slices_info_offset = AV_RL32(buf + buf_size - 8);
        nb_slices = AV_RL32(buf + buf_size - 4);
        slice_height = (height + nb_slices - 1) / nb_slices;

        // Check for valid slice information
        if (slices_info_offset + nb_slices * 8 > buf_size || slice_height <= 0) {
            return AVERROR_INVALIDDATA;
        }
    } else {
        nb_slices = 1;
        slice_height = height;
        slices_info_offset = 0;
    }
