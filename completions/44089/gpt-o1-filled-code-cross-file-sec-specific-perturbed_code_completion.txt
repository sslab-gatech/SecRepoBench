{
    /* Allocate buffer for numeric substring plus null terminator */
    char *num_str = (char *)malloc(str_len + 1);
    if (!num_str) {
        PLIST_JSON_ERR("%s: memory allocation error\n", __func__);
    } else {
        memcpy(num_str, str_val, str_len);
        num_str[str_len] = '\0';

        bool is_float = false;
        for (size_t i = 0; i < str_len; i++) {
            if (num_str[i] == '.' || num_str[i] == 'e' || num_str[i] == 'E') {
                is_float = true;
                break;
            }
        }

        char *endp = NULL;
        if (is_float) {
            double dval = strtod(num_str, &endp);
            if (endp != num_str + str_len) {
                PLIST_JSON_ERR("%s: invalid numeric value '%s'\n", __func__, num_str);
            } else {
                parsed_value = plist_new_real(dval);
            }
        } else {
            long long llval = strtoll(num_str, &endp, 10);
            if (endp != num_str + str_len) {
                PLIST_JSON_ERR("%s: invalid numeric value '%s'\n", __func__, num_str);
            } else if (llval < 0) {
                PLIST_JSON_ERR("%s: negative integer value '%lld' not supported as unsigned\n", __func__, llval);
            } else {
                parsed_value = plist_new_uint((uint64_t)llval);
            }
        }
        free(num_str);
    }
}
