char* endp = NULL;
long long int ival = strtoll(str_val, &endp, 10);
if (endp != str_end) {
    parsed_value = plist_new_uint(ival);
} else {
    double fval = strtod(str_val, &endp);
    if (endp != str_end) {
        parsed_value = plist_new_real(fval);
    } else {
        PLIST_JSON_ERR("%s: invalid numeric value '%.*s' encountered\n", __func__, (int)str_len, str_val);
    }
}
