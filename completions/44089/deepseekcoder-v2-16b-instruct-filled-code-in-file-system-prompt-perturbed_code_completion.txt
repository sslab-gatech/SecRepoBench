char* endp = NULL;
double num_val = strtod(str_val, &endp);
if (endp != str_end) {
    // Check if there are any invalid characters after the number
    if (endp != str_val && *endp != '\0') {
        PLIST_JSON_ERR("%s: invalid character after number\n", __func__);
        return NULL;
    }
    // Create a new real plist node
    plist_t node = plist_new_real(num_val);
    return node;
}
