uint64_t u64val = strtoull(str_val, &endp, 0);
if (endp == str_end) {
    parsed_value = plist_new_uint(u64val);
} else {
    double dval = strtod(str_val, &endp);
    if (endp == str_end) {
        parsed_value = plist_new_real(dval);
    } else {
        PLIST_JSON_ERR("%s: invalid numeric value '%.*s' encountered\n", __func__, (int)str_len, str_val);
    }
}
