double float_val = strtod(str_val, &endp);
if (endp && endp == str_end) {
    if (strchr(str_val, '.') || strchr(str_val, 'e') || strchr(str_val, 'E')) {
        // It's a floating-point number
        parsed_value = plist_new_real(float_val);
    } else {
        // It's an integer
        int64_t int_val = strtoll(str_val, &endp, 10);
        if (endp && endp == str_end) {
            parsed_value = plist_new_uint(int_val);
        } else {
            PLIST_JSON_ERR("%s: invalid integer value '%.*s' encountered\n", __func__, (int)str_len, str_val);
        }
    }
} else {
    PLIST_JSON_ERR("%s: invalid numeric value '%.*s' encountered\n", __func__, (int)str_len, str_val);
}
