long long int int_val = strtoll(str_val, &endp, 10);
if (endp == str_end) {
    // Successfully parsed as integer
    parsed_value = plist_new_uint(int_val);
} else {
    // Try parsing as a floating-point number
    endp = NULL;
    double real_val = strtod(str_val, &endp);
    if (endp == str_end) {
        // Successfully parsed as real number
        parsed_value = plist_new_real(real_val);
    } else {
        // Not a valid number
        PLIST_JSON_ERR("%s: invalid numeric value '%.*s' encountered\n", __func__, (int)str_len, str_val);
    }
}
