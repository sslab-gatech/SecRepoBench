if (str_val[0] == '-' || isdigit(str_val[0])) {
    char* endp = NULL;
    long long value = strtoll(str_val, &endp, 10);
    if (endp!= str_val) {
        // The string was not a valid number
        PLIST_JSON_ERR("%s: invalid numerical value '%.*s' encountered\n", __func__, (int)str_len, str_val);
    } else {
        if (value < 0) {
            // The number is negative, create a new signed integer plist node
            plist_data_t data = plist_new_plist_data();
            data->type = PLIST_SIGNED;
            data->signed_value = (int64_t)value;
            parsed_value = plist_new_node(data);
        } else {
            // The number is non-negative, create a new unsigned integer plist node
            plist_data_t data = plist_new_plist_data();
            data->type = PLIST_UNSIGNED;
            data->unsigned_value = (uint64_t)value;
            parsed_value = plist_new_node(data);
        }
    }
