    double value = strtod(str_val, &endp);
    if (endp != str_end) {
        PLIST_JSON_ERR("%s: invalid characters in number '%.*s'\n", __func__, (int)str_len, str_val);
        return NULL;
    }
    if (value == (unsigned long long)value && value >= 0) {
        parsed_value = plist_new_uint((unsigned long long)value);

        data->type = PLIST_REAL;
        data->realval = value;
