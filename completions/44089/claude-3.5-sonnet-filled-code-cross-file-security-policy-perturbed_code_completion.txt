char buffer[64];
size_t copy_len = (str_len < sizeof(buffer) - 1) ? str_len : sizeof(buffer) - 1;
memcpy(buffer, str_val, copy_len);
buffer[copy_len] = '\0';

long long int_val = strtoll(buffer, &endp, 10);
if (*endp == '\0') {
    parsed_value = plist_new_uint(int_val);
} else {
    double real_val = strtod(buffer, &endp);
    if (*endp == '\0') {
        parsed_value = plist_new_real(real_val);
    } else {
        PLIST_JSON_ERR("%s: invalid numeric value '%s' encountered\n", __func__, buffer);
    }
}
