char *tok_str = strndup(str_val, str_len);
        if (!tok_str) {
            PLIST_JSON_ERR("%s: out of memory\n", __func__);
        } else {
            char *endp = NULL;
            int is_float = 0;
            /* Check if the string contains '.' or 'e'/'E' to determine if it's a float */
            for (size_t i = 0; i < str_len; i++) {
                if (tok_str[i] == '.' || tok_str[i] == 'e' || tok_str[i] == 'E') {
                    is_float = 1;
                    break;
                }
            }

            plist_data_t data = NULL;
            errno = 0;

            if (is_float) {
                /* Parse as a floating-point (real) value */
                double dval = strtod(tok_str, &endp);
                if ((endp != tok_str + str_len) || errno == ERANGE) {
                    PLIST_JSON_ERR("%s: invalid numeric value '%s'\n", __func__, tok_str);
                } else {
                    data = plist_new_plist_data();
                    data->type = PLIST_REAL;
                    data->realval = dval;
                    parsed_value = plist_new_node(data);
                }
            } else {
                /* Parse as an integer value */
                data = plist_new_plist_data();
                data->type = PLIST_UINT;
                if (tok_str[0] == '-') {
                    long long lval = strtoll(tok_str, &endp, 10);
                    if ((endp != tok_str + str_len) || errno == ERANGE) {
                        PLIST_JSON_ERR("%s: invalid numeric value '%s'\n", __func__, tok_str);
                        free(data);
                        data = NULL;
                    } else {
                        data->intval = (uint64_t)lval;
                        data->length = 8; /* negative => treat as signed */
                        parsed_value = plist_new_node(data);
                    }
                } else {
                    unsigned long long ullval = strtoull(tok_str, &endp, 10);
                    if ((endp != tok_str + str_len) || errno == ERANGE) {
                        PLIST_JSON_ERR("%s: invalid numeric value '%s'\n", __func__, tok_str);
                        free(data);
                        data = NULL;
                    } else {
                        data->intval = (uint64_t)ullval;
                        data->length = 16; /* positive => treat as unsigned */
                        parsed_value = plist_new_node(data);
                    }
                }
            }
            free(tok_str);
        }
