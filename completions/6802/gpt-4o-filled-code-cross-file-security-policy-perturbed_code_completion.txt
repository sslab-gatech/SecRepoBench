{
  if (*p == '{' || *p == '[')
  {
    char
      *sub_q;

    level++;
    sub_q=subexpression;
    for (p++; *p != '\0'; )
    {
      if (*p == '{' || *p == '[')
        level++;
      else if (*p == '}' || *p == ']')
      {
        level--;
        if (level == 0)
          break;
      }
      if ((sub_q - subexpression) < (MagickPathExtent - 1))
        *sub_q++=(*p++);
      else
        break;
    }
    *sub_q='\0';
    if (*p != '\0')
      p++;
    alpha=FxEvaluateSubexpression(fx_info,pixchannel,x,y,subexpression,
      &depth,&beta,exception);
    point.x=alpha;
    point.y=beta;
  }
}
