            while (*++p != '\0')
            {
              if (*p == '{' || *p == '[') brace_count++;
              else if (*p == '}' || *p == ']') brace_count--;
              if (brace_count == 0) break;
            }
            if (brace_count == 0)
            {
              size_t length = p - start + 1;
              char subexpression[length];
              strncpy(subexpression, start, length - 1);
              subexpression[length - 1] = '\0';

              point.x = alpha; // Assign the evaluated x value
