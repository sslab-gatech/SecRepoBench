            {
              level++;
              for (p++; *p != '\0'; )
              {
                if (*p == '{')
                  level++;
                else
                  if (*p == '}')
                    {
                      level--;
                      if (level == 0)
                        break;
                    }
                *q++=(*p++);
              }
              *q='\0';
              p++;
            }
          else
            if (*p == '[')
              {
                level++;
                for (p++; *p != '\0'; )
                {
                  if (*p == '[')
                    level++;
                  else
                    if (*p == ']')
                      {
                        level--;
                        if (level == 0)
                          break;
                      }
                  *q++=(*p++);
                }
                *q='\0';
                p++;
              }
            else
              (void) ThrowMagickException(exception,GetMagickModule(),
                OptionError,"UnbalancedParenthesis","`%s'",expression);
