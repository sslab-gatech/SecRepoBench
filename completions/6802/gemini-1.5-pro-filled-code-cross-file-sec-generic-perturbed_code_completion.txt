                if (*p == '{')

                    }
                *q++=(*p++);
              }
              *q='\0';
              p++;
           }

            if (*p == '[')

                level++;
                q=subexpression;
                for (p++; *p != '\0'; )
                {
                  if (*p == '[')
                    level++;
                  else
                    if (*p == ']')
                      {
                        level--;
                        if (level == 0)
                          break;
                      }
                  *q++=(*p++);
                }
                *q='\0';
                p++;
              }
