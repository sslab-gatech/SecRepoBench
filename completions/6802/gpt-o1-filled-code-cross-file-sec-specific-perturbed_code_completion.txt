{
  if (((*(p+1) == '{') || (*(p+1) == '[')))
    {
      char
        bracket_open,
        bracket_close,
        *sep;

      bracket_open=*(p+1);
      bracket_close=(bracket_open == '{') ? '}' : ']';
      level++;
      q=subexpression;
      p+=2;
      for ( ; *p != '\0'; p++)
      {
        if (*p == bracket_open)
          level++;
        else if (*p == bracket_close)
          {
            level--;
            if (level == 0)
              break;
          }
        if ((size_t) (q - subexpression) < (MagickPathExtent-1))
          *q++=(*p);
      }
      *q='\0';
      if (*p != '\0')
        p++;
      sep=strchr(subexpression,',');
      if (sep != (char *) NULL)
        {
          *sep='\0';
          alpha=FxEvaluateSubexpression(fx_info,pixchannel,x,y,subexpression,
            &depth,&beta,exception);
          point.x=alpha;
          alpha=FxEvaluateSubexpression(fx_info,pixchannel,x,y,sep+1,
            &depth,&beta,exception);
          point.y=alpha;
        }
      else
        {
          alpha=FxEvaluateSubexpression(fx_info,pixchannel,x,y,subexpression,
            &depth,&beta,exception);
          point.x=alpha;
          point.y=0.0;
        }
      if (*p == '.')
        p++;
    }
}
