{
  if ((*p == '{') || (*p == '['))
    {
      char
        start = *p,
        stop = (start == '{') ? '}' : ']';

      level++;
      q=subexpression;
      for (p++; *p != '\0'; )
      {
        if (*p == start)
          level++;
        else if (*p == stop)
        {
          level--;
          if (level == 0)
            break;
        }
        *q++=(*p++);
      }
      *q='\0';
      alpha=FxEvaluateSubexpression(fx_info,pixchannel,x,y,subexpression,
        &depth,&beta,exception);
      point.x=alpha;
      point.y=beta;
      if (*p == stop)
        p++;
      if (*p == '.')
        p++;
    }
}
