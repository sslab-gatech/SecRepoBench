size_t
            level;

          level=0;
          q=subexpression;
          for (p++; *p != '\0'; )
          {
            if (*p == '{')
              level++;
            else
              if (*p == '}')
                {
                  level--;
                  if (level == 0)
                    break;
                }
            *q++=(*p++);
          }
          *q='\0';
          alpha=FxEvaluateSubexpression(fx_info,pixchannel,x,y,subexpression,
            &depth,&beta,exception);
          point.x=alpha;
          p=FxOperatorPrecedence(++p,exception);
          if (p == (const char *) NULL)
            {
              (void) ThrowMagickException(exception,GetMagickModule(),
                OptionError,"UnbalancedParenthesis","`%s'",expression);
              return(0.0);
            }
          (void) CopyMagickString(subexpression,p,MagickPathExtent);
          beta=FxEvaluateSubexpression(fx_info,pixchannel,x,y,subexpression,
            &depth,&alpha,exception);
          point.y=beta;
          p++;
        }
