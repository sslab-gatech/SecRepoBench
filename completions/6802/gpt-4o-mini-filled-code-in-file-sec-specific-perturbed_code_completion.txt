if (*p == '{' || *p == '[')
          {
            const char *subexpression_end = FxSubexpression(p, exception);
            if (subexpression_end == NULL)
              return 0.0; // Error handling for unbalanced parentheses

            // Evaluate the subexpression and assign its result to point.x and point.y
            point.x = FxEvaluateSubexpression(fx_info, channel, x, y, subexpression, &depth, &beta, exception);
            point.y = FxEvaluateSubexpression(fx_info, channel, x, y, subexpression_end, &depth, &beta, exception);

            // Update the expression pointer 'p' to continue processing the rest of the expression
            p = subexpression_end + 1; // Move past the closing bracket
          }
