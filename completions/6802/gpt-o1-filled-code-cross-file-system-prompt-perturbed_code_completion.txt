p++;
          if ((*p == '[') || (*p == '{'))
            {
              char
                start_bracket = *p,
                end_bracket = (start_bracket == '[') ? ']' : '}';

              level++;
              q=subexpression;
              for (p++; *p != '\0'; )
              {
                if (*p == start_bracket)
                  level++;
                else if (*p == end_bracket)
                  {
                    level--;
                    if (level == 0)
                      break;
                  }
                *q++ = (*p++);
              }
              *q = '\0';
              p++;
              {
                char
                  *separator;

                separator=strchr(subexpression,',');
                if (separator != (char *) NULL)
                  {
                    *separator='\0';
                    separator++;
                    alpha=FxEvaluateSubexpression(fx_info,pixchannel,x,y,subexpression,
                      &depth,&beta,exception);
                    point.x=alpha;
                    alpha=FxEvaluateSubexpression(fx_info,pixchannel,x,y,separator,
                      &depth,&beta,exception);
                    point.y=alpha;
                  }
                else
                  {
                    alpha=FxEvaluateSubexpression(fx_info,pixchannel,x,y,subexpression,
                      &depth,&beta,exception);
                    point.x=alpha;
                    point.y=0.0;
                  }
              }
            }
          if (*p == '.')
            p++;
