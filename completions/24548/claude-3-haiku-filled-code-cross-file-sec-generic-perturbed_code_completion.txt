unsigned tail_length = c->length () - initial_table_position;
this->record_count = obj_indices.length;
this->record_size = tail_length - sizeof (CmapSubtableFormat14);
c->check_assign (this->length, tail_length, HB_SERIALIZE_ERROR_OVERWRITE);
c->check_assign (this->record_length, this->record_size, HB_SERIALIZE_ERROR_OVERWRITE);

// Reverse the order of the variation selector records
for (unsigned i = 0; i < obj_indices.length / 2; i++)
{
  hb_pair_t<unsigned, unsigned> temp = obj_indices[i];
  obj_indices[i] = obj_indices[obj_indices.length - 1 - i];
  obj_indices[obj_indices.length - 1 - i] = temp;
}
