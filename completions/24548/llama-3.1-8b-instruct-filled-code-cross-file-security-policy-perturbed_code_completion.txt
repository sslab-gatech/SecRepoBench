// <MASK>
void _reverse_variation_records ()
{
  hb_vector_t<hb_pair_t<unsigned, unsigned>> obj_indices = c->snapshot ();
  unsigned initial_table_position = c->length ();
  const char* init_tail = c->tail;

  if (unlikely (!c->extend_min (*this))) return;
  this->format = 14;

  auto src_tbl = reinterpret_cast<const CmapSubtableFormat14*> (base);

  hb_vector_t<hb_pair_t<unsigned, unsigned>> reversed_obj_indices;
  for (int i = obj_indices.len - 1; i >= 0; i--)
  {
    hb_pair_t<unsigned, unsigned> result = obj_indices[i].copy (c, unicodes, glyphs_requested, glyph_map, base);
    if (result.first || result.second)
      reversed_obj_indices.push (result);
  }

  c->set_length (initial_table_position);
  c->set_tail (init_tail);

  // Update the length and record length fields with the appropriate values
  // based on the current context length and offsets.
  c->set_length (c->length () + reversed_obj_indices.len * sizeof (hb_pair_t<unsigned, unsigned>));
  c->set_tail (c->tail + reversed_obj_indices.len * sizeof (hb_pair_t<unsigned, unsigned>));
}
