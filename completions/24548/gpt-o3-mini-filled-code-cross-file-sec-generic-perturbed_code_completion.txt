unsigned tail_length = c->length() - initial_table_position;
/* Patch in the overall subtable length.
 * Assuming the table header layout is:
 *   uint16_t format;                // at offset 0
 *   uint32_t length;                // at offset 2
 *   uint8_t  numVarSelectorRecords; // at offset 6
 */
c->patch32(initial_table_position + 2, tail_length);

/* Patch in the number of variation selector records. */
c->patch8(initial_table_position + 6, obj_indices.length());
