unsigned tail_length = c->length() - initial_table_position;
    if (tail_length > 0 && tail_length <= c->capacity() - initial_table_position)
    {
      this->length = tail_length;
      this->record_length = obj_indices.length;
    }
    else
    {
      // Handle error: tail length is invalid or exceeds buffer capacity
      c->rollback(snap);
