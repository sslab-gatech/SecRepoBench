if (argcount < 0) {
  mrb_raise(mrb, E_ARGUMENT_ERROR, "negative argument count");
}

c = mrb_class(mrb, self);
m = mrb_method_search_vm(mrb, &c, mid);

if (argcount > CALL_MAXARGS) {
  mrb_value args_array = mrb_ary_new_capa(mrb, argcount);
  for (mrb_int i = 0; i < argcount; ++i) {
    mrb_ary_push(mrb, args_array, argv[i]);
  }
  argv = &args_array;
  argcount = 1;
}
  
n += (argcount + 1); // account for blk
mrb_stack_extend(mrb, n);
