if (argcount < 0) {
      mrb_raise(mrb, E_ARGUMENT_ERROR, "negative argument count");
    }

    m = mrb_method_search_vm(mrb, &c, mid);
    if (MRB_METHOD_UNDEF_P(m)) {
      mrb_value args = mrb_ary_new_from_values(mrb, argcount, argv);
      mrb_method_missing(mrb, mid, self, args);
    }

    if (argcount > 14) {
      mrb_value args = mrb_ary_new_from_values(mrb, argcount, argv);
      mrb_stack_extend(mrb, n + 2);
      ci->stack[n + 1] = args;
      argcount = 15;
    } else {
      mrb_stack_extend(mrb, n + argcount + 2);
      stack_copy(ci->stack + 1, argv, argcount);
    }
