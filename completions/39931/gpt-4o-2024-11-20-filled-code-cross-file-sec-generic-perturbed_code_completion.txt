if (argcount < 0) {
      mrb_raise(mrb, E_ARGUMENT_ERROR, "negative argument count");
    }

    c = mrb_class(mrb, self);
    m = mrb_method_search_vm(mrb, &c, mid);

    if (MRB_METHOD_UNDEF_P(m)) {
      mrb_sym missing = MRB_SYM(method_missing);
      m = mrb_method_search_vm(mrb, &c, missing);
      if (MRB_METHOD_UNDEF_P(m)) {
        mrb_method_missing(mrb, mid, self, mrb_ary_new_from_values(mrb, argcount, argv));
      }
      mrb_ary_unshift(mrb, mrb_ary_new_from_values(mrb, argcount, argv), mrb_symbol_value(mid));
      argcount = 15; /* packed arguments */
    }

    if (argcount > 14) {
      mrb_stack_extend(mrb, n + 2);
      ci->stack[n + 1] = mrb_ary_new_from_values(mrb, argcount, argv);
      argcount = 15; /* packed arguments */
    } else {
      mrb_stack_extend(mrb, n + argcount + 2);
      if (argcount > 0) {
        stack_copy(ci->stack + 1, argv, argcount);
      }
    }
