/* check for negative arguments */
if (argcount < 0) {
  mrb_raisef(mrb, E_ARGUMENT_ERROR, "negative argument count %i", argcount);
  goto L_RAISE;
}

/* find class and method */
c = mrb_class(mrb, self);
m = mrb_method_search_vm(mrb, &c, mid);

/* prepare stack */
if (argcount == 15) {
  /* we're passing a packed array of arguments */
  if (mrb_array_p(argv[0])) {
    ci->stack[n+1] = argv[0];
  }
  else {
    /* pack arguments */
    ci->stack[n+1] = mrb_ary_new_from_values(mrb, argcount, argv);
  }
}
else {
  /* make sure the stack is large enough */
  mrb_stack_extend(mrb, n+2);
  /* create an array from the arguments */
  ci->stack[n+1] = mrb_ary_new_from_values(mrb, argcount, argv);
}
