          mrb_raisef(mrb, E_ARGUMENT_ERROR, "negative argc for %n", mid);
        }
        m = mrb_method_search_vm(mrb, &c, mid);
        if (argcount < 15 && mrb->c->ci->nk == 0 && MRB_METHOD_UNDEF_P(m)) {
          do {
            mrb_sym mname = mid;
            if (mname != MRB_SYM(method_missing)) {
              m = mrb_method_search_vm(mrb, &c, MRB_SYM(method_missing));
              if (!MRB_METHOD_UNDEF_P(m)) break;
            }
            mrb_method_missing(mrb, mid, self, mrb_ary_new_capa(mrb, 0));
          } while (FALSE);
        }
        if (argcount == 15) {
          mrb_stack_extend(mrb, n+2);
        }
        else {
          mrb_stack_extend(mrb, n+argcount+2);
