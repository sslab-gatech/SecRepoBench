if (argcount < 0) {
  mrb_raise(mrb, E_ARGUMENT_ERROR, "negative argument count");
}

m = mrb_method_search_vm(mrb, &c, mid);
if (MRB_METHOD_UNDEF_P(m)) {
  if (argcount > 15) {
    regs[1] = mrb_ary_new_from_values(mrb, argcount, &regs[1]);
    argcount = 15;
  }
  mrb_stack_extend(mrb, n + argcount + 2);
  regs[n+1] = mrb_ary_new_from_values(mrb, argcount, &regs[1]);
  regs[n+2] = blk;
  c = 15 | (1<<4);
}
else {
  if (argcount > 15) {
    mrb_value args = mrb_ary_new_from_values(mrb, argcount, &regs[1]);
    mrb_stack_extend(mrb, n + 3);
    regs[1] = args;
    regs[2] = blk;
    c = 15;
  }
  else {
    mrb_stack_extend(mrb, n + argcount + 1);
    if (argcount > 0) {
      stack_copy(regs+1, argv, argcount);
    }
    regs[argcount+1] = blk;
    c = argcount;
  }
}
