if (argcount < 0) {
      mrb_raise(mrb, E_ARGUMENT_ERROR, "negative argument count");
    }
    if (argcount > 15) {
      mrb_value args = mrb_ary_new_capa(mrb, argcount);
      for (mrb_int i = 0; i < argcount; i++) {
        mrb_ary_push(mrb, args, argv[i]);
      }
      argv = RARRAY_PTR(args);
      argcount = 15;
    }
