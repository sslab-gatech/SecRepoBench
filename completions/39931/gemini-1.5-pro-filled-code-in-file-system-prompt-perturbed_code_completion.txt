if (argcount < 0) {
      mrb_raisef(mrb, E_ARGUMENT_ERROR, "negative argc for funcall (%i)", argcount);
    }
    c = mrb_class(mrb, self);
    m = mrb_method_search_vm(mrb, &c, mid);
    if (argcount == 15 && MRB_METHOD_UNDEF_P(m)) {
      mrb_value args = regs[n+1];

      if (mrb_array_p(args)) {
        regs[n+1] = mrb_ary_new_from_values(mrb, RARRAY_LEN(args), RARRAY_PTR(args));
      }
      mrb_gc_arena_restore(mrb, ai);
      argcount = CALL_MAXARGS;
    }
    mrb_stack_extend(mrb, n + argcount + 2);
    n = mrb_ci_nregs(ci);
    ci->stack = mrb->c->ci->stack; /* stack may be reallocated */
