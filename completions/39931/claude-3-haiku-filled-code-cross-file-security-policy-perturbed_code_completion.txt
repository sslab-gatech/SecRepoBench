if (argcount < 0) {
  mrb_raisef(mrb, E_ARGUMENT_ERROR, "negative argument count (%S)", mrb_fixnum_value(argcount));
}

m = mrb_method_search_vm(mrb, &c, mid);
if (MRB_METHOD_UNDEF_P(m)) {
  if (argcount > CALL_MAXARGS) {
    mrb_value args = mrb_ary_new_capa(mrb, argcount);
    for (int i = 0; i < argcount; i++) {
      mrb_ary_push(mrb, args, argv[i]);
    }
    ci->stack[n+1] = args;
    argcount = CALL_MAXARGS;
  }
  mrb_stack_extend(mrb, n + 2 + argcount);
}
