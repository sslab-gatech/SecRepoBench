if (argcount < 0) {
  mrb_raise(mrb, E_ARGUMENT_ERROR, "negative argument count");
}

c = mrb_class(mrb, self);
m = mrb_method_search_vm(mrb, &c, mid);

/* if too many arguments, convert them to an array for mruby */
if (!MRB_METHOD_UNDEF_P(m) && argcount > 14) {
  mrb_value arr = mrb_ary_new_from_values(mrb, argcount, argv);
  ci->stack[n+1] = arr;
  argcount = 15;
}
