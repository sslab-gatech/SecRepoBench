if (argcount < 0) {
  mrb_exc_raise(mrb, mrb_obj_value(mrb->stack_err));
}

c = mrb_class(mrb, self);
m = mrb_method_search_vm(mrb, &c, mid);

/* Prepare arguments array for method_missing */
if (argcount > 0) {
  mrb_value args = mrb_ary_new_from_values(mrb, argcount, argv);
  ci->stack[n+1] = args;
} else {
  ci->stack[n+1] = mrb_ary_new(mrb);
}

/* Special handling for argcount >= 15 */
if (argcount >= 15) {
  argcount = 15;
}
