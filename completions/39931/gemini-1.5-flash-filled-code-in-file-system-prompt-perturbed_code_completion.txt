m = mrb_method_search_vm(mrb, &c, mid);
    if (MRB_METHOD_UNDEF_P(m)) {
      if (c == mrb->object_class) {
        mrb_no_method_error(mrb, mid, mrb_ary_new_from_values(mrb, argcount, argv), "undefined method '%n' for %v", mid, self);
      }
      else {
        mrb_no_method_error(mrb, mid, mrb_ary_new_from_values(mrb, argcount, argv), "undefined method '%n' for %v: %v", mid, self, mrb_obj_value(c));
      }
    }
    if (argcount < 0) {
      mrb_argnum_error(mrb, 0, argcount, -1);
    }
    if (argcount >= 15) {
      mrb_value args = mrb_ary_new_from_values(mrb, argcount, argv);
      mrb_stack_extend(mrb, n + 2);
      ci->stack[n+1] = args;
      argcount = 15;
    }
    else {
      mrb_stack_extend(mrb, n + argcount + 1);
    }
