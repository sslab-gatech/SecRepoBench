if (argcount < 0) {
  mrb_exc_raise(mrb, mrb_obj_value(mrb->eArgError));
}
m = mrb_method_search_vm(mrb, &c, mid);
if (MRB_METHOD_UNDEF_P(m)) {
  m = prepare_missing(mrb, self, mid, &c, 0, &argcount, blk, 0);
  mid = MRB_SYM(method_missing);
}
if (argcount > CALL_MAXARGS) {
  argv = mrb_ary_new_from_values(mrb, argcount, argv);
  argcount = CALL_MAXARGS;
}
mrb_stack_extend(mrb, n + argcount + 1);
