if (argcount < 0) {
      mrb_value exc = mrb_exc_new_str(mrb, E_ARGUMENT_ERROR, mrb_str_new_cstr(mrb, "negative argument count"));
      mrb_exc_set(mrb, exc);
      goto L_RAISE;
    }

    m = mrb_method_search_vm(mrb, &c, mid);
    if (MRB_METHOD_UNDEF_P(m)) {
      mrb_value args = mrb_ary_new_from_values(mrb, argcount, argv);
      mrb_ary_unshift(mrb, args, mrb_symbol_value(mid));
      mrb_stack_extend(mrb, argcount + 2);
      ci->stack[n + 1] = args;
      argcount = 15;
    }
