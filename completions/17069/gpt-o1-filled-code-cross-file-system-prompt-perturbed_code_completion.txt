{
	/* flush local state out to the bitreader before calling the
	 * regular rice-reading function */
	br->consumed_bits = FLAC__BITS_PER_WORD - ucbits;
	br->consumed_words = cwords;

	do {
		/* read a single Rice-coded signed value using the standard function */
		if(!FLAC__bitreader_read_rice_signed(br, val, parameter))
			return false;

		++val;

		/* update our local state to match the bitreader again */
		cwords = br->consumed_words;
		ucbits = FLAC__BITS_PER_WORD - br->consumed_bits;
	} while(cwords >= wordcount && val < end);
}
