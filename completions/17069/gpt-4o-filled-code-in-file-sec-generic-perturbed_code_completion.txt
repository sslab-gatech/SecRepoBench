/* Handle the case where there are unconsumed words left to process. */
		if (cwords < wordcount) {
			b = br->buffer[cwords];
			ucbits = FLAC__BITS_PER_WORD;
		} else {
			if (!bitreader_read_from_client_(br))
				return false;
			cwords = br->consumed_words;
			wordcount = br->words;
			b = br->buffer[cwords];
			ucbits = FLAC__BITS_PER_WORD;
		}
