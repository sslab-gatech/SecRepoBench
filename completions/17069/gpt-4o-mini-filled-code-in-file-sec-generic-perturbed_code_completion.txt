/* read the unary MSBs and end bit */
		if(!FLAC__bitreader_read_unary_unsigned(br, &msbs))
			return false;

		/* read the binary LSBs */
		if(!FLAC__bitreader_read_raw_uint32(br, &lsbs, parameter))
			return false;

		/* compose the value */
		x = (msbs << parameter) | lsbs;
		*val++ = (int)(x >> 1) ^ -(int)(x & 1);
