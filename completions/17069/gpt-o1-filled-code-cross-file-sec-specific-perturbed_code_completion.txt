{
	int tmp;
	/* Restore bitreader's state before attempting to read. */
	br->consumed_words = cwords;
	br->consumed_bits = FLAC__BITS_PER_WORD - ucbits;

	/* Read one rice-signed value. */
	if(!FLAC__bitreader_read_rice_signed(br, &tmp, parameter))
		return false;

	*val++ = tmp;

	/* Update local state from bitreader again. */
	cwords = br->consumed_words;
	if(cwords < wordcount) {
		ucbits = FLAC__BITS_PER_WORD - br->consumed_bits;
		b = br->buffer[cwords] << br->consumed_bits;
	}
	else {
		ucbits = 0;
	}
}
