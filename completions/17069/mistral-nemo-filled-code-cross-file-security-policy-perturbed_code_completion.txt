do {
	// Read the unary most significant bits (MSBs)
	if(!FLAC__bitreader_read_unary_unsigned(br, &msbs))
		return false;

	// Check if we've reached the end of the buffer
	if(cwords >= wordcount) {
		// Handle the case where there are no more words to process
		if(val >= end) {
			break;
		} else {
			// Return false to indicate an incomplete read
			return false;
		}
	}

	// Read the binary least significant bits (LSBs)
	if(!FLAC__bitreader_read_raw_int32(br, &lsbs, parameter))
		return false;

	// Compose the signed value by combining MSBs and LSBs
	x = (msbs << parameter) | lsbs;
	*val++ = (int)(x >> 1) ^ -(int)(x & 1);

	// Update the consumed words and bits count
	cwords++;
	ucbits = FLAC__BITS_PER_WORD;
} while(cwords < wordcount && val < end);
