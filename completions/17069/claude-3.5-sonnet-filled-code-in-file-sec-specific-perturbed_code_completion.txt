if(!FLAC__bitreader_read_unary_unsigned(br, &msbs))
    return false;

if(!FLAC__bitreader_read_raw_uint32(br, &lsbs, parameter))
    return false;

x = (msbs << parameter) | lsbs;
*val++ = (int)(x >> 1) ^ -(int)(x & 1);

br->consumed_bits = FLAC__BITS_PER_WORD - ucbits;
br->consumed_words = cwords;

if(val < end)
    goto process_tail;
