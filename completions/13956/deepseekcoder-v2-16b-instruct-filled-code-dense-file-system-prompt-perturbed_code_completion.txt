// Calculate the position of the string entry in the table using the given index.
  // Return NULL if the string entry is invalid.
  uintptr_t entry_offset = (uintptr_t)str_table_base + (entry_index * sizeof(const char*));
  if (entry_offset < (uintptr_t)str_table_base || entry_offset >= (uintptr_t)str_table_limit) {
    return NULL;
  }
  return (const char*)(entry_offset);
}
