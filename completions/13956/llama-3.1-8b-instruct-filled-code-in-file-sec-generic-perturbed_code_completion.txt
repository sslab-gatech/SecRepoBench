static const char* str_table_entry(
    const char* str_table_base,
    const char* str_table_limit,
    int entry_index)
{
  if (entry_index < 0 || entry_index >= (str_table_limit - str_table_base) / sizeof(uint32_t))
    return NULL;

  uint32_t* entry = (uint32_t*) (str_table_base + entry_index * sizeof(uint32_t));
  if (entry >= str_table_limit)
    return NULL;

  return (const char*) entry;
}
