const char* entry_ptr = str_table_base;
  int current_index = 0;

  while (entry_ptr < str_table_limit && current_index < entry_index)
  {
    // Move to the next string entry by skipping the current string
    entry_ptr += strlen(entry_ptr) + 1;
    current_index++;
  }

  // Check if the entry_ptr is within the valid range
  if (entry_ptr >= str_table_limit || current_index != entry_index)

    return NULL;
