if (str_table_base == NULL || str_table_limit == NULL || entry_index < 0)
    return NULL;

const char* p = str_table_base;
int current_index = 0;

while (p < str_table_limit)
{
    if (current_index == entry_index)
        return p;

    // Advance until the end of the current string or the limit.
    while (p < str_table_limit && *p != '\0')
        p++;

    if (p >= str_table_limit)
        return NULL;

    // Move past the null terminator safely.
    p++;
    current_index++;
}

return NULL;
