if (entry_index < 0)
    return NULL;

  const char* ptr = str_table_base;

  for (int i = 0; i < entry_index; i++)
  {
    while (ptr < str_table_limit && *ptr != '\0')
      ptr++;

    if (ptr >= str_table_limit)
      return NULL;

    ptr++;  // Skip the null terminator

    if (ptr > str_table_limit)
      return NULL;
  }

  if (ptr >= str_table_limit)
    return NULL;

  return ptr;
