static const char* str_table_entry(
    const char* str_table_base,
    const char* str_table_limit,
    int entry_index)
{
    if (entry_index < 0 || entry_index >= (str_table_limit - str_table_base) / sizeof(int))
        return NULL;

    return str_table_base + *(((int*)str_table_base) + entry_index);
}
