if (str_table_base >= str_table_limit || entry_index < 0) return NULL;

const char* entry = str_table_base;

for (int i = 0; i < entry_index; i++) {
  // Skip to the end of the current string
  while (*entry != '\0') {
    entry++;
    if (entry >= str_table_limit) return NULL;
  }
  
  // Move to the beginning of the next string
  entry++;
  if (entry >= str_table_limit) return NULL;
}

return (entry < str_table_limit) ? entry : NULL;
