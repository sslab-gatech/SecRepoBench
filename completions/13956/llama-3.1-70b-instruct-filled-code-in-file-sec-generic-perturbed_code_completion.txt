const char* str_table_entry(
    const char* str_table_base,
    const char* str_table_limit,
    int entry_index)
{
  // Calculate the position of the string entry in the table using the given index.
  const char* entry = str_table_base + entry_index;

  // Check if the entry is within the string table bounds.
  if (entry < str_table_base || entry >= str_table_limit)
    return NULL;

  // Find the null-terminator for the string entry.
  const char* null_terminator = memchr(entry, '\0', str_table_limit - entry);

  // If no null-terminator is found, the entry is invalid.
  if (null_terminator == NULL)
    return NULL;

  // Return the string entry.
  return entry;
}
