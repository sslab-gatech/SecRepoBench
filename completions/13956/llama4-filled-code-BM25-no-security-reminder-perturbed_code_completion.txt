const char* str_table_entry_ptr = str_table_base;
  if (entry_index < 0)
    return NULL;

  // Assuming that each entry in the string table is a null-terminated string
  for (int i = 0; i < entry_index; i++)
  {
    str_table_entry_ptr = strchr(str_table_entry_ptr, '\0') + 1;
    if (str_table_entry_ptr >= str_table_limit)
      return NULL;
  }

  if (str_table_entry_ptr >= str_table_limit)
    return NULL;

  return str_table_entry_ptr;
