const char* entry = NULL;

  if (entry_index >= 0)
  {
    size_t entry_offset = entry_index * sizeof(char*);
    if (entry_offset <= SIZE_MAX - str_table_base && str_table_base + entry_offset <= str_table_limit - sizeof(char*))
    {
      entry = *(const char**)(str_table_base + entry_offset);
      if (entry > str_table_limit)
        entry = NULL;
    }
  }

  return entry;
