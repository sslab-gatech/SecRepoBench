const char* entry_ptr = str_table_base;
  int current_index = 0;

  while (entry_ptr < str_table_limit && current_index < entry_index)
  {
    // Move to the next string entry by finding the null terminator
    while (entry_ptr < str_table_limit && *entry_ptr != '\0')
    {
      entry_ptr++;
    }

    // Move past the null terminator
    if (entry_ptr < str_table_limit)
    {
      entry_ptr++;
      current_index++;
    }
  }

  // Check if we have found the correct entry and it is within bounds
  if (current_index == entry_index && entry_ptr < str_table_limit)
  {
    return entry_ptr;
  }
