const char* current = str_table_base;

// If the table is empty, return NULL.
if (current >= str_table_limit)
    return NULL;

for (int i = 0; i < entry_index; i++)
{
    // Advance to the end of the current string.
    while (current < str_table_limit && *current != '\0')
        current++;

    // If no null terminator found, the entry is invalid.
    if (current >= str_table_limit)
        return NULL;

    // Move past the null terminator to the beginning of the next string.
    current++;
}

// Ensure that the resulting position is within bounds.
if (current >= str_table_limit)
    return NULL;

return current;
