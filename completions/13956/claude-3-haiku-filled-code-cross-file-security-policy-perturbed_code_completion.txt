const char* str_table_entry(
    const char* str_table_base,
    const char* str_table_limit,
    int entry_index)
{
    // Calculate the position of the string entry in the table using the given index.
    // Return NULL if the string entry is invalid.
    if (str_table_base == NULL || str_table_limit == NULL || entry_index < 0)
        return NULL;

    const char* entry = str_table_base + (entry_index * YR_STRING_POOL_CHUNK_SIZE);
    if (entry >= str_table_limit)
        return NULL;

    return entry;
}
