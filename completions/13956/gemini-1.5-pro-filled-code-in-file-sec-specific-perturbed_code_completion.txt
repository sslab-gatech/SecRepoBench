const char* str_entry = str_table_base + entry_index;

  if (str_entry < str_table_base || str_entry >= str_table_limit)
  {
    return NULL;
  }

  // Check if the string entry is null-terminated within the bounds of the string table.
  const char* end = str_entry;
  while (end < str_table_limit && *end != '\0')
  {
    end++;
  }
  if (end >= str_table_limit)
  {
    return NULL;
  }

  return str_entry;
