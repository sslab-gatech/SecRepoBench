if (entry_index < 0)
    return NULL;

const char* p = str_table_base;

for (int i = 0; i < entry_index; i++)
{
    while (p < str_table_limit && *p != '\0')
        p++;

    if (p < str_table_limit)
        p++;
    else
        return NULL;
}

if (p >= str_table_limit)
    return NULL;

return p;
