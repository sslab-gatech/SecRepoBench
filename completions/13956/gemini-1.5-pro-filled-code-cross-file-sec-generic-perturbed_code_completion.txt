const char* entry = NULL;

  if (entry_index >= 0)
  {
    size_t entry_offset = entry_index * sizeof(char*);

    if (entry_offset < str_table_limit - str_table_base)
    {
      entry = str_table_base + entry_offset;

      if (entry < str_table_base || entry >= str_table_limit)
      {
        entry = NULL;
      }
      else if (*entry < str_table_base || *entry >= str_table_limit)
      {
        entry = NULL;
      }
    }
  }

  return entry;
