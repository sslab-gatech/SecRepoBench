if (entry_index < 0 || str_table_base == NULL || str_table_limit == NULL || str_table_base >= str_table_limit)
  {
    return NULL;
  }

  // Assuming strings are null-terminated and the table is an array of strings.
  const char* entry_ptr = str_table_base;
  for (int i = 0; i < entry_index; ++i)
  {
    // Find the end of the current string.
    while (entry_ptr < str_table_limit && *entry_ptr != '\0')
    {
      ++entry_ptr;
    }

    // Check for errors.
    if (entry_ptr >= str_table_limit)
    {
      return NULL;
    }

    // Move to the next string.
    ++entry_ptr;
  }

  // Check if the entry_index is valid.
  if (entry_ptr >= str_table_limit)
  {
    return NULL;
  }

  return entry_ptr;
