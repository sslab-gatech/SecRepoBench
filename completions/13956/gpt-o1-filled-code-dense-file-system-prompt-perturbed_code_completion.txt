{
    // Return NULL for invalid arguments or negative indexes.
    if (str_table_base == NULL ||
        str_table_limit == NULL ||
        entry_index < 0)
    {
        return NULL;
    }

    const char* current = str_table_base;

    // Skip strings until we reach the one at entry_index.
    for (int i = 0; i < entry_index; i++)
    {
        // Move past the current string.
        while (current < str_table_limit && *current != '\0')
            current++;

        // Move past the null terminator.
        if (current < str_table_limit)
            current++;
        else
            return NULL;
    }

    // Ensure we're still within the table.
    if (current >= str_table_limit)
        return NULL;

    return current;
}
