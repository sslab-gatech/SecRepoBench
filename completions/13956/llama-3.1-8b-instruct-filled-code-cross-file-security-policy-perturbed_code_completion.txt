static const char* str_table_entry(
    const char* str_table_base,
    const char* str_table_limit,
    int entry_index)
{
  // Check if the entry index is within the valid range.
  if (entry_index < 0 || entry_index >= (str_table_limit - str_table_base) / sizeof(const char*))
  {
    return NULL;
  }

  // Calculate the position of the string entry in the table using the given index.
  const char* str_entry = str_table_base + entry_index * sizeof(const char*);

  // Return the string entry if it's within the valid range, otherwise return NULL.
  if (str_entry >= str_table_base && str_entry < str_table_limit)
  {
    return str_entry;
  }
  else
  {
    return NULL;
  }
}
