const char* p = str_table_base;
int i = 0;

// Iterate through the string table to find the entry at entry_index
while (i < entry_index && p < str_table_limit)
{
  // Move to the next string (skip current string and its null terminator)
  while (p < str_table_limit && *p != '\0')
    p++;
  if (p < str_table_limit)
    p++; // skip the null terminator
  i++;
}

// Check if we found the entry and it's within bounds
if (p >= str_table_limit || *p == '\0')
  return NULL;

return p;
