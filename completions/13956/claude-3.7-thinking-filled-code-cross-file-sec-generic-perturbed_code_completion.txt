if (entry_index < 0 || str_table_base == NULL || str_table_limit == NULL || str_table_base >= str_table_limit)
  return NULL;

const char* entry = str_table_base;

// Navigate to the entry_index-th string
for (int i = 0; i < entry_index; i++)
{
  // Find the end of the current string (null terminator)
  while (entry < str_table_limit && *entry != '\0')
    entry++;
    
  // Move past the null terminator
  entry++;
  
  // Check if we've gone past the end of the table
  if (entry >= str_table_limit)
    return NULL;
}

return entry;
