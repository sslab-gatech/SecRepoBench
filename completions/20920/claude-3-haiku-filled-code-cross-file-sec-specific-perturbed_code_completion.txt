// Check if the TIFF file is byte-swapped and, if so, swap the byte order of the IPTC profile data.
if (image->endian == LSBEndian)
{
    SwapBytes(profile, sizeof(uint16), length / sizeof(uint16));
}

// Read the IPTC profile into the image.
status = SetImageProfile(image, "iptc", profile, (ssize_t) length, exception);
if (status == MagickFalse)
{
    TIFFClose(tiff);
    return(DestroyImageList(image));
}
