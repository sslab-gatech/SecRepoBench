StringInfo
        *iptc_profile;

      iptc_profile=CloneStringInfo((StringInfo *) profile);
      length=GetStringInfoLength(profile)+4-(GetStringInfoLength(profile) & 0x03);
      SetStringInfoLength(iptc_profile,length);
      if (TIFFIsByteSwapped(tiff))
        TIFFSwabArrayOfLong((uint32 *) GetStringInfoDatum(iptc_profile),
          (unsigned long) (length/4));
      status=ReadProfile(img,"iptc",GetStringInfoDatum(iptc_profile),
        (ssize_t) length,exception);
      iptc_profile=DestroyStringInfo(iptc_profile);
