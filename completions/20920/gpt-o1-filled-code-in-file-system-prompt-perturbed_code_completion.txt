  /*
    Check if the TIFF file is byte-swapped and, if so, swap the byte order of the IPTC profile data.
    Read the IPTC profile into the image.
  */
  StringInfo
    *iptc_profile_info;

  iptc_profile_info=BlobToStringInfo(profile,(size_t) length);
  if (iptc_profile_info != (StringInfo *) NULL)
    {
      if (TIFFIsByteSwapped(tiff))
        TIFFSwabArrayOfLong((uint32 *) GetStringInfoDatum(iptc_profile_info),
          (uint32) (length/4));
      status=ReadProfile(img,"iptc",GetStringInfoDatum(iptc_profile_info),
        (ssize_t) length,exception);
      iptc_profile_info=DestroyStringInfo(iptc_profile_info);
    }
