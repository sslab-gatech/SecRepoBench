if (regex == NULL)
		return REG_INVARG;

	int rc;
#ifdef USE_C_LOCALE
	locale_t old_locale = uselocale(regex->c_lc_ctype);
#else
	char *old_locale = setlocale(LC_CTYPE, "C");
#endif

	rc = regexec(&regex->rx, str, nmatch, pmatch, eflags);

#ifdef USE_C_LOCALE
	uselocale(old_locale);
#else
	setlocale(LC_CTYPE, old_locale);
#endif

	return rc;
