// Ensure the offset is at the start of the image data
  if (offset < (MagickOffsetType) dpx.file.image_offset)
  {
    if (SeekBlob(image,dpx.file.image_offset,SEEK_SET) == -1)
      ThrowReaderException(CorruptImageError,"UnableToReadImageData");
    offset=dpx.file.image_offset;
  }

  // Check for unexpected end of file
  if (EOFBlob(image) != MagickFalse)
  {
    ThrowFileException(exception,CorruptImageError,"UnexpectedEndOfFile",
      image->filename);
    image=DestroyImageList(image);
    return((Image *) NULL);
  }

  // If 'ping' is enabled, close the blob and return the first image
  if (image_info->ping != MagickFalse)
  {
    (void) CloseBlob(image);
    return(GetFirstImageInList(image));
  }

  // Set the image extent
  SetImageExtent(image,image->columns,image->rows,exception);
  if (operationsuccess == MagickFalse)
    return(DestroyImageList(image));
