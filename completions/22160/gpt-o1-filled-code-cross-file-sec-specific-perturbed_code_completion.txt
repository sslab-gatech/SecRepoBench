/* Skip any remaining bytes up to the start of the image data. */
  if (dpx.file.image_offset > offset)
    {
      MagickOffsetType skip_length = (MagickOffsetType) dpx.file.image_offset - offset;
      while (skip_length-- > 0)
        if (ReadBlobByte(image) == EOF)
          ThrowReaderException(CorruptImageError,"UnableToReadImageData");
      offset = dpx.file.image_offset;
    }
  else if (dpx.file.image_offset < offset)
    {
      ThrowReaderException(CorruptImageError,"UnableToReadImageData");
    }

  /* If only pinging, no need to fully decode the raster data. */
  if (image_info->ping != MagickFalse)
    {
      (void) CloseBlob(image);
      return(GetFirstImageInList(image));
    }

  /* Set the image extent to the read dimensions. */
  if (SetImageExtent(image,image->columns,image->rows,exception) == MagickFalse)
    {
      return(DestroyImageList(image));
    }
