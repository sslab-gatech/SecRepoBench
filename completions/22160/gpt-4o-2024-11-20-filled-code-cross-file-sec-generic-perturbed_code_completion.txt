/*
    Skip to the image data offset if necessary.
  */
  if (offset < (MagickOffsetType) dpx.file.image_offset)
  {
    if (SeekBlob(image, dpx.file.image_offset, SEEK_SET) == -1)
      ThrowReaderException(CorruptImageError, "UnableToSeekToImageData");
    offset = dpx.file.image_offset;
  }

  /*
    Handle 'ping' option: if enabled, close the blob and return the first image.
  */
  if (image_info->ping != MagickFalse)
  {
    (void) CloseBlob(image);
    return(GetFirstImageInList(image));
  }

  /*
    Set the image extent to the defined number of columns and rows.
  */
  if ((image->columns == 0) || (image->rows == 0))
    ThrowReaderException(CorruptImageError, "ImproperImageHeader");
  if (SetImageExtent(image, image->columns, image->rows, exception) == MagickFalse)
  {
    (void) CloseBlob(image);
    return(DestroyImageList(image));
  }
