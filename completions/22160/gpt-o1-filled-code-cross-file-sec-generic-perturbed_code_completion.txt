if ((MagickOffsetType) dpx.file.image_offset > offset)
{
  MagickOffsetType skip_count=(MagickOffsetType) dpx.file.image_offset-offset;
  while (skip_count-- > 0)
    if (ReadBlobByte(image) == EOF)
      ThrowReaderException(CorruptImageError,"UnexpectedEndOfFile");
  offset=dpx.file.image_offset;
}

if (EOFBlob(image) != MagickFalse)
  ThrowReaderException(CorruptImageError,"UnexpectedEndOfFile");

/*
  If ping is specified, we only collect the image header information
  and then stop.
*/
if (image_info->ping != MagickFalse)
{
  (void) CloseBlob(image);
  return(GetFirstImageInList(image));
}

/*
  Set image extent to valid geometry.
*/
if (SetImageExtent(image,image->columns,image->rows,exception) == MagickFalse)
  return(DestroyImageList(image));
