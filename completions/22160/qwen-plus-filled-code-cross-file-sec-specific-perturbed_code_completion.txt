if (dpx.file.image_offset > offset)
{
  size_t bytes_to_read = dpx.file.image_offset - offset;
  unsigned char *buffer = (unsigned char *) AcquireQuantumMemory(bytes_to_read, sizeof(unsigned char));
  if (buffer == NULL)
    ThrowReaderException(ResourceLimitError, "MemoryAllocationFailed");
  
  count = ReadBlob(image, bytes_to_read, buffer);
  if (count != (ssize_t)bytes_to_read || EOFBlob(image) != MagickFalse)
  {
    buffer =DestroyString(buffer);
    ThrowReaderException(CorruptImageError, "UnexpectedEndOfFile");
  }
  
  buffer = DestroyString(buffer);
}
