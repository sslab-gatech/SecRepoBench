  {
    MagickOffsetType skip = (MagickOffsetType) dpx.file.image_offset - offset;

  }

  {

      ThrowReaderException(CorruptImageError,"UnexpectedEndOfFile");
    offset = (MagickOffsetType) dpx.file.image_offset;
