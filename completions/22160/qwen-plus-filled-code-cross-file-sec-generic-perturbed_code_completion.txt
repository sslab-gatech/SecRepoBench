if (dpx.file.image_offset > offset)
{
  size_t bytes_to_skip = dpx.file.image_offset - offset;
  while (bytes_to_skip > 0)
  {
    char buffer[4096];
    size_t read_size = (bytes_to_skip > sizeof(buffer)) ? sizeof(buffer) : bytes_to_skip;
    count = ReadBlob(image, read_size, (unsigned char *) buffer);
    if (count != (ssize_t) read_size)
      ThrowReaderException(CorruptImageError, "UnexpectedEndOfFile");
    bytes_to_skip -= count;
    offset += count;
  }
}
else if (dpx.file.image_offset < offset)
{
  ThrowReaderException(CorruptImageError, "InvalidImageOffset");
}

if (image_info->ping != MagickFalse)
{
  (void) CloseBlob(image);
  return(GetFirstImageInList(image));
}
