/*
    Read and skip bytes from the current offset to the specified image offset.
    If an unexpected end of file is encountered, throw an exception.
  */
  for ( ; offset < (MagickOffsetType) dpx.file.image_offset; offset++)
    if (ReadBlobByte(image) == EOF)
      ThrowReaderException(CorruptImageError,"UnexpectedEndOfFile");

  /*
    If the 'ping' option is enabled, close the blob and return.
  */
  if (image_info->ping != MagickFalse)
    {
      (void) CloseBlob(image);
      return(GetFirstImageInList(image));
    }

  /*
    Set image extent to the defined columns and rows.
  */
  if (SetImageExtent(image,image->columns,image->rows,exception) == MagickFalse)
    return(DestroyImageList(image));
