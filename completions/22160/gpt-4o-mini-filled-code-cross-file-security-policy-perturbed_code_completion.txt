if (offset < dpx.file.image_offset)
{
    ssize_t bytes_to_skip = dpx.file.image_offset - offset;
    while (bytes_to_skip > 0)
    {
        ssize_t count = ReadBlob(image, (size_t)bytes_to_skip, (unsigned char *)NULL);
        if (count <= 0)
        {
            ThrowFileException(exception, CorruptImageError, "UnexpectedEndOfFile", image->filename);
            CloseBlob(image);
            return DestroyImageList(image);
        }
        bytes_to_skip -= count;
    }
}
