for (k = 0; k < 10; k++) {
	if (packet->line[i].ptr[j + k] == ' ') {
	  space = k;
	  break;
	}
      }
      if (space == 0)
	continue;
      port = (u_int16_t)ndpi_atoi(&packet->line[i].ptr[j + space + 1]);
      if (port > 1024 && port < 65535) {
	if (src_id != NULL) {
	  ndpi_int_irc_add_port(ndpi_struct, src_id, port);
	} else if (dst != NULL) {
	  ndpi_int_irc_add_port(ndpi_struct, dst, port);
	}
      }
    }
