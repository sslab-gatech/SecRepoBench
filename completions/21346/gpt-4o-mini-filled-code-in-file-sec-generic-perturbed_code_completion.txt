u_int16_t port = 0;
	      while (packet->line[i].ptr[j] != ' ' && j < packet->line[i].len) {
	        port = port * 10 + (packet->line[i].ptr[j] - '0');
	        j++;
	      }
	      if (port > 0 && port <= 65535) {
	        if (!ndpi_is_duplicate(src_id, port)) {
	          if (src_id->irc_number_of_port < NDPI_PROTOCOL_IRC_MAXPORT) {
	            src_id->irc_port[src_id->irc_number_of_port] = port;
	            src_id->last_time_port_used[src_id->irc_number_of_port] = packet->tick_timestamp;
	            src_id->irc_number_of_port++;
	            NDPI_LOG_INFO(ndpi_struct, "Added new DCC port: %u", port);
	          } else {
	            NDPI_LOG_WARN(ndpi_struct, "IRC port array is full, cannot add port: %u", port);
	          }
	        } else {
	          NDPI_LOG_DBG2(ndpi_struct, "Duplicate DCC port: %u", port);
	        }
	      } else {
	        NDPI_LOG_WARN(ndpi_struct, "Invalid port number: %u", port);
	      }
