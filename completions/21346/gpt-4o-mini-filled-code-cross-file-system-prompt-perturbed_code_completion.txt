u_int16_t port_num = 0;
	      while (j < packet->line[i].len - 9) {
	        if (packet->line[i].ptr[j] == ' ') {
	          j++;
	          if (packet->line[i].ptr[j] >= '0' && packet->line[i].ptr[j] <= '9') {
	            port_num = 0;
	            while (j < packet->line[i].len - 9 && packet->line[i].ptr[j] >= '0' && packet->line[i].ptr[j] <= '9') {
	              port_num = port_num * 10 + (packet->line[i].ptr[j] - '0');
	              j++;
	            }
	            if (port_num > 0 && port_num <= 65535) {
	              if (src_id != NULL) {
	                if (src_id->irc_number_of_port < MAX_IRC_PORTS) {
	                  src_id->irc_port[src_id->irc_number_of_port++] = port_num;
	                  src_id->last_time_port_used[src_id->irc_number_of_port - 1] = packet->tick_timestamp;
	                } else {
	                  // Replace the least recently used port
	                  src_id->irc_port[0] = port_num;
	                  src_id->last_time_port_used[0] = packet->tick_timestamp;
	                }
	              }
	              if (dst != NULL) {
	                if (dst->irc_number_of_port < MAX_IRC_PORTS) {
	                  dst->irc_port[dst->irc_number_of_port++] = port_num;
	                  dst->last_time_port_used[dst->irc_number_of_port - 1] = packet->tick_timestamp;
	                } else {
	                  // Replace the least recently used port
	                  dst->irc_port[0] = port_num;
	                  dst->last_time_port_used[0] = packet->tick_timestamp;
	                }
	              }
	              NDPI_LOG_INFO(ndpi_struct, "found DCC command with port: %u", port_num);
	            }
	          }
	        }
	        j++;
	      }
