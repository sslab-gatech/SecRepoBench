/* nothing can be fetched if stream is NULL (e.g. str_getcsv()) */
	if (csv_stream == NULL) {
		if (hunk_begin != line_end) {
			/* add the embedded line end to the field */
			memcpy(tptr, hunk_begin, bptr - hunk_begin);
			tptr += (bptr - hunk_begin);
			hunk_begin = bptr;
		}

		/* add the embedded line end to the field */
		memcpy(tptr, line_end, line_end_len);
		tptr += line_end_len;

		temp_len += line_end_len;
		char *new_temp = erealloc(temp, temp_len);
		tptr = new_temp + (size_t)(tptr - temp);
		temp = new_temp;

		/* nothing can be fetched if stream is NULL (e.g. str_getcsv()) */
		/* so we just return the current field and stop processing */
		break;
	}

	/* read the next line from the stream */
	if (php_stream_get_line(csv_stream, buf, buf_len, &new_len) == NULL) {
		/* reading failed, so handle the unterminated enclosure similarly and exit the loop */
		if (hunk_begin != line_end) {
			/* add the embedded line end to the field */
			memcpy(tptr, hunk_begin, bptr - hunk_begin);
			tptr += (bptr - hunk_begin);
			hunk_begin = bptr;
		}

		/* add the embedded line end to the field */
		memcpy(tptr, line_end, line_end_len);
		tptr += line_end_len;

		temp_len += line_end_len;
		char *new_temp = erealloc(temp, temp_len);
		tptr = new_temp + (size_t)(tptr - temp);
		temp = new_temp;

		/* nothing can be fetched if stream is NULL (e.g. str_getcsv()) */
		/* so we just return the current field and stop processing */
		break;
	}

	/* update the buffer and related state variables for the new line */
	bptr = buf = new_buf;
	hunk_begin = buf;

	line_end = limit = (char *)php_fgetcsv_lookup_trailing_spaces(buf, buf_len);
	line_end_len = buf_len - (size_t)(limit - buf);
