if (csv_stream == NULL) {
	/* In case of string stream, we have to simulate EOF on unterminated enclosure */
	if (bptr == limit) {
		bptr++; /* For unterminated enclosure, do not include the null character at the end of the buffer */
	}
	break;
}

size_t new_len;
char *new_buf = php_stream_get_line(csv_stream, NULL, 0, &new_len);
if (!new_buf) {
	/* simulate EOF */
	break;
