if (Tables == NULL || Tables->TheCurves == NULL) return FALSE;

    for (index = 0; index < Tables->nCurves; index++) {

        cmsToneCurve* curve = Tables->TheCurves[index];
        if (curve == NULL || curve->Table16 == NULL) return FALSE;

        for (j = 0; j < curve->nEntries; j++) {

            if (!_cmsWriteUInt16Number(io, curve->Table16[j]))
                return FALSE;
        }
    }

    return TRUE;
