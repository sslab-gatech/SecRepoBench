if (!Tables) return TRUE;

    for (index = 0; index < Tables->nCurves; index++) {

        if (Tables->TheCurves[index] == NULL ||
            Tables->TheCurves[index]->Table16 == NULL ||
            Tables->TheCurves[index]->nEntries == 0)
        {
            cmsSignalError(ContextID, cmsERROR_RANGE, "Invalid curve data");
            return FALSE;
        }

        for (j = 0; j < Tables->TheCurves[index]->nEntries; j++) {
            val = Tables->TheCurves[index]->Table16[j];
            if (!_cmsWriteUInt16Number(io, val)) return FALSE;
        }
    }

    return TRUE;
