for (index = 0; index < Tables->nCurves; index++) {
        if (Tables->TheCurves[index]->nEntries != 65536) {
            cmsSignalError(ContextID, cmsERROR_RANGE, "LUT16 needs 65536 entries");
            return FALSE;
        }

        for (j = 0; j < 65536; j++) {
            val = Tables->TheCurves[index]->Table16[j];

            if (!_cmsWriteUInt16Number(io, val)) {
                return FALSE;
            }
        }
    }
    return TRUE;
