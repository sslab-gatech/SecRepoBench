{
    for (cmsUInt32Number i = 0; i < Tables->nCurves; i++) {
        cmsToneCurve* curve = Tables->TheCurves[i];
        if (curve == NULL) return FALSE;

        for (j = 0; j < curve->nEntries; j++) {
            index = j;
            val = curve->Table16[index];

            if (!_cmsWriteUInt16Number(io, val)) {
                return FALSE;
            }
        }
    }

    return TRUE;
}
