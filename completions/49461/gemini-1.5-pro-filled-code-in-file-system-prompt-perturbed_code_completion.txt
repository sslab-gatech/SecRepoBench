for (cmsUInt32Number i = 0; i < Tables->nCurves; i++) {
        cmsToneCurve* curve = Tables->TheCurves[i];
        cmsUInt32Number nEntries = curve->nEntries;

        for (cmsUInt32Number j = 0; j < nEntries; j++) {
            val = curve->Table16[j];
            if (!_cmsWriteUInt16Number(io, val)) return FALSE;
        }
    }
    return TRUE;
