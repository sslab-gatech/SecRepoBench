for (index = 0; index < Tables->nCurves; index++) {
        if (Tables->TheCurves[index] != NULL) {
            cmsUInt32Number nEntries = Tables->TheCurves[index]->nEntries;

            // Ensure the number of entries is valid
            if (nEntries > 0xFFFF) {
                cmsSignalError(ContextID, cmsERROR_RANGE, "Curve has too many entries");
                return FALSE;
            }

            for (j = 0; j < nEntries; j++) {
                val = Tables->TheCurves[index]->Table16[j];

                // Write the 16-bit value to the io handler
                if (!_cmsWriteUInt16Number(io, val)) {
                    return FALSE; // Return FALSE if writing fails
                }
            }
        }
    }
    return TRUE;
