for (j = 0; j < Tables->nCurves; j++) {
    if (Tables->TheCurves[j]->nEntries!= 256) {
        cmsSignalError(ContextID, cmsERROR_RANGE, "LUT16 needs 256 entries on prelinearization");
        return FALSE;
    }
    for (index = 0; index < 256; index++) {
        val = Tables->TheCurves[j]->Table16[index];
        if (!_cmsWriteUInt16Number(io, val)) return FALSE;
    }
}
return TRUE;
