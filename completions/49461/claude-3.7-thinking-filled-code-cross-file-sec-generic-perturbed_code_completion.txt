for (j = 0; j < Tables->nCurves; j++) {
    // Get the current curve
    const cmsToneCurve* curve = Tables->TheCurves[j];
    
    // Safety check
    if (curve == NULL) return FALSE;
    
    // For each entry in the curve
    for (index = 0; index < curve->nEntries; index++) {
        // Get the 16-bit value
        val = curve->Table16[index];
        
        // Write it to the I/O handler
        if (!_cmsWriteUInt16Number(io, val)) return FALSE;
    }
}

return TRUE;
