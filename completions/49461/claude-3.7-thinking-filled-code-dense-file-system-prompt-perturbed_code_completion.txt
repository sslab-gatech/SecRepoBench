for (j = 0; j < Tables->nCurves; j++) {
    // For each curve, retrieve the number of entries
    cmsUInt32Number nEntries = Tables->TheCurves[j]->nEntries;
    
    // Loop through each entry and obtain the 16-bit value from the Table16 array
    for (index = 0; index < nEntries; index++) {
        val = Tables->TheCurves[j]->Table16[index];
        
        // Write each 16-bit value to the provided io handler
        if (!_cmsWriteUInt16Number(io, val)) 
            return FALSE;  // If any write operation fails, return FALSE
    }
}

return TRUE;
