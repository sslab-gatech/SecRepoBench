/* Iterate over each curve in the Tables structure */
    for (index = 0; index < Tables->nCurves; index++) {
        cmsToneCurve* curve = Tables->TheCurves[index];

        /* For each curve, retrieve the number of entries */
        if (curve == NULL) {
            /* If something is wrong with the curve pointer, fail outright */
            return FALSE;
        }

        /* Loop through each entry and write the 16-bit value */
        for (j = 0; j < curve->nEntries; j++) {
            val = curve->Table16[j];
            if (!_cmsWriteUInt16Number(io, val)) {
                return FALSE;
            }
        }
    }

    /* If everything is written successfully, return TRUE */
    return TRUE;
