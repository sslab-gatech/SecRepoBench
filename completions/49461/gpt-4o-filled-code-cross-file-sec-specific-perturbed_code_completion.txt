for (index = 0; index < Tables->nCurves; index++) {
        if (Tables->TheCurves[index]->nEntries > 0xFFFF) {
            cmsSignalError(ContextID, cmsERROR_RANGE, "Too many entries in curve");
            return FALSE;
        }

        for (j = 0; j < Tables->TheCurves[index]->nEntries; j++) {
            val = Tables->TheCurves[index]->Table16[j];
            if (!_cmsWriteUInt16Number(io, val)) {
                return FALSE;
            }
        }
    }
    return TRUE;
