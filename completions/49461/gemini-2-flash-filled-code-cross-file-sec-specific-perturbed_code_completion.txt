{
    cmsUInt32Number nCurves = Tables->nCurves;

    for (j = 0; j < nCurves; j++) {
        cmsToneCurve* curve = Tables->TheCurves[j];
        cmsUInt32Number nEntries = curve->nEntries;

        for (index = 0; index < nEntries; index++) {
            val = curve->Table16[index];
            if (!_cmsWriteUInt16Number(io, val)) {
                return FALSE;
            }
        }
    }

    return TRUE;
}
