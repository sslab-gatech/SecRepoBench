uint32_t len, reclen;
	int tapemarks = 0;
	int records = 0;

	while (uc < end_pointer) {
		len = getlen(&uc);
		if (len == 0xffffffff)
			break;
		if (len == 0) {
			tapemarks++;
			if (tapemarks >= SIMH_TAPEMARKS)
				break;
			continue;
		}
		reclen = getlen(&uc);
		if (reclen != len)
			return 0;
		uc += len;
		records++;
	}
	DPRINTF("records %d tapemarks %d\n", records, tapemarks);
	return records > 0 || tapemarks > 0;
