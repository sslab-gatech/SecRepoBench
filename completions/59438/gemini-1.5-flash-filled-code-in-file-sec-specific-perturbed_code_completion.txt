uint32_t len, reclen;
	int tapemarks = 0;
	int records = 0;

	simh_bo.u = 0;
	memcpy(&simh_bo, uc, sizeof(simh_bo));
	uc += sizeof(simh_bo);

	while (uc < end_pointer && tapemarks < SIMH_TAPEMARKS) {
		len = getlen(&uc);
		if (len == 0xffffffff) {
			tapemarks++;
			continue;
		}
		if (uc + len > end_pointer)
			return 0;
		reclen = getlen(&uc);
		if (reclen != len)
			return 0;
		uc += len;
		records++;
	}
	return records > 0;
