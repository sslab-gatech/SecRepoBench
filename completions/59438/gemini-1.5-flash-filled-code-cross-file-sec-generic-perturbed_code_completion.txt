unsigned int len;
	unsigned int records = 0;
	unsigned int tapemarks = 0;

	while (uc < end_pointer) {
		len = getlen(uc, end_pointer);
		if (len == 0xFFFFFFFF) {
			break;
		}
		if (len == 0) {
			return 0;
		}
		uc += len;
		records++;
		if (records > SIMH_RECORDS) {
			return 0;
		}
		len = getlen(uc, end_pointer);
		if (len == 0) {
			return 0;
		}
		if (len != 0x00000000) {
			tapemarks++;
			if (tapemarks >= SIMH_TAPEMARKS) {
				return 0;
			}
		}
		uc += len;
	}
	return records > 0;
