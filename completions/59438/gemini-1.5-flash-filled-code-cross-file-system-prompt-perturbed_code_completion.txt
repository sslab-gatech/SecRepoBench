uint32_t len;
	uint32_t reclen;
	unsigned int tapemarks = 0;
	int records = 0;

	while (uc < end_pointer) {
		len = getlen(uc, end_pointer);
		if (len == 0xFFFFFFFF)
			break;
		if (uc + 4 + len > end_pointer)
			return 0;
		reclen = getlen(uc + 4, end_pointer);
		if (len != reclen)
			return 0;
		uc += 4 + len + 4;
		records++;
		if (len == 0) {
			tapemarks++;
			if (tapemarks >= SIMH_TAPEMARKS)
				break;
		}
	}
	return records > 0;
