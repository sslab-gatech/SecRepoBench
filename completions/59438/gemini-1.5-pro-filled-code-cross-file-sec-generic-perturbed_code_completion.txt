uint32_t len, len2;
	int nrec = 0, ntmk = 0;

	while (uc < end_pointer) {
		len = getlen(uc, end_pointer);
		if (len == 0)
			return 0;
		if (len == 0xFFFFFFFFU)
			break;	// EOM
		uc += 4;
		if (uc + len + 4 > end_pointer)
			return 0;
		len2 = getlen(uc + len, end_pointer);
		if (len != len2)
			return 0;
		nrec++;
		uc += len + 4;
		if (uc + 4 > end_pointer)
			return 0;
		len = getlen(uc, end_pointer);
		if (len == 0)
			return 0;
		if (len == 0xFFFFFFFFU)
			break;	// EOM
		ntmk++;
		if (ntmk == SIMH_TAPEMARKS)
			break;
		uc += 4;
	}
	return nrec > 0;
