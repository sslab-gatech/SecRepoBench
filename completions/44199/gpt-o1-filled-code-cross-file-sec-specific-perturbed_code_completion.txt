jsmntok_info_t tinfo;
    memset(&tinfo, 0, sizeof(tinfo));
    tinfo.tokens = tokens;
    tinfo.count = r;

    switch (tokens[startindex].type) {
        case JSMN_OBJECT:
            *plist = parse_object(jsonstr, &tinfo, &startindex);
            break;
        case JSMN_ARRAY:
            *plist = parse_array(jsonstr, &tinfo, &startindex);
            break;
        case JSMN_STRING:
            *plist = parse_string(jsonstr, &tinfo, &startindex);
            break;
        case JSMN_PRIMITIVE:
            *plist = parse_primitive(jsonstr, &tinfo, &startindex);
            break;
        default:
            *plist = NULL;
            PLIST_JSON_ERR("%s: Unexpected top-level token type\n", __func__);
            free(tokens);
            return PLIST_ERR_PARSE;
    }

    if (!(*plist)) {
        free(tokens);
        return PLIST_ERR_PARSE;
    }
