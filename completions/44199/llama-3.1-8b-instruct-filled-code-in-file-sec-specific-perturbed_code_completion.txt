case JSMN_OBJECT:
    *plist = parse_object(jsonstr, &ti, &startindex);
    break;
case JSMN_ARRAY:
    *plist = parse_array(jsonstr, &ti, &startindex);
    break;
case JSMN_STRING:
    *plist = parse_string(jsonstr, &ti, &startindex);
    break;
case JSMN_PRIMITIVE:
    *plist = parse_primitive(jsonstr, &ti, &startindex);
    break;
default:
    PLIST_JSON_ERR("%s: token type!= JSMN_OBJECT, JSMN_ARRAY, JSMN_STRING, JSMN_PRIMITIVE\n", __func__);
    free(tokens);
    return PLIST_ERR_PARSE;
