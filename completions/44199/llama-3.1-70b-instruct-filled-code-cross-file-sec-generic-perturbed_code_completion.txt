// Parse the JSON token at the current start index based on its type.
// Depending on whether the token is a primitive, string, array, or object,
// call the corresponding parse function to convert the JSON data into a plist structure.
// Assign the parsed plist result to the output pointer `plist`.
switch (tokens[startindex].type) {
    case JSMN_OBJECT:
        *plist = parse_object(jsonstr, (jsmntok_info_t){tokens, r}, &startindex);
        break;
    case JSMN_ARRAY:
        *plist = parse_array(jsonstr, (jsmntok_info_t){tokens, r}, &startindex);
        break;
    case JSMN_STRING:
        *plist = parse_string(jsonstr, (jsmntok_info_t){tokens, r}, &startindex);
        break;
    case JSMN_PRIMITIVE:
        *plist = parse_primitive(jsonstr, (jsmntok_info_t){tokens, r}, &startindex);
        break;
    default:
        PLIST_JSON_ERR("%s: unsupported token type\n", __func__);
        return PLIST_ERR_PARSE;
}
