if (r > 0) {
        jsmntok_info_t ti;
        ti.tokens = tokens;
        ti.count = r;
        if (ti.tokens[0].type == JSMN_OBJECT) {
            *plist = parse_object(jsonstr, &ti, &startindex);
        } else if (ti.tokens[0].type == JSMN_ARRAY) {
            *plist = parse_array(jsonstr, &ti, &startindex);
        } else if (ti.tokens[0].type == JSMN_STRING) {
            *plist = parse_string(jsonstr, &ti, &startindex);
        } else if (ti.tokens[0].type == JSMN_PRIMITIVE) {
            *plist = parse_primitive(jsonstr, &ti, &startindex);
        }
    } else {
        PLIST_JSON_ERR("%s: JSON string is empty\n", __func__);
        free(tokens);
        return PLIST_ERR_INVALID_ARG; // Return an appropriate error code
    }
