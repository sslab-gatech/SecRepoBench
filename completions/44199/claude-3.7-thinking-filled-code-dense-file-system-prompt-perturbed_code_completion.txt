if (r < 1) {
    PLIST_JSON_ERR("%s: No tokens found in JSON\n", __func__);
    free(tokens);
    return PLIST_ERR_PARSE;
}

// Create token info structure
jsmntok_info_t token_info = { tokens };

// Call the appropriate parser based on the type of the first token
switch (tokens[startindex].type) {
    case JSMN_PRIMITIVE:
        *plist = parse_primitive(jsonstr, &token_info, &startindex);
        break;
    case JSMN_STRING:
        *plist = parse_string(jsonstr, &token_info, &startindex);
        break;
    case JSMN_ARRAY:
        *plist = parse_array(jsonstr, &token_info, &startindex);
        break;
    case JSMN_OBJECT:
        *plist = parse_object(jsonstr, &token_info, &startindex);
        break;
    default:
        PLIST_JSON_ERR("%s: Unknown token type\n", __func__);
        free(tokens);
        return PLIST_ERR_PARSE;
}

// Check if parsing was successful
if (*plist == NULL) {
    free(tokens);
    return PLIST_ERR_PARSE;
}
