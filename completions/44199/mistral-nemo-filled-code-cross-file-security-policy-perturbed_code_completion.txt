for (int i = 0; i < r; i++) {
        switch (tokens[i].type) {
            case JSMN_OBJECT:
                *plist = parse_object(jsonstr, &tokens[i], &startindex);
                break;
            case JSMN_ARRAY:
                *plist = parse_array(jsonstr, &tokens[i], &startindex);
                break;
            case JSMN_STRING:
                *plist = parse_string(jsonstr, &tokens[i], &startindex);
                break;
            case JSMN_PRIMITIVE:
                *plist = parse_primitive(jsonstr, &tokens[i], &startindex);
                break;
            default:
                break;
        }
        if (*plist) {
            return PLIST_ERR_SUCCESS;
        }
    }
