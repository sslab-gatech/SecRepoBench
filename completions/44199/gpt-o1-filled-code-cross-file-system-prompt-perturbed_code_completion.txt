jsmntok_info_t ti;
    ti.tokens = tokens;
    ti.count = r;

    int idx = startindex;
    switch (tokens[idx].type) {
        case JSMN_OBJECT:
            *plist = parse_object(jsonstr, &ti, &idx);
            break;
        case JSMN_ARRAY:
            *plist = parse_array(jsonstr, &ti, &idx);
            break;
        case JSMN_STRING:
            *plist = parse_string(jsonstr, &ti, &idx);
            break;
        case JSMN_PRIMITIVE:
            *plist = parse_primitive(jsonstr, &ti, &idx);
            break;
        default:
            *plist = NULL;
            break;
    }

    if (!(*plist)) {
        free(tokens);
        return PLIST_ERR_PARSE;
    }
