    // Depending on whether the token is a primitive, string, array, or object,
    // call the corresponding parse function to convert the JSON data into a plist structure.
    // Assign the parsed plist result to the output pointer `plist`.
    int startindex = 0;
    jsmntok_t* current_token = &tokens[startindex];

    switch (current_token->type) {
        case JSMN_PRIMITIVE:
            // Parse primitive type (integer, float, boolean)
            {
                plist_data_t data = plist_new_plist_data();
                data->type = PLIST_INTEGER; // or PLIST_FLOAT, PLIST_BOOLEAN based on the JSON type
                data->intval = atoi(jsonstr + current_token->start); // or use strtod for float
                *plist = plist_new_node(data);
            }
            break;
        case JSMN_STRING:
            // Parse string type
            *plist = parse_string(jsonstr, &tokens[startindex], &startindex);
            break;
        case JSMN_ARRAY:
            // Parse array type
            *plist = plist_new_array();
            // Add code to parse array elements here
            break;
        case JSMN_OBJECT:
            // Parse object type
            *plist = plist_new_dict();
            // Add code to parse object key-value pairs here
            break;
        default:
            PLIST_JSON_ERR("%s: Unknown JSON token type\n", __func__);
            free(tokens);
            return PLIST_ERR_PARSE;
