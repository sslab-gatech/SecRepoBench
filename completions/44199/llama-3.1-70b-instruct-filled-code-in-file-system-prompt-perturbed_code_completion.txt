jsmntok_info_t ti;
ti.tokens = tokens;
ti.count = r;

plist_t val = NULL;
switch (tokens[startindex].type) {
    case JSMN_OBJECT:
        val = parse_object(jsonstr, &ti, &startindex);
        break;
    case JSMN_ARRAY:
        val = parse_array(jsonstr, &ti, &startindex);
        break;
    case JSMN_STRING:
        val = parse_string(jsonstr, &ti, &startindex);
        break;
    case JSMN_PRIMITIVE:
        val = parse_primitive(jsonstr, &ti, &startindex);
        break;
    default:
        break;
}
*plist = val;
