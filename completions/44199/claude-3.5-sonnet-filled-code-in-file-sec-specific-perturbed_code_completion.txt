jsmntok_info_t ti = { tokens, r };
plist_t result = NULL;

switch (tokens[startindex].type) {
    case JSMN_PRIMITIVE:
        result = parse_primitive(jsonstr, &ti, &startindex);
        break;
    case JSMN_STRING:
        result = parse_string(jsonstr, &ti, &startindex);
        break;
    case JSMN_ARRAY:
        result = parse_array(jsonstr, &ti, &startindex);
        break;
    case JSMN_OBJECT:
        result = parse_object(jsonstr, &ti, &startindex);
        break;
    default:
        PLIST_JSON_ERR("%s: Unexpected token type %d\n", __func__, tokens[startindex].type);
        free(tokens);
        return PLIST_ERR_PARSE;
}

if (!result) {
    free(tokens);
    return PLIST_ERR_PARSE;
}

*plist = result;
