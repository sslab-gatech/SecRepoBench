jsmntok_info_t ti = {tokens, r};
    switch (tokens[startindex].type) {
        case JSMN_PRIMITIVE:
            *plist = parse_primitive(jsonstr, &ti, &startindex);
            break;
        case JSMN_STRING:
            *plist = parse_string(jsonstr, &ti, &startindex);
            break;
        case JSMN_ARRAY:
            *plist = parse_array(jsonstr, &ti, &startindex);
            break;
        case JSMN_OBJECT:
            *plist = parse_object(jsonstr, &ti, &startindex);
            break;
        default:
            PLIST_JSON_ERR("%s: invalid json type %d\n", __func__, tokens[startindex].type);
            free(tokens);
            return PLIST_ERR_PARSE;
    }

    if (!*plist) {
        PLIST_JSON_ERR("%s: failed to parse JSON\n", __func__);
        free(tokens);
        return PLIST_ERR_PARSE;
