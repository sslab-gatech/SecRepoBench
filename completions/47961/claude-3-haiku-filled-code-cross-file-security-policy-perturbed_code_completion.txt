// <MASK>
while (uc < ueend) {
    uc = json_skip_space(uc, ueend);
    if (uc == ueend) {
        goto out;
    }

    if (!json_parse_element(&uc, ueend, st, lvl + 1)) {
        DPRINTF("Bad element", uc, *ucp);
        goto out;
    }

    uc = json_skip_space(uc, ueend);
    if (uc == ueend) {
        goto out;
    }

    if (*uc == ']') {
        uc++;
        DPRINTF("Good array: ", uc, *ucp);
        *ucp = uc;
        return 1;
    } else if (*uc == ',') {
        uc++;
        continue;
    } else {
        DPRINTF("Bad array: ", uc, *ucp);
        goto out;
    }
}
