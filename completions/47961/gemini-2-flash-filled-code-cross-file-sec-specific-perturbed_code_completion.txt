			goto out;
		switch (*uc++) {
		case ',':
			continue;
		case ']':
			DPRINTF("Good array: ", uc, *ucp);
			*ucp = uc;
			return 1;
		default:
			DPRINTF("Bad array syntax: ", uc - 1, *ucp);
