uc = json_skip_space(uc, ue);
if (uc == ue)
	goto out;
if (*uc == ']') {
	uc++;
	goto done;
}
do {
	if (!json_parse(&uc, ue, st, lvl)) {
		DPRINTF("not json", uc, *ucp);
		goto out;
	}
	(*st)[JSON_ARRAYN]++;
	uc = json_skip_space(uc, ue);
	if (uc == ue)
		goto out;
	switch (*uc++) {
	case ',':
		continue;
	case ']': /* [ */
	done:
		DPRINTF("Good array: ", uc, *ucp);
		*ucp = uc;
		return 1;
	default:
		DPRINTF("not more", uc, *ucp);
		*ucp = uc - 1;
		goto out;
	}
} while (1);
