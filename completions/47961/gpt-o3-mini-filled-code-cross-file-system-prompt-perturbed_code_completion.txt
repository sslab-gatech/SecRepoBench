/* Skip any whitespace */
		uc = json_skip_space(uc, ueend);
		if (uc == ueend)
			goto out;
		/* If we encounter a closing bracket, the array ends */
		if (*uc == ']') {
			uc++;
			st[JSON_ARRAYN]++;
			DPRINTF("Good array: ", uc, *ucp);
			*ucp = uc;
			return 1;
		}
		/* Parse the next JSON element */
		if (!json_parse(&uc, ueend, st, lvl + 1))
			goto out;
		/* Skip any whitespace following the element */
		uc = json_skip_space(uc, ueend);
		if (uc == ueend)
			goto out;
		/* After a successful element parse, expect either a comma or a closing bracket */
		if (*uc == ',') {
			uc++;
			continue;
		} else if (*uc == ']') {
			uc++;
			st[JSON_ARRAYN]++;
			DPRINTF("Good array: ", uc, *ucp);
			*ucp = uc;
			return 1;
		} else {
			goto out;
