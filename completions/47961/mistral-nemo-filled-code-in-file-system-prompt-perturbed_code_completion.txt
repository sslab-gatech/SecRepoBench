for (; uc < ue; uc++) {
		uc = json_skip_space(uc, ue);
		if (uc == ue)
			break;
		if (*uc == ']') {
			uc++;
			goto done;
		}
		if (!json_parse(&uc, ue, st, lvl + 1)) {
			DPRINTF("not json", uc, *ucp);
			break;
		}
		uc = json_skip_space(uc, ue);
		if (uc == ue)
			break;
		if (*uc++ != ',') {
			DPRINTF("not comma", uc, *ucp);
			break;
		}
	}
done:
	DPRINTF("Good array: ", uc, *ucp);
	*ucp = uc;
	return 1;
