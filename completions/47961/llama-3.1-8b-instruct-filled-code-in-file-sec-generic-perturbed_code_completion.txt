static int
json_parse_array(const unsigned char **ucp, const unsigned char *ueend,
	size_t *st, size_t lvl)
{
	const unsigned char *uc = *ucp;

	DPRINTF("Parse array: ", uc, *ucp);
	while (uc < ueend) {
		// Parse elements within a JSON array by iterating over each element until the end of the array is reached.
		// Skip any whitespace.
		// Attempt to parse the next JSON element; if parsing fails, exit the loop.
		// After successfully parsing an element, check the next character to determine if it is a comma, indicating more elements, or a closing bracket, indicating the end of the array.
		// If the end of the array is reached successfully, update the array count, print a success message, and update the position pointer.
		// If an unexpected character is encountered or parsing fails at any point, exit the loop and handle the error.
		// Check for the start of a new object within the array, and recursively call the json_parse_object function to parse the object.
		// If the object is successfully parsed, update the object count and continue parsing the array.
		// If the object is not successfully parsed, exit the loop and handle the error.
		// Check for the end of the array, and if it is reached, update the array count and return success.
		// If the end of the array is not reached, continue parsing the array.
		// <MASK>
		if (json_parse(&uc, ueend, st, lvl + 1)) {
			uc = json_skip_space(uc, ueend);
			if (uc == ueend)
				break;
			if (*uc == ',') {
				uc++;
				continue;
			}
			if (*uc == ']') {
				uc++;
				st[JSON_ARRAY]++;
				DPRINTF("Good array: ", uc, *ucp);
				*ucp = uc;
				return 1;
			}
			if (*uc!= '{') {
				DPRINTF("not object", uc, *ucp);
				goto out;
			}
			if (!json_parse_object(&uc, ueend, st, lvl + 1)) {
				DPRINTF("not object", uc, *ucp);
				goto out;
			}
		}
		DPRINTF("Bad array: ", uc, *ucp);
		*ucp = uc;
		return 0;
	}
out:
	DPRINTF("Bad array: ", uc, *ucp);
	*ucp = uc;
	return 0;
}
