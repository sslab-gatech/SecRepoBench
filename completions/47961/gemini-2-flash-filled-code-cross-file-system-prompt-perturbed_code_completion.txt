uc = json_skip_space(uc, ue);
		if (uc == ue)
			goto out;

		if (*uc == ']') {
			uc++;
			st[JSON_ARRAYN]++;
			DPRINTF("Good array: ", uc, *ucp);
			*ucp = uc;
			return 1;
		}

		if (!json_parse(&uc, ue, st, lvl + 1)) {
			DPRINTF("Bad array element: ", uc, *ucp);
			goto out;
		}

		if (uc == ue)
			goto out;

		switch (*uc++) {
		case ',':
			continue;
		case ']':
			st[JSON_ARRAYN]++;
			DPRINTF("Good array: ", uc, *ucp);
			*ucp = uc;
			return 1;
		default:
			DPRINTF("Bad array separator: ", uc - 1, *ucp);
			uc--;
			goto out;
		}
