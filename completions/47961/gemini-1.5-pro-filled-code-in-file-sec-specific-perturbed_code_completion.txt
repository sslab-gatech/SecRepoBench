uc = json_skip_space(uc, ueend);
		if (uc == ueend)
			goto out;
		if (*uc == ']') {
			uc++;
			if (JSON_COUNT) {
				if (lvl == 1)
					st[JSON_ARRAYN]++;
			} else {
				if (lvl == 1)
					goto done;
			}
			DPRINTF("Good array: ", uc, *ucp);
			*ucp = uc;
			return 1;
		}
		if (!json_parse(&uc, ueend, st, lvl + 1))
			goto out;
		if (uc == ueend)
			goto out;

		switch (*uc) {
		case ',':
			uc++;
			break;
		case ']':
			uc++;
			goto done;
		default:
			goto out;
		}
	}
done:
	if (JSON_COUNT && lvl == 1)
		st[JSON_ARRAYN]++;
	DPRINTF("Good array: ", uc, *ucp);
	*ucp = uc;
	return 1;
