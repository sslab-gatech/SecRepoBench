uc = json_skip_space(uc, ueend);
if (uc == ueend)
    goto out;
if (*uc == ']') {
    uc++;
    DPRINTF("Empty array: ", uc, *ucp);
    *ucp = uc;
    return 1;
}

int seenComma = 0;
while (1) {
    if (!json_parse(&uc, ueend, st, lvl)) {
        DPRINTF("not json", uc, *ucp);
        goto out;
    }
    
    uc = json_skip_space(uc, ueend);
    if (uc == ueend)
        goto out;
    
    switch (*uc++) {
    case ',':
        if (!seenComma) {
            st[JSON_ARRAYN]++;
            seenComma = 1;
        }
        continue;
    case ']':
        DPRINTF("Good array: ", uc, *ucp);
        *ucp = uc;
        return 1;
    default:
        DPRINTF("unexpected character", uc, *ucp);
        *ucp = uc - 1;
        goto out;
    }
}
