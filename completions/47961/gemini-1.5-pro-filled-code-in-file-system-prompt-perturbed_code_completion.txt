uc = json_skip_space(uc, ueend);
		if (uc == ueend)
			goto out;
		if (!json_parse(&uc, ueend, st, lvl + 1))
			goto out;
		if (uc == ueend)
			goto out;
		if (JSON_COUNT) {
			if (*uc == ']') {
				uc++;
				st[JSON_ARRAY]++;
				goto done;
			}
		} else if (st[JSON_OBJECT] || st[JSON_ARRAYN])
			return 1;
		if (*uc++ != ',')
			goto out;
		if (JSON_COUNT && *--uc == ']') {
			uc++;
			st[JSON_ARRAYN]++;
			goto done;
		}
	}
done:
	DPRINTF("Good array: ", uc, *ucp);
	*ucp = uc;
	return 1;
