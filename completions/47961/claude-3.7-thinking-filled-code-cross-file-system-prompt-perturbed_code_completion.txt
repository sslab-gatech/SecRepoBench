uc = json_skip_space(uc, ueend);
if (uc == ueend)
    goto out;
if (*uc == ']') {
    uc++;
    goto done;
}

int count = 0;
for (;;) {
    count++;
    if (!json_parse(&uc, ueend, st, lvl)) {
        DPRINTF("not json", uc, *ucp);
        goto out;
    }
    
    uc = json_skip_space(uc, ueend);
    if (uc == ueend)
        goto out;
    
    switch (*uc++) {
    case ',':
        continue;
    case ']':
        goto done;
    default:
        DPRINTF("not more", uc, *ucp);
        *ucp = uc - 1;
        goto out;
    }
}

done:
    if (count > 1)
        st[JSON_ARRAYN]++;
    DPRINTF("Good array: ", uc, *ucp);
    *ucp = uc;
    return 1;
