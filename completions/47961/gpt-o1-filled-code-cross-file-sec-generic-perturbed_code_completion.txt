{
	uc = json_skip_space(uc, ueend);
	if (uc == ueend)
		goto out;

	if (*uc == ']') {
		uc++;
		goto done;
	}

	/* Parse array elements */
	for (;;) {
		/* Parse the next JSON value within the array */
		if (!json_parse(&uc, ueend, st, lvl + 1)) {
			DPRINTF("not json", uc, *ucp);
			goto out;
		}

		uc = json_skip_space(uc, ueend);
		if (uc == ueend)
			goto out;

		/* Check if we have another element or the end of the array */
		switch (*uc++) {
		case ',':
			uc = json_skip_space(uc, ueend);
			if (uc == ueend)
				goto out;
			if (*uc == ']') {
				uc++;
				goto done;
			}
			break;
		case ']':
			goto done;
		default:
			DPRINTF("not more", uc, *ucp);
			*ucp = uc - 1;
			goto out;
		}
	}

done:
	DPRINTF("Good array: ", uc, *ucp);
	*ucp = uc;
	return 1;
}
