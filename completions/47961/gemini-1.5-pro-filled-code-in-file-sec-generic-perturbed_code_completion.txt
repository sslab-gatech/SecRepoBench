uc = json_skip_space(uc, ueend);
		if (uc == ueend)
			goto out;
		if (*uc == ']') {
			uc++;
			if (JSON_COUNT) {
				if (st[JSON_ARRAY] > 1)
					st[JSON_ARRAYN]++;
			}
#if !JSON_COUNT
			else {
				*ucp = uc;
				return 1;
			}
#endif
			DPRINTF("Good array: ", uc, *ucp);
			*ucp = uc;
			return 1;
		}
		if (!json_parse(&uc, ueend, st, lvl + 1))
			goto out;
		if (uc == ueend)
			goto out;

		switch (*uc) {
		case ',':
			uc++;
			break;
		case ']':
			uc++;
			if (JSON_COUNT) {
				if (st[JSON_ARRAY] > 1)
					st[JSON_ARRAYN]++;
			}
			DPRINTF("Good array: ", uc, *ucp);
			*ucp = uc;
			return 1;
		default:
			goto out;
		}
