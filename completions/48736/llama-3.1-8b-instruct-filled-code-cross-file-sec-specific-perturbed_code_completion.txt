    if (*uc == '"') {
        uc++;
        break;
    }
    if (*uc!= *str) {
        DPRINTF("Bad const: ", uc, *ucp);
        return 0;
    }
    uc++;
    str++;
}

    DPRINTF("Good const: ", uc, *ucp);
    return 1;
}
