int64_t ref_seq_start, ref_seq_span;

#ifdef LARGE_POS
        cp += safe_itf8_get((char *)cp, (char *)cp_end, &slice_header->ref_seq_id);
        cp += safe_ltf8_get((char *)cp, (char *)cp_end, &ref_seq_start);
        cp += safe_ltf8_get((char *)cp, (char *)cp_end, &ref_seq_span);
#else
        cp += safe_itf8_get((char *)cp, (char *)cp_end, &slice_header->ref_seq_id);
        cp += safe_itf8_get((char *)cp, (char *)cp_end, (int32_t *)&ref_seq_start);
        cp += safe_itf8_get((char *)cp, (char *)cp_end, (int32_t *)&ref_seq_span);
#endif

        // Validate ref_seq_start and ref_seq_span
        if (ref_seq_start < 0 || ref_seq_span < 0 || ref_seq_start > INT64_MAX - ref_seq_span) {
            free(slice_header);
            return NULL;
        }

        slice_header->ref_seq_start = ref_seq_start;
        slice_header->ref_seq_span = ref_seq_span;
