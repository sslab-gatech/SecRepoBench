cp += safe_itf8_get((char *)cp, (char *)cp_end, &slice_header->ref_seq_id);

#ifdef LARGE_POS
if (safe_ltf8_get((char *)cp, (char *)cp_end, &slice_header->ref_seq_start) < 0 ||
    safe_ltf8_get((char *)cp + 8, (char *)cp_end, &slice_header->ref_seq_span) < 0) {
    free(slice_header);
    return NULL;
}
cp += 16;
#else
int32_t ref_seq_start, ref_seq_span;
if (safe_itf8_get((char *)cp, (char *)cp_end, &ref_seq_start) < 0 ||
    safe_itf8_get((char *)cp + 4, (char *)cp_end, &ref_seq_span) < 0 ||
    ref_seq_start < 0 || ref_seq_span < 0) {
    free(slice_header);
    return NULL;
}
slice_header->ref_seq_start = ref_seq_start;
slice_header->ref_seq_span = ref_seq_span;
cp += 8;
#endif

if (slice_header->ref_seq_start < 0 || slice_header->ref_seq_span < 0) {
    free(slice_header);
    return NULL;
}
