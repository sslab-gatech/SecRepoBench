#ifdef MRB_USE_BIGINT
  if (mrb_bigint_p(v1) || mrb_bigint_p(value)) {
    return mrb_bint_cmp(mrb, v1, value);
  }
#endif

#ifdef MRB_NO_FLOAT
  if (!mrb_integer_p(v1)) {
    mrb_raise(mrb, E_TYPE_ERROR, "non-integer value");
  }
  x = mrb_integer(v1);
#else
  switch (mrb_type(v1)) {
  case MRB_TT_INTEGER:
    x = (mrb_float)mrb_integer(v1);
    break;
  case MRB_TT_FLOAT:
    x = mrb_float(v1);
    break;
#ifdef MRB_USE_RATIONAL
  case MRB_TT_RATIONAL:
    x = mrb_as_float(mrb, v1);
    break;
#endif
  default:
    mrb_raise(mrb, E_TYPE_ERROR, "non-numeric value");
    break;
  }
#endif
