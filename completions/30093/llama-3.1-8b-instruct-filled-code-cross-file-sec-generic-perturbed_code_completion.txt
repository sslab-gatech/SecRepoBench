// Convert the value extracted from the keymap to its string representation
// based on its data type. Handle boolean, integer, floating-point, string, and 
// null values accordingly, appending each to the buffer.
if (v->type == FLB_RA_VALUE_BOOL) {
    if (v->bool_value) {
        sdsappend(buf, "true");
    } else {
        sdsappend(buf, "false");
    }
}
else if (v->type == FLB_RA_VALUE_INT) {
    sdsappend(buf, "%d", v->int_value);
}
else if (v->type == FLB_RA_VALUE_FLOAT) {
    sdsappend(buf, "%f", v->float_value);
}
else if (v->type == FLB_RA_VALUE_STRING) {
    sdsappend(buf, "%s", v->str_value);
}
else if (v->type == FLB_RA_VALUE_NULL) {
    sdsappend(buf, "null");
}
else {
    // Handle other types if needed
}
