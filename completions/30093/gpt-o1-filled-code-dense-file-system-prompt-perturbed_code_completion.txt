tmp = buf;

    switch (v->o.type) {
    case MSGPACK_OBJECT_BOOLEAN:
        if (v->o.via.boolean) {
            tmp = flb_sds_cat(tmp, "true", 4);
        }
        else {
            tmp = flb_sds_cat(tmp, "false", 5);
        }
        break;
    case MSGPACK_OBJECT_POSITIVE_INTEGER:
    case MSGPACK_OBJECT_NEGATIVE_INTEGER:
        str_size = snprintf(str, sizeof(str), "%" PRId64, v->o.via.i64);
        tmp = flb_sds_cat(tmp, str, str_size);
        break;
    case MSGPACK_OBJECT_FLOAT:
        str_size = snprintf(str, sizeof(str), "%f", v->o.via.f64);
        tmp = flb_sds_cat(tmp, str, str_size);
        break;
    case MSGPACK_OBJECT_STR:
        tmp = flb_sds_cat(tmp, v->o.via.str.ptr, v->o.via.str.size);
        break;
    case MSGPACK_OBJECT_NIL:
        tmp = flb_sds_cat(tmp, "null", 4);
        break;
    default:
        /* For unsupported types (array, map, bin, ext), do nothing. */
        tmp = buf;
        break;
    }
