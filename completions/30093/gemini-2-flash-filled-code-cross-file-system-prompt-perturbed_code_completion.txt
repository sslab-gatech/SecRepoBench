if (v->type == MSGPACK_OBJECT_BOOLEAN) {
        if (v->data.boolean == true) {
            tmp = flb_sds_cat(buf, "true", 4);
        }
        else {
            tmp = flb_sds_cat(buf, "false", 5);
        }
    }
    else if (v->type == MSGPACK_OBJECT_POSITIVE_INTEGER) {
        str_size = snprintf(str, sizeof(str) - 1, "%lu", v->data.integer);
        if (str_size > 0) {
            tmp = flb_sds_cat(buf, str, str_size);
        }
    }
    else if (v->type == MSGPACK_OBJECT_NEGATIVE_INTEGER) {
        str_size = snprintf(str, sizeof(str) - 1, "%li", v->data.integer);
        if (str_size > 0) {
            tmp = flb_sds_cat(buf, str, str_size);
        }
    }
    else if (v->type == MSGPACK_OBJECT_FLOAT) {
        str_size = snprintf(str, sizeof(str) - 1, "%f", v->data.float_num);
        if (str_size > 0) {
            tmp = flb_sds_cat(buf, str, str_size);
        }
    }
    else if (v->type == MSGPACK_OBJECT_STR) {
        js = (char *) v->data.string.ptr;
        tmp = flb_sds_cat(buf, js, v->data.string.size);
    }
    else if (v->type == MSGPACK_OBJECT_NIL) {
        tmp = flb_sds_cat(buf, "null", 4);
    }
    else {
        tmp = buf;
    }
