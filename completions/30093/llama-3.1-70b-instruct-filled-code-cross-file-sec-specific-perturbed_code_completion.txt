switch (v->type) {
    case MSGPACK_OBJECT_BOOLEAN:
        tmp = flb_sds_catprintf(buf, "%s", v->via.boolean? "true" : "false");
        break;
    case MSGPACK_OBJECT_POSITIVE_INTEGER:
    case MSGPACK_OBJECT_NEGATIVE_INTEGER:
        str_size = snprintf(str, sizeof(str), "%d", v->via.i64);
        tmp = flb_sds_catprintf(buf, "%s", str);
        break;
    case MSGPACK_OBJECT_FLOAT:
        str_size = snprintf(str, sizeof(str), "%f", v->via.f64);
        tmp = flb_sds_catprintf(buf, "%s", str);
        break;
    case MSGPACK_OBJECT_STR:
        js = flb_json_escape(v->via.str.ptr, v->via.str.size);
        tmp = flb_sds_catprintf(buf, "%s", js);
        flb_free(js);
        break;
    case MSGPACK_OBJECT_NIL:
        tmp = flb_sds_catprintf(buf, "null");
        break;
    default:
        flb_error("[record accessor] unsupported value type");
        flb_ra_key_value_destroy(v);
        return NULL;
}
