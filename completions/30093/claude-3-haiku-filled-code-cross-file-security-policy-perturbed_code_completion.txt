// Convert the value extracted from the keymap to its string representation
// based on its data type. Handle boolean, integer, floating-point, string, and 
// null values accordingly, appending each to the buffer.
switch (v->type) {
    case FLB_RA_VALUE_BOOL:
        str_size = snprintf(str, sizeof(str), "%s", v->data.boolean ? "true" : "false");
        if (str_size < 0 || (size_t)str_size >= sizeof(str)) {
            flb_error("[record accessor] failed to convert boolean value to string");
            return NULL;
        }
        tmp = flb_sds_cat(buf, str, str_size);
        break;
    case FLB_RA_VALUE_INT:
        str_size = snprintf(str, sizeof(str), "%"PRId64, v->data.integer);
        if (str_size < 0 || (size_t)str_size >= sizeof(str)) {
            flb_error("[record accessor] failed to convert integer value to string");
            return NULL;
        }
        tmp = flb_sds_cat(buf, str, str_size);
        break;
    case FLB_RA_VALUE_FLOAT:
        str_size = snprintf(str, sizeof(str), "%f", v->data.floating);
        if (str_size < 0 || (size_t)str_size >= sizeof(str)) {
            flb_error("[record accessor] failed to convert float value to string");
            return NULL;
        }
        tmp = flb_sds_cat(buf, str, str_size);
        break;
    case FLB_RA_VALUE_STRING:
        tmp = flb_sds_cat(buf, v->data.string, flb_sds_len(v->data.string));
        break;
    case FLB_RA_VALUE_NULL:
        tmp = flb_sds_cat(buf, "null", 4);
        break;
    default:
        flb_error("[record accessor] unknown value type");
        return NULL;
}
