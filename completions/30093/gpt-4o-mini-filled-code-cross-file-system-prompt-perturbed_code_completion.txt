switch (v->type) {
        case FLB_RA_VALUE_STRING:
            tmp = flb_sds_create(v->value.str);
            break;
        case FLB_RA_VALUE_BOOL:
            tmp = flb_sds_create(v->value.boolean ? "true" : "false");
            break;
        case FLB_RA_VALUE_INT:
            str_size = snprintf(str, sizeof(str), "%d", v->value.integer);
            tmp = flb_sds_create_size(str_size);
            flb_sds_cat(tmp, str, str_size);
            break;
        case FLB_RA_VALUE_DOUBLE:
            str_size = snprintf(str, sizeof(str), "%f", v->value.dbl);
            tmp = flb_sds_create_size(str_size);
            flb_sds_cat(tmp, str, str_size);
            break;
        case FLB_RA_VALUE_NULL:
            tmp = flb_sds_create("null");
            break;
        default:
            tmp = flb_sds_create("");
            break;
    }
