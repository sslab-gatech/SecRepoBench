switch (v->type) {
    case FLB_RA_BOOL:
        if (v->val.boolean) {
            tmp = flb_sds_cat(buf, "true", 4);
        }
        else {
            tmp = flb_sds_cat(buf, "false", 5);
        }
        break;
    case FLB_RA_INT:
        str_size = snprintf(str, sizeof(str) - 1, "%" PRId64, v->val.i64);
        if (str_size < sizeof(str)) {
            tmp = flb_sds_cat(buf, str, str_size);
        }
        break;
    case FLB_RA_FLOAT:
        str_size = snprintf(str, sizeof(str) - 1, "%f", v->val.f64);
        if (str_size < sizeof(str)) {
            tmp = flb_sds_cat(buf, str, str_size);
        }
        break;
    case FLB_RA_STRING:
        tmp = flb_sds_cat(buf, v->val.string, v->val.string_len);
        break;
    case FLB_RA_NULL:
        tmp = flb_sds_cat(buf, "null", 4);
        break;
    default:
        flb_error("[record accessor] unknown type %i", v->type);
        break;
    }
