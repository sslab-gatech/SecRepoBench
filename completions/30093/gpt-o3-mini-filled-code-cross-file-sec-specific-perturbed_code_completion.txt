switch (v->obj.type) {
case MSGPACK_OBJECT_BOOLEAN:
    js = (v->obj.via.boolean) ? "true" : "false";
    tmp = flb_sds_cat(buf, js, strlen(js));
    break;
case MSGPACK_OBJECT_POSITIVE_INTEGER:
case MSGPACK_OBJECT_NEGATIVE_INTEGER:
    str_size = snprintf(str, sizeof(str), "%" PRIi64, v->obj.via.i64);
    if (str_size < 0 || (size_t)str_size >= sizeof(str)) {
        flb_error("[record accessor] integer conversion error");
        tmp = NULL;
    }
    else {
        tmp = flb_sds_cat(buf, str, str_size);
    }
    break;
case MSGPACK_OBJECT_FLOAT:
    str_size = snprintf(str, sizeof(str), "%.6f", v->obj.via.f64);
    if (str_size < 0 || (size_t)str_size >= sizeof(str)) {
        flb_error("[record accessor] float conversion error");
        tmp = NULL;
    }
    else {
        tmp = flb_sds_cat(buf, str, str_size);
    }
    break;
case MSGPACK_OBJECT_STR:
    tmp = flb_sds_cat(buf, v->obj.via.str.ptr, v->obj.via.str.size);
    break;
case MSGPACK_OBJECT_NIL:
    js = "null";
    tmp = flb_sds_cat(buf, js, 4);
    break;
default:
    /* For unknown types, leave the buffer unchanged */
    tmp = buf;
    break;
}
