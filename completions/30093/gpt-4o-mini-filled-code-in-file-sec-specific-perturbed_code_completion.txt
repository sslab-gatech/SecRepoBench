switch (v->type) {
        case FLB_RA_VALUE_STRING:
            tmp = flb_sds_cat(buf, v->value.str, flb_sds_len(v->value.str));
            break;
        case FLB_RA_VALUE_BOOL:
            tmp = flb_sds_cat_printf(buf, "%s", v->value.boolean ? "true" : "false");
            break;
        case FLB_RA_VALUE_INTEGER:
            tmp = flb_sds_cat_printf(buf, "%lld", v->value.integer);
            break;
        case FLB_RA_VALUE_DOUBLE:
            tmp = flb_sds_cat_printf(buf, "%f", v->value.dbl);
            break;
        case FLB_RA_VALUE_NULL:
            tmp = flb_sds_cat(buf, "null", 4);
            break;
        default:
            *found = FLB_FALSE;
            return buf;
    }
