switch (v->type) {
    case MSGPACK_OBJECT_BOOLEAN:
        if (v->via.boolean) {
            tmp = flb_sds_cat(buf, "true", 4);
        }
        else {
            tmp = flb_sds_cat(buf, "false", 5);
        }
        break;
    case MSGPACK_OBJECT_POSITIVE_INTEGER:
        str_size = snprintf(str, sizeof(str) - 1, "%" PRIu64, v->via.u64);
        tmp = flb_sds_cat(buf, str, str_size);
        break;
    case MSGPACK_OBJECT_NEGATIVE_INTEGER:
        str_size = snprintf(str, sizeof(str) - 1, "%" PRId64, v->via.i64);
        tmp = flb_sds_cat(buf, str, str_size);
        break;
    case MSGPACK_OBJECT_FLOAT32:
    case MSGPACK_OBJECT_FLOAT64:
        str_size = snprintf(str, sizeof(str) - 1, "%f", v->via.f64);
        tmp = flb_sds_cat(buf, str, str_size);
        break;
    case MSGPACK_OBJECT_STR:
        tmp = flb_sds_cat(buf, v->via.str.ptr, v->via.str.size);
        break;
    case MSGPACK_OBJECT_NIL:
        tmp = flb_sds_cat(buf, "null", 4);
        break;
    default:
        tmp = buf;
        break;
    }
