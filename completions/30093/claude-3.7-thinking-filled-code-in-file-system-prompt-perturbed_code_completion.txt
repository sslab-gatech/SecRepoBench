/* Handle value based on its type */
switch (v->type) {
case FLB_RA_BOOL:
    if (v->boolean) {
        tmp = flb_sds_cat(buf, "true", 4);
    }
    else {
        tmp = flb_sds_cat(buf, "false", 5);
    }
    break;
case FLB_RA_INT:
    str_size = snprintf(str, sizeof(str) - 1, "%"PRId64, v->i64);
    tmp = flb_sds_cat(buf, str, str_size);
    break;
case FLB_RA_FLOAT:
    str_size = snprintf(str, sizeof(str) - 1, "%f", v->f64);
    tmp = flb_sds_cat(buf, str, str_size);
    break;
case FLB_RA_STRING:
    tmp = flb_sds_cat(buf, v->str, flb_sds_len(v->str));
    break;
case FLB_RA_NULL:
    tmp = flb_sds_cat(buf, "null", 4);
    break;
default:
    /* Object type not supported for string conversion */
    *found = FLB_FALSE;
    tmp = buf;
    break;
}
