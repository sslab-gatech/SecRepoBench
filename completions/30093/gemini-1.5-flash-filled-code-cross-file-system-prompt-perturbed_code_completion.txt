switch (v->type) {
    case FLB_RA_VALUE_BOOL:
        tmp = flb_sds_cat_printf(buf, "%s", v->value.b ? "true" : "false");
        break;
    case FLB_RA_VALUE_INTEGER:
        tmp = flb_sds_cat_printf(buf, "%" PRId64, v->value.i);
        break;
    case FLB_RA_VALUE_DOUBLE:
        snprintf(str, sizeof(str), "%g", v->value.d);
        str_size = strlen(str);
        tmp = flb_sds_cat(buf, str, str_size);
        break;
    case FLB_RA_VALUE_STRING:
        tmp = flb_sds_cat(buf, v->value.s, strlen(v->value.s));
        break;
    case FLB_RA_VALUE_NULL:
        tmp = flb_sds_cat(buf, "null", 4);
        break;
    default:
        flb_error("[record accessor] unsupported keymap value type");
        flb_sds_destroy(buf);
        return NULL;
    }
