/* Convert value to string representation */
if (v->o.type == MSGPACK_OBJECT_BOOLEAN) {
    /* Boolean */
    if (v->o.via.boolean) {
        tmp = flb_sds_cat(buf, "true", 4);
    }
    else {
        tmp = flb_sds_cat(buf, "false", 5);
    }
}
else if (v->o.type == MSGPACK_OBJECT_POSITIVE_INTEGER) {
    /* Positive integer */
    str_size = snprintf(str, sizeof(str) - 1, "%lu", v->o.via.u64);
    tmp = flb_sds_cat(buf, str, str_size);
}
else if (v->o.type == MSGPACK_OBJECT_NEGATIVE_INTEGER) {
    /* Negative integer */
    str_size = snprintf(str, sizeof(str) - 1, "%ld", v->o.via.i64);
    tmp = flb_sds_cat(buf, str, str_size);
}
else if (v->o.type == MSGPACK_OBJECT_FLOAT) {
    /* Float */
    str_size = snprintf(str, sizeof(str) - 1, "%f", v->o.via.f64);
    tmp = flb_sds_cat(buf, str, str_size);
}
else if (v->o.type == MSGPACK_OBJECT_STR) {
    /* String */
    tmp = flb_sds_cat(buf, v->o.via.str.ptr, v->o.via.str.size);
}
else if (v->o.type == MSGPACK_OBJECT_NIL) {
    /* Null */
    tmp = flb_sds_cat(buf, "null", 4);
}
else {
    /* Unknown type, just return the original buffer */
    tmp = buf;
}
