// Convert the value extracted from the keymap to its string representation
// based on its data type. Handle boolean, integer, floating-point, string, and 
// null values accordingly, appending each to the buffer.
if (v->type == FLB_RA_VALUE_BOOL) {
    if (v->bool_value) {
        str_size = snprintf(str, sizeof(str), "true");
    }
    else {
        str_size = snprintf(str, sizeof(str), "false");
    }
}
else if (v->type == FLB_RA_VALUE_INT) {
    str_size = snprintf(str, sizeof(str), "%" PRId64, v->int_value);
}
else if (v->type == FLB_RA_VALUE_FLOAT) {
    str_size = snprintf(str, sizeof(str), "%f", v->float_value);
}
else if (v->type == FLB_RA_VALUE_STRING) {
    str_size = snprintf(str, sizeof(str), "%s", v->str_value);
}
else if (v->type == FLB_RA_VALUE_NULL) {
    str_size = snprintf(str, sizeof(str), "null");
}
else {
    // Handle other types if needed
    str_size = 0;
}

if (str_size >= sizeof(str)) {
    flb_error("[record accessor] string buffer overflow");
    return NULL;
}

tmp = flb_sds_create_size(str_size + 1);
if (!tmp) {
    flb_error("[record accessor] cannot create outgoing buffer");
    return NULL;
}

flb_sds_cat(tmp, str, str_size);
