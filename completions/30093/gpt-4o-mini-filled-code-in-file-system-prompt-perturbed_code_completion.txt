switch (v->type) {
        case FLB_RA_VALUE_BOOL:
            tmp = flb_sds_cat(buf, v->value.b ? "true" : "false", 4);
            break;
        case FLB_RA_VALUE_INT:
            snprintf(str, sizeof(str), "%d", v->value.i);
            tmp = flb_sds_cat(buf, str, strlen(str));
            break;
        case FLB_RA_VALUE_DOUBLE:
            snprintf(str, sizeof(str), "%f", v->value.d);
            tmp = flb_sds_cat(buf, str, strlen(str));
            break;
        case FLB_RA_VALUE_STRING:
            tmp = flb_sds_cat(buf, v->value.s, flb_sds_len(v->value.s));
            break;
        case FLB_RA_VALUE_NULL:
            tmp = flb_sds_cat(buf, "null", 4);
            break;
        default:
            *found = FLB_FALSE;
            return buf;
    }
