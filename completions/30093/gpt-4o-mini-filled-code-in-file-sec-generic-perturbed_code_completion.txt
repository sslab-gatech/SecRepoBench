switch (v->type) {
        case FLB_RA_VALUE_STRING:
            tmp = flb_sds_cat(buf, v->value.str, flb_sds_len(v->value.str));
            break;
        case FLB_RA_VALUE_INTEGER:
            snprintf(str, sizeof(str), "%d", v->value.i);
            tmp = flb_sds_cat(buf, str, strlen(str));
            break;
        case FLB_RA_VALUE_FLOAT:
            snprintf(str, sizeof(str), "%.2f", v->value.f);
            tmp = flb_sds_cat(buf, str, strlen(str));
            break;
        case FLB_RA_VALUE_BOOLEAN:
            js = v->value.b ? "true" : "false";
            tmp = flb_sds_cat(buf, js, strlen(js));
            break;
        case FLB_RA_VALUE_NULL:
            tmp = flb_sds_cat(buf, "null", 4);
            break;
        default:
            *found = FLB_FALSE;
            return buf;
    }
