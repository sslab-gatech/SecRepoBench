switch (v->type) {
    case MSGPACK_OBJECT_BOOLEAN:
        tmp = flb_sds_catprintf(buf, "%s", v->via.boolean? "true" : "false");
        break;
    case MSGPACK_OBJECT_POSITIVE_INTEGER:
    case MSGPACK_OBJECT_NEGATIVE_INTEGER:
        tmp = flb_sds_catprintf(buf, "%d", v->via.i64);
        break;
    case MSGPACK_OBJECT_FLOAT:
        tmp = flb_sds_catprintf(buf, "%f", v->via.f64);
        break;
    case MSGPACK_OBJECT_STR:
        tmp = flb_sds_catprintf(buf, "%s", v->via.str.ptr);
        break;
    case MSGPACK_OBJECT_NIL:
        tmp = flb_sds_catprintf(buf, "null");
        break;
    default:
        flb_error("[record accessor] unsupported data type");
        flb_ra_key_value_destroy(v);
        return NULL;
}
