switch (v->type) {
    case FLB_RA_VALUE_BOOLEAN:
        snprintf(str, sizeof(str), "%s", v->val.b ? "true" : "false");
        tmp = flb_sds_cat(buf, str, strlen(str));
        break;
    case FLB_RA_VALUE_INTEGER:
        snprintf(str, sizeof(str), "%" PRId64, v->val.i);
        tmp = flb_sds_cat(buf, str, strlen(str));
        break;
    case FLB_RA_VALUE_FLOAT:
        snprintf(str, sizeof(str), "%f", v->val.f);
        tmp = flb_sds_cat(buf, str, strlen(str));
        break;
    case FLB_RA_VALUE_STRING:
        tmp = flb_sds_cat(buf, v->val.s, flb_sds_len(v->val.s));
        break;
    case FLB_RA_VALUE_NULL:
        tmp = flb_sds_cat(buf, "null", 4);
        break;
    default:
        tmp = flb_sds_cat(buf, "unknown", 7);
        break;
    }
