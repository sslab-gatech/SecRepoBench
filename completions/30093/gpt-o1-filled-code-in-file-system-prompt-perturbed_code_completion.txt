tmp = buf;

    switch (v->type) {
    case FLB_RA_BOOL:
        if (v->val.boolean) {
            tmp = flb_sds_cat(tmp, "true", 4);
        }
        else {
            tmp = flb_sds_cat(tmp, "false", 5);
        }
        break;
    case FLB_RA_INT:
        snprintf(str, sizeof(str), "%" PRId64, v->val.i64);
        tmp = flb_sds_cat(tmp, str, strlen(str));
        break;
    case FLB_RA_FLOAT:
        snprintf(str, sizeof(str), "%f", v->val.f64);
        tmp = flb_sds_cat(tmp, str, strlen(str));
        break;
    case FLB_RA_STRING:
        tmp = flb_sds_cat(tmp, v->val.string, flb_sds_len(v->val.string));
        break;
    case FLB_RA_NULL:
        tmp = flb_sds_cat(tmp, "null", 4);
        break;
    default:
        break;
    }
