switch (v->type) {
    case MSGPACK_OBJECT_BOOLEAN:
        js = v->via.boolean? "true" : "false";
        str_size = strlen(js);
        tmp = flb_sds_cat_len(buf, js, str_size);
        break;
    case MSGPACK_OBJECT_POSITIVE_INTEGER:
    case MSGPACK_OBJECT_NEGATIVE_INTEGER:
        str_size = snprintf(str, sizeof(str), "%lld", v->via.i64);
        tmp = flb_sds_cat_len(buf, str, str_size);
        break;
    case MSGPACK_OBJECT_FLOAT:
        str_size = snprintf(str, sizeof(str), "%f", v->via.f64);
        tmp = flb_sds_cat_len(buf, str, str_size);
        break;
    case MSGPACK_OBJECT_STR:
        tmp = flb_sds_cat_len(buf, v->via.str.ptr, v->via.str.size);
        break;
    case MSGPACK_OBJECT_NIL:
        tmp = flb_sds_cat_len(buf, "null", 4);
        break;
    default:
        *found = FLB_FALSE;
        return buf;
}
