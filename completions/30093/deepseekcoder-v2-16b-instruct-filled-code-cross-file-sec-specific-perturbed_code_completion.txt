// Convert the value extracted from the keymap to its string representation
    // based on its data type. Handle boolean, integer, floating-point, string, and 
    // null values accordingly, appending each to the buffer.
    switch (v->type) {
        case FLB_RA_VALUE_TYPE_BOOLEAN:
            snprintf(str, sizeof(str), "%s", v->data.boolean ? "true" : "false");
            break;
        case FLB_RA_VALUE_TYPE_INTEGER:
            snprintf(str, sizeof(str), "%d", v->data.integer);
            break;
        case FLB_RA_VALUE_TYPE_FLOAT:
            snprintf(str, sizeof(str), "%f", v->data.floating);
            break;
        case FLB_RA_VALUE_TYPE_STRING:
            tmp = flb_sds_cat(tmp, v->data.string);
            break;
        case FLB_RA_VALUE_TYPE_NULL:
            snprintf(str, sizeof(str), "null");
            break;
        default:
            break;
    }

    if (tmp != buf) {
        buf = flb_sds_cat(buf, str);
    }
