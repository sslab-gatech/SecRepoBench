const char *sptr = (const char*)src;
  int slen = (int)RSTRING_LEN(src);

  if (count == -1) {
    count = slen * 8;
  }

  dst = mrb_str_new(mrb, NULL, count);
  char *dptr = RSTRING_PTR(dst);

  for (int i = 0; i < slen; i++) {
    unsigned char byte = (unsigned char)*sptr++;
    for (int j = 0; j < 8; j++) {
      *dptr++ = (byte & (1 << (7 - j))) ? '1' : '0';
    }
  }

  dst = mrb_str_resize(mrb, dst, (mrb_int)(dptr - RSTRING_PTR(dst)));
  mrb_ary_push(mrbstate, ary, dst);
  return slen;
