if(networkflow->detected_protocol == NDPI_PROTOCOL_HTTP) {
      if(networkflow->protos.http.url) {
        ndpi_free(networkflow->protos.http.url);
      }
      if(networkflow->protos.http.content_type) {
        ndpi_free(networkflow->protos.http.content_type);
      }
      if(networkflow->protos.http.user_agent) {
        ndpi_free(networkflow->protos.http.user_agent);
      }
      if(networkflow->protos.http.referer) {
        ndpi_free(networkflow->protos.http.referer);
      }
    } else if(networkflow->detected_protocol == NDPI_PROTOCOL_TLS || 
              networkflow->detected_protocol == NDPI_PROTOCOL_QUIC) {
      if(networkflow->protos.stun_ssl.ssl.server_name) {
        ndpi_free(networkflow->protos.stun_ssl.ssl.server_name);
      }
      if(networkflow->protos.stun_ssl.ssl.client_certificate) {
        ndpi_free(networkflow->protos.stun_ssl.ssl.client_certificate);
      }
    }
