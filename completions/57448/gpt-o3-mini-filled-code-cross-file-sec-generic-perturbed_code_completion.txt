if(flow->protos.tls_quic.server_names == NULL) {
  flow->protos.tls_quic.server_names = ndpi_strdup(dNSName);
} else {
  size_t old_len = strlen(flow->protos.tls_quic.server_names);
  size_t dns_len = strlen(dNSName);
  size_t new_len = old_len + 1 + dns_len;
  char *new_str = (char *)malloc(new_len + 1);
  if(new_str != NULL) {
    memcpy(new_str, flow->protos.tls_quic.server_names, old_len);
    new_str[old_len] = ',';
    memcpy(new_str + old_len + 1, dNSName, dns_len);
    new_str[new_len] = '\0';
    free(flow->protos.tls_quic.server_names);
    flow->protos.tls_quic.server_names = new_str;
  }
}
