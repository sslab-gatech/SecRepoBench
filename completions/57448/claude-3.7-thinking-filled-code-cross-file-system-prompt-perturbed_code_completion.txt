  char *new_str = ndpi_malloc(new_len + 1 /* null terminator */);
  
  if(new_str) {
    snprintf(new_str, new_len + 1, "%s,%s", flow->protos.tls_quic.server_names, dNSName);

    flow->protos.tls_quic.server_names = new_str;

  }
