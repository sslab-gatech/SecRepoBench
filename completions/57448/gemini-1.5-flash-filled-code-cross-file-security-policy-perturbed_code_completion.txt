if (flow->protos.tls_quic.server_names == NULL) {
      flow->protos.tls_quic.server_names = ndpi_strdup(dNSName);
      flow->protos.tls_quic.server_names_len = dns_name_length;
    } else {
      size_t new_len = strlen(flow->protos.tls_quic.server_names) + dns_name_length + 2; // +2 for comma and null terminator
      if (new_len > SIZE_MAX) {
        // Handle potential overflow
        ndpi_set_risk(ndpi_struct, flow, NDPI_MEMORY_ERROR, "Overflow in server_names concatenation");
        return;
      }
      char *new_server_names = (char *)ndpi_malloc(new_len);
      if (new_server_names == NULL) {
        ndpi_set_risk(ndpi_struct, flow, NDPI_MEMORY_ERROR, "Memory allocation failed in server_names concatenation");
        return;
      }
      snprintf(new_server_names, new_len, "%s,%s", flow->protos.tls_quic.server_names, dNSName);
      ndpi_free(flow->protos.tls_quic.server_names);
      flow->protos.tls_quic.server_names = new_server_names;
      flow->protos.tls_quic.server_names_len = new_len -1; // Subtract 1 for null terminator
    }
