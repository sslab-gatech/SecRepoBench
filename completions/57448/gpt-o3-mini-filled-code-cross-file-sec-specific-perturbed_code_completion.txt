if(flow->protos.tls_quic.server_names == NULL) {
    flow->protos.tls_quic.server_names = ndpi_strdup(dNSName);
} else {
    size_t old_len = strlen(flow->protos.tls_quic.server_names);
    size_t dns_len = strlen(dNSName);
    size_t new_len = old_len + 1 + dns_len;
    char *tmp = (char *)realloc(flow->protos.tls_quic.server_names, new_len + 1);
    if(tmp != NULL) {
        flow->protos.tls_quic.server_names = tmp;
        flow->protos.tls_quic.server_names[old_len] = ',';
        memcpy(flow->protos.tls_quic.server_names + old_len + 1, dNSName, dns_len);
        flow->protos.tls_quic.server_names[new_len] = '\0';
    }
}
