if (flow->protos.tls_quic.server_names == NULL) {
  flow->protos.tls_quic.server_names = ndpi_strdup(dNSName);
} else {
  size_t old_len = strlen(flow->protos.tls_quic.server_names);
  size_t new_len = old_len + 1 + dns_name_length + 1; /* 1 for comma, 1 for '\0' */
  char *tmp = ndpi_malloc(new_len);

  if (tmp != NULL) {
    snprintf(tmp, new_len, "%s,%s", flow->protos.tls_quic.server_names, dNSName);
    ndpi_free(flow->protos.tls_quic.server_names);
    flow->protos.tls_quic.server_names = tmp;
  }
}
