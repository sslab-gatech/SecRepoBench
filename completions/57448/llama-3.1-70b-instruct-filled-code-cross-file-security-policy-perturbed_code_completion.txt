if(flow->protos.tls_quic.server_names == NULL) {
  flow->protos.tls_quic.server_names = ndpi_strdup(dNSName);
  flow->protos.tls_quic.server_names_len = dns_name_length;
} else {
  u_int32_t new_len = flow->protos.tls_quic.server_names_len + dns_name_length + 2; /* +2 for the comma and the null terminator */
  if(new_len <= UINT32_MAX) { /* Check for overflow */
    char *new_server_names = ndpi_realloc(flow->protos.tls_quic.server_names, new_len);
    if(new_server_names!= NULL) {
      flow->protos.tls_quic.server_names = new_server_names;
      new_server_names[flow->protos.tls_quic.server_names_len] = ',';
      strncpy(&new_server_names[flow->protos.tls_quic.server_names_len + 1], dNSName, dns_name_length);
      new_server_names[new_len - 1] = '\0';
      flow->protos.tls_quic.server_names_len = new_len - 1;
    }
  }
}
