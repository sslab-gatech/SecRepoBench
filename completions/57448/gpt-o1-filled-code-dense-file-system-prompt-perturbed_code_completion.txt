if(flow->protos.tls_quic.server_names == NULL) {
  flow->protos.tls_quic.server_names = ndpi_malloc(dns_name_length + 1);
  if(flow->protos.tls_quic.server_names) {
    memcpy(flow->protos.tls_quic.server_names, dNSName, dns_name_length);
    flow->protos.tls_quic.server_names[dns_name_length] = '\0';
    flow->protos.tls_quic.server_names_len = dns_name_length;
  }
} else {
  uint32_t new_len = flow->protos.tls_quic.server_names_len + 1 /* comma */ + dns_name_length;
  char *tmp = ndpi_malloc(new_len + 1 /* terminator */);
  if(tmp) {
    ndpi_snprintf(tmp, new_len + 1, "%s,%s", flow->protos.tls_quic.server_names, dNSName);
    ndpi_free(flow->protos.tls_quic.server_names);
    flow->protos.tls_quic.server_names = tmp;
    flow->protos.tls_quic.server_names_len = new_len;
  }
}
