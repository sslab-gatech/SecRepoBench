if(flow->protos.tls_quic.server_names == NULL) {
  flow->protos.tls_quic.server_names = ndpi_strdup(dNSName);
} else {
  u_int16_t len = strlen(flow->protos.tls_quic.server_names) + strlen(dNSName) + 2;
  char *new_server_names = ndpi_malloc(len);
  if(new_server_names!= NULL) {
    snprintf(new_server_names, len, "%s, %s", flow->protos.tls_quic.server_names, dNSName);
    ndpi_free(flow->protos.tls_quic.server_names);
    flow->protos.tls_quic.server_names = new_server_names;
  }
