if (flow->protos.tls_quic.server_names == NULL) {
  flow->protos.tls_quic.server_names = ndpi_strdup(dNSName);
  flow->protos.tls_quic.server_names_len = dns_name_length;
} else {
  u_int16_t total_len = flow->protos.tls_quic.server_names_len + 1 /* comma */ + dns_name_length + 1 /* null terminator */;
  char *new_str = (char *)ndpi_malloc(total_len);
  
  if (new_str != NULL) {
    snprintf(new_str, total_len, "%s,%s", flow->protos.tls_quic.server_names, dNSName);
    ndpi_free(flow->protos.tls_quic.server_names);
    flow->protos.tls_quic.server_names = new_str;
    flow->protos.tls_quic.server_names_len = total_len - 1; /* exclude null terminator */
  }
}
