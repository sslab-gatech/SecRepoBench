if(!flow->protos.tls_quic.server_names) {
  flow->protos.tls_quic.server_names = ndpi_strdup(dNSName);
  flow->protos.tls_quic.server_names_len = dns_name_length;
} else {
  size_t new_len = strlen(flow->protos.tls_quic.server_names) + dns_name_length + 2;
  char *new_names = ndpi_malloc(new_len);
  if(new_names) {
    snprintf(new_names, new_len, "%s,%s", flow->protos.tls_quic.server_names, dNSName);
    ndpi_free(flow->protos.tls_quic.server_names);
    flow->protos.tls_quic.server_names = new_names;
    flow->protos.tls_quic.server_names_len = new_len - 1;
  }
}
