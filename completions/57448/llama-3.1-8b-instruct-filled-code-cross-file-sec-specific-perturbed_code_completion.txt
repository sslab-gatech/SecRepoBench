// Check if server_names in the TLS QUIC protocol structure is NULL.
// If it is, set it to a copy of the current dNSName and update the length.
// If not, append the current dNSName to the existing server_names with a comma separator.
// Allocate additional memory for the new concatenated string and update the length accordingly.
if(!flow->protos.tls_quic.server_names) {
  flow->protos.tls_quic.server_names = ndpi_strdup(dNSName);
  flow->protos.tls_quic.server_names_length = strlen(dNSName);
} else {
  u_int16_t new_length = flow->protos.tls_quic.server_names_length + strlen(dNSName) + 1;
  char *new_server_names = ndpi_realloc(flow->protos.tls_quic.server_names, new_length);
  if(new_server_names) {
    strcpy(new_server_names + flow->protos.tls_quic.server_names_length, dNSName);
    flow->protos.tls_quic.server_names = new_server_names;
    flow->protos.tls_quic.server_names_length = new_length;
  }
}
