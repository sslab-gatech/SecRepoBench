if(flow->protos.tls_quic.server_names == NULL) {
  flow->protos.tls_quic.server_names = ndpi_strdup(dNSName);
} else {
  // Need to allocate memory for: existing string + comma + new string + null terminator
  u_int len = strlen(flow->protos.tls_quic.server_names) + 1 + dns_name_length + 1;
  char *new_names = (char*)ndpi_malloc(len);
  
  if(new_names) {
    // Format: "<existing_names>,<new_name>"
    snprintf(new_names, len, "%s,%s", flow->protos.tls_quic.server_names, dNSName);
    ndpi_free(flow->protos.tls_quic.server_names);
    flow->protos.tls_quic.server_names = new_names;
  }
}
