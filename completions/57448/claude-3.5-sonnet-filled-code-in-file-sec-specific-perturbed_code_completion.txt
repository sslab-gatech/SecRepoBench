if(flow->protos.tls_quic.server_names == NULL) {
  flow->protos.tls_quic.server_names = ndpi_strdup(dNSName);
  flow->protos.tls_quic.server_names_len = dns_name_length;
} else {
  char *new_server_names;
  u_int16_t new_len = flow->protos.tls_quic.server_names_len + 1 + dns_name_length;
  
  new_server_names = ndpi_realloc(flow->protos.tls_quic.server_names, 
                                  flow->protos.tls_quic.server_names_len, 
                                  new_len + 1);
  if(new_server_names != NULL) {
    new_server_names[flow->protos.tls_quic.server_names_len] = ',';
    memcpy(&new_server_names[flow->protos.tls_quic.server_names_len + 1], 
           dNSName, dns_name_length);
    new_server_names[new_len] = '\0';
    
    flow->protos.tls_quic.server_names = new_server_names;
    flow->protos.tls_quic.server_names_len = new_len;
  }
}
