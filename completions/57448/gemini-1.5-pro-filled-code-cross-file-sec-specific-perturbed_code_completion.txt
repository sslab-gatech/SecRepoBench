if (flow->protos.tls_quic.server_names == NULL) {
  flow->protos.tls_quic.server_names = ndpi_strdup(dNSName);
  flow->protos.tls_quic.server_names_len = dns_name_length;
} else {
  u_int16_t new_len = flow->protos.tls_quic.server_names_len + dns_name_length + 2;
  char *new_server_names = (char *)malloc(new_len);
  if (new_server_names == NULL) {
    // Handle memory allocation failure
    ndpi_set_risk(ndpi_struct, flow, NDPI_MEMORY_ALLOCATION_FAILED, NULL);
    break;
  }
  snprintf(new_server_names, new_len, "%s,%s", flow->protos.tls_quic.server_names, dNSName);
  free(flow->protos.tls_quic.server_names);
  flow->protos.tls_quic.server_names = new_server_names;
  flow->protos.tls_quic.server_names_len = new_len -1;
}
