if (flow->protos.tls_quic.server_names == NULL) {
  flow->protos.tls_quic.server_names = ndpi_strdup(dNSName);
  flow->protos.tls_quic.server_names_len = dns_name_length;
} else {
  u_int32_t new_len = flow->protos.tls_quic.server_names_len + 1 + dns_name_length;
  
  if (new_len <= UINT32_MAX) {
    char *new_names = (char *)ndpi_realloc(flow->protos.tls_quic.server_names, new_len + 1);
    if (new_names != NULL) {
      new_names[flow->protos.tls_quic.server_names_len] = ',';
      memcpy(new_names + flow->protos.tls_quic.server_names_len + 1, dNSName, dns_name_length);
      new_names[new_len] = '\0';
      flow->protos.tls_quic.server_names = new_names;
      flow->protos.tls_quic.server_names_len = new_len;
    }
  }
}
