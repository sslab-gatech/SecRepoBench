if(flow->protos.tls_quic.server_names == NULL) {
  flow->protos.tls_quic.server_names = ndpi_strdup(dNSName);
  flow->protos.tls_quic.server_names_len = dns_name_length;
} else {
  size_t old_len = flow->protos.tls_quic.server_names_len;
  size_t new_len = old_len + 1 + dns_name_length; /* +1 for comma */
  char *new_names = ndpi_malloc(new_len + 1); /* +1 for null terminator */
  if(new_names) {
    memcpy(new_names, flow->protos.tls_quic.server_names, old_len);
    new_names[old_len] = ',';
    memcpy(new_names + old_len + 1, dNSName, dns_name_length);
    new_names[new_len] = '\0';
    ndpi_free(flow->protos.tls_quic.server_names);
    flow->protos.tls_quic.server_names = new_names;
    flow->protos.tls_quic.server_names_len = new_len;
  }
}
