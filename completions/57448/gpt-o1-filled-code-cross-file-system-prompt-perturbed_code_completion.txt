if(flow->protos.tls_quic.server_names == NULL) {
  flow->protos.tls_quic.server_names = ndpi_strdup(dNSName);
} else {
  size_t old_len = strlen(flow->protos.tls_quic.server_names);
  size_t new_len = old_len + 1 /* comma */ + dns_name_length;
  char *new_ptr = ndpi_malloc(new_len + 1);

  if(new_ptr) {
    snprintf(new_ptr, new_len + 1, "%s,%s",
             flow->protos.tls_quic.server_names, dNSName);
    ndpi_free(flow->protos.tls_quic.server_names);
    flow->protos.tls_quic.server_names = new_ptr;
  }
}
