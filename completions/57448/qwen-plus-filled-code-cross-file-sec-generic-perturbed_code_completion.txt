if (flow->protos.tls_quic.server_names == NULL) {
  flow->protos.tls_quic.server_names = ndpi_strdup(dNSName);
  flow->protos.tls_quic.server_names_len = dns_name_length;
} else {
  u_int16_t combined_length = strlen(flow->protos.tls_quic.server_names) + dns_name_length + 2; // Add 1 for the comma and 1 for the null terminator.
  char *combined_server_names = (char *)ndpi_realloc(NULL, flow->protos.tls_quic.server_names, combined_length);

  if (combined_server_names != NULL) {
    flow->protos.tls_quic.server_names = combined_server_names;
    strcat(flow->protos.tls_quic.server_names, ",");
    strcat(flow->protos.tls_quic.server_names, dNSName);
    flow->protos.tls_quic.server_names_len = combined_length - 1; // Exclude the null terminator from length.
  } else {
    // Handle memory allocation failure gracefully
    ndpi_set_risk(ndpi_struct, flow, NDPI_POSSIBLE_MEMORY_ISSUE, "Failed to allocate memory for server names");
  }
}
