if(flow->protos.tls_quic.server_names == NULL) {
  flow->protos.tls_quic.server_names = ndpi_strdup(dNSName);
  flow->protos.tls_quic.server_names_len = dns_name_length;
} else {
  char *new_server_names = (char*)ndpi_malloc(flow->protos.tls_quic.server_names_len + dns_name_length + 2);
  if(new_server_names) {
    strncpy(new_server_names, flow->protos.tls_quic.server_names, flow->protos.tls_quic.server_names_len);
    new_server_names[flow->protos.tls_quic.server_names_len] = ',';
    strncpy(&new_server_names[flow->protos.tls_quic.server_names_len + 1], dNSName, dns_name_length + 1);
    ndpi_free(flow->protos.tls_quic.server_names);
    flow->protos.tls_quic.server_names = new_server_names;
    flow->protos.tls_quic.server_names_len += dns_name_length + 1;
  }
}
