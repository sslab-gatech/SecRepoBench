// Extract and store the "username" and "password" from the IMAP LOGIN command.
// Copy the packet payload into a temporary string buffer.
char temp_str[flow_packet->payload_packet_len + 1];
memcpy(temp_str, flow_packet->payload, flow_packet->payload_packet_len);
temp_str[flow_packet->payload_packet_len] = '\0';

// Locate the first double quote character, marking the start of the username.
u_int16_t username_start = 0;
while (username_start < flow_packet->payload_packet_len && temp_str[username_start]!= '"') {
  username_start++;
}

// Find the next double quote to identify the end of the username and store it.
u_int16_t username_end = username_start + 1;
while (username_end < flow_packet->payload_packet_len && temp_str[username_end]!= '"') {
  username_end++;
}

// Extract the username and store it within the flow structure.
if (username_start < username_end) {
  memcpy(flow->protos.ftp_imap_pop_smtp.username, &temp_str[username_start + 1], username_end - username_start - 1);
  flow->protos.ftp_imap_pop_smtp.username[username_end - username_start - 1] = '\0';
}

// Locate the next pair of double quotes to extract the password and store it.
u_int16_t password_start = username_end + 1;
while (password_start < flow_packet->payload_packet_len && temp_str[password_start]!= '"') {
  password_start++;
}

u_int16_t password_end = password_start + 1;
while (password_end < flow_packet->payload_packet_len && temp_str[password_end]!= '"') {
  password_end++;
}

// Extract the password and store it within the flow structure.
if (password_start < password_end) {
  memcpy(flow->protos.ftp_imap_pop_smtp.password, &temp_str[password_start + 1], password_end - password_start - 1);
  flow->protos.ftp_imap_pop_smtp.password[password_end - password_start - 1] = '\0';
}
