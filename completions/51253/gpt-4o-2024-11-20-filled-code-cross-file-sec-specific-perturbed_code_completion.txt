for (i = KC; i < (rounds + 1) * 4; i++) {
        uint8_t temp[4];
        memcpy(temp, a->round_key[i - 1].u8, 4);

        if (i % KC == 0) {
            // Rotate the word
            uint8_t t = temp[0];
            temp[0] = temp[1];
            temp[1] = temp[2];
            temp[2] = temp[3];
            temp[3] = t;

            // Apply S-box
            temp[0] = sbox[temp[0]];
            temp[1] = sbox[temp[1]];
            temp[2] = sbox[temp[2]];
            temp[3] = sbox[temp[3]];

            // XOR with Rcon
            temp[0] ^= rcon[rconpointer++];
        } else if (KC > 6 && i % KC == 4) {
            // Apply S-box
            temp[0] = sbox[temp[0]];
            temp[1] = sbox[temp[1]];
            temp[2] = sbox[temp[2]];
            temp[3] = sbox[temp[3]];
        }

        // Generate the round key
        for (t = 0; t < 4; t++) {
            a->round_key[i].u8[t] = a->round_key[i - KC].u8[t] ^ temp[t];
        }
    }

    if (decrypt) {
        // Reverse the round keys for decryption
        for (i = 0; i < rounds + 1; i++) {
            if (i && i != rounds) {
                inv_mix_column(a->round_key[i].u32, dec_multbl);
            }
        }
    }
