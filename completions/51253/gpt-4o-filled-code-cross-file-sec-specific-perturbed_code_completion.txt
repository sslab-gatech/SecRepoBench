for (i = KC; i < (rounds + 1) * 4; i++) {
        uint8_t temp[4];
        memcpy(temp, a->round_key[i - 1].u8, 4);

        if (i % KC == 0) {
            // Rotate the 4 bytes in a word to the left
            uint8_t t = temp[0];
            temp[0] = temp[1];
            temp[1] = temp[2];
            temp[2] = temp[3];
            temp[3] = t;

            // Apply S-box to all 4 bytes
            temp[0] = sbox[temp[0]];
            temp[1] = sbox[temp[1]];
            temp[2] = sbox[temp[2]];
            temp[3] = sbox[temp[3]];

            // XOR the output of the rcon operation with the first byte of the word
            temp[0] ^= rcon[rconpointer++];
        } else if (KC > 6 && i % KC == 4) {
            // Apply S-box to all 4 bytes
            temp[0] = sbox[temp[0]];
            temp[1] = sbox[temp[1]];
            temp[2] = sbox[temp[2]];
            temp[3] = sbox[temp[3]];
        }

        a->round_key[i].u32 = a->round_key[i - KC].u32 ^ AV_RB32(temp);
    }

    if (decrypt) {
