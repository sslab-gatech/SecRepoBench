VALUE_UNISTR:
      if(payload_len < 4)
        return 0;

      u_int32_t size = ntohl(get_u_int32_t(databuffer, 0));
      v->value.ptr.raw = databuffer + 4;
      
      if(size == 0 || size > payload_len - 4)
        return 0;

      v->value_size = size;
      ret = size + 4;

      if(t == VALUE_DATA)
        v->value_size *= 2;

      break;
