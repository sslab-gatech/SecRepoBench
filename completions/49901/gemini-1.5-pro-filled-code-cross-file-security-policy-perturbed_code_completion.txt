VALUE_UNISTR:
      v->value.ptr.value_unistr = (u_int8_t *)databuffer;
      v->value_size = ntohs(get_u_int16_t(databuffer, 0));
      v->value.ptr.value_unistr += 2;

      if(v->value_size == 0 || v->value_size > payload_len - 2)
	return 0;

      ret = v->value_size + 2;
      if(t == VALUE_DATA)
	v->value_size = ret;

      break;
