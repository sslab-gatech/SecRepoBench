  if (payload_len < 2)

  uint16_t field_len;
  memcpy(&field_len, databuffer, sizeof(field_len));
  field_len = ntohs(field_len);

  if (field_len == 0 || (size_t) field_len > payload_len - 2)

  v->value_size = field_len;
  ret = 2 + field_len;
