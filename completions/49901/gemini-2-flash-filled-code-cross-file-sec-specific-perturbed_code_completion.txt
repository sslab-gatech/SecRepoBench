VALUE_UNISTR:
      if(payload_len < 2)
	return 0;

      v->value.ptr.value_size = ntohs(get_u_int16_t(databuffer, 0));

      if(v->value.ptr.value_size > (payload_len - 2))
	return 0;

      v->value.ptr.value_str = (char*)databuffer + 2;
      v->value_size = v->value.ptr.value_size;
      ret = v->value_size + 2;
      break;
