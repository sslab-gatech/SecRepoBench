VALUE_UNISTR:
      if(payload_len < 2)
	return 0;

      v->value.ptr.value_unistr_size = ntohs(get_u_int16_t(databuffer, 0));
      v->value.ptr.value_unistr = (u_int8_t const *)databuffer + 2;

      if(v->value.ptr.value_unistr_size == 0 || v->value.ptr.value_unistr_size > payload_len - 2)
	return 0;

      if(t == VALUE_DATA)
	v->value_size = v->value.ptr.value_unistr_size;
      ret = v->value.ptr.value_unistr_size + 2;
      break;
