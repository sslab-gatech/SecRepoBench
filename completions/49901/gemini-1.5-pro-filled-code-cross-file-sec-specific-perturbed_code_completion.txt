VALUE_UNISTR:
      v->value.ptr.value_unistr = (u_int16_t *)databuffer;
      v->value_size = ntohs(v->value.ptr.value_unistr[0]) * 2 + 2;
      ret = v->value_size;

      if (v->value_size > payload_len || v->value_size == 0) {
          return 0;
      }

      if (t == VALUE_DATA) {
          v->value.ptr.value_str = (char const *)(v->value.ptr.value_unistr + 1);
          v->value_size -= 2;
      }
      break;
