if (m->type == FLB_CONFIG_MAP_STR) {
    entry->val.str = flb_sds_create(kv->val);
    if (!entry->val.str) {
        flb_errno();
        flb_free(entry);
        return -1;
    }
}
else if (m->type == FLB_CONFIG_MAP_STR_PREFIX) {
    if (kv->val) {
        flb_error("[config map] invalid value for prefixed string '%s=%s'",
                  m->name, kv->val);
        flb_free(entry);
        return -1;
    }
}
else if (m->type == FLB_CONFIG_MAP_BOOL) {
    ret = flb_utils_bool(kv->val);
    if (ret == -1) {
        flb_error("[config map] invalid boolean property '%s=%s'",
                  m->name, kv->val);
        flb_free(entry);
        return -1;
    }
    entry->val.boolean = ret;
}
else if (m->type == FLB_CONFIG_MAP_INT) {
    entry->val.i_num = atoi(kv->val);
}
else if (m->type == FLB_CONFIG_MAP_DOUBLE) {
    entry->val.d_num = atof(kv->val);
}
else if (m->type == FLB_CONFIG_MAP_SIZE) {
    entry->val.s_num = flb_utils_size_to_bytes(kv->val);
}
else if (m->type == FLB_CONFIG_MAP_TIME) {
    entry->val.s_num = flb_utils_time_to_seconds(kv->val);
}
else if (m->type >= FLB_CONFIG_MAP_CLIST && m->type <= FLB_CONFIG_MAP_SLIST_4) {
    struct mk_list *list;

    list = parse_string_map_to_list(m, kv->val);
    if (!list) {
        flb_error("[config map] cannot parse list of values '%s'", kv->val);
        flb_free(entry);
        return -1;
    }

    ret = check_list_size(list, m->type);
    if (ret < 0) {
        flb_error("[config map] not enough entries for '%s'", m->name);
        flb_slist_destroy(list);
        flb_free(list);
        flb_free(entry);
        return -1;
    }

    entry->val.list = list;
}
else {
    flb_error("[config map] invalid type for multiple property '%s'",
              m->name);
    flb_free(entry);
    return -1;
}
