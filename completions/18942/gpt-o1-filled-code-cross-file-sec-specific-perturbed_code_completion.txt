{
  /*
   * Roll back all loops and release any resources associated with them.
   * This ensures that if an error has occurred in an inner loop, all
   * resources are correctly freed before abandoning the loop parsing.
   */
  while (compiler->loop_depth > 0)
  {
    compiler->loop_depth--;
    loop_vars_cleanup(compiler->loop_depth);
  }

  /*
   * Re-initialize loop-related variables to safe defaults,
   * avoiding any uninitialized or invalid references.
   */
  compiler->loop_for_of_var_index = -1;

  yyerrok;
  yyclearin;

  /*
   * Signal a parse error so that Bison recovers or aborts parse as needed.
   */
  YYERROR;
}
