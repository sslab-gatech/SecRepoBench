while (compiler->loop_depth > 0)
{
    // Clean up any loop variables associated with the current loop level.
    loop_vars_cleanup(compiler->loop_depth - 1);
    // Decrement the loop depth so that outer loop contexts (if any)
    // will be cleaned up in subsequent iterations.
    compiler->loop_depth--;
}
