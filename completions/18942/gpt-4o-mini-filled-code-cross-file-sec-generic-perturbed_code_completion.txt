// Clean up loop variables for the current loop context
        YR_LOOP_CONTEXT* loop_ctx = &compiler->loop[compiler->loop_depth];
        for (int i = 0; i < loop_ctx->vars_count; i++)
        {
            // Free any resources associated with the loop variables
            yr_free(loop_ctx->vars[i].identifier);
            loop_ctx->vars[i].identifier = NULL;
        }
        loop_ctx->vars_count = 0; // Reset the variable count

        // Reset the loop depth to zero
        compiler->loop_depth = 0;

        // Initialize the frame for local variables in the current loop context
        int var_frame = _yr_compiler_get_var_frame(compiler);
        for (int i = 0; i < YR_MAX_LOOP_VARS; i++)
        {
            // Clear the variable frame for the current loop context
            yr_parser_emit_with_arg(yyscanner, OP_CLEAR_M, var_frame + i, NULL, NULL);
        }
