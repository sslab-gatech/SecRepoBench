{
        int parseresult = ERROR_SUCCESS;
        int loop_depth = compiler->loop_depth;

        loop_vars_cleanup(loop_depth);

        if (loop_depth > 0)
        {
          // If an error occurs within an inner loop, the cleanup will
          // propagate to the outer loops. This ensures that all resources
          // associated with loop variables are properly released before
          // handling the error.
          compiler->loop_depth = 0;
        }

        fail_if_error(parseresult);

        (yyval.expression).type = EXPRESSION_TYPE_BOOLEAN;
