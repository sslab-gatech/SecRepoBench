if (is_generational(gc)) {
  /* disable generational mode to reset the state */
  gc->generational = FALSE;
  /* sweep the heap to find dead objects */
  incremental_sweep_phase(mrbstateptr, gc, MRB_HEAP_PAGE_SIZE);
  /* re-enable generational mode to continue with the generational garbage collection process */
  gc->generational = TRUE;
}
