// Disable generational mode temporarily
gc->generational = FALSE;

// Reset all objects to white color (next GC target)
RVALUE *p = gc->heap.start;
while (p) {
  if (!is_free(&p->as.basic)) {
    paint_white(gc, &p->as.basic); // Mark object as white
  }
  p = next_object(p);
}

// Perform incremental sweep phase to process objects
incremental_sweep_phase(mrbstateptr, gc, SIZE_MAX);

// Re-enable generational mode
gc->generational = TRUE;
