gc->gray_list = NULL; // Reset gray list
  gc->atomic_gray_list = NULL; // Reset atomic gray list
  incremental_sweep_phase(mrbstateptr, gc, MRB_HEAP_PAGE_SIZE); // Perform incremental sweep
  gc->generational = FALSE; // Temporarily disable generational mode
  incremental_gc_finish(mrbstateptr, gc); // Finish the incremental GC
  gc->generational = TRUE; // Re-enable generational mode
