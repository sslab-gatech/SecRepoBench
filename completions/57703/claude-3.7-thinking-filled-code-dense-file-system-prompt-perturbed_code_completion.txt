/* Temporarily disable generational mode */
gc->generational = FALSE;

/* Mark all objects as white (young) for the next collection */
gc->state = MRB_GC_STATE_ROOT;

/* Process all objects */
incremental_sweep(mrbstateptr, gc);
incremental_gc_finish(mrbstateptr, gc);

/* Re-enable generational mode */
gc->generational = TRUE;
