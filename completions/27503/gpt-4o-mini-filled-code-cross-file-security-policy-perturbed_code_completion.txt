// Decode audio data based on the element type specified by the tag
		switch (tag)
		{
			case ID_SCE: // Mono channel
				// Ensure we do not exceed the number of channels
				RequireAction (channelIndex < numChannels, status = kALAC_ParamError; goto Exit;);
				status = DecodeMono(p, bits, sampleBuffer, numSamples, channelIndex, outNumSamples);
				RequireNoErr(status, goto Exit);
				channelIndex++;
				break;

			case ID_CPE: // Stereo channel pair
				// Ensure we do not exceed the number of channels
				RequireAction (channelIndex + 1 < numChannels, status = kALAC_ParamError; goto Exit;);
				status = DecodeStereo(p, bits, sampleBuffer, numSamples, channelIndex, outNumSamples);
				RequireNoErr(status, goto Exit);
				channelIndex += 2;
				break;

			case ID_LFE: // Low-Frequency Effects channel
				// Ensure we do not exceed the number of channels
				RequireAction (channelIndex < numChannels, status = kALAC_ParamError; goto Exit;);
				status = DecodeMono(p, bits, sampleBuffer, numSamples, channelIndex, outNumSamples);
				RequireNoErr(status, goto Exit);
				channelIndex++;
				break;

			default:
				status = kALAC_UnsupportedElement;
				goto Exit;
		}

		// Check if we have processed all requested samples
		if (channelIndex >= numChannels)
		{
			break; // Exit the loop if all channels have been processed
		}
