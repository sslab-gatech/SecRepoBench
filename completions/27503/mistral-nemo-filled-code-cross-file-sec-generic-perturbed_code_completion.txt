switch (tag)
{
	case ID_SCE:
		// Decode mono channel
		BitBufferReadSmall (bits, 4); // Read element instance tag
		DecodeMono (p, bits, sampleBuffer, channelIndex, numSamples);
		channelIndex++;
		break;
	case ID_CPE:
		// Decode stereo channel pair
		BitBufferReadSmall (bits, 4); // Read element instance tag
		DecodeStereo (p, bits, sampleBuffer, channelIndex, numSamples);
		channelIndex += 2;
		break;
	default:
		// Unsupported element, ignore and continue
		BitBufferAlign (bits);
		break;
}
