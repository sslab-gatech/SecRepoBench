switch (tag)
		{
			case ID_SCE: // Mono channel
				BitBufferReadSmall(bits, 4); // Read element instance tag
				status = DecodeMono(p, bits, sampleBuffer, numSamples, &val);
				RequireNoErr(status, goto Exit);
				break;

			case ID_CPE: // Stereo channel
				BitBufferReadSmall(bits, 4); // Read element instance tag
				status = DecodeStereo(p, bits, sampleBuffer, numSamples, &val);
				RequireNoErr(status, goto Exit);
				break;

			case ID_LFE: // LFE channel
				BitBufferReadSmall(bits, 4); // Read element instance tag
				status = DecodeMono(p, bits, sampleBuffer, numSamples, &val);
				RequireNoErr(status, goto Exit);
				break;

			default:
				status = kALAC_UnsupportedElement;
				goto Exit;
		}

		channelIndex++;
		if (channelIndex >= numChannels) {
			break; // Exit if all channels have been processed
		}
