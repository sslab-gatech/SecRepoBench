switch (tag)
{
	case ID_SCE:
		// Decode mono audio data
		BitBufferReadSmall (bits, 4) ; // Read element instance tag
		out32 = sampleBuffer + channelIndex ;
		status = dyn_decomp (&agParams, bits, out32, numSamples, 32, &bits1) ;
		channelIndex++ ;
		break ;

	case ID_CPE:
		// Decode stereo channel pair audio data
		BitBufferReadSmall (bits, 4) ; // Read element instance tag
		out32 = sampleBuffer + channelIndex ;
		status = dyn_decomp (&agParams, bits, out32, numSamples, 32, &bits1) ;
		channelIndex += 2 ;
		break ;

	default:
		// Unsupported element, ignore and continue
		BitBufferAlign (bits) ;
		break ;
}
