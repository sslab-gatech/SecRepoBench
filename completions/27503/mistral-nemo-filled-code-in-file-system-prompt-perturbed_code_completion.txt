switch (tag)
{
	case kALACElement_Header:
		alac_header_element (bits, p) ;
		break ;
	case kALACElement_Frame:
		alac_frame_element (bits, p) ;
		break ;
	case kALACElement_Fill:
		alac_fill_element (bits) ;
		break ;
	case kALACElement_DataStream:
		alac_data_stream_element (bits) ;
		break ;
	default:
		// unsupported element
		status = kALAC_UnsupportedOperation ;
		break ;
}
