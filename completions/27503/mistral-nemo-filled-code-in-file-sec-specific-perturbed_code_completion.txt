switch (tag)
{
	case kALACElement_Header:
		alac_header_element (bits, p) ;
		break ;
	case kALACElement_Filler:
		alac_fill_element (bits) ;
		break ;
	case kALACElement_Uncompressed:
		alac_uncompressed_element (bits, p, sampleBuffer, numSamples, &channelIndex) ;
		break ;
	case kALACElement_Compressed:
		alac_compressed_element (bits, p, sampleBuffer, numSamples, &channelIndex) ;
		break ;
	case kALACElement_Escape:
		alac_escape_element (bits, p, sampleBuffer, numSamples, &channelIndex) ;
		break ;
	case kALACElement_ChannelStream:
		alac_channel_stream_element (bits, p, sampleBuffer, numSamples, &channelIndex) ;
		break ;
	case kALACElement_ChannelStreamEnd:
		alac_channel_stream_end_element (bits, p, sampleBuffer, numSamples, &channelIndex) ;
		break ;
	case kALACElement_ChannelStreamEndWithData:
		alac_channel_stream_end_with_data_element (bits, p, sampleBuffer, numSamples, &channelIndex) ;
		break ;
	case kALACElement_ChannelStreamEndWithDataAndHeader:
		alac_channel_stream_end_with_data_and_header_element (bits, p, sampleBuffer, numSamples, &channelIndex) ;
		break ;
	case kALACElement_ChannelStreamEndWithDataAndHeaderAndFiller:
		alac_channel_stream_end_with_data_and_header_and_filler_element (bits, p, sampleBuffer, numSamples, &channelIndex) ;
		break ;
	case kALACElement_ChannelStreamEndWithDataAndHeaderAndFillerAndEscape:
		alac_channel_stream_end_with_data_and_header_and_filler_and_escape_element (bits, p, sampleBuffer, numSamples, &channelIndex) ;
		break ;
	case kALACElement_ChannelStreamEndWithDataAndHeaderAndFillerAndEscapeAndFill:
		alac_channel_stream_end_with_data_and_header_and_filler_and_escape_and_fill_element (bits, p, sampleBuffer, numSamples, &channelIndex) ;
		break ;
	case kALACElement_ChannelStreamEndWithDataAndHeaderAndFillerAndEscapeAndFillAndDataStream:
		alac_channel_stream_end_with_data_and_header_and_filler_and_escape_and_fill_and_data_stream_element (bits, p, sampleBuffer, numSamples, &channelIndex) ;
		break ;
	default:
		// unsupported element, ignore it
		BitBufferAdvance (bits, 8) ;
		break ;
}
