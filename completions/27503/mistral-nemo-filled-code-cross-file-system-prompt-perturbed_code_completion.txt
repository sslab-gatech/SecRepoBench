switch (tag)
{
	case ID_SCE:
		// Decode mono audio data
		BitBufferReadSmall (bits, 4) ; // Read element instance tag
		DecodeMono (bits, sampleBuffer + channelIndex, numSamples);
		channelIndex++;
		break;
	case ID_CPE:
		// Decode stereo audio data
		BitBufferReadSmall (bits, 4) ; // Read element instance tag
		DecodeStereo (bits, sampleBuffer + channelIndex, numSamples);
		channelIndex += 2;
		break;
	default:
		// Unsupported element, ignore and continue
		break;
}
