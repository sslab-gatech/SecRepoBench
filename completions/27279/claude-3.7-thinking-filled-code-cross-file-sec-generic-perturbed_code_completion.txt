/* Check proxy data if any */
if (proxy_address) {
    /* Duplicate proxy_address */
    c->proxy.host = flb_malloc(strlen(proxy_address) + 1);
    if (!c->proxy.host) {
        flb_errno();
        flb_http_client_destroy(c);
        return NULL;
    }
    strcpy((char *)c->proxy.host, proxy_address);
    
    /* Parse proxy address */
    char *p = strchr((char *)c->proxy.host, ':');
    if (p) {
        *p = '\0';
        p++;
        c->proxy.port = atoi(p);
    }
    else {
        c->proxy.port = 80;
    }
}

/* Response data buffer */
c->resp.data = flb_malloc(FLB_HTTP_DATA_SIZE_MAX);
