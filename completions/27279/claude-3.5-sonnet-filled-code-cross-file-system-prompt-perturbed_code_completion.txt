/* Add Host header */
    flb_http_add_header(c, "Host", 4, c->host, strlen(c->host));

    /* Add Content-Length header */
    if (c->body_len > 0) {
        flb_http_add_header(c, "Content-Length", 14,
                            c->body_buf, c->body_len);
    }

    /* Parse and set proxy data if specified */
    if (proxy_address) {
        c->proxy.host = flb_strdup(proxy_address);
        if (!c->proxy.host) {
            flb_errno();
            flb_http_client_destroy(c);
            return NULL;
        }
    }

    /* Allocate response data buffer */
    c->resp.data = flb_calloc(1, FLB_HTTP_DATA_SIZE_MAX);
