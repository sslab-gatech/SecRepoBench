/* Parse proxy address if specified */
    if (proxy_address) {
        char *proxy_host = NULL;
        int proxy_port = 0;

        /* Parse the proxy address into host and port */
        ret = flb_parse_proxy(proxy_address, &proxy_host, &proxy_port);
        if (ret == -1) {
            flb_error("[http_client] failed to parse proxy address: %s", proxy_address);
            flb_http_client_destroy(c);
            return NULL;
        }

        /* Update the client structure with proxy information */
        c->proxy.host = proxy_host;
        c->proxy.port = proxy_port;
    }

    /* Allocate memory for the response data buffer */
    c->resp.data = flb_calloc(1, FLB_HTTP_DATA_SIZE_MAX);
