if (proxy_address) {
    ret = proxy_parse(proxy_address, c);
    if (ret == -1) {
        flb_error("[http_client] cannot parse proxy address %s", proxy_address);
        return -1;
    }

    /* Add host and content length to the HTTP client structure */
    ret = add_host_and_content_length(c);
    if (ret == -1) {
        flb_error("[http_client] cannot add host and content length to HTTP client");
        return -1;
    }
}
