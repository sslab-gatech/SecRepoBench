/* Allocate memory for the response data buffer */
    c->resp.data = flb_calloc(1, FLB_HTTP_DATA_SIZE_MAX);
    if (!c->resp.data) {
        flb_errno();
        flb_http_client_destroy(c);
        return NULL;
    }

    /* If a proxy is specified, parse its data */
    if (proxy_address) {
        c->proxy.host = flb_strdup(proxy_address);
        if (!c->proxy.host) {
            flb_errno();
            flb_http_client_destroy(c);
            return NULL;
        }
        c->proxy.port = port;
    }
