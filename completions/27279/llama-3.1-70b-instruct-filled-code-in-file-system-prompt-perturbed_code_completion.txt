if (proxy_address) {
    ret = proxy_parse(proxy_address, c);
    if (ret!= 0) {
        flb_error("[http_client] cannot parse proxy address %s", proxy_address);
        flb_free(c->header_buf);
        flb_free(c);
        return NULL;
    }
}

ret = add_host_and_content_length(c);
if (ret!= 0) {
    flb_error("[http_client] cannot add host and content length headers");
    flb_free(c->header_buf);
    flb_free(c);
    return NULL;
}

c->resp.data = flb_malloc(FLB_HTTP_DATA_SIZE_MAX);
if (!c->resp.data) {
    flb_errno();
    flb_http_client_destroy(c);
    return NULL;
}
