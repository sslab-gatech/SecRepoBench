import numpy as np

def fill_masked_region(image, mask, color):
    """
    Fill the masked region of an image with a given color.

    :param image: The input image as a numpy array.
    :param mask: The binary mask indicating the region to fill.
    :param color: The color to fill the masked region with.
    :return: The modified image with the masked region filled.
    """
    # Get the color channel of the image
    color_channel = image[:,:,color]

    # Get the indices of the masked region
    indices = np.where(mask)

    # Fill the masked region with the given color
    color_channel[indices] = color

    # Update the image with the modified color channel
    image[:,:,color] = color_channel

    return image
