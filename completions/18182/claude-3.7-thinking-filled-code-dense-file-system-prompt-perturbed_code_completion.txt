(void) ReadProfile(img,"xmp",profile,(ssize_t) length,exception);
if (length > 19)  // Minimum length to contain "dc:format="image/dng"
{
  register const char *p;
  
  for (p=(const char *) profile; p < ((const char *) profile+length-18); p++)
  {
    if (LocaleNCompare(p,"dc:format=\"image/dng\"",19) == 0)
    {
      (void) CopyMagickString(img->magick,"DNG",MagickPathExtent);
      break;
    }
  }
}
