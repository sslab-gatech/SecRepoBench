(void) ReadProfile(img,"xmp",profile,(ssize_t) length,exception);
{
  char *tempProfile=(char *) AcquireQuantumMemory(length+1,sizeof(*tempProfile));
  if (tempProfile != (char *) NULL)
  {
    (void) memcpy(tempProfile,profile,length);
    tempProfile[length]='\0';
    if (strstr(tempProfile,"dc:format=\"image/dng\"") != (char *) NULL)
      (void) CopyMagickString(img->magick,"DNG",MagickPathExtent);
    tempProfile=DestroyString(tempProfile);
  }
}
