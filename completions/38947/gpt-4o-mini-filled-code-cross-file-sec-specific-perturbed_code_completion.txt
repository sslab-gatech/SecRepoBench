if (elementIter == root.children.end()) {
        throw DeadlyImportError("LWS: Invalid LWS file structure, no elements found");
    }

    // Retrieve the version from the first element
    const std::string &versionToken = (*elementIter).tokens[0];
    if (versionToken.empty()) {
        throw DeadlyImportError("LWS: Invalid version token found");
    }

    // Convert version token to unsigned integer
    version = strtoul10(versionToken.c_str());
    if (version == 0) {
        throw DeadlyImportError("LWS: Invalid version number");
