++elementIter;
if (elementIter == root.children.end()) {
    ASSIMP_LOG_ERROR("LWS: Invalid LWS file format, missing version information.");
    throw DeadlyImportError("LWS: Invalid LWS file format, missing version information.");
}

unsigned int version = strtoul10((*elementIter).tokens[0].c_str());
if (version < 1 || version > 4) {
    ASSIMP_LOG_ERROR("LWS: Unsupported LWS file format version: ", version);
    throw DeadlyImportError("LWS: Unsupported LWS file format version: ", version);
}
ASSIMP_LOG_INFO("LWS file format version is ", version);
