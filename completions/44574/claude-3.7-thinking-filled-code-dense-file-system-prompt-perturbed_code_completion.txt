else if (!strncmp("null", str_val, str_len)) {
    primitive_value = plist_new_null();
} else {
    // Try to parse as a number
    char* number_str = (char*)malloc(str_len + 1);
    if (!number_str) {
        PLIST_JSON_ERR("%s: Out of memory\n", __func__);
        return NULL;
    }
    memcpy(number_str, str_val, str_len);
    number_str[str_len] = '\0';
    
    char* endptr = NULL;
    
    // Try to parse as an integer first
    int64_t int_val = strtoll(number_str, &endptr, 10);
    
    if (endptr == number_str + str_len) {
        // Successfully parsed as an integer
        if (int_val < 0) {
            primitive_value = plist_new_int(int_val);
        } else {
            primitive_value = plist_new_uint((uint64_t)int_val);
        }
    } else {
        // Try to parse as a floating-point number
        endptr = NULL;
        double real_val = strtod(number_str, &endptr);
        
        if (endptr == number_str + str_len) {
            // Successfully parsed as a floating-point number
            primitive_value = plist_new_real(real_val);
        } else {
            // Failed to parse as a number
            PLIST_JSON_ERR("%s: Failed to parse primitive value\n", __func__);
            free(number_str);
            return NULL;
        }
    }
    
    free(number_str);
}
