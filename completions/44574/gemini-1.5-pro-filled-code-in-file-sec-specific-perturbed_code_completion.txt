if (!strncmp("null", str_val, str_len)) {
        primitive_value = plist_new_null();
    } else if (isdigit(str_val[0]) || (str_val[0] == '-' && isdigit(str_val[1]))) {
        char *endp = NULL;
        int64_t intval = parse_decimal(str_val, str_end, &endp);

        if (endp == str_end) {
            primitive_value = plist_new_uint(intval);
        } else if (*endp == '.' || *endp == 'e' || *endp == 'E') {
            double realval = strtod(str_val, &endp);
            if (endp == str_end) {
                primitive_value = plist_new_real(realval);
            } else {
                PLIST_JSON_ERR("%s: invalid number '%.*s'\n", __func__, (int)str_len, str_val);
            }
        } else {
            PLIST_JSON_ERR("%s: invalid number '%.*s'\n", __func__, (int)str_len, str_val);
        }
    } else {
        PLIST_JSON_ERR("%s: invalid primitive '%.*s'\n", __func__, (int)str_len, str_val);
    }
