if (!strncmp("null", str_val, str_len)) {
    primitive_value = plist_new_null();
} else {
    char* endptr = NULL;
    int64_t int_val = parse_decimal(str_val, str_end, &endptr);
    if (endptr == str_end) {
        if (int_val == INT64_MAX || int_val == INT64_MIN) {
            primitive_value = plist_new_uint(int_val);
        } else {
            primitive_value = plist_new_uint(int_val);
        }
    } else {
        double real_val = strtod(str_val, &endptr);
        if (endptr == str_end) {
            primitive_value = plist_new_real(real_val);
        } else {
            PLIST_JSON_ERR("%s: invalid primitive value '%.*s'\n", __func__, (int)str_len, str_val);
            return NULL;
        }
    }
}
