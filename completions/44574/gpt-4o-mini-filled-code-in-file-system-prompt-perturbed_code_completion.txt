if (str_len == 0) {
        PLIST_JSON_ERR("%s: empty primitive value\n", __func__);
        return NULL;
    }

    if (str_len == 4 && !strncmp(str_val, "null", 4)) {
        primitive_value = plist_new_null();
    } else if (str_len == 1 && (str_val[0] == '0' || str_val[0] == '1')) {
        primitive_value = plist_new_bool(str_val[0] - '0');
    } else {
        char* endptr;
        int64_t int_value = parse_decimal(str_val, str_end, &endptr);
        if (endptr != str_end) {
            // If not a valid integer, try parsing as a double
            double double_value = strtod(str_val, &endptr);
            if (endptr == str_end) {
                primitive_value = plist_new_real(double_value);
            } else {
                PLIST_JSON_ERR("%s: invalid primitive value '%.*s'\n", __func__, (int)str_len, str_val);
                return NULL;
            }
        } else {
            primitive_value = plist_new_int(int_value);
        }
