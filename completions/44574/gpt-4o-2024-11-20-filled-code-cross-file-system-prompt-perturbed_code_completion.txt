if (!strncmp("null", str_val, str_len)) {
        primitive_value = plist_new_null();
    } else {
        char* endptr = NULL;
        double num_val = strtod(str_val, &endptr);
        if (endptr == str_end) {
            if (strchr(str_val, '.') || strchr(str_val, 'e') || strchr(str_val, 'E')) {
                primitive_value = plist_new_real(num_val);
            } else {
                long long int_val = strtoll(str_val, &endptr, 10);
                if (endptr == str_end) {
                    primitive_value = plist_new_int(int_val);
                } else {
                    PLIST_JSON_ERR("%s: failed to parse integer value\n", __func__);
                }
            }
        } else {
            PLIST_JSON_ERR("%s: failed to parse primitive value\n", __func__);
        }
    }
