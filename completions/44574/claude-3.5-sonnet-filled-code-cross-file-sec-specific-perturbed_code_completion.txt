if (!strncmp("null", str_val, str_len)) {
    primitive_value = plist_new_null();
} else {
    char* endptr = NULL;
    long long int int_val = strtoll(str_val, &endptr, 10);
    if (endptr == str_end) {
        primitive_value = plist_new_uint(int_val);
    } else {
        double float_val = strtod(str_val, &endptr);
        if (endptr == str_end) {
            primitive_value = plist_new_real(float_val);
        } else {
            PLIST_JSON_ERR("%s: invalid primitive value\n", __func__);
            return NULL;
        }
    }
}
