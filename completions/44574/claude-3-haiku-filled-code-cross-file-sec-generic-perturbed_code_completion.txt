// Check if the primitive value is "null" and create a corresponding plist node for it.
if (!strncmp("null", str_val, str_len)) {
    primitive_value = plist_new_null();
}
// If the value represents a number, parse it as an integer or floating point number.
else if (isdigit(str_val[0]) || (str_val[0] == '-' && isdigit(str_val[1]))) {
    char* endptr;
    long long int int_val = strtoll(str_val, &endptr, 10);
    if (endptr == str_val) {
        PLIST_JSON_ERR("%s: failed to parse integer value\n", __func__);
        return NULL;
    }
    if (*endptr == '\0') {
        primitive_value = plist_new_uint64(int_val);
    } else {
        double float_val = strtod(str_val, &endptr);
        if (endptr == str_val) {
            PLIST_JSON_ERR("%s: failed to parse floating-point value\n", __func__);
            return NULL;
        }
        primitive_value = plist_new_real(float_val);
    }
}
// If the value is an integer, create a plist node for it.
// If the value is a floating-point number, parse the fractional and exponential parts
// and create a plist node for the real number.
else {
    PLIST_JSON_ERR("%s: unsupported primitive value type\n", __func__);
    return NULL;
}
(*index)++;
return primitive_value;
