if (strncmp("null", str_val, str_len) == 0) {
        primitive_value = plist_new_null();
    } else {
        char* endptr;
        int64_t int_value = parse_decimal(str_val, str_end, &endptr);
        if (endptr != str_val) {
            primitive_value = plist_new_uint(int_value);
        } else {
            double real_value = strtod(str_val, &endptr);
            if (endptr != str_val) {
                primitive_value = plist_new_real(real_value);
            } else {
                PLIST_JSON_ERR("%s: invalid primitive value '%.*s'\n", __func__, (int)str_len, str_val);
                return NULL;
            }
        }
    }
