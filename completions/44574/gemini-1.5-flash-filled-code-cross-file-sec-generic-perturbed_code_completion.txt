else if (!strncmp("null", str_val, str_len)) {
        primitive_value = plist_new_null();
    } else {
        char* endptr;
        errno = 0;
        long long int_val = strtoll(str_val, &endptr, 10);
        if (endptr != str_end || errno == ERANGE) {
            double real_val = strtod(str_val, &endptr);
            if (endptr != str_end || errno == ERANGE) {
                PLIST_JSON_ERR("%s: failed to parse primitive value: %.*s\n", __func__, (int)str_len, str_val);
                return NULL;
            }
            primitive_value = plist_new_real(real_val);
        } else {
            primitive_value = plist_new_int(int_val);
        }
    }
