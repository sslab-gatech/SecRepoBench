} else if (!strncmp("null", str_val, str_len)) {
        primitive_value = plist_new_null();
    } else {
        errno = 0;
        long long int_val = strtoll(str_val, NULL, 10);
        if (errno != ERANGE && str_val + str_len == strchr(str_val, '\0')) {
            primitive_value = plist_new_integer(int_val);
        } else {
            double real_val;
            if (sscanf(str_val, "%lf", &real_val) == 1) {
                primitive_value = plist_new_real(real_val);
            } else {
                PLIST_JSON_ERR("%s: cannot interpret '%.*s' as number\n", __func__, (int)str_len, str_val);
            }
        }
