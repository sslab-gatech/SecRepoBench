else if (!strncmp("null", str_val, str_len)) {
        primitive_value = plist_new_null();
    } else {
        char* endptr;
        errno = 0;
        long long int_val = strtoll(str_val, &endptr, 10);
        if (endptr == str_val || errno == ERANGE) {
            double double_val = strtod(str_val, &endptr);
            if (endptr == str_val || errno == ERANGE) {
                PLIST_JSON_ERR("%s: failed to parse number\n", __func__);
                return NULL;
            }
            primitive_value = plist_new_real(double_val);
        } else {
            primitive_value = plist_new_integer(int_val);
        }
    }
