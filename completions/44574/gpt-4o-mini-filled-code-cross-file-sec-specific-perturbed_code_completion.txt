else if (!strncmp("null", str_val, str_len)) {
        primitive_value = plist_new_null();
    } else {
        char* endptr;
        errno = 0; // Reset errno before strtol/strtod
        long int_value = strtol(str_val, &endptr, 10);
        if (errno == ERANGE || *endptr != '\0' || endptr == str_val) {
            // Not a valid integer, try parsing as a double
            double float_value = strtod(str_val, &endptr);
            if (errno == ERANGE || *endptr != '\0' || endptr == str_val) {
                PLIST_JSON_ERR("%s: Unable to parse primitive value: %s\n", __func__, str_val);
                return NULL; // Parsing error
            }
            primitive_value = plist_new_real(float_value);
        } else {
            primitive_value = plist_new_int(int_value);
        }
    }
