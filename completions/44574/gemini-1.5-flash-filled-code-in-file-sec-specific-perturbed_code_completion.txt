else if (!strncmp("null", str_val, str_len)) {
        primitive_value = plist_new_null();
    } else if (isdigit(str_val[0]) || str_val[0] == '-') {
        char *endptr;
        int64_t intval = parse_decimal(str_val, str_end, &endptr);
        if (endptr == str_end) {
            primitive_value = plist_new_int(intval);
        } else if (*endptr == '.') {
            char *exp_ptr;
            double realval = strtod(str_val, &exp_ptr);
            if (exp_ptr == str_end) {
                primitive_value = plist_new_real(realval);
            } else {
                PLIST_JSON_ERR("%s: invalid number format '%.*s'\n", __func__, (int)(str_end - str_val), str_val);
            }
        } else {
            PLIST_JSON_ERR("%s: invalid number format '%.*s'\n", __func__, (int)(str_end - str_val), str_val);
        }
    } else {
        PLIST_JSON_ERR("%s: invalid primitive value '%.*s'\n", __func__, (int)(str_end - str_val), str_val);
    }
