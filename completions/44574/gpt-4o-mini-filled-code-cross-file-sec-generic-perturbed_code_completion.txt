else if (!strncmp("null", str_val, str_len)) {
        primitive_value = plist_new_null();
    } else {
        char* endptr;
        errno = 0; // Reset errno before strtol/strtod
        long int_value = strtol(str_val, &endptr, 10);
        if (endptr != str_end) {
            // Not a valid integer, try to parse as double
            double float_value = strtod(str_val, &endptr);
            if (endptr != str_end || errno == ERANGE) {
                PLIST_JSON_ERR("%s: Unable to parse primitive value: %.*s\n", __func__, (int)str_len, str_val);
                return NULL;
            }
            primitive_value = plist_new_real(float_value);
        } else if (errno == ERANGE || (int_value == LONG_MAX || int_value == LONG_MIN)) {
            PLIST_JSON_ERR("%s: Integer value out of range: %.*s\n", __func__, (int)str_len, str_val);
            return NULL;
        } else {
            primitive_value = plist_new_int(int_value);
        }
