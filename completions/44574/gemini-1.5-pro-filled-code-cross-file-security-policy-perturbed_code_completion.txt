} else if (!strncmp("null", str_val, str_len)) {
        primitive_value = plist_new_null();
    } else if (isdigit(str_val[0]) || (str_val[0] == '-' && isdigit(str_val[1]))) {
        int num_int;
        double num_real;
        char* endptr;
        if (strchr(str_val, '.') || strchr(str_val, 'e') || strchr(str_val, 'E')) {
            num_real = strtod(str_val, &endptr);
            if (endptr != str_end) {
                PLIST_JSON_ERR("%s: Cannot convert to float: %s\n", __func__, str_val);
                return NULL;
            }
            primitive_value = plist_new_real(num_real);

        } else {
            num_int = strtol(str_val, &endptr, 10);
            if (endptr != str_end) {
                PLIST_JSON_ERR("%s: Cannot convert to integer: %s\n", __func__, str_val);
                return NULL;
            }
            primitive_value = plist_new_uint(num_int);
        }
    } else {
        PLIST_JSON_ERR("%s: Unknown primitive type: %s\n", __func__, str_val);
    }
