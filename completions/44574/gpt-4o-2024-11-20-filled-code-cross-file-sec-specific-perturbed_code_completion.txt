if (!strncmp("null", str_val, str_len)) {
        primitive_value = plist_new_null();
    } else {
        char* num_str = strndup(str_val, str_len);
        if (!num_str) {
            PLIST_JSON_ERR("%s: Out of memory while duplicating string\n", __func__);
            return NULL;
        }

        char* endptr = NULL;
        errno = 0;
        long long int_val = strtoll(num_str, &endptr, 10);

        if (endptr == num_str || errno == ERANGE) {
            // Not a valid integer, try parsing as a floating-point number
            errno = 0;
            double real_val = strtod(num_str, &endptr);

            if (endptr == num_str || errno == ERANGE) {
                // Not a valid number
                PLIST_JSON_ERR("%s: Invalid primitive value: %.*s\n", __func__, (int)str_len, str_val);
                free(num_str);
                return NULL;
            } else {
                // Valid floating-point number
                primitive_value = plist_new_real(real_val);
            }
        } else {
            // Valid integer
            primitive_value = plist_new_int(int_val);
        }

        free(num_str);
    }
