if (!strncmp("null", str_val, str_len)) {
        primitive_value = plist_new_null();
    } else {
        char* num_str = strndup(str_val, str_len);
        if (!num_str) {
            PLIST_JSON_ERR("%s: Memory allocation failed\n", __func__);
            return NULL;
        }

        char* endptr = NULL;
        errno = 0;

        // Try parsing as an integer
        long long int_val = strtoll(num_str, &endptr, 10);
        if (errno == 0 && endptr == num_str + str_len) {
            primitive_value = plist_new_int(int_val);
        } else {
            // Try parsing as a floating-point number
            errno = 0;
            double real_val = strtod(num_str, &endptr);
            if (errno == 0 && endptr == num_str + str_len) {
                primitive_value = plist_new_real(real_val);
            } else {
                PLIST_JSON_ERR("%s: Failed to parse primitive value '%.*s'\n", __func__, (int)str_len, str_val);
            }
        }

        free(num_str);
    }
