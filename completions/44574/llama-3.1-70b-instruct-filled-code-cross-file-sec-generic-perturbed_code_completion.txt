// Check if the primitive value is "null" and create a corresponding plist node for it.
else if (!strncmp("null", str_val, str_len)) {
    primitive_value = plist_new_null();
} 
// If the value represents a number, parse it as an integer or floating point number.
else {
    char* endptr;
    errno = 0;
    long long_value = strtoll(str_val, &endptr, 10);
    if (errno == ERANGE) {
        PLIST_JSON_ERR("%s: integer value out of range\n", __func__);
        return NULL;
    }
    if (endptr == str_val + str_len) {
        // If the value is an integer, create a plist node for it.
        primitive_value = plist_new_int(long_value);
    } else {
        double double_value = strtod(str_val, &endptr);
        if (errno == ERANGE) {
            PLIST_JSON_ERR("%s: real value out of range\n", __func__);
            return NULL;
        }
        if (endptr == str_val + str_len) {
            // If the value is a floating-point number, parse the fractional and exponential parts
            // and create a plist node for the real number.
            primitive_value = plist_new_real(double_value);
        } else {
            PLIST_JSON_ERR("%s: invalid number format\n", __func__);
            return NULL;
        }
    }
}
