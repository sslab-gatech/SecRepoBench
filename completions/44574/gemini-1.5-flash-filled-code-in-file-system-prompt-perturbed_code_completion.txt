else if (!strncmp("null", str_val, str_len)) {
        primitive_value = plist_new_null();
    } else {
        char* endptr = NULL;
        int64_t intval = strtoll(str_val, &endptr, 10);
        if (endptr == str_val) {
            double realval = strtod(str_val, &endptr);
            if (endptr == str_val) {
                PLIST_JSON_ERR("%s: could not parse primitive value '%.*s'\n", __func__, (int)str_len, str_val);
                return NULL;
            }
            primitive_value = plist_new_real(realval);
        } else {
            primitive_value = plist_new_int(intval);
        }
    }
