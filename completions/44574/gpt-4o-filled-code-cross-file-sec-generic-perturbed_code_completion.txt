if (!strncmp("null", str_val, str_len)) {
        primitive_value = plist_new_null();
    } else {
        char* num_str = strndup(str_val, str_len);
        if (!num_str) {
            PLIST_JSON_ERR("%s: Out of memory\n", __func__);
            return NULL;
        }
        char* endptr = NULL;
        long long int_val = strtoll(num_str, &endptr, 10);
        if (endptr == num_str + str_len) {
            primitive_value = plist_new_int(int_val);
        } else {
            double real_val = strtod(num_str, &endptr);
            if (endptr == num_str + str_len) {
                primitive_value = plist_new_real(real_val);
            } else {
                PLIST_JSON_ERR("%s: Invalid primitive value\n", __func__);
            }
        }
        free(num_str);
    }
