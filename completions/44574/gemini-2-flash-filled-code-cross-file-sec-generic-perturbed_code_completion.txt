else if (!strncmp("null", str_val, str_len)) {
        primitive_value = plist_new_null();
    } else {
        char* endptr;
        long long_val = strtoll(str_val, &endptr, 10);
        if (endptr == str_end) {
            primitive_value = plist_new_integer(long_val);
        } else {
            double double_val = strtod(str_val, &endptr);
            if (endptr == str_end) {
                primitive_value = plist_new_real(double_val);
            } else {
                PLIST_JSON_ERR("%s: Could not parse primitive value: %.*s\n", __func__, (int)str_len, str_val);
                return NULL;
            }
        }
    }
