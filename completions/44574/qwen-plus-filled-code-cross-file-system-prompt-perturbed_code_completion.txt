else if (!strncmp("null", str_val, str_len)) {
    primitive_value = plist_new_null();
} else {
    char* endptr;
    long int_val = strtol(str_val, &endptr, 10);
    if (endptr != str_end) {
        // Not an integer, try floating point
        double float_val = strtod(str_val, &endptr);
        if (endptr != str_val && endptr == str_end) {
            primitive_value = plist_new_real(float_val);
        } else {
            PLIST_JSON_ERR("%s: failed to parse numeric value\n", __func__);
            return NULL;
        }
    } else {
        primitive_value = plist_new_uint(int_val);
    }
}
