int tp_size = tp->tp_end - bytestream2_tell(&decodercontext->g);
            if (tp_size < 0 || bytestream2_get_bytes_left(&decodercontext->g) < tp_size) {
                av_log(decodercontext->avctx, AV_LOG_ERROR, "Invalid tile-part size\n");
                return AVERROR_INVALIDDATA;
            }
            bytestream2_init(&decodercontext->tile_part_stream, decodercontext->g.buffer + oldpos, tp_size);
            bytestream2_skip(&decodercontext->g, tp_size);
            continue;
