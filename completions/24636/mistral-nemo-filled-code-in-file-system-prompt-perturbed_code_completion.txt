static int get_coc(Jpeg2000DecoderContext *s, int n, Jpeg2000QuantStyle *qntsty, uint8_t *properties)
{
    Jpeg2000QuantStyle tmp;
    int compno, ret;

    memset(&tmp, 0, sizeof(tmp));

    if ((ret = get_qcx(s, n, &tmp)) < 0)
        return ret;
    for (compno = 0; compno < s->ncomponents; compno++)
        if (!(properties[compno] & HAD_QCC))
            memcpy(qntsty + compno, &tmp, sizeof(tmp));
    return 0;
}
