int tp_len = tp->tp_end - decodercontext->g.buffer;
            if (tp_len < 0 || tp_len > bytestream2_get_bytes_left(&decodercontext->g)) {
                av_log(decodercontext->avctx, AV_LOG_ERROR, "Invalid tile-part length\n");
                return AVERROR_INVALIDDATA;
            }
            bytestream2_init(&decodercontext->tile_part_stream, decodercontext->g.buffer, tp_len);
            decodercontext->g.buffer += tp_len; // Move the buffer pointer to the end of the tile-part
