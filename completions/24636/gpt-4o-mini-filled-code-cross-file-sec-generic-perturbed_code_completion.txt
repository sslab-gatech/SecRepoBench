int tp_len = bytestream2_get_be16(&decodercontext->g);
            if (tp_len < 2 || bytestream2_get_bytes_left(&decodercontext->g) < tp_len - 2) {
                av_log(decodercontext->avctx, AV_LOG_ERROR, "Invalid tile-part length %d\n", tp_len);
                return AVERROR_INVALIDDATA;
            }

            // Initialize the bytestream for the current tile-part
            bytestream2_init(&decodercontext->tile_part_stream, decodercontext->g.buffer, tp_len);
            bytestream2_skip(&decodercontext->g, tp_len - 2); // Skip the tile-part header
            continue;
