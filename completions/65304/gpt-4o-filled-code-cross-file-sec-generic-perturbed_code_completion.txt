// Extract the signature algorithms from the packet payload starting from the given offset.
		ja.client.num_signature_algorithms = sa_size;
		if (ndpi_struct->enable_tls_signature_algorithms) {
		  memcpy(ja.client.signature_algorithms, &packet->payload[sessoffset], sa_size * 2);
		}

		// Convert each signature algorithm to a string format and append it to a string buffer for logging or further processing.
		for (id = 0; id < sa_size; id++) {
		  u_int16_t signature_algo = ntohs(*((u_int16_t*)&packet->payload[sessoffset + id * 2]));
		  int rc = ndpi_snprintf(&ja.client.signature_algorithms_str[id * 2], 3, "%02X", signature_algo);
		  if (rc <= 0) break;
		}
