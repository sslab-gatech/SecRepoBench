		  for (size_t sa_idx = 0; sa_idx < sa_size; ++sa_idx) {
		    if (sessoffset + (int)(sa_idx * 2) + 2 < packet->payload_packet_len) {
		      u_int16_t signature_algo = (u_int16_t)ntohs(*((u_int16_t*)&packet->payload[sessoffset+(sa_idx*2)]));
		      ja.client.signature_algorithms[ja.client.num_signature_algorithms++] = signature_algo;
		    }

		    for(i=0; i<tot_signature_algorithms_len && sessoffset + (int)i + 2 < packet->payload_packet_len; i+=2) {

		      NDPI_LOG_INFO(ndpi_struct, "Client TLS [TLS Signature Algorithm] 0x%04X\n", id);
