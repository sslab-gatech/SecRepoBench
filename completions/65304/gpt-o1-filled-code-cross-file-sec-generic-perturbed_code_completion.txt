/* Store the extracted signature algorithm in the client structure */
      if(ja.client.num_signature_algorithms < MAX_NUM_JA) {
        ja.client.signature_algorithms[ja.client.num_signature_algorithms++] = signature_algo;
      }

      /*
       * Convert each signature algorithm to a string and append it to the
       * signature_algorithms_str buffer. Ensure we don't run out of space.
       */
      {
        int rc;
        size_t curr_len = strlen(ja.client.signature_algorithms_str);
        size_t leftover = sizeof(ja.client.signature_algorithms_str) - curr_len;

        if(leftover > 0) {
          if(curr_len == 0) {
            rc = ndpi_snprintf(&ja.client.signature_algorithms_str[curr_len],
                               leftover, "%u", signature_algo);
          } else {
            rc = ndpi_snprintf(&ja.client.signature_algorithms_str[curr_len],
                               leftover, "-%u", signature_algo);
          }
          if(rc < 0) {
#ifdef DEBUG_TLS
            printf("[SIGNATURE] Error building signature_algorithms_str\n");
#endif
            ja.client.signature_algorithms_str[curr_len] = '\0';
          }
        }
      }
