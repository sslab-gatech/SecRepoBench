// Extract the signature algorithms from the packet payload starting from the given offset.
		for (i = 0; i < sa_size; i++) {
			u_int16_t signature_algo = ntohs(*((u_int16_t *)&packet->payload[sessoffset + (i * 2)]));

			// Store the signature algorithm in the client structure if space allows
			if (ja.client.num_signature_algorithms < MAX_NUM_TLS_SIGNATURE_ALGORITHMS) {
				ja.client.signature_algorithms[ja.client.num_signature_algorithms++] = signature_algo;
			} else {
				invalid_ja = 1; // Mark as invalid if the limit is exceeded
#ifdef DEBUG_TLS
				printf("Client TLS Invalid signature algorithm count %u\n", ja.client.num_signature_algorithms);
#endif
				break;
			}

			// Convert the signature algorithm to a string and append it to the buffer
			if (i < sa_size - 1) {
				int rc = ndpi_snprintf(&ja.client.signature_algorithms_str[i * 2], 
					sizeof(ja.client.signature_algorithms_str) - (i * 2), 
					"%02X-", signature_algo);
				if (rc <= 0) {
					break;
				}
			} else {
				int rc = ndpi_snprintf(&ja.client.signature_algorithms_str[i * 2], 
					sizeof(ja.client.signature_algorithms_str) - (i * 2), 
					"%02X", signature_algo);
				if (rc <= 0) {
					break;
				}
