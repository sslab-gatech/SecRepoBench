		  ja.client.signature_algorithms[i] = id;

		  // Convert each signature algorithm to a string format and append it to a string buffer
		  if (i > 0) {
		    ja.client.signature_algorithms_str[i * 2 - 1] = '-';
		  }
		  ja.client.signature_algorithms_str[i * 2] = (id >> 8) & 0xFF;
		  ja.client.signature_algorithms_str[i * 2 + 1] = id & 0xFF;

		  // Optionally, copy the signature algorithms into a dedicated structure if a specific flag is enabled
		  if (flow->protos.tls_quic.browser_heuristics.is_safari_tls) {
		    safari_signature_algorithms++;
		  } else if (flow->protos.tls_quic.browser_heuristics.is_chrome_tls) {
