// Calculate the number of signature algorithms
ja.client.num_signature_algorithms = sa_size;

// Extract the signature algorithms and convert them to string format
for (i = 0; i < tot_signature_algorithms_len && sessoffset + i < packet->payload_packet_len && i*2 + 2 <= sizeof(ja.client.signature_algorithms_str); i++) {
  // Format each byte of the signature algorithm as a 2-character hexadecimal string
  ndpi_snprintf(&ja.client.signature_algorithms_str[i*2], 
                sizeof(ja.client.signature_algorithms_str) - i*2, 
                "%02x", packet->payload[sessoffset + i] & 0xFF);
}
