		  if(ja.client.num_signature_algorithms < MAX_NUM_JA)
		    ja.client.signature_algorithms[ja.client.num_signature_algorithms++] = signature_algo;

		  id = ndpi_snprintf(&ja.client.signature_algorithms_str[i*2],
				       sizeof(ja.client.signature_algorithms_str) - (i*2),
				       "%s%04X", (i > 0) ? ":" : "", signature_algo);
		  if(id > 0) {
		    if((i*2 + id) < sizeof(ja.client.signature_algorithms_str))
		      ja.client.signature_algorithms_str[i*2 + id] = '\0';
