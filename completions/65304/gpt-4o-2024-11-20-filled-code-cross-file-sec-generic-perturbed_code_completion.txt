// Extract the signature algorithms from the packet payload starting from the given offset.
		for (i = 0; i < sa_size; i++) {
		  if (sessoffset + (i * 2) + 1 >= packet->payload_packet_len) {
		    break; // Ensure we don't read beyond the payload length
		  }

		  u_int16_t signature_algo = ntohs(*((u_int16_t *)&packet->payload[sessoffset + (i * 2)]));

		  // Store the signature algorithm in the ja.client structure
		  ja.client.signature_algorithms[i] = signature_algo;

		  // Convert the signature algorithm to a string and append it to the signature_algorithms_str
		  int rc = ndpi_snprintf(&ja.client.signature_algorithms_str[i * 2],
		                         sizeof(ja.client.signature_algorithms_str) - (i * 2),
		                         "%02X", signature_algo);

		  if (rc <= 0) {
		    break; // Stop processing if snprintf fails
		  }
		}

		// Update the number of signature algorithms found
		ja.client.num_signature_algorithms = i;
