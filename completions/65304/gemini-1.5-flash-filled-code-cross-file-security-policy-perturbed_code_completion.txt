if(ja.client.num_signature_algorithms < MAX_NUM_JA) {
		    ja.client.signature_algorithms[ja.client.num_signature_algorithms++] = signature_algo;
		  } else {
		    invalid_ja = 1;
#ifdef DEBUG_TLS
		    printf("Client TLS Invalid signature algorithms %u\n", ja.client.num_signature_algorithms);
#endif
		  }

		  id = ndpi_snprintf(&ja.client.signature_algorithms_str[ja.client.num_signature_algorithms * 2 - 2],
				     sizeof(ja.client.signature_algorithms_str) - (ja.client.num_signature_algorithms * 2 - 2),
				     "%04X", signature_algo);
		  if(id <= 0) break;
