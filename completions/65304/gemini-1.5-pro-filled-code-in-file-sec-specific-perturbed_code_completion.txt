if(sessoffset+tot_signature_algorithms_len > total_len)
		  tot_signature_algorithms_len = total_len - sessoffset;

		ja.client.num_signature_algorithms = tot_signature_algorithms_len / 2;
#ifdef HAVE_TLS_SIGNATURE_ALGORITHMS_STRUCTURE
		if (flow->protos.tls_quic.signature_algorithms == NULL) {
		  flow->protos.tls_quic.signature_algorithms = (u_int16_t*)ndpi_malloc(sa_size * sizeof(u_int16_t));
		  if (flow->protos.tls_quic.signature_algorithms != NULL) {
		    flow->protos.tls_quic.num_signature_algorithms = sa_size;
		  }
		}
#endif

		for (i = 0; i < tot_signature_algorithms_len && sessoffset + (int)i + 1 < packet->payload_packet_len; i += 2) {
		  id = ntohs(*((u_int16_t*)&packet->payload[sessoffset + i]));
#ifdef HAVE_TLS_SIGNATURE_ALGORITHMS_STRUCTURE
		  if (flow->protos.tls_quic.signature_algorithms != NULL &&
		      i / 2 < flow->protos.tls_quic.num_signature_algorithms) {
		    flow->protos.tls_quic.signature_algorithms[i / 2] = id;
		  }
#endif
		  if (i > 0) {
		    int rc = ndpi_snprintf(&ja.client.signature_algorithms_str[i * 2 - 1],
					   sizeof(ja.client.signature_algorithms_str) - i * 2 + 1,
					   "-%04X", id);
		    if (rc <= 0) break;
		  } else {
		    int rc = ndpi_snprintf(&ja.client.signature_algorithms_str[i * 2],
					   sizeof(ja.client.signature_algorithms_str) - i * 2,
					   "%04X", id);
		    if (rc <= 0) break;
		  }

		  if(ja.client.num_signature_algorithms < MAX_NUM_JA)
		    ja.client.signature_algorithms[ja.client.num_signature_algorithms++] = id;
