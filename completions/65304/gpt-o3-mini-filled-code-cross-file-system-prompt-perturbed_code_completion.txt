if(tot_signature_algorithms_len < 2 || (tot_signature_algorithms_len % 2) != 0) {
  // not enough bytes or odd length, adjust the length to be even
  tot_signature_algorithms_len = tot_signature_algorithms_len & ~0x1;
}

#ifdef DEBUG_TLS
printf("Client TLS [SIGNATURE_ALGORITHMS] Extracting %u signature algorithms\n", tot_signature_algorithms_len / 2);
#endif

// Store the number of signature algorithms and reset the signature string buffer
ja.client.num_signature_algorithms = tot_signature_algorithms_len / 2;
ja.client.signature_algorithms_str[0] = '\0';
