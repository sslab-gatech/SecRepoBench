if (attribute->ID != NULL) {
        xmlRemoveID(reader->ctxt->myDoc, attribute->ID);
	xmlFreeID(attribute->ID);
	attribute->ID = NULL;
    }
    if (attribute->ref != NULL) {
        xmlUnref((xmlRefPtr) attribute->ref);
	attribute->ref = NULL;
    }

    if (attribute->children != NULL) {
	xmlTextReaderFreeNodeList(reader, attribute->children);
	attribute->children = NULL;
	attribute->last = NULL;
    }
    DICT_FREE(attribute->name);
