if ((__xmlRegisterCallbacks) && (xmlDeregisterNodeDefaultValue))
    xmlDeregisterNodeDefaultValue((xmlNodePtr) attribute);

if (attribute->children!= NULL)
    xmlTextReaderFreeNodeList(reader, attribute->children);

if (attribute->type == XML_ATTRIBUTE_NODE) {
    if (attribute->name!= NULL) {
        if (attribute->name[0] == '_') {
            if (attribute->name[1] == 'I' && attribute->name[2] == 'D' &&
                attribute->name[3] == '_') {
                xmlTextReaderFreeIDTableEntry(attribute->children,
                                              attribute->children->name);
            } else if (attribute->name[1] == 'R' && attribute->name[2] == 'E' &&
                       attribute->name[3] == 'F' && attribute->name[4] == '_') {
                xmlTextReaderFreeIDTableEntry(attribute->children,
                                              attribute->children->name);
            }
        }
    }
}

if (attribute->name!= NULL)
    DICT_FREE(attribute->name);

if (attribute->children!= NULL)
    attribute->children = NULL;

if (attribute->properties!= NULL)
    xmlTextReaderFreePropList(reader, attribute->properties);
