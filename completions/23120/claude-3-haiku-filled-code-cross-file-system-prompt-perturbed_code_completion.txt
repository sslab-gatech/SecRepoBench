// Handle deregistration of the node if callback functions are enabled.
// Check if the property is an ID or a reference within a document and
// remove it from the document's ID or reference table if applicable.
// If the current property node has children, recursively free the node list.
// Free the memory associated with the property name using the dictionary.
if (attribute->atype == XML_ATTRIBUTE_ID) {
    if ((reader->ctxt != NULL) && (reader->ctxt->id != NULL)) {
        xmlHashRemoveEntry(reader->ctxt->id, attribute->name, xmlTextReaderFreeIDTableEntry);
    }
} else if (attribute->atype == XML_ATTRIBUTE_IDREF || attribute->atype == XML_ATTRIBUTE_IDREFS) {
    if ((reader->ctxt != NULL) && (reader->ctxt->idref != NULL)) {
        xmlHashRemoveEntry(reader->ctxt->idref, attribute->name, xmlTextReaderFreeIDTableEntry);
    }
}

if (attribute->children != NULL) {
    xmlTextReaderFreeNodeList(reader, attribute->children);
}

DICT_FREE(attribute->name);
