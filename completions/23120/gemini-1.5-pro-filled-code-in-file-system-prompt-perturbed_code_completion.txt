if ((reader != NULL) && (reader->ctxt != NULL) &&
	(reader->ctxt->disableSAX != 0))
	return;

    if (attribute->doc != NULL && attribute->parent != NULL &&
        attribute->parent->doc != NULL) {
        if (attribute->parent->type == XML_ELEMENT_NODE) {
            if (xmlIsID(attribute->doc, attribute->parent, attribute))
                xmlTextReaderRemoveID(attribute->doc, attribute);
            else if (xmlIsRef(attribute->doc, attribute->parent, attribute))
                xmlTextReaderRemoveRef(attribute->doc, attribute);
        }
    }
    if ((attribute->children != NULL) &&
	(attribute->type != XML_ENTITY_REF_NODE)) {
	xmlTextReaderFreeNodeList(reader, attribute->children);
	attribute->children = NULL;
	attribute->last = NULL;
    }
    DICT_FREE(attribute->name);
