if (attribute->ID != NULL) {
        xmlRemoveID(reader->ctxt->myDoc, attribute->ID);
	xmlFreeID(attribute->ID);
	attribute->ID = NULL;
    }
    if (attribute->atype == XML_ATTRIBUTE_IDREFS) {
	xmlRemoveRef(reader->ctxt->myDoc, attribute->children);
    } else if (attribute->atype == XML_ATTRIBUTE_IDREF) {
	xmlRemoveRef(reader->ctxt->myDoc, attribute->children);
    }
    if (attribute->children != NULL) {
	xmlTextReaderFreeNodeList(reader, attribute->children);
	attribute->children = NULL;
	attribute->last = NULL;
    }
    if (dict != NULL) {
#ifdef LIBXML_SCHEMAS_ENABLED
        if (attribute->ns != NULL)
	    xmlDictRemove(dict, attribute->name);
	else
#endif
	    xmlDictRemove(dict, attribute->name);
    }
