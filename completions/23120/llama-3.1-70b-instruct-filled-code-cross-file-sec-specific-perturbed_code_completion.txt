if ((__xmlRegisterCallbacks) && (xmlDeregisterNodeDefaultValue))
    xmlDeregisterNodeDefaultValue((xmlNodePtr) attribute);

if (attribute->children!= NULL)
    xmlTextReaderFreeNodeList(reader, attribute->children);

if (attribute->type == XML_ID) {
    if (reader->ctxt->idTable!= NULL)
        xmlFreeIDTableEntry(reader->ctxt->idTable, attribute->name);
} else if (attribute->type == XML_REF) {
    if (reader->ctxt->refTable!= NULL)
        xmlFreeRefTableEntry(reader->ctxt->refTable, attribute->name);
}

DICT_FREE(attribute->name);
DICT_FREE(attribute->children);
DICT_FREE(attribute->content);
