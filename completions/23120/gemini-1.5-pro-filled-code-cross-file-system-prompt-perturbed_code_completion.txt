if (attribute->ID != NULL) {
        xmlRemoveID(reader->ctxt, attribute->ID);
	xmlFreeID(attribute->ID);
	attribute->ID = NULL;
    }
    if (attribute->xref != NULL) {
        xmlRemoveRef(reader->ctxt, attribute->xref);
	xmlFreeRef(attribute->xref);
	attribute->xref = NULL;
    }

    if (attribute->children != NULL) {
	xmlTextReaderFreeNodeList(reader, attribute->children);
	attribute->children = attribute->last = NULL;
    }
    DICT_FREE(attribute->name);
