            (buf[i + 2] == '-') && (buf[i + 3] == '-'))
        {
            incomment = 1;
            i += 3;
            continue;
        }

        if (ignoreAttrValue && !incomment && !invalue &&
            ((buf[i] == '"') || (buf[i] == '\'')))
        {
            invalue = 1;
            valdellim = buf[i];
            continue;
        }

        if (ignoreAttrValue && !incomment && invalue && (buf[i] == valdellim))

            invalue = 0;
            continue;

        {

            {
                incomment = 0;
                i += 2;
            }
            continue;
        }

        if (buf[i] == first)
        {
            if (next == 0)
            {
                ctxt->checkIndex = base + i + 1;
                return (base + i);
            }
            else if (buf[i + 1] == next)
            {
                if (third == 0)
                {
                    ctxt->checkIndex = base + i + 2;
                    return (base + i);
                }

                    ctxt->checkIndex = base + i + 3;
