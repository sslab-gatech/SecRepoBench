int idx, i;
    const xmlChar *buf;
    int len;

    in = ctxt->input;
    if (in == NULL)
        return (-1);

    base = in->cur - in->base;
    if (base < 0)
        return (-1);

    if (ctxt->checkIndex > base)
        base = ctxt->checkIndex;

    if (in->buf == NULL) {
        buf = in->base;
        len = in->length;
    } else {
        buf = xmlBufContent(in->buf->buffer);
        len = xmlBufUse(in->buf->buffer);
    }

    for (idx = base; idx < len; idx++) {
        if (!incomment && (idx + 4 < len)) {
            if ((buf[idx] == '<') && (buf[idx + 1] == '!') &&
                (buf[idx + 2] == '-') && (buf[idx + 3] == '-')) {
                incomment = 1;
                idx += 2;
                continue;
            }
        }
        if (incomment) {
            if (idx + 3 > len)
                return (-1);
            if ((buf[idx] == '-') && (buf[idx + 1] == '-') &&
                (buf[idx + 2] == '>')) {
                incomment = 0;
                idx += 2;
            }
            continue;
        }
        if (ignoreAttrValue && !invalue) {
            if (buf[idx] == '"' || buf[idx] == '\'') {
                invalue = 1;
                valdellim = buf[idx];
                continue;
            }
        }
        if (ignoreAttrValue && invalue) {
            if (buf[idx] == valdellim) {
                invalue = 0;
                valdellim = 0x0;
            }
            continue;
        }
        if (buf[idx] == first) {
            if (next == 0)
                return (idx - (in->cur - in->base));
            if (idx + 1 < len) {
                if (buf[idx + 1] == next) {
                    if (third == 0)
                        return (idx - (in->cur - in->base));
                    if (idx + 2 < len) {
                        if (buf[idx + 2] == third)
                            return (idx - (in->cur - in->base));
                    }
                }
            }
        }
    }
    ctxt->checkIndex = idx;
