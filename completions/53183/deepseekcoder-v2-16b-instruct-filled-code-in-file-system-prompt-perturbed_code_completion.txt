static const char*
get_num(mrb_state *mrb, const char *p, const char *end, int *value_pointer)
{
  int sign = 1;
  int num = 0;

  if (*p == '-') {
    sign = -1;
    p++;
  }

  while (p < end && isdigit(*p)) {
    num = num * 10 + (*p - '0');
    p++;
  }

  *value_pointer = sign * num;
  return p;
}
