  }

  // Check for optional sign
  if (e < end && (*e == '-' || *e == '+')) {
    if (*e == '-') {
      sign = -1;
    }
    e++;
  }

  // Parse digits
  if (e < end && isdigit((unsigned char)*e)) {
    while (e < end && isdigit((unsigned char)*e)) {
      int digit = *e - '0';

      // Check for overflow
      if (value > (INT_MAX - digit) / 10) {
        return NULL; // Overflow occurred
      }

      value = value * 10 + digit;
      e++;
    }
  } else {
    return NULL; // No digits found
  }

  // Apply sign
  value *= sign;

  // Store the parsed value
  *value_pointer = value;
