int base = 10;
  const char *e = p;
  int sign = 1;
  int value = 0;

  if (p < end && *p == '-') {
    sign = -1;
    ++p;
  } else if (p < end && *p == '+') {
    ++p;
  }

  if (p == end) return NULL;

  while (p < end && *p >= '0' && *p <= '9') {
    int digit = *p - '0';
    if (value > (INT_MAX - digit) / base) {
      return NULL;
    }
    value = value * base + digit;
    ++p;
  }

  *value_pointer = sign * value;
  return p;
