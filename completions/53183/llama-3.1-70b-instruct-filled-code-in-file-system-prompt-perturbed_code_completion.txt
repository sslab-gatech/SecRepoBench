const char*
get_num(mrb_state *mrb, const char *p, const char *end, int *value_pointer)
{
  int value = 0;
  int sign = 1;

  if (*p == '-') {
    sign = -1;
    p++;
  }
  else if (*p == '+') {
    p++;
  }

  while (p < end && *p >= '0' && *p <= '9') {
    value = value * 10 + (*p - '0');
    p++;
  }

  *value_pointer = sign * value;
  return p;
}
