static const char*
get_num(mrb_state *mrb, const char *p, const char *end, int *value_pointer)
{
  int value = 0;
  while (p < end && isdigit((unsigned char)*p)) {
    int digit = *p - '0';
    if (value > (INT_MAX - digit) / 10) {
      return NULL; // Integer overflow
    }
    value = value * 10 + digit;
    p++;
  }
  *value_pointer = value;
  return p;
}
