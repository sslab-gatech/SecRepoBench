def pack_char(mrb, o, str, sidx, flags)
  ridx = sidx
  ridx += pack_char(mrb, o, str, ridx, flags)
  ridx
end

def unpack_char(mrb, src, srclen, ary, flags)
  CHECK_UNPACK_LEN(mrb, srclen, ary)

  mrb_value dst

  const char *e = p + srclen

  if (flags & PACK_FLAG_SIGNED)
    mrb_ary_push(mrb, ary, mrb_fixnum_value(*(signed char *)p))
  else

  return 1
end
