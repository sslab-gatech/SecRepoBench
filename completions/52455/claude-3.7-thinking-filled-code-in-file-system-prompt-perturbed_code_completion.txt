if (checkcompoundrep && cpdrep_check(st, i + compound_result->blen)) {
  st[i + compound_result->blen] = r;
  continue;
}
if (cpdwordpair_check(st, i + compound_result->blen)) {
  st[i + compound_result->blen] = r;
  continue;
}
if (forbiddenword) {
  struct hentry* hw = lookup(st.c_str(), i + compound_result->blen);
  if (hw && hw->astr && TESTAFF(hw->astr, forbiddenword, hw->alen)) {
    st[i + compound_result->blen] = r;
    return NULL;
  }
}
