struct file_info *file_info;
		int data_length;

		while (skip_size > 0) {
			file_info = NULL;
			if (next_cache_entry(archivereader, iso9660, &file_info) != ARCHIVE_OK) {
				break;
			}

			data_length = (int)(file_info->size);
			if (data_length < 0 || data_length > skip_size) {
				archive_set_error(&archivereader->archive, ARCHIVE_ERRNO_MISC,
				    "Invalid file size for entry: %s", file_info->name.s);
				return (ARCHIVE_FATAL);
			}

			// Allocate memory for the file_info structure
			struct file_info *new_file_info = malloc(sizeof(struct file_info));
			if (new_file_info == NULL) {
				archive_set_error(&archivereader->archive, ENOMEM,
				    "Memory allocation failed for file_info");
				return (ARCHIVE_FATAL);
			}
			memcpy(new_file_info, file_info, sizeof(struct file_info));

			// Add the new file_info to the multi extent list
			new_file_info->next = multi;
			multi = new_file_info;

			skip_size -= data_length;
		}
