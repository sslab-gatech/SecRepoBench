default:
  switch (method)
  {
    case ReadStripMethod:
    {
      /*
        Read strip-method image.
      */
      if (AcquireMagickResource(WidthResource,rows_per_strip) == MagickFalse)
        ThrowTIFFException(ImageError,"WidthExceedsLimit");
      for (y=0; y < (ssize_t) image->rows; y+=rows_per_strip)
      {
        Quantum
          *magick_restrict q;

        ssize_t
          strip;

        strip=(ssize_t) (y/rows_per_strip);
        if (TIFFReadPixels(tiff,strip,(char *) pixels) == -1)
          break;
        q=QueueAuthenticPixels(image,0,y,image->columns,MagickMin(rows_per_strip,image->rows-y),exception);
        if (q == (Quantum *) NULL)
          break;
        ImportQuantumPixels(image,quantum_info,quantum_type,(const unsigned char *) pixels,q,exception);
        if (SyncAuthenticPixels(image,exception) == MagickFalse)
          break;
        if (image->previous == (Image *) NULL)
        {
          status=SetImageProgress(image,LoadImageTag,(MagickOffsetType) y,
            image->rows);
          if (status == MagickFalse)
            break;
        }
      }
      break;
    }
    case ReadTileMethod:
    {
      /*
        Read tiled image.
      */
      uint32
        columns,
        rows,
        tile_width,
        tile_length;

      if ((TIFFGetField(tiff,TIFFTAG_TILEWIDTH,&columns) != 1) ||
          (TIFFGetField(tiff,TIFFTAG_TILELENGTH,&rows) != 1))
        ThrowTIFFException(CoderError,"ImageIsNotTiled");
      tile_width=(uint32) image->columns;
      tile_length=(uint32) image->rows;
      if (AcquireMagickResource(WidthResource,tile_width) == MagickFalse)
        ThrowTIFFException(ImageError,"WidthExceedsLimit");
      if (AcquireMagickResource(HeightResource,tile_length) == MagickFalse)
        ThrowTIFFException(ImageError,"HeightExceedsLimit");
      if (TIFFGetField(tiff,TIFFTAG_TILEWIDTH,&tile_width) == 1)
        if (AcquireMagickResource(WidthResource,tile_width) == MagickFalse)
          ThrowTIFFException(ImageError,"WidthExceedsLimit");
      if (TIFFGetField(tiff,TIFFTAG_TILELENGTH,&tile_length) == 1)
        if (AcquireMagickResource(HeightResource,tile_length) == MagickFalse)
          ThrowTIFFException(ImageError,"HeightExceedsLimit");
      for (y=0; y < (ssize_t) image->rows; y+=tile_length)
      {
        Quantum
          *magick_restrict q;

        ssize_t
          tile;

        tile=(ssize_t) (y/tile_length);
        if (TIFFReadTile(tiff,(char *) pixels,tile,0,y) == -1)
          break;
        q=QueueAuthenticPixels(image,0,y,image->columns,MagickMin(tile_length,image->rows-y),exception);
        if (q == (Quantum *) NULL)
          break;
        ImportQuantumPixels(image,quantum_info,quantum_type,(const unsigned char *) pixels,q,exception);
        if (SyncAuthenticPixels(image,exception) == MagickFalse)
          break;
        if (image->previous == (Image *) NULL)
        {
          status=SetImageProgress(image,LoadImageTag,(MagickOffsetType) y,
            image->rows);
          if (status == MagickFalse)
            break;
        }
      }
      break;
    }
  }
  break;
}
