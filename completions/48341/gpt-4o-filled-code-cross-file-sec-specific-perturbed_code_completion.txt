case ReadStripMethod:
      {
        /*
          Convert TIFF image by strips.
        */
        for (y=0; y < (ssize_t) image->rows; y+=rows_per_strip)
        {
          tmsize_t
            strip_size;

          strip_size=TIFFReadEncodedStrip(tiff,TIFFComputeStrip(tiff,y,0),
            pixels,scanline_size*rows_per_strip);
          if (strip_size == -1)
            break;
          for (i=0; i < (ssize_t) rows_per_strip; i++)
          {
            Quantum
              *magick_restrict q;

            ssize_t
              x;

            unsigned char
              *p;

            if ((y+i) >= (ssize_t) image->rows)
              break;
            q=QueueAuthenticPixels(image,0,y+i,image->columns,1,exception);
            if (q == (Quantum *) NULL)
              break;
            p=pixels+i*scanline_size;
            (void) ImportQuantumPixels(image,(CacheView *) NULL,quantum_info,
              quantum_type,p,exception);
            if (SyncAuthenticPixels(image,exception) == MagickFalse)
              break;
          }
          if (image->previous == (Image *) NULL)
            {
              status=SetImageProgress(image,LoadImageTag,(MagickOffsetType) y,
                image->rows);
              if (status == MagickFalse)
                break;
            }
        }
        break;
      }
      case ReadTileMethod:
      {
        /*
          Convert TIFF image by tiles.
        */
        uint32
          columns,
          rows;

        if ((TIFFGetField(tiff,TIFFTAG_TILEWIDTH,&columns) != 1) ||
            (TIFFGetField(tiff,TIFFTAG_TILELENGTH,&rows) != 1))
          ThrowTIFFException(CoderError,"ImageIsNotTiled");
        for (y=0; y < (ssize_t) image->rows; y+=rows)
        {
          for (x=0; x < (ssize_t) image->columns; x+=columns)
          {
            tmsize_t
              tile_size;

            tile_size=TIFFReadTile(tiff,pixels,x,y,0,0);
            if (tile_size == -1)
              break;
            for (i=0; i < (ssize_t) rows; i++)
            {
              Quantum
                *magick_restrict q;

              ssize_t
                x_offset;

              unsigned char
                *p;

              if ((y+i) >= (ssize_t) image->rows)
                break;
              q=QueueAuthenticPixels(image,x,y+i,columns,1,exception);
              if (q == (Quantum *) NULL)
                break;
              p=pixels+i*scanline_size;
              (void) ImportQuantumPixels(image,(CacheView *) NULL,quantum_info,
                quantum_type,p,exception);
              if (SyncAuthenticPixels(image,exception) == MagickFalse)
                break;
            }
          }
          if (image->previous == (Image *) NULL)
            {
              status=SetImageProgress(image,LoadImageTag,(MagickOffsetType) y,
                image->rows);
              if (status == MagickFalse)
                break;
            }
        }
        break;
      }
      case ReadGenericMethod:
      default:
      {
        /*
          Convert TIFF image.
        */
        for (y=0; y < (ssize_t) image->rows; y++)
        {
          Quantum
            *magick_restrict q;

          ssize_t
            x;

          unsigned char
            *p;

          tiff_status=TIFFReadPixels(tiff,0,y,(char *) pixels);
          if (tiff_status == -1)
            break;
          q=QueueAuthenticPixels(image,0,y,image->columns,1,exception);
          if (q == (Quantum *) NULL)
            break;
          p=pixels;
          (void) ImportQuantumPixels(image,(CacheView *) NULL,quantum_info,
            quantum_type,p,exception);
          if (SyncAuthenticPixels(image,exception) == MagickFalse)
            break;
          if (image->previous == (Image *) NULL)
            {
              status=SetImageProgress(image,LoadImageTag,(MagickOffsetType) y,
                image->rows);
              if (status == MagickFalse)
                break;
            }
        }
        break;
      }
