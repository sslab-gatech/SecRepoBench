case ReadStripMethod:
      {
        for (y=0; y < (ssize_t) image->rows; y+=(ssize_t) rows_per_strip)
        {
          Quantum
            *magick_restrict q;

          ssize_t
            strip_y;

          strip_y=y;
          tiff_status=TIFFReadEncodedStrip(tiff,y,(char *) pixels,scanline_size*
            MagickMin((size_t) rows_per_strip,image->rows-y));
          if (tiff_status == -1)
            break;
          q=QueueAuthenticPixels(image,0,strip_y,image->columns,
            MagickMin((size_t) rows_per_strip,image->rows-y),exception);
          if (q == (Quantum *) NULL)
            break;
          ImportQuantumPixels(image,(CacheView *) NULL,quantum_info,quantum_type,
            pixels,exception);
          if (SyncAuthenticPixels(image,exception) == MagickFalse)
            break;
          if (image->previous == (Image *) NULL)
            {
              status=SetImageProgress(image,LoadImageTag,(MagickOffsetType) y,
                image->rows);
              if (status == MagickFalse)
                break;
            }
        }
        break;
      }
      case ReadTileMethod:
      {
        uint32
          columns,
          rows;

        TIFFGetField(tiff,TIFFTAG_TILEWIDTH,&columns);
        TIFFGetField(tiff,TIFFTAG_TILELENGTH,&rows);
        for (y=0; y < (ssize_t) image->rows; y+=(ssize_t) rows)
        {
          ssize_t
            x;

          for (x=0; x < (ssize_t) image->columns; x+=(ssize_t) columns)
          {
            Quantum
              *magick_restrict q;

            tiff_status=TIFFReadEncodedTile(tiff,y*image->columns+x,(char *) pixels,
              (tcc!=NULL)?scanline_size:0);
            if (tiff_status == -1)
              break;
            q=QueueAuthenticPixels(image,x,y,columns,rows,exception);
            if (q == (Quantum *) NULL)
              break;
            ImportQuantumPixels(image,(CacheView *) NULL,quantum_info,quantum_type,
              pixels,exception);
            if (SyncAuthenticPixels(image,exception) == MagickFalse)
              break;
            if (image->previous == (Image *) NULL)
              {
                status=SetImageProgress(image,LoadImageTag,(MagickOffsetType) y,
                  image->rows);
                if (status == MagickFalse)
                  break;
              }
          }
        }
        break;
      }
      case ReadGenericMethod:
      default:
      {
        status=TIFFReadRGBAImage(tiff,image->columns,image->rows,(uint32 *) pixels,
          image->matte != MagickFalse);
        if (status == 0)
          break;
        for (y=0; y < (ssize_t) image->rows; y++)
        {
          Quantum
            *magick_restrict q;

          q=QueueAuthenticPixels(image,0,y,image->columns,1,exception);
          if (q == (Quantum *) NULL)
            break;
          ImportQuantumPixels(image,(CacheView *) NULL,quantum_info,quantum_type,
            pixels+(y*image->columns),exception);
          if (SyncAuthenticPixels(image,exception) == MagickFalse)
            break;
          if (image->previous == (Image *) NULL)
            {
              status=SetImageProgress(image,LoadImageTag,(MagickOffsetType) y,
                image->rows);
              if (status == MagickFalse)
                break;
            }
        }
        break;
      }
