case ReadStripMethod:
      {
        /*
          Convert TIFF image to pixel packets.
        */
        for (y=0; y < (ssize_t) image->rows; y++)
        {
          Quantum
            *magick_restrict q;

          ssize_t
            x;

          unsigned char
            *p;

          tiff_status=TIFFReadScanline(tiff,pixels,y);
          if (tiff_status == -1)
            break;
          q=QueueAuthenticPixels(image,0,y,image->columns,1,exception);
          if (q == (Quantum *) NULL)
            break;
          p=pixels;
          for (x=0; x < (ssize_t) image->columns; x++)
          {
            ImportQuantumPixels(image,quantum_info,quantum_type,p,q,exception);
            p+=quantum_info->pad;
            q+=GetPixelChannels(image);
          }
          if (SyncAuthenticPixels(image,exception) == MagickFalse)
            break;
          if (image->previous == (Image *) NULL)
            {
              status=SetImageProgress(image,LoadImageTag,(MagickOffsetType) y,
                image->rows);
              if (status == MagickFalse)
                break;
            }
        }
        break;
      }
      case ReadTileMethod:
      {
        /*
          Convert tiled TIFF image to pixel packets.
        */
        for (y=0; y < (ssize_t) image->rows; y+=rows_per_strip)
        {
          Quantum
            *magick_restrict q;

          size_t
            columns,
            rows;

          ssize_t
            x;

          unsigned char
            *p;

          uint32
            tile_height,
            tile_width;

          tiff_status=TIFFGetField(tiff,TIFFTAG_TILEWIDTH,&tile_width);
          tiff_status&=TIFFGetField(tiff,TIFFTAG_TILELENGTH,&tile_height);
          if (tiff_status != 1)
            break;
          columns=tile_width;
          rows=tile_height;
          if (y+rows > (ssize_t) image->rows)
            rows=(size_t) (image->rows-y);
          q=QueueAuthenticPixels(image,0,y,image->columns,rows,exception);
          if (q == (Quantum *) NULL)
            break;
          p=pixels;
          for (x=0; x < (ssize_t) image->columns; x+=columns)
          {
            size_t
              tile_columns,
              tile_rows;

            tiff_status=TIFFReadTile(tiff,pixels,x,y,0,0);
            if (tiff_status == -1)
              break;
            tile_columns=columns;
            tile_rows=rows;
            if (x+columns > (ssize_t) image->columns)
              tile_columns=(size_t) (image->columns-x);
            ImportQuantumPixels(image,quantum_info,quantum_type,p,q,exception);
            p+=quantum_info->pad;
            q+=GetPixelChannels(image)*tile_columns;
          }
          if (SyncAuthenticPixels(image,exception) == MagickFalse)
            break;
          if (image->previous == (Image *) NULL)
            {
              status=SetImageProgress(image,LoadImageTag,(MagickOffsetType) y,
                image->rows);
              if (status == MagickFalse)
                break;
            }
        }
        break;
      }
      case ReadGenericMethod:
      default:
      {
        /*
          Convert TIFF image to pixel packets.
        */
        if (TIFFReadRGBAImage(tiff,image->columns,image->rows,pixels,0) == 0)
          ThrowTIFFException(CorruptImageError,"UnableToReadImageData");
        ImportQuantumPixels(image,quantum_info,quantum_type,pixels,
          QueueAuthenticPixels(image,0,0,image->columns,image->rows,exception),
          exception);
        if (SyncAuthenticPixels(image,exception) == MagickFalse)
          break;
        break;
      }
    }
