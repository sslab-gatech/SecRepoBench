case ReadStripMethod:
      {
        /*
          Convert stripped TIFF image.
        */
        for (y=0; y < (ssize_t) image->rows; y++)
        {
          Quantum
            *magick_restrict q;

          tiff_status=TIFFReadPixels(tiff,0,y,(char *) pixels);
          if (tiff_status == -1)
            break;
          q=QueueAuthenticPixels(image,0,y,image->columns,1,exception);
          if (q == (Quantum *) NULL)
            break;
          (void) ImportQuantumPixels(image,(CacheView *) NULL,quantum_info,
            quantum_type,pixels,exception);
          if (SyncAuthenticPixels(image,exception) == MagickFalse)
            break;
          if (image->previous == (Image *) NULL)
            {
              status=SetImageProgress(image,LoadImageTag,(MagickOffsetType) y,
                image->rows);
              if (status == MagickFalse)
                break;
            }
        }
        break;
      }
      case ReadTileMethod:
      {
        /*
          Convert tiled TIFF image.
        */
        for (y=0; y < (ssize_t) image->rows; y+=tiff_info.tile_geometry.height)
        {
          Quantum
            *magick_restrict q,
            *magick_restrict tile_pixels;

          size_t
            rows;

          ssize_t
            x;

          rows=tiff_info.tile_geometry.height;
          if ((y+rows) > image->rows)
            rows=image->rows-y;
          for (x=0; x < (ssize_t) image->columns; x+=tiff_info.tile_geometry.width)
          {
            size_t
              columns;

            columns=tiff_info.tile_geometry.width;
            if ((x+columns) > image->columns)
              columns=image->columns-x;
            tiff_status=TIFFReadTile(tiff,pixels,(uint32) x,(uint32) y,0,0);
            if (tiff_status == -1)
              break;
            q=QueueAuthenticPixels(image,x,y,columns,rows,exception);
            if (q == (Quantum *) NULL)
              break;
            tile_pixels=pixels+(rows-tiff_info.tile_geometry.height)*
              TIFFTileRowSize(tiff)+(columns-tiff_info.tile_geometry.width)*
              ((ssize_t) samples_per_pixel*((bits_per_sample+7) >> 3));
            (void) ImportQuantumPixels(image,(CacheView *) NULL,quantum_info,
              quantum_type,tile_pixels,exception);
            if (SyncAuthenticPixels(image,exception) == MagickFalse)
              break;
          }
          if (tiff_status == -1)
            break;
          if (image->previous == (Image *) NULL)
            {
              status=SetImageProgress(image,LoadImageTag,(MagickOffsetType) y,
                image->rows);
              if (status == MagickFalse)
                break;
            }
        }
        break;
      }
      case ReadGenericMethod:
      default:
      {
        status=TIFFReadRGBAImage(tiff,(uint32) image->columns,(uint32)
          image->rows,(uint32 *) pixels,0);
        if (status == 0)
          break;
        for (y=0; y < (ssize_t) image->rows; y++)
        {
          Quantum
            *magick_restrict q;

          q=QueueAuthenticPixels(image,0,y,image->columns,1,exception);
          if (q == (Quantum *) NULL)
            break;
          (void) ImportQuantumPixels(image,(CacheView *) NULL,quantum_info,
            quantum_type,pixels+y*image->columns*sizeof(uint32),exception);
          if (SyncAuthenticPixels(image,exception) == MagickFalse)
            break;
          if (image->previous == (Image *) NULL)
            {
              status=SetImageProgress(image,LoadImageTag,(MagickOffsetType) y,
                image->rows);
              if (status == MagickFalse)
                break;
            }
        }
        break;
      }
