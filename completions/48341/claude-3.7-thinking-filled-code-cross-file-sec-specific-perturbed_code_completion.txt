case ReadStripMethod:
{
  /*
    Read TIFF image as strips.
  */
  size_t
    extent,
    length;

  ssize_t
    strip_id,
    strips_per_image;

  strips_per_image=(ssize_t) TIFFNumberOfStrips(tiff);
  for (y=0; y < (ssize_t) image->rows; y+=(ssize_t) rows_per_strip)
  {
    strip_id=TIFFComputeStrip(tiff,(uint32) y,0);
    extent=(size_t) TIFFStripSize(tiff);
    if (((strips_per_image > 0) && (strip_id >= strips_per_image)) ||
        (extent > number_pixels))
      break;
    if (TIFFReadEncodedStrip(tiff,(uint32) strip_id,pixels,(size_t) extent) == -1)
      break;
    rows_per_strip=(uint32) MagickMin((ssize_t) rows_per_strip,
      (ssize_t) image->rows-y);
    if ((image->previous == (Image *) NULL) && (rows_per_strip >= 8))
      {
        status=SetImageProgress(image,LoadImageTag,(MagickOffsetType) y,
          image->rows);
        if (status == MagickFalse)
          break;
      }
    length=ImportQuantumPixels(image,(CacheView *) NULL,quantum_info,
      quantum_type,pixels,exception);
    if (length == 0)
      break;
    if (SyncAuthenticPixels(image,exception) == MagickFalse)
      break;
  }
  break;
}
case ReadTileMethod:
{
  /*
    Read TIFF image as tiles.
  */
  uint32
    tile_height,
    tile_width;

  if (TIFFGetField(tiff,TIFFTAG_TILEWIDTH,&tile_width) == 0)
    ThrowTIFFException(CoderError,"ImageIsNotTiled");
  if (TIFFGetField(tiff,TIFFTAG_TILELENGTH,&tile_height) == 0)
    ThrowTIFFException(CoderError,"ImageIsNotTiled");
  for (y=0; y < (ssize_t) image->rows; y+=(ssize_t) tile_height)
  {
    ssize_t
      tile_y;

    tile_y=y;
    for (x=0; x < (ssize_t) image->columns; x+=(ssize_t) tile_width)
    {
      ssize_t
        tile_x,
        tiles_per_row;

      size_t
        extent;

      tile_x=x;
      extent=(size_t) TIFFTileSize(tiff);
      tiles_per_row=(ssize_t) (image->columns+tile_width-1)/tile_width;
      if ((size_t) (((tiles_per_row+1)*(y/tile_height)+
          (x/tile_width)+1)*TIFFTileSize(tiff)) > number_pixels)
        break;
      if (TIFFReadTile(tiff,pixels,(uint32) tile_x,(uint32) tile_y,0,0) == -1)
        break;
      if ((image->previous == (Image *) NULL) && (((tile_y/tile_height) % 8) == 0) &&
          (((tile_x/tile_width) % 8) == 0))
        {
          status=SetImageProgress(image,LoadImageTag,(MagickOffsetType)
            (tile_y*image->columns+tile_x),image->rows*image->columns);
          if (status == MagickFalse)
            break;
        }
      (void) ImportQuantumPixels(image,(CacheView *) NULL,quantum_info,
        quantum_type,pixels,exception);
      if (SyncAuthenticPixels(image,exception) == MagickFalse)
        break;
    }
    if (x < (ssize_t) image->columns)
      break;
  }
  break;
}
case ReadGenericMethod:
default:
{
  /*
    Read TIFF image with TIFFReadRGBAImage.
  */
  size_t
    pad;

  uint32
    *p;

  pad=(size_t) (image->columns*sizeof(uint32));
  if (pad > number_pixels)
    ThrowTIFFException(ResourceLimitError,"MemoryAllocationFailed");
  if (TIFFReadRGBAImage(tiff,(uint32) image->columns,(uint32) image->rows,
      (uint32 *) pixels,0) == 0)
    break;
  /*
    Convert image to DirectClass MIFF format.
  */
  p=(uint32 *) pixels;
  for (y=0; y < (ssize_t) image->rows; y++)
  {
    ssize_t
      x;

    Quantum
      *q;

    q=QueueAuthenticPixels(image,0,(ssize_t) (image->rows-y-1),
      image->columns,1,exception);
    if (q == (Quantum *) NULL)
      break;
    for (x=0; x < (ssize_t) image->columns; x++)
    {
      SetPixelRed(image,ScaleCharToQuantum(TIFFGetR(*p)),q);
      SetPixelGreen(image,ScaleCharToQuantum(TIFFGetG(*p)),q);
      SetPixelBlue(image,ScaleCharToQuantum(TIFFGetB(*p)),q);
      if (image->alpha_trait != UndefinedPixelTrait)
        SetPixelAlpha(image,ScaleCharToQuantum(TIFFGetA(*p)),q);
      p++;
      q+=GetPixelChannels(image);
    }
    if (SyncAuthenticPixels(image,exception) == MagickFalse)
      break;
    if (image->previous == (Image *) NULL)
      {
        status=SetImageProgress(image,LoadImageTag,(MagickOffsetType) y,
          image->rows);
        if (status == MagickFalse)
          break;
      }
  }
  break;
}
