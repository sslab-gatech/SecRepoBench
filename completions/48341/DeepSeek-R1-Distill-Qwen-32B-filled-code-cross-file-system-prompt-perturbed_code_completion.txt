case ReadStripMethod:
      {
        /*
          Convert strip-based TIFF image.
        */
        for (y=0; y < (ssize_t) image->rows; y += rows_per_strip)
        {
          Quantum
            *magick_restrict q;

          ssize_t
            x;

          unsigned char
            *p;

          size_t
            strip_height = MagickMin(rows_per_strip, image->rows - y);

          tiff_status=TIFFReadPixels(tiff,0,y,(char *) pixels);
          if (tiff_status == -1)
            break;
          q=QueueAuthenticPixels(image,0,y,image->columns,strip_height,exception);
          if (q == (Quantum *) NULL)
            break;
          p=pixels;
          for (x=0; x < (ssize_t) image->columns; x++)
          {
            ImportQuantumPixels(image,quantum_info,quantum_type,p,q);
            p += GetQuantumPixelsPerSample(quantum_info);
            q += GetPixelChannels(image);
          }
          if (SyncAuthenticPixels(image,exception) == MagickFalse)
            break;
          if (image->previous == (Image *) NULL)
            {
              status=SetImageProgress(image,LoadImageTag,(MagickOffsetType) y,
                image->rows);
              if (status == MagickFalse)
                break;
            }
        }
        break;
      }
      case ReadTileMethod:
      {
        /*
          Convert tile-based TIFF image.
        */
        uint32
          columns,
          rows;

        if ((TIFFGetField(tiff,TIFFTAG_TILEWIDTH,&columns) != 1) ||
            (TIFFGetField(tiff,TIFFTAG_TILELENGTH,&rows) != 1))
          ThrowTIFFException(CoderError,"ImageIsNotTiled");
        pixel_info=AcquireVirtualMemory(image->columns*image->rows,
          sizeof(unsigned char));
        if (pixel_info == (MemoryInfo *) NULL)
          ThrowTIFFException(ResourceLimitError,"MemoryAllocationFailed");
        pixels=(unsigned char *) GetVirtualMemoryBlob(pixel_info);
        for (y=0; y < (ssize_t) image->rows; y += rows)
        {
          for (uint32 x=0; x < (uint32) image->columns; x += columns)
          {
            tiff_status=TIFFReadTile(tiff,pixels,x,y,0);
            if (tiff_status == -1)
              break;
            ImportQuantumPixels(image,quantum_info,quantum_type,pixels,
              image->pixels + y*image->columns + x);
          }
          if (SyncAuthenticPixels(image,exception) == MagickFalse)
            break;
          if (image->previous == (Image *) NULL)
            {
              status=SetImageProgress(image,LoadImageTag,(MagickOffsetType) y,
                image->rows);
              if (status == MagickFalse)
                break;
            }
        }
        break;
      }
      default:
      {
        /*
          Convert generic TIFF image.
        */
        if (TIFFReadRGBAImage(tiff,image->columns,image->rows,pixels) == 0)
          ThrowTIFFException(CorruptImageError,"CorruptImage");
        ImportQuantumPixels(image,quantum_info,quantum_type,pixels,
          image->pixels);
        break;
      }
