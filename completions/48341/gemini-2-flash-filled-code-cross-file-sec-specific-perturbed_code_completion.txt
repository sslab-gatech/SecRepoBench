case ReadStripMethod:
      {
        /*
          Read TIFF image by strips.
        */
        for (y=0; y < (ssize_t) image->rows; y+=rows_per_strip)
        {
          Quantum
            *magick_restrict q;

          ssize_t
            height,
            row;

          /*
            Import strip into image.
          */
          height=(ssize_t) rows_per_strip;
          if ((y+height) > (ssize_t) image->rows)
            height=(ssize_t) image->rows-y;
          tiff_status=TIFFReadRGBAStrip(tiff,0,y,(uint32) height,
            (uint32 *) pixels);
          if (tiff_status == -1)
            break;
          q=QueueAuthenticPixels(image,0,y,image->columns,height,exception);
          if (q == (Quantum *) NULL)
            break;
          for (row=0; row < height; row++)
          {
            ssize_t
              x;

            unsigned char
              *p;

            p=pixels+(ssize_t) image->columns*row*4;
            for (x=0; x < (ssize_t) image->columns; x++)
            {
              SetPixelRed(image,ScaleCharToQuantum(p[0]),q);
              SetPixelGreen(image,ScaleCharToQuantum(p[1]),q);
              SetPixelBlue(image,ScaleCharToQuantum(p[2]),q);
              if (image->alpha_trait != UndefinedPixelTrait)
                SetPixelAlpha(image,ScaleCharToQuantum(p[3]),q);
              q+=GetPixelChannels(image);
              p+=4;
            }
          }
          if (SyncAuthenticPixels(image,exception) == MagickFalse)
            break;
          if (image->previous == (Image *) NULL)
            {
              status=SetImageProgress(image,LoadImageTag,(MagickOffsetType) y,
                image->rows);
              if (status == MagickFalse)
                break;
            }
        }
        break;
      }
      case ReadTileMethod:
      {
        /*
          Read TIFF image by tiles.
        */
        for (y=0; y < (ssize_t) image->rows; y+=rows_per_strip)
        {
          Quantum
            *magick_restrict q;

          ssize_t
            height,
            row;

          /*
            Import tile into image.
          */
          height=(ssize_t) rows_per_strip;
          if ((y+height) > (ssize_t) image->rows)
            height=(ssize_t) image->rows-y;
          tiff_status=TIFFReadRGBATile(tiff,0,y,(uint32 *) pixels);
          if (tiff_status == -1)
            break;
          q=QueueAuthenticPixels(image,0,y,image->columns,height,exception);
          if (q == (Quantum *) NULL)

          {
            ssize_t
              x;

            unsigned char
              *p;

            p=pixels+(ssize_t) image->columns*row*4;
            for (x=0; x < (ssize_t) image->columns; x++)
            {
              SetPixelRed(image,ScaleCharToQuantum(p[0]),q);
              SetPixelGreen(image,ScaleCharToQuantum(p[1]),q);
              SetPixelBlue(image,ScaleCharToQuantum(p[2]),q);
              if (image->alpha_trait != UndefinedPixelTrait)
                SetPixelAlpha(image,ScaleCharToQuantum(p[3]),q);
              q+=GetPixelChannels(image);

            {
              status=SetImageProgress(image,LoadImageTag,(MagickOffsetType) y,
                image->rows);
              if (status == MagickFalse)
                break;
            }
        }
        break;
      }
      case ReadGenericMethod:
      default:
      {
        /*
          Read entire image.
        */
        tiff_status=TIFFReadRGBAImage(tiff,image->columns,image->rows,
          (uint32 *) pixels,0);
        if (tiff_status != 1)
          break;
        for (y=0; y < (ssize_t) image->rows; y++)
        {
          Quantum
            *magick_restrict q;

          ssize_t
            x;

          unsigned char
            *p;

          q=QueueAuthenticPixels(image,0,y,image->columns,1,exception);
          if (q == (Quantum *) NULL)
            break;
          p=pixels+(ssize_t) image->columns*y*4;
          for (x=0; x < (ssize_t) image->columns; x++)
          {
            SetPixelRed(image,ScaleCharToQuantum(p[0]),q);
            SetPixelGreen(image,ScaleCharToQuantum(p[1]),q);
            SetPixelBlue(image,ScaleCharToQuantum(p[2]),q);
            if (image->alpha_trait != UndefinedPixelTrait)
              SetPixelAlpha(image,ScaleCharToQuantum(p[3]),q);
            q+=GetPixelChannels(image);
            p+=4;
          }
          if (SyncAuthenticPixels(image,exception) == MagickFalse)
            break;
          if (image->previous == (Image *) NULL)
            {
              status=SetImageProgress(image,LoadImageTag,(MagickOffsetType) y,
                image->rows);
              if (status == MagickFalse)
                break;
            }
        }
        break;
