case ReadStripMethod:
{
  ssize_t
    y;

  unsigned char
    *pixels;

  /*
    Read image data by processing each strip of the TIFF file.
  */
  for (y=0; y < (ssize_t) image->rows; y++)
  {
    Quantum
      *magick_restrict q;

    ssize_t
      x;

    tiff_status=TIFFReadStrip(tiff,y);
    if (tiff_status == -1)
      break;
    q=QueueAuthenticPixels(image,0,y,image->columns,1,exception);
    if (q == (Quantum *) NULL)
      break;
    pixels=pixels;
    for (x=0; x < (ssize_t) image->columns; x++)
    {
      switch (quantum_type)
      {
        case RedQuantum:
        {
          SetPixelRed(image,ScaleCharToQuantum((unsigned char) *pixels),q);
          break;
        }
        case GreenQuantum:
        {
          SetPixelGreen(image,ScaleCharToQuantum((unsigned char) *(pixels+1)),q);
          break;
        }
        case BlueQuantum:
        {
          SetPixelBlue(image,ScaleCharToQuantum((unsigned char) *(pixels+2)),q);
          break;
        }
        case AlphaQuantum:
        {
          SetPixelAlpha(image,ScaleCharToQuantum((unsigned char) *(pixels+3)),q);
          break;
        }
        case RGBQuantum:
        {
          SetPixelRed(image,ScaleCharToQuantum((unsigned char) *pixels),q);
          SetPixelGreen(image,ScaleCharToQuantum((unsigned char) *(pixels+1)),q);
          SetPixelBlue(image,ScaleCharToQuantum((unsigned char) *(pixels+2)),q);
          break;
        }
        case RGBAQuantum:
        {
          SetPixelRed(image,ScaleCharToQuantum((unsigned char) *pixels),q);
          SetPixelGreen(image,ScaleCharToQuantum((unsigned char) *(pixels+1)),q);
          SetPixelBlue(image,ScaleCharToQuantum((unsigned char) *(pixels+2)),q);
          SetPixelAlpha(image,ScaleCharToQuantum((unsigned char) *(pixels+3)),q);
          break;
        }
        case CMYKQuantum:
        {
          SetPixelCyan(image,ScaleCharToQuantum((unsigned char) *pixels),q);
          SetPixelMagenta(image,ScaleCharToQuantum((unsigned char) *(pixels+1)),q);
          SetPixelYellow(image,ScaleCharToQuantum((unsigned char) *(pixels+2)),q);
          SetPixelBlack(image,ScaleCharToQuantum((unsigned char) *(pixels+3)),q);
          break;
        }
        case CMYKAQuantum:
        {
          SetPixelCyan(image,ScaleCharToQuantum((unsigned char) *pixels),q);
          SetPixelMagenta(image,ScaleCharToQuantum((unsigned char) *(pixels+1)),q);
          SetPixelYellow(image,ScaleCharToQuantum((unsigned char) *(pixels+2)),q);
          SetPixelBlack(image,ScaleCharToQuantum((unsigned char) *(pixels+3)),q);
          SetPixelAlpha(image,ScaleCharToQuantum((unsigned char) *(pixels+4)),q);
          break;
        }
        case GrayQuantum:
        {
          SetPixelGray(image,ScaleCharToQuantum((unsigned char) *pixels),q);
          break;
        }
        case GrayAlphaQuantum:
        {
          SetPixelGray(image,ScaleCharToQuantum((unsigned char) *pixels),q);
          SetPixelAlpha(image,ScaleCharToQuantum((unsigned char) *(pixels+1)),q);
          break;
        }
        case IndexQuantum:
        {
          SetPixelIndex(image,ScaleCharToQuantum((unsigned char) *pixels),q);
          break;
        }
        case IndexAlphaQuantum:
        {
          SetPixelIndex(image,ScaleCharToQuantum((unsigned char) *pixels),q);
          SetPixelAlpha(image,ScaleCharToQuantum((unsigned char) *(pixels+1)),q);
          break;
        }
        case MultispectralQuantum:
        {
          SetPixelMultispectral(image,ScaleCharToQuantum((unsigned char) *pixels),q);
          break;
        }
        default:
        {
          break;
        }
      }
      q+=GetPixelChannels(image);
      pixels+=samples_per_pixel;
    }
    if (SyncAuthenticPixels(image,exception) == MagickFalse)
      break;
    if (image->previous == (Image *) NULL)
      {
        status=SetImageProgress(image,LoadImageTag,(MagickOffsetType) y,
          image->rows);
        if (status == MagickFalse)
          break;
      }
  }
  break;
}
