case ReadStripMethod:
      {
        tiff_status=1;
        for (y=0; y < (ssize_t) image->rows; y++)
        {
          tiff_status=TIFFReadPixels(tiff,0,y,(char *) pixels);
          if (tiff_status == -1)
            break;
          if (ImportQuantumPixels(image,(CacheView *) NULL,quantum_info,
              quantum_type,pixels,exception) == MagickFalse)
            break;
          if (SyncAuthenticPixels(image,exception) == MagickFalse)
            break;
          if (image->previous == (Image *) NULL)
            {
              status=SetImageProgress(image,LoadImageTag,(MagickOffsetType) y,
                image->rows);
              if (status == MagickFalse)
                break;
            }
        }
        break;
      }
      case ReadTileMethod:
      {
        uint32
          tile_columns,
          tile_rows;

        tiff_status=1;
        TIFFGetField(tiff,TIFFTAG_TILEWIDTH,&tile_columns);
        TIFFGetField(tiff,TIFFTAG_TILELENGTH,&tile_rows);
        for (y=0; y < (ssize_t) image->rows; y+=tile_rows)
        {
          ssize_t
            x;

          for (x=0; x < (ssize_t) image->columns; x+=tile_columns)
          {
            tiff_status=TIFFReadTile(tiff,pixels,x,y,0,0);
            if (tiff_status == -1)
              break;
            if (ImportQuantumPixels(image,(CacheView *) NULL,quantum_info,
                quantum_type,pixels,exception) == MagickFalse)
              break;
          }
          if (tiff_status == -1)
            break;
          if (SyncAuthenticPixels(image,exception) == MagickFalse)
            break;
          if (image->previous == (Image *) NULL)
            {
              status=SetImageProgress(image,LoadImageTag,(MagickOffsetType) y,
                image->rows);
              if (status == MagickFalse)
                break;
            }
        }
        break;
      }
      default:
      case ReadGenericMethod:
      {
        uint32
          *raster;

        raster=(uint32 *) AcquireQuantumMemory(image->columns,image->rows*
          sizeof(*raster));
        if (raster != (uint32 *) NULL)
          {
            tiff_status=TIFFReadRGBAImage(tiff,width,height,raster,0);
            if (tiff_status != 0)
              {
                ssize_t
                  x;

                uint32
                  *p;

                p=raster;
                for (y=0; y < (ssize_t) image->rows; y++)
                {
                  Quantum
                    *q;

                  q=QueueAuthenticPixels(image,0,y,image->columns,1,exception);
                  if (q == (Quantum *) NULL)
                    break;
                  for (x=0; x < (ssize_t) image->columns; x++)
                  {
                    SetPixelRed(image,ScaleCharToQuantum(TIFFGetR(*p)),q);
                    SetPixelGreen(image,ScaleCharToQuantum(TIFFGetG(*p)),q);
                    SetPixelBlue(image,ScaleCharToQuantum(TIFFGetB(*p)),q);
                    if (photometric == PHOTOMETRIC_RGB)
                      SetPixelAlpha(image,ScaleCharToQuantum(TIFFGetA(*p)),q);
                    p++;
                    q+=GetPixelChannels(image);
                  }
                  if (SyncAuthenticPixels(image,exception) == MagickFalse)
                    break;
                  if (image->previous == (Image *) NULL)
                    {
                      status=SetImageProgress(image,LoadImageTag,
                        (MagickOffsetType) y,image->rows);
                      if (status == MagickFalse)
                        break;
                    }
                }
              }
            raster=(uint32 *) RelinquishMagickMemory(raster);
          }
        break;
      }
