case ReadStripMethod:
      {
        /*
          Read TIFF image data by strips.
        */
        for (y=0; y < (ssize_t) image->rows; y += rows_per_strip)
        {
          Quantum
            *magick_restrict q;

          ssize_t
            rows_to_read;

          rows_to_read=(ssize_t) MagickMin((size_t) rows_per_strip,
            image->rows-y);
          tiff_status=TIFFReadEncodedStrip(tiff,TIFFComputeStrip(tiff,y,0),
            pixels,(tsize_t) (scanline_size*rows_to_read));
          if (tiff_status == -1)
            break;
          q=QueueAuthenticPixels(image,0,y,image->columns,rows_to_read,
            exception);
          if (q == (Quantum *) NULL)
            break;
          status=ImportQuantumPixels(image,(CacheView *) NULL,quantum_info,
            quantum_type,pixels,exception);
          if (status == MagickFalse)
            break;
          if (image->previous == (Image *) NULL)
            {
              status=SetImageProgress(image,LoadImageTag,(MagickOffsetType) y,
                image->rows);
              if (status == MagickFalse)
                break;
            }
        }
        break;
      }
      case ReadTileMethod:
      {
        /*
          Read TIFF image data by tiles.
        */
        uint32
          tile_width,
          tile_length;

        tiff_status=TIFFGetField(tiff,TIFFTAG_TILEWIDTH,&tile_width);
        tiff_status&=TIFFGetField(tiff,TIFFTAG_TILELENGTH,&tile_length);
        if (tiff_status == 0)
          ThrowTIFFException(CoderError,"ImageIsNotTiled");
        for (y=0; y < (ssize_t) image->rows; y += tile_length)
        {
          ssize_t
            x;

          for (x=0; x < (ssize_t) image->columns; x += tile_width)
          {
            Quantum
              *magick_restrict q;

            tiff_status=TIFFReadTile(tiff,pixels,(uint32) x,(uint32) y,0,0);
            if (tiff_status == -1)
              break;
            q=QueueAuthenticPixels(image,x,y,MagickMin(tile_width,
              image->columns-x),MagickMin(tile_length,image->rows-y),
              exception);
            if (q == (Quantum *) NULL)
              break;
            status=ImportQuantumPixels(image,(CacheView *) NULL,quantum_info,
              quantum_type,pixels,exception);
            if (status == MagickFalse)
              break;
          }
          if (image->previous == (Image *) NULL)
            {
              status=SetImageProgress(image,LoadImageTag,(MagickOffsetType) y,
                image->rows);
              if (status == MagickFalse)
                break;
            }
        }
        break;
      }
      case ReadGenericMethod:
      default:
      {
        /*
          Read TIFF image data using TIFFReadRGBAImage.
        */
        uint32
          *raster;

        raster=(uint32 *) pixels;
        tiff_status=TIFFReadRGBAImage(tiff,(uint32) image->columns,
          (uint32) image->rows,raster,0);
        if (tiff_status == 0)
          ThrowTIFFException(CorruptImageError,"UnableToReadImageData");
        for (y=0; y < (ssize_t) image->rows; y++)
        {
          Quantum
            *magick_restrict q;

          ssize_t
            x;

          q=QueueAuthenticPixels(image,0,(ssize_t) image->rows-y-1,
            image->columns,1,exception);
          if (q == (Quantum *) NULL)
            break;
          for (x=0; x < (ssize_t) image->columns; x++)
          {
            uint32
              pixel;

            pixel=raster[y*image->columns+x];
            SetPixelRed(image,ScaleCharToQuantum((unsigned char) TIFFGetR(pixel)),q);
            SetPixelGreen(image,ScaleCharToQuantum((unsigned char) TIFFGetG(pixel)),q);
            SetPixelBlue(image,ScaleCharToQuantum((unsigned char) TIFFGetB(pixel)),q);
            if (image->alpha_trait != UndefinedPixelTrait)
              SetPixelAlpha(image,ScaleCharToQuantum((unsigned char) TIFFGetA(pixel)),q);
            q+=GetPixelChannels(image);
          }
          if (SyncAuthenticPixels(image,exception) == MagickFalse)
            break;
          if (image->previous == (Image *) NULL)
            {
              status=SetImageProgress(image,LoadImageTag,(MagickOffsetType) y,
                image->rows);
              if (status == MagickFalse)
                break;
            }
        }
        break;
      }
