case ReadStripMethod:
{
  /*
    Read TIFF image by processing each strip of the TIFF file.
  */
  size_t
    strip_size;

  ssize_t
    i,
    j;

  unsigned char
    *strip;

  strip_size=TIFFStripSize(tiff);
  strip=(unsigned char *) AcquireQuantumMemory((size_t) strip_size,
    sizeof(*strip));
  if (strip == (unsigned char *) NULL)
    ThrowTIFFException(ResourceLimitError,"MemoryAllocationFailed");
  for (i=0; i < (ssize_t) TIFFNumberOfStrips(tiff); i++)
  {
    ssize_t
      count;

    count=(ssize_t) TIFFReadRawStrip(tiff,(uint32) i,strip,strip_size);
    if (count < 0)
      break;
    for (j=0; j < (ssize_t) count; j++)
      pixels[j]=strip[j];
    tiff_status=TIFFWritePixels(tiff,&tiff_info,i*rows_per_strip,0,image);
    if (tiff_status == -1)
      break;
  }
  strip=(unsigned char *) RelinquishMagickMemory(strip);
  break;
}
case ReadTileMethod:
{
  /*
    Read TIFF image by processing each tile of the TIFF file.
  */
  size_t
    tile_size;

  ssize_t
    i,
    j;

  unsigned char
    *tile;

  tile_size=TIFFTileSize(tiff);
  tile=(unsigned char *) AcquireQuantumMemory((size_t) tile_size,
    sizeof(*tile));
  if (tile == (unsigned char *) NULL)
    ThrowTIFFException(ResourceLimitError,"MemoryAllocationFailed");
  for (i=0; i < (ssize_t) TIFFNumberOfTiles(tiff); i++)
  {
    ssize_t
      count;

    count=(ssize_t) TIFFReadRawTile(tiff,(uint32) i,tile,tile_size);
    if (count < 0)
      break;
    for (j=0; j < (ssize_t) count; j++)
      pixels[j]=tile[j];
    tiff_status=TIFFWritePixels(tiff,&tiff_info,i*tile_rows,0,image);
    if (tiff_status == -1)
      break;
  }
  tile=(unsigned char *) RelinquishMagickMemory(tile);
  break;
}
case ReadGenericMethod:
default:
{
  /*
    Read TIFF image using TIFFReadRGBAImage.
  */
  tiff_status=TIFFReadRGBAImage(tiff,image->columns,image->rows,pixels,
    0,0);
  if (tiff_status == -1)
    break;
  tiff_status=TIFFWritePixels(tiff,&tiff_info,0,0,image);
  if (tiff_status == -1)
    break;
  break;
}
