case ReadStripMethod:
      {
        /*
          Convert TIFF image to runlength encoded pixels.
        */
        for (y=0; y < (ssize_t) image->rows; y++)
        {
          Quantum
            *magick_restrict q;

          tiff_status=TIFFReadPixels(tiff,0,y,(char *) pixels);
          if (tiff_status == -1)
            break;
          q=QueueAuthenticPixels(image,0,y,image->columns,1,exception);
          if (q == (Quantum *) NULL)
            break;
          ImportQuantumPixels(image,(CacheView *) NULL,quantum_info,
            quantum_type,pixels,exception);
          if (SyncAuthenticPixels(image,exception) == MagickFalse)
            break;
          if (image->previous == (Image *) NULL)
            {
              status=SetImageProgress(image,LoadImageTag,(MagickOffsetType) y,
                image->rows);
              if (status == MagickFalse)
                break;
            }
        }
        break;
      }
      case ReadTileMethod:
      {
        /*
          Convert tiled TIFF image.
        */
        for (y=0; y < (ssize_t) image->rows; y++)
        {
          Quantum
            *magick_restrict q;

          size_t
            tile_width;

          ssize_t
            x;

          uint32
            tile_y;

          tile_y=(uint32) (y/tiff_info->tile_geometry.height)*
            tiff_info->tile_geometry.height;
          tile_width=(size_t) (image->columns-(
            (image->columns/tiff_info->tile_geometry.width)*
            tiff_info->tile_geometry.width));
          tiff_status=TIFFReadPixels(tiff,0,y,(char *) pixels);
          if (tiff_status == -1)
            break;
          q=QueueAuthenticPixels(image,0,y,image->columns,1,exception);
          if (q == (Quantum *) NULL)
            break;
          ImportQuantumPixels(image,(CacheView *) NULL,quantum_info,
            quantum_type,pixels,exception);
          if (SyncAuthenticPixels(image,exception) == MagickFalse)
            break;
          if (image->previous == (Image *) NULL)
            {
              status=SetImageProgress(image,LoadImageTag,(MagickOffsetType) y,
                image->rows);
              if (status == MagickFalse)
                break;
            }
        }
        break;
      }
      case ReadGenericMethod:
      default:
      {
        /*
          Read entire image at once.
        */
        if (TIFFReadRGBAImage(tiff,image->columns,image->rows,pixels,0) == 0)
          ThrowTIFFException(CoderError,"UnableToReadImageData");
        ImportQuantumPixels(image,(CacheView *) NULL,quantum_info,
          quantum_type,pixels,exception);
        break;
      }
