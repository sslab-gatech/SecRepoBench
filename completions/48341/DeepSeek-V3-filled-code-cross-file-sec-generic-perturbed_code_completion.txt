case ReadStripMethod:
      {
        /*
          Read image by strips.
        */
        for (y=0; y < (ssize_t) image->rows; y+=rows_per_strip)
        {
          ssize_t
            count;

          count=(ssize_t) MagickMin(rows_per_strip,(uint32) (image->rows-y));
          tiff_status=TIFFReadEncodedStrip(tiff,TIFFComputeStrip(tiff,(uint32) y,0),
            pixels,count*scanline_size);
          if (tiff_status == -1)
            break;
          status=ImportQuantumPixels(image,(CacheView *) NULL,quantum_info,
            quantum_type,pixels,exception);
          if (status == MagickFalse)
            break;
          if (image->previous == (Image *) NULL)
            {
              status=SetImageProgress(image,LoadImageTag,(MagickOffsetType) y,
                image->rows);
              if (status == MagickFalse)
                break;
            }
        }
        break;
      }
      case ReadTileMethod:
      {
        /*
          Read image by tiles.
        */
        uint32
          tile_height,
          tile_width;

        if ((TIFFGetField(tiff,TIFFTAG_TILEWIDTH,&tile_width) != 1) ||
            (TIFFGetField(tiff,TIFFTAG_TILELENGTH,&tile_height) != 1))
          ThrowTIFFException(CoderError,"ImageIsNotTiled");
        for (y=0; y < (ssize_t) image->rows; y+=tile_height)
        {
          ssize_t
            x;

          for (x=0; x < (ssize_t) image->columns; x+=tile_width)
          {
            tiff_status=TIFFReadTile(tiff,pixels,(uint32) x,(uint32) y,0,0);
            if (tiff_status == -1)
              break;
            status=ImportQuantumPixels(image,(CacheView *) NULL,quantum_info,
              quantum_type,pixels,exception);
            if (status == MagickFalse)
              break;
          }
          if (image->previous == (Image *) NULL)
            {
              status=SetImageProgress(image,LoadImageTag,(MagickOffsetType) y,
                image->rows);
              if (status == MagickFalse)
                break;
            }
        }
        break;
      }
      case ReadGenericMethod:
      default:
      {
        /*
          Read image generically.
        */
        uint32
          *raster;

        raster=(uint32 *) AcquireQuantumMemory((size_t) image->columns,
          image->rows*sizeof(*raster));
        if (raster == (uint32 *) NULL)
          ThrowTIFFException(ResourceLimitError,"MemoryAllocationFailed");
        if (TIFFReadRGBAImage(tiff,(uint32) image->columns,(uint32) image->rows,
            raster,0) == 0)
          {
            raster=(uint32 *) RelinquishMagickMemory(raster);
            ThrowTIFFException(CorruptImageError,"UnableToReadImageData");
          }
        for (y=0; y < (ssize_t) image->rows; y++)
        {
          Quantum
            *q;

          ssize_t
            x;

          uint32
            *p;

          q=QueueAuthenticPixels(image,0,y,image->columns,1,exception);
          if (q == (Quantum *) NULL)
            break;
          p=raster+y*image->columns;
          for (x=0; x < (ssize_t) image->columns; x++)
          {
            SetPixelRed(image,ScaleCharToQuantum((unsigned char) TIFFGetR(*p)),q);
            SetPixelGreen(image,ScaleCharToQuantum((unsigned char) TIFFGetG(*p)),q);
            SetPixelBlue(image,ScaleCharToQuantum((unsigned char) TIFFGetB(*p)),q);
            SetPixelAlpha(image,ScaleCharToQuantum((unsigned char) TIFFGetA(*p)),q);
            q+=GetPixelChannels(image);
            p++;
          }
          if (SyncAuthenticPixels(image,exception) == MagickFalse)
            break;
          if (image->previous == (Image *) NULL)
            {
              status=SetImageProgress(image,LoadImageTag,(MagickOffsetType) y,
                image->rows);
              if (status == MagickFalse)
                break;
            }
        }
        raster=(uint32 *) RelinquishMagickMemory(raster);
        break;
      }
