case ReadStripMethod:
      {
        /*
          Read image one strip at a time.
        */
        for (y=0; y < (ssize_t) image->rows; y+=rows_per_strip)
        {
          Quantum
            *magick_restrict q;

          ssize_t
            height;

          /*
            Import pixel row.
          */
          height=(ssize_t) rows_per_strip;
          if ((y+height) > (ssize_t) image->rows)
            height=(ssize_t) image->rows-y;
          tiff_status=TIFFReadPixels(tiff,0,y,(char *) pixels);
          if (tiff_status == -1)
            break;
          q=QueueAuthenticPixels(image,0,y,image->columns,height,exception);
          if (q == (Quantum *) NULL)
            break;
          if (ImportQuantumPixels(image,quantum_info,quantum_type,pixels,
              exception) == MagickFalse)
            break;
          if (SyncAuthenticPixels(image,exception) == MagickFalse)
            break;
          if (image->previous == (Image *) NULL)
            {
              status=SetImageProgress(image,LoadImageTag,(MagickOffsetType) y,
                image->rows);
              if (status == MagickFalse)
                break;
            }
        }
        break;
      }
      case ReadTileMethod:
      {
        /*
          Read image one tile at a time.
        */
        for (y=0; y < (ssize_t) image->rows; y+=rows_per_strip)
        {
          Quantum
            *magick_restrict q;

          ssize_t
            height;

          /*
            Import pixel row.
          */
          height=(ssize_t) rows_per_strip;
          if ((y+height) > (ssize_t) image->rows)
            height=(ssize_t) image->rows-y;
          tiff_status=TIFFReadRGBAImage(tiff,0,0,(uint32) image->columns,
            (uint32) height,(uint32 *) pixels,0);
          if (tiff_status == -1)
            break;
          q=QueueAuthenticPixels(image,0,y,image->columns,height,exception);
          if (q == (Quantum *) NULL)
            break;
          if (ImportQuantumPixels(image,quantum_info,quantum_type,pixels,
              exception) == MagickFalse)
            break;
          if (SyncAuthenticPixels(image,exception) == MagickFalse)
            break;
          if (image->previous == (Image *) NULL)
            {
              status=SetImageProgress(image,LoadImageTag,(MagickOffsetType) y,
                image->rows);
              if (status == MagickFalse)
                break;
            }
        }
        break;
      }
      case ReadGenericMethod:
      default:
      {
        /*
          Read entire image via TIFFReadRGBAImage().
        */
        tiff_status=TIFFReadRGBAImage(tiff,0,0,(uint32) image->columns,
          (uint32) image->rows,(uint32 *) pixels,0);
        if (tiff_status == -1)
          break;
        if (ImportQuantumPixels(image,quantum_info,quantum_type,pixels,
            exception) == MagickFalse)
          break;
        if (image->previous == (Image *) NULL)
          status=SetImageProgress(image,LoadImageTag,image->rows,image->rows);
        break;
      }
