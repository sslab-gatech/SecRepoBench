case ReadStripMethod:
      {
        /*
          Convert TIFF image to pixel packets.
        */
        for (y=0; y < (ssize_t) image->rows; y++)
        {
          Quantum
            *magick_restrict q;

          ssize_t
            x;

          unsigned char
            *p;

          if (TIFFReadScanline(tiff,pixels,y) == -1)
            break;
          q=QueueAuthenticPixels(image,0,y,image->columns,1,exception);
          if (q == (Quantum *) NULL)
            break;
          p=pixels;
          for (x=0; x < (ssize_t) image->columns; x++)
          {
            ImportQuantumPixels(image,quantum_info,quantum_type,p,q,exception);
            p+=quantum_info->pad;
            q+=GetPixelChannels(image);
          }
          if (SyncAuthenticPixels(image,exception) == MagickFalse)
            break;
          if (image->previous == (Image *) NULL)
            {
              status=SetImageProgress(image,LoadImageTag,(MagickOffsetType) y,
                image->rows);
              if (status == MagickFalse)
                break;
            }
        }
        break;
      }
      case ReadTileMethod:
      {
        /*
          Convert tiled TIFF image to pixel packets.
        */
        for (y=0; y < (ssize_t) image->rows; y+=rows_per_strip)
        {
          Quantum
            *magick_restrict q;

          size_t
            columns,
            rows;

          ssize_t
            x;

          unsigned char
            *p;

          uint32
            tile_width,
            tile_length;

          if (TIFFGetField(tiff,TIFFTAG_TILEWIDTH,&tile_width) != 1)
            break;
          if (TIFFGetField(tiff,TIFFTAG_TILELENGTH,&tile_length) != 1)
            break;
          columns=tile_width;
          rows=tile_length;
          if (y+rows > (ssize_t) image->rows)
            rows=(size_t) (image->rows-y);
          if (TIFFReadTile(tiff,pixels,0,y,0,rows) == -1)
            break;
          q=QueueAuthenticPixels(image,0,y,image->columns,rows,exception);
          if (q == (Quantum *) NULL)
            break;
          p=pixels;
          for (x=0; x < (ssize_t) image->columns; x++)
          {
            ImportQuantumPixels(image,quantum_info,quantum_type,p,q,exception);
            p+=quantum_info->pad;
            q+=GetPixelChannels(image);
          }
          if (SyncAuthenticPixels(image,exception) == MagickFalse)
            break;
          if (image->previous == (Image *) NULL)
            {
              status=SetImageProgress(image,LoadImageTag,(MagickOffsetType) y,
                image->rows);
              if (status == MagickFalse)
                break;
            }
        }
        break;
      }
      case ReadGenericMethod:
      default:
      {
        /*
          Convert TIFF image to pixel packets.
        */
        if (TIFFReadRGBAImage(tiff,image->columns,image->rows,pixels,0) == 0)
          ThrowTIFFException(CorruptImageError,"UnableToReadImageData");
        for (y=0; y < (ssize_t) image->rows; y++)
        {
          Quantum
            *magick_restrict q;

          ssize_t
            x;

          unsigned char
            *p;

          q=QueueAuthenticPixels(image,0,y,image->columns,1,exception);
          if (q == (Quantum *) NULL)
            break;
          p=pixels+y*(ssize_t) (image->columns*4);
          for (x=0; x < (ssize_t) image->columns; x++)
          {
            SetPixelRed(image,ScaleCharToQuantum(*p),q);
            SetPixelGreen(image,ScaleCharToQuantum(*(p+1)),q);
            SetPixelBlue(image,ScaleCharToQuantum(*(p+2)),q);
            SetPixelAlpha(image,ScaleCharToQuantum(*(p+3)),q);
            q+=GetPixelChannels(image);
            p+=4;
          }
          if (SyncAuthenticPixels(image,exception) == MagickFalse)
            break;
          if (image->previous == (Image *) NULL)
            {
              status=SetImageProgress(image,LoadImageTag,(MagickOffsetType) y,
                image->rows);
              if (status == MagickFalse)
                break;
            }
        }
        break;
      }
    }
