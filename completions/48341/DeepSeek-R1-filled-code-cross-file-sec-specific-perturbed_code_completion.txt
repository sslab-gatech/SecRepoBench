case ReadStripMethod:
      {
        for (y=0; y < (ssize_t) image->rows; y++)
        {
          tiff_status=TIFFReadScanline(tiff,pixels,y);
          if (tiff_status == -1)
            break;
          Quantum
            *q;

          q=QueueAuthenticPixels(image,0,y,image->columns,1,exception);
          if (q == (Quantum *) NULL)
            break;
          status=ImportQuantumPixels(image,(CacheView *) NULL,quantum_info,
            quantum_type,pixels,exception);
          if (status == MagickFalse)
            break;
          if (SyncAuthenticPixels(image,exception) == MagickFalse)
            break;
          if (image->previous == (Image *) NULL)
            {
              status=SetImageProgress(image,LoadImageTag,(MagickOffsetType) y,
                image->rows);
              if (status == MagickFalse)
                break;
            }
        }
        break;
      }
      case ReadTileMethod:
      {
        uint32
          tile_width,
          tile_height;

        tsize_t
          tile_size;

        if ((TIFFGetField(tiff,TIFFTAG_TILEWIDTH,&tile_width) != 1) ||
            (TIFFGetField(tiff,TIFFTAG_TILELENGTH,&tile_height) != 1))
          ThrowTIFFException(CoderError,"ImageIsNotTiled");
        tile_size=TIFFTileSize(tiff);
        if (tile_size > (tsize_t) GetBlobSize(image))
          ThrowTIFFException(CorruptImageError,"CorruptImage");
        for (y=0; y < (ssize_t) image->rows; y += tile_height)
        {
          for (ssize_t x=0; x < (ssize_t) image->columns; x += tile_width)
          {
            tiff_status=TIFFReadTile(tiff,pixels,x,y,0,0);
            if (tiff_status == -1)
              break;
            for (ssize_t ty=0; ty < (ssize_t) tile_height; ty++)
            {
              ssize_t
                current_y;

              current_y=y+ty;
              if (current_y >= (ssize_t) image->rows)
                break;
              Quantum
                *q;

              q=QueueAuthenticPixels(image,x,current_y,
                MagickMin(tile_width,image->columns-x),1,exception);
              if (q == (Quantum *) NULL)
                break;
              status=ImportQuantumPixels(image,(CacheView *) NULL,quantum_info,
                quantum_type,pixels+ty*TIFFTileRowSize(tiff),exception);
              if (status == MagickFalse)
                break;
              if (SyncAuthenticPixels(image,exception) == MagickFalse)
                break;
            }
          }
        }
        break;
      }
      case ReadGenericMethod:
      default:
      {
        uint32
          *raster;

        raster=(uint32 *) _TIFFmalloc((size_t) width*(size_t) height*sizeof(uint32));
        if (raster == (uint32 *) NULL)
          ThrowTIFFException(ResourceLimitError,"MemoryAllocationFailed");
        if (TIFFReadRGBAImage(tiff,width,height,raster,0) != 0)
        {
          for (y=0; y < (ssize_t) height; y++)
          {
            Quantum
              *q;

            uint32
              *p;

            q=QueueAuthenticPixels(image,0,y,image->columns,1,exception);
            if (q == (Quantum *) NULL)
              break;
            p=raster+(height-y-1)*width;
            for (ssize_t x=0; x < (ssize_t) width; x++)
            {
              SetPixelRed(image,ScaleCharToQuantum(TIFFGetR(*p)),q);
              SetPixelGreen(image,ScaleCharToQuantum(TIFFGetG(*p)),q);
              SetPixelBlue(image,ScaleCharToQuantum(TIFFGetB(*p)),q);
              if (image->alpha_trait != UndefinedPixelTrait)
                SetPixelAlpha(image,ScaleCharToQuantum(TIFFGetA(*p)),q);
              p++;
              q+=GetPixelChannels(image);
            }
            if (SyncAuthenticPixels(image,exception) == MagickFalse)
              break;
            if (image->previous == (Image *) NULL)
              {
                status=SetImageProgress(image,LoadImageTag,(MagickOffsetType) y,
                  image->rows);
                if (status == MagickFalse)
                  break;
              }
          }
        }
        _TIFFfree(raster);
        break;
      }
