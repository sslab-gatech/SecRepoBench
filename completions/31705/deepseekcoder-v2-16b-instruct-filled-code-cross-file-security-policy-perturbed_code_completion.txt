// Close the file pointer if it is open.
  // If there was an error, free the allocated chunk if necessary and return the error code.
  // If successful, return the size of the lazy chunk.
  if (fp != NULL) {
    fclose(fp);
  }
  if (rc < 0) {
    if (*needs_free) {
      free(*chunk);
    }
    *chunk = NULL;
    *needs_free = false;
    return rc;
  }

  return lazy_chunk_bytes;
}
