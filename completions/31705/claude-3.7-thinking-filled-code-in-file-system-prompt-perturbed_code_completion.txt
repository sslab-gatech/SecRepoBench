  // Close the file pointer if it is open.
  if (fp != NULL) {
    fclose(fp);
  }

  if (rc < 0) {
    if (*needs_free && *chunk != NULL) {
      free(*chunk);
      *chunk = NULL;
      *needs_free = false;
    }
