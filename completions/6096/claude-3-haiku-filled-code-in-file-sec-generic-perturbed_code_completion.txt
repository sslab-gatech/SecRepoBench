// <MASK>
switch (*expression)
{
  case '(':
    (*recursion_depth)++;
    if (*recursion_depth > FxMaxParenthesisDepth)
    {
      (void) ThrowMagickException(exception,GetMagickModule(),OptionError,
        "UnableToParseExpression","`%s'",expression);
      FxResult(0.0);
    }
    expression++;
    alpha=FxEvaluateSubexpression(fx_info,channel,x,y,expression,recursion_depth,beta,exception);
    (*recursion_depth)--;
    FxResult(alpha);
  default:
    break;
}
