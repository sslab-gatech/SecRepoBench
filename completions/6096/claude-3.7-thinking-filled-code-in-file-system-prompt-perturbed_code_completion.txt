if (*expression == '(')
{
  const char
    *closing;

  (*recursion_depth)++;
  if (*recursion_depth > FxMaxParenthesisDepth)
    {
      (void) ThrowMagickException(exception,GetMagickModule(),OptionError,
        "UnableToParseExpression","`%s'",expression);
      FxResult(0.0);
    }
  closing=FxSubexpression(expression,exception);
  if ((*closing != ')') || (exception->severity != UndefinedException))
    FxResult(0.0);
  (void) CopyMagickString(subexpression,expression+1,(size_t)(closing-expression));
  alpha=FxEvaluateSubexpression(fx_info,channel,x,y,subexpression,recursion_depth,
    beta,exception);
  (*recursion_depth)--;
  FxResult(alpha);
}
