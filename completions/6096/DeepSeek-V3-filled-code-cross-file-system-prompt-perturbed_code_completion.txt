if (*expression == '(')
    {
      (*recursion_depth)++;
      if (*recursion_depth > FxMaxParenthesisDepth)
        {
          (void) ThrowMagickException(exception,GetMagickModule(),OptionError,
            "ParenthesisDepthExceedsLimit","`%s'",expression);
          FxResult(0.0);
        }
      (void) CopyMagickString(subexpression,expression+1,MagickPathExtent);
      q=subexpression;
      p=StringToken(")",&q);
      if (p == (char *) NULL)
        {
          (void) ThrowMagickException(exception,GetMagickModule(),OptionError,
            "UnmatchedParenthesis","`%s'",expression);
          FxResult(0.0);
        }
      *q='\0';
      gamma=FxEvaluateSubexpression(fx_info,channel,x,y,subexpression,
        recursion_depth,beta,exception);
      (*recursion_depth)--;
      FxResult(gamma);
    }
