if (*expression == '(')
  {
    size_t
      old_recursion_depth;

    old_recursion_depth = *recursion_depth;
    (*recursion_depth)++;
    if (*recursion_depth > FxMaxParenthesisDepth)
    {
      (void) ThrowMagickException(exception, GetMagickModule(), OptionError,
        "UnableToParseExpression", "`%s'", expression);
      FxResult(0.0);
    }
    *subexpression = '\0';
    p = FxOperatorPrecedence(expression + 1, exception);
    if (p == (const char *) NULL || *p != ')')
    {
      (void) ThrowMagickException(exception, GetMagickModule(), OptionError,
        "UnableToParseExpression", "`%s'", expression);
      FxResult(0.0);
    }
    (void) CopyMagickString(subexpression, expression + 1, (size_t) (p - expression));
    alpha = FxEvaluateSubexpression(fx_info, channel, x, y, subexpression, recursion_depth,
      beta, exception);
    (*recursion_depth)--;
    FxResult(alpha);
  }
