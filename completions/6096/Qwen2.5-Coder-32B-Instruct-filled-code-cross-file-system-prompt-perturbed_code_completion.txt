if (*expression == '(')
{
  (*recursion_depth)++;
  if (*recursion_depth > FxMaxParenthesisDepth)
  {
    (void) ThrowMagickException(exception,GetMagickModule(),OptionError,
      "TooManyNestedParentheses","`%s'",expression);
    FxResult(0.0);
  }
  (void) CopyMagickString(subexpression,expression+1,MagickPathExtent);
  q=subexpression;
  p=FxOperatorPrecedence(subexpression,exception);
  if (p != (const char *) NULL)
    q=(char *) p;
  while (*q != ')')
  {
    if (*q == '\0')
      {
        (void) ThrowMagickException(exception,GetMagickModule(),OptionError,
          "MissingParenthesis","`%s'",expression);
        FxResult(0.0);
      }
    q++;
  }
  *q='\0';
  alpha=FxEvaluateSubexpression(fx_info,channel,x,y,subexpression,recursion_depth,
    beta,exception);
  (*recursion_depth)--;
  FxResult(alpha);
}
