if (*expression == '(')
    {
      (*recursion_depth)++;
      expression++;
      p = FxOperatorPrecedence(expression, exception);
      if (p == (const char *) NULL)
      {
        (void) ThrowMagickException(exception, GetMagickModule(),
          OptionError, "UnableToParseExpression", "`%s'", expression);
        (*recursion_depth)--;
        FxResult(0.0);
      }
      CopyMagickString(subexpression, expression, (size_t)(p - expression + 1));
      alpha = FxEvaluateSubexpression(fx_info, channel, x, y, subexpression, recursion_depth, beta, exception);
      (*recursion_depth)--;
      FxResult(alpha);
    }
