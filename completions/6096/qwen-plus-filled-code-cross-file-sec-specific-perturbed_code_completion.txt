if (*expression == '(')
    {
      (*recursion_depth)++;
      if (*recursion_depth > FxMaxParenthesisDepth)
        {
          (void) ThrowMagickException(exception,GetMagickModule(),OptionError,
            "MaximumParenthesesExceeded","`%s'",expression);
          FxResult(0.0);
        }
      p=strchr(expression+1,')');
      if (p == NULL)
        {
          (void) ThrowMagickException(exception,GetMagickModule(),OptionError,
            "UnableToFindMatchingParentheses","`%s'",expression);
          FxResult(0.0);
        }
      *beta = FxEvaluateSubexpression(fx_info, channel, x, y, expression + 1,
        recursion_depth, beta, exception);
      (*recursion_depth)--;
      FxResult(*beta);
    }
