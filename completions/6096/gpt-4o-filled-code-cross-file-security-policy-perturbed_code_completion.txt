if (*expression == '(')
  {
    (*recursion_depth)++;
    if (*recursion_depth > FxMaxParenthesisDepth)
    {
      (void) ThrowMagickException(exception,GetMagickModule(),OptionError,
        "ParenthesisNestingDepthExceeded","`%s'",expression);
      FxResult(0.0);
    }
    expression++;
    p=expression;
    while ((*p != '\0') && (*p != ')'))
      p++;
    if (*p == '\0')
    {
      (void) ThrowMagickException(exception,GetMagickModule(),OptionError,
        "UnableToParseExpression","`%s'",expression);
      FxResult(0.0);
    }
    (void) CopyMagickString(subexpression,expression,(size_t) (p-expression+1));
    subexpression[p-expression]='\0';
    gamma=FxEvaluateSubexpression(fx_info,channel,x,y,subexpression,
      recursion_depth,beta,exception);
    (*recursion_depth)--;
    FxResult(gamma);
  }
