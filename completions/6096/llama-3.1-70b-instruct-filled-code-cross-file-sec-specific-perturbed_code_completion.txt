if (*expression == '(')
{
  (*recursion_depth)++;
  if (*recursion_depth > FxMaxParenthesisDepth)
    {
      (void) ThrowMagickException(exception,GetMagickModule(),OptionError,
        "ParenthesisDepthOverflow","`%s'",expression);
      FxResult(0.0);
    }
  (void) CopyMagickString(subexpression,++expression,MagickPathExtent);
  q=subexpression;
  p=StringToken(")",&q);
  if (q == (char *) NULL)
    {
      (void) ThrowMagickException(exception,GetMagickModule(),OptionError,
        "UnableToParseExpression","`%s'",subexpression);
      FxResult(0.0);
    }
  *q='\0';
  alpha=FxEvaluateSubexpression(fx_info,channel,x,y,subexpression,recursion_depth,
    beta,exception);
  (*recursion_depth)--;
  FxResult(alpha);
}
