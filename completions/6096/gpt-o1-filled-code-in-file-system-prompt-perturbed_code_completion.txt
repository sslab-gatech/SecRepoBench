if (*expression == '(')
{
  (*recursion_depth)++;
  if (*recursion_depth > FxMaxParenthesisDepth)
    {
      (void) ThrowMagickException(exception,GetMagickModule(),OptionError,
        "TooManyParenthesis","`%s'",expression);
      FxResult(0.0);
    }
  {
    register ssize_t
      level = 1;

    register const char
      *r;

    size_t
      length;

    r=expression+1;
    while ((*r != '\0') && (level > 0))
    {
      if (*r == '(')
        level++;
      else
        if (*r == ')')
          level--;
      r++;
    }
    if (level != 0)
      {
        (void) ThrowMagickException(exception,GetMagickModule(),OptionError,
          "UnbalancedParenthesis","`%s'",expression);
        FxResult(0.0);
      }
    length=(size_t) (r-(expression+1)-1);
    (void) CopyMagickString(subexpression,expression+1,length+1);
    subexpression[length]='\0';
    alpha=FxEvaluateSubexpression(fx_info,channel,x,y,subexpression,recursion_depth,
      beta,exception);
  }
  (*recursion_depth)--;
  FxResult(alpha);
}
