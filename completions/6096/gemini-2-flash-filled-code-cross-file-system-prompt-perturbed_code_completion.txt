if (*expression == '(')
    {
      size_t
        extent;

      (*recursion_depth)++;
      if (*recursion_depth > FxMaxParenthesisDepth)
        {
          (void) ThrowMagickException(exception,GetMagickModule(),OptionError,
            "ParenthesisNestedTooDeep","`%s'",expression);
          FxResult(0.0);
        }
      extent=0;
      p=expression+1;
      while ((*p != '\0') && (extent >= 0))
      {
        if (*p == '(')
          extent++;
        if (*p == ')')
          extent--;
        if (extent == (size_t) (-1))
          break;
        p++;
      }
      if (*p != ')')
        {
          (void) ThrowMagickException(exception,GetMagickModule(),OptionError,
            "UnbalancedParenthesis","`%s'",expression);
          FxResult(0.0);
        }
      (void) CopyMagickString(subexpression,expression+1,(size_t) (p-expression));
      gamma=FxEvaluateSubexpression(fx_info,channel,x,y,subexpression,recursion_depth,
        beta,exception);
      (*recursion_depth)--;
      FxResult(gamma);
    }
