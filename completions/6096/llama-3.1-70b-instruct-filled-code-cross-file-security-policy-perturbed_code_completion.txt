if (*expression == '(')
{
  (*recursion_depth)++;
  if (*recursion_depth > FxMaxParenthesisDepth)
    {
      (void) ThrowMagickException(exception,GetMagickModule(),OptionError,
        "ParenthesisOverflow","`%s'",expression);
      FxResult(0.0);
    }
  p=FxOperatorPrecedence(expression+1,exception);
  if (p == (const char *) NULL)
    p=expression+strlen(expression)-1;
  if (*p!= ')')
    {
      (void) ThrowMagickException(exception,GetMagickModule(),OptionError,
        "UnbalancedParenthesis","`%s'",expression);
      FxResult(0.0);
    }
  if (strlen(expression+1) < strlen(p))
    {
      (void) CopyMagickString(subexpression,expression+1,(size_t) (p-expression));
      if (strlen(subexpression) > 0)
        subexpression[strlen(subexpression)-1]='\0';
    }
  else
    subexpression[0]='\0';
  alpha=FxEvaluateSubexpression(fx_info,channel,x,y,subexpression,recursion_depth,
    beta,exception);
  (*recursion_depth)--;
  FxResult(alpha);
}
