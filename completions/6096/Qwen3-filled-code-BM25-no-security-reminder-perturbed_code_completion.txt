if (*expression == '(')
    {
      (*recursion_depth)++;
      if (*recursion_depth > MagickMaxRecursionDepth)
        {
          (void) ThrowMagickException(exception,GetMagickModule(),OptionError,
            "UnableToParseExpression","`%s'",expression);
          FxResult(0.0);
        }
      const char *closing=FxSubexpression(expression,exception);
      if (closing == expression)
        {
          (void) ThrowMagickException(exception,GetMagickModule(),OptionError,
            "UnbalancedParenthesis","`%s'",expression);
          FxResult(0.0);
        }
      size_t inner_length=(size_t) (closing-expression-1);
      if (inner_length >= MagickPathExtent)
        inner_length=MagickPathExtent-1;
      (void) memcpy(subexpression,expression+1,inner_length);
      subexpression[inner_length]='\0';
      gamma=FxEvaluateSubexpression(fx_info,channel,x,y,subexpression,recursion_depth,beta,exception);
      (*recursion_depth)--;
      FxResult(gamma);
    }
