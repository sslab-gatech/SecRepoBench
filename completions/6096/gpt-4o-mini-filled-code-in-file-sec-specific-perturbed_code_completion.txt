if (*expression == '(')
  {
    (*recursion_depth)++;
    subexpression=AcquireString(expression+1);
    while (*subexpression != '\0' && *subexpression != ')')
      subexpression++;
    if (*subexpression == '\0')
    {
      (void) ThrowMagickException(exception,GetMagickModule(),OptionError,
        "UnbalancedParenthesis","`%s'",expression);
      return(0.0);
    }
    *subexpression='\0';
    alpha=FxEvaluateSubexpression(fx_info,channel,x,y,expression+1,recursion_depth,beta,exception);
    (*recursion_depth)--;
    return(alpha);
  }
