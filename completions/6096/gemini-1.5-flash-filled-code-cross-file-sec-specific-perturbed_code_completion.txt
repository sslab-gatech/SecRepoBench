if (*expression == '(')
    {
      register const char
        *q;

      size_t
        close_parenthesis;

      (*recursion_depth)++;
      close_parenthesis=1;
      for (q=expression+1; *q != '\0'; q++)
      {
        if (*q == '(')
          close_parenthesis++;
        else
          if (*q == ')')
            {
              close_parenthesis--;
              if (close_parenthesis == 0)
                break;
            }
      }
      if (*q == '\0')
        {
          (void) ThrowMagickException(exception,GetMagickModule(),OptionError,
            "UnbalancedParenthesis","`%s'",expression);
          (*recursion_depth)--;
          FxResult(0.0);
        }
      (void) CopyMagickString(subexpression,expression+1,(size_t) (q-
        expression));
      alpha=FxEvaluateSubexpression(fx_info,channel,x,y,subexpression,
        recursion_depth,beta,exception);
      (*recursion_depth)--;
      FxResult(alpha);
    }
