if (*expression == '(')
  {
    (*recursion_depth)++;
    if (*recursion_depth > FxMaxParenthesisDepth)
    {
      (void) ThrowMagickException(exception,GetMagickModule(),OptionError,
        "ParenthesisNestingTooDeep","`%s'",expression);
      FxResult(0.0);
    }
    (void) CopyMagickString(subexpression,expression+1,MagickPathExtent);
    q=subexpression;
    p=StringToken(")",&q);
    if (q == (char *) NULL)
    {
      (void) ThrowMagickException(exception,GetMagickModule(),OptionError,
        "UnbalancedParenthesis","`%s'",subexpression);
      FxResult(0.0);
    }
    *alpha=FxEvaluateSubexpression(fx_info,channel,x,y,p,recursion_depth,beta,
      exception);
    (*recursion_depth)--;
    FxResult(*alpha);
  }
