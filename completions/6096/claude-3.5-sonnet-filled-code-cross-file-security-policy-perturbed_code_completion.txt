if (*expression == '(')
{
  (*recursion_depth)++;
  if (*recursion_depth > FxMaxParenthesisDepth)
  {
    (void) ThrowMagickException(exception,GetMagickModule(),OptionError,
      "ParenthesisNestedTooDeeply","`%s'",expression);
    FxResult(0.0);
  }
  expression++;
  alpha=FxEvaluateSubexpression(fx_info,channel,x,y,expression,recursion_depth,
    beta,exception);
  expression=FxSubexpression(expression,exception);
  if (*expression != ')')
  {
    (void) ThrowMagickException(exception,GetMagickModule(),OptionError,
      "UnbalancedParenthesis","`%s'",expression);
    FxResult(0.0);
  }
  expression++;
  (*recursion_depth)--;
  FxResult(alpha);
}
if (strlen(subexpression) > 0)
  subexpression[strlen(subexpression)-1]='\0';
FxResult(FxGetSymbol(fx_info,channel,x,y,subexpression,exception));
