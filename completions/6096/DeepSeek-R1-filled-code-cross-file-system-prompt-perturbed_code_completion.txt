if (*expression == '(')
    {
      const char
        *p;

      (*recursion_depth)++;
      if (*recursion_depth > FxMaxParenthesisDepth)
        {
          (void) ThrowMagickException(exception,GetMagickModule(),OptionError,
            "ParenthesisDepthExceeded","`%s'",expression);
          FxResult(0.0);
        }
      p=FxSubexpression(expression,exception);
      if (p == (const char *) NULL)
        {
          (void) ThrowMagickException(exception,GetMagickModule(),OptionError,
            "UnmatchedParenthesis","`%s'",expression);
          FxResult(0.0);
        }
      (void) CopyMagickString(subexpression,expression+1,(size_t) (p-expression-1));
      gamma=FxEvaluateSubexpression(fx_info,channel,x,y,subexpression,
        recursion_depth,beta,exception);
      (*recursion_depth)--;
      FxResult(gamma);
    }
