if (*expression == '(')
  {
    (*recursion_depth)++;
    if (*recursion_depth > FxMaxParenthesisDepth)
      {
        (void) ThrowMagickException(exception,GetMagickModule(),OptionError,
          "MaximumParenthesesExceeded","`%s'",expression);
        FxResult(0.0);
      }
    q=(char *) expression+1;
    p=FxSubexpression(q,exception);
    if (p == (const char *) NULL)
      {
        (void) ThrowMagickException(exception,GetMagickModule(),OptionError,
          "UnableToParseExpression","`%s'",expression);
        FxResult(0.0);
      }
    *p='\0';
    *beta=FxEvaluateSubexpression(fx_info,channel,x,y,q,recursion_depth,beta,
      exception);
    (*recursion_depth)--;
    FxResult(*beta);
  }
