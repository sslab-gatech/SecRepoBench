if (*expression == '(')
    {
      size_t
        length;

      (*recursion_depth)++;
      if (*recursion_depth > FxMaxParenthesisDepth)
        {
          (void) ThrowMagickException(exception,GetMagickModule(),OptionError,
            "ParenthesisNestedTooDeep","`%s'",expression);
          FxResult(0.0);
        }
      expression++;
      length=strlen(expression);
      if ((length > 0) && (expression[length-1] == ')'))
        {
          (void) CopyMagickString(subexpression,expression,length);
          subexpression[length-1]='\0';
        }
      else
        (void) CopyMagickString(subexpression,expression,MagickPathExtent);
      gamma=FxEvaluateSubexpression(fx_info,channel,x,y,subexpression,recursion_depth,
        beta,exception);
      (*recursion_depth)--;
      FxResult(gamma);
    }
