if (*expression == '(')
    {
      register const char
        *p;

      size_t
        depth;

      p=FxSubexpression(expression+1,exception);
      if (p == (const char *) NULL)
        FxResult(0.0);
      depth=*recursion_depth;
      (*recursion_depth)++;
      alpha=FxEvaluateSubexpression(fx_info,channel,x,y,expression+1,recursion_depth,
        beta,exception);
      (*recursion_depth)=depth;
      if (*p != ')')
        {
          (void) ThrowMagickException(exception,GetMagickModule(),OptionError,
            "UnbalancedParenthesis","`%s'",expression);
          FxResult(0.0);
        }
      FxResult(alpha);
    }
