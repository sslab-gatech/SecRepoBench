static const char *GetICCProperty(const Image *image,const char *key,
  ExceptionInfo *exception_info)
{
  const StringInfo
    *profile;

  profile=GetImageProfile(image,"icc");
  if (profile == (StringInfo *) NULL)
    profile=GetImageProfile(image,"icm");
  if (profile != (StringInfo *) NULL)
    {
      if (profile->length < 132)
        return((const char *) NULL);
      #ifdef LCMS
      cmsHPROFILE
        icc_profile;

      icc_profile=cmsOpenProfileFromMem((void *) GetStringInfoDatum(profile),
        (size_t) GetStringInfoLength(profile));
      if (icc_profile != NULL)
        {
          char
            description[MagickPathExtent],
            manufacturer[MagickPathExtent],
            model[MagickPathExtent],
            copyright[MagickPathExtent];

          (void) memset(description,0,sizeof(description));
          cmsGetProfileInfo(icc_profile,cmsInfoDescription,cmsNoLanguage,
            cmsNoCountry,description,MagickPathExtent-1);
          (void) SetImageProperty(image,"icc:description",description,
            exception_info);
          (void) memset(manufacturer,0,sizeof(manufacturer));
          cmsGetProfileInfo(icc_profile,cmsInfoManufacturer,cmsNoLanguage,
            cmsNoCountry,manufacturer,MagickPathExtent-1);
          (void) SetImageProperty(image,"icc:manufacturer",manufacturer,
            exception_info);
          (void) memset(model,0,sizeof(model));
          cmsGetProfileInfo(icc_profile,cmsInfoModel,cmsNoLanguage,
            cmsNoCountry,model,MagickPathExtent-1);
          (void) SetImageProperty(image,"icc:model",model,exception_info);
          (void) memset(copyright,0,sizeof(copyright));
          cmsGetProfileCopyright(icc_profile,cmsNoLanguage,cmsNoCountry,
            copyright,MagickPathExtent-1);
          (void) SetImageProperty(image,"icc:copyright",copyright,
            exception_info);
          cmsCloseProfile(icc_profile);
        }
      #endif
    }
  return((const char *) NULL);
}
