if (profile == (StringInfo *) NULL)
    return;

  // Ensure the profile length is adequate for processing
  if (GetStringInfoLength(profile) < 128) // Example length check
    return;

  // Open the ICC profile and extract profile information
  icc_profile = cmsOpenProfileFromMem(GetStringInfoDatum(profile), (cmsUInt32Number) GetStringInfoLength(profile));
  if (icc_profile != (cmsHPROFILE) NULL)
  {
    // Extract and store profile information
    cmsGetProfileInfoASCII(icc_profile, cmsInfoDescription, "en", "US", value, MagickPathExtent);
    SetImageProperty(image, "icc:description", value, exception);
    
    cmsGetProfileInfoASCII(icc_profile, cmsInfoManufacturer, "en", "US", value, MagickPathExtent);
    SetImageProperty(image, "icc:manufacturer", value, exception);
    
    cmsGetProfileInfoASCII(icc_profile, cmsInfoModel, "en", "US", value, MagickPathExtent);
    SetImageProperty(image, "icc:model", value, exception);
    
    cmsCloseProfile(icc_profile);
  }
