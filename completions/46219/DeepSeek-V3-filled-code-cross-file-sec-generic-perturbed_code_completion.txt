const StringInfo
    *profile;

  const char
    *property;

  (void) exception_info;
  if (image == (Image *) NULL)
    return((const char *) NULL);
  profile=GetImageProfile(image,"icc");
  if (profile == (StringInfo *) NULL)
    profile=GetImageProfile(image,"icm");
  if (profile == (StringInfo *) NULL)
    return((const char *) NULL);
  if (profile->length < 128)
    return((const char *) NULL);
#if defined(MAGICKCORE_LCMS_DELEGATE)
  {
    cmsHPROFILE
      icc_profile;

    icc_profile=cmsOpenProfileFromMem(profile->datum,profile->length);
    if (icc_profile == (cmsHPROFILE) NULL)
      return((const char *) NULL);
    property=(const char *) NULL;
    if (LocaleCompare(key,"description") == 0)
      property=GetICCProfileDescription(icc_profile);
    else if (LocaleCompare(key,"manufacturer") == 0)
      property=GetICCProfileManufacturer(icc_profile);
    else if (LocaleCompare(key,"model") == 0)
      property=GetICCProfileModel(icc_profile);
    else if (LocaleCompare(key,"copyright") == 0)
      property=GetICCProfileCopyright(icc_profile);
    (void) cmsCloseProfile(icc_profile);
    if (property != (const char *) NULL)
      {
        (void) SetImageProperty(image,key,property,exception_info);
        return(property);
      }
  }
#endif
  return((const char *) NULL);
