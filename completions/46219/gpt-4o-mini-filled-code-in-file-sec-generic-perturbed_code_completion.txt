    return((const char *) NULL);
  
  const StringInfo *icc_profile = GetImageProfile(image, key);
  if (icc_profile == (StringInfo *) NULL)
    return((const char *) NULL);

  cmsHPROFILE profile = cmsOpenProfileFromMem(GetStringInfoDatum(icc_profile), (cmsUInt32Number) GetStringInfoLength(icc_profile));
  if (profile != (cmsHPROFILE) NULL)
  {
    char description[MagickPathExtent];
    (void) cmsGetProfileInfoASCII(profile, cmsInfoDescription, "en", "US", description, MagickPathExtent);
    (void) SetImageProperty((Image *) image, key, description, exception);
    cmsCloseProfile(profile);
  }
  return((const char *) NULL);
