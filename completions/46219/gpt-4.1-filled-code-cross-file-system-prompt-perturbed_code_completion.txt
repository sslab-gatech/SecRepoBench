#if defined(MAGICKCORE_LCMS_DELEGATE)
  const StringInfo
    *icc_profile;

  icc_profile=GetImageProfile(image,"icc");
  if (icc_profile == (const StringInfo *) NULL)
    icc_profile=GetImageProfile(image,"icm");
  if (icc_profile != (const StringInfo *) NULL)
    {
      const unsigned char
        *profile;

      size_t
        length;

      profile=GetStringInfoDatum(icc_profile);
      length=GetStringInfoLength(icc_profile);
      if (length > 128)
        {
          cmsHPROFILE
            lcms_profile;

          lcms_profile=cmsOpenProfileFromMem((void *) profile,length);
          if (lcms_profile != (cmsHPROFILE) NULL)
            {
              char
                property[MagickPathExtent];

              char
                buffer[256];

              if (cmsGetProfileInfoASCII(lcms_profile,cmsInfoDescription,"en","US",buffer,sizeof(buffer)) > 0)
                {
                  (void) FormatLocaleString(property,MagickPathExtent,"icc:description");
                  (void) SetImageProperty(image,property,buffer,exception_info);
                }
              if (cmsGetProfileInfoASCII(lcms_profile,cmsInfoManufacturer,"en","US",buffer,sizeof(buffer)) > 0)
                {
                  (void) FormatLocaleString(property,MagickPathExtent,"icc:manufacturer");
                  (void) SetImageProperty(image,property,buffer,exception_info);
                }
              if (cmsGetProfileInfoASCII(lcms_profile,cmsInfoModel,"en","US",buffer,sizeof(buffer)) > 0)
                {
                  (void) FormatLocaleString(property,MagickPathExtent,"icc:model");
                  (void) SetImageProperty(image,property,buffer,exception_info);
                }
              if (cmsGetProfileInfoASCII(lcms_profile,cmsInfoCopyright,"en","US",buffer,sizeof(buffer)) > 0)
                {
                  (void) FormatLocaleString(property,MagickPathExtent,"icc:copyright");
                  (void) SetImageProperty(image,property,buffer,exception_info);
                }
              cmsCloseProfile(lcms_profile);
            }
        }
    }
#endif
  return(GetImageProperty(image,key,exception_info));
