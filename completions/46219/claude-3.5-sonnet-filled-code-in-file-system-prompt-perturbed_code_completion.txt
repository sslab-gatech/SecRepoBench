const StringInfo
    *profile;

  cmsHPROFILE
    icc_profile;

  assert(image != (Image *) NULL);
  assert(image->signature == MagickCoreSignature);
  if (image->debug != MagickFalse)
    (void) LogMagickEvent(TraceEvent,GetMagickModule(),"%s",image->filename);
  if (key == (const char *) NULL)
    return((const char *) NULL);
  profile=GetImageProfile(image,"icc");
  if (profile == (StringInfo *) NULL)
    profile=GetImageProfile(image,"icm");
  if (profile == (StringInfo *) NULL)
    return((const char *) NULL);
  icc_profile=cmsOpenProfileFromMem(GetStringInfoDatum(profile),
    (cmsUInt32Number) GetStringInfoLength(profile));
  if (icc_profile == (cmsHPROFILE) NULL)
    return((const char *) NULL);
  if (LocaleCompare(key,"icc:description") == 0)
    {
      char
        info[MagickPathExtent];

      (void) cmsGetProfileInfoASCII(icc_profile,cmsInfoDescription,"en","US",
        info,MagickPathExtent);
      (void) cmsCloseProfile(icc_profile);
      return(GetImageProperty(image,key,exception));
    }
  if (LocaleCompare(key,"icc:manufacturer") == 0)
    {
      char
        info[MagickPathExtent];

      (void) cmsGetProfileInfoASCII(icc_profile,cmsInfoManufacturer,"en","US",
        info,MagickPathExtent);
      (void) cmsCloseProfile(icc_profile);
      return(GetImageProperty(image,key,exception));
    }
  if (LocaleCompare(key,"icc:model") == 0)
    {
      char
        info[MagickPathExtent];

      (void) cmsGetProfileInfoASCII(icc_profile,cmsInfoModel,"en","US",info,
        MagickPathExtent);
      (void) cmsCloseProfile(icc_profile);
      return(GetImageProperty(image,key,exception));
    }
  if (LocaleCompare(key,"icc:copyright") == 0)
    {
      char
        info[MagickPathExtent];

      (void) cmsGetProfileInfoASCII(icc_profile,cmsInfoCopyright,"en","US",
        info,MagickPathExtent);
      (void) cmsCloseProfile(icc_profile);
      return(GetImageProperty(image,key,exception));
    }
  (void) cmsCloseProfile(icc_profile);
  return((const char *) NULL);
