const char *icc_profile = NULL;
  size_t profile_length = 0;

  // Attempt to retrieve the ICC profile using the provided keys
  icc_profile = GetImageProperty(image, "icc", exception);
  if (icc_profile == NULL) {
    icc_profile = GetImageProperty(image, "icm", exception);
  }

  // Check if the ICC profile was found
  if (icc_profile != NULL) {
    profile_length = strlen(icc_profile);
    if (profile_length > 0) {
      // Open the ICC profile and extract properties
      // (Assuming OpenICCProfile is a function that handles this securely)
      ICCProfileInfo *icc_info = OpenICCProfile(icc_profile, exception);
      if (icc_info != NULL) {
        // Store properties in the image
        SetImageProperty(image, "icc_description", icc_info->description, exception);
        SetImageProperty(image, "icc_manufacturer", icc_info->manufacturer, exception);
        SetImageProperty(image, "icc_model", icc_info->model, exception);
        SetImageProperty(image, "icc_copyright", icc_info->copyright, exception);
        
        // Clean up
        CloseICCProfile(icc_info);
      } else {
        // Handle error if ICC profile could not be opened
        (void) ThrowMagickException(exception, GetMagickModule(), OptionError,
          "UnableToOpenICCProfile", "'%s': %s", icc_profile, "Failed to open ICC profile");
      }
    }
  }
