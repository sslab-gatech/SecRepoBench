#if defined(MAGICKCORE_LCMS_DELEGATE)
  {
    const StringInfo
      *profile;

    /*
      Attempt to retrieve profile from "icc" or "icm" key.
    */
    profile=GetImageProfile(image,"icc");
    if (profile == (const StringInfo *) NULL)
      profile=GetImageProfile(image,"icm");
    /*
      If profile doesn't exist or is too small, abort.
    */
    if ((profile == (const StringInfo *) NULL) ||
        (GetStringInfoLength(profile) < 132))
      return((const char *) NULL);

    /*
      Use Little CMS to parse the profile and set ICC properties.
    */
    {
      cmsHPROFILE
        cms_profile;

      cms_profile=cmsOpenProfileFromMem(
        GetStringInfoDatum(profile),
        (cmsUInt32Number) GetStringInfoLength(profile));

      if (cms_profile != (cmsHPROFILE) NULL)
        {
          char
            prop_value[MagickPathExtent];

          /*
            Description
          */
          if (cmsGetProfileInfoASCII(cms_profile,cmsInfoDescription,
              "en","US",prop_value,sizeof(prop_value)) != 0)
            (void) SetImageProperty(image,"icc:description",prop_value);
          /*
            Manufacturer
          */
          if (cmsGetProfileInfoASCII(cms_profile,cmsInfoManufacturer,
              "en","US",prop_value,sizeof(prop_value)) != 0)
            (void) SetImageProperty(image,"icc:manufacturer",prop_value);
          /*
            Model
          */
          if (cmsGetProfileInfoASCII(cms_profile,cmsInfoModel,
              "en","US",prop_value,sizeof(prop_value)) != 0)
            (void) SetImageProperty(image,"icc:model",prop_value);
          /*
            Copyright
          */
          if (cmsGetProfileInfoASCII(cms_profile,cmsInfoCopyright,
              "en","US",prop_value,sizeof(prop_value)) != 0)
            (void) SetImageProperty(image,"icc:copyright",prop_value);

          cmsCloseProfile(cms_profile);
        }
    }
  }
#endif
  /*
    Return the requested ICC property, if it exists. Otherwise NULL.
  */
  return(GetImageProperty(image,key,exception_info));
