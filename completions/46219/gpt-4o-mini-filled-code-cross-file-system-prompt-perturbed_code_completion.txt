const char *profile;
  size_t length;

  // Attempt to retrieve the ICC profile using the provided key
  profile = GetImageProperty(image, key, exception);
  if (profile == (const char *) NULL)
    return((const char *) NULL);

  // Check the length of the profile
  length = strlen(profile);
  if (length == 0)
    return((const char *) NULL);

#ifdef LCMS
  // Open the ICC profile and extract properties
  ICCProfileInfo *icc_info = OpenICCProfile(profile, exception);
  if (icc_info == (ICCProfileInfo *) NULL)
    return((const char *) NULL);

  // Store properties in the image
  SetImageProperty(image, "icc:description", icc_info->description, exception);
  SetImageProperty(image, "icc:manufacturer", icc_info->manufacturer, exception);
  SetImageProperty(image, "icc:model", icc_info->model, exception);
  SetImageProperty(image, "icc:copyright", icc_info->copyright, exception);

  // Close the ICC profile
  CloseICCProfile(icc_info);
#endif

  return(profile);
