const StringInfo
    *profile;

  cmsHPROFILE
    icc_profile;

  char
    *name;

  assert(image != (Image *) NULL);
  assert(image->signature == MagickCoreSignature);
  if (image->debug != MagickFalse)
    (void) LogMagickEvent(TraceEvent,GetMagickModule(),"%s",image->filename);
  if (key == (const char *) NULL)
    return((const char *) NULL);
  profile=GetImageProfile(image,"icc");
  if (profile == (StringInfo *) NULL)
    profile=GetImageProfile(image,"icm");
  if (profile == (StringInfo *) NULL)
    return((const char *) NULL);
  icc_profile=cmsOpenProfileFromMem(GetStringInfoDatum(profile),
    (cmsUInt32Number) GetStringInfoLength(profile));
  if (icc_profile == (cmsHPROFILE) NULL)
    return((const char *) NULL);
  name=(char *) NULL;
#if defined(MAGICKCORE_LCMS_DELEGATE)
  if (LocaleCompare(key,"icc:description") == 0)
    name=(char *) cmsTakeProductDesc(icc_profile);
  else if (LocaleCompare(key,"icc:manufacturer") == 0)
    name=(char *) cmsTakeManufacturer(icc_profile);
  else if (LocaleCompare(key,"icc:model") == 0)
    name=(char *) cmsTakeModel(icc_profile);
  else if (LocaleCompare(key,"icc:copyright") == 0)
    name=(char *) cmsTakeCopyright(icc_profile);
#endif
  cmsCloseProfile(icc_profile);
  if (name != (char *) NULL)
    {
      (void) SetImageProperty((Image *) image,key,name,exception_info);
      name=DestroyString(name);
    }
  return(GetImageProperty(image,key,exception_info));
