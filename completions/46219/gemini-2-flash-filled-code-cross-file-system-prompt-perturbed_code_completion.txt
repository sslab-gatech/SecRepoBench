const char
    *profile;

  ExceptionInfo
    *exception;

  size_t
    length;

  /*
    Verify arguments.
  */
  if (image == (Image *) NULL)
    return((const char *) NULL);
  if (key == (const char *) NULL)
    return((const char *) NULL);
  /*
    Get the ICC profile from the image.
  */
  profile=GetImageProfile(image,key,exception_info);
  if (profile == (const char *) NULL)
    return((const char *) NULL);
  length=strlen(profile);
  if (length < 8)
    return((const char *) NULL);
#if defined(MAGICKCORE_LCMS_DELEGATE)
  {
    cmsHPROFILE
      hProfile;

    cmsInfoType
      info;

    const char
      *property;

    char
      *value;

    /*
      Open the ICC profile and extract profile information.
    */
    hProfile=cmsOpenProfileFromMem(profile,(cmsUInt32Number) length);
    if (hProfile != (cmsHPROFILE) NULL)
      {
        /*
          Extract profile information.
        */
        value=(char *) AcquireQuantumMemory(MagickPathExtent,sizeof(*value));
        if (value != (char *) NULL)
          {
            *value='\0';
            info=cmsInfoDescription;
            (void) cmsGetProfileInfoASCII(hProfile,ptDescription,info,"en",
              "US",value,(cmsUInt32Number) MagickPathExtent);
            if (*value != '\0')
              {
                property=GetImageProperty(image,"icc:description",
                  exception_info);
                if (property == (const char *) NULL)
                  (void) SetImageProperty(image,"icc:description",value,
                    exception_info);
              }
            *value='\0';
            info=cmsInfoManufacturer;
            (void) cmsGetProfileInfoASCII(hProfile,ptManufacturer,info,"en",
              "US",value,(cmsUInt32Number) MagickPathExtent);
            if (*value != '\0')
              {
                property=GetImageProperty(image,"icc:manufacturer",
                  exception_info);
                if (property == (const char *) NULL)
                  (void) SetImageProperty(image,"icc:manufacturer",value,
                    exception_info);
              }
            *value='\0';
            info=cmsInfoModel;
            (void) cmsGetProfileInfoASCII(hProfile,ptModel,info,"en","US",
              value,(cmsUInt32Number) MagickPathExtent);
            if (*value != '\0')
              {
                property=GetImageProperty(image,"icc:model",exception_info);
                if (property == (const char *) NULL)
                  (void) SetImageProperty(image,"icc:model",value,
                    exception_info);
              }
            *value='\0';
            info=cmsInfoCopyright;
            (void) cmsGetProfileInfoASCII(hProfile,ptCopyright,info,"en","US",
              value,(cmsUInt32Number) MagickPathExtent);
            if (*value != '\0')
              {
                property=GetImageProperty(image,"icc:copyright",
                  exception_info);
                if (property == (const char *) NULL)
                  (void) SetImageProperty(image,"icc:copyright",value,
                    exception_info);
              }
            value=(char *) RelinquishMagickMemory(value);
          }
        (void) cmsCloseProfile(hProfile);
      }
  }
#endif
  return(profile);
