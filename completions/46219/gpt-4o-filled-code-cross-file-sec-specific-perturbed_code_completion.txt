const StringInfo
    *profile;

  const char
    *property;

  profile=GetImageProfile(image,"icc");
  if (profile == (StringInfo *) NULL)
    profile=GetImageProfile(image,"icm");
  if (profile == (StringInfo *) NULL)
    return((const char *) NULL);

  if (GetStringInfoLength(profile) < 132)
    return((const char *) NULL);

#if defined(MAGICKCORE_LCMS_DELEGATE)
  {
    cmsHPROFILE
      icc_profile;

    icc_profile=cmsOpenProfileFromMem(GetStringInfoDatum(profile),
      (cmsUInt32Number) GetStringInfoLength(profile));
    if (icc_profile == (cmsHPROFILE) NULL)
      return((const char *) NULL);

    if (LocaleCompare(key,"icc:description") == 0)
      {
        char
          description[MagickPathExtent];

        if (cmsGetProfileInfoASCII(icc_profile,cmsInfoDescription,"en","US",
            description,sizeof(description)) != 0)
          property=ConstantString(description);
      }
    else if (LocaleCompare(key,"icc:manufacturer") == 0)
      {
        char
          manufacturer[MagickPathExtent];

        if (cmsGetProfileInfoASCII(icc_profile,cmsInfoManufacturer,"en","US",
            manufacturer,sizeof(manufacturer)) != 0)
          property=ConstantString(manufacturer);
      }
    else if (LocaleCompare(key,"icc:model") == 0)
      {
        char
          model[MagickPathExtent];

        if (cmsGetProfileInfoASCII(icc_profile,cmsInfoModel,"en","US",
            model,sizeof(model)) != 0)
          property=ConstantString(model);
      }
    else if (LocaleCompare(key,"icc:copyright") == 0)
      {
        char
          copyright[MagickPathExtent];

        if (cmsGetProfileInfoASCII(icc_profile,cmsInfoCopyright,"en","US",
            copyright,sizeof(copyright)) != 0)
          property=ConstantString(copyright);
      }
    cmsCloseProfile(icc_profile);
  }
#endif
  return(property);
