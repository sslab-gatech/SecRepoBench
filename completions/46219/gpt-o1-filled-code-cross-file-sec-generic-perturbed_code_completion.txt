{
  const StringInfo *icc_profile_info = GetImageProfile(image, "icc");
  if (icc_profile_info == (const StringInfo *) NULL)
    icc_profile_info = GetImageProfile(image, "icm");
  if ((icc_profile_info == (const StringInfo *) NULL) ||
      (icc_profile_info->length < 132))
    return (const char *) NULL;

#if defined(MAGICKCORE_LCMS_DELEGATE)
  {
    cmsHPROFILE icc_profile = cmsOpenProfileFromMem(icc_profile_info->datum,
      (cmsUInt32Number) icc_profile_info->length);
    if (icc_profile != (cmsHPROFILE) NULL)
      {
        char property[256];

        (void) memset(property, 0, sizeof(property));
        if (cmsGetProfileInfoASCII(icc_profile, cmsInfoDescription, "en", "US",
            property, sizeof(property)) > 0 && *property != '\0')
          (void) SetImageProperty((Image *) image,"icc:description",property,
            exception_info);

        (void) memset(property, 0, sizeof(property));
        if (cmsGetProfileInfoASCII(icc_profile, cmsInfoManufacturer, "en", "US",
            property, sizeof(property)) > 0 && *property != '\0')
          (void) SetImageProperty((Image *) image,"icc:manufacturer",property,
            exception_info);

        (void) memset(property, 0, sizeof(property));
        if (cmsGetProfileInfoASCII(icc_profile, cmsInfoModel, "en", "US",
            property, sizeof(property)) > 0 && *property != '\0')
          (void) SetImageProperty((Image *) image,"icc:model",property,
            exception_info);

        (void) memset(property, 0, sizeof(property));
        if (cmsGetProfileInfoASCII(icc_profile, cmsInfoCopyright, "en", "US",
            property, sizeof(property)) > 0 && *property != '\0')
          (void) SetImageProperty((Image *) image,"icc:copyright",property,
            exception_info);

        cmsCloseProfile(icc_profile);
      }
  }
#endif

  /*
    Return the specific ICC-based property requested, if it exists.
  */
  return GetImageProperty((Image *) image, key, exception_info);
}
