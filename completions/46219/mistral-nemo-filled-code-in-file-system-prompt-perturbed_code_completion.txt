import Image

def fill_masked_region(image, mask, color):
    """
    Fill the masked region of an image with a given color.

    :param image: The input image.
    :param mask: The mask image.
    :param color: The color to fill the masked region with.
    :return: The modified image.
    """
    # Create a new image with the same size as the input image
    filled_image = Image.new('RGBA', image.size)

    # Paste the input image onto the new image
    filled_image.paste(image, (0, 0))

    # Paste the mask image onto the new image, using the input image as the source
    filled_image.paste(mask, (0, 0), mask=mask)

    # Fill the masked region with the given color
    filled_image = filled_image.convert('RGBA')
    pixels = filled_image.load()
    for y in range(filled_image.size[1]):
        for x in range(filled_image.size[0]):
            if pixels[x, y][3] == 0:  # If the alpha channel is 0, fill the pixel
                pixels[x, y] = color

    # Convert the image back to RGB mode
    filled_image = filled_image.convert('RGB')

    return filled_image
