MagickExport int LocaleLowercase(const int charcode)
{
  if (charcode == EOF)
    return(charcode);
#if defined(MAGICKCORE_LOCALE_SUPPORT) && defined(MAGICKCORE_HAVE_TOLOWER_L)
  if (c_locale != (locale_t) NULL)
    return(tolower_l((int) ((unsigned char) charcode),c_locale));
#endif
  return(tolower((int) ((unsigned char) charcode)));
}
