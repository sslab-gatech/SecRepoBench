{

  if (charcode < 0 || charcode > 0x10FFFF)
    return charcode;

  if (charcode == EOF)
    return EOF;

    return charcode + 0x0020; // Convert to lowercase

    return charcode + 0x0020; // Convert to lowercase

    return charcode + 0x0020; // Convert to lowercase
  else if (charcode >= 0x037F && charcode <= 0x1FFF) // Greek letters
    return charcode + 0x0020; // Convert to lowercase
  else if (charcode >= 0x200C && charcode <= 0x200D) // Combining characters
    return charcode;
  else if (charcode >= 0x202A && charcode <= 0x202E) // Combining characters
    return charcode;
  else if (charcode >= 0x203F && charcode <= 0x2040) // Combining characters
    return charcode;
  else if (charcode >= 0x2054 && charcode <= 0x206F) // Combining characters
    return charcode;
  else if (charcode >= 0x2070 && charcode <= 0x218F) // Combining characters
    return charcode;

    return charcode;

    return charcode;

    return charcode;

  else if (charcode >= 0xF900 && charcode <= 0xFAFF) // CJK characters
    return charcode;
  else if (charcode >= 0xFE20 && charcode <= 0xFE2F) // CJK characters
