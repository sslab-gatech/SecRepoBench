if (infostring->datum == NULL)
{
  infostring->datum = (unsigned char *) AcquireQuantumMemory(length + MagickPathExtent, sizeof(unsigned char));
  if (infostring->datum == NULL)
    ThrowFatalException(ResourceLimitFatalError,"MemoryAllocationFailed");
}
else
{
  infostring->datum = (unsigned char *) ResizeQuantumMemory(infostring->datum, infostring->length + MagickPathExtent, length + MagickPathExtent, sizeof(unsigned char));
  if (infostring->datum == NULL)
    ThrowFatalException(ResourceLimitFatalError,"MemoryReallocationFailed");
}

if (source->datum != NULL)
{
  (void) memcpy((void *) (infostring->datum + infostring->length - source->length), (const void *) source->datum, source->length);
}

infostring->length = length;
