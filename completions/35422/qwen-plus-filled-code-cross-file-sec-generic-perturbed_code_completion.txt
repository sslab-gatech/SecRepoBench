if (infostring->datum == NULL)
{
  infostring->datum = (unsigned char *) AcquireQuantumMemory(length + MagickPathExtent, sizeof(*infostring->datum));
  if (infostring->datum == NULL)
    ThrowFatalException(ResourceLimitFatalError,"MemoryAllocationFailed");
}
else
{
  infostring->datum = (unsigned char *) ResizeQuantumMemory(infostring->datum, infostring->length + MagickPathExtent, length + MagickPathExtent, sizeof(*infostring->datum));
  if (infostring->datum == NULL)
    ThrowFatalException(ResourceLimitFatalError,"MemoryReallocationFailed");
}

(void) memcpy((char *) infostring->datum + infostring->length, source->datum, source->length);
infostring->length = length;
