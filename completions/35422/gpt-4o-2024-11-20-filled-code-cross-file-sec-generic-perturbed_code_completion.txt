if (infostring->datum == (unsigned char *) NULL)
  {
    // Allocate memory for the concatenated data
    infostring->datum = (unsigned char *) AcquireQuantumMemory(length + 1, sizeof(*infostring->datum));
    if (infostring->datum == (unsigned char *) NULL)
      ThrowFatalException(ResourceLimitFatalError, "MemoryAllocationFailed");
    // Copy the source data into the newly allocated memory
    (void) memcpy(infostring->datum, source->datum, source->length);
  }
  else
  {
    // Resize the existing memory to accommodate the concatenated data
    unsigned char *resized_datum = (unsigned char *) ResizeQuantumMemory(infostring->datum, length + 1, sizeof(*infostring->datum));
    if (resized_datum == (unsigned char *) NULL)
      ThrowFatalException(ResourceLimitFatalError, "MemoryAllocationFailed");
    infostring->datum = resized_datum;
    // Append the source data to the existing data
    (void) memcpy(infostring->datum + infostring->length, source->datum, source->length);
  }
  // Update the length of the destination StringInfo
  infostring->length = length;
  // Null-terminate the concatenated data
  infostring->datum[length] = '\0';
