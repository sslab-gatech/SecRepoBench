// Concatenate the source StringInfo's data to the destination StringInfo's data.
  // If the current destination data is NULL, allocate new memory; otherwise, 
  // resize existing memory. 
  // Copy the source data to the end of the destination's data.
  if (infostring->datum == NULL) {
    infostring->datum = (unsigned char *) AcquireQuantumMemory(length + 1, sizeof(char));
    if (infostring->datum == NULL) {
      ThrowFatalException(ResourceLimitFatalError, "MemoryAllocationFailed");
    }
    (void) memcpy(infostring->datum, source->datum, source->length);
  } else {
    unsigned char *new_datum = (unsigned char *) AcquireQuantumMemory(length + 1, sizeof(char));
    if (new_datum == NULL) {
      ThrowFatalException(ResourceLimitFatalError, "MemoryAllocationFailed");
    }
    (void) memcpy(new_datum, infostring->datum, infostring->length);
    (void) memcpy(new_datum + infostring->length, source->datum, source->length);
    (void) AcquireQuantumMemory(length + 1, sizeof(char));
    infostring->datum = new_datum;
  }
  infostring->length = length;
