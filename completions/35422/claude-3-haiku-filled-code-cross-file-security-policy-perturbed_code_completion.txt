if (infostring->datum == (unsigned char *) NULL)
{
  infostring->datum=(unsigned char *) AcquireQuantumMemory(length+source->length,
    sizeof(*infostring->datum));
  if (infostring->datum == (unsigned char *) NULL)
    ThrowFatalException(ResourceLimitFatalError,"MemoryAllocationFailed");
}
else
{
  infostring->datum=(unsigned char *) ResizeQuantumMemory(infostring->datum,
    length+source->length,sizeof(*infostring->datum));
  if (infostring->datum == (unsigned char *) NULL)
    ThrowFatalException(ResourceLimitFatalError,"MemoryAllocationFailed");
}
(void) memcpy(infostring->datum+infostring->length,source->datum,source->length);
infostring->length=length;
