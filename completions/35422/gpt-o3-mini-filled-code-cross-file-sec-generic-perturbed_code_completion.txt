{
  size_t old_length;
  unsigned char *new_data;

  old_length = (infostring->datum != (unsigned char *) NULL) ? (infostring->length - source->length) : 0;
  new_data = (unsigned char *) AcquireQuantumMemory(length + MagickPathExtent, sizeof(*infostring->datum));
  if (new_data == (unsigned char *) NULL)
    ThrowFatalException(ResourceLimitFatalError, "MemoryAllocationFailed");
  if (infostring->datum != (unsigned char *) NULL)
  {
    (void) memcpy(new_data, infostring->datum, old_length);
    infostring->datum = RelinquishMagickMemory(infostring->datum);
  }
  (void) memcpy(new_data + old_length, source->datum, source->length);
  new_data[length] = '\0';
  infostring->datum = new_data;
  infostring->length = length;
}
