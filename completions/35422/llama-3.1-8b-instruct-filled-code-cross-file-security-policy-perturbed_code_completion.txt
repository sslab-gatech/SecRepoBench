// Concatenate the source StringInfo's data to the destination StringInfo's data.
// If the current destination data is NULL, allocate new memory; otherwise, 
// resize existing memory. 
// Copy the source data to the end of the destination's data.
if (infostring->datum == (unsigned char *) NULL)
  {
    infostring->datum=(unsigned char *) AcquireQuantumMemory(length+source->length,
      sizeof(*infostring->datum));
  }
else
  {
    size_t
      new_length;

    new_length=infostring->length+source->length;
    if (~new_length < MagickPathExtent)
      ThrowFatalException(ResourceLimitFatalError,"MemoryAllocationFailed");
    infostring->datum=(unsigned char *) AcquireQuantumMemory(new_length,
      sizeof(*infostring->datum));
    (void) memcpy(infostring->datum+infostring->length,source->datum,source->length);
  }
infostring->length+=source->length;
infostring->path=ConcatenateMagickString(infostring->path,source->path,infostring->length);
infostring->name=ConcatenateMagickString(infostring->name,source->name,infostring->length);
