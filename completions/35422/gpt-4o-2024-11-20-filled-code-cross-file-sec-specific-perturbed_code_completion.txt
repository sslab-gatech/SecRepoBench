if (infostring->datum == (unsigned char *) NULL)
  {
    // Allocate new memory for the concatenated data
    infostring->datum = (unsigned char *) AcquireCriticalMemory(length + 1);
    if (infostring->datum == (unsigned char *) NULL)
      ThrowFatalException(ResourceLimitFatalError, "MemoryAllocationFailed");
    (void) memcpy(infostring->datum, source->datum, source->length);
  }
  else
  {
    // Resize the existing memory to accommodate the concatenated data
    unsigned char *new_datum = (unsigned char *) ResizeQuantumMemory(
      infostring->datum, length + 1, sizeof(*infostring->datum));
    if (new_datum == (unsigned char *) NULL)
      ThrowFatalException(ResourceLimitFatalError, "MemoryAllocationFailed");
    infostring->datum = new_datum;
    (void) memcpy(infostring->datum + infostring->length, source->datum, source->length);
  }
  infostring->length = length;
  infostring->datum[length] = '\0'; // Null-terminate the concatenated string
