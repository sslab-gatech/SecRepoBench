{
  unsigned char
    *new_datum;
  size_t
    new_length = infostring->length + source->length,
    allocation = new_length + MagickPathExtent;

  if (infostring->datum == (unsigned char *) NULL)
  {
    new_datum = (unsigned char *) AcquireQuantumMemory(allocation, sizeof(*new_datum));
    if (new_datum == (unsigned char *) NULL)
      ThrowFatalException(ResourceLimitFatalError, "MemoryAllocationFailed");
    if (source->length > 0)
      (void) memcpy(new_datum, source->datum, source->length);
  }
  else
  {
    new_datum = (unsigned char *) ResizeQuantumMemory(infostring->datum, infostring->length + MagickPathExtent, allocation, sizeof(*new_datum));
    if (new_datum == (unsigned char *) NULL)
      ThrowFatalException(ResourceLimitFatalError, "MemoryAllocationFailed");
    (void) memcpy(new_datum + infostring->length, source->datum, source->length);
  }
  new_datum[new_length] = '\0';
  infostring->datum = new_datum;
  infostring->length = new_length;
}
