{
  size_t old_length = infostring->length - source->length;
  unsigned char *new_data;
  /* If there is no existing data, allocate new memory */
  if (infostring->datum == (unsigned char *) NULL)
  {
    new_data=(unsigned char *) AcquireQuantumMemory(length+MagickPathExtent,
      sizeof(*new_data));
    if (new_data == (unsigned char *) NULL)
      ThrowFatalException(ResourceLimitFatalError,"MemoryAllocationFailed");
    if (source->length > 0)
      (void) memcpy(new_data, source->datum, source->length);
    new_data[source->length]='\0';
  }
  /* Otherwise, resize the existing memory and append the source data */
  else
  {
    new_data=(unsigned char *) ResizeQuantumMemory(infostring->datum,
      length+MagickPathExtent, sizeof(*new_data));
    if (new_data == (unsigned char *) NULL)
      ThrowFatalException(ResourceLimitFatalError,"MemoryAllocationFailed");
    (void) memcpy(new_data+old_length, source->datum, source->length);
    new_data[length]='\0';
  }
  infostring->datum=new_data;
  infostring->length = length;
}
