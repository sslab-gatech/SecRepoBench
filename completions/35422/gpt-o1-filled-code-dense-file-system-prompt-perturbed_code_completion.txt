{
  size_t old_length = infostring->length;
  infostring->length = length;
  if (infostring->datum == (unsigned char *) NULL)
    {
      infostring->datum=(unsigned char *) AcquireQuantumMemory(
        infostring->length+MagickPathExtent,sizeof(*infostring->datum));
      if (infostring->datum == (unsigned char *) NULL)
        ThrowFatalException(ResourceLimitFatalError,"MemoryAllocationFailed");
      (void) memcpy(infostring->datum,source->datum,source->length);
    }
  else
    {
      infostring->datum=(unsigned char *) ResizeQuantumMemory(infostring->datum,
        infostring->length+MagickPathExtent,sizeof(*infostring->datum));
      if (infostring->datum == (unsigned char *) NULL)
        ThrowFatalException(ResourceLimitFatalError,"MemoryAllocationFailed");
      (void) memcpy(infostring->datum+old_length,source->datum,source->length);
    }
