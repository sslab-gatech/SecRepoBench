if (infostring->datum == NULL)
  {
    infostring->datum = (unsigned char *) AcquireQuantumMemory(length + source->length + 1, sizeof(*(infostring->datum)));
    if (infostring->datum == NULL)
      ThrowFatalException(ResourceLimitFatalError, "MemoryAllocationFailed");
    memcpy(infostring->datum, source->datum, source->length);
    infostring->datum[source->length] = '\0';
  }
  else
  {
    unsigned char *new_datum = (unsigned char *) ResizeQuantumMemory(infostring->datum, length + source->length + 1, sizeof(*(infostring->datum)));
    if (new_datum == NULL)
      ThrowFatalException(ResourceLimitFatalError, "MemoryAllocationFailed");
    infostring->datum = new_datum;
    memcpy(infostring->datum + length, source->datum, source->length);
  }
  infostring->length = length + source->length;
  infostring->datum[infostring->length] = '\0';
