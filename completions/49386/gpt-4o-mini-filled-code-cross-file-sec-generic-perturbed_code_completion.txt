cmsTagEntry TagEntries[MAX_TABLE_TAG];
    cmsTagSignature TagSignatures[MAX_TABLE_TAG];
    cmsUInt32Number TagOffsets[MAX_TABLE_TAG];
    cmsUInt32Number TagSizes[MAX_TABLE_TAG];

    for (i = 0; i < TagCount; i++) {
        cmsTagEntry* entry = &TagEntries[i];

        // Read the tag signature
        if (!_cmsReadUInt32Number(io, &entry->Signature)) return FALSE;

        // Read the tag offset
        if (!_cmsReadUInt32Number(io, &entry->Offset)) return FALSE;

        // Read the tag size
        if (!_cmsReadUInt32Number(io, &entry->Size)) return FALSE;

        // Validate the tag offset and size
        if (entry->Offset + entry->Size > HeaderSize) {
            cmsSignalError(Icc->ContextID, cmsERROR_RANGE, "Invalid tag offset/size");
            return FALSE;
        }

        // Store the tag information
        Icc->TagSignatures[Icc->TagCount] = entry->Signature;
        Icc->TagOffsets[Icc->TagCount] = entry->Offset;
        Icc->TagSizes[Icc->TagCount] = entry->Size;
        Icc->TagCount++;
    }

    // Set links for the tags
    SetLinks(Icc);

    return TRUE;
