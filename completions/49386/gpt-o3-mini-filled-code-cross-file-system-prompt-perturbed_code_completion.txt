for (i = 0; i < TagCount; i++) {

    // Read one tag entry from the IO handler
    if (io->Read(io, &TagEntry, sizeof(cmsTagEntry), 1) != 1)
        return FALSE;

    // Adjust endianness for tag signature, offset, and size
    TagEntry.sig    = _cmsAdjustEndianess32(TagEntry.sig);
    TagEntry.offset = _cmsAdjustEndianess32(TagEntry.offset);
    TagEntry.size   = _cmsAdjustEndianess32(TagEntry.size);

    // Sanity check: tag must lie within the file bounds
    if ((TagEntry.offset + TagEntry.size) > io->ReportedSize)
        continue;

    // Record the tag entry in the Icc structure
    Icc->TagSignatures[Icc->TagCount] = TagEntry.sig;
    Icc->TagOffsets[Icc->TagCount]      = TagEntry.offset;
    Icc->TagSizes[Icc->TagCount]        = TagEntry.size;
    Icc->TagLinked[Icc->TagCount]       = 0;  // initialize as not linked

    // Check for previously recorded tags with the same offset and size.
    for (j = 0; j < Icc->TagCount; j++) {
        if (Icc->TagOffsets[j] == TagEntry.offset &&
            Icc->TagSizes[j]   == TagEntry.size) {
            Icc->TagLinked[Icc->TagCount] = Icc->TagSignatures[j];
            break;
        }
    }

    Icc->TagCount++;
}

return TRUE;
