for (i = 0; i < TagCount; i++) {
    if (io->Read(io, &TagEntry, sizeof(cmsTagEntry), 1)!= 1) {
        return FALSE;
    }

    // Adjust endianness of the tag entry
    TagEntry.offset = _cmsAdjustEndianess32(TagEntry.offset);
    TagEntry.size = _cmsAdjustEndianess32(TagEntry.size);

    // Check if the tag offset and size are within the bounds of the file
    if (TagEntry.offset + TagEntry.size > Icc->IOhandler->ReportedSize) {
        cmsSignalError(Icc->ContextID, cmsERROR_RANGE, "Tag offset or size out of bounds");
        return FALSE;
    }

    // Record the tag signature, offset, and size in the Icc structure
    Icc->TagSignatures[i] = TagEntry.signature;
    Icc->TagOffsets[i] = TagEntry.offset;
    Icc->TagSizes[i] = TagEntry.size;

    // Search for links between tags that share the same offset and size
    Icc->TagLinked[i] = (cmsTagSignature) 0;
    for (j = 0; j < i; j++) {
        if (Icc->TagOffsets[j] == TagEntry.offset && Icc->TagSizes[j] == TagEntry.size) {
            Icc->TagLinked[i] = Icc->TagSignatures[j];
            break;
        }
    }

    // Increment the tag count
    Icc->TagCount++;
}
