for (i = 0; i < TagCount; i++) {
        if (io->Read(io, &TagEntry, sizeof(cmsTagEntry), 1) != 1) {
            return FALSE;
        }

        TagEntry.sig = (cmsTagSignature)_cmsAdjustEndianess32(TagEntry.sig);
        TagEntry.offset = _cmsAdjustEndianess32(TagEntry.offset);
        TagEntry.size = _cmsAdjustEndianess32(TagEntry.size);

        // Sanity check
        if (TagEntry.offset + TagEntry.size > HeaderSize) {
            cmsSignalError(Icc->ContextID, cmsERROR_CORRUPTION_DETECTED, "Tag '%4.4s' is out of profile bounds", (char*)&TagEntry.sig);
            return FALSE;
        }

        Icc->TagNames[i] = TagEntry.sig;
        Icc->TagOffsets[i] = TagEntry.offset;
        Icc->TagSizes[i] = TagEntry.size;
        Icc->TagPtrs[i] = NULL;
        Icc->TagTypeHandlers[i] = NULL;
        Icc->TagSaveAsRaw[i] = FALSE;
        Icc->TagLinked[i] = (cmsTagSignature)0;
        Icc->TagCount++;

        // Check for linked tags
        for (j = 0; j < i; j++) {
            if (Icc->TagOffsets[j] == Icc->TagOffsets[i] && Icc->TagSizes[j] == Icc->TagSizes[i]) {
                Icc->TagLinked[i] = Icc->TagNames[j];
                break;
            }
        }
    }

    return TRUE;
