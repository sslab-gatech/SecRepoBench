for (i = 0; i < TagCount; i++) {
    if (!_cmsReadTagSignature(io, &TagEntry.sig)) return FALSE;
    if (!_cmsReadUInt32Number(io, &TagEntry.offset)) return FALSE;
    if (!_cmsReadUInt32Number(io, &TagEntry.size)) return FALSE;

    if (TagEntry.offset + TagEntry.size > HeaderSize) {
        cmsSignalError(Icc->ContextID, cmsERROR_PARSE, "Tag offset/size outside the profile size");
        return FALSE;
    }

    if (Icc->TagCount >= MAX_TABLE_TAG) {
        cmsSignalError(Icc->ContextID, cmsERROR_RANGE, "Too many tags");
        return FALSE;
    }

    Icc->TagSignatures[Icc->TagCount] = TagEntry.sig;
    Icc->TagOffsets[Icc->TagCount] = TagEntry.offset;
    Icc->TagSizes[Icc->TagCount] = TagEntry.size;
    Icc->TagLinked[Icc->TagCount] = (cmsTagSignature)0;

    for (j = 0; j < Icc->TagCount; j++) {
        if (TagEntry.offset == Icc->TagOffsets[j] && TagEntry.size == Icc->TagSizes[j]) {
            Icc->TagLinked[Icc->TagCount] = Icc->TagSignatures[j];
            break;
        }
    }

    Icc->TagCount++;
}

return SetLinks(Icc);
