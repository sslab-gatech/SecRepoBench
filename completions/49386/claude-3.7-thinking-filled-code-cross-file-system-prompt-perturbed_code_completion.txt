for (i=0; i < TagCount; i++) {
    cmsTagSignature sig;
    cmsUInt32Number offset, size;

    // Read tag signature, offset, and size
    if (!_cmsReadUInt32Number(io, &sig)) return FALSE;
    if (!_cmsReadUInt32Number(io, &offset)) return FALSE;
    if (!_cmsReadUInt32Number(io, &size)) return FALSE;

    // Adjust endianness
    sig = (cmsTagSignature) _cmsAdjustEndianess32((cmsUInt32Number) sig);
    offset = _cmsAdjustEndianess32(offset);
    size = _cmsAdjustEndianess32(size);

    // Check for integrity
    if (offset >= HeaderSize || (offset + size) > HeaderSize) {
        continue; // Skip this tag as it's out of bounds
    }

    // Store in ICC directory
    Icc->TagNames[Icc->TagCount] = sig;
    Icc->TagOffsets[Icc->TagCount] = offset;
    Icc->TagSizes[Icc->TagCount] = size;
    Icc->TagLinked[Icc->TagCount] = (cmsTagSignature) 0;

    // Check for linked tags
    for (j=0; j < Icc->TagCount; j++) {
        if (Icc->TagOffsets[j] == offset && Icc->TagSizes[j] == size) {
            Icc->TagLinked[Icc->TagCount] = Icc->TagNames[j];
            break;
        }
    }

    Icc->TagCount++;
}

return TRUE;
