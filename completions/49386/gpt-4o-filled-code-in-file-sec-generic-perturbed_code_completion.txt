// Loop through each tag entry in the profile's tag directory
    for (i = 0; i < TagCount; i++) {
        // Read the tag signature, offset, and size from the IO handler
        if (!io->Read(io, &TagEntry, sizeof(cmsTagEntry), 1)) {
            return FALSE;
        }

        // Adjust endianness for the tag entry fields
        TagEntry.sig = _cmsAdjustEndianess32(TagEntry.sig);
        TagEntry.offset = _cmsAdjustEndianess32(TagEntry.offset);
        TagEntry.size = _cmsAdjustEndianess32(TagEntry.size);

        // Perform basic sanity checks to ensure the tag falls within the bounds of the file
        if (TagEntry.offset + TagEntry.size > HeaderSize) {
            cmsSignalError(Icc->ContextID, cmsERROR_CORRUPTION_DETECTED, "Tag %d is out of profile bounds", i);
            return FALSE;
        }

        // Record the tag signature, offset, and size in the Icc structure
        Icc->TagNames[i] = TagEntry.sig;
        Icc->TagOffsets[i] = TagEntry.offset;
        Icc->TagSizes[i] = TagEntry.size;
        Icc->TagPtrs[i] = NULL;
        Icc->TagSaveAsRaw[i] = FALSE;
        Icc->TagLinked[i] = (cmsTagSignature)0;
        Icc->TagTypeHandlers[i] = NULL;

        // Search for and record any links between tags that share the same offset and size
        for (j = 0; j < i; j++) {
            if (Icc->TagOffsets[j] == TagEntry.offset && Icc->TagSizes[j] == TagEntry.size) {
                Icc->TagLinked[i] = Icc->TagNames[j];
                break;
            }
        }

        // Increment the tag count for each valid tag entry processed
        Icc->TagCount++;
    }

    return TRUE;
