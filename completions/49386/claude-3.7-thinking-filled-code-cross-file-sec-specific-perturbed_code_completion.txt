for (i=0; i < TagCount; i++) {
    
    // Read the tag entry
    if (!_cmsReadUInt32Number(io, &TagEntry.sig)) return FALSE;
    if (!_cmsReadUInt32Number(io, &TagEntry.offset)) return FALSE;
    if (!_cmsReadUInt32Number(io, &TagEntry.size)) return FALSE;
    
    // Adjust endianness
    TagEntry.sig = _cmsAdjustEndianess32(TagEntry.sig);
    TagEntry.offset = _cmsAdjustEndianess32(TagEntry.offset);
    TagEntry.size = _cmsAdjustEndianess32(TagEntry.size);
    
    // Check for zero size
    if (TagEntry.size == 0) continue;
    
    // Validate offset and size within file bounds, preventing integer overflow
    if (TagEntry.offset >= HeaderSize || 
        TagEntry.size >= HeaderSize || 
        HeaderSize - TagEntry.offset < TagEntry.size) {
        continue; // Skip invalid tag
    }
    
    // Ensure we don't exceed the maximum number of tags
    if (Icc->TagCount >= MAX_TABLE_TAG) break;
    
    // Store tag in ICC profile
    Icc->TagNames[Icc->TagCount] = (cmsTagSignature) TagEntry.sig;
    Icc->TagOffsets[Icc->TagCount] = TagEntry.offset;
    Icc->TagSizes[Icc->TagCount] = TagEntry.size;
    Icc->TagLinked[Icc->TagCount] = (cmsTagSignature) 0;
    
    // Search for linked tags (tags with same offset and size)
    for (j=0; j < Icc->TagCount; j++) {
        if (Icc->TagOffsets[j] == TagEntry.offset && 
            Icc->TagSizes[j] == TagEntry.size) {
            Icc->TagLinked[Icc->TagCount] = Icc->TagNames[j];
            break;
        }
    }
    
    Icc->TagCount++;
}

// Make sure all links are properly set
SetLinks(Icc);

return TRUE;
