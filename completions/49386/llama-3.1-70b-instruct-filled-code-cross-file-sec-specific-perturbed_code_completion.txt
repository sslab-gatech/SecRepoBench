for (i = 0; i < TagCount; i++) {
    if (!_cmsReadTagEntry(io, &TagEntry)) return FALSE;

    // Basic sanity checks
    if (_cmsAdjustEndianess32(TagEntry.offset) + _cmsAdjustEndianess32(TagEntry.size) > HeaderSize) {
        cmsSignalError(Icc ->ContextID, cmsERROR_RANGE, "Tag %d out of bounds", i);
        return FALSE;
    }

    // Record tag signature, offset, and size
    Icc ->TagSignatures[i] = _cmsAdjustEndianess32(TagEntry.signature);
    Icc ->TagOffsets[i]    = _cmsAdjustEndianess32(TagEntry.offset);
    Icc ->TagSizes[i]      = _cmsAdjustEndianess32(TagEntry.size);

    // Search for links between tags
    Icc ->TagLinked[i] = (cmsTagSignature) 0;
    for (j = 0; j < i; j++) {
        if (Icc ->TagOffsets[i] == Icc ->TagOffsets[j] &&
            Icc ->TagSizes[i] == Icc ->TagSizes[j]) {
            Icc ->TagLinked[i] = Icc ->TagSignatures[j];
            break;
        }
    }

    Icc -> TagCount++;
}
