for (i = 0; i < TagCount; i++) {
    if (!_cmsReadUInt32Number(io, &TagEntry.sig)) return FALSE;
    if (!_cmsReadUInt32Number(io, &TagEntry.offset)) return FALSE;
    if (!_cmsReadUInt32Number(io, &TagEntry.size)) return FALSE;
    
    // Adjust endianness
    TagEntry.sig = _cmsAdjustEndianess32(TagEntry.sig);
    TagEntry.offset = _cmsAdjustEndianess32(TagEntry.offset);
    TagEntry.size = _cmsAdjustEndianess32(TagEntry.size);
    
    // Validate offset is within file
    if (TagEntry.offset < HeaderSize || TagEntry.offset > Icc->IOhandler->ReportedSize) {
        cmsSignalError(Icc->ContextID, cmsERROR_CORRUPTION, "Tag offset beyond file");
        continue;  // Skip this tag but try to read others
    }

    // Validate size is reasonable
    if (TagEntry.size == 0 || TagEntry.size > Icc->IOhandler->ReportedSize) {
        cmsSignalError(Icc->ContextID, cmsERROR_CORRUPTION, "Tag size is invalid");
        continue;
    }

    // Validate tag doesn't extend beyond file bounds (check for integer overflow too)
    if (TagEntry.offset > Icc->IOhandler->ReportedSize - TagEntry.size) {
        cmsSignalError(Icc->ContextID, cmsERROR_CORRUPTION, "Tag extends beyond file bounds");
        continue;
    }
    
    // Make sure we don't exceed array bounds
    if (Icc->TagCount >= MAX_TABLE_TAG) {
        cmsSignalError(Icc->ContextID, cmsERROR_RANGE, "Too many tags");
        return FALSE;
    }
    
    // Store the tag
    Icc->TagNames[Icc->TagCount] = (cmsTagSignature) TagEntry.sig;
    Icc->TagOffsets[Icc->TagCount] = TagEntry.offset;
    Icc->TagSizes[Icc->TagCount] = TagEntry.size;
    Icc->TagLinked[Icc->TagCount] = (cmsTagSignature) 0;  // No link by default
    
    // Check for links
    for (j = 0; j < Icc->TagCount; j++) {
        if (Icc->TagOffsets[j] == TagEntry.offset && Icc->TagSizes[j] == TagEntry.size) {
            // This tag is linked to a previous one
            Icc->TagLinked[Icc->TagCount] = Icc->TagNames[j];
            break;
        }
    }
    
    Icc->TagCount++;
}

return TRUE;
