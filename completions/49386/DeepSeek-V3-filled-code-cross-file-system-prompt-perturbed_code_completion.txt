for (i = 0; i < TagCount; i++) {

    if (!_cmsReadUInt32Number(io, (cmsUInt32Number*) &TagEntry.sig)) return FALSE;
    if (!_cmsReadUInt32Number(io, &TagEntry.offset)) return FALSE;
    if (!_cmsReadUInt32Number(io, &TagEntry.size)) return FALSE;

    // Adjust endianness
    TagEntry.sig    = (cmsTagSignature) _cmsAdjustEndianess32(TagEntry.sig);
    TagEntry.offset = _cmsAdjustEndianess32(TagEntry.offset);
    TagEntry.size   = _cmsAdjustEndianess32(TagEntry.size);

    // Basic sanity check
    if (TagEntry.offset + TagEntry.size > HeaderSize) {
        cmsSignalError(Icc ->ContextID, cmsERROR_CORRUPTION_DETECTED, "Corrupted tag in profile");
        return FALSE;
    }

    // Check if this tag is already in the directory (possible linked tags)
    for (j = 0; j < Icc -> TagCount; j++) {
        if (TagEntry.offset == Icc ->TagOffsets[j] && 
            TagEntry.size == Icc ->TagSizes[j]) {
            Icc ->TagLinked[Icc ->TagCount] = Icc ->TagNames[j];
            break;
        }
    }

    // Store the tag entry
    Icc ->TagNames[Icc ->TagCount]   = TagEntry.sig;
    Icc ->TagOffsets[Icc ->TagCount] = TagEntry.offset;
    Icc ->TagSizes[Icc ->TagCount]   = TagEntry.size;
    Icc ->TagCount++;
}

// Resolve any linked tags
if (!SetLinks(Icc)) return FALSE;

return TRUE;
