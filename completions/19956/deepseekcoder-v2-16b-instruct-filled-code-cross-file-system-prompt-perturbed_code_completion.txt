// Serialize each VarRegion in the region map
    for (const auto &item : regionmapping) {
        const hb_bimap_t::const_iterator &src_region = item.first;
        const hb_bimap_t::const_iterator &dst_region = item.second;

        // Serialize the VarRegion object
        if (!serialize(c, src->get_region(src_region), dst_region)) {
            return false;
        }
    }

    // Serialize the VarRegionList object
    if (!serialize_VarRegionList(c, src)) {
        return false;
    }

    return true;
