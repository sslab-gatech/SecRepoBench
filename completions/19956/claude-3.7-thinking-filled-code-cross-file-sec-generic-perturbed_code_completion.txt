TRACE_SERIALIZE (this);
VarRegionList *dest = c->extend_min (*this);
if (unlikely (dest == nullptr)) return false;

dest->axisCount = src->axisCount;
dest->regionCount = regionmapping.get_population ();

if (unlikely (!c->extend (dest))) return false;

for (unsigned int r = 0; r < regionmapping.get_population (); r++)
{
  unsigned int region = regionmapping.backward (r);
  for (unsigned int i = 0; i < src->axisCount; i++)
    dest->get_region_coords ()[r * src->axisCount + i] = src->get_region_coords ()[region * src->axisCount + i];
}

return true;
