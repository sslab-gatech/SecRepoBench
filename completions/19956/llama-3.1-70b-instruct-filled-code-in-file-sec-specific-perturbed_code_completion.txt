bool serialize (hb_serialize_context_t *c,
                const VarRegionList *src,
                const hb_bimap_t &regionmapping)
{
    TRACE_SERIALIZE (this);
    if (unlikely (!c->extend_min (*this))) return_trace (false);

    axisCount = src->axisCount;
    regionCount = src->regionCount;

    unsigned int size = min_size + VarRegionAxis::static_size * axisCount * regionCount;
    if (unlikely (!c->allocate_size<HBUINT32> (size))) return_trace (false);

    VarRegionAxis *axes = c->copy (src->axesZ.arrayZ, axisCount * regionCount);
    if (unlikely (!axes)) return_trace (false);

    for (unsigned int r = 0; r < regionCount; r++)
    {
        for (unsigned int a = 0; a < axisCount; a++)
        {
            axes[r * axisCount + a] = src->axesZ.arrayZ[r * axisCount + a];
        }
    }

    return_trace (true);
}
