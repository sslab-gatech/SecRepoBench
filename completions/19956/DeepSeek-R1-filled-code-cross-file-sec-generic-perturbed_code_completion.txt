VarRegionList *dest = c->extend_min<VarRegionList> ();
    if (unlikely (!dest)) return false;
    dest->axisCount = src->axisCount;
    dest->regionCount = regionmapping.get_population ();

    unsigned int regions_size = dest->regionCount * dest->axisCount * F2DOT14::static_size;
    if (unlikely (!c->allocate_size<F2DOT14> (regions_size))) return false;

    F2DOT14 *dest_regions = reinterpret_cast<F2DOT14*> (dest + 1);
    for (unsigned int i = 0; i < dest->regionCount; i++)
    {
      hb_codepoint_t src_idx = regionmapping.backward (i);
      const VarRegion *src_region = &src->regions[src_idx];
      for (unsigned int j = 0; j < dest->axisCount; j++)
        dest_regions[i * dest->axisCount + j] = src_region->axes[j];
    }

    return true;
