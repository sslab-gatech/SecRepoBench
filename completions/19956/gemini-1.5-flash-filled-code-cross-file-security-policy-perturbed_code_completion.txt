VarRegionList *dest = c->allocate_size<VarRegionList> (VarRegionList::get_size (src->axes.length, src->regions.length));
    if (unlikely (dest == nullptr)) return false;
    dest->axes.length = src->axes.length;
    dest->regions.length = src->regions.length;
    for (unsigned int i = 0; i < src->regions.length; i++) {
      unsigned int mapped_region = regionmapping.get (i);
      if (mapped_region >= src->regions.length) return false; //Check index is within bounds
      dest->regions[mapped_region] = src->regions[i];
    }
    return true;
