// Serialize the VarRegionList object
    VarRegionList *dest = c->extend_min (*src);
    if (unlikely (dest == nullptr)) return_trace (false);

    dest->axis_count = src->axis_count;
    dest->region_count = src->region_count;

    if (unlikely (!c->allocate_size<VarRegion> (dest->region_count * VarRegion::min_size))) return_trace (false);

    for (unsigned int i = 0; i < dest->region_count; i++)
    {
      VarRegion *region = &dest->regions[i];
      const VarRegion *src_region = &src->regions[regionmapping[i]];

      region->axis_count = src_region->axis_count;
      region->min_value = src_region->min_value;
      region->max_value = src_region->max_value;
      region->default_value = src_region->default_value;

      if (unlikely (!c->allocate_size<VarAxis> (region->axis_count * VarAxis::min_size))) return_trace (false);

      for (unsigned int j = 0; j < region->axis_count; j++)
      {
	VarAxis *axis = &region->axes[j];
	const VarAxis *src_axis = &src_region->axes[j];

	axis->tag = src_axis->tag;
	axis->name = src_axis->name;
	axis->min_value = src_axis->min_value;
	axis->max_value = src_axis->max_value;
	axis->default_value = src_axis->default_value;
      }
    }

    return_trace (true);
