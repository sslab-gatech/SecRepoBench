OffsetListOf<Lookup> LookupList::subset (hb_subset_context_t *c) const
{
  TRACE_SUBSET (this);
  auto *out = c->serializer->embed (*this);
  if (unlikely (!c->serializer->extend_min (out))) return_trace (false);

  unsigned int count = len;
  for (unsigned int i = 0; i < count; i++)
  {
    auto *lookup = c->serializer->embed (subTable[i]);
    if (unlikely (!lookup)) continue;
    lookup->serialize_subset (c, subTable[i], this, out, get_type ());
  }

  return_trace (true);
}
