if (!c || !src) {
        return false; // Check for null pointers
    }

    c->varRegionList = (VarRegionList *)malloc(sizeof(VarRegionList));
    if (!c->varRegionList) {
        return false; // Check for memory allocation failure
    }

    c->varRegionList->axisCount = src->axisCount;
    c->varRegionList->regionCount = regionmapping.size();

    // Allocate memory for axis data based on the axis count
    c->varRegionList->axisData = (AxisData *)malloc(c->varRegionList->axisCount * sizeof(AxisData));
    if (!c->varRegionList->axisData) {
        free(c->varRegionList); // Free previously allocated memory
        return false; // Check for memory allocation failure
    }

    // Iterate over each region in the region map
    for (const auto &mapping : regionmapping) {
        int srcIndex = mapping.first; // Source index
        int destIndex = mapping.second; // Destination index

        // Copy axis data from source to context
        if (srcIndex < src->axisCount) {
            c->varRegionList->axisData[destIndex] = src->axisData[srcIndex];
        }
    }

    return true; // Successful serialization
