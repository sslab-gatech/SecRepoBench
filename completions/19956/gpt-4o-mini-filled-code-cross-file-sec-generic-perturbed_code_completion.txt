if (src == nullptr || regionmapping.empty())
      return_trace (false);

    // Allocate memory for the VarRegionList in the context
    VarRegionList *dest = c->allocate_size<VarRegionList>(sizeof(VarRegionList) + src->size() * sizeof(Region));
    if (unlikely (dest == nullptr)) return_trace (false);

    // Set the axis and region counts
    dest->axis_count = src->axis_count;
    dest->region_count = src->region_count;

    // Iterate over each region in the region map
    for (const auto &mapping : regionmapping)
    {
      const VarRegion &src_region = src->at(mapping.first);
      Region &dest_region = dest->at(mapping.second);

      // Copy axis data from the source to the destination
      dest_region.axis_data = src_region.axis_data; // Assuming axis_data is a simple copyable type
      dest_region.size = src_region.size; // Assuming size is a simple copyable type
    }

    return_trace (true);
