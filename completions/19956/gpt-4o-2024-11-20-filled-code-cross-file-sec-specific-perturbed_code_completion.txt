TRACE_SERIALIZE(this);

    // Calculate the size of the VarRegionList to allocate
    unsigned int axisCount = src->axisCount;
    unsigned int regionCount = src->regionCount;
    unsigned int size = sizeof(VarRegionList) + regionCount * sizeof(VarRegion);

    // Allocate memory for the VarRegionList
    VarRegionList *dest = c->allocate_size<VarRegionList>(size);
    if (unlikely(dest == nullptr)) return_trace(false);

    // Set the axis and region counts
    dest->axisCount = axisCount;
    dest->regionCount = regionCount;

    // Iterate over each region in the region mapping
    for (unsigned int i = 0; i < regionCount; i++) {
        const VarRegion &srcRegion = src->regions[i];
        VarRegion &destRegion = dest->regions[i];

        // Copy axis data from the source to the destination
        for (unsigned int j = 0; j < axisCount; j++) {
            destRegion.axes[j].startCoord = srcRegion.axes[j].startCoord;
            destRegion.axes[j].peakCoord = srcRegion.axes[j].peakCoord;
            destRegion.axes[j].endCoord = srcRegion.axes[j].endCoord;
        }
    }

    return_trace(true);
