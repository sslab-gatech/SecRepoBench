// Allocate memory for the VarRegionList in the context and check for allocation success.
    VarRegionList *dest = c->extend_min<VarRegionList> ();
    if (unlikely (dest == nullptr)) return false;

    // Set the axis and region counts based on the source and region map.
    dest->axis_count = src->axis_count;
    dest->region_count = regionmapping.get_population();

    // Allocate additional memory as required based on the object's size calculations.
    // Ensure successful completion by returning true if the serialization process finishes without errors.
    if (unlikely (!c->allocate_size<HBUINT16> (dest->axis_count * sizeof(VarRegionList::Axis)))) return false;
    if (unlikely (!c->allocate_size<HBUINT16> (dest->region_count * sizeof(VarRegionList::Region)))) return false;

    // Iterate over each region in the region map, using its mapping to copy axis data from the source to the destination.
    for (const auto &entry : regionmapping) {
      unsigned int region_index = entry.second;
      if (region_index >= src->regions.size()) return false; // Ensure the index is within valid range
      const VarRegion &src_region = src->regions[region_index];
      VarRegion &dest_region = dest->regions[entry.first];

      // Copy axis data
      for (unsigned int axis_index = 0; axis_index < dest->axis_count; ++axis_index) {
        dest_region.axis_values[axis_index] = src_region.axis_values[axis_index];
      }
    }

    return true;
