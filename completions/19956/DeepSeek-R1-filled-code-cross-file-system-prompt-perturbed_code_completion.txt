VarRegionList *dest = c->extend_min<VarRegionList>();
    if (unlikely (!dest)) return_trace (false);
    dest->axisCount = src->axisCount;
    dest->regionCount = regionmapping.get_population ();

    unsigned int regions_size = dest->regionCount * dest->axisCount * HBFixed::static_size;
    HBFixed *regions = c->allocate_size<HBFixed> (regions_size);
    if (unlikely (!regions)) return_trace (false);

    for (unsigned int i = 0; i < dest->regionCount; i++)
    {
      unsigned int src_idx = regionmapping.backward (i);
      const VarRegion &src_region = src->regions[src_idx];
      HBFixed *dest_region = regions + i * dest->axisCount;
      for (unsigned int j = 0; j < dest->axisCount; j++)
        dest_region[j] = src_region.axes[j];
    }

    return_trace (true);
