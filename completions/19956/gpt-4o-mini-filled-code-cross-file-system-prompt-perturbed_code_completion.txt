if (!c || !src) {
        return false; // Check for null context or source
    }

    // Allocate memory for the VarRegionList in the context
    c->varRegionList = (VarRegionList *)malloc(sizeof(VarRegionList));
    if (!c->varRegionList) {
        return false; // Allocation failed
    }

    // Set axis and region counts
    c->varRegionList->axisCount = src->axisCount;
    c->varRegionList->regionCount = regionmapping.size();

    // Allocate additional memory for regions based on the count
    c->varRegionList->regions = (Region *)malloc(c->varRegionList->regionCount * sizeof(Region));
    if (!c->varRegionList->regions) {
        free(c->varRegionList); // Free previously allocated memory
        return false; // Allocation failed
    }

    // Iterate over each region in the region map
    for (const auto &mapping : regionmapping) {
        int srcIndex = mapping.first; // Get source index
        int destIndex = mapping.second; // Get destination index

        // Copy axis data from the source to the destination
        c->varRegionList->regions[destIndex] = src->regions[srcIndex];
    }

    return true; // Serialization completed successfully
