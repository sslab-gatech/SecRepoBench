TRACE_SERIALIZE (this);
    VarRegionList *dest = c->extend_min (*this);
    if (unlikely (dest == nullptr)) return_trace (false);
    
    dest->axis_count = src->axis_count;
    dest->region_count = regionmapping.size();
    
    if (unlikely (!c->allocate_size<VarRegion> (dest->region_count * sizeof(VarRegion)))) return_trace (false);
    
    for (size_t i = 0; i < regionmapping.size(); i++)
    {
        const auto &mapping = regionmapping[i];
        const VarRegion &src_region = src->regions[mapping.src_index];
        VarRegion &dest_region = dest->regions[i];
        
        dest_region.axis_data = c->allocate_size<AxisData>(src_region.axis_data_count);
        if (unlikely (dest_region.axis_data == nullptr)) return_trace (false);
        
        memcpy(dest_region.axis_data, src_region.axis_data, src_region.axis_data_count * sizeof(AxisData));
    }
    
    return_trace (true);
