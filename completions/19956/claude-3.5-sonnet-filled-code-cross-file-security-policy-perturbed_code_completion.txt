TRACE_SERIALIZE (this);
VarRegionList *dest = c->extend_min (*this);
if (unlikely (dest == nullptr)) return_trace (false);

unsigned int region_count = regionmapping.get_population ();
dest->axisCount = src->axisCount;
dest->regionCount = region_count;

if (unlikely (!c->extend (dest))) return_trace (false);

for (unsigned int i = 0; i < region_count; i++)
{
  unsigned int j = regionmapping.backward (i);
  if (unlikely (j >= src->regionCount)) return_trace (false);
  
  for (unsigned int axis = 0; axis < src->axisCount; axis++)
  {
    dest->varRegionAxes[i * src->axisCount + axis] = src->varRegionAxes[j * src->axisCount + axis];
  }
}

return_trace (true);
