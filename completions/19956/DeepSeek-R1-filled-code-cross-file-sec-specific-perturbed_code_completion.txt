VarRegionList *dest = c->extend_min (*src);
    if (unlikely (!dest)) return_trace (false);
    dest->axisCount = src->axisCount;
    dest->regionCount = regionmapping.get_population ();

    unsigned int regions_size = dest->regionCount * dest->axisCount * HBUINT16::static_size;
    HBUINT16 *regions = c->allocate_size<HBUINT16> (regions_size);
    if (unlikely (!regions)) return_trace (false);

    for (unsigned int i = 0; i < dest->regionCount; i++)
    {
        unsigned src_idx = regionmapping.backward (i);
        if (unlikely (src_idx >= src->regionCount)) return_trace (false);
        
        const HBUINT16 *src_region = src->get_region (src_idx);
        if (unlikely (!src_region)) return_trace (false);
        
        unsigned int offset = i * dest->axisCount;
        for (unsigned int j = 0; j < dest->axisCount; j++)
        {
            if (unlikely (j >= src->axisCount)) return_trace (false);
            regions[offset + j] = src_region[j];
        }
    }

    return_trace (true);
