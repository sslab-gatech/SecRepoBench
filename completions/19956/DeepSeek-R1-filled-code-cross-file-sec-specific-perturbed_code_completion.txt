VarRegionList *dest = c->extend_min<VarRegionList> ();
    if (unlikely (!dest)) return_trace (false);
    dest->axisCount = src->axisCount;
    dest->regionCount = regionmapping.get_population ();

    if (!c->allocate_size<HBINT16> (dest->regionCount * dest->axisCount))
      return_trace (false);

    for (unsigned int i = 0; i < dest->regionCount; i++)
    {
      hb_codepoint_t region_idx;
      if (!regionmapping.backward (i, &region_idx) || region_idx >= src->regionCount)
        return_trace (false);

      const VarRegion *src_region = &src->regions[region_idx];
      HBINT16 *dest_region = &dest->regions[i * dest->axisCount];
      for (unsigned int j = 0; j < dest->axisCount; j++)
        dest_region[j] = src_region->axisValues[j];
    }

    return_trace (true);
