TRACE_SERIALIZE (this);
VarRegionList *dest = c->extend_min (*this);
if (unlikely (dest == nullptr)) return false;

unsigned int axisCount = src->axisCount;
unsigned int regionCount = regionmapping.get_population ();

dest->axisCount = axisCount;
dest->regionCount = regionCount;

if (unlikely (!c->extend (dest))) return false;

VarRegionAxis *axes = c->allocate_size<VarRegionAxis> (axisCount * regionCount);
if (unlikely (axes == nullptr)) return false;

for (unsigned int r = 0; r < regionmapping.get_population (); r++)
{
  for (unsigned int a = 0; a < axisCount; a++)
  {
    unsigned int mapped_index = regionmapping.get (r);
    if (unlikely (mapped_index >= src->regionCount)) return false;
    
    axes[r * axisCount + a] = src->get_region_axes ()[mapped_index * axisCount + a];
  }
}

return true;
