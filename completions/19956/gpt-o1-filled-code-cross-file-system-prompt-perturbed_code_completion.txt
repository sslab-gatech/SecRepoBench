TRACE_SERIALIZE (this);
VarRegionList *dest = c->extend_min (*this);
if (unlikely (!dest)) return_trace (false);

// Set axis and region counts from source and mapping
dest->axisCount = src->axisCount;
dest->regionCount = regionmapping.get_count ();

// Allocate space for regions if needed
if (dest->regionCount)
{
  VarRegion *regions = c->allocate_size<VarRegion> (dest->regionCount);
  if (unlikely (!regions)) return_trace (false);

  // Copy axis data for each mapped region
  for (unsigned int i = 0; i < dest->regionCount; i++)
  {
    unsigned int sIdx = regionmapping[i];
    for (unsigned int ax = 0; ax < dest->axisCount; ax++)
      regions[i].axes[ax] = src->regions[sIdx].axes[ax];
  }
}

return_trace (true);
