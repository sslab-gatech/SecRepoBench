TRACE_SERIALIZE (this);
auto *out = c->serializer->start_embed (*this);
if (unlikely (!c->serializer->extend_min (out))) return_trace (false);

out->axisCount = src->axisCount;
out->regionCount = regionmapping.get_population ();

if (unlikely (!c->serializer->allocate_size<VarRegionAxis> (out,
							  (unsigned) out->axisCount * (unsigned) out->regionCount)))
  return_trace (false);

for (hb_codepoint_t i = 0; i < regionmapping.get_population (); i++)
{
  hb_codepoint_t j = regionmapping[i];
  for (unsigned int axis_index = 0; axis_index < src->axisCount; axis_index++)
    out->axesZ[i * out->axisCount + axis_index] = src->axesZ[j * src->axisCount + axis_index];
}

return_trace (true);
