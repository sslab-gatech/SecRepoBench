TRACE_SERIALIZE (this);
    VarRegionList *dest = c->extend_min (*this);
    if (unlikely (dest == nullptr)) return_trace (false);
    
    unsigned int region_count = regionmapping.get_population ();
    if (unlikely (!c->check_assign (dest->regionCount, region_count))) return_trace (false);
    
    unsigned int axis_count = src->get_axis_count ();
    if (unlikely (!c->check_assign (dest->axisCount, axis_count))) return_trace (false);
    
    unsigned int size = VarRegionList::min_size + VarRegion::static_size * region_count * axis_count;
    VarRegionList *new_region_list = c->allocate_size<VarRegionList> (size);
    if (unlikely (new_region_list == nullptr)) return_trace (false);
    
    for (unsigned int i = 0; i < region_count; i++)
    {
      hb_codepoint_t region_idx;
      if (!regionmapping.has (i, &region_idx)) return_trace (false);
      
      const VarRegion *src_region = &src->regions[region_idx];
      VarRegion *dest_region = &new_region_list->regions[i];
      
      for (unsigned int j = 0; j < axis_count; j++)
        dest_region->coords[j] = src_region->coords[j];
    }
    
    return_trace (true);
