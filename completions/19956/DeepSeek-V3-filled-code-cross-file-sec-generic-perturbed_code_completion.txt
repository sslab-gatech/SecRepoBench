TRACE_SERIALIZE (this);
    if (unlikely (!c->extend_min (this))) return false;

    unsigned axis_count = src->get_axis_count ();
    unsigned region_count = regionmapping.get_population ();

    if (unlikely (!c->check_assign (axisCount, axis_count))) return false;
    if (unlikely (!c->check_assign (regionCount, region_count))) return false;

    unsigned size = VarRegionList::min_size + region_count * VarRegion::get_size (axis_count);
    if (unlikely (!c->allocate_size<VarRegionList> (size))) return false;

    for (unsigned i = 0; i < region_count; i++)
    {
        hb_codepoint_t region_idx;
        if (!regionmapping.has (i, &region_idx)) return false;

        const VarRegion *src_region = &src->get_region (region_idx);
        VarRegion *dst_region = &get_region (i);

        for (unsigned j = 0; j < axis_count; j++)
            dst_region->set_axis (j, src_region->get_axis (j));
    }

    return true;
