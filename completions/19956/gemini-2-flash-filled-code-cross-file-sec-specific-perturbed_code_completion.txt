TRACE_SERIALIZE (this);

    VarRegionList *dest = c->extend_min (*this);
    if (unlikely (dest == nullptr)) return_trace (false);

    dest->axis_count = src->axis_count;
    dest->region_count = regionmapping.get_length ();

    size_t size = VarRegionList::static_size +
                  dest->axis_count * sizeof (hb_var_axis_t) +
                  dest->region_count * dest->axis_count * sizeof (float);

    if (unlikely (!c->allocate_size<char> (size - VarRegionList::static_size)))
      return_trace (false);

    /* Copy axis data */
    memcpy (dest->axes, src->axes, dest->axis_count * sizeof (hb_var_axis_t));

    /* Copy region data */
    for (unsigned int i = 0; i < dest->region_count; ++i)
    {
      unsigned int src_region_index;
      if (!regionmapping.get (i, &src_region_index))
        return_trace (false);

      memcpy (dest->region (i), src->region (src_region_index), dest->axis_count * sizeof (float));
    }

    return_trace (true);
