TRACE_SERIALIZE (this);
VarRegionList *dest = c->extend_min (*this);
if (unlikely (dest == nullptr)) return_trace (false);

dest->axisCount = src->axisCount;
dest->regionCount = regionmapping.get_population ();

unsigned int size = VarRegionList::min_size + src->axisCount * regionmapping.get_population () * VarRegion::static_size;
if (unlikely (!c->allocate_size<HBUINT8> (size))) return_trace (false);

for (auto _ : + regionmapping)
{
  unsigned int src_index = regionmapping[_];
  const VarRegion& src_region = src->regions[src_index];
  VarRegion& dest_region = dest->regions[_.index];

  for (unsigned int axis_index = 0; axis_index < src->axisCount; axis_index++)
  {
    dest_region.axes[axis_index] = src_region.axes[axis_index];
  }
}

return_trace (true);
