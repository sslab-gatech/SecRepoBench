VarRegionList *dest = c->extend_min (*this);
if (unlikely (dest == nullptr)) return false;
dest->axisCount = src->axisCount;
dest->regionCount = regionmapping.get_length ();
if (unlikely (!c->allocate_size<VarRegion> (dest->regionCount))) return false;
for (unsigned int i = 0; i < dest->regionCount; i++)
{
  const VarRegion &region = regionmapping[i];
  if (unlikely (!c->allocate_size<VarRegion> (1))) return false;
  dest->regions[i] = region;
  for (unsigned int j = 0; j < src->axisCount; j++)
    dest->regions[i].axes[j] = src->regions[regionmapping.get_mapping (i)].axes[j];
}
return true;
