TRACE_SERIALIZE (this);
    VarRegionList *dest = c->extend_min (*this);
    if (unlikely (dest == nullptr)) return_trace (false);

    unsigned region_count = regionmapping.get_population ();
    if (unlikely (!c->check_assign (dest->regionCount, region_count, HB_SERIALIZE_ERROR_INT_OVERFLOW)))
      return_trace (false);

    unsigned axis_count = src->get_axis_count ();
    if (unlikely (!c->check_assign (dest->axisCount, axis_count, HB_SERIALIZE_ERROR_INT_OVERFLOW)))
      return_trace (false);

    unsigned size = VarRegionList::min_size + VarRegion::static_size * region_count;
    VarRegionList *new_region_list = c->allocate_size<VarRegionList> (size);
    if (unlikely (new_region_list == nullptr)) return_trace (false);

    for (unsigned i = 0; i < region_count; i++)
    {
      hb_codepoint_t region_idx;
      if (!regionmapping.has (i, &region_idx)) continue;

      const VarRegion *src_region = &src->regions[region_idx];
      VarRegion *dest_region = &new_region_list->regions[i];

      for (unsigned j = 0; j < axis_count; j++)
        dest_region->coords[j] = src_region->coords[j];
    }

    return_trace (true);
