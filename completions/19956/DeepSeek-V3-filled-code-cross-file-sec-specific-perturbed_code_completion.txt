    unsigned axis_count = src->get_axis_count ();

    if (unlikely (!regionmapping.get_population ())) return false;

    if (unlikely (!c->extend_min (this))) return false;

    this->axisCount = axis_count;
    this->regionCount = count;

    unsigned size = VarRegionList::min_size + VarRegion::get_size (axis_count) * count;
    if (unlikely (!c->allocate_size<VarRegionList> (size))) return false;

    for (unsigned i = 0; i < count; i++)

        const VarRegion *src_region = &src->get_region (region_idx);
        VarRegion *dest_region = &this->get_region (i);
        dest_region->copy (src_region, axis_count);
    }
