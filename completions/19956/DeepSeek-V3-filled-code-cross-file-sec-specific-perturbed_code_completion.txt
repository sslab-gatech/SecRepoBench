      unsigned int region_idx = regionmapping.backward (i);

        return_trace (false);

      const VarRegion *src_region = &src->regions[region_idx];
      VarRegion *dest_region = &dest->regions[i];

      for (unsigned int j = 0; j < src->get_axis_count (); j++)
        dest_region->coords[j] = src_region->coords[j];
    }
