TRACE_SERIALIZE (this);
VarRegionList *dest = c->extend_min (*this);
if (unlikely (dest == nullptr)) return_trace (false);

dest->axisCount = src->axisCount;
dest->regionCount = regionmapping.get_population ();

if (unlikely (!c->extend (dest))) return_trace (false);

for (unsigned int i = 0; i < regionmapping.get_population (); i++)
{
  unsigned int j = regionmapping.backward (i);
  if (unlikely (j >= src->regionCount)) return_trace (false);

  if (unlikely (!dest->regions[i].serialize (c, src->regions[j])))
    return_trace (false);
}

return_trace (true);
