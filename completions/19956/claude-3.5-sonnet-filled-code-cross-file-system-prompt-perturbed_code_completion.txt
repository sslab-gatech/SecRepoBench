TRACE_SERIALIZE (this);
VarRegionList *dest = c->extend_min (*this);
if (unlikely (!dest)) return_trace (false);

unsigned int region_count = regionmapping.get_population ();
dest->axisCount = src->axisCount;
dest->regionCount = region_count;

if (unlikely (!c->extend (dest))) return_trace (false);

for (unsigned int i = 0; i < region_count; i++)
{
  hb_pair_t<unsigned, unsigned> mapping = regionmapping.get_next ();
  unsigned int region_index = mapping.second;
  
  for (unsigned int j = 0; j < src->axisCount; j++)
  {
    dest->varRegionRecords[i].regionAxes[j] = src->varRegionRecords[region_index].regionAxes[j];
  }
}

return_trace (true);
