VarRegionList *dest = c->allocate_size<VarRegionList> (src->get_size (regionmapping.size ()));
    if (unlikely (dest == nullptr)) return false;
    dest->axes_count = src->axes_count;
    dest->regions_count = regionmapping.size ();
    for (unsigned int i = 0; i < regionmapping.size (); i++) {
      unsigned int src_region_index = regionmapping.get_value (i);
      if (src_region_index >= src->regions_count) return false;
      dest->regions[i] = src->regions[src_region_index];
    }
    return true;
