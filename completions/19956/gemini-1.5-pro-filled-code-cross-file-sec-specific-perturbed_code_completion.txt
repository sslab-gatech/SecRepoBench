unsigned int count = src->axis_count;
    if (unlikely (!c->extend_min (*src))) return_trace (false);
    src->axes.resize (count);
    for (unsigned int i = 0; i < count; i++)
    {
      unsigned int region_index = regionmapping.get (i);
      if (region_index == HB_BIMA_VALUE_INVALID)
      {
	src->axes[i].tag = HB_TAG_NONE;
	src->axes[i].avar_table = nullptr;
	continue;
      }

      const hb_vector_t<HBUINT8> &axis_data = src->axes[region_index].axis_data;
      unsigned int axis_size = axis_data.length;

      HBUINT8 *dest_axis_data = c->allocate_size<HBUINT8> (axis_size);
      if (unlikely (!dest_axis_data)) return_trace (false);

      memcpy (dest_axis_data, axis_data.arrayZ, axis_size);
      src->axes[i].tag = src->axes[region_index].tag;
      src->axes[i].axis_data.reference_array (dest_axis_data, axis_size);
      src->axes[i].avar_table = src->axes[region_index].avar_table;
    }
    src->region_count = regionmapping.get_population ();
    return_trace (true);
