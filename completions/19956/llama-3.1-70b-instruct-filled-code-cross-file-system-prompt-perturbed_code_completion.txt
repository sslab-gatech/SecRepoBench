VarRegionList *dest = c->extend_min (*src);
if (unlikely (dest == nullptr)) return false;
dest->axis_count = regionmapping.get_size ();
dest->region_count = src->region_count;
if (unlikely (!c->allocate_size<VarRegionAxis> (dest->axis_count))) return false;
if (unlikely (!c->allocate_size<VarRegion> (dest->region_count))) return false;
for (unsigned int i = 0; i < regionmapping.get_size (); i++)
{
  unsigned int axis_index = regionmapping.get_value (i);
  dest->axes[i] = src->axes[axis_index];
}
for (unsigned int i = 0; i < src->region_count; i++)
{
  dest->regions[i] = src->regions[i];
}
return true;
