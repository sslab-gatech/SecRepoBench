TRACE_SERIALIZE (this);
    if (unlikely (!c->extend_min (this))) return false;

    unsigned axis_count = src->get_axis_count ();
    unsigned region_count = regionmapping.get_population ();

    if (unlikely (!regions.serialize (c, region_count))) return false;
    if (unlikely (!c->allocate_size<VarRegionAxis> (axis_count * region_count * VarRegionAxis::static_size))) return false;

    for (unsigned i = 0; i < region_count; i++)
    {
        hb_codepoint_t region_idx = regionmapping.get (i);
        const VarRegion &src_region = src->get_region (region_idx);
        VarRegion &dest_region = regions[i];
        for (unsigned j = 0; j < axis_count; j++)
            dest_region.get_axis (j).set_coord (src_region.get_axis (j).get_coord ());
    }

    return true;
