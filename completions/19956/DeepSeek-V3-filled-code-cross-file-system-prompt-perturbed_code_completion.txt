    dest->axisCount = src->axisCount;
    dest->regionCount = regionmapping.get_population ();

    unsigned int size = src->get_size (src->regionCount);
    VarRegionList *new_regions = c->allocate_size<VarRegionList> (size);

    for (unsigned int i = 0; i < regionmapping.get_population (); i++)

      unsigned int region_idx = regionmapping.backward (i);
      const VarRegion *src_region = &src->regions[region_idx];
      VarRegion *dst_region = &new_regions->regions[i];
      memcpy (dst_region, src_region, src->get_region_size ());
    }
