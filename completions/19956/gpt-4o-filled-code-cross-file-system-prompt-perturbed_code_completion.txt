TRACE_SERIALIZE (this);
    VarRegionList *dest = c->extend_min (*this);
    if (unlikely (dest == nullptr)) return_trace (false);

    dest->axisCount = src->axisCount;
    dest->regionCount = regionmapping.length;

    if (unlikely (!c->allocate_size<VarRegion> (dest->regionCount)))
      return_trace (false);

    for (unsigned int i = 0; i < regionmapping.length; i++)
    {
      const VarRegion &srcRegion = src->regions[regionmapping[i]];
      VarRegion *destRegion = c->allocate_size<VarRegion> ();
      if (unlikely (destRegion == nullptr)) return_trace (false);

      memcpy(destRegion, &srcRegion, sizeof(VarRegion));
    }

    return_trace (true);
