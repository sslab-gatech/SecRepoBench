// Allocate memory for the VarRegionList in the context
    if (!c->allocate(sizeof(VarRegionList)))
      return false;

    // Set the axis and region counts
    uint16_t axisCount = src->axisCount;
    uint16_t regionCount = regionmapping.size();

    // Allocate additional memory for the regions
    if (!c->allocate(regionCount * sizeof(VarRegion)))
      return false;

    // Iterate over each region in the region map
    for (uint16_t i = 0; i < regionCount; ++i)
    {
      const VarRegion &srcRegion = src->regions[regionmapping[i]];
      VarRegion *dstRegion = c->get<VarRegion>(i);

      // Copy axis data from the source to the destination
      for (uint16_t j = 0; j < axisCount; ++j)
      {
        dstRegion->axes[j] = srcRegion.axes[j];
      }
    }

    // Ensure successful completion
    return true;
