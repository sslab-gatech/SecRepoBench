VarRegionList *dst = c->allocate<VarRegionList>();
    if (!dst) return false;

    dst->axisCount = src->axisCount;
    dst->regionCount = src->regionCount;

    // Allocate memory for axes
    if (unlikely(!c->allocate_size<VarRegionAxis>(dst->get_size() - HBUINT16::static_size))) return false;

    // Copy axes data from source to destination
    for (unsigned int i = 0; i < src->regionCount; i++) {
        const VarRegionAxis &src_axis = src->axesZ[i];
        VarRegionAxis *dst_axis = &dst->axesZ[i];
        dst_axis->startCoord = src_axis.startCoord;
        dst_axis->peakCoord = src_axis.peakCoord;
        dst_axis->endCoord = src_axis.endCoord;
    }

    return true;
