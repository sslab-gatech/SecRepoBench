TRACE_SERIALIZE(this);
    VarRegionList *dest = c->extend_min(*this);
    if (unlikely(dest == nullptr)) return_trace(false);

    dest->axisCount = src->axisCount;
    dest->regionCount = regionmapping.length;

    if (unlikely(!c->allocate_size<VarRegion>(regionmapping.length)))
        return_trace(false);

    for (unsigned int i = 0; i < regionmapping.length; i++) {
        if (unlikely(i >= src->regionCount)) return_trace(false);
        const VarRegion &srcRegion = src->regions[i];
        VarRegion &destRegion = dest->regions[i];

        for (unsigned int j = 0; j < src->axisCount; j++) {
            destRegion.axis[j].startCoord = srcRegion.axis[j].startCoord;
            destRegion.axis[j].peakCoord = srcRegion.axis[j].peakCoord;
            destRegion.axis[j].endCoord = srcRegion.axis[j].endCoord;
        }
    }

    return_trace(true);
