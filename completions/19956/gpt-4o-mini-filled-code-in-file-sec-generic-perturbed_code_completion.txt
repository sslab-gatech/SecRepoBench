VarRegionList *dst = c->embed<VarRegionList>(src);
    if (unlikely (!dst)) return false;

    dst->axisCount = src->axisCount;
    dst->regionCount = src->regionCount;

    if (unlikely (!dst->axesZ.resize(src->axesZ.len))) return false;

    for (unsigned int i = 0; i < src->axesZ.len; i++)
    {
        const VarRegionAxis &src_axis = src->axesZ[i];
        VarRegionAxis *dst_axis = &dst->axesZ[i];

        dst_axis->startCoord = src_axis.startCoord;
        dst_axis->peakCoord = src_axis.peakCoord;
        dst_axis->endCoord = src_axis.endCoord;
    }

    return true;
