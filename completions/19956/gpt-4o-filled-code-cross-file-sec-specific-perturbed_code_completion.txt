// Calculate the number of regions and axes
    unsigned int num_regions = regionmapping.size();
    unsigned int num_axes = src->axisCount;

    // Calculate the size needed for the VarRegionList
    size_t size_needed = sizeof(VarRegionList) + num_regions * sizeof(Region) + num_regions * num_axes * sizeof(AxisData);

    // Allocate memory for the VarRegionList in the context
    VarRegionList *dest = (VarRegionList *)malloc(size_needed);
    if (!dest) return false; // Check for allocation success

    // Initialize the VarRegionList
    dest->regionCount = num_regions;
    dest->axisCount = num_axes;

    // Copy data from the source to the destination using the region mapping
    for (unsigned int i = 0; i < num_regions; ++i) {
        const Region &src_region = src->regions[regionmapping[i]];
        Region &dest_region = dest->regions[i];

        // Copy each axis data
        for (unsigned int j = 0; j < num_axes; ++j) {
            dest_region.axisData[j] = src_region.axisData[j];
        }
    }

    // Assign the serialized VarRegionList to the context
    c->varRegionList = dest;

    return true; // Serialization successful
