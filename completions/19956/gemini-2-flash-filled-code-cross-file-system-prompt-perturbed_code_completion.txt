TRACE_SERIALIZE (this);

    VarRegionList *dest = c->extend_min (*this);
    if (unlikely (dest == nullptr)) return_trace (false);

    dest->axis_count = src->axis_count;
    dest->region_count = regionmapping.get_length ();

    size_t size = VarRegionList::get_size (dest->axis_count, dest->region_count);
    if (size > c->remaining ())
      return_trace (false);

    /* now allocate the rest of the object */
    if (unlikely (!c->allocate_size<char> (size - VarRegionList::min_size)))
      return_trace (false);

    for (unsigned int i = 0; i < regionmapping.get_length (); ++i)
    {
      unsigned int src_index = regionmapping.get (i);
      memcpy (&dest->data[i * dest->axis_count], &src->data[src_index * src->axis_count], dest->axis_count * sizeof (VarRegionAxis));
    }

    return_trace (true);
