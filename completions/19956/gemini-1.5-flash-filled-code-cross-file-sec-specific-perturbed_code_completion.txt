size_t size = sizeof(VarRegionList) + src->num_regions * sizeof(VarRegion);
    VarRegionList *dest = (VarRegionList *)hb_malloc(c, size);
    if (!dest) return false;
    dest->num_regions = src->num_regions;
    dest->num_axes = src->num_axes;
    for (size_t i = 0; i < src->num_regions; i++) {
        size_t region_index = regionmapping.get(i);
        if (region_index >= src->num_regions) return false; //Handle out of bounds
        dest->regions[i] = src->regions[region_index];
    }
