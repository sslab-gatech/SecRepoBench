TRACE_SERIALIZE (this);
VarRegionList *dest = c->extend_min (*this);
if (unlikely (dest == nullptr)) return false;

dest->axisCount = src->axisCount;
dest->regionCount = regionmapping.get_population ();

if (unlikely (!c->extend (dest))) return false;

for (unsigned i = 0; i < regionmapping.get_population (); i++)
{
  hb_pair_t<unsigned, unsigned> mapping = regionmapping.get_min_major (i);
  unsigned idx = mapping.second;
  
  for (unsigned j = 0; j < src->axisCount; j++)
    dest->get_region_array ()[i][j] = src->get_region_array ()[idx][j];
}

return true;
