TRACE_SERIALIZE (this);
    VarRegionList *dest = c->extend_min (*this);
    if (unlikely (dest == nullptr)) return_trace (false);
    
    dest->axis_count = src->get_axis_count();
    dest->region_count = regionmapping.get_size();
    
    if (unlikely (dest->region_count > 0)) {
        VarRegion *regions = c->allocate_size<VarRegion> (VarRegion::static_size * dest->region_count);
        if (unlikely (regions == nullptr)) return_trace (false);
        dest->regions = regions;
        
        for (unsigned int i = 0; i < dest->region_count; i++) {
            const VarRegion *src_region = src->get_region(i);
            if (unlikely (src_region == nullptr)) return_trace (false);
            memcpy(&dest->regions[i], src_region, VarRegion::static_size);
        }
    }
    
    return_trace (true);
