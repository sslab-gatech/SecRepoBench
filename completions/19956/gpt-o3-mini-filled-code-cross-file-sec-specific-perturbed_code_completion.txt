VarRegionList *dest = c->allocate_size<VarRegionList> (VarRegionList::min_size + src->axisCount * sizeof(uint32_t));
if (unlikely(dest == nullptr))
  return_trace(false);

dest->axisCount = src->axisCount;
dest->regionCount = regionmapping.length;

uint32_t *dest_axes = dest->axes;
for (unsigned int i = 0; i < regionmapping.length; i++)
{
  unsigned int src_index = regionmapping[i];
  if (unlikely(src_index >= src->axisCount))
    return_trace(false);
  dest_axes[i] = src->axes[src_index];
}

return_trace(true);
