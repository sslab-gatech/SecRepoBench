{
  TRACE_SERIALIZE (this);
  VarRegionList *dest = c->extend_min (*this);
  if (unlikely (!dest)) return_trace (false);

  // Set axis and region counts
  dest->axisCount = src->axisCount;
  dest->regionCount = static_cast<uint16_t> (regionmapping.size ());

  // Calculate required size for regions
  size_t totalRegions = dest->regionCount;
  size_t regionStorageSize = totalRegions * sizeof (VarRegion);
  VarRegion *regions = c->allocate_size<VarRegion> (regionStorageSize);
  if (unlikely (!regions)) return_trace (false);

  // Copy axis data for each mapped region
  for (unsigned int i = 0; i < totalRegions; i++)
  {
    unsigned int mappedIndex = regionmapping.forward (i);
    if (unlikely (mappedIndex >= src->regionCount)) return_trace (false);
    memcpy (&regions[i], &src->regions[mappedIndex], sizeof (VarRegion));
  }

  return_trace (true);
}
