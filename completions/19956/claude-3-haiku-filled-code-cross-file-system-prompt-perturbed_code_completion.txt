unsigned int axis_count = src->get_axis_count ();
unsigned int region_count = regionmapping.get_population ();

VarRegionList *dest = c->allocate_size<VarRegionList> (VarRegionList::get_size (axis_count, region_count));
if (!dest)
  return false;

dest->init (axis_count, region_count);

for (unsigned int i = 0; i < region_count; i++)
{
  unsigned int src_region_index = regionmapping.get_value (i);
  const VarRegion *src_region = src->get_region (src_region_index);
  VarRegion *dest_region = dest->get_region (i);
  memcpy (dest_region, src_region, VarRegion::get_size (axis_count));
}

return true;
