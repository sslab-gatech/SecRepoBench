if (c == NULL || src == NULL) {
        return false; // Check for null pointers
    }

    size_t regionCount = regionmapping.size();
    c->regions = (VarRegion *)malloc(regionCount * sizeof(VarRegion));
    if (c->regions == NULL) {
        return false; // Check for successful memory allocation
    }

    c->axisCount = src->axisCount; // Assuming src has an axisCount member
    c->regionCount = regionCount;

    for (size_t i = 0; i < regionCount; ++i) {
        const auto &mapping = regionmapping[i];
        if (mapping < src->regions.size()) { // Ensure mapping is within bounds
            c->regions[i] = src->regions[mapping]; // Copy region data
        } else {
            free(c->regions); // Free allocated memory on error
            return false; // Invalid mapping
        }
    }

    return true; // Serialization successful
