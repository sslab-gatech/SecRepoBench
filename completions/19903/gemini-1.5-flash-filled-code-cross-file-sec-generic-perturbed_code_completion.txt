if (isalpha((int) ((unsigned char) *expression)) != 0)
    {
      value=FxGetSymbol(fx_info,channel,x,y,expression,depth+1,exception);
      FxReturn(value);
    }
  if (isdigit((int) ((unsigned char) *expression)) != 0)
    {
      alpha=InterpretSiPrefixValue(expression,&q);
      if (q == expression)
        {
          (void) ThrowMagickException(exception,GetMagickModule(),OptionError,
            "UnableToParseExpression","`%s'",expression);
          FxReturn(0.0);
        }
      FxReturn(alpha);
    }
  if (*expression == '(')
    {
      expression++;
      alpha=FxEvaluateSubexpression(fx_info,channel,x,y,expression,depth+1,
        beta,exception);
      while (isspace((int) ((unsigned char) *expression)) != 0)
        expression++;
      if (*expression != ')')
        {
          (void) ThrowMagickException(exception,GetMagickModule(),OptionError,
            "UnableToParseExpression","`%s'",expression);
          FxReturn(0.0);
        }
      expression++;
      FxReturn(alpha);
    }
  (void) ThrowMagickException(exception,GetMagickModule(),OptionError,
    "UnableToParseExpression","`%s'",expression);
  FxReturn(0.0);
