if (*expression == '(')
  {
    const char
      *end;

    end=FxSubexpression(expression,exception);
    if (end == (const char *) NULL)
      FxReturn(0.0);
    (void) CopyMagickString(subexpression,expression+1,(size_t) (end-expression-
      1));
    alpha=FxEvaluateSubexpression(fx_info,channel,x,y,subexpression,depth+1,beta,
      exception);
    expression=end+1;
    FxReturn(alpha);
  }
  if (IsFxFunction(expression,"abs",3) != MagickFalse)
  {
    expression+=3;
    if (*expression != '(')
      {
        (void) ThrowMagickException(exception,GetMagickModule(),OptionError,
          "UnableToParseExpression","`%s'",expression);
        FxReturn(0.0);
      }
    (void) CopyMagickString(subexpression,expression+1,MagickPathExtent);
    q=(char *) strchr(subexpression,')');
    if (q == (char *) NULL)
      {
        (void) ThrowMagickException(exception,GetMagickModule(),OptionError,
          "UnableToParseExpression","`%s'",subexpression);
        FxReturn(0.0);
      }
    *q='\0';
    alpha=FxEvaluateSubexpression(fx_info,channel,x,y,subexpression,depth+1,beta,
      exception);
    alpha=fabs(alpha);
    expression+=strlen(subexpression)+2;
    FxReturn(alpha);
  }
  if (LocaleCompare(expression,"pi") == 0)
  {
    alpha=M_PI;
    expression+=2;
    FxReturn(alpha);
  }
  if (LocaleCompare(expression,"e") == 0)
  {
    alpha=M_E;
    expression++;
    FxReturn(alpha);
  }
