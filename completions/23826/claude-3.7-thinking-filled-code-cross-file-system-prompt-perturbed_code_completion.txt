if (envvar != NULL) {
  // BLOSC_NOLOCK is set, use a new context
  dctx = blosc2_create_dctx(dparams);
  result = blosc2_decompress_ctx(dctx, src, srcsize, outputbuffer, destsize);
  blosc2_free_dctx(dctx);
  return result;
} else {
  // BLOSC_NOLOCK is not set, use mutex and global context
  pthread_mutex_lock(&global_comp_mutex);
  result = blosc2_decompress_nolock(src, srcsize, outputbuffer, destsize);
  pthread_mutex_unlock(&global_comp_mutex);
  return result;
}
