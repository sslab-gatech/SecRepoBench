import blosc2

# Define the compressor and decompression functions
def compress(data):
    # Compress the data using blosc2
    return blosc2.compress(data, typesize=1, clevel=5, nthreads=1)

def decompress(compressed_data):
    # Decompress the data using blosc2
    return blosc2.decompress(compressed_data)

# Compress the data
compressed_data = compress(data)

# Decompress the data
decompressed_data = decompress(compressed_data)
