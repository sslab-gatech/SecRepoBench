if (envvar != NULL) {
    dctx = blosc2_create_dctx(dparams);
    if (dctx == NULL) {
        return -1;  // Error creating a decompression context
    }
    result = blosc2_decompress_ctx(dctx, src, srcsize, outputbuffer, destsize);
    blosc2_free_ctx(dctx);
    return result;
} else {
    pthread_mutex_lock(&g_decompression_mutex);
    result = blosc2_decompress_ctx(g_dctx, src, srcsize, outputbuffer, destsize);
    pthread_mutex_unlock(&g_decompression_mutex);
    return result;
}
