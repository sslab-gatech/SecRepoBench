return flags;
}


static int initialize_context_decompression(
  blosc2_context* context, const void* src, int32_t srcsize, void* dest,
  int32_t destsize, int32_t typesize, int nthreads, blosc2_schunk* schunk) {

  /* Set parameters */
  context->do_compress = 0;
  context->src = (const uint8_t*)src;
  context->srcsize = srcsize;
  context->dest = (uint8_t*)dest;
  context->output_bytes = 0;
  context->destsize = destsize;
  context->sourcesize = srcsize;
  context->typesize = typesize;
  context->filter_flags = get_filter_flags(context->header_flags, typesize);
  flags_to_filters(context->filter_flags, context->filters);
  context->nthreads = nthreads;
  context->schunk = schunk;

  /* Check buffer size limits */
  if (srcsize > BLOSC_MAX_BUFFERSIZE) {
    fprintf(stderr, "Input buffer size cannot exceed %d bytes\n",
            BLOSC_MAX_BUFFERSIZE);
    return 0;
  }

  if (destsize < BLOSC_MAX_OVERHEAD) {
    fprintf(stderr, "Output buffer size should be larger than %d bytes\n",
            BLOSC_MAX_OVERHEAD);
    return -2;
  }

  /* Compute number of blocks in buffer */
  context->nblocks = context->sourcesize / context->blocksize;
  context->leftover = context->sourcesize % context->blocksize;
  context->nblocks = (context->leftover > 0) ?
                     (context->nblocks + 1) : context->nblocks;

  return 1;
}

/* Release the context for non-contextual API */
void blosc_destroy() {
  if (g_initlib == 1) {
    if (g_global_context != NULL) {
      if (g_global_context->serial_context != NULL) {
        free_thread_context(g_global_context->serial_context);
      }
      if (g_global_context->thread_contexts != NULL) {
        for (int i = 0; i < g_global_context->nthreads; i++) {
          free_thread_context(g_global_context->thread_contexts[i]);
        }
        my_free(g_global_context->thread_contexts);
      }
      my_free(g_global_context);
      g_initlib = 0;
    }
  }
}


/* Compress a buffer */
int blosc_compress(int clevel, const void* src, int32_t srcsize, void* dest,
                   int32_t destsize, int32_t typesize, const void* filters,
                   int32_t filter_nelements, void* filter_meta,
                   int32_t filter_meta_nelements, int32_t blocksize,
                   int nthreads) {
  blosc2_context context;
  int32_t ntbytes;

  /* Initialize the context */
  initialize_context_compression(&context, src, srcsize, dest, destsize,
                                 clevel, (const uint8_t*)filters,
                                 (const uint8_t*)filter_meta, typesize,
                                 g_compressor, blocksize, nthreads, nthreads,
                                 g_schunk);

  /* Do the compression */
  ntbytes = do_job(&context);

  /* Return the number of bytes compressed */
  return ntbytes;
}


/* Decompress a buffer */
int32_t blosc_decompress(const void* src, int32_t srcsize, void* dest,
                         int32_t destsize, int32_t typesize, int nthreads) {
  blosc2_context context;
  int32_t ntbytes;

  /* Initialize the context */
  initialize_context_decompression(&context, src, srcsize, dest, destsize,
                                   typesize, nthreads, g_schunk);

  /* Do the decompression */
  ntbytes = do_job(&context);

  /* Return the number of bytes decompressed */
  return ntbytes;
}
