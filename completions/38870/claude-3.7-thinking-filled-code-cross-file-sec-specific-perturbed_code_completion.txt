unz_file_info fileInfo;
// Get information about the current file
if (unzGetCurrentFileInfo(m_ZipFileHandle, &fileInfo, filename, max_filename_length - 1, nullptr, 0, nullptr, 0) != UNZ_OK) {
    continue; // Skip this file if we can't get its info
}
filename[max_filename_length - 1] = '\0'; // Ensure null-termination

// Check if the uncompressed size is not zero
if (fileInfo.uncompressed_size == 0) {
    continue; // Skip empty files
}

// Construct a string from the filename
std::string strFilename(filename);

// Simplify the filename (remove unnecessary components)
SimplifyFilename(strFilename);

// Add the file info to the archive map
m_ArchiveMap[strFilename] = ZipFileInfo(m_ZipFileHandle, fileInfo.uncompressed_size);
