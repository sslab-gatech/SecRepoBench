        char temp_filename[max_filename_length];
        if (unzGetCurrentFileInfo(m_ZipFileHandle, &file_info, temp_filename, sizeof(temp_filename), nullptr, 0, nullptr, 0) != UNZ_OK)
            continue;

        if (file_info.uncompressed_size == 0)
            continue;

        SimplifyFilename(simplified_filename);

        zip_file_info.file_handle = m_ZipFileHandle;
        zip_file_info.uncompressed_size = file_info.uncompressed_size;
