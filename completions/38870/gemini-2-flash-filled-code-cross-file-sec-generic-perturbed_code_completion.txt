unz_file_info fileInfo;
        int err = unzGetCurrentFileInfo(m_ZipFileHandle, &fileInfo, filename, sizeof(filename), nullptr, 0, nullptr, 0);
        if (err != UNZ_OK)
            continue;

        if (fileInfo.uncompressed_size == 0)
            continue;

        std::string simplifiedFilename = filename;
        SimplifyFilename(simplifiedFilename);

        m_ArchiveMap[simplifiedFilename] = ZipFileInfo(m_ZipFileHandle, fileInfo.uncompressed_size);
