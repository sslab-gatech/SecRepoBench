unz_file_info file_info;
        char temp_filename[max_filename_length];
        if (unzGetCurrentFileInfo(m_ZipFileHandle, &file_info, temp_filename, sizeof(temp_filename), nullptr, 0, nullptr) != UNZ_OK)
            continue;

        if (file_info.uncompressed_size == 0)
            continue;

        std::string simplified_filename(temp_filename);
        ReplaceAll(simplified_filename, "\\", "/");
        ReplaceAll(simplified_filename, "//", "/");

        m_ArchiveMap[simplified_filename] = ZipFileInfo(m_ZipFileHandle, file_info.uncompressed_size);
