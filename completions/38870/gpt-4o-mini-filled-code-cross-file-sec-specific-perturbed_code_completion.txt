unz_file_info fileInfo;
        char filenameBuffer[max_filename_length];
        if (unzGetCurrentFileInfo(m_ZipFileHandle, &fileInfo, filenameBuffer, sizeof(filenameBuffer), nullptr, 0, nullptr, 0) != UNZ_OK)
            continue;

        if (fileInfo.uncompressed_size == 0)
            continue;

        std::string simplifiedFilename(filenameBuffer);
        ReplaceAll(simplifiedFilename, "/", "_"); // Simplify filename by replacing slashes

        // Store the file information in the archive map
        m_ArchiveMap[simplifiedFilename] = ZipFileInfo(m_ZipFileHandle, fileInfo.uncompressed_size);
