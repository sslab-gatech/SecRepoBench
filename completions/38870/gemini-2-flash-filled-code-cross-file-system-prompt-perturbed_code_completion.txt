unz_file_info fileInfo;
        int err = unzGetCurrentFileInfo(m_ZipFileHandle, &fileInfo, filename, max_filename_length, nullptr, 0, nullptr, 0);
        if (err != UNZ_OK)
            continue;

        if (fileInfo.uncompressed_size == 0)
            continue;

        std::string simplifiedFilename = filename;
        SimplifyFilename(simplifiedFilename);

        ZipFileInfo zipInfo;
        zipInfo.m_UncompressedSize = fileInfo.uncompressed_size;
        m_ArchiveMap[simplifiedFilename] = zipInfo;
