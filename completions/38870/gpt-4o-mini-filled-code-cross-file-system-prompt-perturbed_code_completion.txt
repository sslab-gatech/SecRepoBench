unz_file_info fileInfo;
        if (unzGetCurrentFileInfo(m_ZipFileHandle, &fileInfo, filename, sizeof(filename), nullptr, 0, nullptr, 0) != UNZ_OK)
            continue;

        if (fileInfo.uncompressed_size == 0)
            continue;

        SimplifyFilename(filename);

        ZipFileInfo zipFileInfo;
        zipFileInfo.Handle = m_ZipFileHandle;
        zipFileInfo.UncompressedSize = fileInfo.uncompressed_size;

        m_ArchiveMap[filename] = zipFileInfo;
