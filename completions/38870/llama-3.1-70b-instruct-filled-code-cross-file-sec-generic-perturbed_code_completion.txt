unz_s zip_file_info;
memset(&zip_file_info, 0, sizeof(unz_s));
if (unzGetCurrentFileInfo(m_ZipFileHandle, &zip_file_info, filename, max_filename_length, NULL, 0, NULL, 0)!= UNZ_OK)
    continue;

if (zip_file_info.uncompressed_size == 0)
    continue;

std::string simplified_filename(filename);
SimplifyFilename(simplified_filename);
ReplaceAll(simplified_filename, "\\", "/");

m_ArchiveMap[simplified_filename] = ZipFileInfo(m_ZipFileHandle, zip_file_info.uncompressed_size);
