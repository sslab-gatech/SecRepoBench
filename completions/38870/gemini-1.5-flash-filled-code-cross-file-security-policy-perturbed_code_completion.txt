unz_file_info fileInfo;
        if (unzGetCurrentFileInfo(m_ZipFileHandle, &fileInfo, filename, max_filename_length, nullptr, 0, nullptr, 0) == UNZ_OK && fileInfo.uncompressed_size > 0) {
            std::string simplifiedFilename = filename;
            ReplaceAll(simplifiedFilename, "\\", "/");
            size_t pos = simplifiedFilename.find_last_of('/');
            if (pos != std::string::npos) {
                simplifiedFilename = simplifiedFilename.substr(pos + 1);
            }
            if (fileInfo.size_filename <= max_filename_length) {
                m_ArchiveMap[simplifiedFilename] = ZipFileInfo(m_ZipFileHandle, fileInfo.uncompressed_size);
            }
        }
