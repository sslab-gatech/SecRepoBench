char filename[FileNameSize];
        unz_file_info file_info;
        if (unzGetCurrentFileInfo(zip_handle, &file_info, filename, sizeof(filename), nullptr, 0, nullptr, 0) != UNZ_OK) {
            return;
        }

        if (file_info.uncompressed_size == 0) {
            return;
        }

        SimplifyFilename(filename);
        m_ArchiveMap.emplace(filename, ZipFileInfo(zip_handle, file_info.uncompressed_size));
