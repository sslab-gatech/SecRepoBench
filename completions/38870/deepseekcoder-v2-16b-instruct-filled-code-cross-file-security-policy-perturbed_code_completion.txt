unz_file_info fileInfo;
        char filePath[max_filename_length];
        if (unzGetCurrentFileInfo(m_ZipFileHandle, &fileInfo, filePath, max_filename_length, nullptr, 0, nullptr, 0) != UNZ_OK)
            continue;

        if (fileInfo.uncompressed_size <= 0)
            continue;

        if (filePath[0] == '\0')
            continue;

        std::string fullPath(filePath);
        SimplifyFilename(fullPath);

        ZipFileInfo zipInfo;
        zipInfo.fileHandle = m_ZipFileHandle;
        zipInfo.uncompressedSize = fileInfo.uncompressed_size;

        m_ArchiveMap[fullPath] = zipInfo;
