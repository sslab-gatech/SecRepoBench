// Retrieves information for the current file within the ZIP archive
unz_s unz_file_info;
if (unzGetCurrentFileInfo(m_ZipFileHandle, &unz_file_info, filename, max_filename_length, nullptr, 0, nullptr, 0)!= UNZ_OK)
    continue;

// Check the file's uncompressed size to ensure it's not zero
if (unz_file_info.uncompressed_size == 0)
    continue;

// Simplify the filename by removing unnecessary components
SimplifyFilename(filename);

// Map the file information in the archive map
m_ArchiveMap[filename] = ZipFileInfo(m_ZipFileHandle, unz_file_info.uncompressed_size);
