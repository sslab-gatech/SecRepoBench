unz_file_info file_info;
if (unzGetCurrentFileInfo(m_ZipFileHandle, &file_info, filename, sizeof(filename), nullptr, 0, nullptr, 0) != UNZ_OK)
    continue;

if (file_info.uncompressed_size == 0)
    continue;

std::string filename_str(filename);
ReplaceAll(filename_str, "\\", "/");
size_t start_pos = 0;
if (filename_str.compare(0, 2, "./") == 0)
    start_pos = 2;

filename_str = filename_str.substr(start_pos);

if (filename_str.empty())
    continue;

m_ArchiveMap[filename_str] = ZipFileInfo(filename, file_info.uncompressed_size);
