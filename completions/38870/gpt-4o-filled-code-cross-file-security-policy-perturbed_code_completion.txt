unz_file_info fileInfo;
        if (unzGetCurrentFileInfo(m_ZipFileHandle, &fileInfo, filename, max_filename_length, nullptr, 0, nullptr, 0) != UNZ_OK)
            continue;

        // Ensure the filename size does not exceed the buffer length
        if (fileInfo.size_filename > max_filename_length)
            continue;

        // Simplify the filename
        std::string simplifiedFilename(filename, fileInfo.size_filename);
        ReplaceAll(simplifiedFilename, "\\", "/");

        // Add to the archive map
        if (fileInfo.uncompressed_size > 0) {
            ZipFileInfo zipFileInfo = {m_ZipFileHandle, fileInfo.uncompressed_size};
            m_ArchiveMap[simplifiedFilename] = zipFileInfo;
        }
