unz_file_info file_info;
        if (unzGetCurrentFileInfo(m_ZipFileHandle, &file_info, filename, max_filename_length, nullptr, 0, nullptr, 0) != UNZ_OK)
            continue;

        // Skip files with zero uncompressed size
        if (file_info.uncompressed_size == 0)
            continue;

        // Convert filename to a std::string and simplify it
        std::string simplified_filename(filename);
        ReplaceAll(simplified_filename, "\\", "/");

        // Add the file information to the archive map
        ZipFileInfo zip_file_info = {m_ZipFileHandle, file_info.uncompressed_size};
        m_ArchiveMap[simplified_filename] = zip_file_info;
