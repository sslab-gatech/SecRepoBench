unz_file_info file_info;
        if (unzGetCurrentFileInfo(m_ZipFileHandle, &file_info, filename, sizeof(filename), nullptr, 0, nullptr, 0) != UNZ_OK) {
            continue;
        }

        if (file_info.uncompressed_size == 0) {
            continue;
        }

        std::string fileNameString(filename);
        SimplifyFilename(fileNameString);

        m_ArchiveMap[fileNameString] = ZipFileInfo(m_ZipFileHandle, file_info.uncompressed_size);
