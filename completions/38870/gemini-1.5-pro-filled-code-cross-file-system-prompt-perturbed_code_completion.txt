unz_file_info64 file_info;
        if (unzGetCurrentFileInfo64(m_ZipFileHandle, &file_info, filename, max_filename_length, nullptr, 0, nullptr, 0) != UNZ_OK) {
            continue;
        }

        if (file_info.uncompressed_size == 0) {
            continue;
        }

        std::string sFilename(filename);
        SimplifyFilename(sFilename);

        ZipFileInfo info;
        info.m_UncompressedSize = file_info.uncompressed_size;
        m_ArchiveMap[sFilename] = info;
