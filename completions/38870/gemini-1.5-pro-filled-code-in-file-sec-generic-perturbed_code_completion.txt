unz_file_info zip_info;
        if (unzGetCurrentFileInfo(m_ZipFileHandle, &zip_info, filename, max_filename_length, nullptr, 0, nullptr, 0) != UNZ_OK) {
            continue;
        }

        if (zip_info.uncompressed_size == 0) {
            continue;
        }

        std::string name = std::string(filename);
        SimplifyFilename(name);

        m_ArchiveMap.emplace(name, ZipFileInfo(m_ZipFileHandle, zip_info.uncompressed_size));
