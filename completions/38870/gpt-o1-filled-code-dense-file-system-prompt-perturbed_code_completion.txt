unz_file_info fileInfo;
if (unzGetCurrentFileInfo(m_ZipFileHandle, &fileInfo, filename, max_filename_length,
                          nullptr, 0, nullptr, 0) == UNZ_OK)
{
    if (fileInfo.uncompressed_size == 0)
        continue;

    std::string simplifiedName(filename);
    SimplifyFilename(simplifiedName);

    m_ArchiveMap[simplifiedName] = ZipFileInfo(m_ZipFileHandle, fileInfo.uncompressed_size);
}
