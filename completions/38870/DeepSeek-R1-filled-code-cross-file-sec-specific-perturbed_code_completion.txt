unz_file_info file_info;
        if (unzGetCurrentFileInfo(m_ZipFileHandle, &file_info, filename, sizeof(filename), nullptr, 0, nullptr, 0) != UNZ_OK)
            continue;

        if (file_info.uncompressed_size == 0)
            continue;

        std::string filenameStr(filename);
        ReplaceAll(filenameStr, "\\", "/");

        size_t start = filenameStr.find_first_not_of("./");
        if (start != std::string::npos) {
            filenameStr = filenameStr.substr(start);
        }

        if (!filenameStr.empty()) {
            m_ArchiveMap.emplace(filenameStr, ZipFileInfo(m_ZipFileHandle, file_info.uncompressed_size));
        }
