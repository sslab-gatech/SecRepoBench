unz_s zip_stat;
if (unzGetCurrentFileInfo(m_ZipFileHandle, &zip_stat, filename, FileNameSize, nullptr, 0, nullptr, 0)!= UNZ_OK)
    continue;

    continue;

SimplifyFilename(filenameStr);
