unz_file_info file_info;
if (unzGetCurrentFileInfo(m_ZipFileHandle, &file_info, filename, sizeof(filename), nullptr, 0, nullptr, 0) != UNZ_OK)
    continue;

if (file_info.uncompressed_size == 0)
    continue;

std::string simplified_filename(filename);
SimplifyFilename(simplified_filename);

ZipFileInfo zip_file_info;
zip_file_info.file_pos = unzGetOffset(m_ZipFileHandle);
zip_file_info.size = file_info.uncompressed_size;

m_ArchiveMap[simplified_filename] = zip_file_info;
