proto_tree_add_uint_format(field_tree, hf_tcp_option_sack_sre, tvb,
                                   offset + 4, 4, rightedge,
                                   "right edge = %u%s", rightedge,
                                   (tcp_analyze_seq && tcp_relative_seq) ? " (relative)" : "");

        if (tcp_analyze_seq) {
            if (tcpd) {
                if (tcp_track_bytes_in_flight && tcpd->fwd->tcp_analyze_seq_info) {
                    if (sack_range_count < MAX_SACK_RANGES) {
                        tcpd->fwd->tcp_analyze_seq_info->sack_left_edge[sack_range_count] = leftedge;
                        tcpd->fwd->tcp_analyze_seq_info->sack_right_edge[sack_range_count] = rightedge;
                    }
                }
            }
        }


        sack_range_count++;
