proto_tree_add_item(field_tree, hf_tcp_option_sack_sle, tvb, offset, 4, ENC_BIG_ENDIAN);
proto_tree_add_item(field_tree, hf_tcp_option_sack_sre, tvb, offset + 4, 4, ENC_BIG_ENDIAN);
offset += 8;
sack_range_count++;

if (sack_range_count > MAX_TCP_SACK_RANGES) {
    expert_add_info_format(pinfo, field_tree, &ei_tcp_option_sack_range_count,
                           "Too many SACK ranges (%u)", sack_range_count);
    break;
}

if (tcph != NULL && (tcph->num_sack_ranges < MAX_TCP_SACK_RANGES)) {
    tcph->sack_left_edge[tcph->num_sack_ranges] = tvb_get_ntohl(tvb, offset - 8);
    tcph->sack_right_edge[tcph->num_sack_ranges] = tvb_get_ntohl(tvb, offset - 4);
    tcph->num_sack_ranges++;
}

proto_item_append_text(field_tree, " %u-%u", tvb_get_ntohl(tvb, offset - 8), tvb_get_ntohl(tvb, offset - 4));
