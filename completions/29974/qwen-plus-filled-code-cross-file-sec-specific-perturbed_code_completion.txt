proto_tree_add_uint_format(field_tree, hf_tcp_option_sack_sre, tvb,
                           offset + 4, 4, rightedge,
                           "right edge = %u%s", rightedge,
                           (tcp_analyze_seq && tcp_relative_seq) ? " (relative)" : "");

if (tcpd && tcp_track_bytes_in_flight && tcpd->fwd->tcp_analyze_seq_info) {
    tcpd->fwd->tcp_analyze_seq_info->sack_left_edges[sack_range_count] = leftedge;
    tcpd->fwd->tcp_analyze_seq_info->sack_right_edges[sack_range_count] = rightedge;
    sack_range_count++;
    if (sack_range_count >= MAX_TCP_SACK_RANGES) {
        expert_add_info(pinfo, ti, &ei_tcp_option_sack_too_many_ranges);
        break;
    }
}
