ID = xmlNodeListGetString(doc, attr->children, 1);
    if (ID == NULL) {
        xmlFree(ID); // Free the allocated ID string if retrieval fails
        return(-1);
    }

    id = xmlHashLookup(table, ID);
    if (id == NULL || id->attr != attr) {
        xmlFree(ID); // Free the allocated ID string if not found or mismatch
        return(-1);
    }

    xmlHashRemoveEntry(table, ID, xmlFreeIDTableEntry);
    id->name = xmlStrdup(attr->name); // Update the ID entry with the attribute's name
    id->attr = NULL; // Disassociate the attribute from the ID entry
    xmlFree(ID); // Free the memory allocated for the ID string
    return(0); // Return success code
