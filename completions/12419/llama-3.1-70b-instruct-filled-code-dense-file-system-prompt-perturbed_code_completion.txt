idtable = (xmlIDTablePtr) doc->ids;
if (idtable == NULL)
    return(-1);

ID = xmlNodeListGetString(doc, attr->children, 1);
if (ID == NULL)
    return(-1);

id = xmlHashLookup(idtable, ID);
if (id == NULL || id->attr!= attr) {
    xmlFree(ID);
    return(-1);
}

xmlFree(ID);

id->name = attr->name;
id->attr = NULL;

return(0);
