            time(&time_now);

        else {
            time_now = now;

        gmtime_r(&time_now, &tmy);

        tmp[sizeof(tmp) - 1] = '\0';

        fmt = timeparser->time_fmt_year;
        p = flb_strptime(tmp, fmt, tm);
