        time_now = now;
        gmtime_r(&time_now, &tmy);

        /* Concatenate the current year with the time string */
        snprintf(tmp, sizeof(tmp) - 1, "%d %.*s",

        tmp[sizeof(tmp) - 1] = '\0';

        /* Parse the time string with the year */
        p = flb_strptime(tmp, timeparser->time_fmt_year, tm);
    }

        /* Copy the time string to a temporary buffer */
        memcpy(tmp, time_str, tsize);
