        /* Get the current year */
        time_now = now;
        gmtime_r(&time_now, &tmy);

        /* Create a temporary buffer with the year prepended */
        snprintf(tmp, sizeof(tmp), "%d %.*s", tmy.tm_year + 1900, (int)tsize, time_str);
        fmt = timeparser->time_fmt_year;
        time_ptr = tmp;
        time_len = strlen(tmp);
    }
    else {
        /* Copy the time string to a temporary buffer ensuring null-termination */
        memcpy(tmp, time_str, tsize);
        tmp[tsize] = '\0';
        fmt = timeparser->time_fmt;
    }

    /* Parse the time string */
    p = flb_strptime(time_ptr, fmt, tm);
