const int max_matrices = m->avctx->codec_id == AV_CODEC_ID_MLP ? MAX_MATRICES_MLP : MAX_MATRICES_TRUEHD;

if (m->matrix_changed++ > 1) {
    av_log(m->avctx, AV_LOG_ERROR, "Matrix parameters may change only once per access unit.\n");
    return AVERROR_INVALIDDATA;
}

s->num_primitive_matrices = get_bits(bit_context, 4);

if (s->num_primitive_matrices > max_matrices) {
    av_log(m->avctx, AV_LOG_ERROR,
           "Number of primitive matrices %d is greater than maximum %d.\n",
           s->num_primitive_matrices, max_matrices);
    return AVERROR_INVALIDDATA;
}
