unsigned int max_matrices = (m->avctx->codec_id == AV_CODEC_ID_MLP) ? 16 : 256;
    
    if (m->matrix_changed >= 1) {
        av_log(m->avctx, AV_LOG_ERROR, "Matrix configuration may change only once per access unit.\n");
        return AVERROR_INVALIDDATA;
    }

    s->num_primitive_matrices = get_bits(bit_context, 8);
    
    if (s->num_primitive_matrices > max_matrices) {
        av_log(m->avctx, AV_LOG_ERROR, "Too many primitive matrices specified.\n");
        return AVERROR_INVALIDDATA;
    }
