int max_primitive_matrices;
int num_primitive_matrices;

if (m->avctx->codec_id == AV_CODEC_ID_MLP) {
    max_primitive_matrices = 3;
} else {
    max_primitive_matrices = 4;
}

if (s->matrix_changed > 0) {
    if (s->matrix_changed > 1) {
        av_log(m->avctx, AV_LOG_ERROR,
                "Matrix configuration may change only once per access unit.\n");
        return AVERROR_INVALIDDATA;
    }
    s->matrix_changed++;
}

num_primitive_matrices = get_bits(bit_context, 2);

if (num_primitive_matrices > max_primitive_matrices) {
    av_log(m->avctx, AV_LOG_ERROR,
            "Too many primitive matrices specified.\n");
    return AVERROR_INVALIDDATA;
}

s->num_primitive_matrices = num_primitive_matrices;
