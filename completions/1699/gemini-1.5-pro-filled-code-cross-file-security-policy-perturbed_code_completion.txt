int max_primitive_matrices = (m->avctx->codec_id == AV_CODEC_ID_TRUEHD) ? MAX_TRUEHD_PRIMITIVE_MATRICES : MAX_MLP_PRIMITIVE_MATRICES;
    if (m->matrix_changed++ > 1) {
        av_log(m->avctx, AV_LOG_ERROR, "Matrix configuration may change only once per access unit.\n");
        return AVERROR_INVALIDDATA;
    }
    s->num_primitive_matrices = get_bits(bit_context, 4);

    if (s->num_primitive_matrices > max_primitive_matrices) {
        av_log(m->avctx, AV_LOG_ERROR, "Too many primitive matrices: %d.\n", s->num_primitive_matrices);
        return AVERROR_INVALIDDATA;
    } else if (s->num_primitive_matrices == 0) {
        // Handle the case where no primitive matrices are specified.
        // Set a default value to prevent issues in later processing.
        s->num_primitive_matrices = 1;
