unsigned int max_primitive_matrices;
    /* Determine the maximum allowed primitive matrices based on the codec type. */
    if (m->avctx->codec_id == AV_CODEC_ID_TRUEHD)
        max_primitive_matrices = 2;
    else
        max_primitive_matrices = 3;

    /* Ensure that the matrix configuration has not already been changed. */
    if (m->matrix_changed) {
        av_log(m->avctx, AV_LOG_ERROR,
               "Matrix configuration changed more than once per access unit.\n");
        return AVERROR_INVALIDDATA;
    }
    m->matrix_changed = 1;

    /* Retrieve the number of primitive matrices from the bitstream.
       Assume a 2-bit field with an offset of 1, yielding a value from 1 to 4. */
    s->num_primitive_matrices = get_bits(bit_context, 2) + 1;

    /* Validate that the number does not exceed the allowed maximum. */
    if (s->num_primitive_matrices > max_primitive_matrices) {
        av_log(m->avctx, AV_LOG_ERROR,
               "Invalid number of primitive matrices: %u (allowed: %u).\n",
               s->num_primitive_matrices, max_primitive_matrices);
        return AVERROR_INVALIDDATA;
    }
