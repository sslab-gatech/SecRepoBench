unsigned int max_primitive_matrices = (m->avctx->codec_id == AV_CODEC_ID_MLP) ? 16 : 32;
    if (s->num_primitive_matrices > max_primitive_matrices) {
        av_log(m->avctx, AV_LOG_ERROR,
                "Too many primitive matrices specified (%u), maximum is %u.\n",
                s->num_primitive_matrices, max_primitive_matrices);
        return AVERROR_INVALIDDATA;
    }

    if (s->matrix_config_changes > 1) {
        av_log(m->avctx, AV_LOG_ERROR,
                "Matrix configuration has been changed more than once per access unit.\n");
        return AVERROR_INVALIDDATA;
    }

    s->num_primitive_matrices = get_bits(gbp, 8);
    if (s->num_primitive_matrices > max_primitive_matrices) {
        av_log(m->avctx, AV_LOG_ERROR,
                "Too many primitive matrices specified (%u), maximum is %u.\n",
                s->num_primitive_matrices, max_primitive_matrices);
        return AVERROR_INVALIDDATA;
    }
