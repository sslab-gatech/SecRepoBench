static int read_matrix_params(MLPDecodeContext *m, unsigned int substr, GetBitContext *bit_context)
{
    SubStream *s = &m->substream[substr];
    unsigned int mat, ch;
    // Determine the maximum number of primitive matrices allowed based on the codec type.
    // Check if the matrix configuration has been changed more than once per access unit.
    // Retrieve the number of primitive matrices from the bitstream.
    // Validate that the number of primitive matrices does not exceed the allowed maximum.
    // If validation fails, log an error message and return an error code.
    if (m->avctx->codec_id == AV_CODEC_ID_MLP) {
        s->num_primitive_matrices = FFMIN(get_bits(bit_context, 4), MAX_MATRIX_CHANNEL_MLP);
    } else {
        s->num_primitive_matrices = FFMIN(get_bits(bit_context, 4), MAX_MATRIX_CHANNEL_TRUEHD);
    }
    if (s->num_primitive_matrices > 1 && s->noise_type) {
        av_log(m->avctx, AV_LOG_ERROR, "Noise type is not supported for more than one matrix.\n");
        return AVERROR_INVALIDDATA;
    }
    if (s->num_primitive_matrices > 0 && s->noise_type == 0) {
        av_log(m->avctx, AV_LOG_ERROR, "Noise type is not supported for zero matrices.\n");
        return AVERROR_INVALIDDATA;
    }
    if (s->num_primitive_matrices > 0 && s->noise_type == 1) {
        av_log(m->avctx, AV_LOG_ERROR, "Noise type is not supported for one matrix.\n");
        return AVERROR_INVALIDDATA;
    }
    if (s->num_primitive_matrices > 0 && s->noise_type == 2) {
        av_log(m->avctx, AV_LOG_ERROR, "Noise type is not supported for two matrices.\n");
        return AVERROR_INVALIDDATA;
    }
    if (s->num_primitive_matrices > 0 && s->noise_type == 3) {
        av_log(m->avctx, AV_LOG_ERROR, "Noise type is not supported for three matrices.\n");
        return AVERROR_INVALIDDATA;
    }
    if (s->num_primitive_matrices > 0 && s->noise_type == 4) {
        av_log(m->avctx, AV_LOG_ERROR, "Noise type is not supported for four matrices.\n");
        return AVERROR_INVALIDDATA;
    }
    if (s->num_primitive_matrices > 0 && s->noise_type == 5) {
        av_log(m->avctx, AV_LOG_ERROR, "Noise type is not supported for five matrices.\n");
        return AVERROR_INVALIDDATA;
    }
    if (s->num_primitive_matrices > 0 && s->noise_type == 6) {
        av_log(m->avctx, AV_LOG_ERROR, "Noise type is not supported for six matrices.\n");
        return AVERROR_INVALIDDATA;
    }
    if (s->num_primitive_matrices > 0 && s->noise_type == 7) {
        av_log(m->avctx, AV_LOG_ERROR, "Noise type is not supported for seven matrices.\n");
        return AVERROR_INVALIDDATA;
    }
    if (s->num_primitive_matrices > 0 && s->noise_type == 8) {
        av_log(m->avctx, AV_LOG_ERROR, "Noise type is not supported for eight matrices.\n");
        return AVERROR_INVALIDDATA;
    }
    if (s->num_primitive_matrices > 0 && s->noise_type == 9) {
        av_log(m->avctx, AV_LOG_ERROR, "Noise type is not supported for nine matrices.\n");
        return AVERROR_INVALIDDATA;
    }
    if (s->num_primitive_matrices > 0 && s->noise_type == 10) {
        av_log(m->avctx, AV_LOG_ERROR, "Noise type is not supported for ten matrices.\n");
        return AVERROR_INVALIDDATA;
    }
    if (s->num_primitive_matrices > 0 && s->noise_type == 11) {
        av_log(m->avctx, AV_LOG_ERROR, "Noise type is not supported for eleven matrices.\n");
        return AVERROR_INVALIDDATA;
    }
    if (s->num_primitive_matrices > 0 && s->noise_type == 12) {
        av_log(m->avctx, AV_LOG_ERROR, "Noise type is not supported for twelve matrices.\n");
        return AVERROR_INVALIDDATA;
    }
    if (s->num_primitive_matrices > 0 && s->noise_type == 13) {
        av_log(m->avctx, AV_LOG_ERROR, "Noise type is not supported for thirteen matrices.\n");
        return AVERROR_INVALIDDATA;
    }
    if (s->num_primitive_matrices > 0 && s->noise_type == 14) {
        av_log(m->avctx, AV_LOG_ERROR, "Noise type is not supported for fourteen matrices.\n");
        return AVERROR_INVALIDDATA;
    }
    if (s->num_primitive_matrices > 0 && s->noise_type == 15) {
        av_log(m->avctx, AV_LOG_ERROR, "Noise type is not supported for fifteen matrices.\n");
        return AVERROR_INVALIDDATA;
    }
    if (s->num_primitive_matrices > 0 && s->noise_type == 16) {
        av_log(m->avctx, AV_LOG_ERROR, "Noise type is not supported for sixteen matrices.\n");
        return AVERROR_INVALIDDATA;
    }
    if (s->num_primitive_matrices > 0 && s->noise_type == 17) {
        av_log(m->avctx, AV_LOG_ERROR, "Noise type is not supported for seventeen matrices.\n");
        return AVERROR_INVALIDDATA;
    }
    if (s->num_primitive_matrices > 0 && s->noise_type == 18) {
        av_log(m->avctx, AV_LOG_ERROR, "Noise type is not supported for eighteen matrices.\n");
        return AVERROR_INVALIDDATA;
    }
    if (s->num_primitive_matrices > 0 && s->noise_type == 19) {
        av_log(m->avctx, AV_LOG_ERROR, "Noise type is not supported for nineteen matrices.\n");
        return AVERROR_INVALIDDATA;
    }
    if (s->num_primitive_matrices > 0 && s->noise_type == 20) {
        av_log(m->avctx, AV_LOG_ERROR, "Noise type is not supported for twenty matrices.\n");
        return AVERROR_INVALIDDATA;
    }
    if (s->num_primitive_matrices > 0 && s->noise_type == 21) {
        av_log(m->avctx, AV_LOG_ERROR, "Noise type is not supported for twenty-one matrices.\n");
        return AVERROR_INVALIDDATA;
    }
    if (s->num_primitive_matrices > 0 && s->noise_type == 22) {
        av_log(m->avctx, AV_LOG_ERROR, "Noise type is not supported for twenty-two matrices.\n");
        return AVERROR_INVALIDDATA;
    }
    if (s->num_primitive_matrices > 0 && s->noise_type == 23) {
        av_log(m->avctx, AV_LOG_ERROR, "Noise type is not supported for twenty-three matrices.\n");
        return AVERROR_INVALIDDATA;
    }
    if (s->num_primitive_matrices > 0 && s->noise_type == 24) {
        av_log(m->avctx, AV_LOG_ERROR, "Noise type is not supported for twenty-four matrices.\n");
        return AVERROR_INVALIDDATA;
    }
    if (s->num_primitive_matrices > 0 && s->noise_type == 25) {
        av_log(m->avctx, AV_LOG_ERROR, "Noise type is not supported for twenty-five matrices.\n");
        return AVERROR_INVALIDDATA;
    }
    if (s->num_primitive_matrices > 0 && s->noise_type == 26) {
        av_log(m->avctx, AV_LOG_ERROR, "Noise type is not supported for twenty-six matrices.\n");
        return AVERROR_INVALIDDATA;
    }
    if (s->num_primitive_matrices > 0 && s->noise_type == 27) {
        av_log(m->avctx, AV_LOG_ERROR, "Noise type is not supported for twenty-seven matrices.\n");
        return AVERROR_INVALIDDATA;
    }
    if (s->num_primitive_matrices > 0 && s->noise_type == 28) {
        av_log(m->avctx, AV_LOG_ERROR, "Noise type is not supported for twenty-eight matrices.\n");
        return AVERROR_INVALIDDATA;
    }
    if (s->num_primitive_matrices > 0 && s->noise_type == 29) {
        av_log(m->avctx, AV_LOG_ERROR, "Noise type is not supported for twenty-nine matrices.\n");
        return AVERROR_INVALIDDATA;
    }
    if (s->num_primitive_matrices > 0 && s->noise_type == 30) {
        av_log(m->avctx, AV_LOG_ERROR, "Noise type is not supported for thirty matrices.\n");
        return AVERROR_INVALIDDATA;
    }
    if (s->num_primitive_matrices > 0 && s->noise_type == 31) {
        av_log(m->avctx, AV_LOG_ERROR, "Noise type is not supported for thirty-one matrices.\n");
        return AVERROR_INVALIDDATA;
    }
    for (mat = 0; mat < s->num_primitive_matrices; mat++) {
        int frac_bits, max_chan;
        s->matrix_out_ch[mat] = get_bits(bit_context, 4);
        frac_bits             = get_bits(bit_context, 4);
        s->lsb_bypass   [mat] = get_bits1(bit_context);

        if (s->matrix_out_ch[mat] > s->max_matrix_channel) {
            av_log(m->avctx, AV_LOG_ERROR,
                    "Invalid channel %d specified as output from matrix.\n",
                    s->matrix_out_ch[mat]);
            return AVERROR_INVALIDDATA;
        }
        if (frac_bits > 14) {
            av_log(m->avctx, AV_LOG_ERROR,
                    "Too many fractional bits specified.\n");
            return AVERROR_INVALIDDATA;
        }

        max_chan = s->max_matrix_channel;
        if (!s->noise_type)
            max_chan+=2;

        for (ch = 0; ch <= max_chan; ch++) {
            int coeff_val = 0;
            if (get_bits1(bit_context))
                coeff_val = get_sbits(bit_context, frac_bits + 2);

            s->matrix_coeff[mat][ch] = coeff_val * (1 << (14 - frac_bits));
        }

        if (s->noise_type)
            s->matrix_noise_shift[mat] = get_bits(bit_context, 4);
        else
            s->matrix_noise_shift[mat] = 0;
    }

    return 0;
}
