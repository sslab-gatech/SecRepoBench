// <MASK>
if (m->matrix_changed > 1) {
    av_log(m->avctx, AV_LOG_ERROR, "Matrix configuration has been changed more than once per access unit.\n");
    return AVERROR_INVALIDDATA;
}

mat = get_bits(bit_context, 4);
if (mat > m->max_decoded_substream) {
    av_log(m->avctx, AV_LOG_ERROR, "Invalid matrix index %d specified.\n", mat);
    return AVERROR_INVALIDDATA;
}

if (mat >= m->num_substreams) {
    av_log(m->avctx, AV_LOG_ERROR, "Number of primitive matrices %d exceeds the number of substreams %d.\n",
           mat, m->num_substreams);
    return AVERROR_INVALIDDATA;
}

s->num_primitive_matrices = get_bits(bit_context, 4);
if (s->num_primitive_matrices > m->max_decoded_substream) {
    av_log(m->avctx, AV_LOG_ERROR, "Number of primitive matrices %d exceeds the maximum allowed %d.\n",
           s->num_primitive_matrices, m->max_decoded_substream);
    return AVERROR_INVALIDDATA;
}
