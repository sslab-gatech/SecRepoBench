unsigned int num_primitive_matrices;

    // Retrieve the number of primitive matrices from the bitstream.
    num_primitive_matrices = get_bits(bit_context, 4);

    // Validate that the number of primitive matrices does not exceed the allowed maximum.
    if (num_primitive_matrices > MAX_PRIMITIVE_MATRICES) {
        av_log(m->avctx, AV_LOG_ERROR, "Number of primitive matrices exceeds maximum allowed.\n");
        return AVERROR_INVALIDDATA;
    }

    // Set the number of primitive matrices, capped to the permissible limit.
    s->num_primitive_matrices = FFMIN(num_primitive_matrices, MAX_PRIMITIVE_MATRICES);
