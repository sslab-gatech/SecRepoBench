if (m->avctx->codec_id == AV_CODEC_ID_TRUEHD) {
        s->max_primitive_matrices = 16;
    } else {
        s->max_primitive_matrices = 8;
    }

    if (m->matrix_changed++ > 1) {
        av_log(m->avctx, AV_LOG_ERROR,
               "Primitive matrix configuration changed more than once per access unit.\n");
        return AVERROR_INVALIDDATA;
    }

    s->num_primitive_matrices = get_bits(bit_context, 4);
    if (s->num_primitive_matrices > s->max_primitive_matrices) {
        av_log(m->avctx, AV_LOG_ERROR,
               "Number of primitive matrices %d exceeds maximum allowed %d.\n",
               s->num_primitive_matrices, s->max_primitive_matrices);
        return AVERROR_INVALIDDATA;
    }
