    // Determine the maximum number of primitive matrices allowed based on the codec type.
    const int max_primitive_matrices = m->avctx->codec_id == AV_CODEC_ID_TRUEHD? MAX_TRUEHD_MATRICES : MAX_MLP_MATRICES;

    // Check if the matrix configuration has been changed more than once per access unit.
    if (m->matrix_changed > 1) {
        av_log(m->avctx, AV_LOG_ERROR, "Matrix configuration changed more than once per access unit.\n");
        return AVERROR_INVALIDDATA;
    }

    // Retrieve the number of primitive matrices from the bitstream.
    unsigned int num_primitive_matrices = get_bits(bit_context, 4);

    // Validate that the number of primitive matrices does not exceed the allowed maximum.
    if (num_primitive_matrices > max_primitive_matrices) {
        num_primitive_matrices = max_primitive_matrices;
        av_log(m->avctx, AV_LOG_WARNING, "Number of primitive matrices exceeds maximum allowed.\n");
    }

    // If validation fails, log an error message and return an error code.
    if (num_primitive_matrices == 0) {
        av_log(m->avctx, AV_LOG_ERROR, "Number of primitive matrices is zero.\n");
        return AVERROR_INVALIDDATA;
    }

    // Set the number of primitive matrices to the safe default value if the limit is exceeded.
    s->num_primitive_matrices = FFMIN(num_primitive_matrices, max_primitive_matrices);
