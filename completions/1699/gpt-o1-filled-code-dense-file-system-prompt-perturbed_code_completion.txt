if (m->matrix_changed[substr]++ > 1) {
    av_log(m->avctx, AV_LOG_ERROR,
           "Matrix configuration may change only once per access unit.\n");
    return AVERROR_INVALIDDATA;
}

s->num_primitive_matrices = get_bits(bit_context, 4);

if (s->num_primitive_matrices > 8) {
    av_log(m->avctx, AV_LOG_ERROR,
           "Invalid number of primitive matrices: %u.\n",
           s->num_primitive_matrices);
    return AVERROR_INVALIDDATA;
}
