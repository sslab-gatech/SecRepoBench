if (m->avctx->codec_id == AV_CODEC_ID_TRUEHD) {
        s->max_matrix_channel = 15;
    } else {
        s->max_matrix_channel = 7;
    }

    if (m->matrix_changed++ > 1) {
        av_log(m->avctx, AV_LOG_ERROR,
               "Primitive matrix configuration changed more than once per access unit.\n");
        return AVERROR_INVALIDDATA;
    }

    s->num_primitive_matrices = get_bits(bit_context, 6);
    if (s->num_primitive_matrices > MAX_PRIMITIVE_MATRICES) {
        av_log(m->avctx, AV_LOG_ERROR,
               "Number of primitive matrices exceeds maximum allowed (%d > %d).\n",
               s->num_primitive_matrices, MAX_PRIMITIVE_MATRICES);
        return AVERROR_INVALIDDATA;
    }
