unsigned int max_primitive_matrices = m->avctx->codec_id == AV_CODEC_ID_MLP ? 3 : 8;
    if (s->matrix_changed++ > 1) {
        av_log(m->avctx, AV_LOG_ERROR, "Matrix configuration has changed more than once per access unit.\n");
        return AVERROR_INVALIDDATA;
    }
    s->num_primitive_matrices = get_bits(bit_context, 4);
    if (s->num_primitive_matrices > max_primitive_matrices) {
        av_log(m->avctx, AV_LOG_ERROR, "Number of primitive matrices %d exceeds maximum %d.\n",
               s->num_primitive_matrices, max_primitive_matrices);
        return AVERROR_INVALIDDATA;
    }
