int max_matrices;

    /* Check if matrix has already been changed in this access unit. */
    if (m->matrix_changed++) {
        av_log(m->avctx, AV_LOG_ERROR, "Matrix changed more than once per access unit.\n");
        return AVERROR_INVALIDDATA;
    }

    /* Determine maximum allowed primitive matrices based on the codec. */
    max_matrices = (m->avctx->codec_id == AV_CODEC_ID_MLP) ? 8 : 6;

    /* Retrieve number of primitive matrices from the bitstream. */
    s->num_primitive_matrices = get_bits(bit_context, 4);

    /* Validate number of matrices. */
    if (s->num_primitive_matrices > max_matrices) {
        av_log(m->avctx, AV_LOG_ERROR,
               "Invalid number of primitive matrices (%d) exceeds max (%d).\n",
               s->num_primitive_matrices, max_matrices);
        return AVERROR_INVALIDDATA;
    }
