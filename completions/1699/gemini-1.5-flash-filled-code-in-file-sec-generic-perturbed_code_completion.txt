int max_matrices;

    if (m->avctx->codec_id == AV_CODEC_ID_MLP)
        max_matrices = MAX_MATRICES_MLP;
    else
        max_matrices = MAX_MATRICES_TRUEHD;

    if (m->matrix_changed++ > 1) {
        av_log(m->avctx, AV_LOG_ERROR, "Matrices may change only once per access unit.\n");
        return AVERROR_INVALIDDATA;
    }

    s->num_primitive_matrices = get_bits(bit_context, 4);
    if (s->num_primitive_matrices > max_matrices) {
        av_log(m->avctx, AV_LOG_ERROR,
               "Number of primitive matrices %d exceeds maximum %d.\n",
               s->num_primitive_matrices, max_matrices);
        return AVERROR_INVALIDDATA;
    }
