unsigned int max_primitive_matrices = m->codec_type == MLP_CODEC_TRUEHD? 2 : 1;
if (s->param_presence_flags & PARAM_MATRIX) {
    if (get_bits1(bit_context)) {
        s->num_primitive_matrices = get_bits(bit_context, 4);
        if (s->num_primitive_matrices > max_primitive_matrices) {
            av_log(m->avctx, AV_LOG_ERROR,
                   "Number of primitive matrices (%d) exceeds the maximum allowed (%d).\n",
                   s->num_primitive_matrices, max_primitive_matrices);
            return AVERROR_INVALIDDATA;
        }
    }
}
