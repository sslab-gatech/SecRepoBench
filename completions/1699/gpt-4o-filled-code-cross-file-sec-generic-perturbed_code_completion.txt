unsigned int max_primitive_matrices;

    // Determine the maximum number of primitive matrices allowed based on the codec type.
    if (m->avctx->codec_id == AV_CODEC_ID_MLP) {
        max_primitive_matrices = 8;
    } else if (m->avctx->codec_id == AV_CODEC_ID_TRUEHD) {
        max_primitive_matrices = 12;
    } else {
        av_log(m->avctx, AV_LOG_ERROR, "Unsupported codec type.\n");
        return AVERROR_INVALIDDATA;
    }

    // Check if the matrix configuration has been changed more than once per access unit.
    if (m->matrix_changed++ > 1) {
        av_log(m->avctx, AV_LOG_ERROR, "Matrix configuration changed more than once per access unit.\n");
        return AVERROR_INVALIDDATA;
    }

    // Retrieve the number of primitive matrices from the bitstream.
    s->num_primitive_matrices = get_bits(bit_context, 4);

    // Validate that the number of primitive matrices does not exceed the allowed maximum.
    if (s->num_primitive_matrices > max_primitive_matrices) {
        av_log(m->avctx, AV_LOG_ERROR, "Number of primitive matrices exceeds maximum allowed.\n");
        return AVERROR_INVALIDDATA;
    }
