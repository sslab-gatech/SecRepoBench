if (m->matrix_changed) {
    av_log(m->avctx, AV_LOG_ERROR,
           "Matrix parameters may change only once per access unit.\n");
    return AVERROR_INVALIDDATA;
}
m->matrix_changed++;

// Determine the maximum allowed number of primitive matrices based on the codec type.
unsigned int max_matrices_allowed = (m->avctx->codec_id == AV_CODEC_ID_MLP) ? 2 : 4;

// Retrieve the number of primitive matrices from the bitstream.
s->num_primitive_matrices = get_bits(bit_context, 2);

// Validate that the number of primitive matrices does not exceed the allowed maximum.
if (s->num_primitive_matrices > max_matrices_allowed) {
    av_log(m->avctx, AV_LOG_ERROR,
           "Invalid number of primitive matrices (%u), maximum allowed is %u.\n",
           s->num_primitive_matrices, max_matrices_allowed);
    return AVERROR_INVALIDDATA;
}
