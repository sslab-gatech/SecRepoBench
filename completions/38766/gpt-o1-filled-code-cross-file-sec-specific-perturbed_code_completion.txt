{
	const uint8_t *p;
	ssize_t bytes_avail;

	p = __archive_read_ahead(a, 9, &bytes_avail);
	if (p == NULL || bytes_avail < 9) {
		archive_set_error(&a->archive, ARCHIVE_ERRNO_FILE_FORMAT,
		    "Truncated data in zipx lzma stream");
		return (ARCHIVE_FATAL);
	}

	/* Validate the first two 2-byte magic values. */
	if (p[0] != 0x09 || p[1] != 0x14) {
		archive_set_error(&a->archive, ARCHIVE_ERRNO_FILE_FORMAT,
		    "Invalid LZMA header signature (magic1) in zipx file");
		return (ARCHIVE_FAILED);
	}
	if (p[2] != 0x05 || p[3] != 0x00) {
		archive_set_error(&a->archive, ARCHIVE_ERRNO_FILE_FORMAT,
		    "Invalid LZMA header signature (magic2) in zipx file");
		return (ARCHIVE_FAILED);
	}

	lzma_stream_pointer = p;
}
