/* Prepare the lzma alone header */
	if (bytes_avail < 9) {
		archive_set_error(&a->archive, ARCHIVE_ERRNO_FILE_FORMAT,
		    "Truncated lzma alone header");
		return ARCHIVE_FATAL;
	}

	/* Check the magic numbers */
	if (p[0] != 0x5d || p[1] != 0x00) {
		archive_set_error(&a->archive, ARCHIVE_ERRNO_FILE_FORMAT,
		    "Invalid lzma alone header");
		return ARCHIVE_FATAL;
	}

	/* Copy the LZMA parameters */
	memcpy(&alone_header.bytes[0], p + 4, 5);
