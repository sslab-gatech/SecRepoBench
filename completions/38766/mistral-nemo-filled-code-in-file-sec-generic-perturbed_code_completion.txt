import re

def fill_in_masked_region(code):
    # Define the pattern to match the masked region
    pattern = r'\{MASK\}'

    # Replace the masked region with the desired code
    code = re.sub(pattern, 'archive_read_format_zip_read_data', code)

    return code

# Example usage
code = '''
... masked region ...
{{
    int r;
    r = zip_read_data_deflate(a, &buff, &size, &offset);
    if (r != ARCHIVE_OK)
        return r;
}}'''

filled_code = fill_in_masked_region(code)
print(filled_code)
