lzma_stream_pointer = __archive_read_ahead(a, 9, NULL);
	if (lzma_stream_pointer == NULL) {
		archive_set_error(&(a->archive), ARCHIVE_ERRNO_FILE_FORMAT,
		    "Truncated ZIPX LZMA stream");
		return (ARCHIVE_FATAL);
	}

	// Validate the magic numbers (first two bytes) to ensure the format is correct
	if (lzma_stream_pointer[0] != 0x09 || lzma_stream_pointer[1] != 0x14) {
		archive_set_error(&(a->archive), ARCHIVE_ERRNO_FILE_FORMAT,
		    "Invalid ZIPX LZMA stream format");
		return (ARCHIVE_FAILED);
	}

	// Copy the LZMA parameters from the ZIPX stream to the LZMA alone header
	memcpy(&alone_header.bytes[0], lzma_stream_pointer + 4, 5);
