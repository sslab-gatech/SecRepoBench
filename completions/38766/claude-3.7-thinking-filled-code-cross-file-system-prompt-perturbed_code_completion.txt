const void* p;
ssize_t bytes_avail;

/* Read the ZIPX LZMA stream header (9 bytes). */
p = __archive_read_ahead(a, 9, &bytes_avail);
if (p == NULL || bytes_avail < 9) {
    archive_set_error(&(a->archive), ARCHIVE_ERRNO_FILE_FORMAT,
        "Truncated LZMA header in ZIPX stream");
    return (ARCHIVE_FATAL);
}

/* Get a pointer to the LZMA stream data */
lzma_stream_pointer = (const uint8_t*)p;

/* Validate magic numbers (first 4 bytes) */
if (lzma_stream_pointer[0] != 0x09 || lzma_stream_pointer[1] != 0x14 || 
    lzma_stream_pointer[2] != 0x05 || lzma_stream_pointer[3] != 0x00) {
    archive_set_error(&(a->archive), ARCHIVE_ERRNO_FILE_FORMAT,
        "Invalid LZMA header in ZIPX stream");
    return (ARCHIVE_FATAL);
}
