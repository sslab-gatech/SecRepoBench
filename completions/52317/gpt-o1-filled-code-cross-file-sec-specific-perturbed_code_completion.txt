if (langnum == LANG_hu) {
    // calculate the number of syllables in the second word starting at index i
    int second_word_syllables = get_syllable(st.substr(i));
    
    // subtract the syllables possibly introduced by a suffix (inflections/derivations)
    if (sfx && sfx->getKey()) {
        second_word_syllables -= get_syllable(sfx->getKey());
        if (second_word_syllables < 0) {
            second_word_syllables = 0;
        }
    }
    
    // add the resulting count
    syllable_count += second_word_syllables;
    
    // if the prefix part has more than one syllable, treat it as an extra word
    if (pfx && pfx->getKey()) {
        int prefix_syllables = get_syllable(pfx->getKey());
        if (prefix_syllables > 1) {
            wordnum++;
        }
    }
    
    // adjust syllable count based on specific Hungarian flags in the dictionary
    if (rv && rv->astr &&
        TESTAFF(rv->astr, (unsigned short)'I', rv->alen) &&
        !TESTAFF(rv->astr, (unsigned short)'J', rv->alen)) {
        syllable_count--;
        if (syllable_count < 0) {
            syllable_count = 0;
        }
    }
}
