// Check if a word is a compound word
bool AffixMgr::is_compound_word(const std::string& word) {
  struct hentry* rv = NULL;
  rv = compound_check(word, 0, word.size(), 0, NULL, 0, 0, 0);
  return rv != NULL;
}

// Check if a word is a compound word with a specific flag
bool AffixMgr::is_compound_word(const std::string& word, FLAG flag) {
  struct hentry* rv = NULL;
  rv = compound_check(word, 0, word.size(), 0, NULL, 0, 0, flag);
  return rv != NULL;
}
