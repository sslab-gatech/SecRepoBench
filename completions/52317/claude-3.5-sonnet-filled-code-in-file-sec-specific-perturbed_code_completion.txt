  numsyllable += get_syllable(word.c_str() + i);

  // subtract syllable count of the suffix, considering inflections and derivations
  if (sfxappnd) {
    std::string tmp(sfxappnd);
    reverseword(tmp);
    numsyllable -= short(get_syllable(tmp) + sfxextra);
  } else {
    numsyllable -= short(sfxextra);
  }

  // increment word count if prefix has more than one syllable
  if (pfx && (get_syllable(pfx->getKey()) > 1))
    wordnum++;

  // adjust syllable count based on specific flags
  if (!cpdsyllablenum.empty()) {
    switch (sfxflag) {
      case 'c':
        numsyllable += 2;
        break;
      case 'J':
        numsyllable += 1;
        break;
      case 'I':
        if (rv && TESTAFF(rv->astr, 'J', rv->alen))
          numsyllable += 1;
        break;
    }
  }
