          }

          // + 1 word, if syllable number of the prefix > 1 (hungarian

          if (pfx && (get_syllable(pfx->getKey()) > 1))
            wordnum++;

          // and the suffix is beginning `s'
          if (!cpdsyllablenum.empty()) {
            switch (sfxflag) {
              case 'c': {
                numsyllable += 2;
                break;
              }
              case 'J': {
                numsyllable += 1;
                break;
              }
              case 'I': {
                if (rv && TESTAFF(rv->astr, 'J', rv->alen))
                  numsyllable += 1;
                break;
              }
