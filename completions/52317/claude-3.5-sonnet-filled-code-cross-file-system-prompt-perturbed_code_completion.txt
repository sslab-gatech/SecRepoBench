if (langnum == LANG_hu) {
    // calculate syllable number of the word
    syllable_count += get_syllable(word.substr(i));

    // - affix syllable num.
    if (sfx && sfx->getKey()) {
        std::string tmp(sfx->getKey());
        reverseword(tmp);
        syllable_count -= get_syllable(tmp);
    }

    // + 1 word, if prefix syllable num. > 1 (hungarian convention)
    if (pfx && (get_syllable(pfx->getKey()) > 1)) {
        wordnum++;
    }

    // increment syllable num, if last word has a SYLLABLENUM flag
    // and the suffix is beginning `s`
    if (cpdsyllablenum) {
        if (sfx && sfx->getFlag() && TESTAFF(sfx->getFlag(), cpdsyllablenum, sfx->getFlagLen())) {
            syllable_count++;
        }

        // decrement syllable num, if last word has a SYLLABLENUM flag
        // and the suffix is beginning `s`
        if (sfx && sfx->getKey() && (sfx->getKey()[0] == 's') &&
            TESTAFF(rv->astr, cpdsyllablenum, rv->alen)) {
            syllable_count--;
        }
    }

    // increment word number, if the second word has a compoundroot flag
    if ((rv) && (compoundroot) &&
        (TESTAFF(rv->astr, compoundroot, rv->alen))) {
        wordnum++;
    }
}
