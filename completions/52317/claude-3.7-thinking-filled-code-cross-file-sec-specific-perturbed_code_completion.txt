    syllable_count += get_syllable(st.substr(i));
    
    // subtract syllable number of the suffix
    if (sfx && sfx->getKey()) {
        syllable_count -= get_syllable(std::string(sfx->getKey()));
    }
    
    // increment word number, if the prefix has more than 1 syllable (hungarian convention)
    if (pfx && pfx->getKey() && (get_syllable(std::string(pfx->getKey())) > 1)) {
        wordnum++;
    }
    
    // adjust syllable count for specific Hungarian rules
    if (rv && sfx && sfx->getCont() &&
        (TESTAFF(sfx->getCont(), (unsigned short)'x', sfx->getContLen()) ||
         TESTAFF(sfx->getCont(), (unsigned short)'%', sfx->getContLen()))) {
        syllable_count--;
    }
}
