            if (pfx && pfx->getCont()) {
              syllable_count += get_syllable(pfx->getKey());
              if (get_syllable(pfx->getKey()) > 1) {
                wordnum++;
              }
            }

            // Check if the suffix exists and adjust the syllable count
            if (sfx && sfx->getCont()) {
              syllable_count -= get_syllable(sfx->getCont());
