  // Calculate syllables for the current part (word starting from index i)
  int sec_word_syllables = get_syllable(st.substr(i));
  // If there is a suffix, subtract its syllable count from the current part
  int suffix_syllables = 0;
  if (sfx && sfx->getCont()) {
    suffix_syllables = get_syllable(std::string(sfx->getCont(), sfx->getContLen()));
  }
  sec_word_syllables -= suffix_syllables;
  syllable_count += sec_word_syllables;
  // According to Hungarian conventions: if the prefix has more than one syllable,
  // the prefix+word counts as two words.
  if (pfx && (get_syllable(pfx->getKey()) > 1)) {
    wordnum++;
  }
}
