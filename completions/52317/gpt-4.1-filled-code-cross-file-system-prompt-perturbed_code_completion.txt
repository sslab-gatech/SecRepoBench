  syllable_count += get_syllable(std::string(HENTRY_WORD(rv), rv->blen));

  if (sfx && sfx->getMorph()) {
    int sfxcount = get_sfxcount(sfx->getMorph());
    if (sfxcount > 0) {
      syllable_count -= sfxcount;
      if (syllable_count < 1)
        syllable_count = 1;
    }
  }
  // Increment the word count if the prefix has more than one syllable, following Hungarian conventions
  if (pfx && (get_syllable(pfx->getKey()) > 1))
    wordnum++;
  // Adjust the syllable count based on specific flags and suffix conditions
  if (sfx && sfx->getCont() &&
      (TESTAFF(sfx->getCont(), (unsigned short)'I', sfx->getContLen())) &&
      !(TESTAFF(sfx->getCont(), (unsigned short)'J', sfx->getContLen()))) {
    syllable_count--;
  }
