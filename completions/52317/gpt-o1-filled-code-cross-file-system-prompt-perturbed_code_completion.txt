    int sylSegment = get_syllable(st.substr(i));

    // subtract the number of suffix-based derivational or inflectional syllables
    if (rv && HENTRY_DATA(rv)) {
        int sfxCount = get_sfxcount(HENTRY_DATA(rv));
        sylSegment -= sfxCount;
        if (sylSegment < 0) sylSegment = 0;
    }

    // update the overall syllable count
    syllable_count += sylSegment;

    // for Hungarian: if the prefix exists and has more than one syllable, count as two words
    if (pfx && (get_syllable(pfx->getKey()) > 1)) {
        wordnum++;
    }
