if (langnum == LANG_hu) {
    // Store the prefix string for reference (if any)
    pfxappnd = pfx ? pfx->getKey() : NULL;

    // Calculate how many syllables appear in the remainder of the word (from index i)
    syllable_count += get_syllable(word.substr(i));

    // Subtract the syllable count from the suffix, if we have one
    if (sfxappnd) {
        std::string tmp(sfxappnd);
        reverseword(tmp);
        syllable_count -= short(get_syllable(tmp) + sfxextra);
    } else {
        syllable_count -= short(sfxextra);
    }

    // If the prefix contains more than one syllable, treat it as if we added a word
    if (pfxappnd && get_syllable(pfxappnd) > 1) {
        wordnum++;
    }

    // Optionally adjust syllable count if we have a special SYLLABLENUM suffix flag
    if (!cpdsyllablenum.empty()) {
        switch (sfxflag) {
          case 'c':
            syllable_count += 2; // for -c type suffix
            break;
          case 'J':
            syllable_count += 1; // for -J type suffix
            break;
          case 'I':
            // If the resulting suffix has the J flag, add one more syllable
            if (rv && TESTAFF(rv->astr, 'J', rv->alen)) {
                syllable_count += 1;
            }
            break;
        }
    }
}
