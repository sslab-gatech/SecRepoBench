    int numsyllable = get_syllable(std::string(HENTRY_WORD(rv), rv->blen));
    // subtract syllable number of suffix
    if (sfx) {
        std::string tmp;
        tmp.assign(sfx->getKey());
        numsyllable -= get_syllable(tmp);
    }
    // +1 word, if syllable number of the word > 1 (hungarian convention)
    if (numsyllable > 1) wordnum++;
    // increment syllable count, if syllable number of the prefix > 1 (hungarian convention)
    if (pfx && (get_syllable(pfx->getKey()) > 1)) syllable_count += numsyllable;
}
