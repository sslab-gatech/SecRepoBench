  int sfx_syllable = 0;
  if (sfx) {
    sfx_syllable = get_syllable(sfx->getKey());
    if (sfx->getCont() && TESTAFF(sfx->getCont(), 'I', sfx->getContLen()) &&
        !TESTAFF(sfx->getCont(), 'J', sfx->getContLen()))
      sfx_syllable--;
  }
  syllable_count -= sfx_syllable;

  if (pfx && (get_syllable(pfx->getKey()) > 1)) wordnum++;

  if (sfx && sfx->getCont() && TESTAFF(sfx->getCont(), 'K', sfx->getContLen()))
    syllable_count--;
  if (sfx && sfx->getCont() && TESTAFF(sfx->getCont(), 'L', sfx->getContLen()))
    wordnum++;
  if (sfx && sfx->getCont() && TESTAFF(sfx->getCont(), 'M', sfx->getContLen()))
    syllable_count++;
