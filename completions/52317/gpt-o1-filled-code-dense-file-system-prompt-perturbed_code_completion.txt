    // calculate syllable number of the word from index i
    syllable_count += get_syllable(word.substr(i));

    // subtract the syllables contributed by the suffix
    if (sfxappnd) {
        std::string tmp(sfxappnd);
        reverseword(tmp);
        syllable_count -= short(get_syllable(tmp) + sfxextra);
    } else {
        syllable_count -= short(sfxextra);
    }

    // +1 word if the prefix has more than one syllable (Hungarian convention)
    if (pfx && (get_syllable(pfx->getKey()) > 1)) {
        wordnum++;
    }

    // adjust the syllable count based on specific suffix flags
    if (!cpdsyllablenum.empty()) {
        switch (sfxflag) {
            case 'c':
                syllable_count += 2;
                break;
            case 'J':
                syllable_count += 1;
                break;
            case 'I':
                if (rv && TESTAFF(rv->astr, 'J', rv->alen)) {
                    syllable_count += 1;
                }
                break;
        }
    }
}
