// LANG_hu section: spec. Hungarian rule
if (langnum == LANG_hu) {
  // calculate syllable number of the word starting from index i
  syllable_count += get_syllable(std::string(st.c_str() + i));
  
  // subtract syllable count of the suffix
  if (sfx && sfx->getKey()) {
    syllable_count -= get_syllable(sfx->getKey());
  }
  
  // increment word number, if prefix syllable count > 1 (Hungarian convention)
  if (pfx && pfx->getKey()) {
    if (get_syllable(pfx->getKey()) > 1) {
      wordnum++;
    }
  }
  
  // adjust syllable count based on specific flags
  if (sfx && sfx->getCont()) {
    // Special Hungarian rules with hardcoded flags
    if (TESTAFF(sfx->getCont(), (unsigned short)'I', sfx->getContLen()) &&
        !TESTAFF(sfx->getCont(), (unsigned short)'J', sfx->getContLen())) {
      syllable_count--;
    }
  }
}
