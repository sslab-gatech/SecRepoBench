if (langnum == LANG_hu && rv) {
    // Calculate the syllables in the portion of the word starting from index i
    int hunSyllables = get_syllable(st.substr(i));
    
    // Subtract syllables contributed by any suffix
    if (sfx && sfx->getKey()) {
        int sfxSyll = get_syllable(std::string(sfx->getKey()));
        hunSyllables -= sfxSyll;
    }

    // Ensure we don't go below zero
    if (hunSyllables < 0) {
        hunSyllables = 0;
    }

    // If there's a prefix, check if it has more than one syllable
    // Hungarian convention: the prefix alone may be counted as an extra word
    if (pfx && pfx->getKey()) {
        int pfxSyll = get_syllable(std::string(pfx->getKey()));
        if (pfxSyll > 1) {
            wordnum++;
        }
    }

    // Add these syllables to the total
    syllable_count += hunSyllables;
}
