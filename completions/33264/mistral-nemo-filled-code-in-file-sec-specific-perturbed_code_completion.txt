import blosc2

# Create a context for compression
cparams = blosc2.blosc2_cparams()
cparams.compcode = blosc2.BLOSC_LZ4
cparams.clevel = 5
cparams.typesize = 8
cparams.blocksize = 1024
cparams.nthreads = 1

# Create a chunk made of zeros
nbytes = 1024 * 1024
dest = bytearray(nbytes)
destsize = len(dest)
error = blosc2.blosc2_chunk_zeros(cparams, nbytes, dest, destsize)
if error == 0:
    print("Chunk of zeros created successfully")
else:
    print(f"Error creating chunk of zeros: {error}")
