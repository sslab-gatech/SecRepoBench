if(network_packet_struct->line[network_packet_struct->parsed_lines].len > NDPI_STATICSTRING_LEN("Content-Type:") + 1 &&
   strncasecmp((const char *) network_packet_struct->line[network_packet_struct->parsed_lines].ptr,
               "Content-Type:", NDPI_STATICSTRING_LEN("Content-Type:")) == 0) {
  // Parse the "Content-Type" header value, trimming any leading spaces.
  network_packet_struct->content_line.ptr =
    &network_packet_struct->line[network_packet_struct->parsed_lines].ptr[NDPI_STATICSTRING_LEN("Content-Type:") + 1];
  network_packet_struct->content_line.len =
    network_packet_struct->line[network_packet_struct->parsed_lines].len - NDPI_STATICSTRING_LEN("Content-Type:") - 1;

  // Trim leading spaces from the content type value.
  while (network_packet_struct->content_line.len > 0 && isspace(network_packet_struct->content_line.ptr[0])) {
    network_packet_struct->content_line.ptr++;
    network_packet_struct->content_line.len--;
  }

  network_packet_struct->http_num_headers++;
}
