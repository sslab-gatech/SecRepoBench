{
  /*
   * We already know this line starts with "Authorization: " per the outer if-condition,
   * so parse out and store its value, then handle other recognized headers similarly.
   */

  char *line_ptr = (char *)network_packet_struct->line[network_packet_struct->parsed_lines].ptr;
  uint16_t line_len = network_packet_struct->line[network_packet_struct->parsed_lines].len;
  char *val_ptr = line_ptr + 15; /* skip "Authorization: " */
  uint16_t val_len;

  /* Trim leading spaces in the header's value */
  while ((val_ptr < line_ptr + line_len) && (*val_ptr == ' ' || *val_ptr == '\t')) {
    val_ptr++;
  }
  val_len = (uint16_t)(line_len - (val_ptr - line_ptr));

  /* If desired, a dedicated struct field could be used; here we simply track count. */
  /* For demonstration, store it in a hypothetical authorization_line field: */
  network_packet_struct->authorization_line.ptr = (u_int8_t*)val_ptr;
  network_packet_struct->authorization_line.len = val_len;
  network_packet_struct->http_num_headers++;

  /*
   * Recognize additional HTTP headers if needed. In practice, each line is handled
   * by a similar if/else chain or elsewhere. Below is an example pattern showing
   * how you could handle some other headers in-line. Note that in this codebase,
   * headers like "Server:" and "Host:" have already been handled above.
   */

  /* Example: Accept: */
  if ((line_len >= 7) && (strncasecmp(line_ptr, "Accept:", 7) == 0)) {
    char *accept_val = line_ptr + 7;
    while ((accept_val < line_ptr + line_len) && (*accept_val == ' ' || *accept_val == '\t')) {
      accept_val++;
    }
    /* Store or process value if you have a field for it */
    network_packet_struct->http_num_headers++;
  }

  /* Example: Referer: */
  else if ((line_len >= 8) && (strncasecmp(line_ptr, "Referer:", 8) == 0)) {
    network_packet_struct->http_num_headers++;
  }

  /* Example: User-Agent: */
  else if ((line_len >= 11) && (strncasecmp(line_ptr, "User-Agent:", 11) == 0)) {
    network_packet_struct->http_num_headers++;
  }

  /*
   * Handle Content-Type specifically, even if there is no space after the colon.
   */
  else if ((line_len >= 11) && (strncasecmp(line_ptr, "Content-Type", 11) == 0)) {
    char *colon_ptr = memchr(line_ptr, ':', line_len);
    if (colon_ptr != NULL) {
      char *ctype_val = colon_ptr + 1;
      while ((ctype_val < line_ptr + line_len) && (*ctype_val == ' ' || *ctype_val == '\t')) {
        ctype_val++;
      }
      uint16_t ctype_len = (uint16_t)(line_len - (ctype_val - line_ptr));
      network_packet_struct->content_line.ptr = (u_int8_t*)ctype_val;
      network_packet_struct->content_line.len = ctype_len;
    }
    network_packet_struct->http_num_headers++;
  }

  /* Fallback: increment header count for unrecognized lines */
  else {
    network_packet_struct->http_num_headers++;
  }
}
