{
	  size_t authorization_header_len = NDPI_STATICSTRING_LEN("Authorization: ");
	  char *authorization_value = (char *) network_packet_struct->line[network_packet_struct->parsed_lines].ptr + authorization_header_len;
	  size_t authorization_value_len = network_packet_struct->line[network_packet_struct->parsed_lines].len - authorization_header_len;

	  // Trim leading spaces from the authorization value
	  while (authorization_value_len > 0 && isspace(*authorization_value)) {
		authorization_value++;
		authorization_value_len--;
	  }

	  network_packet_struct->authorization_line.ptr = authorization_value;
	  network_packet_struct->authorization_line.len = authorization_value_len;
	  network_packet_struct->http_num_headers++;
	}
