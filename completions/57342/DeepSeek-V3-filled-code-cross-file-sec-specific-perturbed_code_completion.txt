bool axis_not_pinned = false;
  unsigned old_idx_to_tag_map_size = 0;
  for (unsigned i = 0; i < axes.length; i++)
  {
    hb_tag_t tag = axes[i].get_axis_tag ();
    plan->axes_old_index_tag_map.set (i, tag);

    if (plan->user_axes_location.has (tag))
    {
      float val = plan->user_axes_location.get (tag);
      int normalized_val = axes[i].normalize_value (val);
      if (has_avar)
      {
        seg_maps = &fontface->table.avar->get_segment_maps (i);
        normalized_val = seg_maps->map (normalized_val);
      }
      plan->normalized_coords[i] = normalized_val;
      if (normalized_val != 0)
        plan->pinned_at_default = false;
    }
    else
    {
      axis_not_pinned = true;
      plan->axes_index_map.set (old_idx_to_tag_map_size++, i);
    }
  }
