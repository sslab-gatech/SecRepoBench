hb_map_t old_index_to_tag;
  hb_map_t tag_to_new_index;
  bool axis_not_pinned = false;

  for (unsigned int old_axis_idx = 0; old_axis_idx < axes.length; old_axis_idx++)
  {
    hb_tag_t tag = axes[old_axis_idx].tag;
    old_index_to_tag.set(old_axis_idx, tag);

    int user_value;
    if (!plan->user_axes_location.get(tag, &user_value))
    {
      axis_not_pinned = true;
      tag_to_new_index.set(tag, plan->normalized_coords.length);
      plan->normalized_coords[old_axis_idx] = 0;
    }
    else
    {
      int normalized_value = fontface->table.fvar->normalize_axis_value(old_axis_idx, user_value);
      if (has_avar && old_axis_idx < fontface->table.avar->get_axis_count())
      {
        seg_maps = fontface->table.avar->get_axis_segment_maps(old_axis_idx);
        if (seg_maps)
          normalized_value = seg_maps->map(normalized_value);
      }
      plan->normalized_coords[old_axis_idx] = normalized_value;
      if (normalized_value != 0)
        axis_not_pinned = true;
    }
  }
