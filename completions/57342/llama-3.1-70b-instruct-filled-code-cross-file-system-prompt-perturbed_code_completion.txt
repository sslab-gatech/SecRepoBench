bool axis_not_pinned = false;
unsigned int axis_index = 0;
hb_map_t old_index_to_tag;
if (has_avar)
  seg_maps = &face->table.avar->get_segment_maps ();

for (unsigned int i = 0; i < axes.length; i++)
{
  const OT::AxisRecord &axis = axes[i];
  hb_tag_t tag = axis.get_axis_tag ();
  old_index_to_tag.set (i, tag);

  if (plan->user_axes_location.has (tag))
  {
    float value = plan->user_axes_location.get (tag);
    int normalized_value = face->table.fvar->normalize_axis_value (i, value);
    if (has_avar)
      seg_maps->remap_value (axis_index, &normalized_value);
    plan->normalized_coords[axis_index] = normalized_value;
    if (normalized_value!= 0)
      plan->pinned_at_default = false;
  }
  else
  {
    axis_not_pinned = true;
    plan->normalized_coords[axis_index] = 0;
  }
  axis_index++;
}
