unsigned old_index = 0;
  bool axis_not_pinned = false;
  if (has_avar)
    seg_maps = &fontface->table.avar->get_segment_maps ();

  for (unsigned i = 0; i < axes.length; i++)
  {
    hb_tag_t axis_tag = axes[i].axisTag;
    plan->axes_old_index_tag_map.set (old_index++, axis_tag);

    if (!plan->user_axes_location.has (axis_tag))
    {
      axis_not_pinned = true;
      plan->axes_index_map.set (i, HB_OT_VAR_NO_AXIS_INDEX);
      continue;
    }

    float v = plan->user_axes_location.get (axis_tag);
    float normalized = axes[i].normalize_axis_value (v);

    if (has_avar)
    {
      normalized = seg_maps->map (normalized);
      seg_maps++;
    }

    plan->axes_location.set (axis_tag, v);
    plan->normalized_coords[i] = normalized;
    plan->axes_index_map.set (i, i);

    if (normalized != 0.f)
      plan->pinned_at_default = false;
  }
