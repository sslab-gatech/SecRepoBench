unsigned old_axis_idx = 0;
  bool axis_not_pinned = false;
  const OT::SegmentMaps *seg_maps = fontface->table.avar->get_segment_maps();
  unsigned avar_axis_count = seg_maps ? seg_maps->get_axis_count() : 0;

  for (unsigned i = 0; i < axes.length; i++) {
    hb_tag_t axis_tag = axes[i].axisTag;
    auto user_axis_location_iter = plan->user_axes_location.find(axis_tag);

    if (user_axis_location_iter == plan->user_axes_location.end()) {
      axis_not_pinned = true;
      old_axis_idx++;
      continue;
    }

    float normalized_value = plan->user_axes_location[axis_tag];
    plan->normalized_coords[i] = normalized_value;

    if (has_avar && seg_maps) {
      normalized_value = seg_maps->remap_value(i, normalized_value);
      plan->normalized_coords[i] = normalized_value;
    }

    if (normalized_value != 0) {
      plan->all_axes_pinned = false;
    }

    old_axis_idx++;
  }
