bool axis_not_pinned = false;
  unsigned int axis_count = axes.length;
  hb_map_t old_index_tag_map;
  hb_map_t new_index_map;

  for (unsigned int i = 0; i < axis_count; ++i)
  {
    hb_tag_t tag = axes[i].axisTag;
    old_index_tag_map.set (i, tag);

    float user_val;
    if (!plan->user_axes_location.has (tag, &user_val))
    {
      axis_not_pinned = true;
      new_index_map.set (i, i);
      plan->normalized_coords[i] = 0;
    }
    else
    {
      int normalized_val = axes[i].normalize_axis_value (user_val);
      if (has_avar)
      {
        if (i > 0)
          seg_maps = &fontface->table.avar->get_axis_segment_maps (i - 1);
        normalized_val = seg_maps->map_segment (i, normalized_val);
      }

      plan->normalized_coords[i] = normalized_val;
      plan->axes_location.set (tag, user_val);

      if (normalized_val != 0)
      {
        plan->pinned_at_default = false;
      }
    }
  }
