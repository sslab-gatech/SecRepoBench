if (has_avar)
    seg_maps = &fontface->table.avar->get_segment_maps ();

  for (unsigned i = 0; i < axes.length; i++)
  {
    const OT::AxisRecord &axis = axes[i];
    plan->normalized_coords[i] = plan->user_axes_location.get (axis.tag, axis.default_value);

    if (seg_maps)
    {
      const OT::SegmentMap &seg_map = (*seg_maps)[i];
      if (seg_map.has_data ())
        plan->normalized_coords[i] = seg_map.map (plan->normalized_coords[i]);
    }

    if (plan->normalized_coords[i] != axis.default_value)
      pinned_at_default = false;
  }
