bool axis_not_pinned = false;
  for (unsigned i = 0; i < axes.length; i++)
  {
    const OT::AxisRecord &axis = axes[i];
    plan->axis_index_map[i] = axis.tag;

    if (!plan->user_axes_location.has(axis.tag))
    {
      axis_not_pinned = true;
      continue;
    }

    float user_value = plan->user_axes_location.get(axis.tag);
    float normalized_value = user_value;

    if (has_avar)
    {
      normalized_value = fontface->table.avar->map_value(axis.tag, user_value);
    }

    plan->normalized_coords[i] = normalized_value;

    if (normalized_value != 0)
      plan->all_axes_pinned = false;
  }
