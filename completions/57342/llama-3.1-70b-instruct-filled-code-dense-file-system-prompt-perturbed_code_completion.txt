unsigned int axis_index = 0;
bool axis_not_pinned = false;
unsigned int old_to_tag_map[axes.length];

for (unsigned int i = 0; i < axes.length; i++) {
  const OT::AxisRecord &axis = axes[i];
  old_to_tag_map[i] = axis.axis_index;
  axis_index = axis.axis_index;

  if (plan->user_axes_location.has (axis.axis_tag)) {
    float value = plan->user_axes_location.get (axis.axis_tag);
    value = axis.normalize_value (value);
    if (has_avar)
      value = seg_maps->remap_value (axis.axis_index, value);
    plan->normalized_coords[axis_index] = value;
    if (value!= 0)
      plan->pinned_at_default = false;
  } else {
    plan->normalized_coords[axis_index] = 0;
    axis_not_pinned = true;
  }

  if (has_avar)
    seg_maps = seg_maps->get_next (axis.axis_index);
}
