bool axis_not_pinned = false;
const OT::SegmentMaps *seg_maps = nullptr;

if (has_avar)
  seg_maps = fontface->table.avar->get_segment_maps ().arrayZ;

for (unsigned i = 0; i < axes.length; i++)
{
  hb_tag_t tag = axes[i].axisTag;
  plan->axes_old_index_tag_map.set (i, tag);

  if (plan->user_axes_location.has (tag))
  {
    float user_val = plan->user_axes_location.get (tag);
    int normalized_val = axes[i].normalize_axis_value (user_val);
    if (has_avar)
      normalized_val = seg_maps[i].map (normalized_val);
    plan->normalized_coords[i] = normalized_val;
    plan->axes_location.set (tag, user_val);
    if (normalized_val != 0)
      plan->pinned_at_default = false;
  }
  else
  {
    axis_not_pinned = true;
    plan->axes_index_map.add (i);
  }
}
