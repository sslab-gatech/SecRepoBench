unsigned axis_index = 0;
  bool axis_not_pinned = false;
  if (has_avar)
    seg_maps = &fontface->table.avar->get_segment_maps ();

  for (const auto& axis : axes)
  {
    hb_tag_t axis_tag = axis.get_axis_tag ();
    plan->axes_old_index_tag_map.set (axis_index, axis_tag);

    if (!plan->user_axes_location.has (axis_tag))
    {
      axis_not_pinned = true;
      plan->axes_index_map.set (axis_index, axis_index);
      axis_index++;
      if (has_avar && seg_maps)
        seg_maps = &StructAfter<OT::SegmentMaps> (*seg_maps);
      continue;
    }

    float v = plan->user_axes_location[axis_tag];
    v = OT::VariationSpace::normalize_axis_value (v, axis);
    if (has_avar && seg_maps)
    {
      v = seg_maps->map (v);
      seg_maps = &StructAfter<OT::SegmentMaps> (*seg_maps);
    }

    plan->axes_location.set (axis_tag, v);
    plan->normalized_coords[axis_index] = v;
    if (v != 0.f)
      plan->pinned_at_default = false;
  }
