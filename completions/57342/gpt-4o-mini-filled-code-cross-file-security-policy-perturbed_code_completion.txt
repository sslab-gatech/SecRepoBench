unsigned old_axis_idx = 0;
  bool axis_not_pinned = false;

  for (unsigned i = 0; i < axes.length; i++) {
    const OT::AxisRecord &axis = axes[i];
    hb_tag_t tag = axis.tag;

    if (plan->user_axes_location.has(tag)) {
      // Axis is set by the user, normalize its value.
      float user_value = plan->user_axes_location.get(tag);
      plan->normalized_coords[old_axis_idx] = user_value;

      if (user_value != 0)
        plan->all_axes_pinned = false;

      if (has_avar && old_axis_idx < avar_axis_count) {
        // Remap using segment maps if available.
        plan->normalized_coords[old_axis_idx] = seg_maps->map_value(old_axis_idx, user_value);
      }
    } else {
      // Axis is not set by the user, mark as not pinned.
      axis_not_pinned = true;
      plan->normalized_coords[old_axis_idx] = axis.default_value; // Use default value.
    }

    if (has_avar && old_axis_idx < avar_axis_count) {
      // Update segment maps for 'avar'.
      seg_maps->update(old_axis_idx, plan->normalized_coords[old_axis_idx]);
    }

    old_axis_idx++;
  }
