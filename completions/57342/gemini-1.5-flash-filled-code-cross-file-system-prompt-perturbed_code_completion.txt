bool axis_not_pinned = false;
  for (unsigned int i = 0; i < axes.length; i++)
  {
    hb_tag_t tag = axes[i].axisTag;
    plan->old_to_new_axis_index_map.set (i, tag);

    bool user_set = plan->user_axes_location.has (tag);
    if (!user_set)
    {
      axis_not_pinned = true;
      plan->old_to_new_axis_index_map.set (i, HB_MAP_VALUE_INVALID);
      continue;
    }

    float user_value = plan->user_axes_location.get (tag);
    float normalized_value = user_value;
    if (has_avar)
    {
      seg_maps = &fontface->table.avar->get_segment_maps (i);
      normalized_value = seg_maps->map_value (normalized_value);
    }
    plan->normalized_coords[i] = normalized_value;
    plan->axes_location.set (tag, normalized_value);
    if (normalized_value != 0)
      axis_not_pinned = true;
  }
