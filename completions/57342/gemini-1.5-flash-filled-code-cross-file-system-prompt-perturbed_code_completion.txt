bool axis_not_pinned = false;
  hb_map_t old_to_new_index;
  for (unsigned int i = 0; i < axes.length; i++)
  {
    hb_tag_t tag = axes[i].axis.tag;
    old_to_new_index.set (i, i);
    bool user_set = plan->user_axes_location.has (tag);
    if (!user_set)
    {
      axis_not_pinned = true;
      continue;
    }
    float design_coord = plan->user_axes_location.get (tag);
    int normalized_coord = axes[i].axis.normalize_value (design_coord);
    if (has_avar)
    {
      seg_maps = &face->table.avar->get_segment_maps (i);
      normalized_coord = seg_maps->remap (normalized_coord);
    }
    plan->normalized_coords[i] = normalized_coord;
    plan->axes_location.set (tag, normalized_coord);
    if (normalized_coord != 0)
      axis_not_pinned = true;
  }
