bool axis_not_pinned = false;
  hb_map_t<unsigned, hb_tag_t> axis_tag_map;
  for (unsigned i = 0; i < axes.length; i++) {
    hb_tag_t axis_tag = axes[i].axisTag;
    axis_tag_map.set(i, axis_tag);
    if (plan->user_axes_location.get(axis_tag) == 0) {
      axis_not_pinned = true;
    } else {
      float normalized_value = plan->user_axes_location.get(axis_tag);
      plan->normalized_coords[i] = fvar.normalize_axis_value(i, normalized_value);
      if (has_avar) {
        seg_maps = fontface->table.avar->get_segment_maps();
        if (seg_maps != nullptr) {
          plan->normalized_coords[i] = seg_maps->map_value(plan->normalized_coords[i]);
        }
      }
      if (plan->normalized_coords[i] != 0) {
        plan->all_axes_pinned = false;
      }
    }
  }
