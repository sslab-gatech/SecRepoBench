unsigned int axis_index = 0;
  bool axis_not_pinned = false;
  if (has_avar)
    seg_maps = &fontface->table.avar->get_segment_maps();

  for (const OT::AxisRecord &axis : axes)
  {
    hb_tag_t tag = axis.axisTag;
    plan->axes_old_index_tag_map.set (axis_index, tag);

    if (!plan->user_axes_location.has (tag))
    {
      axis_not_pinned = true;
      plan->axes_location.set (axis_index, axis.defaultValue);
      plan->axes_index_map.set (axis_index, axis_index);
    }
    else
    {
      float normalized_value = plan->user_axes_location.get (tag);
      normalized_value = (normalized_value - axis.minValue) / (axis.maxValue - axis.minValue) * 2.0f - 1.0f;

      if (has_avar)
        normalized_value = seg_maps[axis_index].map(normalized_value);

      plan->axes_location.set (axis_index, normalized_value);
      plan->normalized_coords[axis_index] = normalized_value;

      if (normalized_value != 0.0f)
        plan->pinned_at_default = false;
    }

    if (has_avar)
      seg_maps++;
    axis_index++;
  }
