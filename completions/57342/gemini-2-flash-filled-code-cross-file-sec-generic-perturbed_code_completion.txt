unsigned axis_index = 0;
  bool axis_not_pinned = false;
  hb_map_t old_to_new_index_map;

  for (unsigned i = 0; i < axes.length; ++i)
  {
    hb_tag_t axis_tag = axes[i].axisTag;
    old_to_new_index_map.set (i, axis_index);

    if (!plan->user_axes_location.has (axis_tag))
    {
      axis_not_pinned = true;
      old_to_new_index_map.remove (i);
      continue;
    }

    float user_value = plan->user_axes_location.get (axis_tag);
    int normalized_value = axes[i].normalize_value (user_value);

    if (has_avar)
    {
      if (!seg_maps)
        seg_maps = fontface->table.avar->get_segment_maps (axes.length);
      normalized_value = seg_maps[i].map (normalized_value);
    }

    plan->normalized_coords[axis_index] = normalized_value;
    plan->axes_location.set (axis_tag, normalized_value / (float)0x8000);

    if (normalized_value != 0)
      plan->all_axes_at_default = false;

    axis_index++;
  }
  plan->num_user_axes = axis_index;
