unsigned axis_index = 0;
  bool axis_not_pinned = false;
  hb_map_t old_index_to_tag;

  for (unsigned i = 0; i < axes.length; ++i)
  {
    hb_tag_t axis_tag = axes[i].axisTag;
    old_index_to_tag.set (i, axis_tag);

    if (!plan->user_axes_location.has (axis_tag))
    {
      axis_not_pinned = true;
      plan->old_index_to_new_index.set (i, axis_index++);
      continue;
    }

    float user_value = plan->user_axes_location.get (axis_tag);
    int normalized_value = fontface->table.fvar->normalize_axis_value (i, user_value);

    if (has_avar)
    {
      if (!seg_maps)
        seg_maps = fontface->table.avar->get_segment_maps (axes.length);
      normalized_value = seg_maps->map (i, normalized_value);
    }

    plan->normalized_coords[i] = normalized_value;
    plan->axes_location.set (axis_tag, normalized_value);

    if (normalized_value != 0)
      plan->at_least_one_axis_not_at_default = true;

    if (has_avar)
      seg_maps = fontface->table.avar->get_segment_maps (axes.length);
  }
