#ifndef HB_NO_VAR
  bool axis_not_pinned = false;
  for (auto _ : plan->user_axes_location)
  {
    hb_tag_t tag = _.first;
    float value = _.second;
    unsigned axis_index = HB_MAP_VALUE_INVALID;
    for (unsigned i = 0; i < axes.length; i++)
    {
      if (axes[i].axis_tag == tag)
      {
        axis_index = i;
        break;
      }
    }
    if (axis_index == HB_MAP_VALUE_INVALID)
    {
      axis_not_pinned = true;
      continue;
    }

    if (!has_avar)
    {
      plan->normalized_coords[axis_index] = value;
    }
    else
    {
      if (!seg_maps)
        seg_maps = &fontface->table.avar->get_segment_maps ();
      plan->normalized_coords[axis_index] = seg_maps->map_value_forward (axis_index, value);
    }

    if (plan->normalized_coords[axis_index] != 0.f)
      plan->pinned_at_default = false;
  }
