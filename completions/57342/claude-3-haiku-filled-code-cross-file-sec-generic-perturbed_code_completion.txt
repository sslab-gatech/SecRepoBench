bool axis_not_pinned = false;
for (unsigned i = 0; i < axes.length; i++)
{
  hb_tag_t axis_tag = axes[i].axisTag;
  plan->axis_location_map.set (axis_tag, i);

  if (!plan->user_axes_location.has (axis_tag))
  {
    axis_not_pinned = true;
    plan->axis_location_map.set (axis_tag, HB_MAP_VALUE_INVALID);
  }
  else
  {
    float user_value = plan->user_axes_location.get (axis_tag);
    float normalized_value = hb_ot_var_normalize_coordinate (fontface, axis_tag, user_value);
    if (has_avar)
    {
      if (!seg_maps)
        seg_maps = &fontface->table.avar->get_segment_maps ();
      normalized_value = seg_maps->map_value_forward (i, normalized_value);
    }
    plan->normalized_coords[i] = normalized_value;
    plan->all_axes_pinned = plan->all_axes_pinned && (normalized_value == 0.f);
  }
}
