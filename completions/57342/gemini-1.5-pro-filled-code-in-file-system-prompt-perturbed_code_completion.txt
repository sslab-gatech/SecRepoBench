if (has_avar)
  {
    seg_maps = &fontface->table.avar->get_segment_maps ();
  }

  bool axis_not_pinned = false;
  unsigned axis_idx = 0;
  for (const auto& axis : axes)
  {
    hb_tag_t axis_tag = axis.tag;
    axes_old_index_tag_map.set (axis_idx, axis_tag);

    float axis_value;
    if (!user_axes_location.has (axis_tag, &axis_value))
    {
      axis_not_pinned = true;
      axes_location.set (axis_tag, axis.get_default ());
      axes_index_map.set (axis_idx, axis_idx);
    }
    else
    {
      float normalized_axis_value = (axis_value - axis.get_min_value ()) / (axis.get_max_value () - axis.get_min_value ());
      if (has_avar)
      {
        normalized_axis_value = seg_maps[axis_idx].map (normalized_axis_value);
      }
      axes_location.set (axis_tag, normalized_axis_value);
      normalized_coords[axis_idx] = normalized_axis_value;
      if (normalized_axis_value != 0.f)
        pinned_at_default = false;
      axes_index_map.set (axis_idx, axis_idx);
    }
    if (has_avar)
    {
      seg_maps = &fontface->table.avar->get_segment_maps ();
    }
    axis_idx++;
  }
