bool axis_not_pinned = false;
  for (unsigned i = 0; i < axes.length; i++)
  {
    hb_tag_t tag = axes[i].tag;
    plan->axes_old_index_tag_map.set(i, tag);

    if (!plan->user_axes_location.has(tag))
    {
      axis_not_pinned = true;
      plan->axes_index_map.set(i, plan->axes_location.get_population());
      plan->axes_location.set(tag, 0.0);
      plan->normalized_coords[i] = 0;
    }
    else
    {
      float value = plan->user_axes_location.get(tag);
      plan->axes_index_map.set(i, plan->axes_location.get_population());
      plan->axes_location.set(tag, value);

      if (has_avar)
      {
        seg_maps = fontface->table.avar->get_segment_maps(i);
        value = seg_maps->map(value);
      }

      plan->normalized_coords[i] = value;
      if (value != 0)
        plan->pinned_at_default = false;
    }
  }
