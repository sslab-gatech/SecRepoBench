bool axis_not_pinned = false;

if (has_avar)
  seg_maps = fontface->table.avar->get_segment_maps ().arrayZ;

for (unsigned i = 0; i < axes.length; i++)
{
  const auto& axis = axes[i];
  hb_tag_t tag = axis.get_axis_tag ();

  plan->axes_old_index_tag_map.set (i, tag);

  if (plan->user_axes_location.has (tag))
  {
    float design_coord = plan->user_axes_location.get (tag);
    int normalized = axis.normalize_axis_value (design_coord);
    if (has_avar)
      normalized = seg_maps[i].map (normalized);

    plan->normalized_coords[i] = normalized;
    plan->axes_location.set (tag, normalized);

    if (normalized != 0)
      plan->pinned_at_default = false;
  }
  else
  {
    axis_not_pinned = true;
    plan->axes_old_index_tag_map.set (i, tag);
  }
}
