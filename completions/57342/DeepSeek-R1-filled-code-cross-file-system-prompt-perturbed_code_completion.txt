bool axis_not_pinned = false;
  plan->axes_old_index_tag_map.init ();
  plan->axes_index_map.init ();

  for (unsigned i = 0; i < axes.length; i++)
  {
    hb_tag_t tag = axes[i].get_axis_tag ();
    plan->axes_old_index_tag_map.set (i, tag);

    if (plan->user_axes_location.has (tag))
    {
      float design_coord = plan->user_axes_location.get (tag);
      int normalized = axes[i].normalize_axis_value (design_coord);

      if (has_avar)
        normalized = seg_maps->map (normalized);

      plan->axes_location.set (tag, normalized);
      plan->normalized_coords.arrayZ[i] = normalized;

      if (normalized != 0)
        plan->pinned_at_default = false;

      if (has_avar)
        seg_maps++;
    }
    else
    {
      axis_not_pinned = true;
      plan->axes_index_map.set (i, plan->axes_index_map.length);
    }
  }
