bool axis_not_pinned = false;
plan->axes_index_map.resize (axes.length);
plan->axes_old_index_to_tag.resize (axes.length);

if (has_avar)
  seg_maps = &fontface->table.avar->get_segment_maps ();

for (unsigned i = 0; i < axes.length; i++)
{
  hb_tag_t tag = axes[i].axisTag;
  plan->axes_old_index_to_tag[i] = tag;

  int32_t normalized = 0;
  if (!plan->user_axes_location.has (tag))
  {
    axis_not_pinned = true;
    plan->axes_index_map[i] = HB_OT_LAYOUT_NO_AXIS_INDEX;
  }
  else
  {
    float user_value = plan->user_axes_location.get (tag);
    normalized = axes[i].normalize_axis_value (user_value);

    if (has_avar && seg_maps)
      normalized = (*seg_maps)[i].map (normalized);

    plan->axes_index_map[i] = i;
  }

  plan->axes_location.set (tag, normalized);
  plan->normalized_coords[i] = normalized;

  if (normalized != 0)
    plan->all_axes_pinned_at_default = false;
}
