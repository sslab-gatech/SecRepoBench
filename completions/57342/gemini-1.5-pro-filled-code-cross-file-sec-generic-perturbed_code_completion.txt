unsigned axis_count = axes.length;
  unsigned pinned_count = plan->user_axes_location.get_population ();
  bool axis_not_pinned = pinned_count < axis_count;

  if (has_avar)
  {
    seg_maps = &fontface->table.avar->get_axis_segment_maps ();
  }

  hb_map_t old_index_tag_map;
  hb_map_t new_index_tag_map;
  for (unsigned i = 0; i < axis_count; ++i)
  {
    hb_tag_t tag = axes[i].axisTag;
    old_index_tag_map.set (tag, i);

    float axis_value;
    if (!plan->user_axes_location.has (tag, &axis_value))
    {
      axis_not_pinned = true;
      new_index_tag_map.set (tag, i);
      plan->normalized_coords[i] = 0;
      plan->axes_location.set (tag, 0);
      continue;
    }

    int normalized_value = fontface->table.fvar->normalize_axis_value (i, axis_value);
    if (has_avar)
    {
      normalized_value = seg_maps->map_axis_value (i, normalized_value);
    }

    plan->axes_location.set (tag, normalized_value);
    plan->normalized_coords[i] = normalized_value;

    if (normalized_value != 0)
    {
      plan->pinned_at_default = false;
    }
  }

  if (has_avar && axis_not_pinned)
  {
    plan->avar_maps = hb_ot_var_avar_subset_maps (fontface->table.avar.get (),
						 old_index_tag_map,
						 new_index_tag_map);
  }
