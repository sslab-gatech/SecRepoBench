unsigned axis_count = axes.length;
  bool axis_not_pinned = false;
  hb_map_t old_index_tag_map;
  hb_map_t new_index_tag_map;

  if (has_avar)
  {
    seg_maps = &fontface->table.avar->get_axis_segment_maps ();
  }

  for (unsigned i = 0; i < axis_count; ++i)
  {
    hb_tag_t tag = axes[i].axisTag;
    old_index_tag_map.set (i, tag);

    float user_val;
    if (!plan->user_axes_location.has (tag, &user_val))
    {
      axis_not_pinned = true;
      new_index_tag_map.set (i, tag);
      plan->normalized_coords[i] = 0;
      if (has_avar)
      {
        seg_maps = &seg_maps->get_next_segment_maps (0);
      }
      continue;
    }

    int normalized_val;
    OT::AxisRecord axis = axes[i];
    normalized_val = axis.normalize_axis_value (user_val);

    if (has_avar)
    {
      normalized_val = seg_maps->map_segment (normalized_val);
      seg_maps = &seg_maps->get_next_segment_maps (normalized_val);
    }

    plan->axes_location.set (tag, user_val);
    plan->normalized_coords[i] = normalized_val;

    if (normalized_val != 0)
    {
      plan->pinned_at_default = false;
    }
  }

  plan->old_index_tag_map = old_index_tag_map;
  plan->new_index_tag_map = new_index_tag_map;
