bool axis_not_pinned = false;
if (has_avar)
  seg_maps = fontface->table.avar->get_segment_maps ();

for (unsigned i = 0; i < axes.length; i++)
{
  const OT::AxisRecord &axis = axes[i];
  hb_tag_t tag = axis.tag;
  float user_val = 0;
  int normalized = 0;
  if (plan->user_axes_location.get (tag, &user_val))
  {
    normalized = fontface->table.fvar->normalize_axis_value (i, user_val);
    if (has_avar && seg_maps)
      normalized = seg_maps->map_coord (i, normalized);
  }
  else
  {
    /* Axis not specified by the user: keep default value and mark as not pinned */
    axis_not_pinned = true;
  }
  if (normalized)
    axis_not_pinned = true;

  plan->normalized_coords[i] = normalized;
  plan->axes_location.set (tag, normalized);
}
