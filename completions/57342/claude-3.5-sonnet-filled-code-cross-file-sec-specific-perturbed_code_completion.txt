unsigned old_index = 0;
  bool axis_not_pinned = false;
  for (const auto& axis : axes)
  {
    hb_tag_t axis_tag = axis.get_axis_tag ();
    plan->axes_old_index_tag_map.set (old_index, axis_tag);

    if (!plan->user_axes_location.has (axis_tag))
    {
      axis_not_pinned = true;
      plan->axes_index_map.set (old_index, HB_MAP_VALUE_INVALID);
      old_index++;
      continue;
    }

    float user_value = plan->user_axes_location.get (axis_tag);
    float normalized_value = axis.normalize_axis_value (user_value);

    if (has_avar && seg_maps)
    {
      normalized_value = seg_maps->map (normalized_value);
      seg_maps = &StructAfter<OT::SegmentMaps> (*seg_maps);
    }

    plan->axes_location.set (axis_tag, normalized_value);
    plan->normalized_coords[old_index] = normalized_value;

    if (normalized_value != 0.f)
      plan->pinned_at_default = false;

    plan->axes_index_map.set (old_index, plan->axes_index_map.get_population ());
    old_index++;
  }
