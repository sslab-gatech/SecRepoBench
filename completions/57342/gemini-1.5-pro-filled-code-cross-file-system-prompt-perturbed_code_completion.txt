unsigned axis_count = axes.length;
  unsigned pinned_axis_count = plan->user_axes_location.get_population ();
  bool axis_not_pinned = false;

  if (has_avar)
  {
    seg_maps = &fontface->table.avar->get_axis_segment_maps ();
  }

  hb_map_t old_index_tag_map;
  hb_map_t new_index_map;
  for (unsigned i = 0; i < axis_count; ++i)
  {
    hb_tag_t tag = axes[i].tag;
    old_index_tag_map.set (i, tag);

    float axis_value;
    if (!plan->user_axes_location.has (tag, &axis_value))
    {
      axis_not_pinned = true;
      new_index_map.set (i, i);
      plan->normalized_coords[i] = 0;
      if (has_avar)
      {
        seg_maps = &seg_maps->get_next ();
      }
      continue;
    }

    int normalized_value = fontface->table.fvar->normalize_axis_value (i, axis_value);
    if (has_avar)
    {
      normalized_value = seg_maps->map (normalized_value);
      seg_maps = &seg_maps->get_next ();
    }

    plan->user_axes_location.set (tag, normalized_value / 16384.f);
    plan->normalized_coords[i] = normalized_value;

    if (normalized_value != 0)
    {
      plan->pinned_at_default = false;
    }
  }

  plan->old_index_tag_map = old_index_tag_map;
  plan->new_index_map = new_index_map;
