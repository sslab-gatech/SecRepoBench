unsigned axis_count = axes.length;
  hb_tag_t axes_tags[axis_count];
  bool axis_not_pinned = false;
  unsigned pinned_axis_count = 0;

  if (has_avar)
  {
    seg_maps = &fontface->table.avar->get_axis_segment_maps ();
  }

  for (unsigned i = 0; i < axis_count; ++i)
  {
    hb_tag_t tag = axes[i].axisTag;
    axes_tags[i] = tag;
    plan->axes_old_index_tag_map.set (i, tag);

    float v = 0.f;
    if (!plan->user_axes_location.get (tag, &v))
    {
      axis_not_pinned = true;
      plan->axes_index_map.set (i, i);
    }
    else
    {
      plan->axes_index_map.set (i, pinned_axis_count++);
      float normalized_val = axes[i].normalize_axis_value (v);
      if (has_avar)
      {
	normalized_val = seg_maps[i].map (normalized_val);
      }
      plan->normalized_coords[i] = normalized_val;
      plan->axes_location.set (tag, normalized_val);
      if (normalized_val != 0.f)
      {
	plan->pinned_at_default = false;
      }
    }
  }
