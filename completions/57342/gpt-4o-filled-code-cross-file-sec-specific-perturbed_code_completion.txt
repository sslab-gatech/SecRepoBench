unsigned int axis_index = 0;
  bool axis_not_pinned = false;
  hb_map_t<unsigned, hb_tag_t> old_index_to_tag;
  hb_map_t<unsigned, unsigned> old_index_to_new_index;

  for (const OT::AxisRecord &axis : axes)
  {
    hb_tag_t tag = axis.axisTag;
    old_index_to_tag.set(axis_index, tag);

    if (!plan->user_axes_location.has(tag))
    {
      axis_not_pinned = true;
      old_index_to_new_index.set(axis_index, plan->normalized_coords.length);
      plan->normalized_coords[axis_index] = 0;
    }
    else
    {
      float user_value = plan->user_axes_location.get(tag);
      int normalized_value = axis.normalize_value(user_value);

      if (has_avar)
      {
        if (!seg_maps)
          seg_maps = fontface->table.avar->get_segment_maps();

        normalized_value = seg_maps->remap(axis_index, normalized_value);
      }

      plan->normalized_coords[axis_index] = normalized_value;
      if (normalized_value != 0)
        axis_not_pinned = true;
    }

    axis_index++;
  }
