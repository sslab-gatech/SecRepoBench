hb_map_t axis_index_map;
  bool axis_not_pinned = false;

  for (unsigned int i = 0; i < axes.length; i++)
  {
    hb_tag_t tag = axes[i].axisTag;
    axis_index_map.set(i, tag);

    if (!plan->user_axes_location.has(tag))
    {
      axis_not_pinned = true;
      plan->normalized_coords[i] = 0;
    }
    else
    {
      float user_value = plan->user_axes_location.get(tag);
      plan->normalized_coords[i] = fontface->table.fvar->normalize_axis_value(i, user_value);

      if (has_avar)
      {
        if (!seg_maps)
          seg_maps = fontface->table.avar->get_segment_maps();

        plan->normalized_coords[i] = seg_maps[i].map(plan->normalized_coords[i]);
      }

      if (plan->normalized_coords[i] != 0)
        axis_not_pinned = true;
    }
  }
