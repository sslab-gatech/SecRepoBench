bool axis_not_pinned = false;
  if (has_avar)
    seg_maps = fontface->table.avar->get_segment_maps();

  for (unsigned i = 0; i < axes.length; i++)
  {
    hb_tag_t tag = axes[i].get_axis_tag();
    plan->axes_old_index_tag_map.set (i, tag);

    if (!plan->user_axes_location.has (tag))
    {
      axis_not_pinned = true;
      plan->axes_index_map.add (i);
      continue;
    }

    float value = plan->user_axes_location.get (tag);
    int normalized_value = axes[i].normalize_axis_value (value);

    if (has_avar)
      normalized_value = seg_maps[i].map (normalized_value);

    plan->axes_location.set (tag, normalized_value);
    plan->normalized_coords[i] = normalized_value;

    if (normalized_value != 0)
      plan->pinned_at_default = false;
  }
