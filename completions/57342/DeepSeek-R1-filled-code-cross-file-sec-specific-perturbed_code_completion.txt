bool axis_not_pinned = false;
  unsigned old_idx = 0;
  if (has_avar)
    seg_maps = fontface->table.avar->get_segment_maps();

  for (const auto& axis : axes)
  {
    hb_tag_t tag = axis.get_axis_tag();
    plan->axes_old_index_tag_map.set (old_idx, tag);

    if (plan->user_axes_location.has (tag))
    {
      float user_val = plan->user_axes_location.get (tag);
      int normalized = axis.normalize_axis_value (user_val);

      if (has_avar && seg_maps)
        normalized = seg_maps->map (normalized);

      plan->normalized_coords[old_idx] = normalized;
      plan->axes_location.set (tag, normalized);
      if (normalized != 0)
        plan->pinned_at_default = false;

      if (has_avar)
        seg_maps++;
    }
    else
    {
      axis_not_pinned = true;
      plan->axes_index_map.set (old_idx, plan->axes_index_map.get_population ());
    }
    old_idx++;
  }
