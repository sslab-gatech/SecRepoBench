bool axis_not_pinned = false;
  const OT::SegmentMaps *seg_maps = nullptr;
  if (has_avar)
  {
    if (fontface->table.avar->get_axis_count () == axes.length)
      seg_maps = fontface->table.avar->get_segment_maps ();
    else
      has_avar = false;
  }

  for (unsigned i = 0; i < axes.length; i++)
  {
    hb_tag_t tag = axes[i].axisTag;
    plan->axes_old_index_tag_map.set (i, tag);

    if (!plan->user_axes_location.has (tag))
    {
      axis_not_pinned = true;
      if (has_avar) seg_maps++;
      continue;
    }

    float user_value = plan->user_axes_location.get (tag);
    int normalized_value = axes[i].normalize_axis_value (user_value);

    if (has_avar)
    {
      normalized_value = seg_maps->map (normalized_value);
      seg_maps++;
    }

    plan->normalized_coords.arrayZ[i] = normalized_value;
    plan->axes_location.set (tag, normalized_value);

    if (normalized_value != 0)
      plan->pinned_at_default = false;
  }
