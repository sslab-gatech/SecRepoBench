if (has_avar)
  {
    seg_maps = &fontface->table.avar->get_segment_maps ();
  }

  bool axis_not_pinned = false;
  unsigned axis_idx = 0;
  for (const auto& axis : axes)
  {
    hb_tag_t tag = axis.tag;
    axes_old_index_tag_map.set (axis_idx, tag);

    float v = user_axes_location.get (tag);
    if (v == HB_MAP_VALUE_INVALID)
    {
      axes_location.set (tag, axis.get_default ());
      axis_not_pinned = true;
      axes_index_map.set (axis_idx, axis_idx);
    }
    else
    {
      axes_location.set (tag, v);
      float normalized_v = (v - axis.get_min ()) / (axis.get_max () - axis.get_min ());

      if (has_avar)
      {
        normalized_v = seg_maps->get_mapped_axis_value (axis_idx, normalized_v);
      }

      normalized_coords[axis_idx] = normalized_v;
      axes_location.set (tag, normalized_v * (axis.get_max () - axis.get_min ()) + axis.get_min ());
      if (normalized_v != 0.f)
        pinned_at_default = false;
      axes_index_map.set (axis_idx, axis_idx);
    }

    if (has_avar)
    {
      seg_maps = &fontface->table.avar->get_segment_maps (axis_idx);
    }
    axis_idx++;
  }
