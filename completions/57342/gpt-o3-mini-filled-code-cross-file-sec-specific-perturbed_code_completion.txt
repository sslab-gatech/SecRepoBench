{
  bool axis_not_pinned = false;
  if (has_avar)
    seg_maps = fontface->table.avar->get_segment_maps ();

  for (unsigned int i = 0; i < axes.length; i++)
  {
    hb_tag_t tag = axes[i].get_tag();
    int norm_value;
    if (plan->user_axes_location.has (tag))
    {
      float user_value = plan->user_axes_location.get (tag);
      norm_value = fontface->table.fvar->normalize_axis_value (i, user_value);
    }
    else
    {
      norm_value = fontface->table.fvar->normalize_axis_value (i, 0.f);
      axis_not_pinned = true;
    }

    if (has_avar && seg_maps)
      norm_value = seg_maps->map (i, norm_value);

    plan->normalized_coords[i] = norm_value;
    if (norm_value != 0)
      axis_not_pinned = true;
  }
}
