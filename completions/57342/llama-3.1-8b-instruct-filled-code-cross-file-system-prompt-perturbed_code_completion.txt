// <MASK>
bool axis_not_pinned = false;
for (unsigned int i = 0; i < axes.length; i++)
{
  const OT::AxisRecord &axis = axes[i];
  if (!plan->user_axes_location.has (axis.tag))
  {
    axis_not_pinned = true;
    plan->normalized_coords[i] = axis.get_default ();
  }
  else
  {
    plan->normalized_coords[i] = plan->user_axes_location.get (axis.tag);
  }
}

if (has_avar)
{
  const OT::SegmentMaps *seg_maps = fontface->table.avar->get_segment_maps ();
  for (unsigned int i = 0; i < axes.length; i++)
  {
    const OT::AxisRecord &axis = axes[i];
    if (axis_not_pinned)
    {
      plan->normalized_coords[i] = seg_maps->remap (plan->normalized_coords[i]);
    }
  }
}

plan->all_axes_pinned =!axis_not_pinned;
