if (has_avar)
  {
    seg_maps = &fontface->table.avar->get_segment_maps ();
    plan->avar_maps.resize (axes.length);
  }

  unsigned axis_index = 0;
  bool axis_not_pinned = false;
  for (const auto& axis : axes)
  {
    hb_tag_t tag = axis.tag;
    plan->axes_old_index_tag_map.set (axis_index, tag);

    float v = user_axes_location.get (tag);
    if (v == HB_MAP_VALUE_INVALID)
    {
      axis_not_pinned = true;
      plan->axes_index_map.set (axis_index, axis_index);
    }
    else
    {
      float normalized_v = (v - axis.min_value) / (axis.max_value - axis.min_value);
      if (has_avar)
      {
	plan->avar_maps[axis_index] = seg_maps;
        normalized_v = seg_maps->map (normalized_v);
	seg_maps++;
      }

      plan->axes_location.set (tag, normalized_v);
      plan->normalized_coords[axis_index] = normalized_v;

      if (normalized_v != 0.f)
        pinned_at_default = false;

      plan->axes_index_map.set (axis_index, axis_index);
    }
    axis_index++;
  }
