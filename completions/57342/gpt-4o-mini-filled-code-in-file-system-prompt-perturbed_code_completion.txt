if (has_avar)
    seg_maps = &fontface->table.avar->get_segment_maps ();

  bool axis_not_pinned = false;
  for (unsigned i = 0; i < axes.length; i++)
  {
    const OT::AxisRecord &axis = axes[i];
    plan->normalized_coords[i] = user_axes_location.get (axis.tag, axis.default_value);

    if (plan->normalized_coords[i] != axis.default_value)
      pinned_at_default = false;

    if (has_avar)
      plan->normalized_coords[i] = seg_maps->remap (i, plan->normalized_coords[i]);

    if (plan->normalized_coords[i] != axis.default_value)
      axis_not_pinned = true;

    plan->axes_location.set (axis.tag, plan->normalized_coords[i]);
  }
