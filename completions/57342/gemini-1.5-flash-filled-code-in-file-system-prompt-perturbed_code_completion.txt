bool axis_not_pinned = false;
  for (unsigned i = 0; i < axes.length; i++) {
    hb_tag_t tag = axes[i].tag;
    plan->axes_old_index_tag_map.set (i, tag);
    if (!plan->user_axes_location.has (tag)) {
      axis_not_pinned = true;
      plan->axes_location.set (tag, 0);
      plan->normalized_coords[i] = 0;
      continue;
    }
    float val = plan->user_axes_location.get (tag);
    plan->normalized_coords[i] = val;
    plan->axes_location.set (tag, val);
    if (has_avar) {
      seg_maps = &(fontface->table.avar->get_axis_segment_maps (i));
      val = seg_maps->remap_value (val);
    }
    plan->normalized_coords[i] = val;
    plan->axes_location.set (tag, val);
    if (val != 0)
      plan->pinned_at_default = false;
  }
