unsigned int axis_index = 0;
  bool axis_not_pinned = false;
  hb_map_t old_index_to_tag;
  old_index_to_tag.init ();

  for (unsigned int i = 0; i < axes.length; ++i)
  {
    hb_tag_t axis_tag = axes[i].axisTag;
    old_index_to_tag.set (i, axis_tag);

    float user_value;
    bool user_set = plan->user_axes_location.get (axis_tag, &user_value);

    float normalized_value;
    if (!user_set)
    {
      axis_not_pinned = true;
      plan->old_index_to_new_index.set (i, axis_index++);
    }
    else
    {
      normalized_value = axes[i].normalize_axis_value (user_value);

      if (has_avar)
      {
        seg_maps = fontface->table.avar->get_segment_maps (i);
        if (seg_maps)
          normalized_value = seg_maps->map (normalized_value);
      }

      plan->normalized_coords[i] = normalized_value;
      plan->axes_location.set (axis_tag, normalized_value);

      if (normalized_value != 0.f)
        plan->all_axes_at_default = false;
    }

    if (has_avar)
    {
      if (seg_maps)
        seg_maps++;
      else if (fontface->table.avar->axis_count > 1)
        seg_maps = fontface->table.avar->get_segment_maps (i + 1);
    }
  }
