bool axis_not_pinned = false;
  for (unsigned i = 0; i < axes.length; i++) {
    hb_tag_t tag = axes[i].tag;
    plan->axes_old_index_tag_map.set (i, tag);
    float user_val = 0.f;
    if (plan->user_axes_location.has (tag, &user_val)) {
      float normalized_val = user_val;
      if (has_avar) {
        seg_maps = &(fontface->table.avar->get_segment_maps (i));
        normalized_val = seg_maps->remap_value (normalized_val);
      }
      plan->axes_location.set (tag, normalized_val);
      plan->normalized_coords[i] = normalized_val;
      if (normalized_val != 0.f)
        plan->pinned_at_default = false;
    } else {
      axis_not_pinned = true;
      plan->axes_location.set (tag, 0.f);
      plan->normalized_coords[i] = 0.f;
    }
    if (has_avar)
      seg_maps = nullptr;
  }
