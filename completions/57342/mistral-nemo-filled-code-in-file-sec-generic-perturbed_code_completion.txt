// <MASK>
for (unsigned i = 0; i < axes.length; i++)
{
  OT::Axis axis = axes[i];
  plan->old_index_to_tag_map.set (axis.index, axis.tag);
  if (plan->user_axes_location.has (axis.tag))
  {
    plan->normalized_coords[i] = plan->user_axes_location.get (axis.tag);
    if (plan->normalized_coords[i] != axis.default_value)
      plan->pinned_at_default = false;
  }
  else
  {
    plan->normalized_coords[i] = axis.default_value;
    plan->all_axes_pinned = true;
  }

  if (has_avar)
  {
    const OT::SegmentMaps *seg_maps = fontface->table.avar->get_segment_maps (axis.index);
    if (seg_maps)
    {
      plan->normalized_coords[i] = seg_maps->remap (plan->normalized_coords[i]);
      if (plan->normalized_coords[i] != axis.default_value)
        plan->pinned_at_default = false;
    }
  }
}
