seg_maps = has_avar ? &fontface->table.avar->get_segment_maps() : nullptr;
  unsigned int axis_index = 0;
  bool axis_not_pinned = false;

  for (const OT::AxisRecord &axis : axes)
  {
    hb_tag_t tag = axis.axisTag;
    plan->axes_old_index_tag_map.set(axis_index, tag);

    if (!plan->user_axes_location.has(tag))
    {
      axis_not_pinned = true;
      plan->axes_index_map.set(axis_index, plan->axes_location.get_population());
      plan->axes_location.set(tag, axis.defaultValue);
      plan->normalized_coords[axis_index] = 0;
    }
    else
    {
      float normalized_value = plan->user_axes_location.get(tag);
      normalized_value = (normalized_value - axis.minValue) / (axis.maxValue - axis.minValue) * 2 - 1;

      if (has_avar)
        normalized_value = seg_maps->remap(axis_index, normalized_value);

      plan->axes_location.set(tag, normalized_value);
      plan->normalized_coords[axis_index] = normalized_value;

      if (normalized_value != 0)
        plan->pinned_at_default = false;
    }

    if (has_avar)
      seg_maps = seg_maps->next();
    axis_index++;
  }
