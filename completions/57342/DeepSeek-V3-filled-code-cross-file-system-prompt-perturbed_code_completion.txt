bool axis_not_pinned = false;
  unsigned old_idx = 0;
  for (const OT::AxisRecord& axis : axes)
  {
    hb_tag_t tag = axis.get_axis_tag ();
    plan->axes_old_index_tag_map.set (old_idx, tag);

    if (plan->user_axes_location.has (tag))
    {
      float val = plan->user_axes_location.get (tag);
      int normalized_val = axis.normalize_value (val);
      if (has_avar)
      {
        if (seg_maps == nullptr)
          seg_maps = fontface->table.avar->get_segment_maps (axes.length);
        normalized_val = seg_maps[old_idx].map (normalized_val);
      }
      plan->axes_location.set (tag, normalized_val);
      plan->normalized_coords[old_idx] = normalized_val;
      if (normalized_val != 0)
        plan->pinned_at_default = false;
      if (has_avar)
        seg_maps++;
    }
    else
    {
      axis_not_pinned = true;
      plan->axes_index_map.set (old_idx, plan->axes_index_map.get_population ());
    }
    old_idx++;
  }
