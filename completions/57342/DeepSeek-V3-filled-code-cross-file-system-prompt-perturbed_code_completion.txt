bool axis_not_pinned = false;
  unsigned old_idx = 0;
  for (const auto& axis : axes)
  {
    hb_tag_t tag = axis.axisTag;
    plan->axes_old_index_tag_map.set (old_idx, tag);

    if (plan->user_axes_location.has (tag))
    {
      float user_val = plan->user_axes_location.get (tag);
      float normalized_val = axis.normalize_value (user_val);

      if (has_avar)
      {
        seg_maps = &fontface->table.avar->get_segment_map (old_idx);
        normalized_val = seg_maps->map (normalized_val);
      }

      plan->axes_location.set (tag, normalized_val);
      plan->normalized_coords[old_idx] = normalized_val * 16384.f; // 14.14 fixed point

      if (normalized_val != 0.f)
        plan->pinned_at_default = false;
    }
    else
    {
      axis_not_pinned = true;
      plan->axes_index_map.set (old_idx, plan->axes_index_map.get_population ());
    }
    old_idx++;
  }
