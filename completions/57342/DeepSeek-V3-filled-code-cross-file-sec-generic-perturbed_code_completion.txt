bool axis_not_pinned = false;
  unsigned old_idx = 0;
  for (const auto& axis : axes)
  {
    hb_tag_t tag = axis.axisTag;
    plan->axes_old_index_tag_map.set (old_idx, tag);

    if (plan->user_axes_location.has (tag))
    {
      float normalized_value = axis.normalize_value (plan->user_axes_location.get (tag));
      if (has_avar)
      {
        seg_maps = &fontface->table.avar->get_segment_maps ()[old_idx];
        normalized_value = seg_maps->map (normalized_value);
      }
      plan->axes_location.set (tag, normalized_value);
      plan->normalized_coords[old_idx] = normalized_value;
      if (normalized_value != 0.f)
        plan->all_axes_pinned = false;
    }
    else
    {
      axis_not_pinned = true;
      plan->axes_index_map.set (old_idx, plan->axes_index_map.get_population ());
    }
    old_idx++;
  }
