if ((rv) && ((!words && !wordnum &&
              defcpd_check(&words, wnum, rv, rwords, 0)) ||
             (words &&
              defcpd_check(&words, wnum, rv, rwords, 0)))) {
  // forbid compound word, if it is a non-compound word with typical
  // fault
  if ((checkcompoundrep && cpdrep_check(word, len)) ||
      cpdwordpair_check(word, len))
    return NULL;
  return rv_first;
