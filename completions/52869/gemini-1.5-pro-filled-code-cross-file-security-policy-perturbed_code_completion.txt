                                      include_context->incTab[i]);
            include_context->incTotal++;
            if (ret == 0)
                include_context->incTab[i]->direct = 1;
            if (include_context->incTotal > 10000) {
                xmlXIncludeErr(include_context, cur,
                               XML_XINCLUDE_RECURSION_LIMIT,
                               "Detected a recursive inclusion import\n",
