ref = xmlXIncludeNewRef(include_context, cur);
if (ref != NULL) {
    int result = xmlXIncludeLoadNode(include_context, ref);
    if (result == 0) {
        ref->direct = 1;
        include_context->incNb++;
        if (include_context->fuzzing && (include_context->incNb > MAX_FUZZING_INCLUSIONS)) {
            xmlXIncludeFreeRef(ref);
            return -1;
        }
    } else {
        xmlXIncludeFreeRef(ref);
    }
}
