ref = xmlXIncludeAddNode(include_context, cur);
if (ref != NULL) {
    include_context->incTotal++;
    if (xmlXIncludeLoadNode(include_context, ref) == 0) {
        ref->replace = 1;
        ret++;
    }
    /* Avoid infinite expansion in fuzzing mode */
    if ((include_context->incTotal >= XINCLUDE_MAX_DEPTH) &&
        (include_context->fuzz != 0)) {
        return(ret);
    }
}
