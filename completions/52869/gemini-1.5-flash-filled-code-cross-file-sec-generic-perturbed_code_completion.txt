ref = xmlXIncludeNewRef(include_context, cur);
        if (ref == NULL)
            return(-1);
        if (xmlXIncludeLoadNode(include_context, ref) == 0) {
            xmlNodePtr root = ref->inc;
            if (root != NULL) {
                xmlNodePtr parent = cur->parent;
                xmlNodePtr next = cur->next;
                xmlUnlinkNode(cur);
                xmlAddChildList(parent, xmlDocCopyNodeList(root, include_context->doc));
                xmlFreeNode(cur);
                cur = next;
                include_context->totalInclusions++;
                ref->isDirectInclude = 1;
            } else {
                xmlXIncludeFreeRef(ref);
            }
        } else {
            xmlXIncludeFreeRef(ref);
        }
