} else if ((cur->children != NULL) &&
                   ((cur->type == XML_DOCUMENT_NODE) ||
                    (cur->type == XML_ELEMENT_NODE))) {
            xmlNodePtr expanded_node = xmlXIncludeCopyRange(include_context, cur->children);
            if (expanded_node != NULL) {
                xmlAddChild(cur, expanded_node);
                include_context->incTab[i]->inc = expanded_node;
            }
            cur = cur->children;
            continue;
        }
