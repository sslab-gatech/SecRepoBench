ref = xmlXIncludeAddRef(include_context, cur, NULL);
if (ref!= NULL) {
    if (include_context->fuzzingMode) {
        if (include_context->totalInclusions >= include_context->maxInclusions) {
            xmlXIncludeErr(include_context, cur, XML_XINCLUDE_TOO_MANY_INCLUSIONS,
                           "Too many inclusions, stopping at %d\n",
                           include_context->maxInclusions);
            return(-1);
        }
        include_context->totalInclusions++;
    }
    if (xmlXIncludeLoadNode(include_context, ref) < 0) {
        xmlXIncludeRemoveRef(include_context, ref);
        ref = NULL;
    } else {
        include_context->incNr++;
        cur->flags |= XML_XINCLUDE_START;
    }
}
