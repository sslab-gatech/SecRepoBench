ref = xmlMemMalloc(sizeof(xmlXIncludeRef));
            if (ref == NULL) {
                xmlXIncludeErr(include_context, cur, XML_XINCLUDE_MEMORY,
                               "cannot allocate XInclude reference\n", NULL);
                return(-1);
            }
            memset(ref, 0, sizeof(xmlXIncludeRef));
            ref->elem = cur;
            ref->doc = include_context->doc;
            ref->inc = NULL;
            ref->emptyFb = 0;
            ref->replace = 1;
            if (include_context->incNr == include_context->incMax) {
                include_context->incMax *= 2;
                include_context->incTab = (xmlXIncludeRefPtr *) xmlRealloc(include_context->incTab,
                                                                            include_context->incMax * sizeof(xmlXIncludeRefPtr));
                if (include_context->incTab == NULL) {
                    xmlXIncludeErr(include_context, cur, XML_XINCLUDE_MEMORY,
                                   "cannot allocate XInclude references\n", NULL);
                    xmlFree(ref);
                    return(-1);
                }
            }
            include_context->incTab[include_context->incNr++] = ref;
            if (xmlXIncludeLoadNode(include_context, ref) == 0)
                ref->flags |= XML_XINCLUDE_INCLUDE_DIRECT;
#ifdef LIBXML_XINCLUDE_FUZZ
            include_context->nb_process++;
            if (include_context->nb_process > 200)
                return(0);
#endif
