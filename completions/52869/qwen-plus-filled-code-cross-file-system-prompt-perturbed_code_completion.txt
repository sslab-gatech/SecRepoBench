ref = xmlXIncludeLoadNode(include_context, cur);
if (ref != NULL) {
    include_context->incNr++;
    include_context->incTab = xmlRealloc(include_context->incTab, include_context->incNr * sizeof(xmlXIncludeRefPtr));
    if (include_context->incTab == NULL) {
        xmlXIncludeErr(include_context, cur, XML_XINCLUDE_REPLACEMENT,
                       "Failed to allocate memory for incTab\n", NULL);
        return(-1);
    }
    include_context->incTab[include_context->incNr - 1] = ref;
    ref->direct = 1;
}
#ifdef FUZZING_BUILD_MODE_UNSAFE_FOR_PRODUCTION
if (include_context->incNr > MAX_INCLUDE_COUNT) {
    /* Prevent excessive growth in fuzzing mode */
    return(-1);
}
#endif
