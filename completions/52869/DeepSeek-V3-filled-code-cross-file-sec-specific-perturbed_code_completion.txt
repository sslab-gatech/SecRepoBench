ref = xmlXIncludeAddNode(include_context, cur);
            if (ref != NULL) {
                if (include_context->incNr >= include_context->incMax) {
                    include_context->incMax *= 2;
                    include_context->incTab = (xmlXIncludeRefPtr *) xmlRealloc(
                        include_context->incTab,
                        include_context->incMax * sizeof(xmlXIncludeRefPtr));
                    if (include_context->incTab == NULL) {
                        xmlXIncludeErr(include_context, cur, XML_XINCLUDE_NO_MEMORY,
                                     "failed to allocate memory for include table\n");
                        return(-1);
                    }
                }
                include_context->incTab[include_context->incNr++] = ref;
                if (include_context->options & XML_XINCLUDE_LIMIT_INCLUDES) {
                    if (include_context->incNr > XML_XINCLUDE_MAX_INCLUDES) {
                        xmlXIncludeErr(include_context, cur, XML_XINCLUDE_TOO_MANY_INCLUDES,
                                     "too many includes, limit reached\n");
                        return(-1);
                    }
                }
                ref->direct = 1;
            }
