} else if ((cur->children != NULL) &&
                   ((cur->type == XML_DOCUMENT_NODE) ||
                    (cur->type == XML_ELEMENT_NODE))) {
            xmlNodePtr expandedNode = xmlXIncludeCopyRange(include_context, cur->children);
            if (expandedNode != NULL) {
                xmlAddChild(include_context->doc, expandedNode);
                include_context->incTab[i]->inc = expandedNode;
            }
            cur = cur->children;
            continue;
        }
