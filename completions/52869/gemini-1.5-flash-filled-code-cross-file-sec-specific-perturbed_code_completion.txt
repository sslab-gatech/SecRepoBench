ref = xmlXIncludeExpandNode(include_context, cur);
            if (ref != NULL) {
                include_context->incNr++;
                include_context->incTab = (xmlXIncludeRefPtr *)
                    xmlRealloc(include_context->incTab,
                               include_context->incNr *
                               sizeof(xmlXIncludeRefPtr));
                if (include_context->incTab == NULL) {
                    xmlXIncludeErr(include_context, cur,
                                   XML_XINCLUDE_ALLOC_FAIL,
                                   "Memory allocation failed\n");
                    xmlFree(ref);
                    return(-1);
                }
                include_context->incTab[include_context->incNr - 1] = ref;
                ref->direct = 1;
            }
