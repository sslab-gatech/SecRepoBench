ref->replace = 1; // Mark this node for replacement
            include_context->incTotal++; // Increment the total inclusions processed
            if (include_context->incTotal > XINCLUDE_MAX_DEPTH) {
                xmlXIncludeErr(ctxt, node, XML_XINCLUDE_RECURSION,
                               "maximum recursion depth exceeded\n", NULL);
                return NULL;
            }
            return ref; // Return the reference for the included node
