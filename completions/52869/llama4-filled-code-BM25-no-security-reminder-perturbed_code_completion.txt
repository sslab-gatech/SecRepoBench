int include_count = include_context->incNr;
ref = xmlXIncludeProcessNode(include_context, cur);
if (ref != NULL) {
    include_context->incTab[include_context->incNr] = ref;
    include_context->incNr++;
    if (include_context->incNr > include_context->incMax) {
        include_context->incMax *= 2;
        include_context->incTab = (xmlXIncludeRefPtr *) xmlRealloc(include_context->incTab,
            include_context->incMax * sizeof(xmlXIncludeRefPtr));
    }
    if (include_count != include_context->incNr) {
        cur->include = 1;
    }
#ifdef FUZZ_XINCLUDE
    if (include_context->fuzzing && include_context->incNr > 1000) {
        return -1;
    }
#endif
}
