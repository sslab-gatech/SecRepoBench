if (xmlXIncludeTestNode(include_context, cur) == 1) {
    // Expand the current node and increment the total inclusion count.
    // Mark the node as a direct include if the expansion is successful.
    // If in fuzzing mode, limit the total number of expansions to prevent excessive growth.
    if (include_context->fuzzing) {
        if (include_context->incNr >= include_context->maxInclusions) {
            xmlXIncludeErr(include_context, cur, XML_XINCLUDE_MAX_INCLUDES,
                           "Maximum number of inclusions reached\n");
            break;
        }
    }
    if (xmlXIncludeLoadNode(include_context, cur) < 0) {
        xmlXIncludeErr(include_context, cur, XML_XINCLUDE_LOAD_ERROR,
                       "Failed to load %s\n", cur->name);
        break;
    }
    include_context->incNr++;
    include_context->incTab[include_context->incNr - 1] = cur;
    if (cur->type == XML_ELEMENT_NODE) {
        xmlNodePtr child = cur->children;
        while (child!= NULL) {
            if (xmlXIncludeTestNode(include_context, child) == 1) {
                xmlXIncludeDoProcess(include_context, child);
            }
            child = child->next;
        }
    }
    break;
} else if ((cur->children!= NULL) &&
           ((cur->type == XML_DOCUMENT_NODE) ||
            (cur->type == XML_ELEMENT_NODE))) {
    cur = cur->children;
    continue;
}
