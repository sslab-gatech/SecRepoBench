ref = xmlXIncludeNewRef(include_context, cur);
            if (ref == NULL) {
                xmlXIncludeErrMemory(include_context, cur, "processing XInclude");
                return(-1);
            }
            ref->inc = xmlXIncludeLoadNode(include_context, ref);
            if (ref->inc == NULL) {
                xmlXIncludeErr(include_context, cur, XML_XINCLUDE_NO_FALLBACK,
                               "could not load %s\n", ref->URI);
                xmlXIncludeFreeRef(ref);
                return(-1);
            }
            include_context->incTab[include_context->incNr++] = ref;
