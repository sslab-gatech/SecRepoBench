if (include_context->options & XML_XINCLUDE_RECURSE) {
                ref = xmlXIncludeAddRef(include_context, cur);
                if (ref == NULL) {
                    ret = -1;
                    break;
                }
                if (include_context->incNr > XML_XINCLUDE_MAX_DEPTH) {
                    xmlXIncludeErr(include_context, cur,
                                   XML_XINCLUDE_MAX_RECURSION,
                                   "Maximum recursion depth reached\n", NULL);
                    ret = -1;
                    break;
                }
		if (include_context->options & XML_XINCLUDE_FUZZ) {
		    if (include_context->total_inc > 1000) {
			xmlXIncludeErr(include_context, cur,
					   XML_XINCLUDE_MAX_RECURSION,
					   "Fuzzing: Maximum recursion depth reached\n", NULL);
			ret = -1;
			break;
		    }
		}
                ref->direct_include = 1;
		include_context->total_inc++;
                ret = xmlXIncludeLoadNode(include_context, ref);
                if (ret < 0) {
                    /*
                     * An error occurred, don't process further includes
                     */
                    break;
                }
            }
