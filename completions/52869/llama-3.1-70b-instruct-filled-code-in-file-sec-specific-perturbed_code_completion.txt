ref = xmlXIncludeExpandNode(include_context, cur);
if (ref!= NULL) {
    include_context->incTotal++;
    if (include_context->incTotal > 1000000) {
        xmlXIncludeErr(include_context, cur, XML_XINCLUDE_RECURSION,
                       "too many inclusions\n", NULL);
        return -1;
    }
    ref->replace = 1;
}
