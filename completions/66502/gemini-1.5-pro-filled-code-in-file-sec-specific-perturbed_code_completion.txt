        if (ret != NULL) {
            if (streaming) {
                if (ret->attr == NULL) {
                    if (doc->dict != NULL)
                        ret->name = xmlDictLookup(doc->dict, attribute->name, -1);
                    else
                        ret->name = xmlStrdup(attribute->name);
                    if (ret->name == NULL)
                        return (-1);
                }
            } else {
                ret->attr = attribute;
            }
            attribute->atype = XML_ATTRIBUTE_ID;
            if (!streaming)
                attribute->id = ret;
            if (id != NULL)
                *id = ret;

            return (0);
