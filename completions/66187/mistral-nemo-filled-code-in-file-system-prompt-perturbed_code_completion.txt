# Solution for the given problem

def is_palindrome(s):
    return s == s[::-1]

def is_palindrome_number(n):
    return str(n) == str(n)[::-1]

def is_palindrome_list(lst):
    return lst == lst[::-1]

def is_palindrome_dict(dct):
    return dct == {k: v for k, v in dct.items()}

def is_palindrome_set(s):
    return s == {i for i in s}

def is_palindrome_frozenset(frz):
    return frozenset(frz) == {i for i in frozenset(frz)}

def is_palindrome_tuple(tpl):
    return tpl == tuple(reversed(tpl))

def is_palindrome_range(start, end):
    return range(start, end) == range(end, start, -1)

def is_palindrome_reverse(s):
    return s[::-1] == s

def is_palindrome_reverse_number(n):
    return int(str(n)[::-1]) == n

def is_palindrome_reverse_list(lst):
