// <MASK>
else if ( (linebuffer[i] & 0xf0) == 0xe0) {
    if (j >= GF_ARRAY_LENGTH(szLineConv))
        break;
    szLineConv[j] = 0xc0 | ( (linebuffer[i] >> 6) & 0x3 );
    j++;
    i++;
    if (i >= lineSize)
        break;
    if ((linebuffer[i] & 0xc0)!= 0x80) {
        if (j >= GF_ARRAY_LENGTH(szLineConv))
            break;
        szLineConv[j] = 0x80 | ( (linebuffer[i] >> 6) & 0x3 );
        j++;
        linebuffer[i] &= 0x3f;
    }
}
