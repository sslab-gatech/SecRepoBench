else if ((linebuffer[i] & 0xf0) == 0xe0) {
	if (j + 1 >= GF_ARRAY_LENGTH(szLineConv)) {
		break;
	}
	szLineConv[j] = linebuffer[i];
	i++;
	szLineConv[j+1] = linebuffer[i];
	i++;
	j+=2;
}
/*UTF8 4 bytes char*/
else if ((linebuffer[i] & 0xf8) == 0xf0) {
	if (j + 2 >= GF_ARRAY_LENGTH(szLineConv)) {
		break;
	}
	szLineConv[j] = linebuffer[i];
	i++;
	szLineConv[j+1] = linebuffer[i];
	i++;
	szLineConv[j+2] = linebuffer[i];
	i++;
	j+=3;
