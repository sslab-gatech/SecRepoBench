if (!a)
    return GF_OK;

if (container->size < a->size) {
    if (!skip_logs)
        fprintf(stderr, "Warning: Container size (%u) is less than parsed box size (%u).\n", container->size, a->size);
    container->size = 0;
} else {
    container->size -= a->size;
}

if (container->registry && container->registry->is_valid_box) {
    if (!container->registry->is_valid_box(container, a)) {
        if (!skip_logs)
            fprintf(stderr, "Warning: Box type 0x%X is not valid in container type 0x%X.\n", a->type, container->type);
    }
}
