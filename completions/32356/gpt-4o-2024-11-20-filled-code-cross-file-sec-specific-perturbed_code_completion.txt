if (!a) {
            return GF_OK; // No further processing needed if the parsed box is NULL
        }

        if (container->size < a->size) {
            if (!skip_logs) {
                gf_log(GF_LOG_WARNING, GF_LOG_CONTAINER, 
                       "Parsed box size exceeds parent box size. Parent type: %s, Parsed box type: %s\n", 
                       gf_isom_4cc_to_str(parent_type), gf_isom_4cc_to_str(a->type));
            }
            container->size = 0; // Reset parent box size to prevent further processing
            gf_isom_box_del(a); // Free the parsed box to avoid memory leaks
            return GF_OK;
        }

        container->size -= a->size; // Adjust parent box size

        if (!gf_isom_is_box_valid_in_container(parent_type, a->type)) {
            if (!skip_logs) {
                gf_log(GF_LOG_WARNING, GF_LOG_CONTAINER, 
                       "Invalid box type %s in container type %s\n", 
                       gf_isom_4cc_to_str(a->type), gf_isom_4cc_to_str(parent_type));
            }
            gf_isom_box_del(a); // Free the invalid box
            continue; // Skip to the next iteration
        }
