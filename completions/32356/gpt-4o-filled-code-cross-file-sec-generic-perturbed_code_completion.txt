if (!a) {
            return GF_OK;
        }

        if (container->size < a->size) {
            if (!skip_logs) {
                gf_log(GF_LOG_WARNING, GF_LOG_CONTAINER, "Parsed box size exceeds parent box size\n");
            }
            container->size = 0;
            return GF_OK;
        }

        container->size -= a->size;

        if (!gf_isom_check_box_compatibility(container, a)) {
            if (!skip_logs) {
                gf_log(GF_LOG_WARNING, GF_LOG_CONTAINER, "Parsed box is not compatible with its parent container\n");
            }
            gf_isom_box_del(a);
            return GF_OK;
        }
