if (!a) {
	// parsed box is NULL, no further boxes to process
	return GF_OK;
}

if (container->size < a->size) {
	// the parsed box is larger than the remaining size of the parent box
	if (!skip_logs) {
		GF_LOG(GF_LOG_WARNING, GF_LOG_CONTAINER,
			("[iso file] Box %s (size %u) exceeds parent box leftover %llu\n",
			gf_4cc_to_str(a->type), a->size, (unsigned long long)container->size));
	}
	// reset any leftover parent size
	container->size = 0;
} else {
	// update remaining parent box size
	container->size -= a->size;
}

// validate the parsed box within its parent type using the registry
if (container->registry && container->registry->is_box_valid_in &&
	!container->registry->is_box_valid_in(parent_type, a->type)) {
	if (!skip_logs) {
		GF_LOG(GF_LOG_WARNING, GF_LOG_CONTAINER,
			("[iso file] Box %s is not valid in container %s\n",
			gf_4cc_to_str(a->type), gf_4cc_to_str(parent_type)));
	}
}
