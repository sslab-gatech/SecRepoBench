GF_Err gf_isom_box_array_dump(GF_List *boxlist, FILE * trace)
{
	u32 count, i;
	if (!boxlist) return GF_OK;
	count = gf_list_count(boxlist);
	for (i = 0; i < count; i++) {
		GF_Box *a = (GF_Box *)gf_list_get(boxlist, i);
		if (a) {
			gf_isom_box_dump_start(a, gf_4cc_to_str(a->type), trace);
			gf_isom_box_dump(a, trace);
			gf_isom_box_dump_done(gf_4cc_to_str(a->type), a, trace);
		}
	}
	return GF_OK;
}
