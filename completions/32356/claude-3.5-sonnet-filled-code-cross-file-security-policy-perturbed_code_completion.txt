if (!a) return GF_OK;

if (container->size < a->size) {
    if (!skip_logs) {
        GF_LOG(GF_LOG_WARNING, GF_LOG_CONTAINER, ("[iso file] Box \"%s\" size %d exceeds parent box size %d\n", gf_4cc_to_str(a->type), a->size, container->size));
    }
    container->size = 0;
} else {
    container->size -= a->size;
}

if (container->registry->valid_in) {
    Bool is_valid = GF_FALSE;
    u32 i, count = container->registry->valid_in_count;
    for (i=0; i<count; i++) {
        if (container->registry->valid_in[i] == a->type) {
            is_valid = GF_TRUE;
            break;
        }
    }
    if (!is_valid && !skip_logs) {
        GF_LOG(GF_LOG_WARNING, GF_LOG_CONTAINER, ("[iso file] Box \"%s\" is not valid in container \"%s\"\n", gf_4cc_to_str(a->type), gf_4cc_to_str(container->type)));
    }
}
