if (!a) {
	GF_LOG(GF_LOG_ERROR, GF_LOG_CONTAINER, ("[iso file] Read Box failed to allocate box\n"));
	return GF_OUT_OF_MEM;
}

if (container->size < a->size) {
	if (!skip_logs) {
		GF_LOG(GF_LOG_WARNING, GF_LOG_CONTAINER, ("[iso file] Box \"%s\" size "LLU" invalid (child \"%s\" size "LLU" larger than remaining size "LLU")\n", gf_4cc_to_str(container->type), container->size, gf_4cc_to_str(a->type), a->size, container->size));
	}
}
container->size -= a->size;
if (container->size < 0) container->size = 0;

if (!gf_box_valid_in_parent(a, gf_4cc_to_str(parent_type))) {
	if (!skip_logs) {
		GF_LOG(GF_LOG_WARNING, GF_LOG_CONTAINER, ("[iso file] Box \"%s\" not valid in container \"%s\"\n", gf_4cc_to_str(a->type), gf_4cc_to_str(parent_type)));
	}
}
