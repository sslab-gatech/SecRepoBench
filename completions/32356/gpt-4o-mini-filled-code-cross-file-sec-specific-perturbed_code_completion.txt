if (a) {
    if (container->size < a->size) {
        if (!skip_logs) {
            // Log a warning about the size mismatch
            gf_log_warning("Parsed box size exceeds parent box size.");
        }
        // Reset the parent box size to zero or handle accordingly
        container->size = 0;
    } else {
        container->size -= a->size;
    }

    // Validate the box type compatibility
    if (!is_box_type_compatible(container->type, a->type)) {
        if (!skip_logs) {
            // Log a warning about the invalid box type
            gf_log_warning("Parsed box type is not compatible with the container.");
        }
        gf_isom_box_del(a);
        return GF_ISOM_INVALID_MEDIA;
    }
