if (!a) return GF_OK;
if (container->size < a->size) {
    if (!skip_logs) {
        gf_isom_log(GF_LOG_WARNING, GF_LOG_CONTAINER, ("box size exceeded (0x%08X) - ignoring\n", container->size));
    }
    container->size = 0;
} else {
    container->size -= a->size;
}
if (!gf_isom_box_type_compatible(container->type, a->type)) {
    if (!skip_logs) {
        gf_isom_log(GF_LOG_WARNING, GF_LOG_CONTAINER, ("invalid box type 0x%08X in container 0x%08X\n", a->type, container->type));
    }
}
