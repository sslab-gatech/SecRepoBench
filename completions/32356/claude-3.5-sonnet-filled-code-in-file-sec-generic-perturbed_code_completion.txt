if (!a) return GF_OK;

if (container->size < a->size) {
    if (!skip_logs) {
        GF_LOG(GF_LOG_WARNING, GF_LOG_CONTAINER, ("[iso file] Box %s size "LLU" larger than container size "LLU"\n", gf_4cc_to_str(a->type), a->size, container->size));
    }
    container->size = 0;
} else {
    container->size -= a->size;
}

if (!gf_box_valid_in_parent(a, gf_4cc_to_str(parent_type))) {
    if (!skip_logs) {
        GF_LOG(GF_LOG_WARNING, GF_LOG_CONTAINER, ("[iso file] Box \"%s\" is not valid in container %s\n", gf_4cc_to_str(a->type), gf_4cc_to_str(parent_type)));
    }
}
