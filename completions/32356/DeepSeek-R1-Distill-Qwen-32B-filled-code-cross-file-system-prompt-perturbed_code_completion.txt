if (!a) return GF_OK;
if (container->size < a->size) {
    if (!skip_logs) gf_log(GF_LOG_WARNING, "Box size %d exceeds parent size %d", a->size, container->size);
    container->size -= a->size;
    if (container->size <0) container->size = 0;
}
if (container->registry->validate_box) {
    e = container->registry->validate_box(container, a);
    if (e && !skip_logs) gf_log(GF_LOG_WARNING, "Box type %s is invalid in container %s", gf_isom_get_box_type(a->type), gf_isom_get_box_type(container->type));
}
