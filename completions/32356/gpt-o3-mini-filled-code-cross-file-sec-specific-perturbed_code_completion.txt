if (!a)
	return GF_OK;

if (container->size < a->size) {
	if (!skip_logs)
		gf_log("Warning: Parsed box size (%u) exceeds remaining container size (%u)", a->size, container->size);
	container->size = 0;
} else {
	container->size -= a->size;
}

if (container->registry && container->registry->validate_child_box) {
	/* validate_child_box returns non-zero (true) if the box is valid */
	if (!container->registry->validate_child_box(container, a)) {
		if (!skip_logs)
			gf_log("Warning: Box type 0x%X is not valid within container type 0x%X", a->type, container->type);
		gf_isom_box_del(a);
		continue;
	}
}
