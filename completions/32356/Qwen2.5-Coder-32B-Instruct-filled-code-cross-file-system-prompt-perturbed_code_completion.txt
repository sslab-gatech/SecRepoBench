if (!a) return GF_OK;

if (container->size < a->size) {
    if (!skip_logs) {
        GF_LOG(GF_LOG_WARNING, GF_LOG_CONTAINER, ("[iso file] Parent box size is smaller than parsed box size\n"));
    }
    container->size = 0; // Reset size if exceeded
} else {
    container->size -= a->size;
}

if (!gf_isom_box_type_compatible(parent_type, a->type)) {
    if (!skip_logs) {
        GF_LOG(GF_LOG_WARNING, GF_LOG_CONTAINER, ("[iso file] Box type %s is invalid in container type %s\n", gf_4cc_to_str(a->type), gf_4cc_to_str(parent_type)));
    }
}
