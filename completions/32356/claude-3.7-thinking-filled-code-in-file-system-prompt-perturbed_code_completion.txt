if (!a) {
    return GF_OK;
}
if (container->size < a->size) {
    if (!skip_logs) {
        GF_LOG(GF_LOG_WARNING, GF_LOG_CONTAINER, ("[iso file] Box \"%s\" larger than container %s\n", gf_4cc_to_str(a->type), gf_4cc_to_str(container->type)));
    }
    container->size = 0;
} else {
    container->size -= a->size;
}

// Check if the box is valid for this container
if (a->registry && !gf_box_valid_in_parent(a, gf_4cc_to_str(container->type))) {
    if (!skip_logs) {
        GF_LOG(GF_LOG_WARNING, GF_LOG_CONTAINER, ("[iso file] Box \"%s\" is not a valid box in container %s\n", gf_4cc_to_str(a->type), gf_4cc_to_str(container->type)));
    }
}
