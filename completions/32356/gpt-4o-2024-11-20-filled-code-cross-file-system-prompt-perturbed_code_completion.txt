// Ensure the parsed box `a` is not NULL
		if (!a) return GF_OK;

		// Check if the size of the parent box is smaller than the size of the newly parsed box `a`
		if (container->size < a->size) {
			if (!skip_logs) {
				gf_log(GF_LOG_WARNING, GF_LOG_CONTAINER, 
					"Box size mismatch: parent box size (%"PRIu64") is smaller than child box size (%"PRIu64")\n", 
					container->size, a->size);
			}
			container->size = 0; // Reset the parent box size
		} else {
			container->size -= a->size; // Adjust the parent box size
		}

		// Validate if the container box type is compatible with the parsed box `a`
		if (!gf_isom_check_box_compatibility(container->type, a->type)) {
			if (!skip_logs) {
				gf_log(GF_LOG_WARNING, GF_LOG_CONTAINER, 
					"Incompatible box type: parent type 0x%08x, child type 0x%08x\n", 
					container->type, a->type);
			}
			gf_isom_box_del(a); // Delete the invalid box
			continue; // Skip further processing for this box
		}
