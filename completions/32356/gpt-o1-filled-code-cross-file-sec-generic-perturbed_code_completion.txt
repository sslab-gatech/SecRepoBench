if (!a) {
	// sub box parsing aborted with no error
	// Return success indicating no further processing is needed
	return GF_OK;
}

// Check if the size of the parent box is smaller than the size of the newly parsed box 'a'
if (container->size < a->size) {
	// Log a warning if logging is enabled
	if (!skip_logs) {
		gf_log(GF_LOG_WARNING, GF_LOG_CONTAINER, "Parsed box size (%u) exceeds remaining container size (%u).", a->size, container->size);
	}
}

// Adjust the parent box size by subtracting the size of the parsed box 'a', or reset it if the size was exceeded
if (container->size >= a->size) {
	container->size -= a->size;
} else {
	container->size = 0;
}

// Validate if the parent box type is valid for the parsed box 'a'
if (container->registry && container->registry->is_box_valid_in_container &&
	!container->registry->is_box_valid_in_container(container, a))
{
	// Log a warning if the parsed box is invalid in its current container, but only if logging is not skipped
	if (!skip_logs) {
		gf_log(GF_LOG_WARNING, GF_LOG_CONTAINER, "Box type 0x%08x is invalid in container 0x%08x.", a->type, parent_type);
	}
}
