if (!a) return GF_OK;
if (container->size < a->size) {
    if (!skip_logs) {
        GF_LOG(GF_LOG_WARNING, GF_LOG_CONTAINER, ("[iso file] Box \"%s\" size %d larger than container size %d\n", gf_4cc_to_str(a->type), a->size, container->size));
    }
    container->size = 0;
} else {
    container->size -= a->size;
}
if (container->registry && container->registry->box_type_valid) {
    if (!container->registry->box_type_valid(container->type, a->type)) {
        if (!skip_logs) {
            GF_LOG(GF_LOG_WARNING, GF_LOG_CONTAINER, ("[iso file] Box \"%s\" is not a valid box in container %s\n", gf_4cc_to_str(a->type), gf_4cc_to_str(container->type)));
        }
    }
}
