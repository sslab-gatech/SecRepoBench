if (!a) return GF_OK;

		if (container->size < a->size) {
			if (!skip_logs) {
				GF_LOG(GF_LOG_WARNING, GF_LOG_CONTAINER, ("[iso file] Box size exceeds container size\n"));
			}
			container->size = 0;
			return GF_OK;
		}

		container->size -= a->size;

		if (!gf_box_valid_in_parent(a, gf_4cc_to_str(container->type))) {
			if (!skip_logs) {
				GF_LOG(GF_LOG_WARNING, GF_LOG_CONTAINER, ("[iso file] Box %s is invalid in container %s\n", gf_4cc_to_str(a->type), gf_4cc_to_str(container->type)));
			}
			gf_isom_box_del(a);
			continue;
		}
