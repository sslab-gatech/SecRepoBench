if (!a) return GF_OK;

if (container->size < a->size) {
    if (!skip_logs) {
        GF_LOG(GF_LOG_WARNING, GF_LOG_CONTAINER, ("[iso file] Box \"%s\" size %d larger than remaining parent box size %d\n", gf_4cc_to_str(a->type), a->size, container->size));
    }
    container->size = 0;
} else {
    container->size -= a->size;
}

// Check if the box is valid in this container
if (container->registry && container->registry->valid_box && !container->registry->valid_box(container, a)) {
    if (!skip_logs) {
        GF_LOG(GF_LOG_WARNING, GF_LOG_CONTAINER, ("[iso file] Box \"%s\" is not valid in container \"%s\"\n", gf_4cc_to_str(a->type), gf_4cc_to_str(container->type)));
    }
}
