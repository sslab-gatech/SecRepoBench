if (!a) return GF_OK;

if (container->size < a->size) {
    if (!skip_logs) {
        GF_LOG(GF_LOG_WARNING, GF_LOG_CONTAINER, ("[iso file] Box \"%s\" size %d exceeds parent box \"%s\" remaining size %d\n",
            gf_4cc_to_str(a->type), a->size, gf_4cc_to_str(container->type), container->size));
    }
    container->size = 0;
} else {
    container->size -= a->size;
}

if (container->registry->valid_in) {
    if (!container->registry->valid_in(container->type, a->type)) {
        if (!skip_logs) {
            GF_LOG(GF_LOG_WARNING, GF_LOG_CONTAINER, ("[iso file] Box \"%s\" is not allowed in container \"%s\"\n",
                gf_4cc_to_str(a->type), gf_4cc_to_str(container->type)));
        }
    }
}
