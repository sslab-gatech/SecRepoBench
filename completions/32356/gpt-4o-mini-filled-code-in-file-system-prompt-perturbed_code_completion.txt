    if (container->size < a->size) {
        if (!skip_logs) {
            GF_LOG(GF_LOG_WARNING, GF_LOG_CONTAINER, ("[iso file] Box \"%s\" size "LLU" exceeds parent box size "LLU"\n", gf_4cc_to_str(a->type), a->size, container->size));
