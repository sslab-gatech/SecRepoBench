unsigned int n_regions = interp_env.argStack.pop_uint ();
unsigned int n = interp_env.argStack.pop_uint ();
if (n_regions > interp_env.get_region_count () || n > interp_env.argStack.get_count ())
  return;

hb_array_t<const blend_arg_t> blend_args = interp_env.blendArg.as_array (n);
for (unsigned int i = 0; i < n; i++)
{
  const blend_arg_t &arg = blend_args[i];
  hb_array_t<const number_t> values = interp_env.argStack.sub_array (arg.deltas_idx, n_regions);
  if (unlikely (!arg.blending.set_values (values)))
    return;
}
interp_env.argStack.pop (n * (1 + n_regions));
