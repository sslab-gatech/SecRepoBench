// Retrieve the number of blend regions from the argument stack
    unsigned int num_regions = interp_env.get_num_regions();

    // Ensure there are enough arguments on the stack for blending
    if (param.argStack.size() < num_regions + 1)
      throw std::runtime_error("Insufficient arguments for blend operation");

    // Retrieve the default value (first argument) from the stack
    float default_value = param.argStack.pop();

    // Prepare a blend array to store the blended values
    std::vector<float> blend_values(num_regions, default_value);

    // Populate the blend array with values from the argument stack
    for (unsigned int i = 0; i < num_regions; ++i)
    {
      blend_values[i] += param.argStack.pop();
    }

    // Apply the blended values to the environment
    interp_env.apply_blend(blend_values);

    // Clear any remaining arguments from the stack
    interp_env.clear_args();
