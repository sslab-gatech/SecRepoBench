unsigned int numBlends = interp_env.pop_arg ().to_int ();
    unsigned int valueIndex = interp_env.pop_arg ().to_int ();
    unsigned int numValues = interp_env.pop_arg ().to_int ();

    if (unlikely (!numValues || !interp_env.check_args (numValues + numBlends * numValues)))
    {
      interp_env.set_error ();
      return;
    }

    interp_env.process_blend ();

    unsigned int i = 0;
    do {
      BlendArg  v = interp_env.argStack[i];
      for (unsigned int j = 0; j < numBlends; j++)
      {
	interp_env.argStack[i + 1 + j * numValues].set_blends (numValues, valueIndex,
							     interp_env.get_region_count (),
							     &interp_env.argStack[i + numValues + j * numValues]);
      }
      interp_env.argStack[i] = v;
      i++;
    } while (i < numValues);
    interp_env.argStack.pop (numBlends * numValues);
