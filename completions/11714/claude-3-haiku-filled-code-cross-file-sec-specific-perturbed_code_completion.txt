size_t blend_count = param.getBlendCount();
if (blend_count > 0) {
    float* blend_values = param.getBlendValues();
    if (blend_values != nullptr && blend_count <= interp_env.getBlendArraySize()) {
        for (size_t i = 0; i < blend_count; i++) {
            interp_env.setBlendValue(i, blend_values[i]);
        }
        param.popBlendValues(blend_count);
    } else {
        // Handle error case: blend_count is greater than the blend array size
        // or blend_values is a null pointer
    }
}
