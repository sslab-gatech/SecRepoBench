int region_count = interp_env.get_region_count();
    float blend_value = interp_env.get_blend_value();

    std::vector<float> blend_array(region_count);
    for (int i = 0; i < region_count; ++i) {
        blend_array[i] = interp_env.get_arg_value(i) * blend_value;
    }

    interp_env.clear_args(); // Clear the argument stack after processing

    // Apply blend values to the default values
    for (int i = 0; i < region_count; ++i) {
        float default_value = interp_env.get_default_value(i);
        interp_env.set_blended_value(i, default_value + blend_array[i]);
    }
