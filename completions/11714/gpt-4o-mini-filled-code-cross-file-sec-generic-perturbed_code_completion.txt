int region_count = param.get_region_count();
    float blend_value = param.get_blend_value();
    
    // Assuming blend_array is a member of interp_env that holds the blend values
    for (int i = 0; i < region_count; ++i) {
        float default_value = interp_env.get_default_value(i);
        interp_env.blend_array[i] = default_value + (blend_value * default_value);
    }

    // Clear the argument stack after processing
    param.clear_blend_values();
