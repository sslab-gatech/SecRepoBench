unsigned int region_count = interp_env.argStack.pop_uint ();
    unsigned int num_value = interp_env.argStack.pop_uint ();

    if (region_count > param.blend_count)
      return; // Ensure we don't exceed the blend array size

    for (unsigned int i = 0; i < region_count; i++)
    {
      if (i >= param.blend_count)
        break; // Prevent buffer overflow
      param.blend_array[i] = interp_env.argStack.pop_num ();
    }

    for (unsigned int i = 0; i < num_value; i++)
    {
      if (i >= param.default_count)
        break; // Prevent buffer overflow
      param.default_array[i] += param.blend_array[i % region_count];
    }

    interp_env.argStack.pop (num_value + region_count);
