{
    unsigned int regionCount = interp_env.get_region_count();
    unsigned int blendValueCount = interp_env.get_blend_value_count();

    // Ensure we have enough space for the blend array to avoid buffer overflow
    if (regionCount > param.blend_array_size() || blendValueCount > param.default_array_size()) {
        // Handle error: insufficient space in blend or default arrays
        interp_env.set_error(CFF2_interp_error::BUFFER_OVERFLOW);
        return;
    }

    // Transfer blend values from argument stack to blend array
    for (unsigned int i = 0; i < blendValueCount; ++i) {
        for (unsigned int j = 0; j < regionCount; ++j) {
            param.set_blend_value(i, j, interp_env.pop_arg());
        }
    }

    // Apply blending modifications to default values
    for (unsigned int i = 0; i < blendValueCount; ++i) {
        float defaultValue = param.get_default_value(i);
        float blendedValue = defaultValue; // Start with default value

        for (unsigned int j = 0; j < regionCount; ++j) {
            blendedValue += param.get_blend_value(i, j) * interp_env.get_region_weight(j);
        }

        param.set_default_value(i, blendedValue);
    }

    // Clear the argument stack after processing
    interp_env.clear_args();
}
