unsigned int num_regions = interp_env.argStack.pop_uint ();
    float blend_value = interp_env.argStack.pop_fixed ();

    /* Blend num_regions values from the stack */
    if (unlikely (!interp_env.blendStack.push (num_regions)))
      return;

    float *blend = &interp_env.blendStack[interp_env.blendStack.len - num_regions];
    for (unsigned int i = 0; i < num_regions; i++)
      blend[i] = interp_env.argStack.pop_fixed ();

    interp_env.argStack.push_fixed (blend_value);

    /* Pop off the original values and push back blended ones */
    float default_value = 1.f;
    for (unsigned int i = 0; i < num_regions; i++)
    {
      float value = interp_env.argStack.pop_fixed ();
      blend[i] = default_value * (1 - blend[i]) + value * blend[i];
      interp_env.argStack.push_fixed (blend[i]);
    }

    interp_env.blendStack.pop (num_regions);
