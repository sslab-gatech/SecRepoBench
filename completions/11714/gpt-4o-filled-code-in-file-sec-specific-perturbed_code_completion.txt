unsigned int numBlends = interp_env.argStack.pop_uint();
    unsigned int numValues = interp_env.argStack.len - numBlends * numBlends;
    if (unlikely(numValues > interp_env.argStack.len))
    {
      interp_env.set_error();
      return;
    }

    interp_env.process_blend();

    for (unsigned int i = 0; i < numBlends; i++)
    {
      BlendArg &arg = interp_env.argStack[numValues + i];
      arg.set_blends(numValues, i, numBlends, &interp_env.argStack[numValues]);
    }

    interp_env.argStack.resize(numValues);
