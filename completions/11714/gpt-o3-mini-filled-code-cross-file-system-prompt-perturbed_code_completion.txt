{
  // The last value on the argument stack is the blend count.
  int blendCount = static_cast<int>(interp_env.argStack.back());
  interp_env.argStack.pop_back();

  // Compute the number of default values remaining.
  size_t defaultCount = interp_env.argStack.size() - blendCount;

  // For each blend region, add the corresponding blend delta.
  for (int i = 0; i < blendCount; ++i)
    interp_env.argStack[defaultCount - blendCount + i] += interp_env.argStack[defaultCount + i];

  // Remove the blend delta values from the argument stack,
  // leaving only the blended default values.
  interp_env.argStack.resize(defaultCount);
}
