unsigned int n_regions = interp_env.get_region_count ();
unsigned int n = interp_env.argStack.pop_uint ();
if (n_regions == 0 || n == 0)
{
  interp_env.clear_args ();
  return;
}

hb_array_t<const blend_arg_t> blend_args = interp_env.argStack.pop_blend_args (n_regions, n);
if (!blend_args)
{
  interp_env.set_error ();
  return;
}

for (unsigned int i = 0; i < n; i++)
{
  const blend_arg_t &arg = blend_args[i];
  interp_env.argStack[i] = arg.get_blend_value (interp_env);
}

interp_env.argStack.resize (n);
