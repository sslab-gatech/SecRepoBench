unsigned int n_regions = interp_env.get_region_count();
unsigned int n = interp_env.argStack.pop_uint();
unsigned int default_idx = interp_env.argStack.get_count() - n - 1;

for (unsigned int i = 0; i < n; i++)
{
  for (unsigned int j = 0; j < n_regions; j++)
  {
    interp_env.blendArray[i][j] = interp_env.argStack.pop_num();
  }
  interp_env.defaultArray[default_idx + i] += interp_env.blendArray[i].get_blend();
}

interp_env.argStack.pop(n);
