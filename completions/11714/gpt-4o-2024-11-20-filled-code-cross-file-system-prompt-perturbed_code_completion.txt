// Retrieve the number of blend regions
    unsigned int num_regions = interp_env.get_num_regions();

    // Retrieve the default value from the argument stack
    float default_value = interp_env.argStack.pop();

    // Prepare a blend array to store the blend values
    std::vector<float> blend_values(num_regions, default_value);

    // Populate the blend array with values from the argument stack
    for (unsigned int i = 0; i < num_regions; ++i)
    {
        blend_values[i] = interp_env.argStack.pop();
    }

    // Apply the blend values to the parameter
    param.apply_blend(default_value, blend_values);

    // Clear the argument stack after processing
    interp_env.clear_args();
