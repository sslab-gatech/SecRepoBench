// Retrieve the number of blend regions from the argument stack
    int num_regions = interp_env.get_num_regions();
    if (num_regions <= 0 || num_regions > MAX_BLEND_REGIONS) {
        // Handle invalid number of regions
        return;
    }

    // Retrieve the default value from the argument stack
    float default_value = interp_env.pop_arg();

    // Allocate a buffer for blend values
    std::vector<float> blend_values(num_regions, 0.0f);

    // Safely populate the blend values from the argument stack
    for (int i = 0; i < num_regions; ++i) {
        if (interp_env.has_args()) {
            blend_values[i] = interp_env.pop_arg();
        } else {
            // Handle missing blend values
            blend_values[i] = 0.0f; // Default to 0 if not enough arguments
        }
    }

    // Apply the blend values to the parameter
    param.apply_blend(default_value, blend_values);

    // Clear any remaining arguments from the stack
    interp_env.clear_args();
