unsigned int num_regions = interp_env.argStack.pop_uint ();
    float default_val = interp_env.argStack.pop_float ();
    float blend_vals[16]; // Assuming a maximum of 16 regions for blending
    for (unsigned int i = 0; i < num_regions; i++)
    {
      blend_vals[i] = interp_env.argStack.pop_float ();
    }
    // Apply blend modifications to default values
    for (unsigned int i = 0; i < num_regions; i++)
    {
      // Assuming a simple blend application for demonstration
      param.blend_array[i] = default_val * blend_vals[i];
    }
    // Clear the argument stack of used blend values
    interp_env.clear_args ();
