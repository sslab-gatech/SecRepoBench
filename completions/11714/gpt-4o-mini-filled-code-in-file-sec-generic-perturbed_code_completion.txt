unsigned int count = interp_env.get_region_count();
    if (count == 0) return;

    hb_vector_t<BlendArg> blendArgs;
    blendArgs.resize(count);

    for (unsigned int i = 0; i < count; i++) {
        blendArgs[i] = interp_env.pop_arg();
    }

    for (unsigned int i = 0; i < count; i++) {
        if (blendArgs[i].blending()) {
            // Apply blending logic
            double blendValue = blendArgs[i].to_real();
            for (unsigned int j = 0; j < blendArgs[i].deltas.len; j++) {
                blendValue += blendArgs[i].deltas[j].to_real();
            }
            blendArgs[i].set_real(blendValue);
        }
    }

    // Push the modified blend arguments back onto the argument stack
    for (unsigned int i = 0; i < count; i++) {
        interp_env.argStack.push(blendArgs[i]);
    }
