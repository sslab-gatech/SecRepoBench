interp_env.process_blend ();

    unsigned int numBlends = interp_env.pop_arg ().to_uint ();
    unsigned int numValues = interp_env.pop_arg ().to_uint ();
    unsigned int valueIndex = interp_env.argStack.len - numValues;

    if (unlikely (numBlends > interp_env.get_region_count ()))
    {
      interp_env.set_error ();
      return;
    }

    for (unsigned int i = 0; i < numValues; i++)
    {
      BlendArg &arg = interp_env.argStack[valueIndex + i];
      arg.set_blends (numValues, i, numBlends, &interp_env.argStack[interp_env.argStack.len - numBlends]);
    }

    interp_env.argStack.resize (valueIndex + numValues);
