// Retrieve the count of regions and a numeric value from the argument stack.
    unsigned int region_count = env.argStack.pop_uint32();
    unsigned int numeric_value = env.argStack.pop_uint32();

    // Transfer blend values from the argument stack to a blend array associated
    // with default values, based on the retrieved counts.
    std::vector<float> blend_values(region_count, numeric_value);

    // Remove the blend values from the argument stack, leaving only the default
    // values with their blend modifications applied.
    for (unsigned int i = 0; i < region_count; ++i) {
        env.argStack.pop();
    }

    // Apply the blend modifications to the default values.
    for (auto &value : blend_values) {
        value *= param.get_blend_factor();
    }

    // Update the interp_env with the modified blend values.
    interp_env.set_blend_values(blend_values);
