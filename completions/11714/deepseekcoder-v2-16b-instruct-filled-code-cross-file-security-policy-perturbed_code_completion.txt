    int value = arg_stack.pop();

    // Initialize the blend array with default values.
    std::vector<int> blend_array(count, value);

    for (int i = 0; i < count; ++i) {
        blend_array[i] = arg_stack.pop();
    }

    // Apply blend modifications to the interp_env and param.
    for (int i = 0; i < count; ++i) {
        interp_env.blend_values[i] = blend_array[i];
        param.blend_values[i] = blend_array[i];
    }

    // Remove the blend values from the argument stack.
    for (int i = 0; i < count; ++i) {
