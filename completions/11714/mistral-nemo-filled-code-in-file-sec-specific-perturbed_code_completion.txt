static inline void process_blend(CFF2CSInterpEnv &interp_env, PARAM& param)
{
  unsigned int region_count = interp_env.get_region_count();
  BlendArg& arg = interp_env.pop_arg();
  if (arg.blending() && arg.deltas.len == region_count)
  {
    for (unsigned int i = 0; i < region_count; i++)
    {
      arg.deltas[i].set_real(arg.deltas[i].to_real() * param.scalars[i]);
    }
    arg.deltas.resize(0);
  }
  interp_env.argStack.push(arg);
}
