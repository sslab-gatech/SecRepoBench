unsigned int num_regions = interp_env.argStack.pop_uint ();
    float blend_value = interp_env.argStack.pop_fixed ();

    /* Blend num_regions values. */
    unsigned int i;
    for (i = 0; i < num_regions && i < param.blend.get_size (); i++)
    {
      param.blend[i] = interp_env.argStack.pop_fixed () * blend_value + param.blend[i] * (1.f - blend_value);
    }
    /* Clear remaining args from stack. */
    interp_env.clear_args ();
