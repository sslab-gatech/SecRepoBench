int num_regions;
float blend_values[MAX_BLEND_REGIONS];

// Retrieve the count of regions and a numeric value from the argument stack
interp_env.PopInteger(num_regions);
interp_env.PopReal(param.numeric_value);

// Transfer blend values from the argument stack to a blend array
for (int i = 0; i < num_regions; i++) {
    float blend_value;
    interp_env.PopReal(blend_value);
    blend_values[i] = blend_value;
}

// Apply the blend values to the default values
for (int i = 0; i < num_regions; i++) {
    param.default_values[i] = param.default_values[i] * (1.0f - blend_values[i]);
}

// Remove the blend values from the argument stack, leaving only the default values
for (int i = 0; i < num_regions; i++) {
    interp_env.PopReal();
}
