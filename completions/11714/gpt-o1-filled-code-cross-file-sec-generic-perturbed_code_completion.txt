// Ensure we have at least two arguments: one for the region count and one for the base value
    if (interp_env.argStack.size() < 2)
      return;

    // Retrieve the region count
    int regionCount = (int)interp_env.argStack.back();
    interp_env.argStack.pop_back();

    // Retrieve the default/base value
    double baseValue = interp_env.argStack.back();
    interp_env.argStack.pop_back();

    // Check if we have enough blend values on the stack
    if (regionCount < 0 || interp_env.argStack.size() < (size_t)regionCount)
      return;

    // Transfer blend values from the stack
    std::vector<double> blendValues(regionCount);
    for (int i = 0; i < regionCount; i++)
    {
      blendValues[i] = interp_env.argStack.back();
      interp_env.argStack.pop_back();
    }

    // Apply the blend values to the base value
    for (int i = 0; i < regionCount; i++)
      baseValue += blendValues[i];

    // Leave only the blended result on the stack
    interp_env.argStack.push_back(baseValue);
