// Retrieve the number of regions and the numeric value from the argument stack
    int numRegions = interp_env.getNumRegions();
    float defaultValue = interp_env.popArg();

    // Create a vector to store blend values
    std::vector<float> blendValues(numRegions, defaultValue);

    // Transfer blend values from the argument stack to the blend array
    for (int i = 0; i < numRegions; ++i) {
        blendValues[i] = interp_env.popArg();
    }

    // Apply blend modifications to the default value
    for (int i = 0; i < numRegions; ++i) {
        defaultValue += blendValues[i] * param.getRegionWeight(i);
    }

    // Push the modified default value back onto the argument stack
    interp_env.pushArg(defaultValue);
