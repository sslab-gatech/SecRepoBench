int num_regions = interp_env.stack.back();
    interp_env.stack.pop_back();
    int blend_val = interp_env.stack.back();
    interp_env.stack.pop_back();

    if (num_regions < 0 || num_regions > param.max_regions) {
        // Handle error: Invalid number of regions.  Could throw exception or return error code.
        return;
    }

    std::vector<int> blend_array(param.max_regions, param.default_blend_value);
    for (int i = 0; i < num_regions; ++i) {
        if (!interp_env.stack.empty()) {
            blend_array[i] = interp_env.stack.back() + blend_val; // Apply blend value
            interp_env.stack.pop_back();
        } else {
            //Handle error: Stack underflow. Could throw exception or return error code.
            return;
        }
    }

    //The blend_array now contains the modified blend values.  Use it as needed.
    //Example:  Store it in interp_env for later use.
    interp_env.blend_values = blend_array;
