// Retrieve the number of blend regions from the argument stack
    int numRegions = interp_env.argStack.pop();
    
    // Retrieve the default value from the argument stack
    float defaultValue = interp_env.argStack.pop();
    
    // Prepare a blend array to store the blend values
    std::vector<float> blendValues(numRegions, defaultValue);
    
    // Transfer blend values from the argument stack to the blend array
    for (int i = 0; i < numRegions; ++i) {
      blendValues[i] = interp_env.argStack.pop();
    }
    
    // Apply blend modifications to the default value
    for (int i = 0; i < numRegions; ++i) {
      blendValues[i] += defaultValue;
    }
    
    // Clear the argument stack and push the modified default values back
    interp_env.clear_args();
    for (int i = 0; i < numRegions; ++i) {
      interp_env.argStack.push(blendValues[i]);
    }
