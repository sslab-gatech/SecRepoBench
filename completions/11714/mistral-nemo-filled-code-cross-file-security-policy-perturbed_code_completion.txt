    uint32_t region_count = env.argStack.pop();
    float numeric_value = env.argStack.pop();

    if (region_count > param.blend_array.size()) {
      throw std::runtime_error("Not enough space in blend array to store blend values.");
    }

    // Transfer blend values from the argument stack to the blend array.
    for (uint32_t i = 0; i < region_count; ++i) {
      param.blend_array.push_back(env.argStack.pop());
    }

    // Remove the blend values from the argument stack.
