int num_regions;
float blend_values[MAX_REGIONS];

// Retrieve the count of regions and a numeric value from the argument stack
num_regions = static_cast<int>(interp_env.PopReal());
float default_value = interp_env.PopReal();

// Transfer blend values from the argument stack to a blend array associated
// with default values, based on the retrieved counts
for (int i = 0; i < num_regions; i++) {
  blend_values[i] = interp_env.PopReal();
}

// Remove the blend values from the argument stack, leaving only the default
// values with their blend modifications applied
interp_env.StackCleanup(num_regions + 1);

// Apply the blend values to the default value
for (int i = 0; i < num_regions; i++) {
  default_value = blend_values[i] * default_value;
}

// Store the final blended value back into the parameter context
param = default_value;
