unsigned int num_regions = interp_env.argStack.pop_uint ();
    float blend_value = interp_env.argStack.pop_fixed ();

    /* save default operands before blend */
    interp_env.push_defaults ();

    unsigned int i = 0;
    while (i < num_regions && !interp_env.argStack.is_empty ())
    {
      float blend = interp_env.argStack.pop_fixed ();
      interp_env.blend (blend_value, blend);
      ++i;
    }

    /* pop remaining blend values */
    while (i < num_regions)
    {
      interp_env.argStack.pop ();
      ++i;
    }
    interp_env.pop_defaults ();
