unsigned int region_count = interp_env.get_region_count();
unsigned int num_blends = interp_env.argStack.peek_uint();
if (unlikely (num_blends!= region_count))
{
  interp_env.set_error ();
  return;
}
BlendArg default_value = interp_env.pop_arg ();
hb_vector_t<BlendArg> blends;
blends.resize (num_blends);
for (unsigned int i = 0; i < num_blends; i++)
  blends[i] = interp_env.pop_arg ();
default_value.set_blends (region_count, 0, num_blends, blends.array);
interp_env.push_arg (default_value);
