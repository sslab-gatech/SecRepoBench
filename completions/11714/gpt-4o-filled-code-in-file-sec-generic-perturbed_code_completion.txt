interp_env.process_blend();

    unsigned int numBlends = interp_env.pop_arg().to_uint();
    unsigned int numValues = interp_env.argStack.len - numBlends * numValues;

    if (unlikely(numValues > interp_env.argStack.len))
    {
      interp_env.set_error();
      return;
    }

    for (unsigned int i = 0; i < numValues; i++)
    {
      BlendArg &arg = interp_env.argStack[i];
      arg.set_blends(numValues, i, numBlends, &interp_env.argStack[numValues + i * numBlends]);
    }

    interp_env.argStack.resize(numValues);
