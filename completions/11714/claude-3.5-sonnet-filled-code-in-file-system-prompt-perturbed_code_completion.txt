unsigned int n = interp_env.get_region_count ();
unsigned int k = interp_env.argStack.pop_uint ();
if (unlikely (n == 0 || k == 0 || interp_env.argStack.get_count () < n * k + k))
{
  interp_env.set_error ();
  return;
}
interp_env.process_blend ();
for (unsigned int i = 0; i < k; i++)
{
  const BlendArg &arg = interp_env.argStack[interp_env.argStack.get_count () - k + i];
  BlendArg &blendArg = interp_env.argStack[interp_env.argStack.get_count () - (n + 1) * k + i];
  blendArg.set_blends (k, i, n, &arg);
}
interp_env.argStack.pop (n * k);
