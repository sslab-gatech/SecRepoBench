// Retrieve the count of regions and a numeric value from the argument stack.
int region_count = interp_env.get_arg_count();
int blend_value = interp_env.pop_arg();

// Transfer blend values from the argument stack to a blend array associated
// with default values, based on the retrieved counts.
if (region_count > 0 && region_count <= interp_env.get_stack_size()) {
  for (int i = 0; i < region_count; i++) {
    interp_env.push_arg(interp_env.get_default_blend_value() + blend_value);
  }
} else {
  // Handle invalid region count or stack size
  interp_env.push_arg(0); // or some other default value
}

// Remove the blend values from the argument stack, leaving only the default
// values with their blend modifications applied.
interp_env.pop_args(region_count);
