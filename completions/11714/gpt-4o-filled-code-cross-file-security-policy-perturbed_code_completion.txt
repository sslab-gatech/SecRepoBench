// Retrieve the number of blend regions from the argument stack
    int numRegions = interp_env.argStack.pop();
    
    // Ensure there are enough elements on the stack for the blend operation
    if (interp_env.argStack.size() < numRegions + 1) {
        // Handle error: not enough arguments on the stack
        return;
    }

    // Retrieve the default value from the argument stack
    float defaultValue = interp_env.argStack.pop();

    // Prepare an array to hold the blend values
    std::vector<float> blendValues(numRegions);

    // Retrieve blend values from the argument stack
    for (int i = 0; i < numRegions; ++i) {
        blendValues[i] = interp_env.argStack.pop();
    }

    // Apply the blend operation using the default value and blend values
    interp_env.applyBlend(defaultValue, blendValues);

    // Clear the argument stack after processing
    interp_env.clear_args();
