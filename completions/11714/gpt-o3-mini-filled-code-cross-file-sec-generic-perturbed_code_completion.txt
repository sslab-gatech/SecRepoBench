if (interp_env.argStack.size() < 2)
    return;  // Insufficient arguments

// Retrieve the region count and the default value from the argument stack.
// Assumes that the top of the stack is the region count and the one below is the default value.
unsigned int regionCount = static_cast<unsigned int>(interp_env.argStack.back());
interp_env.argStack.pop_back();

double defaultValue = interp_env.argStack.back();
interp_env.argStack.pop_back();

// Make sure there are enough blend values on the stack.
if (interp_env.argStack.size() < regionCount)
    return;  // Insufficient blend values

// Transfer the blend values from the argument stack into param.blend.
// (Assumes that PARAM has an STL container named "blend".)
param.blend.clear();
param.blend.reserve(regionCount);
for (unsigned int i = 0; i < regionCount; ++i)
{
    // The blend values are assumed to be stored in order at the end of the argument stack.
    param.blend.push_back(interp_env.argStack[interp_env.argStack.size() - regionCount + i]);
}

// Remove the blend values from the argument stack.
interp_env.argStack.erase(interp_env.argStack.end() - regionCount, interp_env.argStack.end());

// Update the default value by applying the blend modifications.
// Here we simply add all the blend values to the default; adjust the blending logic as needed.
for (unsigned int i = 0; i < regionCount; ++i)
{
    defaultValue += param.blend[i];
}

// Push the modified default value back onto the argument stack.
interp_env.argStack.push_back(defaultValue);
