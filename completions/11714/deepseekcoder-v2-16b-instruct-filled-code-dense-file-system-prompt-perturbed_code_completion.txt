// <MASK>
    process_blend(interp_env, param);
    // After processing blend, remove the blend values from the argument stack
    for (unsigned int i = 0; i < interp_env.argStack.get_count();) {
        const BlendArg &arg = interp_env.argStack[i];
        if (arg.blending()) {
            assert((arg.numValues > 0) && (interp_env.argStack.get_count() >= arg.numValues));
            interp_env.argStack.remove(i, arg.numValues);
        } else {
            i++;
        }
    }
