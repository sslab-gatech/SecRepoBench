static inline void process_blend(CFF2CSInterpEnv &interp_env, PARAM& param)
{
    unsigned int region_count = interp_env.get_region_count();
    BlendArg& arg = interp_env.argStack.pop();

    if (region_count > 0 && arg.blending())
    {
        unsigned int num_blends = arg.deltas.len;
        if (num_blends == region_count)
        {
            arg.set_blends(1, 0, num_blends, arg.deltas.arrayZ);
            arg.deltas.resize(0);
        }
        else
        {
            interp_env.set_error();
        }
    }
}
