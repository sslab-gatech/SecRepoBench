// Retrieve the number of blend regions and the default value from the argument stack
    int numRegions = interp_env.argStack.pop();
    float defaultValue = interp_env.argStack.pop();

    // Ensure the blend array has enough space to store the blend values
    if (numRegions > interp_env.blendArray.size()) {
        interp_env.blendArray.resize(numRegions);
    }

    // Transfer blend values from the argument stack to the blend array
    for (int i = 0; i < numRegions; ++i) {
        interp_env.blendArray[i] = interp_env.argStack.pop();
    }

    // Apply blend modifications to the default value
    for (int i = 0; i < numRegions; ++i) {
        interp_env.blendArray[i] += defaultValue;
    }

    // Clear the argument stack, leaving only the modified default values
    interp_env.clear_args();
