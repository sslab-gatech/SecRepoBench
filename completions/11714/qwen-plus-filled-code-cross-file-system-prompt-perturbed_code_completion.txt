uint16_t region_count = interp_env.get_region_count();
double blend_value = interp_env.pop_arg(); // Get a numeric value from the argument stack

if (region_count > 0 && !std::isnan(blend_value)) {
    std::vector<double> blend_array(region_count, 0.0);

    for (uint16_t i = 0; i < region_count; ++i) {
        if (!interp_env.argStack.empty()) {
            blend_array[i] = interp_env.pop_arg(); // Transfer blend values to blend array
        }
    }

    double default_value = param.get_default_value();
    for (uint16_t i = 0; i < region_count; ++i) {
        default_value += blend_array[i]; // Apply blend modifications to default values
    }

    param.set_value(default_value); // Update parameter with blended value
}

interp_env.clear_args(); // Remove blend values from the argument stack
