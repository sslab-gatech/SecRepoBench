unsigned int numBlends = env.argStack.pop_uint();
    unsigned int valueIndex = env.argStack.pop_uint();
    unsigned int numValues = env.argStack.pop_uint();

    hb_vector_t<BlendArg> blends;
    blends.resize(numBlends);
    for (unsigned int i = 0; i < numBlends; i++) {
        blends[i] = env.argStack.pop();
    }

    BlendArg defaultArg;
    defaultArg.set_blends(numValues, valueIndex, numBlends, blends.begin());
    env.argStack.push(defaultArg);
