unsigned int n = interp_env.argStack.pop_uint ();
unsigned int numValues = interp_env.argStack.get_count () - n;
if (numValues < 1)
  return;
  
unsigned int blendVals = numValues * n;
if (blendVals > interp_env.argStack.get_count () - 1)
  return;

/* copy the blend values into blend array of the default values */
unsigned int iDefaults = interp_env.argStack.get_count () - numValues - blendVals;
for (unsigned int i = 0; i < numValues; i++)
{
  const unsigned int iValue = iDefaults + i;
  const unsigned int iBlend = iValue + numValues;
  interp_env.argStack.set_blends (iValue, n);
  for (unsigned int j = 0; j < n; j++)
  {
    interp_env.argStack[iValue].blendArg (j) = interp_env.argStack[iBlend + j * numValues];
  }
}

/* pop off blend values leaving defaults with blend values attached */
interp_env.argStack.pop (blendVals);
