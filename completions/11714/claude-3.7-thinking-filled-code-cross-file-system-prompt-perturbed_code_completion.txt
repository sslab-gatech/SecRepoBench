unsigned int n_regions = interp_env.get_region_count ();
unsigned int k = interp_env.argStack.pop_uint ();
if (unlikely (k < 1))
  return;
if (unlikely (!interp_env.argStack.get_count () || interp_env.argStack.get_count () < k * n_regions + 1))
  return;

const hb_codepoint_t blends_base = interp_env.argStack.get_count () - (k * n_regions + 1);
interp_env.blendVF (blends_base, k, n_regions);
