{
    unsigned int region_count = interp_env.get_region_count();
    if (region_count == 0)
    {
        // No blending needed, simply pop the default value.
        param.set_blend_value(interp_env.pop_arg());
        return;
    }

    // Prepare an array to hold blend values.
    std::vector<double> blend_values(region_count);
    for (unsigned int i = 0; i < region_count; ++i)
    {
        if (interp_env.argStack_empty())
        {
            // If the stack is unexpectedly empty, handle the error or fallback.
            param.set_invalid();
            return;
        }
        blend_values[i] = interp_env.pop_arg();
    }

    // Retrieve the default value from the argument stack.
    double default_value = interp_env.pop_arg();

    // Perform the blending operation.
    double blended_value = default_value;
    for (unsigned int i = 0; i < region_count; ++i)
    {
        blended_value += blend_values[i] * interp_env.get_region_weight(i);
    }

    // Set the final blended value into the parameter.
    param.set_blend_value(blended_value);

    // Clear any remaining blend-related arguments from the stack.
    interp_env.clear_args();
}
