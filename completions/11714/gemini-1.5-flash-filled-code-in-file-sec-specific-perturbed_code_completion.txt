unsigned int num_blends = interp_env.argStack.pop_uint ();
    unsigned int valueIndex = interp_env.argStack.pop_uint ();
    unsigned int numValues = interp_env.argStack.pop_uint ();
    interp_env.process_blend ();
    if (num_blends > 0)
    {
      hb_vector_t<BlendArg> blends(num_blends);
      for (int i = num_blends - 1; i >= 0; --i)
      {
	blends[i] = interp_env.pop_arg ();
      }
      BlendArg& arg = interp_env.argStack.peek ();
      arg.set_blends (numValues, valueIndex, num_blends, &blends[0]);
    }
