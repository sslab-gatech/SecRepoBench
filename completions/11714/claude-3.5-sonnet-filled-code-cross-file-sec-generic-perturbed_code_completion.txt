unsigned int n_regions = interp_env.argStack.pop_uint ();
unsigned int n = interp_env.argStack.pop_uint ();
if (n_regions == 0 || n == 0 || (n_regions + 1) * n > interp_env.argStack.get_count ())
  return;

hb_array_t<const BlendArg> blend_args = interp_env.argStack.tail<BlendArg> ((n_regions + 1) * n);
hb_array_t<const BlendArg> default_args = blend_args.sub_array (0, n);
hb_array_t<const BlendArg> deltas = blend_args.sub_array (n);

for (unsigned int i = 0; i < n; i++)
{
  BlendArg arg = default_args[i];
  for (unsigned int j = 0; j < n_regions; j++)
    arg += deltas[i * n_regions + j] * interp_env.normalized_coords[j];
  interp_env.argStack.push (arg);
}

interp_env.argStack.pop ((n_regions + 1) * n);
