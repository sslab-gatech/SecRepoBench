if (interp_env.argStack.size() < 2)
  return;

int regionCount = static_cast<int>(interp_env.argStack.back());
interp_env.argStack.pop_back();

float defaultVal = interp_env.argStack.back();
interp_env.argStack.pop_back();

// Make sure regionCount is valid and there are enough values on the stack
if (regionCount < 0 || static_cast<size_t>(regionCount) > interp_env.argStack.size())
  return;

std::vector<float> blendValues(regionCount);
for (int i = regionCount - 1; i >= 0; i--)
{
  blendValues[i] = interp_env.argStack.back();
  interp_env.argStack.pop_back();
}

// Example blending: accumulate all blendValues into defaultVal
for (int i = 0; i < regionCount; i++)
{
  defaultVal += blendValues[i];
}

// Push the blended value (or any final computed value)
interp_env.argStack.push_back(defaultVal);
