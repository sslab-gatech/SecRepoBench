if (object.type == MSGPACK_OBJECT_STR) {
        result->type = FLB_RA_STRING;
        result->val.string = flb_sds_create_len((char *) object.via.str.ptr,
                                                    object.via.str.size);
        if (!result->val.string) {
            return -1;
        }
        return 0;
    }
    else if (object.type == MSGPACK_OBJECT_MAP) {
        /*
         * if the key exists, we just want to know if it exists, so
         * we return a boolean true.
         */
        result->type = FLB_RA_BOOL;
        result->val.boolean = FLB_TRUE;
        return 0;
    }
    else if (object.type == MSGPACK_OBJECT_NIL) {
        result->type = FLB_RA_NULL;
        return 0;
    }
