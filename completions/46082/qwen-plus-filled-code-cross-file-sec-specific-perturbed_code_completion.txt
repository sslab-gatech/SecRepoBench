else if (object.type == MSGPACK_OBJECT_STR) {
        result->type = FLB_RA_STRING;
        // Allocate memory for the string and copy the content
        result->val.str.ptr = flb_malloc(object.via.str.size + 1); // +1 for null terminator
        if (!result->val.str.ptr) {
            return -1; // Memory allocation failed
        }
        memcpy(result->val.str.ptr, object.via.str.ptr, object.via.str.size);
        result->val.str.ptr[object.via.str.size] = '\0'; // Null-terminate the string
        result->val.str.size = object.via.str.size;
        return 0;
    }
    else if (object.type == MSGPACK_OBJECT_MAP) {
        result->type = FLB_RA_BOOL;
        result->val.boolean = true; // Indicate presence of the key
        return 0;
    }
    else if (object.type == MSGPACK_OBJECT_NIL) {
        result->type = FLB_RA_NULL;
        return 0;
    }
