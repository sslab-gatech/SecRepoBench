// Ensure the attribute name is not NULL
	if (attribute == NULL) {
		return NULL;
	}

	// Set the attribute name
	attr->name = persistent ? zend_string_copy(attribute) : zend_string_dup(attribute, persistent);

	// Convert the attribute name to lowercase and store it
	attr->lcname = zend_string_tolower(attr->name);

	// Assign the offset and argument count
	attr->offset = offset;
	attr->argc = argc;

	// Initialize the arguments array if argc > 0
	if (argc > 0) {
		attr->args = pemalloc(sizeof(zval) * argc, persistent);
	} else {
		attr->args = NULL;
	}

	// Insert the new attribute into the attributes hash table
	zend_hash_next_index_insert_ptr(*attributes, attr);

	// Return the pointer to the newly created attribute
	return attr;
