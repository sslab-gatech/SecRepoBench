if (attribute) {
		attr->lcname = zend_string_dup(attribute);
	} else {
		attr->lcname = NULL;
	}
	attr->offset = offset;
	attr->argc = argc;

	zend_hash_add(*attributes, ZSTR_VAL(attr->lcname), attr->lcname, sizeof(zend_attribute), (void *)attr, sizeof(zend_attribute), persistent);

	return attr;
}
