		attr->name = zend_string_copy(attribute);
		attr->lcname = zend_string_tolower(attr->name);
	} else {
		attr->name = zend_string_dup(attribute, false);
		attr->lcname = zend_string_tolower(attr->name);
		zend_string_release(attr->name);
		attr->name = attribute;
	}
	attr->offset = offset;
	attr->argc = argc;

	zend_hash_add_ptr(*attributes, attr->lcname, attr);

	return attr;
