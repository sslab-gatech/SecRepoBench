attr->lcname = zend_string_tolower(zend_string_copy(attribute));
attr->offset = offset;
attr->argc = argc;

if (get_attribute_str(*attributes, ZSTR_VAL(attr->lcname), ZSTR_LEN(attr->lcname), offset)) {
	pefree(attr, persistent);
} else {
	zend_hash_next_index_insert_new(*attributes, attr);
}

return attr;
