attr->name = persistent ? zend_string_dup(attribute, 1) : zend_string_copy(attribute);
attr->lcname = zend_string_tolower_ex(attribute, persistent);
attr->offset = offset;
attr->argc = argc;

// Initialize arguments to undefined
for (uint32_t i = 0; i < argc; i++) {
    ZVAL_UNDEF(&attr->argv[i]);
}

zend_hash_next_index_insert_ptr(*attributes, attr);
return attr;
