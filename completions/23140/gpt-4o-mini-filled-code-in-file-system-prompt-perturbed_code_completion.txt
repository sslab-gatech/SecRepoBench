attr->name = zend_string_copy(attribute);
	attr->lcname = zend_string_tolower(attribute);
	attr->offset = offset;
	attr->argc = argc;
	attr->argv = (zval *)pemalloc(sizeof(zval) * argc, persistent);

	for (uint32_t i = 0; i < argc; i++) {
		ZVAL_UNDEF(&attr->argv[i]);
	}

	zend_hash_index_update_ptr(*attributes, offset, attr);
	return attr;
