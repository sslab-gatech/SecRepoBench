attr->name = persistent ? zend_string_copy(attribute) : zend_string_dup(attribute);
	attr->lcname = zend_string_tolower(attr->name);
	attr->offset = offset;
	attr->argc = argc;
	attr->argv = (zval *) pemalloc(sizeof(zval) * argc, persistent);
	memset(attr->argv, 0, sizeof(zval) * argc);

	zend_hash_next_index_insert(*attributes, &attr, sizeof(attr), NULL);

	return attr;
