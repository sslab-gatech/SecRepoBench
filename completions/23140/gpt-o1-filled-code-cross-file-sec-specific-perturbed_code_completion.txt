if (!attribute) {
	pefree(attr, persistent);
	return NULL;
}

zend_string *lcname = zend_string_alloc(ZSTR_LEN(attribute), persistent);
zend_str_tolower_copy(ZSTR_VAL(lcname), ZSTR_VAL(attribute), ZSTR_LEN(attribute));
attr->lcname = lcname;
attr->offset = offset;
attr->argc = argc;

zend_hash_next_index_insert_ptr(*attributes, attr);

return attr;
