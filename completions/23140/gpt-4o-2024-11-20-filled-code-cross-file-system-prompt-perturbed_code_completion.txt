attr->lcname = zend_string_tolower(attribute);
	attr->name = persistent ? zend_string_copy(attribute) : zend_string_dup(attribute, 0);
	attr->offset = offset;
	attr->argc = argc;

	if (argc > 0) {
		attr->args = (zend_attribute_arg*) ((char*) attr + ZEND_ATTRIBUTE_SIZE(0));
	} else {
		attr->args = NULL;
	}

	zend_hash_next_index_insert_ptr(*attributes, attr);

	return attr;
