attr->name = persistent ? zend_string_dup(attribute) : zend_string_copy(attribute);
	attr->lcname = zend_string_tolower_ex(attr->name, 1);
	attr->offset = offset;
	attr->argc = argc;
	attr->argv = persistent ? pemalloc(sizeof(zval) * argc, persistent) : ecalloc(argc, sizeof(zval));

	zend_hash_update_ptr(attributes, attr->lcname, attr);
	return attr;
