if (persistent) {
		attr->name = zend_string_dup(attribute, 1);
		attr->lcname = zend_string_tolower_ex(attribute, 1);
	} else {
		attr->name = attribute;
		zend_string_addref(attribute);
		attr->lcname = zend_string_tolower(attribute);
	}
	attr->offset = offset;
	attr->argc = argc;
	if (argc > 0) {
		attr->argv = (zval *) ((char *) attr + ZEND_ATTRIBUTE_SIZE(0));
	} else {
		attr->argv = NULL;
	}

	return zend_hash_next_index_insert_ptr(*attributes, attr);
