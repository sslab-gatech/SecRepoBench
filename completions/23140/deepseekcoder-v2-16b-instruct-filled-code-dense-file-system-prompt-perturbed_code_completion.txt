// <MASK>
	attr->name = zend_string_dup(attribute, persistent);
	attr->lcname = zend_string_tolower(attr->name);
	attr->offset = offset;
	attr->argc = argc;

	// Initialize argv array
	attr->argv = pemalloc(sizeof(zval) * argc, persistent);
	for (uint32_t i = 0; i < argc; i++) {
		INIT_ZVAL(attr->argv[i]);
	}

	zend_hash_add(attributes, attr->lcname, attr);
	return attr;
}
