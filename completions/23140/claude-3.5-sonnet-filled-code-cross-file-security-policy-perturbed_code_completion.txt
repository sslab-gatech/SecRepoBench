attr->name = persistent ? zend_string_copy(attribute) : zend_string_dup(attribute, persistent);
attr->lcname = zend_string_tolower(attr->name);
attr->offset = offset;
attr->argc = argc;

if (attr && *attributes) {
    return zend_hash_add_ptr(*attributes, attr->lcname, attr);
} else {
    return NULL;
}
