zend_string *lcname = zend_string_tolower(attribute);
	attr->lcname = lcname;
	attr->offset = offset;
	attr->argc = argc;

	if (zend_hash_add(*attributes, lcname, attr) == NULL) {
		// Handle the error case where the attribute could not be added
		pemalloc_free(attr);
		zend_string_release(lcname);
		return NULL;
	}

	return attr;
