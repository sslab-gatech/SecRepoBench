// Calculate the total size of the fields in SF_CART_INFO_16K
size_t total_field_size = sizeof(c->version) + sizeof(c->title) + sizeof(c->artist) +
                          sizeof(c->cut_id) + sizeof(c->client_id) + sizeof(c->category) +
                          sizeof(c->classification) + sizeof(c->out_cue) + sizeof(c->start_date) +
                          sizeof(c->start_time) + sizeof(c->end_date) + sizeof(c->end_time) +
                          sizeof(c->producer_app_id) + sizeof(c->producer_app_version) +
                          sizeof(c->user_def) + sizeof(c->level_reference) +
                          ARRAY_LEN(c->post_timers) * (sizeof(c->post_timers[0].usage) + sizeof(c->post_timers[0].value)) +
                          sizeof(c->reserved) + sizeof(c->url);

// Check if the chunk size is too large to be handled
if (chunksize < total_field_size)
{
    psf_log_printf(audio_file_private, "cart : %u (too small for expected fields)\n", chunksize);
    psf_binheader_readf(audio_file_private, "j", chunksize);
    return 0;
}
