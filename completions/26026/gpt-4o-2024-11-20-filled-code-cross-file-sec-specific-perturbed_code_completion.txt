// Ensure the chunk size is not excessively large to prevent potential heap overflow.
if (chunksize > WAV_CART_MAX_CHUNK_SIZE)
{
    psf_log_printf(audio_file_private, "cart : %u (exceeds maximum allowed size of %d)\n", chunksize, WAV_CART_MAX_CHUNK_SIZE);
    psf_binheader_readf(audio_file_private, "j", chunksize); // Skip the chunk by reading and discarding its contents.
    return 0;
}
