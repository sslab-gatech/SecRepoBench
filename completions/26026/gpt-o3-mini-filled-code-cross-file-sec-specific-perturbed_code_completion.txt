if (chunksize - WAV_CART_MIN_CHUNK_SIZE > sizeof(((SF_CART_INFO_16K *)0)->tag_text))
{
    psf_log_printf(audio_file_private, "cart : %u (tag text size %u exceeds maximum allowed %u, skipping)\n",
                   chunksize,
                   chunksize - WAV_CART_MIN_CHUNK_SIZE,
                   (unsigned)sizeof(((SF_CART_INFO_16K *)0)->tag_text));
    psf_binheader_readf(audio_file_private, "j", chunksize);
    return 0;
}
