// <MASK>
if (chunksize > WAV_CART_MAX_CHUNK_SIZE)
{
    // Allocate a buffer to read the chunk contents into.
    uint8_t *chunk_buffer = malloc(chunksize);
    if (chunk_buffer == NULL)
    {
        psf_log_printf(audio_file_private, "cart : %u (malloc failed)\n", chunksize);
        return 0;
    }

    // Read the chunk contents into the buffer.
    if (psf_binheader_readf(audio_file_private, "m", chunk_buffer, chunksize)!= chunksize)
    {
        psf_log_printf(audio_file_private, "cart : %u (read failed)\n", chunksize);
        free(chunk_buffer);
        return 0;
    }

    // Discard the chunk contents.
    free(chunk_buffer);

    return 0;
}
