sf_count_t current_pos = psf_fseek (audio_file_private, 0, SEEK_CUR);

if (current_pos + chunksize > audio_file_private->filelength)
{
	psf_log_printf (audio_file_private,
		"cart : chunk size (%u) goes beyond file length, skipping.\n", chunksize) ;
	psf_binheader_readf (audio_file_private, "j", chunksize) ;
	return 0 ;
}
