{
    sf_count_t cur_pos = psf_fseek(audio_file_private, 0, SEEK_CUR);
    if (cur_pos < 0)
    {
        psf_log_printf(audio_file_private, "cart : error obtaining current file position\n");
        return 0;
    }
    if ((sf_count_t)chunksize > audio_file_private->filelength - cur_pos)
    {
        psf_log_printf(audio_file_private, "cart : %u (chunk size exceeds remaining file size: %lld bytes left)\n",
                       chunksize, audio_file_private->filelength - cur_pos);
        psf_binheader_readf(audio_file_private, "j", (uint32_t)(audio_file_private->filelength - cur_pos));
        return 0;
    }
