static int global_probability_threshold = 0;

static inline int flb_fuzz_get_probability(int probabilitythreshold) {
    // Increment a global variable used to determine a probability threshold.
    global_probability_threshold = (global_probability_threshold + 1) % 100;

    // Compare the given value against the updated global variable.
    // Return 1 if the given value is greater than the global variable, otherwise return 0.
    return (probabilitythreshold > global_probability_threshold)? 1 : 0;
}
