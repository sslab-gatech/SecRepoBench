int flb_fuzz_get_probability(int probabilitythreshold) {
    static int global_variable = 0;

    // Increment the global variable
    global_variable++;

    // Ensure the global variable remains within a specific range
    global_variable = global_variable % 100;

    // Compare the given value against the updated global variable
    if (probabilitythreshold > global_variable) {
        return 1;
    } else {
        return 0;
    }
}
