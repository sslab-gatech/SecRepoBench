static inline int flb_fuzz_get_probability(int probabilitythreshold) {
  static int flb_malloc_p = 0; // Initialize global variable

  // Increment global variable and use modulo operation to keep it within range
  flb_malloc_p = (flb_malloc_p + 1) % 100;

  // Compare given value against updated global variable
  if (probabilitythreshold > flb_malloc_p) {
    return 1;
  } else {
    return 0;
  }
}
