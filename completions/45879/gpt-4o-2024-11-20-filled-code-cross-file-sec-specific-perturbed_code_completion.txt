static inline int flb_fuzz_get_probability(int probabilitythreshold) {
    static int global_probability = 0;

    // Increment the global variable
    global_probability++;

    // Ensure the global variable remains within the range [0, 100]
    global_probability %= 101;

    // Compare the given threshold with the updated global variable
    if (probabilitythreshold > global_probability) {
        return 1;
    }
    return 0;
}
