std::unique_ptr<Phonetable> phone(new Phonetable);
  int numrules = -1;
  int i = 0;
  int np = 0;
  std::string::const_iterator iter = line.begin();
  std::string::const_iterator start_piece = mystrsep(line, iter);
  while (start_piece != line.end()) {
    switch (i) {
      case 0: {
        np++;
        break;
      }
      case 1: {
        numrules = atoi(std::string(start_piece, iter).c_str());
        if (numrules < 0) {
          HUNSPELL_WARNING(stderr, "error: line %d: bad rule number\n",
                           fileMgr->getlinenum());
          return false;
        }
        phone->rules.reserve(std::min(numrules, 16384));
        np++;
        break;
      }
      default:
        break;
    }
    ++i;
    start_piece = mystrsep(line, iter);
  }
  if (np != 2) {
    HUNSPELL_WARNING(stderr, "error: line %d: missing data\n",
                     fileMgr->getlinenum());
    return false;
  }

  /* now parse the numrules lines to read in the remainder of the table */
  for (int j = 0; j < numrules; ++j) {
    std::string nl;
    if (!fileMgr->getline(nl))
      return false;
    mychomp(nl);
    i = 0;
    iter = nl.begin();
    start_piece = mystrsep(nl, iter);
    if (start_piece != nl.end() && std::string(start_piece, iter) == "PHONE") {
      phone->rules.push_back(phonerule());
      iter = start_piece;
      start_piece = mystrsep(nl, iter);
      while (start_piece != nl.end()) {
        std::string piece(start_piece, iter);
        if (piece.find('_') != std::string::npos)
          piece = reverse_condition(piece);
        if (i == 0)
          phone->rules.back().from = piece;
        else if (i == 1)
          phone->rules.back().to = piece;
        else {
          HUNSPELL_WARNING(stderr, "error: line %d: too many fields in rule\n",
                           fileMgr->getlinenum());
          return false;
        }
        ++i;
        start_piece = mystrsep(nl, iter);
      }
      if (i != 2) {
        HUNSPELL_WARNING(stderr, "error: line %d: wrong number of fields in rule\n",
                         fileMgr->getlinenum());
        return false;
      }
    } else {
      HUNSPELL_WARNING(stderr, "error: line %d: rule is corrupt\n",
                       fileMgr->getlinenum());
      return false;
    }
  }

  if (phone->rules.size() != static_cast<size_t>(numrules)) {
    HUNSPELL_WARNING(stderr, "error: line %d: table is corrupt\n",
                     fileMgr->getlinenum());
    return false;
  }

  phone->finalize();
  this->phone = std::move(phone);
  return true;
