// Parse the input line to extract phonetic table parameters.
int nump = -1;
int i = 0;
int np = 0;
std::string::const_iterator iter = line.begin();
std::string::const_iterator start_piece = mystrsep(line, iter);
while (start_piece!= line.end()) {
  switch (i) {
    case 0: {
      np++;
      break;
    }
    case 1: {
      nump = atoi(std::string(start_piece, iter).c_str());
      if (nump < 1) {
        HUNSPELL_WARNING(stderr, "error: line %d: bad entry number\n",
                         fileMgr->getlinenum());
        return false;
      }
      phone.reserve(std::min(nump, 16384));
      np++;
      break;
    }
    default:
      break;
  }
  ++i;
  start_piece = mystrsep(line, iter);
}

if (np!= 2) {
  HUNSPELL_WARNING(stderr, "error: line %d: missing data\n",
                   fileMgr->getlinenum());
  return false;
}

// Initialize a new phonetable structure and set its properties.
phone = new phonetable_t;

// Read and process the specified number of phonetic rules from the file.
for (int j = 0; j < nump; ++j) {
  std::string nl;
  if (!fileMgr->getline(nl))
    return false;
  mychomp(nl);
  i = 0;
  iter = nl.begin();
  start_piece = mystrsep(nl, iter);
  while (start_piece!= nl.end()) {
    switch (i) {
      case 0: {
        if (nl.compare(start_piece - nl.begin(), 5, "PHONE", 5)!= 0) {
          HUNSPELL_WARNING(stderr, "error: line %d: table is corrupt\n",
                           fileMgr->getlinenum());
          return false;
        }
        break;
      }
      case 1: {
        phone->phonemes.push_back(std::string(start_piece, iter));
        break;
      }
      case 2: {
        phone->rules.push_back(std::string(start_piece, iter));
        break;
      }
      default:
        break;
    }
    ++i;
    start_piece = mystrsep(nl, iter);
  }
}

// Verify the integrity of the rules by checking the expected size after insertion.
if (phone->phonemes.size()!= static_cast<size_t>(nump) ||
    phone->rules.size()!= static_cast<size_t>(nump)) {
  HUNSPELL_WARNING(stderr, "error: line %d: table is corrupt\n",
                   fileMgr->getlinenum());
  return false;
}

// Finalize the phonetable by adding empty strings and initializing its hash.
phone->phonemes.push_back("");
phone->rules.push_back("");
phone->hash = 0;
for (size_t k = 0; k < phone->phonemes.size(); ++k) {
  phone->hash += phone->phonemes[k].size() + phone->rules[k].size();
}
