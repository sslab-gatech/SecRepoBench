{
  int tokenIndex = 0;
  int numRules = -1;
  int np = 0;
  std::string::const_iterator iter = line.begin();
  std::string::const_iterator tokenStart = mystrsep(line, iter);
  while (tokenStart != line.end()) {
    switch (tokenIndex) {
      case 0:
        // First token (e.g. a tag or identifier) â€“ no validation needed here.
        np++;
        break;
      case 1: {
        numRules = atoi(std::string(tokenStart, iter).c_str());
        if (numRules < 1) {
          HUNSPELL_WARNING(stderr, "error: line %d: bad number of phonetic rules\n",
                           fileMgr->getlinenum());
          return false;
        }
        np++;
        break;
      }
      default:
        break;
    }
    tokenIndex++;
    tokenStart = mystrsep(line, iter);
  }
  if (np != 2) {
    HUNSPELL_WARNING(stderr, "error: line %d: missing phonetable parameters\n",
                     fileMgr->getlinenum());
    return false;
  }

  // Initialize the phonetable structure.
  phone = new Phonetable();
  phone->rules.reserve(std::min(numRules, 16384));

  // Read and process each of the phonetic rules.
  for (int j = 0; j < numRules; ++j) {
    std::string nl;
    if (!fileMgr->getline(nl))
      return false;
    mychomp(nl);

    std::string::const_iterator it = nl.begin();
    std::string::const_iterator token = mystrsep(nl, it);
    // The first token of the rule must be "PHONE".
    std::string keyword(token, it);
    if (keyword != "PHONE") {
      HUNSPELL_WARNING(stderr, "error: line %d: phonetic rule missing PHONE keyword\n",
                       fileMgr->getlinenum());
      return false;
    }
    // The remainder of the line is the rule detail.
    std::string ruleDetail(it, nl.end());
    // Strip underscores from the rule detail.
    ruleDetail.erase(std::remove(ruleDetail.begin(), ruleDetail.end(), '_'),
                     ruleDetail.end());
    phone->rules.push_back(ruleDetail);
  }

  if (phone->rules.size() != static_cast<size_t>(numRules)) {
    HUNSPELL_WARNING(stderr, "error: line %d: phonetic rules count mismatch\n",
                     fileMgr->getlinenum());
    return false;
  }

  // Finalize the phonetable: add an extra empty string and initialize its hash.
  phone->rules.push_back("");
  phone->hash = std::hash<std::string>()(phone->rules.back());
}
