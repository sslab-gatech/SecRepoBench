// parse in the keyboard string for suggestions
bool AffixMgr::parse_try_string(const std::string& line, FileMgr* af) {
  if (!trystring.empty()) {
    HUNSPELL_WARNING(stderr, "error: line %d: multiple definitions of a parameter\n",
                     af->getlinenum());
    return false;
  }
  std::string s;
  if (!parse_string(line, s, af->getlinenum()))
    return false;
  trystring = s;
  return true;
}
