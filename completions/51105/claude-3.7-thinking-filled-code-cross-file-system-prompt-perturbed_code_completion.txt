phone = new phonetable();
  if (!phone) {
    HUNSPELL_WARNING(stderr, "error: line %d: could not allocate phone table\n",
                     fileMgr->getlinenum());
    return false;
  }
  
  int numphonetic = -1;
  int i = 0;
  int np = 0;
  
  std::string::const_iterator iter = line.begin();
  std::string::const_iterator start_piece = mystrsep(line, iter);
  
  while (start_piece != line.end()) {
    switch (i) {
      case 0: {
        np++;
        break;
      }
      case 1: {
        numphonetic = atoi(std::string(start_piece, iter).c_str());
        if (numphonetic < 1) {
          HUNSPELL_WARNING(stderr, "error: line %d: bad entry number\n",
                           fileMgr->getlinenum());
          return false;
        }
        phone->rules.reserve(std::min(numphonetic, 16384));
        np++;
        break;
      }
      case 2: {
        phone->utf8 = atoi(std::string(start_piece, iter).c_str());
        np++;
        break;
      }
      default:
        break;
    }
    ++i;
    start_piece = mystrsep(line, iter);
  }
  
  if (np < 2) {
    HUNSPELL_WARNING(stderr, "error: line %d: missing phone information\n",
                     fileMgr->getlinenum());
    return false;
  }
  
  /* now parse the numphonetic lines to read in the remainder of the table */
  for (int j = 0; j < numphonetic; ++j) {
    std::string nl;
    if (!fileMgr->getline(nl))
      return false;
    mychomp(nl);
    i = 0;
    phoneentry rule;
    iter = nl.begin();
    start_piece = mystrsep(nl, iter);
    
    while (start_piece != nl.end()) {
      switch (i) {
        case 0: {
          if (nl.compare(start_piece - nl.begin(), 5, "PHONE", 5) != 0) {
            HUNSPELL_WARNING(stderr, "error: line %d: table is corrupt\n",
                             fileMgr->getlinenum());
            return false;
          }
          break;
        }
        case 1: {
          rule.pattern.assign(start_piece, iter);
          if (rule.pattern.find('_') != std::string::npos) {
            mystrrep(rule.pattern, "_", "");
          }
          break;
        }
        case 2: {
          rule.replacement.assign(start_piece, iter);
          if (rule.replacement.find('_') != std::string::npos) {
            mystrrep(rule.replacement, "_", "");
          }
          break;
        }
        default:
          break;
      }
      ++i;
      start_piece = mystrsep(nl, iter);
    }
    
    if (i < 3) {
      HUNSPELL_WARNING(stderr, "error: line %d: missing phonetic rule data\n",
                       fileMgr->getlinenum());
      return false;
    }
    
    phone->rules.push_back(rule);
  }
  
  if (phone->rules.size() != static_cast<size_t>(numphonetic)) {
    HUNSPELL_WARNING(stderr, "error: line %d: table is corrupt\n",
                     fileMgr->getlinenum());
    return false;
  }
  
  // Add empty entries at the end of the table
  phone->rules.push_back(phoneentry());
  phone->rules.push_back(phoneentry());
  phone->rules.push_back(phoneentry());
  
  // Initialize hash
  phone->hash.reset(new phonehash());
