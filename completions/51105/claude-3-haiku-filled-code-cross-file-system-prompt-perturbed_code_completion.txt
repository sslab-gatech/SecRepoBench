phone = new phonetable();
std::string::const_iterator iter = line.begin();
std::string::const_iterator start_piece = mystrsep(line, iter);
int np = 0;
while (start_piece != line.end()) {
  switch (np) {
    case 0: {
      phone->num = atoi(std::string(start_piece, iter).c_str());
      break;
    }
    case 1: {
      phone->max_size = atoi(std::string(start_piece, iter).c_str());
      break;
    }
    default: {
      if (np % 2 == 0) {
        std::string rule(start_piece, iter);
        if (rule.compare(0, 5, "PHONE") == 0) {
          rule = rule.substr(6);
          std::string::size_type pos = rule.find_first_of("/");
          if (pos != std::string::npos) {
            std::string cond = rule.substr(pos + 1);
            rule = rule.substr(0, pos);
            reverse_condition(cond);
            phone->rules.push_back(std::make_pair(rule, cond));
          } else {
            phone->rules.push_back(std::make_pair(rule, ""));
          }
          for (std::string::iterator it = rule.begin(); it != rule.end(); ++it) {
            if (*it == '_')
              *it = ' ';
          }
        } else {
          HUNSPELL_WARNING(stderr, "error: line %d: missing PHONE keyword\n",
                           fileMgr->getlinenum());
          delete phone;
          return false;
        }
      }
      break;
    }
  }
  ++np;
  start_piece = mystrsep(line, iter);
}

if (phone->rules.size() != static_cast<size_t>(phone->num)) {
  HUNSPELL_WARNING(stderr, "error: line %d: table is corrupt\n",
                   fileMgr->getlinenum());
  delete phone;
  return false;
}

phone->init_hash();
return true;
