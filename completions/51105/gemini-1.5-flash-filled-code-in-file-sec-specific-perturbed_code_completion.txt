phone = new phonetable;
  if (!phone) return false;
  int numphone = 0;
  int i = 0;
  int np = 0;
  std::string::const_iterator iter = line.begin();
  std::string::const_iterator start_piece = mystrsep(line, iter);
  while (start_piece != line.end()) {
    switch (i) {
      case 0: {
        np++;
        break;
      }
      case 1: {
        numphone = atoi(std::string(start_piece, iter).c_str());
        if (numphone < 1) {
          HUNSPELL_WARNING(stderr, "error: line %d: bad entry number\n",
                           fileMgr->getlinenum());
          delete phone;
          phone = NULL;
          return false;
        }
        phone->numrules = numphone;
        np++;
        break;
      }
      default:
        break;
    }
    ++i;
    start_piece = mystrsep(line, iter);
  }
  if (np != 2) {
    HUNSPELL_WARNING(stderr, "error: line %d: missing data\n",
                     fileMgr->getlinenum());
    delete phone;
    phone = NULL;
    return false;
  }

  /* now parse the numphone lines to read in the remainder of the table */
  for (int j = 0; j < numphone; ++j) {
    std::string nl;
    if (!fileMgr->getline(nl)) {
      delete phone;
      phone = NULL;
      return false;
    }
    mychomp(nl);
    i = 0;
    iter = nl.begin();
    start_piece = mystrsep(nl, iter);
    while (start_piece != nl.end()) {
      switch (i) {
        case 0: {
          if (nl.compare(start_piece - nl.begin(), 5, "PHONE", 5) != 0) {
            HUNSPELL_WARNING(stderr, "error: line %d: table is corrupt\n",
                             fileMgr->getlinenum());
            delete phone;
            phone = NULL;
            return false;
          }
          break;
        }
        case 1: {
          phone->rules[j].pattern.assign(start_piece, iter);
          break;
        }
        case 2: {
          phone->rules[j].outstring.assign(start_piece, iter);
          break;
        }
        default:
          break;
      }
      ++i;
      start_piece = mystrsep(nl, iter);
    }
    if (phone->rules[j].pattern.empty() ||
        phone->rules[j].outstring.empty()) {
      HUNSPELL_WARNING(stderr, "error: line %d: table is corrupt\n",
                       fileMgr->getlinenum());
      delete phone;
      phone = NULL;
      return false;
    }
    size_t pos = phone->rules[j].pattern.find('_');
    while (pos != std::string::npos) {
      phone->rules[j].pattern.replace(pos, 1, "");
      pos = phone->rules[j].pattern.find('_');
    }
    pos = phone->rules[j].outstring.find('_');
    while (pos != std::string::npos) {
      phone->rules[j].outstring.replace(pos, 1, "");
      pos = phone->rules[j].outstring.find('_');
    }
    phone->rules[j].patternlen = phone->rules[j].pattern.size();
    phone->rules[j].outstringlen = phone->rules[j].outstring.size();
  }
  if (phone->numrules != static_cast<int>(numphone)) {
    HUNSPELL_WARNING(stderr, "error: line %d: table is corrupt\n",
                     fileMgr->getlinenum());
    delete phone;
    phone = NULL;
    return false;
  }
  phone->rules[numphone].pattern = "";
  phone->rules[numphone].outstring = "";
  phone->rules[numphone].patternlen = 0;
  phone->rules[numphone].outstringlen = 0;
  phone->hash = new unsigned int[phone->numrules + 1];
  for (int j = 0; j < phone->numrules; ++j) {
    phone->hash[j] = hash_string(phone->rules[j].pattern.c_str());
  }
  phone->hash[numphone] = 0;
