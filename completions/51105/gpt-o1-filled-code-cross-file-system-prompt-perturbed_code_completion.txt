int i = 0;
  int np = 0;
  int num_rules = -1;

  // Parse the line to get the number of phonetic rules.
  std::string::const_iterator iter = line.begin();
  std::string::const_iterator start_piece = mystrsep(line, iter);
  while (start_piece != line.end()) {
    switch (i) {
      case 0: {
        // First piece is just a marker (e.g., "PHONETABLE"), ignore content
        np++;
        break;
      }
      case 1: {
        // Second piece should be the number of phonetic rules
        num_rules = atoi(std::string(start_piece, iter).c_str());
        if (num_rules < 1) {
          HUNSPELL_WARNING(stderr, "error: line %d: invalid phonetic table size\n",
                           fileMgr->getlinenum());
          return false;
        }
        phone = new PHonetable(); // hypothetical structure for phonetic data
        phone->rules.reserve(num_rules);
        np++;
        break;
      }
      default:
        break;
    }
    i++;
    start_piece = mystrsep(line, iter);
  }

  // Validate we got the correct number of entries
  if (np != 2) {
    HUNSPELL_WARNING(stderr, "error: line %d: missing phonetable data\n",
                     fileMgr->getlinenum());
    return false;
  }

  // Read and process the specified number of phonetic rules
  for (int j = 0; j < num_rules; ++j) {
    std::string rule_line;
    if (!fileMgr->getline(rule_line)) {
      HUNSPELL_WARNING(stderr, "error: line %d: unable to read phonetable rule\n",
                       fileMgr->getlinenum());
      return false;
    }
    mychomp(rule_line);

    // Parse the rule line pieces
    int rule_idx = 0;
    iter = rule_line.begin();
    start_piece = mystrsep(rule_line, iter);
    std::string rule_data;

    while (start_piece != rule_line.end()) {
      switch (rule_idx) {
        case 0: {
          // Each rule must start with the "PHONE" keyword
          if (rule_line.compare(start_piece - rule_line.begin(), 5, "PHONE", 5) != 0) {
            HUNSPELL_WARNING(stderr, "error: line %d: phonetable rule is corrupt\n",
                             fileMgr->getlinenum());
            return false;
          }
          break;
        }
        case 1: {
          // Extract and store the actual rule content
          rule_data.assign(start_piece, iter);
          // Strip underscores
          rule_data.erase(std::remove(rule_data.begin(), rule_data.end(), '_'), rule_data.end());
          break;
        }
        default:
          break;
      }
      rule_idx++;
      start_piece = mystrsep(rule_line, iter);
    }

    // Simple integrity check before pushing
    if (rule_data.empty()) {
      HUNSPELL_WARNING(stderr, "error: line %d: empty phonetable rule\n",
                       fileMgr->getlinenum());
      return false;
    }
    phone->rules.push_back(rule_data);
  }

  // Verify the final number of inserted rules
  if (phone->rules.size() != static_cast<size_t>(num_rules)) {
    HUNSPELL_WARNING(stderr, "error: line %d: phonetable size mismatch\n",
                     fileMgr->getlinenum());
    return false;
  }

  phone->rules.push_back(std::string());
