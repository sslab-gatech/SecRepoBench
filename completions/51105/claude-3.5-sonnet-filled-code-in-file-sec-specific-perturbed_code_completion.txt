if (phone) {
    HUNSPELL_WARNING(stderr, "error: line %d: multiple table definitions\n",
                     fileMgr->getlinenum());
    return false;
  }
  
  int numphon = -1;
  int i = 0;
  int np = 0;
  std::string::const_iterator iter = line.begin();
  std::string::const_iterator start_piece = mystrsep(line, iter);
  
  while (start_piece != line.end()) {
    switch (i) {
      case 0: {
        np++;
        break;
      }
      case 1: {
        numphon = atoi(std::string(start_piece, iter).c_str());
        if (numphon < 0) {
          HUNSPELL_WARNING(stderr, "error: line %d: bad entry number\n",
                           fileMgr->getlinenum());
          return false;
        }
        phone = (struct phonetable*) malloc(sizeof(struct phonetable));
        if (!phone) return false;
        phone->num = numphon;
        phone->rules = (char**) calloc(numphon, sizeof(char*));
        phone->utf8 = utf8;
        if (!phone->rules) {
          free(phone);
          phone = NULL;
          return false;
        }
        np++;
        break;
      }
      default:
        break;
    }
    i++;
    start_piece = mystrsep(line, iter);
  }
  
  if (np != 2) {
    HUNSPELL_WARNING(stderr, "error: line %d: missing data\n",
                     fileMgr->getlinenum());
    return false;
  }

  for (int j = 0; j < numphon; j++) {
    std::string nl;
    if (!fileMgr->getline(nl))
      return false;
    mychomp(nl);
    
    i = 0;
    iter = nl.begin();
    start_piece = mystrsep(nl, iter);
    
    while (start_piece != nl.end()) {
      switch (i) {
        case 0: {
          if (nl.compare(start_piece - nl.begin(), 5, "PHONE", 5) != 0) {
            HUNSPELL_WARNING(stderr, "error: line %d: table is corrupt\n",
                             fileMgr->getlinenum());
            return false;
          }
          break;
        }
        case 1: {
          phone->rules[j] = mystrdup(std::string(start_piece, iter).c_str());
          if (!phone->rules[j]) {
            for (int k = 0; k < j; k++)
              free(phone->rules[k]);
            free(phone->rules);
            free(phone);
            phone = NULL;
            return false;
          }
          break;
        }
        default:
          break;
      }
      i++;
      start_piece = mystrsep(nl, iter);
    }
    
    if (i != 2) {
      HUNSPELL_WARNING(stderr, "error: line %d: table is corrupt\n",
                       fileMgr->getlinenum());
      return false;
    }
  }

  phone->rules[numphon] = NULL;
  init_phonet_hash(*phone);
  return true;
