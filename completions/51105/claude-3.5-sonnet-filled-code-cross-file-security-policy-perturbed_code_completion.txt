if (!phone)
  phone = std::make_unique<phonetable>();

int num_rules = -1;
int i = 0;
int np = 0;
std::string::const_iterator iter = line.begin();
std::string::const_iterator start_piece = mystrsep(line, iter);
while (start_piece != line.end()) {
  switch (i) {
    case 0:
      np++;
      break;
    case 1:
      num_rules = atoi(std::string(start_piece, iter).c_str());
      if (num_rules < 1) {
        HUNSPELL_WARNING(stderr, "error: line %d: bad entry number\n",
                         fileMgr->getlinenum());
        return false;
      }
      phone->rules.reserve(std::min(num_rules, 16384));
      np++;
      break;
    default:
      break;
  }
  ++i;
  start_piece = mystrsep(line, iter);
}

if (np != 2) {
  HUNSPELL_WARNING(stderr, "error: line %d: missing data\n",
                   fileMgr->getlinenum());
  return false;
}

for (int j = 0; j < num_rules; ++j) {
  std::string nl;
  if (!fileMgr->getline(nl))
    return false;
  mychomp(nl);
  i = 0;
  iter = nl.begin();
  start_piece = mystrsep(nl, iter);
  std::string rule;
  while (start_piece != nl.end()) {
    switch (i) {
      case 0:
        if (nl.compare(start_piece - nl.begin(), 5, "PHONE", 5) != 0) {
          HUNSPELL_WARNING(stderr, "error: line %d: table is corrupt\n",
                           fileMgr->getlinenum());
          return false;
        }
        break;
      case 1:
        rule = std::string(start_piece, iter);
        break;
      case 2:
        rule += " " + std::string(start_piece, iter);
        break;
      default:
        break;
    }
    ++i;
    start_piece = mystrsep(nl, iter);
  }

  if (i != 3) {
    HUNSPELL_WARNING(stderr, "error: line %d: bad rule format\n",
                     fileMgr->getlinenum());
    return false;
  }

  // Strip underscores from the second part of the rule
  size_t space_pos = rule.find(' ');
  if (space_pos != std::string::npos) {
    rule.erase(std::remove(rule.begin() + space_pos + 1, rule.end(), '_'), rule.end());
  }

  phone->rules.push_back(rule);
}

if (phone->rules.size() != static_cast<size_t>(num_rules)) {
  HUNSPELL_WARNING(stderr, "error: line %d: table is corrupt\n",
                   fileMgr->getlinenum());
  return false;
}
