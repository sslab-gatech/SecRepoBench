phone = new PhoneTable();
  int i = 0;
  int np = 0;
  std::string::const_iterator iter = line.begin();
  std::string::const_iterator start_piece = mystrsep(line, iter);
  while (start_piece != line.end()) {
    switch (i) {
      case 0: {
        np++;
        break;
      }
      case 1: {
        phone->phonenum = atoi(std::string(start_piece, iter).c_str());
        if (phone->phonenum < 1) {
          HUNSPELL_WARNING(stderr, "error: line %d: bad entry number\n",
                           fileMgr->getlinenum());
          return false;
        }
        phone->phonearr.reserve(std::min(phone->phonenum, 16384));
        np++;
        break;
      }
      default:
        break;
    }
    ++i;
    start_piece = mystrsep(line, iter);
  }
  if (np != 2) {
    HUNSPELL_WARNING(stderr, "error: line %d: missing data\n",
                     fileMgr->getlinenum());
    return false;
  }

  /* now parse the phone->phonenum lines to read in the remainder of the table */
  for (int j = 0; j < phone->phonenum; ++j) {
    std::string nl;
    if (!fileMgr->getline(nl))
      return false;
    mychomp(nl);
    i = 0;
    phone->phonearr.push_back(phoneentry());
    iter = nl.begin();
    start_piece = mystrsep(nl, iter);
    while (start_piece != nl.end()) {
      switch (i) {
        case 0: {
          if (nl.compare(start_piece - nl.begin(), 5, "PHONE", 5) != 0) {
            HUNSPELL_WARNING(stderr, "error: line %d: table is corrupt\n",
                             fileMgr->getlinenum());
            return false;
          }
          break;
        }
        case 1: {
          phone->phonearr.back().key.assign(start_piece, iter);
          size_t under_pos = phone->phonearr.back().key.find('_');
          if (under_pos != std::string::npos)
            phone->phonearr.back().key.resize(under_pos);
          break;
        }
        case 2: {
          phone->phonearr.back().value.assign(start_piece, iter);
          size_t under_pos = phone->phonearr.back().value.find('_');
          if (under_pos != std::string::npos)
            phone->phonearr.back().value.resize(under_pos);
          break;
        }
        default:
          break;
      }
      ++i;
      start_piece = mystrsep(nl, iter);
    }
    if (phone->phonearr.back().key.empty() || phone->phonearr.back().value.empty()) {
        HUNSPELL_WARNING(stderr, "error: line %d: bad phone entry\n", fileMgr->getlinenum());
        return false;
    }
  }

  if (phone->phonearr.size() != static_cast<size_t>(phone->phonenum)) {
    HUNSPELL_WARNING(stderr, "error: line %d: table is corrupt\n",
                     fileMgr->getlinenum());
    return false;
  }

  phone->phonearr.push_back(phoneentry());  // Add an empty string terminator.
  phone->phonearr.back().key = "";
  phone->phonearr.back().value = "";

  phone->make_hash();
