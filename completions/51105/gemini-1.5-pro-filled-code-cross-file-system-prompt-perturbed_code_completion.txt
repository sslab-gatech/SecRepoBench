phone = new PhoneTable();
  int nump = -1;
  int i = 0;
  int np = 0;
  std::string::const_iterator iter = line.begin();
  std::string::const_iterator start_piece = mystrsep(line, iter);
  while (start_piece != line.end()) {
    switch (i) {
      case 0: {
        np++;
        break;
      }
      case 1: {
        nump = atoi(std::string(start_piece, iter).c_str());
        if (nump < 1) {
          HUNSPELL_WARNING(stderr, "error: line %d: bad entry number\n",
                           fileMgr->getlinenum());
          return false;
        }
        phone->rules.reserve(std::min(nump, 16384));
        np++;
        break;
      }
      default:
        break;
    }
    ++i;
    start_piece = mystrsep(line, iter);
  }
  if (np != 2) {
    HUNSPELL_WARNING(stderr, "error: line %d: missing data\n",
                     fileMgr->getlinenum());
    return false;
  }

  /* now parse the nump lines to read in the remainder of the table */
  for (int j = 0; j < nump; ++j) {
    std::string nl;
    if (!fileMgr->getline(nl))
      return false;
    mychomp(nl);
    i = 0;
    phone->rules.push_back(phonerule());
    iter = nl.begin();
    start_piece = mystrsep(nl, iter);
    while (start_piece != nl.end()) {
      switch (i) {
        case 0: {
          if (nl.compare(start_piece - nl.begin(), 5, "PHONE", 5) != 0) {
            HUNSPELL_WARNING(stderr, "error: line %d: table is corrupt\n",
                             fileMgr->getlinenum());
            return false;
          }
          break;
        }
        case 1: {
          phone->rules.back().letters.assign(start_piece, iter);
          std::string l(phone->rules.back().letters);
          stripUnderscores(l);
          phone->rules.back().letters = l;

          break;
        }
        case 2: {
          phone->rules.back().replacement.assign(start_piece, iter);
          std::string r(phone->rules.back().replacement);
          stripUnderscores(r);
          phone->rules.back().replacement = r;
          break;
        }
        default:
          break;
      }
      ++i;
      start_piece = mystrsep(nl, iter);
    }
    if (phone->rules.back().letters.size() != phone->rules.back().replacement.size()) {
        HUNSPELL_WARNING(stderr, "error: line %d: PHONE string length mismatch\n", fileMgr->getlinenum());
        return false;
    }
  }

  phone->rules.push_back(phonerule());  // add empty strings
  phone->rules.back().letters = "";
  phone->rules.back().replacement = "";

  phone->create_hash();
