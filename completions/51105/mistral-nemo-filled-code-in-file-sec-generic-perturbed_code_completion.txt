// parse in the keyboard string for suggestions
bool AffixMgr::parse_keystring(const std::string& line, FileMgr* af) {
  if (!keystring.empty()) {
    HUNSPELL_WARNING(stderr, "error: line %d: multiple definitions\n",
                     af->getlinenum());
    return false;
  }
  std::string s;
  if (!parse_string(line, s, af->getlinenum()))
    return false;
  keystring = s;
  return true;
}
