bool AffixMgr::parse_phonetable(const std::string& line, FileMgr* fileMgr) {
  if (phone) {
    HUNSPELL_WARNING(stderr, "error: line %d: multiple table definitions\n",
                     fileMgr->getlinenum());
    return false;
  }

  int numrules = -1;
  int i = 0;
  int np = 0;
  std::string::const_iterator iter = line.begin();
  std::string::const_iterator start_piece = mystrsep(line, iter);
  while (start_piece != line.end()) {
    switch (i) {
      case 0: {
        np++;
        break;
      }
      case 1: {
        np++;
        numrules = atoi(std::string(start_piece, iter).c_str());
        if (numrules < 1) {
          HUNSPELL_WARNING(stderr, "error: line %d: bad entry number\n",
                           fileMgr->getlinenum());
          return false;
        }
        phone = new phonetable(numrules);
        break;
      }
      default:
        break;
    }
    ++i;
    start_piece = mystrsep(line, iter);
  }
  if (np != 2) {
    HUNSPELL_WARNING(stderr, "error: line %d: missing data\n",
                     fileMgr->getlinenum());
    return false;
  }

  for (int j = 0; j < numrules; ++j) {
    std::string nl;
    if (!fileMgr->getline(nl))
      return false;
    mychomp(nl);
    i = 0;
    iter = nl.begin();
    start_piece = mystrsep(nl, iter);
    while (start_piece != nl.end()) {
      switch (i) {
        case 0: {
          if (nl.compare(start_piece - nl.begin(), 5, "PHONE", 5) != 0) {
            HUNSPELL_WARNING(stderr, "error: line %d: table is corrupt\n",
                             fileMgr->getlinenum());
            delete phone;
            phone = NULL;
            return false;
          }
          break;
        }
        case 1: {
          phone->rules[j].pattern.assign(start_piece, iter);
          break;
        }
        case 2: {
          phone->rules[j].pattern2.assign(start_piece, iter);
          break;
        }
        default:
          break;
      }
      ++i;
      start_piece = mystrsep(nl, iter);
    }
    if (phone->rules[j].pattern.empty() || phone->rules[j].pattern2.empty()) {
      HUNSPELL_WARNING(stderr, "error: line %d: table is corrupt\n",
                       fileMgr->getlinenum());
      delete phone;
      phone = NULL;
      return false;
    }
    // remove underscores from pattern and pattern2
    if (!ignorechars.empty()) {
      if (utf8) {
        remove_ignored_chars_utf(phone->rules[j].pattern, ignorechars_utf16);
        remove_ignored_chars_utf(phone->rules[j].pattern2, ignorechars_utf16);
      } else {
        remove_ignored_chars(phone->rules[j].pattern, ignorechars);
        remove_ignored_chars(phone->rules[j].pattern2, ignorechars);
      }
    }
    if (phone->rules[j].pattern.size() != phone->rules[j].pattern2.size()) {
      HUNSPELL_WARNING(stderr, "error: line %d: table is corrupt\n",
                       fileMgr->getlinenum());
      delete phone;
      phone = NULL;
      return false;
    }
  }

  // finalize the phonetable
  phone->hash();
  return true;
}
