  int np = 0;
  std::string::const_iterator iter = line.begin();
  std::string::const_iterator start_piece = mystrsep(line, iter);
  while (start_piece != line.end()) {
    switch (i) {
      case 0: {
        np++;
        break;
      }
      case 1: {
        numphone = atoi(std::string(start_piece, iter).c_str());
        if (numphone < 0) {
          HUNSPELL_WARNING(stderr, "error: line %d: bad entry number\n",
                           fileMgr->getlinenum());
          delete phone;
          phone = nullptr;
          return false;
        }
        phone->phonelist.reserve(std::min(numphone, 16384));
        np++;
        break;
      }
      default:
        break;
    }
    ++i;
    start_piece = mystrsep(line, iter);
  }
  if (np != 2) {
    HUNSPELL_WARNING(stderr, "error: line %d: missing data\n",
                     fileMgr->getlinenum());
    delete phone;
    phone = nullptr;
    return false;
  }

  /* now parse the numphone lines to read in the remainder of the table */
  for (int j = 0; j < numphone; ++j) {
    std::string nl;
    if (!fileMgr->getline(nl)) {
      delete phone;
      phone = nullptr;
      return false;
    }
    mychomp(nl);
    i = 0;
    iter = nl.begin();
    start_piece = mystrsep(nl, iter);
    std::string phoneme;
    std::string replacement;
    while (start_piece != nl.end()) {
      switch (i) {
        case 0: {
          if (nl.compare(start_piece - nl.begin(), 5, "PHONE", 5) != 0) {
            HUNSPELL_WARNING(stderr, "error: line %d: table is corrupt\n",
                             fileMgr->getlinenum());
            delete phone;
            phone = nullptr;
            return false;
          }
          break;
        }
        case 1: {
          phoneme.assign(start_piece, iter);
          break;
        }
        case 2: {
          replacement.assign(start_piece, iter);
          break;
        }
        default:
          break;
      }
      ++i;
      start_piece = mystrsep(nl, iter);
    }
    if (i != 3) {
      HUNSPELL_WARNING(stderr, "error: line %d: bad phonetable line\n",
                       fileMgr->getlinenum());
      delete phone;
      phone = nullptr;
      return false;
    }
    size_t under_pos = phoneme.find('_');
    if (under_pos != std::string::npos) {
      phoneme.erase(under_pos, 1);
    }
    under_pos = replacement.find('_');
    if (under_pos != std::string::npos) {
      replacement.erase(under_pos, 1);
    }
    phone->phonelist.push_back(std::make_pair(phoneme, replacement));
  }

  if (phone->phonelist.size() != static_cast<size_t>(numphone)) {

    phone = nullptr;
    return false;
