phone = new struct phonetable();
  if (!phone) {
    HUNSPELL_WARNING(stderr, "error: line %d: memory allocation failed\n", af->getlinenum());
    return false;
  }

  int i = 0;
  int np = 0;
  std::string::const_iterator iter = line.begin();
  std::string::const_iterator start_piece = mystrsep(line, iter);
  while (start_piece != line.end()) {
    switch (i) {
      case 0: {
        if (line.compare(start_piece - line.begin(), 5, "PHONE") != 0) {
          HUNSPELL_WARNING(stderr, "error: line %d: table is corrupt\n", af->getlinenum());
          delete phone;
          phone = NULL;
          return false;
        }
        np++;
        break;
      }
      case 1: {
        int numrules = atoi(std::string(start_piece, iter).c_str());
        if (numrules < 1) {
          HUNSPELL_WARNING(stderr, "error: line %d: bad entry number\n", af->getlinenum());
          delete phone;
          phone = NULL;
          return false;
        }
        phone->rules.reserve(numrules);
        np++;
        break;
      }
      default:
        break;
    }
    ++i;
    start_piece = mystrsep(line, iter);
  }

  if (np != 2) {
    HUNSPELL_WARNING(stderr, "error: line %d: missing data\n", af->getlinenum());
    delete phone;
    phone = NULL;
    return false;
  }

  for (int j = 0; j < phone->rules.capacity(); ++j) {
    std::string nl;
    if (!af->getline(nl)) {
      delete phone;
      phone = NULL;
      return false;
    }
    mychomp(nl);
    phone->rules.push_back(nl);
  }

  return true;
