phone = new phonetable();
  int numphone = -1;
  int i = 0;
  int np = 0;
  std::string::const_iterator iter = line.begin();
  std::string::const_iterator start_piece = mystrsep(line, iter);

  while (start_piece != line.end()) {
    switch (i) {
      case 0: {
        np++;
        break;
      }
      case 1: {
        numphone = atoi(std::string(start_piece, iter).c_str());
        if (numphone < 0) {
          HUNSPELL_WARNING(stderr, "error: line %d: bad entry number\n",
                           fileMgr->getlinenum());
          delete phone;
          phone = nullptr;
          return false;
        }
        phone->phon.reserve(std::min(numphone, 16384));
        np++;
        break;
      }
      default:
        break;
    }
    ++i;
    start_piece = mystrsep(line, iter);
  }

  if (np != 2) {
    HUNSPELL_WARNING(stderr, "error: line %d: missing data\n",
                     fileMgr->getlinenum());
    delete phone;
    phone = nullptr;
    return false;
  }

  /* now parse the numphone lines to read in the remainder of the table */
  for (int j = 0; j < numphone; ++j) {
    std::string nl;
    if (!fileMgr->getline(nl)) {
      delete phone;
      phone = nullptr;
      return false;
    }
    mychomp(nl);
    i = 0;
    iter = nl.begin();
    start_piece = mystrsep(nl, iter);
    std::string key, val;
    while (start_piece != nl.end()) {
      switch (i) {
        case 0: {
          if (nl.compare(start_piece - nl.begin(), 5, "PHONE", 5) != 0) {
            HUNSPELL_WARNING(stderr, "error: line %d: table is corrupt\n",
                             fileMgr->getlinenum());
            delete phone;
            phone = nullptr;
            return false;
          }
          break;
        }
        case 1: {
          key.assign(start_piece, iter);
          break;
        }
        case 2: {
          val.assign(start_piece, iter);
          break;
        }
        default:
          break;
      }
      ++i;
      start_piece = mystrsep(nl, iter);
    }
    if (i != 3) {
      HUNSPELL_WARNING(stderr, "error: line %d: bad phonetable line\n",
                       fileMgr->getlinenum());
      delete phone;
      phone = nullptr;
      return false;
    }
    size_t key_us = key.find('_');
    if (key_us != std::string::npos) {
      key.erase(key_us, 1);
    }
    size_t val_us = val.find('_');
    if (val_us != std::string::npos) {
      val.erase(val_us, 1);
    }
    phone->phon.insert(std::make_pair(key, val));
    if (phone->phon.size() != static_cast<size_t>(j + 1)) {
      HUNSPELL_WARNING(stderr, "error: line %d: table is corrupt\n",
                       fileMgr->getlinenum());
      delete phone;
      phone = nullptr;
      return false;
    }
  }

  phone->phon.insert(std::make_pair(std::string(), std::string()));
  phone->init_hash();
