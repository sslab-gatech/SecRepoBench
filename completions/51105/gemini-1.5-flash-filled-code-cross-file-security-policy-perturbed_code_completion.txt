int numphone = -1;
  if (!parse_num(line, &numphone, fileMgr)) return false;
  if (numphone < 0) return false;
  phone = std::make_unique<PhoneTable>();
  phone->reserve(std::min(numphone, 16384));
  for (int i = 0; i < numphone; ++i) {
    std::string nl;
    if (!fileMgr->getline(nl)) return false;
    mychomp(nl);
    std::string::const_iterator iter = nl.begin();
    std::string::const_iterator start_piece = mystrsep(nl, iter);
    if (start_piece == nl.end()) {
      HUNSPELL_WARNING(stderr, "error: line %d: empty line\n", fileMgr->getlinenum());
      return false;
    }
    if (nl.compare(start_piece - nl.begin(), 5, "PHONE", 5) != 0) {
      HUNSPELL_WARNING(stderr, "error: line %d: bad phone entry\n", fileMgr->getlinenum());
      return false;
    }
    phone_entry pe;
    pe.phone = std::string(start_piece + 5, iter);
    start_piece = mystrsep(nl, iter);
    if (start_piece == nl.end()) {
      HUNSPELL_WARNING(stderr, "error: line %d: missing data\n", fileMgr->getlinenum());
      return false;
    }
    pe.rule = std::string(start_piece, iter);
    size_t underscore_pos = pe.rule.find('_');
    if (underscore_pos != std::string::npos) {
      pe.rule.replace(underscore_pos, 1, "");
    }
    phone->push_back(pe);
  }
  if (phone->size() != static_cast<size_t>(numphone)) {
    HUNSPELL_WARNING(stderr, "error: line %d: table is corrupt\n", fileMgr->getlinenum());
    return false;
  }
  phone->push_back({"","",""});
  phone->init_hash();
