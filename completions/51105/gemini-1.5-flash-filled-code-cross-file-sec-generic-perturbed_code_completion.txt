int numphone = -1;
  if (!parse_num(line, &numphone, fileMgr)) return false;
  if (numphone < 0) return false;
  phone = new phonetable_t[numphone];
  for (int i = 0; i < numphone; ++i) {
    std::string nl;
    if (!fileMgr->getline(nl)) return false;
    mychomp(nl);
    std::string::const_iterator iter = nl.begin();
    std::string::const_iterator start_piece = mystrsep(nl, iter);
    int j = 0;
    while (start_piece != nl.end()) {
      switch (j) {
        case 0: {
          if (nl.compare(start_piece - nl.begin(), 5, "PHONE", 5) != 0) {
            HUNSPELL_WARNING(stderr, "error: line %d: bad phone entry\n",
                             fileMgr->getlinenum());
            return false;
          }
          break;
        }
        case 1: {
          phone[i].phonetic.assign(start_piece, iter);
          break;
        }
        case 2: {
          phone[i].rule.assign(start_piece, iter);
          size_t underscore_pos = phone[i].rule.find('_');
          if (underscore_pos != std::string::npos) {
            phone[i].rule.replace(underscore_pos, 1, "");
          }
          break;
        }
        default:
          break;
      }
      ++j;
      start_piece = mystrsep(nl, iter);
    }
  }
  if (phone) {
    phone[numphone].phonetic = "";
    phone[numphone].rule = "";
    phone_hash = new hash_t[numphone + 1];
    for (int i = 0; i <= numphone; ++i) {
      phone_hash[i] = 0;
    }
  }
