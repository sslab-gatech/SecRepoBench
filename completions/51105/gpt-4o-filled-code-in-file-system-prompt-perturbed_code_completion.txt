if (phone) {
    HUNSPELL_WARNING(stderr, "error: line %d: multiple table definitions\n",
                     fileMgr->getlinenum());
    return false;
  }
  int numphone = -1;
  int i = 0;
  int np = 0;
  std::string::const_iterator iter = line.begin();
  std::string::const_iterator start_piece = mystrsep(line, iter);
  while (start_piece != line.end()) {
    switch (i) {
      case 0: {
        np++;
        break;
      }
      case 1: {
        numphone = atoi(std::string(start_piece, iter).c_str());
        if (numphone < 1) {
          HUNSPELL_WARNING(stderr, "error: line %d: bad entry number\n",
                           fileMgr->getlinenum());
          return false;
        }
        phone = new struct phonetable;
        if (!phone)
          return false;
        phone->num = numphone;
        phone->p = new struct phentry[numphone];
        if (!phone->p)
          return false;
        np++;
        break;
      }
      default:
        break;
    }
    ++i;
    start_piece = mystrsep(line, iter);
  }
  if (np != 2) {
    HUNSPELL_WARNING(stderr, "error: line %d: missing data\n",
                     fileMgr->getlinenum());
    return false;
  }

  /* now parse the numphone lines to read in the remainder of the table */
  for (int j = 0; j < numphone; ++j) {
    std::string nl;
    if (!fileMgr->getline(nl))
      return false;
    mychomp(nl);
    i = 0;
    iter = nl.begin();
    start_piece = mystrsep(nl, iter);
    while (start_piece != nl.end()) {
      switch (i) {
        case 0: {
          if (nl.compare(start_piece - nl.begin(), 5, "PHONE", 5) != 0) {
            HUNSPELL_WARNING(stderr, "error: line %d: table is corrupt\n",
                             fileMgr->getlinenum());
            return false;
          }
          break;
        }
        case 1: {
          phone->p[j].pattern.assign(start_piece, iter);
          break;
        }
        case 2: {
          phone->p[j].ph.assign(start_piece, iter);
          break;
        }
        case 3: {
          phone->p[j].num = atoi(std::string(start_piece, iter).c_str());
          break;
        }
        default:
          break;
      }
      ++i;
      start_piece = mystrsep(nl, iter);
    }
    if (phone->p[j].pattern.empty() || phone->p[j].ph.empty()) {
      HUNSPELL_WARNING(stderr, "error: line %d: table is corrupt\n",
                       fileMgr->getlinenum());
      return false;
    }
    // remove underscores from pattern and ph
    phone->p[j].pattern.erase(
        std::remove(phone->p[j].pattern.begin(), phone->p[j].pattern.end(), '_'),
        phone->p[j].pattern.end());
    phone->p[j].ph.erase(
        std::remove(phone->p[j].ph.begin(), phone->p[j].ph.end(), '_'),
        phone->p[j].ph.end());
  }
  phone->p[numphone].pattern.clear();
  phone->p[numphone].ph.clear();
  phone->p[numphone].num = 0;
  phone->init();
  return true;
