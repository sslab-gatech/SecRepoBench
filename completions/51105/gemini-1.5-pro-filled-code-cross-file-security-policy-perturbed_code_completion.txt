std::unique_ptr<PhoneTable> pt(new PhoneTable());
  phone = pt.get();

  int i = 0;
  int np = 0;
  std::string::const_iterator iter = line.begin();
  std::string::const_iterator start_piece = mystrsep(line, iter);
  while (start_piece != line.end()) {
    switch (i) {
      case 0: {
        np++;
        break;
      }
      case 1: {
        phone->nump = atoi(std::string(start_piece, iter).c_str());
        if (phone->nump < 1) {
          HUNSPELL_WARNING(stderr, "error: line %d: bad entry number\n",
                           fileMgr->getlinenum());
          return false;
        }
        np++;
        break;
      }
      default:
        break;
    }
    ++i;
    start_piece = mystrsep(line, iter);
  }
  if (np != 2) {
    HUNSPELL_WARNING(stderr, "error: line %d: missing data\n",
                     fileMgr->getlinenum());
    return false;
  }

  /* now parse the phone->nump lines to read in the remainder of the table */
  for (int j = 0; j < phone->nump; ++j) {
    std::string nl;
    if (!fileMgr->getline(nl))
      return false;
    mychomp(nl);
    i = 0;
    iter = nl.begin();
    start_piece = mystrsep(nl, iter);
    while (start_piece != nl.end()) {
      switch (i) {
        case 0: {
          if (nl.compare(start_piece - nl.begin(), 5, "PHONE", 5) != 0) {
            HUNSPELL_WARNING(stderr, "error: line %d: table is corrupt\n",
                             fileMgr->getlinenum());
            return false;
          }
          break;
        }
        case 1: {
          std::string piece(start_piece, iter);
          size_t slash_pos = piece.find('/');
          if (slash_pos != std::string::npos) {
            std::string chunk(piece, slash_pos + 1);
            piece.resize(slash_pos);
            phone->phonetbl.push_back(PhoneEntry());
            phone->phonetbl.back().cond = pHMgr->decode_flag(chunk);
          } else {
            phone->phonetbl.push_back(PhoneEntry());
          }
          stripUnderscores(piece);
          phone->phonetbl.back().key = piece;
          break;
        }
        case 2: {
          std::string piece(start_piece, iter);
          stripUnderscores(piece);
          phone->phonetbl.back().value = piece;
          break;
        }
        default:
          break;
      }
      ++i;
      start_piece = mystrsep(nl, iter);
    }
    if (phone->phonetbl.back().key.size() != phone->phonetbl.back().value.size()) {
      HUNSPELL_WARNING(stderr, "error: line %d: mismatched PHONE rules\n",
                       fileMgr->getlinenum());
      return false;
    }
  }

  phone->phonetbl.push_back(PhoneEntry());  // add empty strings
  phone->make_hash();
