{
  // Tokenize the header line.
  std::string::const_iterator iter = line.begin();
  std::string::const_iterator token = mystrsep(line, iter);
  int num_rules = 0;
  int token_count = 0;
  while (token != line.end()) {
    switch (token_count) {
      case 0:
        // Optionally, check that the command is "PHONETABLE"
        break;
      case 1:
        num_rules = atoi(std::string(token, iter).c_str());
        break;
      default:
        break;
    }
    ++token_count;
    token = mystrsep(line, iter);
  }
  if (token_count != 2 || num_rules < 0) {
    HUNSPELL_WARNING(stderr,
                     "error: line %d: invalid phonetable parameters\n",
                     fileMgr->getlinenum());
    return false;
  }

  // Initialize the phonetable structure.
  phone = new Phonetable();
  phone->rules.reserve(num_rules);

  // Read and process each phonetic rule.
  for (int i = 0; i < num_rules; ++i) {
    std::string ruleLine;
    if (!fileMgr->getline(ruleLine))
      return false;
    mychomp(ruleLine);
    std::string::const_iterator riter = ruleLine.begin();
    std::string::const_iterator rule_token = mystrsep(ruleLine, riter);
    if (ruleLine.compare(rule_token - ruleLine.begin(), 5, "PHONE", 5) != 0) {
      HUNSPELL_WARNING(stderr,
                       "error: line %d: phonetic rule does not start with PHONE\n",
                       fileMgr->getlinenum());
      return false;
    }
    // Extract the phonetic rule details.
    rule_token = mystrsep(ruleLine, riter);
    if (rule_token == ruleLine.end()) {
      HUNSPELL_WARNING(stderr,
                       "error: line %d: incomplete phonetic rule\n",
                       fileMgr->getlinenum());
      return false;
    }
    std::string rule_detail(rule_token, riter);
    // Strip underscores from the rule.
    rule_detail.erase(std::remove(rule_detail.begin(), rule_detail.end(), '_'),
                      rule_detail.end());
    phone->rules.push_back(rule_detail);
  }
  if (phone->rules.size() != static_cast<size_t>(num_rules)) {
    HUNSPELL_WARNING(stderr,
                     "error: line %d: phonetable rule count mismatch\n",
                     fileMgr->getlinenum());
    return false;
  }

  // Finalize the phonetable by adding empty strings and initializing its hash.
  phone->rules.push_back("");
  phone->rules.push_back("");
  phone->init_hash();
}
