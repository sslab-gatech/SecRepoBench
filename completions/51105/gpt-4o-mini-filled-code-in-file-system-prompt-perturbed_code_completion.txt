phone = new struct phonetable();
  if (!phone) {
    HUNSPELL_WARNING(stderr, "error: line %d: could not allocate memory for phonetic table\n", af->getlinenum());
    return false;
  }
  phone->num = 0;
  phone->entries.reserve(256); // Reserve space for phonetic entries

  // Parse the phonetic rules
  std::string::const_iterator iter = line.begin();
  std::string::const_iterator start_piece = mystrsep(line, iter);
  while (start_piece != line.end()) {
    std::string rule(start_piece, iter);
    phone->entries.push_back(rule);
    phone->num++;
    start_piece = mystrsep(line, iter);
  }

  return true;
