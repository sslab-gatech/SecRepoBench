if (phone) {
    HUNSPELL_WARNING(stderr, "error: line %d: multiple table definitions\n",
                     fileMgr->getlinenum());
    return false;
  }

  int numphone = -1;
  int i = 0;
  int np = 0;
  std::string::const_iterator iter = line.begin();
  std::string::const_iterator start_piece = mystrsep(line, iter);
  while (start_piece != line.end()) {
    switch (i) {
      case 0: {
        np++;
        break;
      }
      case 1: {
        numphone = atoi(std::string(start_piece, iter).c_str());
        if (numphone < 1) {
          HUNSPELL_WARNING(stderr, "error: line %d: bad entry number\n",
                           fileMgr->getlinenum());
          return false;
        }
        phone = new phonetable;
        phone->num = numphone;
        phone->phon = new phentry[numphone];
        np++;
        break;
      }
      default:
        break;
    }
    ++i;
    start_piece = mystrsep(line, iter);
  }
  if (np != 2) {
    HUNSPELL_WARNING(stderr, "error: line %d: missing data\n",
                     fileMgr->getlinenum());
    return false;
  }

  for (int j = 0; j < numphone; ++j) {
    std::string nl;
    if (!fileMgr->getline(nl))
      return false;
    mychomp(nl);
    i = 0;
    iter = nl.begin();
    start_piece = mystrsep(nl, iter);
    while (start_piece != nl.end()) {
      switch (i) {
        case 0: {
          if (nl.compare(start_piece - nl.begin(), 5, "PHONE", 5) != 0) {
            HUNSPELL_WARNING(stderr, "error: line %d: table is corrupt\n",
                             fileMgr->getlinenum());
            return false;
          }
          break;
        }
        case 1: {
          phone->phon[j].pattern.assign(start_piece, iter);
          break;
        }
        case 2: {
          phone->phon[j].replacement.assign(start_piece, iter);
          break;
        }
        default:
          break;
      }
      ++i;
      start_piece = mystrsep(nl, iter);
    }
    if (phone->phon[j].pattern.empty() || phone->phon[j].replacement.empty()) {
      HUNSPELL_WARNING(stderr, "error: line %d: table is corrupt\n",
                       fileMgr->getlinenum());
      return false;
    }
    // Remove underscores from the pattern and replacement
    phone->phon[j].pattern.erase(
        std::remove(phone->phon[j].pattern.begin(), phone->phon[j].pattern.end(), '_'),
        phone->phon[j].pattern.end());
    phone->phon[j].replacement.erase(
        std::remove(phone->phon[j].replacement.begin(), phone->phon[j].replacement.end(), '_'),
        phone->phon[j].replacement.end());
  }

  // Add empty string to the end of the phon table
  phone->phon[numphone].pattern.clear();
  phone->phon[numphone].replacement.clear();
  phone->init_phon_hash();
  return true;
