phone = new Phonetable();
  int i = 0;
  int np = 0;
  std::string::const_iterator iter = line.begin();
  std::string::const_iterator start_piece = mystrsep(line, iter);
  while (start_piece != line.end()) {
    switch (i) {
      case 0:
        np++;
        break;
      case 1:
        phone->pnum = atoi(std::string(start_piece, iter).c_str());
        if (phone->pnum < 1) {
          HUNSPELL_WARNING(stderr, "error: line %d: bad entry number\n",
                           fileMgr->getlinenum());
          return false;
        }
        np++;
        break;
      default:
        break;
    }
    ++i;
    start_piece = mystrsep(line, iter);
  }
  if (np != 2) {
    HUNSPELL_WARNING(stderr, "error: line %d: missing data\n",
                     fileMgr->getlinenum());
    return false;
  }

  /* now parse the phone->pnum lines to read in the remainder of the table */
  phone->rules.reserve(phone->pnum);
  for (int j = 0; j < phone->pnum; ++j) {
    std::string nl;
    if (!fileMgr->getline(nl))
      return false;
    mychomp(nl);
    i = 0;
    phone->rules.push_back(phonerule());
    iter = nl.begin();
    start_piece = mystrsep(nl, iter);
    while (start_piece != nl.end()) {
      switch (i) {
        case 0:
          if (nl.compare(start_piece - nl.begin(), 5, "PHONE", 5) != 0) {
            HUNSPELL_WARNING(stderr, "error: line %d: table is corrupt\n",
                             fileMgr->getlinenum());
            return false;
          }
          break;
        case 1: {
          std::string piece(start_piece, iter);
          size_t underscore_pos = piece.find('_');
          if (underscore_pos != std::string::npos) {
            phone->rules.back().phon.assign(piece, 0, underscore_pos);
          } else {
            phone->rules.back().phon.assign(piece);
          }
          break;
        }
        case 2: {
          std::string piece(start_piece, iter);
          size_t underscore_pos = piece.rfind('_');
          if (underscore_pos != std::string::npos) {
            phone->rules.back().acceptor.assign(piece, 0, underscore_pos);
          } else {
            phone->rules.back().acceptor.assign(piece);
          }
          break;
        }
        default:
          break;
      }
      ++i;
      start_piece = mystrsep(nl, iter);
    }
    if (phone->rules.back().phon.size() != phone->rules.back().acceptor.size()) {
      phone->rules.pop_back();
      HUNSPELL_WARNING(stderr, "error: line %d: size mismatch\n",
                       fileMgr->getlinenum());
      return false;
    }
  }

  phone->rules.push_back(phonerule());  // add an empty rule to the end
  phone->rules.back().phon = "";
  phone->rules.back().acceptor = "";
  phone->create_hash();
