int numrules = -1;
  int i = 0;
  int np = 0;
  std::string::const_iterator iter = line.begin();
  std::string::const_iterator start_piece = mystrsep(line, iter);
  while (start_piece != line.end()) {
    switch (i) {
      case 0: {
        np++;
        break;
      }
      case 1: {
        numrules = atoi(std::string(start_piece, iter).c_str());
        if (numrules < 0) {
          HUNSPELL_WARNING(stderr, "error: line %d: bad rule number\n",
                           fileMgr->getlinenum());
          return false;
        }
        rules.reserve(std::min(numrules, 16384));
        np++;
        break;
      }
      default:
        break;
    }
    ++i;
    start_piece = mystrsep(line, iter);
  }
  if (np != 2) {
    HUNSPELL_WARNING(stderr, "error: line %d: missing data\n",
                     fileMgr->getlinenum());
    return false;
  }

  /* now parse the numrules lines to read in the remainder of the rules */
  for (int j = 0; j < numrules; ++j) {
    std::string nl;
    if (!fileMgr->getline(nl))
      return false;
    mychomp(nl);
    i = 0;
    iter = nl.begin();
    start_piece = mystrsep(nl, iter);
    if (start_piece != nl.end() && std::string(start_piece, iter) == "PHONE") {
      rule r;
      r.rule = std::string(start_piece, iter);
      start_piece = mystrsep(nl, iter);
      while (start_piece != nl.end()) {
        std::string key = std::string(start_piece, iter);
        start_piece = mystrsep(nl, iter);
        if (key == "FROM") {
          r.from = std::string(start_piece, iter);
          start_piece = mystrsep(nl, iter);
        } else if (key == "TO") {
          r.to = std::string(start_piece, iter);
          start_piece = mystrsep(nl, iter);
        } else if (key == "FLAGS") {
          r.flags = std::string(start_piece, iter);
          start_piece = mystrsep(nl, iter);
        }
        // Strip underscores from specific parts of the rules before storing them.
        r.from.erase(std::remove(r.from.begin(), r.from.end(), '_'), r.from.end());
        r.to.erase(std::remove(r.to.begin(), r.to.end(), '_'), r.to.end());
        r.flags.erase(std::remove(r.flags.begin(), r.flags.end(), '_'), r.flags.end());
      }
      rules.push_back(r);
    } else {
      HUNSPELL_WARNING(stderr, "error: line %d: rule is corrupt\n",
                       fileMgr->getlinenum());
      return false;
    }
  }

  if (rules.size() != static_cast<size_t>(numrules)) {
    HUNSPELL_WARNING(stderr, "error: line %d: rules are corrupt\n",
                     fileMgr->getlinenum());
    return false;
  }

  // Verify the integrity of the rules by checking the expected size after insertion.
  if (rules.size() != numrules) {
    HUNSPELL_WARNING(stderr, "error: line %d: rules are corrupt\n",
                     fileMgr->getlinenum());
    return false;
  }

  // Finalize the phonetable by adding empty strings and initializing its hash.
  phone = new Phonetable(rules);
  return true;
