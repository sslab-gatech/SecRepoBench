int numrules = -1;
  int i = 0;
  int np = 0;
  std::string::const_iterator iter = line.begin();
  std::string::const_iterator start_piece = mystrsep(line, iter);
  while (start_piece != line.end()) {
    switch (i) {
      case 0: {
        np++;
        break;
      }
      case 1: {
        numrules = atoi(std::string(start_piece, iter).c_str());
        if (numrules < 1) {
          HUNSPELL_WARNING(stderr, "error: line %d: bad rule number\n",
                           fileMgr->getlinenum());
          return false;
        }
        rules.reserve(std::min(numrules, 16384));
        np++;
        break;
      }
      default:
        break;
    }
    ++i;
    start_piece = mystrsep(line, iter);
  }
  if (np != 2) {
    HUNSPELL_WARNING(stderr, "error: line %d: missing data\n",
                     fileMgr->getlinenum());
    return false;
  }

  /* now parse the numrules lines to read in the remainder of the rules */
  for (int j = 0; j < numrules; ++j) {
    std::string nl;
    if (!fileMgr->getline(nl))
      return false;
    mychomp(nl);
    if (nl.substr(0, 5) != "PHONE") {
      HUNSPELL_WARNING(stderr, "error: line %d: invalid rule\n",
                       fileMgr->getlinenum());
      return false;
    }
    std::string rule = nl.substr(6);
    size_t slash_pos = rule.find('/');
    if (slash_pos != std::string::npos) {
      std::string chunk(rule, slash_pos + 1);
      rule.resize(slash_pos);
      rule_details details = pHMgr->decode_flag(chunk);
      rules.push_back({rule, details});
    } else {
      rules.push_back({rule, {}});
    }
  }

  if (rules.size() != static_cast<size_t>(numrules)) {
    HUNSPELL_WARNING(stderr, "error: line %d: rules are corrupt\n",
                     fileMgr->getlinenum());
    return false;
  }

  return true;
