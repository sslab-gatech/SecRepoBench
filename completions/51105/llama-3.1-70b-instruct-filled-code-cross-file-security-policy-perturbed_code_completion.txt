// Parse the input line to extract phonetic table parameters.
int n = 0;
int np = 0;
std::string::const_iterator iter = line.begin();
std::string::const_iterator start_piece = mystrsep(line, iter);
while (start_piece!= line.end()) {
  switch (n) {
    case 0: {
      np++;
      break;
    }
    case 1: {
      phone = new phonetable();
      phone->num = atoi(std::string(start_piece, iter).c_str());
      if (phone->num < 1) {
        HUNSPELL_WARNING(stderr, "error: line %d: bad entry number\n",
                         fileMgr->getlinenum());
        delete phone;
        phone = NULL;
        return false;
      }
      phone->table.reserve(std::min(phone->num, 16384));
      np++;
      break;
    }
    default:
      break;
  }
  ++n;
  start_piece = mystrsep(line, iter);
}

if (np!= 2) {
  HUNSPELL_WARNING(stderr, "error: line %d: missing data\n",
                   fileMgr->getlinenum());
  delete phone;
  phone = NULL;
  return false;
}

// Read and process the specified number of phonetic rules from the file.
for (int j = 0; j < phone->num; ++j) {
  std::string nl;
  if (!fileMgr->getline(nl))
    return false;
  mychomp(nl);
  n = 0;
  iter = nl.begin();
  start_piece = mystrsep(nl, iter);
  while (start_piece!= nl.end()) {
    switch (n) {
      case 0: {
        if (nl.compare(start_piece - nl.begin(), 5, "PHONE", 5)!= 0) {
          HUNSPELL_WARNING(stderr, "error: line %d: table is corrupt\n",
                           fileMgr->getlinenum());
          delete phone;
          phone = NULL;
          return false;
        }
        break;
      }
      case 1: {
        phone->table.push_back(std::string(start_piece, iter));
        break;
      }
      default:
        break;
    }
    ++n;
    start_piece = mystrsep(nl, iter);
  }
}

// Verify the integrity of the rules by checking the expected size after insertion.
if (phone->table.size()!= static_cast<size_t>(phone->num)) {
  HUNSPELL_WARNING(stderr, "error: line %d: table is corrupt\n",
                   fileMgr->getlinenum());
  delete phone;
  phone = NULL;
  return false;
}

// Finalize the phonetable by adding empty strings and initializing its hash.
phone->table.push_back("");
phone->hash = new hashset();
for (size_t i = 0; i < phone->table.size(); ++i) {
  phone->hash->insert(phone->table[i]);
}
