int numphone = -1;
  int i = 0;
  int np = 0;
  std::string::const_iterator iter = line.begin();
  std::string::const_iterator start_piece = mystrsep(line, iter);
  
  // Parse the header line
  while (start_piece != line.end()) {
    switch (i) {
      case 0: {
        np++;
        break;
      }
      case 1: {
        numphone = atoi(std::string(start_piece, iter).c_str());
        if (numphone < 1) {
          HUNSPELL_WARNING(stderr, "error: line %d: bad entry number\n",
                           fileMgr->getlinenum());
          return false;
        }
        np++;
        break;
      }
      default:
        break;
    }
    ++i;
    start_piece = mystrsep(line, iter);
  }
  
  if (np != 2) {
    HUNSPELL_WARNING(stderr, "error: line %d: missing data\n",
                     fileMgr->getlinenum());
    return false;
  }
  
  // Initialize the phonetable structure
  phone = (phonetable*)malloc(sizeof(phonetable));
  if (!phone)
    return false;
  
  phone->utf8 = utf8;
  phone->langnum = numphone;
  phone->rules = (phoneentry**)malloc(sizeof(phoneentry*) * (numphone + 1));
  if (!phone->rules) {
    free(phone);
    phone = NULL;
    return false;
  }
  
  // Read and process the phonetic rules
  for (int j = 0; j < numphone; ++j) {
    std::string nl;
    if (!fileMgr->getline(nl)) {
      for (int k = 0; k < j; k++) {
        free(phone->rules[k]->pattern);
        free(phone->rules[k]->replace);
        free(phone->rules[k]);
      }
      free(phone->rules);
      free(phone);
      phone = NULL;
      return false;
    }
    mychomp(nl);
    i = 0;
    phone->rules[j] = (phoneentry*)malloc(sizeof(phoneentry));
    if (!phone->rules[j]) {
      for (int k = 0; k < j; k++) {
        free(phone->rules[k]->pattern);
        free(phone->rules[k]->replace);
        free(phone->rules[k]);
      }
      free(phone->rules);
      free(phone);
      phone = NULL;
      return false;
    }
    
    iter = nl.begin();
    start_piece = mystrsep(nl, iter);
    std::string chunk;
    
    while (start_piece != nl.end()) {
      switch (i) {
        case 0: {
          if (nl.compare(start_piece - nl.begin(), 5, "PHONE", 5) != 0) {
            HUNSPELL_WARNING(stderr, "error: line %d: table is corrupt\n",
                             fileMgr->getlinenum());
            numphone = 0;
            for (int k = 0; k < j; k++) {
              free(phone->rules[k]->pattern);
              free(phone->rules[k]->replace);
              free(phone->rules[k]);
            }
            free(phone->rules);
            free(phone);
            phone = NULL;
            return false;
          }
          break;
        }
        case 1: {
          chunk.assign(start_piece, iter);
          // Strip underscores from the pattern
          if (!chunk.empty() && chunk[chunk.size() - 1] == '_')
            chunk.resize(chunk.size() - 1);
          phone->rules[j]->pattern = mystrdup(chunk.c_str());
          if (!phone->rules[j]->pattern) {
            for (int k = 0; k < j; k++) {
              free(phone->rules[k]->pattern);
              free(phone->rules[k]->replace);
              free(phone->rules[k]);
            }
            free(phone->rules[j]);
            free(phone->rules);
            free(phone);
            phone = NULL;
            return false;
          }
          break;
        }
        case 2: {
          chunk.assign(start_piece, iter);
          // Strip underscores from the replace pattern
          if (!chunk.empty() && chunk[0] == '_')
            chunk.erase(0, 1);
          phone->rules[j]->replace = mystrdup(chunk.c_str());
          if (!phone->rules[j]->replace) {
            free(phone->rules[j]->pattern);
            for (int k = 0; k < j; k++) {
              free(phone->rules[k]->pattern);
              free(phone->rules[k]->replace);
              free(phone->rules[k]);
            }
            free(phone->rules[j]);
            free(phone->rules);
            free(phone);
            phone = NULL;
            return false;
          }
          break;
        }
        default:
          break;
      }
      ++i;
      start_piece = mystrsep(nl, iter);
    }
    
    if (i < 3) {
      HUNSPELL_WARNING(stderr, "error: line %d: table is corrupt\n",
                       fileMgr->getlinenum());
      for (int k = 0; k <= j; k++) {
        if (k < j || i > 1) free(phone->rules[k]->pattern);
        if (k < j || i > 2) free(phone->rules[k]->replace);
        free(phone->rules[k]);
      }
      free(phone->rules);
      free(phone);
      phone = NULL;
      return false;
    }
  }
  
  // Finalize the phonetable
  phone->rules[numphone] = NULL;
  init_phonet_hash(*phone);
