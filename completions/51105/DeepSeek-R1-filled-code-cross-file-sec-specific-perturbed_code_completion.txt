int numphone = 0;
  int np = 0;
  int i = 0;
  std::string::const_iterator iter = line.begin();
  std::string::const_iterator start_piece = mystrsep(line, iter);
  while (start_piece != line.end()) {
    switch (i) {
      case 0: {
        std::string key(start_piece, iter);
        if (key != "PHONE") {
          HUNSPELL_WARNING(stderr, "error: line %d: missing PHONE keyword\n",
                           fileMgr->getlinenum());
          return false;
        }
        np++;
        break;
      }
      case 1: {
        numphone = atoi(std::string(start_piece, iter).c_str());
        if (numphone <= 0) {
          HUNSPELL_WARNING(stderr, "error: line %d: bad entry count\n",
                           fileMgr->getlinenum());
          return false;
        }
        np++;
        break;
      }
      default:
        break;
    }
    ++i;
    start_piece = mystrsep(line, iter);
  }
  if (np != 2) {
    HUNSPELL_WARNING(stderr, "error: line %d: missing PHONE data\n",
                     fileMgr->getlinenum());
    return false;
  }

  phone = new phonetable;
  phone->num_rules = numphone;
  phone->rules.reserve(numphone);

  for (int j = 0; j < numphone; ++j) {
    std::string nl;
    if (!fileMgr->getline(nl)) {
      HUNSPELL_WARNING(stderr, "error: line %d: truncated table\n",
                       fileMgr->getlinenum());
      return false;
    }
    mychomp(nl);
    i = 0;
    std::string pattern;
    std::string::const_iterator rule_iter = nl.begin();
    std::string::const_iterator rule_start = mystrsep(nl, rule_iter);
    while (rule_start != nl.end()) {
      switch (i) {
        case 0: {
          std::string key(rule_start, rule_iter);
          if (key != "PHONE") {
            HUNSPELL_WARNING(stderr, "error: line %d: missing rule PHONE\n",
                             fileMgr->getlinenum());
            return false;
          }
          break;
        }
        case 1: {
          pattern.assign(rule_start, rule_iter);
          pattern.erase(std::remove(pattern.begin(), pattern.end(), '_'),
                        pattern.end());
          phone->rules.push_back(pattern);
          break;
        }
        default:
          break;
      }
      ++i;
      rule_start = mystrsep(nl, rule_iter);
    }
    if (i < 2) {
      HUNSPELL_WARNING(stderr, "error: line %d: truncated rule\n",
                       fileMgr->getlinenum());
      return false;
    }
  }

  if (static_cast<int>(phone->rules.size()) != numphone) {
    HUNSPELL_WARNING(stderr, "error: line %d: table size mismatch\n",
                     fileMgr->getlinenum());
    return false;
  }

  phone->rules.push_back("");
  phone->hash.init();
  for (size_t k = 0; k < phone->rules.size(); ++k) {
    phone->hash.add(phone->rules[k].c_str());
  }
