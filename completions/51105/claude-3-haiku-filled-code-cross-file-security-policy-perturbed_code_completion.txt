phone = new phonetable();
std::string::const_iterator iter = line.begin();
std::string::const_iterator start_piece = mystrsep(line, iter);
int np = 0;
while (start_piece != line.end()) {
  switch (np) {
    case 0: {
      phone->num = atoi(std::string(start_piece, iter).c_str());
      break;
    }
    case 1: {
      phone->rules = new char*[phone->num];
      for (int i = 0; i < phone->num; i++) {
        std::string nl;
        if (!fileMgr->getline(nl))
          return false;
        mychomp(nl);
        if (nl.compare(0, 5, "PHONE", 5) != 0) {
          HUNSPELL_WARNING(stderr, "error: line %d: table is corrupt\n",
                           fileMgr->getlinenum());
          return false;
        }
        std::string::const_iterator iter2 = nl.begin() + 5;
        std::string::const_iterator start_piece2 = mystrsep(nl, iter2);
        phone->rules[i] = mystrdup(std::string(start_piece2, iter2).c_str());
        reverse_condition(phone->rules[i]);
        std::replace(phone->rules[i], phone->rules[i] + strlen(phone->rules[i]), '_', ' ');
      }
      break;
    }
    default:
      break;
  }
  np++;
  start_piece = mystrsep(line, iter);
}
if (np != 2) {
  HUNSPELL_WARNING(stderr, "error: line %d: missing data\n",
                   fileMgr->getlinenum());
  return false;
}
if (phone->num != static_cast<int>(phone->rules.size())) {
  HUNSPELL_WARNING(stderr, "error: line %d: table is corrupt\n",
                   fileMgr->getlinenum());
  return false;
}
phone->hash = new std::vector<std::vector<int>>(phone->num);
for (int i = 0; i < phone->num; i++) {
  (*phone->hash)[i].push_back(0);
}
return true;
