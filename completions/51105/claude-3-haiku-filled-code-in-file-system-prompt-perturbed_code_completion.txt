bool AffixMgr::parse_phonetable(const std::string& line, FileMgr* fileMgr) {
  if (phone) {
    HUNSPELL_WARNING(stderr, "error: line %d: multiple table definitions\n",
                     fileMgr->getlinenum());
    return false;
  }
  // Parse the input line to extract phonetic table parameters.
  // Initialize a new phonetable structure and set its properties.
  // Validate the line to ensure it contains the correct number of entries.
  // Read and process the specified number of phonetic rules from the file.
  // For each rule, ensure it starts with a "PHONE" keyword, then extract and store the rule details.
  // Strip underscores from specific parts of the rules before storing them.
  // Verify the integrity of the rules by checking the expected size after insertion.
  // Finalize the phonetable by adding empty strings and initializing its hash.
  // Return false if any errors or inconsistencies are encountered during parsing.

  int numrules = -1;
  int i = 0;
  int np = 0;
  std::string::const_iterator iter = line.begin();
  std::string::const_iterator start_piece = mystrsep(line, iter);
  while (start_piece != line.end()) {
    switch (i) {
      case 0: {
        np++;
        break;
      }
      case 1: {
        numrules = atoi(std::string(start_piece, iter).c_str());
        if (numrules < 1) {
          HUNSPELL_WARNING(stderr, "error: line %d: bad entry number\n",
                           fileMgr->getlinenum());
          return false;
        }
        phone = new phonetable(numrules);
        np++;
        break;
      }
      default:
        break;
    }
    ++i;
    start_piece = mystrsep(line, iter);
  }
  if (np != 2) {
    HUNSPELL_WARNING(stderr, "error: line %d: missing data\n",
                     fileMgr->getlinenum());
    return false;
  }

  // now parse the numrules lines to read in the remainder of the table
  for (int j = 0; j < numrules; ++j) {
    std::string nl;
    if (!fileMgr->getline(nl))
      return false;
    mychomp(nl);
    i = 0;
    iter = nl.begin();
    start_piece = mystrsep(nl, iter);
    while (start_piece != nl.end()) {
      switch (i) {
        case 0: {
          if (nl.compare(start_piece - nl.begin(), 5, "PHONE", 5) != 0) {
            HUNSPELL_WARNING(stderr, "error: line %d: table is corrupt\n",
                             fileMgr->getlinenum());
            return false;
          }
          break;
        }
        case 1: {
          phone->rules[j].pattern.assign(start_piece, iter);
          std::replace(phone->rules[j].pattern.begin(), phone->rules[j].pattern.end(), '_', ' ');
          break;
        }
        case 2: {
          phone->rules[j].pattern2.assign(start_piece, iter);
          std::replace(phone->rules[j].pattern2.begin(), phone->rules[j].pattern2.end(), '_', ' ');
          break;
        }
        default:
          break;
      }
      ++i;
      start_piece = mystrsep(nl, iter);
    }
    if (phone->rules[j].pattern.empty() || phone->rules[j].pattern2.empty()) {
      HUNSPELL_WARNING(stderr, "error: line %d: table is corrupt\n",
                       fileMgr->getlinenum());
      return false;
    }
  }

  // Finalize the phonetable by adding empty strings and initializing its hash
  phone->init_hash();

  return true;
}
