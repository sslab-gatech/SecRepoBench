int numphone = -1;
  int i = 0;
  int np = 0;
  std::string::const_iterator iter = line.begin();
  std::string::const_iterator start_piece = mystrsep(line, iter);
  while (start_piece != line.end()) {
    switch (i) {
      case 0: {
        np++;
        break;
      }
      case 1: {
        numphone = atoi(std::string(start_piece, iter).c_str());
        if (numphone < 1) {
          HUNSPELL_WARNING(stderr, "error: line %d: bad entry number\n",
                           fileMgr->getlinenum());
          return false;
        }
        phone = new phonetable();
        if (!phone) {
          HUNSPELL_WARNING(stderr, "error: line %d: can't allocate phone table\n",
                           fileMgr->getlinenum());
          return false;
        }
        phone->rules.reserve(std::min(numphone, 16384));
        np++;
        break;
      }
      default:
        break;
    }
    ++i;
    start_piece = mystrsep(line, iter);
  }
  if (np != 2) {
    HUNSPELL_WARNING(stderr, "error: line %d: missing data\n",
                     fileMgr->getlinenum());
    delete phone;
    phone = NULL;
    return false;
  }

  /* now parse the numphone lines to read in the remainder of the table */
  for (int j = 0; j < numphone; ++j) {
    std::string nl;
    if (!fileMgr->getline(nl)) {
      delete phone;
      phone = NULL;
      return false;
    }
    mychomp(nl);
    i = 0;
    phonentry entry;
    entry.first = "";  // Initialize to empty string
    entry.second = ""; // Initialize to empty string
    iter = nl.begin();
    start_piece = mystrsep(nl, iter);
    while (start_piece != nl.end()) {
      switch (i) {
        case 0: {
          if (nl.compare(start_piece - nl.begin(), 5, "PHONE", 5) != 0) {
            HUNSPELL_WARNING(stderr, "error: line %d: table is corrupt\n",
                             fileMgr->getlinenum());
            delete phone;
            phone = NULL;
            return false;
          }
          break;
        }
        case 1: {
          // Store the first part of the rule, stripping underscores
          entry.first.assign(start_piece, iter);
          size_t pos = entry.first.find('_');
          while (pos != std::string::npos) {
            entry.first.erase(pos, 1);
            pos = entry.first.find('_');
          }
          break;
        }
        case 2: {
          // Store the second part of the rule, stripping underscores
          entry.second.assign(start_piece, iter);
          size_t pos = entry.second.find('_');
          while (pos != std::string::npos) {
            entry.second.erase(pos, 1);
            pos = entry.second.find('_');
          }
          break;
        }
        default:
          break;
      }
      ++i;
      start_piece = mystrsep(nl, iter);
    }
    phone->rules.push_back(entry);
  }

  // Verify the integrity of the phonetic table
  if (phone->rules.size() != static_cast<size_t>(numphone)) {
    HUNSPELL_WARNING(stderr, "error: line %d: table is corrupt\n",
                     fileMgr->getlinenum());
    delete phone;
    phone = NULL;
    return false;
  }
