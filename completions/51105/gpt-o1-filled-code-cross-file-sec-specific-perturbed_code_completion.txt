{
  // parse the initial line to extract the number of phonetic rules
  int i = 0, np = 0;
  int numphone = -1;
  std::string::const_iterator iter = line.begin();
  std::string::const_iterator start_piece = mystrsep(line, iter);

  while (start_piece != line.end()) {
    switch (i) {
      case 0: {
        // first piece is just a keyword placeholder
        np++;
        break;
      }
      case 1: {
        // second piece contains the number of phone rules
        numphone = atoi(std::string(start_piece, iter).c_str());
        if (numphone < 1) {
          HUNSPELL_WARNING(stderr, "error: line %d: invalid phonetable size\n",
                           fileMgr->getlinenum());
          return false;
        }
        np++;
        break;
      }
      default:
        break;
    }
    ++i;
    start_piece = mystrsep(line, iter);
  }

  if (np != 2) {
    HUNSPELL_WARNING(stderr, "error: line %d: missing phonetable data\n",
                     fileMgr->getlinenum());
    return false;
  }

  // allocate and prepare the phonetable structure
  phone = new Phonetable();
  phone->rules.reserve(std::min(numphone, 16384));

  // read and parse each phonetic rule line
  for (int j = 0; j < numphone; j++) {
    std::string nl;
    if (!fileMgr->getline(nl))
      return false;
    mychomp(nl);

    int idx = 0;
    iter = nl.begin();
    start_piece = mystrsep(nl, iter);

    while (start_piece != nl.end()) {
      if (idx == 0) {
        // verify the rule starts with "PHONE"
        if (nl.compare(start_piece - nl.begin(), 5, "PHONE", 5) != 0) {
          HUNSPELL_WARNING(stderr, "error: line %d: table is corrupt\n",
                           fileMgr->getlinenum());
          return false;
        }
      } else if (idx == 1) {
        // store the rule after stripping underscores
        std::string rule(start_piece, iter);
        rule.erase(std::remove(rule.begin(), rule.end(), '_'), rule.end());
        phone->rules.push_back(rule);
      }
      ++idx;
      start_piece = mystrsep(nl, iter);
    }
  }

  // verify we have the expected number of rules
  if (phone->rules.size() != static_cast<size_t>(numphone)) {
    HUNSPELL_WARNING(stderr, "error: line %d: phonetable incomplete\n",
                     fileMgr->getlinenum());
    return false;
  }

  // finalize the phonetable by adding empty entries and initializing
  phone->rules.push_back("");
  phone->rules.push_back("");
  phone->init_hash();

  return true;
}
