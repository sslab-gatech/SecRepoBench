phone = new phonetable();
  if (!phone) {
    HUNSPELL_WARNING(stderr, "error: line %d: can't allocate memory for phone\n",
                     fileMgr->getlinenum());
    return false;
  }
  
  int numents = 0;
  int np = 0;
  int i = 0;
  
  // Parse the first line to get the number of phonetic entries
  std::string::const_iterator iter = line.begin();
  std::string::const_iterator start_piece = mystrsep(line, iter);
  while (start_piece != line.end()) {
    switch (i) {
      case 0: {
        np++;
        break;
      }
      case 1: {
        numents = atoi(std::string(start_piece, iter).c_str());
        if (numents < 1) {
          HUNSPELL_WARNING(stderr, "error: line %d: bad entry number\n",
                           fileMgr->getlinenum());
          delete phone;
          phone = nullptr;
          return false;
        }
        np++;
        break;
      }
      default:
        break;
    }
    ++i;
    start_piece = mystrsep(line, iter);
  }
  
  if (np != 2) {
    HUNSPELL_WARNING(stderr, "error: line %d: missing data\n",
                     fileMgr->getlinenum());
    delete phone;
    phone = nullptr;
    return false;
  }
  
  // Allocate memory for phone table entries
  phone->num = numents;
  phone->phonets = new char*[numents];
  phone->pindex = new char*[numents];
  phone->plength = new int[numents];
  
  if (!phone->phonets || !phone->pindex || !phone->plength) {
    HUNSPELL_WARNING(stderr, "error: line %d: can't allocate memory for phone\n",
                     fileMgr->getlinenum());
    if (phone->phonets) delete[] phone->phonets;
    if (phone->pindex) delete[] phone->pindex;
    if (phone->plength) delete[] phone->plength;
    delete phone;
    phone = nullptr;
    return false;
  }
  
  // Initialize arrays to null/zero for safety
  for (int j = 0; j < numents; j++) {
    phone->phonets[j] = nullptr;
    phone->pindex[j] = nullptr;
    phone->plength[j] = 0;
  }
  
  /* now parse the numents lines to read in the remainder of the table */
  for (int j = 0; j < numents; j++) {
    std::string nl;
    if (!fileMgr->getline(nl)) {
      HUNSPELL_WARNING(stderr, "error: line %d: missing phone data\n",
                       fileMgr->getlinenum());
      for (int k = 0; k < j; k++) {
        free(phone->phonets[k]);
        free(phone->pindex[k]);
      }
      delete[] phone->phonets;
      delete[] phone->pindex;
      delete[] phone->plength;
      delete phone;
      phone = nullptr;
      return false;
    }
    
    mychomp(nl);
    i = 0;
    
    iter = nl.begin();
    start_piece = mystrsep(nl, iter);
    
    std::string rule;
    std::string replace;
    
    while (start_piece != nl.end()) {
      switch (i) {
        case 0: {
          if (nl.compare(start_piece - nl.begin(), 5, "PHONE", 5) != 0) {
            HUNSPELL_WARNING(stderr, "error: line %d: table is corrupt\n",
                             fileMgr->getlinenum());
            for (int k = 0; k < j; k++) {
              free(phone->phonets[k]);
              free(phone->pindex[k]);
            }
            delete[] phone->phonets;
            delete[] phone->pindex;
            delete[] phone->plength;
            delete phone;
            phone = nullptr;
            return false;
          }
          break;
        }
        case 1: {
          rule.assign(start_piece, iter);
          break;
        }
        case 2: {
          replace.assign(start_piece, iter);
          break;
        }
        default:
          break;
      }
      ++i;
      start_piece = mystrsep(nl, iter);
    }
    
    // Validate that we got the expected number of fields
    if (i < 3) {
      HUNSPELL_WARNING(stderr, "error: line %d: missing phone rule data\n",
                       fileMgr->getlinenum());
      for (int k = 0; k < j; k++) {
        free(phone->phonets[k]);
        free(phone->pindex[k]);
      }
      delete[] phone->phonets;
      delete[] phone->pindex;
      delete[] phone->plength;
      delete phone;
      phone = nullptr;
      return false;
    }
    
    // Strip underscores from the rule if present
    size_t pos = 0;
    while ((pos = rule.find('_', pos)) != std::string::npos) {
      rule.erase(pos, 1);
    }
    
    // Allocate memory and store the rule and replacement
    phone->phonets[j] = strdup(rule.c_str());
    phone->pindex[j] = strdup(replace.c_str());
    phone->plength[j] = rule.length();
    
    if (!phone->phonets[j] || !phone->pindex[j]) {
      HUNSPELL_WARNING(stderr, "error: line %d: can't allocate memory for phone\n",
                       fileMgr->getlinenum());
      for (int k = 0; k <= j; k++) {
        if (phone->phonets[k]) free(phone->phonets[k]);
        if (phone->pindex[k]) free(phone->pindex[k]);
      }
      delete[] phone->phonets;
      delete[] phone->pindex;
      delete[] phone->plength;
      delete phone;
      phone = nullptr;
      return false;
    }
  }
  
  // Finalize the phonetic table
  phone->utf8 = utf8;
