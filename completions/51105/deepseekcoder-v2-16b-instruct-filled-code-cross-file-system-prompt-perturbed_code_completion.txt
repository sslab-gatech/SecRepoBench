  // Validate the line to ensure it contains the correct number of entries.
  // Read and process the specified number of phonetic rules from the file.
  // For each rule, ensure it starts with a "PHONE" keyword, then extract and store the rule details.
  // Strip underscores from specific parts of the rules before storing them.
  // Verify the integrity of the rules by checking the expected size after insertion.
  // Finalize the phonetable by adding empty strings and initializing its hash.
  // Return false if any errors or inconsistencies are encountered during parsing.

  // Extract the phonetic table parameters from the line
  std::vector<std::string> phonetable_params = split(line, ' ');
  if (phonetable_params.size() < 3) {
    HUNSPELL_WARNING(stderr, "error: line %d: invalid phonetable format\n", fileMgr->getlinenum());
    return false;
  }

  // Initialize a new phonetable structure
  phone = new Phonetable();

  // Set the properties of the phonetable structure
  phone->num_rules = std::stoi(phonetable_params[0]);
  phone->max_length = std::stoi(phonetable_params[1]);
  phone->vowels = phonetable_params[2];

  // Validate the line to ensure it contains the correct number of entries
  if (phonetable_params.size() != 3 + phone->num_rules * 2) {
    HUNSPELL_WARNING(stderr, "error: line %d: incorrect number of phonetic rules\n", fileMgr->getlinenum());
    delete phone;
    phone = nullptr;
    return false;
  }

  // Read and process the specified number of phonetic rules from the file
  for (int i = 0; i < phone->num_rules; ++i) {
    std::string rule_line;
    if (!fileMgr->getline(rule_line)) {
      HUNSPELL_WARNING(stderr, "error: line %d: premature end of file\n", fileMgr->getlinenum());
      delete phone;
      phone = nullptr;
      return false;
    }

    // Ensure the rule starts with a "PHONE" keyword
    if (rule_line.substr(0, 5) != "PHONE") {
      HUNSPELL_WARNING(stderr, "error: line %d: invalid rule format\n", fileMgr->getlinenum());
      delete phone;
      phone = nullptr;
      return false;
    }

    // Extract and store the rule details
    PhoneticRule rule;
    rule.keyword = "PHONE";
    rule.parameters = split(rule_line.substr(5), ' ');

    // Strip underscores from specific parts of the rules before storing them
    for (auto& param : rule.parameters) {
      std::string::size_type underscore_pos = param.find('_');
      if (underscore_pos != std::string::npos) {
        param.erase(underscore_pos, 1);
      }
    }

    // Verify the integrity of the rules by checking the expected size after insertion
    if (rule.parameters.size() != 2) {
      HUNSPELL_WARNING(stderr, "error: line %d: invalid rule parameters\n", fileMgr->getlinenum());
      delete phone;
      phone = nullptr;
      return false;
    }

    // Add the rule to the phonetable
    phone->rules.push_back(rule);
  }

  // Finalize the phonetable by adding empty strings and initializing its hash
  phone->empty_strings.resize(phone->max_length, "");
  phone->hash_init();
