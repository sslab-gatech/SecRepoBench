int numphone = -1;
int i = 0;
int np = 0;
std::string::const_iterator iter = line.begin();
std::string::const_iterator start_piece = mystrsep(line, iter);

while (start_piece != line.end()) {
    switch (i) {
        case 0: {
            np++;
            break;
        }
        case 1: {
            numphone = atoi(std::string(start_piece, iter).c_str());
            if (numphone < 1) {
                HUNSPELL_WARNING(stderr, "error: line %d: invalid phonetic rule count\n", fileMgr->getlinenum());
                return false;
            }
            phonetable.reserve(std::min(numphone, 16384));
            np++;
            break;
        }
        default:
            break;
    }
    ++i;
    start_piece = mystrsep(line, iter);
}

if (np != 2) {
    HUNSPELL_WARNING(stderr, "error: line %d: missing phonetic table information\n", fileMgr->getlinenum());
    return false;
}

/* Parse the specified number of phonetic rules */
for (int j = 0; j < numphone; ++j) {
    std::string nl;
    if (!fileMgr->getline(nl))
        return false;
    mychomp(nl);
    i = 0;
    iter = nl.begin();
    start_piece = mystrsep(nl, iter);

    while (start_piece != nl.end()) {
        switch (i) {
            case 0: {
                if (nl.compare(start_piece - nl.begin(), 5, "PHONE", 5) != 0) {
                    HUNSPELL_WARNING(stderr, "error: line %d: invalid phonetic rule format\n", fileMgr->getlinenum());
                    return false;
                }
                break;
            }
            case 1: {
                std::string piece(start_piece, iter);
                size_t underscore_pos = piece.find('_');
                if (underscore_pos != std::string::npos) {
                    piece.erase(underscore_pos, 1); // Strip underscore
                }
                phonetable.push_back(piece);
                break;
            }
            default:
                break;
        }
        ++i;
        start_piece = mystrsep(nl, iter);
    }
}

if (phonetable.size() != static_cast<size_t>(numphone)) {
    HUNSPELL_WARNING(stderr, "error: line %d: phonetic table is corrupt or incomplete\n", fileMgr->getlinenum());
    return false;
}

// Finalize phonetable by adding empty strings and initializing its hash
phonetable.push_back("");
init_phonetable_hash();
