switch (FndArgs)
      {
        case 1:
          firstArgIndex = pfx->usedElements - 1;
          if (fe == fIf || fe == fWhile)
          {
            (void) AddAddressingElement(pfx, rIfZeroGoto, 0);
            ndx2 = pfx->usedElements - 1;
          }
          else if (fe == fDo)
          {
            /* firstArgIndex is the statement block for "do" */
          }
          else if (fe == fFor)
          {
            /* firstArgIndex is the initialization for "for" */
          }
          break;

        case 2:
          if (fe == fIf)
          {
            /* after if's "true" block, add unconditional jump over "else" */
            (void) AddAddressingElement(pfx, rGoto, 0);
            ndx3 = pfx->usedElements - 1;
          }
          else if (fe == fWhile)
          {
            /* loop body finished, jump back to condition */
            (void) AddAddressingElement(pfx, rGoto, ndx0);
            ndx3 = pfx->usedElements - 1;
          }
          else if (fe == fDo)
          {
            /* second argument is condition for "do ... while" */
            (void) AddAddressingElement(pfx, rIfZeroGoto, ndx0);
            ndx2 = pfx->usedElements - 1;
          }
          else if (fe == fFor)
          {
            /* second argument is condition for "for" */
            (void) AddAddressingElement(pfx, rIfZeroGoto, 0);
            ndx2 = pfx->usedElements - 1;
          }
          break;

        case 3:
          if (fe == fIf)
          {
            /* third argument is the "else" block, no extra code needed here */
          }
          else if (fe == fFor)
          {
            /* third argument is the increment expression */
            (void) AddAddressingElement(pfx, rGoto, ndx0 + 1);
            ndx3 = pfx->usedElements - 1;
          }
          break;

        default:
          break;
      }
