switch (FndArgs) {
  case 0:
    ndx0 = pfx->usedElements;
    if (fe==fDo) {
      (void) AddAddressingElement (pfx, rGoto, NULL_ADDRESS); /* address will be ndx1+1 */
    }
    break;
  case 1:
    ndx1 = pfx->usedElements;
    if (fe==fFor || fe==fWhile) {
      (void) AddAddressingElement (pfx, rGoto, NULL_ADDRESS); /* address will be ndx2 */
    } else if (fe==fIf) {
      (void) AddAddressingElement (pfx, rIfZeroGoto, NULL_ADDRESS); /* address will be ndx2 */
    }
    break;
  case 2:
    ndx2 = pfx->usedElements;
    if (fe==fFor) {
      (void) AddAddressingElement (pfx, rGoto, NULL_ADDRESS); /* address will be ndx1+1 */
    } else if (fe==fIf) {
      (void) AddAddressingElement (pfx, rGoto, NULL_ADDRESS); /* address will be ndx3 */
    }
    break;
  case 3:
    ndx3 = pfx->usedElements;
    if (fe==fIf) {
      pfx->Elements[ndx1].EleNdx = ndx2;
      pfx->Elements[ndx2].EleNdx = ndx3;
    }
    break;
  default:
    (void) ThrowMagickException (
      pfx->exception, GetMagickModule(), OptionError,
      "Too many args for function", "'%s' found %i, expected %i at '%s'",
      Functions[fe-FirstFunc].str, FndArgs, Functions[fe-FirstFunc].nArgs, SetShortExp(pfx));
    return isCoordQualifier;
}
