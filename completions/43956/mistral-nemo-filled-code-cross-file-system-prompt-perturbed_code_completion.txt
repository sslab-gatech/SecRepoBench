switch (FndArgs) {
  case 1:
    ndx1 = pfx->usedElements;
    if (fe == fIf) {
      (void) AddAddressingElement (pfx, rIfZeroGoto, NULL_ADDRESS); /* address will be ndx2+1 */
    } else if (fe == fWhile) {
      (void) AddAddressingElement (pfx, rGoto, NULL_ADDRESS); /* address will be ndx2+1 */
    }
    break;
  case 2:
    ndx2 = pfx->usedElements;
    if (fe == fIf) {
      (void) AddAddressingElement (pfx, rIfNotZeroGoto, ndx1); /* jump back to ndx1 */
    } else if (fe == fWhile) {
      (void) AddAddressingElement (pfx, rGoto, ndx1); /* jump back to ndx1 */
    }
    break;
  case 3:
    ndx3 = pfx->usedElements;
    if (fe == fFor) {
      (void) AddAddressingElement (pfx, rGoto, ndx2); /* jump back to ndx2 */
    }
    break;
  default:
    break;
}
