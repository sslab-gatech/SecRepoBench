switch (FndArgs)
{
  case 1:
    if (fe == fDo)
    {
      ndx0 = pfx->usedElements - 1;
      (void) AddAddressingElement(pfx, rGoto, NULL_ADDRESS);
      /* address will be firstArgIndex+1 */
    }
    else if ((fe == fFor) || (fe == fIf) || (fe == fWhile))
    {
      firstArgIndex = pfx->usedElements - 1;
      (void) AddAddressingElement(pfx, rIfZeroGoto, NULL_ADDRESS);
      /* address will be ndx2+1 */
    }
    break;

  case 2:
    if ((fe == fIf) || (fe == fWhile))
    {
      ndx2 = pfx->usedElements - 1;
      (void) AddAddressingElement(pfx, rGoto, NULL_ADDRESS);
      /* address will be ndx3 (if fIf) or the next statement (if fWhile) */
    }
    else if (fe == fDo)
    {
      firstArgIndex = pfx->usedElements - 1;
      (void) AddAddressingElement(pfx, rIfZeroGoto, NULL_ADDRESS);
      /* address will be ndx2+1 */
    }
    else if (fe == fFor)
    {
      ndx0 = pfx->usedElements - 1;
      (void) AddAddressingElement(pfx, rIfZeroGoto, NULL_ADDRESS);
      /* address will be ndx2+1 for next argument */
    }
    break;

  case 3:
    if (fe == fIf)
    {
      ndx3 = pfx->usedElements - 1;
      /* the jump addresses are fixed after the loop */
    }
    else if (fe == fFor)
    {
      ndx2 = pfx->usedElements - 1;
      (void) AddAddressingElement(pfx, rGoto, ndx0);
      /* loop back to initialization after increment */
    }
    break;

  default:
    break;
}
