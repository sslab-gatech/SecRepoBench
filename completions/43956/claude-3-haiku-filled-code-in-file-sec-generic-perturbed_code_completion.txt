switch (FndArgs) {
  case 0:
    ndx0 = pfx->usedElements;
    if (fe == fDo) {
      (void) AddAddressingElement (pfx, rGoto, NULL_ADDRESS); /* address will be ndx1+1 */
    }
    break;
  case 1:
    ndx1 = pfx->usedElements;
    if (fe == fDo) {
      (void) AddAddressingElement (pfx, rIfNotZeroGoto, NULL_ADDRESS); /* address will be ndx2+1 */
    } else if (fe == fFor) {
      (void) AddAddressingElement (pfx, rGoto, NULL_ADDRESS); /* address will be ndx3 */
    } else if (fe == fIf) {
      (void) AddAddressingElement (pfx, rIfZeroGoto, NULL_ADDRESS); /* address will be ndx2+1 */
    } else if (fe == fWhile) {
      (void) AddAddressingElement (pfx, rIfZeroGoto, NULL_ADDRESS); /* address will be ndx2+1 */
    }
    break;
  case 2:
    ndx2 = pfx->usedElements;
    if (fe == fFor) {
      (void) AddAddressingElement (pfx, rGoto, NULL_ADDRESS); /* address will be ndx1+1 */
    } else if (fe == fIf) {
      (void) AddAddressingElement (pfx, rGoto, NULL_ADDRESS); /* address will be ndx3 */
    }
    break;
  case 3:
    ndx3 = pfx->usedElements;
    if (fe == fFor) {
      (void) AddAddressingElement (pfx, rGoto, ndx1); /* address will be ndx1+1 */
    }
    break;

    (void) ThrowMagickException (
      pfx->exception, GetMagickModule(), OptionError,
      "Too many args for function", "'%s' found %i at '%s'",
      Functions[fe-FirstFunc].str, FndArgs, SetShortExp(pfx));
