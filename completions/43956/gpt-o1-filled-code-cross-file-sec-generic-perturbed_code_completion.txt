switch(FndArgs)
{
  case 1:
    /* Handle single-argument cases for flow-control functions */
    if ((fe == fWhile) || (fe == fIf) || (fe == fFor))
    {
      if (!AddAddressingElement(pfx, rIfZeroGoto, NULL_ADDRESS))
        return MagickFalse;
      firstArgIndex = pfx->usedElements - 1;
    }
    /* 'do' has its jump set up before the loop, so nothing here */
    break;

  case 2:
    if (fe == fWhile)
    {
      /* After the body, jump back to condition */
      if (!AddAddressingElement(pfx, rGoto, ndx0))
        return MagickFalse;
      ndx2 = pfx->usedElements - 1;
    }
    else if (fe == fIf)
    {
      /* If condition is true, skip to the next argument or end */
      if (!AddAddressingElement(pfx, rGoto, NULL_ADDRESS))
        return MagickFalse;
      ndx2 = pfx->usedElements - 1;
    }
    else if (fe == fDo)
    {
      /* Second argument is the condition for do..while */
      if (!AddAddressingElement(pfx, rIfZeroGoto, NULL_ADDRESS))
        return MagickFalse;
      firstArgIndex = pfx->usedElements - 1;
      ndx2 = firstArgIndex;
    }
    else if (fe == fFor)
    {
      /* Second argument is the condition check */
      if (!AddAddressingElement(pfx, rIfZeroGoto, NULL_ADDRESS))
        return MagickFalse;
      ndx2 = pfx->usedElements - 1;
    }
    break;

  case 3:
    /* Some functions (if/for) have an optional third argument */
    if ((fe == fIf) || (fe == fFor))
    {
      ndx3 = pfx->usedElements;
    }
    break;

  default:
    break;
}
