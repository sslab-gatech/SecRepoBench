switch (FndArgs) {
        case 1:
          /*
            For "if" or "while", the first argument is the condition. We insert an
            rIfZeroGoto placeholder to be fixed up later. For "do", the first argument
            is the loop body, so we just record its index. For "for", the first argument
            is the initialization expression.
          */
          if (fe == fIf || fe == fWhile)
            {
              firstArgIndex = pfx->usedElements - 1;
              (void) AddAddressingElement(pfx, rIfZeroGoto, 0);
              ndx2 = pfx->usedElements - 1;
            }
          else if (fe == fDo)
            {
              firstArgIndex = pfx->usedElements - 1;
            }
          else if (fe == fFor)
            {
              firstArgIndex = pfx->usedElements - 1;
            }
          break;

        case 2:
          /*
            For "if", once we have the condition and first expression, we add a jump
            (rGoto) to skip the optional else-part. For "while", we insert a jump back
            to the start (ndx0) after the body. For "do", this second argument is the
            condition, so we add rIfZeroGoto. For "for", the second argument holds the
            loop condition, so insert rIfZeroGoto.
          */
          if (fe == fIf)
            {
              (void) AddAddressingElement(pfx, rGoto, 0);
              ndx3 = pfx->usedElements - 1;
            }
          else if (fe == fWhile)
            {
              (void) AddAddressingElement(pfx, rGoto, ndx0);
              ndx3 = pfx->usedElements - 1;
            }
          else if (fe == fDo)
            {
              (void) AddAddressingElement(pfx, rIfZeroGoto, 0);
              ndx2 = pfx->usedElements - 1;
            }
          else if (fe == fFor)
            {
              (void) AddAddressingElement(pfx, rIfZeroGoto, 0);
              ndx2 = pfx->usedElements - 1;
            }
          break;

        case 3:
          /*
            For "if" with three arguments, we already have the jump placeholders inserted.
            For "for", the third argument is the increment expression, so we add a jump
            back to the condition.
          */
          if (fe == fIf)
            {
              /* The else part was just parsed; nothing extra here. */
            }
          else if (fe == fFor)
            {
              (void) AddAddressingElement(pfx, rGoto, firstArgIndex + 1);
              ndx3 = pfx->usedElements - 1;
            }
          break;

        default:
          break;
      }
