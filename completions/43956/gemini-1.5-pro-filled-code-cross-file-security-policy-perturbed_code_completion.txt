case 1:
        firstArgIndex = pfx->usedElements-1;
        if (fe==fFor) {
          (void) AddAddressingElement (pfx, rGoto, NULL_ADDRESS); /* address will be ndx2 */
          ndx2 = pfx->usedElements-1;
          (void) AddAddressingElement (pfx, rIfZeroGoto, NULL_ADDRESS); /* address will be outside the loop */
          ndx3 = pfx->usedElements-1;
        } else if (fe==fIf) {
          (void) AddAddressingElement (pfx, rIfZeroGoto, NULL_ADDRESS); /* address will be ndx2 */
          ndx2 = pfx->usedElements-1;
        } else if (fe==fWhile) {
          /* While is really a do-while: we push a goto to the condition at the end. */
          (void) AddAddressingElement (pfx, rGoto, NULL_ADDRESS); /* address will be ndx2 */
          ndx2 = pfx->usedElements-1;
        }
        break;
      case 2:
        ndx2 = pfx->usedElements-1;
        if (fe==fFor) {
          (void) AddAddressingElement (pfx, rGoto, firstArgIndex+1);
        } else if (fe==fDo) {
          (void) AddAddressingElement (pfx, rIfNotZeroGoto, NULL_ADDRESS);
        }
        break;
      case 3:
        ndx3 = pfx->usedElements-1;
        if (fe==fFor) {
          (void) AddAddressingElement (pfx, rGoto, ndx0+1);
          pfx->Elements[ndx3].EleNdx = pfx->usedElements; /* outside the loop */
        }
        break;
