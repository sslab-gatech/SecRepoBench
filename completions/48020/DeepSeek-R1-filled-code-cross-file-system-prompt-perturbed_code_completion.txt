else if(is_ndpi_proto(flow_info, NDPI_PROTOCOL_SSH)) {
    flow_info->info_type = INFO_SSH;
    ndpi_snprintf(flow_info->ssh.client_signature, sizeof(flow_info->ssh.client_signature), "%s", 
                  flow_info->ndpi_flow->protos.ssh.client_signature);
    ndpi_snprintf(flow_info->ssh.server_signature, sizeof(flow_info->ssh.server_signature), "%s", 
                  flow_info->ndpi_flow->protos.ssh.server_signature);
    if(flow_info->ndpi_flow->protos.ssh.client_hassh[0] != '\0')
        flow_info->ssh.client_hassh = ndpi_strdup(flow_info->ndpi_flow->protos.ssh.client_hassh);
    if(flow_info->ndpi_flow->protos.ssh.server_hassh[0] != '\0')
        flow_info->ssh.server_hassh = ndpi_strdup(flow_info->ndpi_flow->protos.ssh.server_hassh);
} else if((is_ndpi_proto(flow_info, NDPI_PROTOCOL_TLS) || 
          is_ndpi_proto(flow_info, NDPI_PROTOCOL_QUIC) ||
          flow_info->detected_protocol.master_protocol == NDPI_PROTOCOL_TLS)) {
    flow_info->info_type = INFO_TLS;
    ndpi_snprintf(flow_info->tls_quic.ssl_version, sizeof(flow_info->tls_quic.ssl_version), "%s",
                 ndpi_ssl_version2str(flow_info->ndpi_flow->protos.tls_quic.ssl_version, 1));
    ndpi_snprintf(flow_info->tls_quic.client_requested_server_name, sizeof(flow_info->tls_quic.client_requested_server_name),
                 "%s", flow_info->ndpi_flow->protos.tls_quic.client_requested_server_name);
    ndpi_snprintf(flow_info->tls_quic.server_names, sizeof(flow_info->tls_quic.server_names),
                 "%s", flow_info->ndpi_flow->protos.tls_quic.server_names);
    ndpi_snprintf(flow_info->tls_quic.notBefore, sizeof(flow_info->tls_quic.notBefore), "%s",
                 flow_info->ndpi_flow->protos.tls_quic.notBefore);
    ndpi_snprintf(flow_info->tls_quic.notAfter, sizeof(flow_info->tls_quic.notAfter), "%s",
                 flow_info->ndpi_flow->protos.tls_quic.notAfter);
    if(flow_info->ndpi_flow->protos.tls_quic.ja3_client[0] != '\0')
        flow_info->tls_quic.ja3_client = ndpi_strdup(flow_info->ndpi_flow->protos.tls_quic.ja3_client);
    if(flow_info->ndpi_flow->protos.tls_quic.ja3_server[0] != '\0')
        flow_info->tls_quic.ja3_server = ndpi_strdup(flow_info->ndpi_flow->protos.tls_quic.ja3_server);
    ndpi_snprintf(flow_info->tls_quic.alpn, sizeof(flow_info->tls_quic.alpn), "%s",
                 flow_info->ndpi_flow->protos.tls_quic.alpn ? flow_info->ndpi_flow->protos.tls_quic.alpn : "");
    ndpi_snprintf(flow_info->tls_quic.tls_supported_versions, sizeof(flow_info->tls_quic.tls_supported_versions), "%s",
                 flow_info->ndpi_flow->protos.tls_quic.tls_supported_versions ? flow_info->ndpi_flow->protos.tls_quic.tls_supported_versions : "");
    if(flow_info->ndpi_flow->protos.tls_quic.alpn != NULL) {
        flow_info->info_type = INFO_TLS_ALPN;
        ndpi_snprintf(flow_info->info, sizeof(flow_info->info), "%s", flow_info->ndpi_flow->protos.tls_quic.alpn);
    }
    if(enable_payload_analyzer) {
        ndpi_reset_payload_len_stats(&flow_info->payload_len);
        ndpi_update_payload_len_stats(&flow_info->payload_len, flow_info->ndpi_flow->protos.tls_quic.tls_block_len);
    }
}
