else if(is_ndpi_proto(flow_info, NDPI_PROTOCOL_SSH)) {
    flow_info->info_type = INFO_SSH;
    ndpi_snprintf(flow_info->ssh.client_signature, sizeof(flow_info->ssh.client_signature),
                  "%s", flow_info->ndpi_flow->protos.ssh.client_signature);
    ndpi_snprintf(flow_info->ssh.server_signature, sizeof(flow_info->ssh.server_signature),
                  "%s", flow_info->ndpi_flow->protos.ssh.server_signature);
    flow_info->ssh.client_hash = flow_info->ndpi_flow->protos.ssh.client_hash;
    flow_info->ssh.server_hash = flow_info->ndpi_flow->protos.ssh.server_hash;
  } else if(is_ndpi_proto(flow_info, NDPI_PROTOCOL_TLS) || is_ndpi_proto(flow_info, NDPI_PROTOCOL_QUIC)) {
    flow_info->info_type = INFO_TLS_QUIC;
    ndpi_snprintf(flow_info->tls.version, sizeof(flow_info->tls.version),
                  "%s", flow_info->ndpi_flow->tls.version);
    ndpi_snprintf(flow_info->tls.server_name, sizeof(flow_info->tls.server_name),
                  "%s", flow_info->ndpi_flow->tls.server_name);
    flow_info->tls.cert_validity_start = flow_info->ndpi_flow->tls.cert_validity_start;
    flow_info->tls.cert_validity_end = flow_info->ndpi_flow->tls.cert_validity_end;
    if(flow_info->ndpi_flow->tls.ja3_fingerprint[0] != '\0') {
      flow_info->tls.ja3_fingerprint = ndpi_strdup(flow_info->ndpi_flow->tls.ja3_fingerprint);
    }
    if(flow_info->ndpi_flow->tls.alpn[0] != '\0') {
      flow_info->tls.alpn = ndpi_strdup(flow_info->ndpi_flow->tls.alpn);
    }
