else if(is_ndpi_proto(flow_info, NDPI_PROTOCOL_SSH) ||
	  is_ndpi_proto(flow_info, NDPI_PROTOCOL_TLS) ||
	  is_ndpi_proto(flow_info, NDPI_PROTOCOL_QUIC) ||
	  (flow_info->detected_protocol.master_protocol == NDPI_PROTOCOL_TLS) ||
	  (flow_info->detected_protocol.master_protocol == NDPI_PROTOCOL_SSH) ||
	  (flow_info->detected_protocol.master_protocol == NDPI_PROTOCOL_QUIC)) {
    flow_info->info_type = INFO_TLS_SSH_QUIC;
    if(flow_info->ndpi_flow->tls.server_names[0] != '\0')
      flow_info->ssh_tls.server_names = ndpi_strdup(flow_info->ndpi_flow->tls.server_names);
    if(flow_info->ndpi_flow->tls.tls_alpn[0] != '\0')
      flow_info->ssh_tls.tls_alpn = ndpi_strdup(flow_info->ndpi_flow->tls.tls_alpn);
    if(flow_info->ndpi_flow->tls.tls_supported_versions[0] != '\0')
      flow_info->ssh_tls.tls_supported_versions = ndpi_strdup(flow_info->ndpi_flow->tls.tls_supported_versions);
    if(flow_info->ndpi_flow->tls.tls_issuerDN[0] != '\0')
      flow_info->ssh_tls.tls_issuerDN = ndpi_strdup(flow_info->ndpi_flow->tls.tls_issuerDN);
    if(flow_info->ndpi_flow->tls.tls_subjectDN[0] != '\0')
      flow_info->ssh_tls.tls_subjectDN = ndpi_strdup(flow_info->ndpi_flow->tls.tls_subjectDN);
    if(flow_info->ndpi_flow->tls.ja3_fingerprint[0] != '\0')
      flow_info->ssh_tls.ja3_fingerprint = ndpi_strdup(flow_info->ndpi_flow->tls.ja3_fingerprint);
    if(flow_info->ndpi_flow->tls.encrypted_sni.esni[0] != '\0') {
      flow_info->ssh_tls.encrypted_sni.esni = ndpi_strdup(flow_info->ndpi_flow->tls.encrypted_sni.esni);
      flow_info->ssh_tls.encrypted_sni.esni_len = flow_info->ndpi_flow->tls.encrypted_sni.esni_len;
    }
    flow_info->ssh_tls.tls_version = flow_info->ndpi_flow->tls.tls_version;
    flow_info->ssh_tls.quic_version = flow_info->ndpi_flow->tls.quic_version;
    flow_info->ssh_tls.tls_cipher_suite = flow_info->ndpi_flow->tls.tls_cipher_suite;
    flow_info->ssh_tls.tls_client_random = flow_info->ndpi_flow->tls.tls_client_random;
    flow_info->ssh_tls.tls_server_random = flow_info->ndpi_flow->tls.tls_server_random;
    flow_info->ssh_tls.tls_client_cert_sha256 = flow_info->ndpi_flow->tls.tls_client_cert_sha256;
    flow_info->ssh_tls.tls_server_cert_sha256 = flow_info->ndpi_flow->tls.tls_server_cert_sha256;
    flow_info->ssh_tls.tls_client_cert_sha1 = flow_info->ndpi_flow->tls.tls_client_cert_sha1;
    flow_info->ssh_tls.tls_server_cert_sha1 = flow_info->ndpi_flow->tls.tls_server_cert_sha1;
    flow_info->ssh_tls.tls_client_cert_md5 = flow_info->ndpi_flow->tls.tls_client_cert_md5;
    flow_info->ssh_tls.tls_server_cert_md5 = flow_info->ndpi_flow->tls.tls_server_cert_md5;
    if(flow_info->ssh_tls.tls_alpn[0] != '\0' || flow_info->ssh_tls.tls_supported_versions[0] != '\0') {
      flow_info->info_type = INFO_TLS_SSH_QUIC;
    }
  }
