else if(is_ndpi_proto(flow_info, NDPI_PROTOCOL_SSH)) {
    flow_info->info_type = INFO_SSH;
    if(flow_info->ndpi_flow->protos.ssh.client_signature[0] != '\0')
      ndpi_snprintf(flow_info->ssh.client_signature, sizeof(flow_info->ssh.client_signature), "%s",
                    flow_info->ndpi_flow->protos.ssh.client_signature);
    if(flow_info->ndpi_flow->protos.ssh.server_signature[0] != '\0')
      ndpi_snprintf(flow_info->ssh.server_signature, sizeof(flow_info->ssh.server_signature), "%s",
                    flow_info->ndpi_flow->protos.ssh.server_signature);
    if(flow_info->ndpi_flow->protos.ssh.hassh_client_hash[0] != '\0')
      flow_info->ssh.client_hash = ndpi_strdup(flow_info->ndpi_flow->protos.ssh.hassh_client_hash);
    if(flow_info->ndpi_flow->protos.ssh.hassh_server_hash[0] != '\0')
      flow_info->ssh.server_hash = ndpi_strdup(flow_info->ndpi_flow->protos.ssh.hassh_server_hash);
  } else if(is_ndpi_proto(flow_info, NDPI_PROTOCOL_TLS) ||
            is_ndpi_proto(flow_info, NDPI_PROTOCOL_QUIC)) {
    flow_info->info_type = INFO_TLS;
    ndpi_snprintf(flow_info->tls.ssl_version, sizeof(flow_info->tls.ssl_version), "%s",
                  flow_info->ndpi_flow->protos.tls_quic.ssl_version ? flow_info->ndpi_flow->protos.tls_quic.ssl_version : "");
    ndpi_snprintf(flow_info->tls.server_names, sizeof(flow_info->tls.server_names), "%s",
                  flow_info->ndpi_flow->protos.tls_quic.server_names ? flow_info->ndpi_flow->protos.tls_quic.server_names : "");
    ndpi_snprintf(flow_info->tls.issuerDN, sizeof(flow_info->tls.issuerDN), "%s",
                  flow_info->ndpi_flow->protos.tls_quic.issuerDN ? flow_info->ndpi_flow->protos.tls_quic.issuerDN : "");
    ndpi_snprintf(flow_info->tls.subjectDN, sizeof(flow_info->tls.subjectDN), "%s",
                  flow_info->ndpi_flow->protos.tls_quic.subjectDN ? flow_info->ndpi_flow->protos.tls_quic.subjectDN : "");
    flow_info->tls.not_before = flow_info->ndpi_flow->protos.tls_quic.not_before;
    flow_info->tls.not_after = flow_info->ndpi_flow->protos.tls_quic.not_after;
    if(flow_info->ndpi_flow->protos.tls_quic.ja3_client[0] != '\0')
      flow_info->tls.ja3_client = ndpi_strdup(flow_info->ndpi_flow->protos.tls_quic.ja3_client);
    if(flow_info->ndpi_flow->protos.tls_quic.ja3_server[0] != '\0')
      flow_info->tls.ja3_server = ndpi_strdup(flow_info->ndpi_flow->protos.tls_quic.ja3_server);
    if(flow_info->ndpi_flow->protos.tls_quic.alpn != NULL) {
      ndpi_snprintf(flow_info->tls.alpn, sizeof(flow_info->tls.alpn), "%s",
                    flow_info->ndpi_flow->protos.tls_quic.alpn);
      flow_info->info_type = INFO_TLS_ALPN;
    }
    if(flow_info->ndpi_flow->protos.tls_quic.tls_supported_versions != NULL) {
      ndpi_snprintf(flow_info->tls.tls_supported_versions, sizeof(flow_info->tls.tls_supported_versions), "%s",
                    flow_info->ndpi_flow->protos.tls_quic.tls_supported_versions);
      flow_info->info_type = INFO_TLS_VERSION;
    }
    ndpi_snprintf(flow_info->tls.client_requested_server_name, sizeof(flow_info->tls.client_requested_server_name), "%s",
                  flow_info->ndpi_flow->host_server_name);
  }
