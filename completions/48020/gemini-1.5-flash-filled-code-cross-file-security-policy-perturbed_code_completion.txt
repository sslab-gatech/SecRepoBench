else if(is_ndpi_proto(flow_info, NDPI_PROTOCOL_SSH)) {
    flow_info->info_type = INFO_SSH;
    ndpi_snprintf(flow_info->ssh.client_signature, sizeof(flow_info->ssh.client_signature), "%s", flow_info->ndpi_flow->protos.ssh.client_signature);
    ndpi_snprintf(flow_info->ssh.server_signature, sizeof(flow_info->ssh.server_signature), "%s", flow_info->ndpi_flow->protos.ssh.server_signature);
    ndpi_snprintf(flow_info->ssh.client_hash, sizeof(flow_info->ssh.client_hash), "%s", flow_info->ndpi_flow->protos.ssh.client_hash);
    ndpi_snprintf(flow_info->ssh.server_hash, sizeof(flow_info->ssh.server_hash), "%s", flow_info->ndpi_flow->protos.ssh.server_hash);
  } else if(is_ndpi_proto(flow_info, NDPI_PROTOCOL_TLS) || is_ndpi_proto(flow_info, NDPI_PROTOCOL_QUIC) || is_ndpi_proto(flow_info, NDPI_PROTOCOL_DTLS)) {
    flow_info->info_type = INFO_TLS;
    flow_info->tls.ssl_version = flow_info->ndpi_flow->tls.ssl_version;
    ndpi_snprintf(flow_info->tls.server_name, sizeof(flow_info->tls.server_name), "%s", flow_info->ndpi_flow->tls.server_name);
    ndpi_snprintf(flow_info->tls.ja3_fingerprint, sizeof(flow_info->tls.ja3_fingerprint), "%s", flow_info->ndpi_flow->tls.ja3_fingerprint);
    ndpi_snprintf(flow_info->tls.alpn, sizeof(flow_info->tls.alpn), "%s", flow_info->ndpi_flow->tls.alpn);
    if (flow_info->ndpi_flow->tls.cert_not_before != 0) {
      flow_info->tls.cert_not_before = flow_info->ndpi_flow->tls.cert_not_before;
    }
    if (flow_info->ndpi_flow->tls.cert_not_after != 0) {
      flow_info->tls.cert_not_after = flow_info->ndpi_flow->tls.cert_not_after;
    }
    if (flow_info->ndpi_flow->tls.supported_versions != NULL) {
      size_t len = strlen(flow_info->ndpi_flow->tls.supported_versions);
      if (len < sizeof(flow_info->tls.supported_versions)) {
        strcpy(flow_info->tls.supported_versions, flow_info->ndpi_flow->tls.supported_versions);
      }
    }
    if (flow_info->ndpi_flow->tls.alpn != NULL && strlen(flow_info->ndpi_flow->tls.alpn) > 0) {
      flow_info->info_type = INFO_TLS_ALPN;
    }
    if (flow_info->ndpi_flow->tls.ja3_fingerprint != NULL && strlen(flow_info->ndpi_flow->tls.ja3_fingerprint) > 0) {
      flow_info->info_type = INFO_TLS_JA3;
    }
    is_quic = is_ndpi_proto(flow_info, NDPI_PROTOCOL_QUIC);
    if (is_quic) {
      flow_info->quic.version = flow_info->ndpi_flow->quic.version;
      ndpi_snprintf(flow_info->quic.fingerprint, sizeof(flow_info->quic.fingerprint), "%s", flow_info->ndpi_flow->quic.fingerprint);
    }
  }
