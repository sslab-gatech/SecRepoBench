else if(is_ndpi_proto(flow_info, NDPI_PROTOCOL_SSH)) {
    flow_info->info_type = INFO_SSH;
    ndpi_snprintf(flow_info->ssh.client_signature, sizeof(flow_info->ssh.client_signature), "%s", flow_info->ndpi_flow->protos.ssh.client_signature);
    ndpi_snprintf(flow_info->ssh.server_signature, sizeof(flow_info->ssh.server_signature), "%s", flow_info->ndpi_flow->protos.ssh.server_signature);
    ndpi_snprintf(flow_info->ssh.client_hash, sizeof(flow_info->ssh.client_hash), "%s", flow_info->ndpi_flow->protos.ssh.client_hash);
    ndpi_snprintf(flow_info->ssh.server_hash, sizeof(flow_info->ssh.server_hash), "%s", flow_info->ndpi_flow->protos.ssh.server_hash);
  } else if(is_ndpi_proto(flow_info, NDPI_PROTOCOL_TLS) || is_ndpi_proto(flow_info, NDPI_PROTOCOL_QUIC)) {
    flow_info->info_type = INFO_TLS_QUIC;
    flow_info->tls_quic.ssl_version = flow_info->ndpi_flow->tls.ssl_version;
    ndpi_snprintf(flow_info->tls_quic.server_name, sizeof(flow_info->tls_quic.server_name), "%s", flow_info->ndpi_flow->tls.server_name);
    ndpi_snprintf(flow_info->tls_quic.ja3_fingerprint, sizeof(flow_info->tls_quic.ja3_fingerprint), "%s", flow_info->ndpi_flow->tls.ja3_fingerprint);
    flow_info->tls_quic.cert_valid_from = flow_info->ndpi_flow->tls.cert_valid_from;
    flow_info->tls_quic.cert_valid_until = flow_info->ndpi_flow->tls.cert_valid_until;
    ndpi_snprintf(flow_info->tls_quic.alpn, sizeof(flow_info->tls_quic.alpn), "%s", flow_info->ndpi_flow->tls.alpn);
    ndpi_snprintf(flow_info->tls_quic.supported_versions, sizeof(flow_info->tls_quic.supported_versions), "%s", flow_info->ndpi_flow->tls.supported_versions);
    if (flow_info->tls_quic.alpn[0] != '\0' || flow_info->tls_quic.supported_versions[0] != '\0') {
      flow_info->info_type = INFO_TLS_QUIC_DETAILS;
    }
