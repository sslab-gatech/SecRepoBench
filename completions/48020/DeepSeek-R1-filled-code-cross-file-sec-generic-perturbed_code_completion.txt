} else if(is_ndpi_proto(flow_info, NDPI_PROTOCOL_SSH)) {
    flow_info->info_type = INFO_SSH;
    if(flow_info->ndpi_flow->protos.ssh.client_signature[0] != '\0')
      flow_info->ssh.client_signature = ndpi_strdup(flow_info->ndpi_flow->protos.ssh.client_signature);
    if(flow_info->ndpi_flow->protos.ssh.server_signature[0] != '\0')
      flow_info->ssh.server_signature = ndpi_strdup(flow_info->ndpi_flow->protos.ssh.server_signature);
    if(flow_info->ndpi_flow->protos.ssh.hassh_client[0] != '\0')
      flow_info->ssh.hassh_client = ndpi_strdup(flow_info->ndpi_flow->protos.ssh.hassh_client);
    if(flow_info->ndpi_flow->protos.ssh.hassh_server[0] != '\0')
      flow_info->ssh.hassh_server = ndpi_strdup(flow_info->ndpi_flow->protos.ssh.hassh_server);
  } else if((flow_info->detected_protocol.master_protocol == NDPI_PROTOCOL_TLS) ||
            is_ndpi_proto(flow_info, NDPI_PROTOCOL_TLS) ||
            is_ndpi_proto(flow_info, NDPI_PROTOCOL_DTLS) ||
            is_ndpi_proto(flow_info, NDPI_PROTOCOL_QUIC)) {
    flow_info->info_type = INFO_TLS;
    ndpi_snprintf(flow_info->tls.version, sizeof(flow_info->tls.version), "%s",
                 flow_info->ndpi_flow->protos.ssl.version ? flow_info->ndpi_flow->protos.ssl.version : "");
    ndpi_snprintf(flow_info->tls.server_names, sizeof(flow_info->tls.server_names), "%s",
                 flow_info->ndpi_flow->protos.ssl.server_names ? flow_info->ndpi_flow->protos.ssl.server_names : "");
    ndpi_snprintf(flow_info->tls.ja3_client, sizeof(flow_info->tls.ja3_client), "%s",
                 flow_info->ndpi_flow->protos.ssl.ja3_client ? flow_info->ndpi_flow->protos.ssl.ja3_client : "");
    ndpi_snprintf(flow_info->tls.ja3_server, sizeof(flow_info->tls.ja3_server), "%s",
                 flow_info->ndpi_flow->protos.ssl.ja3_server ? flow_info->ndpi_flow->protos.ssl.ja3_server : "");
    ndpi_snprintf(flow_info->tls.issuerDN, sizeof(flow_info->tls.issuerDN), "%s",
                 flow_info->ndpi_flow->protos.ssl.issuerDN ? flow_info->ndpi_flow->protos.ssl.issuerDN : "");
    ndpi_snprintf(flow_info->tls.subjectDN, sizeof(flow_info->tls.subjectDN), "%s",
                 flow_info->ndpi_flow->protos.ssl.subjectDN ? flow_info->ndpi_flow->protos.ssl.subjectDN : "");
    flow_info->tls.not_before = flow_info->ndpi_flow->protos.ssl.not_before;
    flow_info->tls.not_after = flow_info->ndpi_flow->protos.ssl.not_after;

    if(flow_info->ndpi_flow->protos.ssl.alpn != NULL) {
      ndpi_snprintf(flow_info->tls.alpn, sizeof(flow_info->tls.alpn), "%s", flow_info->ndpi_flow->protos.ssl.alpn);
      flow_info->info_type = INFO_TLS_ALPN;
    }
    if(flow_info->ndpi_flow->protos.ssl.tls_supported_versions != NULL) {
      ndpi_snprintf(flow_info->tls.supported_versions, sizeof(flow_info->tls.supported_versions), "%s",
                   flow_info->ndpi_flow->protos.ssl.tls_supported_versions);
    }
    if(flow_info->ndpi_flow->protos.ssl.tls_block_length) {
      memcpy(flow_info->payload_len_distribution, flow_info->ndpi_flow->protos.ssl.tls_block_length,
            sizeof(flow_info->payload_len_distribution));
    }
