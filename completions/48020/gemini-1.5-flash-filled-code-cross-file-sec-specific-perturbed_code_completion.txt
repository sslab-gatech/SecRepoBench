  } else if(is_ndpi_proto(flow_info, NDPI_PROTOCOL_TLS) || is_ndpi_proto(flow_info, NDPI_PROTOCOL_QUIC) || is_ndpi_proto(flow_info, NDPI_PROTOCOL_DTLS)) {

    if (flow_info->ndpi_flow->protos.tls.alpn[0] != '\0' && flow_info->ndpi_flow->protos.tls.ssl_version[0] != '\0') {
      flow_info->info_type = INFO_ALPN;
