else if(is_ndpi_proto(flow_info, NDPI_PROTOCOL_TLS) ||
	  is_ndpi_proto(flow_info, NDPI_PROTOCOL_QUIC) ||
	  is_ndpi_proto(flow_info, NDPI_PROTOCOL_SSH)) {
    if(flow_info->ndpi_flow->l4.tcp.tls.server_names[0] != '\0')
      flow_info->ssh_tls.server_names = ndpi_strdup(flow_info->ndpi_flow->l4.tcp.tls.server_names);
    if(flow_info->ndpi_flow->l4.tcp.tls.tls_alpn[0] != '\0')
      flow_info->ssh_tls.tls_alpn = ndpi_strdup(flow_info->ndpi_flow->l4.tcp.tls.tls_alpn);
    if(flow_info->ndpi_flow->l4.tcp.tls.tls_supported_versions[0] != '\0')
      flow_info->ssh_tls.tls_supported_versions = ndpi_strdup(flow_info->ndpi_flow->l4.tcp.tls.tls_supported_versions);
    if(flow_info->ndpi_flow->l4.tcp.tls.tls_issuerDN[0] != '\0')
      flow_info->ssh_tls.tls_issuerDN = ndpi_strdup(flow_info->ndpi_flow->l4.tcp.tls.tls_issuerDN);
    if(flow_info->ndpi_flow->l4.tcp.tls.tls_subjectDN[0] != '\0')
      flow_info->ssh_tls.tls_subjectDN = ndpi_strdup(flow_info->ndpi_flow->l4.tcp.tls.tls_subjectDN);
    if(flow_info->ndpi_flow->l4.tcp.tls.ja3_fingerprint[0] != '\0')
      flow_info->ssh_tls.ja3_fingerprint = ndpi_strdup(flow_info->ndpi_flow->l4.tcp.tls.ja3_fingerprint);
    if(flow_info->ndpi_flow->l4.tcp.tls.encrypted_sni.esni[0] != '\0') {
      flow_info->ssh_tls.encrypted_sni.esni = ndpi_strdup(flow_info->ndpi_flow->l4.tcp.tls.encrypted_sni.esni);
      flow_info->ssh_tls.encrypted_sni.esni_len = flow_info->ndpi_flow->l4.tcp.tls.encrypted_sni.esni_len;
    }
    if(flow_info->ssh_tls.tls_alpn[0] != '\0' || flow_info->ssh_tls.tls_supported_versions[0] != '\0') {
      flow_info->info_type = INFO_TLS;
    }
  }
