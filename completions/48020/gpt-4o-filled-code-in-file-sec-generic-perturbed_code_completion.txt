else if(is_ndpi_proto(flow_info, NDPI_PROTOCOL_SSH)) {
    flow_info->info_type = INFO_SSH;
    ndpi_snprintf(flow_info->ssh.client_signature,
                  sizeof(flow_info->ssh.client_signature),
                  "%s", flow_info->ndpi_flow->protos.ssh.client_signature);
    ndpi_snprintf(flow_info->ssh.server_signature,
                  sizeof(flow_info->ssh.server_signature),
                  "%s", flow_info->ndpi_flow->protos.ssh.server_signature);
    ndpi_snprintf(flow_info->ssh.client_hash,
                  sizeof(flow_info->ssh.client_hash),
                  "%s", flow_info->ndpi_flow->protos.ssh.client_hash);
    ndpi_snprintf(flow_info->ssh.server_hash,
                  sizeof(flow_info->ssh.server_hash),
                  "%s", flow_info->ndpi_flow->protos.ssh.server_hash);
  } else if(is_ndpi_proto(flow_info, NDPI_PROTOCOL_TLS) ||
            is_ndpi_proto(flow_info, NDPI_PROTOCOL_QUIC) ||
            is_ndpi_proto(flow_info, NDPI_PROTOCOL_HTTP2) ||
            is_ndpi_proto(flow_info, NDPI_PROTOCOL_HTTP3)) {
    flow_info->info_type = INFO_TLS;
    flow_info->tls.ssl_version = flow_info->ndpi_flow->protos.ssl.ssl_version;
    ndpi_snprintf(flow_info->tls.server_names,
                  sizeof(flow_info->tls.server_names),
                  "%s", flow_info->ndpi_flow->protos.ssl.server_names);
    ndpi_snprintf(flow_info->tls.ja3_client,
                  sizeof(flow_info->tls.ja3_client),
                  "%s", flow_info->ndpi_flow->protos.ssl.ja3_client);
    ndpi_snprintf(flow_info->tls.ja3_server,
                  sizeof(flow_info->tls.ja3_server),
                  "%s", flow_info->ndpi_flow->protos.ssl.ja3_server);
    ndpi_snprintf(flow_info->tls.ja3_client_hash,
                  sizeof(flow_info->tls.ja3_client_hash),
                  "%s", flow_info->ndpi_flow->protos.ssl.ja3_client_hash);
    ndpi_snprintf(flow_info->tls.ja3_server_hash,
                  sizeof(flow_info->tls.ja3_server_hash),
                  "%s", flow_info->ndpi_flow->protos.ssl.ja3_server_hash);
    ndpi_snprintf(flow_info->tls.validity,
                  sizeof(flow_info->tls.validity),
                  "%s", flow_info->ndpi_flow->protos.ssl.validity);
    ndpi_snprintf(flow_info->tls.issuer,
                  sizeof(flow_info->tls.issuer),
                  "%s", flow_info->ndpi_flow->protos.ssl.issuer);
    ndpi_snprintf(flow_info->tls.subject,
                  sizeof(flow_info->tls.subject),
                  "%s", flow_info->ndpi_flow->protos.ssl.subject);
    ndpi_snprintf(flow_info->tls.alpn,
                  sizeof(flow_info->tls.alpn),
                  "%s", flow_info->ndpi_flow->protos.ssl.tls_alpn);
    ndpi_snprintf(flow_info->tls.supported_versions,
                  sizeof(flow_info->tls.supported_versions),
                  "%s", flow_info->ndpi_flow->protos.ssl.tls_supported_versions);
    if(flow_info->tls.alpn[0] != '\0' || flow_info->tls.supported_versions[0] != '\0') {
      flow_info->info_type = INFO_TLS_ALPN;
    }
    if(flow_info->ndpi_flow->protos.ssl.tls_block_len > 0) {
      ndpi_reset_bin(&flow_info->payload_len_bin);
      ndpi_inc_bin(&flow_info->payload_len_bin, plen2slot(flow_info->ndpi_flow->protos.ssl.tls_block_len), 1);
    }
