    if (TIFFGetField(tiff,TIFFTAG_DOCUMENTNAME,&text_val) == 1)

        (void) SetImageProperty(img,"tiff:document",text_val,exception);

    if (TIFFGetField(tiff,TIFFTAG_HOSTCOMPUTER,&text_val) == 1)
      if ((text_val != (char *) NULL) && (*text_val != '\0'))

    if (TIFFGetField(tiff,33423,&text_val) == 1)

    /*
      Handle subfiletype with specific artifact values.
    */
    {
      uint32
        subfiletype = 0;

      if (TIFFGetField(tiff,TIFFTAG_SUBFILETYPE,&subfiletype) == 1)
        {
          if ((subfiletype & FILETYPE_REDUCEDIMAGE) != 0)
            (void) SetImageArtifact(img,"tiff:subfiletype","REDUCEDIMAGE",exception);
          else if ((subfiletype & FILETYPE_PAGE) != 0)
            (void) SetImageArtifact(img,"tiff:subfiletype","PAGE",exception);
          else if ((subfiletype & FILETYPE_MASK) != 0)
            (void) SetImageArtifact(img,"tiff:subfiletype","MASK",exception);
        }
    }

    /*
      Example: special handling for OPIIMAGEID (if needed).

    {

        opi_count = 0;

      if (TIFFGetField(tiff,32781,&opi_count,&text_val) == 1)
        {
          if ((text_val != (char *) NULL) && (opi_count > 0))
            {
              char opi_value[MagickPathExtent];
              (void) CopyMagickString(opi_value,text_val,
                (size_t) MagickMin(opi_count,MagickPathExtent-1));
              (void) SetImageProperty(img,"tiff:OPIIMAGEID",opi_value,exception);
            }
        }
