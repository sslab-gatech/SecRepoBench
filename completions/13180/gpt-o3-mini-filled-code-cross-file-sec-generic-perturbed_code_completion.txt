{
  const char *value = NULL;
  /* Document Name */
  if (TIFFGetField(tiff, TIFFTAG_DOCUMENTNAME, &value))
    if (value != NULL)
      SetImageProperty(img, "tiff:document", value, exception);

  /* Host Computer */
  if (TIFFGetField(tiff, TIFFTAG_HOSTCOMPUTER, &value))
    if (value != NULL)
      SetImageProperty(img, "tiff:hostcomputer", value, exception);

  /* Artist */
  if (TIFFGetField(tiff, TIFFTAG_ARTIST, &value))
    if (value != NULL)
      SetImageProperty(img, "tiff:artist", value, exception);

  /* Date/Time */
  if (TIFFGetField(tiff, TIFFTAG_DATETIME, &value))
    if (value != NULL)
      SetImageProperty(img, "tiff:timestamp", value, exception);

  /* Make */
  if (TIFFGetField(tiff, TIFFTAG_MAKE, &value))
    if (value != NULL)
      SetImageProperty(img, "tiff:make", value, exception);

  /* Model */
  if (TIFFGetField(tiff, TIFFTAG_MODEL, &value))
    if (value != NULL)
      SetImageProperty(img, "tiff:model", value, exception);

  /* Software */
  if (TIFFGetField(tiff, TIFFTAG_SOFTWARE, &value))
    if (value != NULL)
      SetImageProperty(img, "tiff:software", value, exception);

  /* Copyright */
  if (TIFFGetField(tiff, TIFFTAG_COPYRIGHT, &value))
    if (value != NULL)
      SetImageProperty(img, "tiff:copyright", value, exception);

  /* Kodak private tags */
  if (TIFFGetField(tiff, 33423, &value))
    if (value != NULL)
      SetImageProperty(img, "kodak-33423", value, exception);
  if (TIFFGetField(tiff, 36867, &value))
    if (value != NULL)
      SetImageProperty(img, "kodak-36867", value, exception);

  /* Page Name (set as label) */
  if (TIFFGetField(tiff, TIFFTAG_PAGENAME, &value))
    if (value != NULL)
      SetImageProperty(img, "label", value, exception);

  /* Image Description (set as comment) */
  if (TIFFGetField(tiff, TIFFTAG_IMAGEDESCRIPTION, &value))
    if (value != NULL)
      SetImageProperty(img, "comment", value, exception);

  /* OPI Image ID with count validation */
  {
    uint32 opi_count = 0;
    char *opi_text = NULL;
    if (TIFFGetField(tiff, TIFFTAG_OPIIMAGEID, &opi_count, &opi_text))
    {
      if (opi_text != NULL && opi_count < MagickPathExtent)
      {
        strncpy(message, opi_text, opi_count);
        message[opi_count] = '\0';
        SetImageProperty(img, "tiff:OPIImageID", message, exception);
      }
    }
  }

  /* Subfiletype handling */
  {
    uint32 subfiletype = 0;
    if (TIFFGetField(tiff, TIFFTAG_SUBFILETYPE, &subfiletype))
    {
      if (subfiletype == FILETYPE_REDUCEDIMAGE)
        SetImageProperty(img, "tiff:subfiletype", "REDUCEDIMAGE", exception);
      else if (subfiletype == FILETYPE_PAGE)
        SetImageProperty(img, "tiff:subfiletype", "PAGE", exception);
      else if (subfiletype == FILETYPE_MASK)
        SetImageProperty(img, "tiff:subfiletype", "MASK", exception);
      else
      {
        snprintf(message, MagickPathExtent, "%u", subfiletype);
        SetImageProperty(img, "tiff:subfiletype", message, exception);
      }
    }
  }
}
