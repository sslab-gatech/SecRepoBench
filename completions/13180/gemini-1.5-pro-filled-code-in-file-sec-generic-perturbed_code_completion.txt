void
  *sans[2] = { NULL, NULL };

  uint32
    length;

  unsigned char
    *profile;

  length = 0;
  if (TIFFGetField(tiff, TIFFTAG_ARTIST, &text, sans) == 1)
    (void) SetImageProperty(img, "tiff:artist", text, exception);
  if (TIFFGetField(tiff, TIFFTAG_COPYRIGHT, &text, sans) == 1)
    (void) SetImageProperty(img, "tiff:copyright", text, exception);

  if (TIFFGetField(tiff, TIFFTAG_DOCUMENTNAME, &text, sans) == 1)
    (void) SetImageProperty(img, "tiff:document", text, exception);

  if (TIFFGetField(tiff, TIFFTAG_HOSTCOMPUTER, &text, sans) == 1)
    (void) SetImageProperty(img, "tiff:hostcomputer", text, exception);

  if (TIFFGetField(tiff, TIFFTAG_IMAGEDESCRIPTION, &text, sans) == 1)
    (void) SetImageProperty(img, "comment", text, exception);

  if (TIFFGetField(tiff, TIFFTAG_MAKE, &text, sans) == 1)
    (void) SetImageProperty(img, "tiff:make", text, exception);

  if (TIFFGetField(tiff, TIFFTAG_MODEL, &text, sans) == 1)
    (void) SetImageProperty(img, "tiff:model", text, exception);

  if (TIFFGetField(tiff, TIFFTAG_PAGENAME, &text, sans) == 1)
    (void) SetImageProperty(img, "label", text, exception);


  if (TIFFGetField(tiff, TIFFTAG_SOFTWARE, &text, sans) == 1)
    (void) SetImageProperty(img, "tiff:software", text, exception);

  count = 0;
  if (TIFFGetField(tiff, TIFFTAG_OPIIMAGEID, &count, &profile) == 1)
  {
    if (count > 0)
    {
      text = (char *) AcquireQuantumMemory(count + 1, sizeof(char));
      if (text != (char *) NULL)
      {
        (void) memcpy(text, profile, count);
        text[count] = '\0';
        (void) SetImageProperty(img, "tiff:OPIIMAGEID", text, exception);
        text = (char *) RelinquishMagickMemory(text);
      }
    }
  }

  if (TIFFGetField(tiff, TIFFTAG_SUBFILETYPE, &length) == 1)
  {
    switch (length)
    {
      case FILETYPE_REDUCEDIMAGE:
        (void) SetImageProperty(img, "tiff:subfiletype", "REDUCEDIMAGE", exception);
        break;
      case FILETYPE_PAGE:
        (void) SetImageProperty(img, "tiff:subfiletype", "PAGE", exception);
        break;
      case FILETYPE_MASK:
        (void) SetImageProperty(img, "tiff:subfiletype", "MASK", exception);
        break;
      default:
        break;
    }
  }

  (void) TIFFGetFieldDefaulted(tiff, TIFFTAG_DATETIME, &text, sans);
  (void) SetImageProperty(img, "tiff:timestamp", text, exception);
