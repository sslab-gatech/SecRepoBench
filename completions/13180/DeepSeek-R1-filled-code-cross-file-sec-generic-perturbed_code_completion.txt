{
  const char *type="none";
  switch (subfiletype)
  {
    case FILETYPE_REDUCEDIMAGE: type="REDUCEDIMAGE"; break;
    case FILETYPE_PAGE: type="PAGE"; break;
    case FILETYPE_MASK: type="MASK"; break;
    default: break;
  }
  (void) SetImageProperty(img,"tiff:subfiletype",type,exception);
}
