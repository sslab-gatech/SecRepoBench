// Initialize TIFF tags for properties
  const char *value;

  value = GetImageArtifact(img, "tiff:document");
  if (value != NULL)
    (void) TIFFSetField(tiff, TIFFTAG_DOCUMENTNAME, value);

  value = GetImageArtifact(img, "tiff:hostcomputer");
  if (value != NULL)
    (void) TIFFSetField(tiff, TIFFTAG_HOSTCOMPUTER, value);

  value = GetImageArtifact(img, "tiff:artist");
  if (value != NULL)
    (void) TIFFSetField(tiff, TIFFTAG_ARTIST, value);

  value = GetImageArtifact(img, "tiff:timestamp");
  if (value != NULL)
    (void) TIFFSetField(tiff, TIFFTAG_DATETIME, value);

  value = GetImageArtifact(img, "tiff:make");
  if (value != NULL)
    (void) TIFFSetField(tiff, TIFFTAG_MAKE, value);

  value = GetImageArtifact(img, "tiff:model");
  if (value != NULL)
    (void) TIFFSetField(tiff, TIFFTAG_MODEL, value);

  value = GetImageArtifact(img, "tiff:software");
  if (value != NULL)
    (void) TIFFSetField(tiff, TIFFTAG_SOFTWARE, value);

  value = GetImageArtifact(img, "tiff:copyright");
  if (value != NULL)
    (void) TIFFSetField(tiff, TIFFTAG_COPYRIGHT, value);

  value = GetImageArtifact(img, "kodak-33423");
  if (value != NULL)
    (void) TIFFSetField(tiff, 33423, value);

  value = GetImageArtifact(img, "kodak-36867");
  if (value != NULL)
    (void) TIFFSetField(tiff, 36867, value);

  value = GetImageProperty(img, "label", exception);
  if (value != NULL)
    (void) TIFFSetField(tiff, TIFFTAG_PAGENAME, value);

  value = GetImageProperty(img, "comment", exception);
  if (value != NULL)
    (void) TIFFSetField(tiff, TIFFTAG_IMAGEDESCRIPTION, value);

  value = GetImageArtifact(img, "tiff:subfiletype");
  if (value != NULL) {
    if (LocaleCompare(value, "REDUCEDIMAGE") == 0)
      (void) TIFFSetField(tiff, TIFFTAG_SUBFILETYPE, FILETYPE_REDUCEDIMAGE);
    else if (LocaleCompare(value, "PAGE") == 0)
      (void) TIFFSetField(tiff, TIFFTAG_SUBFILETYPE, FILETYPE_PAGE);
    else if (LocaleCompare(value, "MASK") == 0)
      (void) TIFFSetField(tiff, TIFFTAG_SUBFILETYPE, FILETYPE_MASK);
  } else {
    uint16 page = (uint16) img->scene;
    uint16 pages = (uint16) GetImageListLength(img);
    if ((adjoin != MagickFalse) && (pages > 1))
      (void) TIFFSetField(tiff, TIFFTAG_SUBFILETYPE, FILETYPE_PAGE);
    (void) TIFFSetField(tiff, TIFFTAG_PAGENUMBER, page, pages);
  }
