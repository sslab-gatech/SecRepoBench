// Get standard TIFF tags
  if (TIFFGetField(tiff,TIFFTAG_DOCUMENTNAME,&text) == 1)
    if (text != (char *) NULL)
      (void) SetImageProperty(img,"tiff:document",text,exception);
  if (TIFFGetField(tiff,TIFFTAG_HOSTCOMPUTER,&text) == 1)
    if (text != (char *) NULL)
      (void) SetImageProperty(img,"tiff:hostcomputer",text,exception);
  if (TIFFGetField(tiff,TIFFTAG_ARTIST,&text) == 1)
    if (text != (char *) NULL)
      (void) SetImageProperty(img,"tiff:artist",text,exception);
  if (TIFFGetField(tiff,TIFFTAG_DATETIME,&text) == 1)
    if (text != (char *) NULL)
      (void) SetImageProperty(img,"tiff:timestamp",text,exception);
  if (TIFFGetField(tiff,TIFFTAG_MAKE,&text) == 1)
    if (text != (char *) NULL)
      (void) SetImageProperty(img,"tiff:make",text,exception);
  if (TIFFGetField(tiff,TIFFTAG_MODEL,&text) == 1)
    if (text != (char *) NULL)
      (void) SetImageProperty(img,"tiff:model",text,exception);
  if (TIFFGetField(tiff,TIFFTAG_SOFTWARE,&text) == 1)
    if (text != (char *) NULL)
      (void) SetImageProperty(img,"tiff:software",text,exception);
  if (TIFFGetField(tiff,TIFFTAG_COPYRIGHT,&text) == 1)
    if (text != (char *) NULL)
      (void) SetImageProperty(img,"tiff:copyright",text,exception);
  
  // Get Kodak specific tags
  if (TIFFGetField(tiff,33423,&text) == 1)
    if (text != (char *) NULL)
      (void) SetImageProperty(img,"kodak-33423",text,exception);
  if (TIFFGetField(tiff,36867,&text) == 1)
    if (text != (char *) NULL)
      (void) SetImageProperty(img,"kodak-36867",text,exception);
  
  // Get special OPI Image ID tag
  if (TIFFGetField(tiff,32781,&count,&text) == 1)
  {
    if ((text != (char *) NULL) && (count <= MagickPathExtent))
    {
      char *opiid = (char *) AcquireQuantumMemory(count+1,sizeof(*opiid));
      if (opiid != (char *) NULL)
      {
        (void) memcpy(opiid,text,count);
        opiid[count]='\0';
        (void) SetImageProperty(img,"tiff:opiimageid",opiid,exception);
        opiid=(char *) RelinquishMagickMemory(opiid);
      }
    }
  }
  
  // Get page name (label)
  if (TIFFGetField(tiff,TIFFTAG_PAGENAME,&text) == 1)
    if (text != (char *) NULL)
      (void) SetImageProperty(img,"label",text,exception);
  
  // Get image description (comment)
  if (TIFFGetField(tiff,TIFFTAG_IMAGEDESCRIPTION,&text) == 1)
    if (text != (char *) NULL)
      (void) SetImageProperty(img,"comment",text,exception);
  
  // Handle subfiletype property
  uint16 subfiletype;
  if (TIFFGetField(tiff,TIFFTAG_SUBFILETYPE,&subfiletype) == 1)
  {
    if (subfiletype & FILETYPE_REDUCEDIMAGE)
      (void) SetImageProperty(img,"tiff:subfiletype","REDUCEDIMAGE",exception);
    else if (subfiletype & FILETYPE_PAGE)
      (void) SetImageProperty(img,"tiff:subfiletype","PAGE",exception);
    else if (subfiletype & FILETYPE_MASK)
      (void) SetImageProperty(img,"tiff:subfiletype","MASK",exception);
  }
  
  // Get page number information
  uint16 page, pages;
  if (TIFFGetField(tiff,TIFFTAG_PAGENUMBER,&page,&pages) == 1)
  {
    (void) FormatLocaleString(message,MagickPathExtent,"%.20g",(double) page);
    (void) SetImageProperty(img,"tiff:page",message,exception);
    (void) FormatLocaleString(message,MagickPathExtent,"%.20g",(double) pages);
    (void) SetImageProperty(img,"tiff:pages",message,exception);
  }
