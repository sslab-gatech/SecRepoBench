/* OPIIMAGEID is a special case: it's a 32-byte string */
if (TIFFGetField(tiff,TIFFTAG_OPIIMAGEID,&count,&text) && (text != (char *) NULL) && (count > 0))
{
  char
    opiid[40];

  (void) CopyMagickString(opiid,text,(size_t) MagickMin(count,sizeof(opiid)-1));
  (void) SetImageProperty(img,"tiff:opiimageid",opiid,exception);
}
if (TIFFGetField(tiff,TIFFTAG_SUBFILETYPE,&type))
{
  const char *subfiletype = "none";
  if ((type & FILETYPE_REDUCEDIMAGE) != 0)
    subfiletype = "REDUCEDIMAGE";
  else if ((type & FILETYPE_PAGE) != 0)
    subfiletype = "PAGE";
  else if ((type & FILETYPE_MASK) != 0)
    subfiletype = "MASK";
  (void) SetImageProperty(img,"tiff:subfiletype",subfiletype,exception);
}
