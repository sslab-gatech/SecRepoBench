def fill_masked_region(image, mask, color):
    """
    Fill a masked region in an image with a specified color.

    :param image: The input image.
    :param mask: A binary image representing the mask.
    :param color: The color to fill the masked region with.
    :return: The modified image.
    """
    # Get the image dimensions
    width, height = image.size

    # Create a new image with the same dimensions as the input image
    filled_image = Image.new('RGB', (width, height))

    # Paste the input image onto the new image
    filled_image.paste(image, (0, 0))

    # Get the pixel access object for the new image
    pixels = filled_image.load()

    # Iterate over each pixel in the mask
    for y in range(height):
        for x in range(width):
            # If the pixel in the mask is set, fill the corresponding pixel
            # in the new image with the specified color
            if mask.getpixel((x, y))[0] == 0:
                pixels[x, y] = color

    # Return the modified image
    return filled_image
