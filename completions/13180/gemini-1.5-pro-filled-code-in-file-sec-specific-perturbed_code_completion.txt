uint32
    count;

  void
    *sans[2] = { NULL, NULL };

  if (TIFFGetField(tiff,TIFFTAG_ARTIST,&count,&text,sans) == 1)
    {
      if (text != (char *) NULL)
        (void) SetImageProperty(img,"tiff:artist",text,exception);
    }
  if (TIFFGetField(tiff,TIFFTAG_COPYRIGHT,&count,&text,sans) == 1)
    {
      if (text != (char *) NULL)
        (void) SetImageProperty(img,"tiff:copyright",text,exception);
    }
  if (TIFFGetField(tiff,TIFFTAG_DOCUMENTNAME,&count,&text,sans) == 1)
    {
      if (text != (char *) NULL)
        (void) SetImageProperty(img,"tiff:document",text,exception);
    }
  if (TIFFGetField(tiff,TIFFTAG_HOSTCOMPUTER,&count,&text,sans) == 1)
    {
      if (text != (char *) NULL)
        (void) SetImageProperty(img,"tiff:hostcomputer",text,exception);
    }
  if (TIFFGetField(tiff,TIFFTAG_IMAGEDESCRIPTION,&count,&text,sans) == 1)
    {
      if (text != (char *) NULL)
        (void) SetImageProperty(img,"comment",text,exception);
    }
  if (TIFFGetField(tiff,TIFFTAG_MAKE,&count,&text,sans) == 1)
    {
      if (text != (char *) NULL)
        (void) SetImageProperty(img,"tiff:make",text,exception);
    }
  if (TIFFGetField(tiff,TIFFTAG_MODEL,&count,&text,sans) == 1)
    {
      if (text != (char *) NULL)
        (void) SetImageProperty(img,"tiff:model",text,exception);
    }
  if (TIFFGetField(tiff,TIFFTAG_OPIIMAGEID,&count,&text,sans) == 1)
    {
      if (text != (char *) NULL)
        {
          if (count >= MagickPathExtent)
            count = MagickPathExtent-1;
          (void) strncpy(message,text,count);
          message[count] = '\0';
          (void) SetImageProperty(img,"tiff:image-id",message,exception);
        }
    }
  if (TIFFGetField(tiff,TIFFTAG_PAGENAME,&count,&text,sans) == 1)
    {
      if (text != (char *) NULL)
        (void) SetImageProperty(img,"label",text,exception);
    }
  if (TIFFGetField(tiff,TIFFTAG_SOFTWARE,&count,&text,sans) == 1)
    {
      if (text != (char *) NULL)
        (void) SetImageProperty(img,"tiff:software",text,exception);
    }


  if (TIFFGetField(tiff,TIFFTAG_SUBFILETYPE,&type,sans) == 1)
  {
    switch (type)
    {
      case FILETYPE_REDUCEDIMAGE:
        (void) SetImageProperty(img,"tiff:subfiletype","REDUCEDIMAGE",
          exception);
        break;
      case FILETYPE_PAGE:
        (void) SetImageProperty(img,"tiff:subfiletype","PAGE",exception);
        break;
      case FILETYPE_MASK:
        (void) SetImageProperty(img,"tiff:subfiletype","MASK",exception);
        break;
      default:
        break;
    }
  }
