// Extract and set TIFF properties as image properties
  if (TIFFGetField(tiff, TIFFTAG_ARTIST, &text) && text != NULL)
    (void) SetImageProperty(img, "tiff:artist", text, exception);

  if (TIFFGetField(tiff, TIFFTAG_COPYRIGHT, &text) && text != NULL)
    (void) SetImageProperty(img, "tiff:copyright", text, exception);

  if (TIFFGetField(tiff, TIFFTAG_DOCUMENTNAME, &text) && text != NULL)
    (void) SetImageProperty(img, "tiff:document", text, exception);

  if (TIFFGetField(tiff, TIFFTAG_IMAGEDESCRIPTION, &text) && text != NULL)
    (void) SetImageProperty(img, "tiff:description", text, exception);

  if (TIFFGetField(tiff, TIFFTAG_MAKE, &text) && text != NULL)
    (void) SetImageProperty(img, "tiff:make", text, exception);

  if (TIFFGetField(tiff, TIFFTAG_MODEL, &text) && text != NULL)
    (void) SetImageProperty(img, "tiff:model", text, exception);

  if (TIFFGetField(tiff, TIFFTAG_SOFTWARE, &text) && text != NULL)
    (void) SetImageProperty(img, "tiff:software", text, exception);

  if (TIFFGetField(tiff, TIFFTAG_DATETIME, &text) && text != NULL)
    (void) SetImageProperty(img, "tiff:timestamp", text, exception);

  if (TIFFGetField(tiff, TIFFTAG_PAGENAME, &text) && text != NULL)
    (void) SetImageProperty(img, "tiff:page-name", text, exception);

  if (TIFFGetField(tiff, TIFFTAG_SUBFILETYPE, &type))
  {
    if (type & FILETYPE_REDUCEDIMAGE)
      (void) SetImageProperty(img, "tiff:subfiletype", "REDUCEDIMAGE", exception);
    else if (type & FILETYPE_PAGE)
      (void) SetImageProperty(img, "tiff:subfiletype", "PAGE", exception);
    else if (type & FILETYPE_MASK)
      (void) SetImageProperty(img, "tiff:subfiletype", "MASK", exception);
  }

  if (TIFFGetField(tiff, TIFFTAG_OPIIMAGEID, &count, &text) && count > 0 && text != NULL)
  {
    length = MagickMin(count, MagickPathExtent - 1);
    (void) CopyMagickString(message, text, length + 1);
    (void) SetImageProperty(img, "tiff:opi-image-id", message, exception);
  }
