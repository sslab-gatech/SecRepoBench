const char
  *text;

uint32
  count;

for (text=TIFFGetField(tiff,TIFFTAG_ARTIST,&count,&text); text != (char *) NULL; )
  {
    (void) SetImageProperty(image,"tiff:artist",text,exception);
    break;
  }
for (text=TIFFGetField(tiff,TIFFTAG_COPYRIGHT,&count,&text); text != (char *) NULL; )
  {
    (void) SetImageProperty(image,"tiff:copyright",text,exception);
    break;
  }
for (text=TIFFGetField(tiff,TIFFTAG_DATETIME,&count,&text); text != (char *) NULL; )
  {
    (void) SetImageProperty(image,"tiff:timestamp",text,exception);
    break;
  }
for (text=TIFFGetField(tiff,TIFFTAG_DOCUMENTNAME,&count,&text); text != (char *) NULL; )
  {
    (void) SetImageProperty(image,"tiff:document",text,exception);
    break;
  }
for (text=TIFFGetField(tiff,TIFFTAG_HOSTCOMPUTER,&count,&text); text != (char *) NULL; )
  {
    (void) SetImageProperty(image,"tiff:hostcomputer",text,exception);
    break;
  }
for (text=TIFFGetField(tiff,TIFFTAG_IMAGEDESCRIPTION,&count,&text); text != (char *) NULL; )
  {
    (void) SetImageProperty(image,"comment",text,exception);
    break;
  }
for (text=TIFFGetField(tiff,TIFFTAG_MAKE,&count,&text); text != (char *) NULL; )
  {
    (void) SetImageProperty(image,"tiff:make",text,exception);
    break;
  }
for (text=TIFFGetField(tiff,TIFFTAG_MODEL,&count,&text); text != (char *) NULL; )
  {
    (void) SetImageProperty(image,"tiff:model",text,exception);
    break;
  }
for (text=TIFFGetField(tiff,TIFFTAG_PAGENAME,&count,&text); text != (char *) NULL; )
  {
    (void) SetImageProperty(image,"label",text,exception);
    break;
  }
for (text=TIFFGetField(tiff,TIFFTAG_SOFTWARE,&count,&text); text != (char *) NULL; )
  {
    (void) SetImageProperty(image,"tiff:software",text,exception);
    break;
  }
for (text=TIFFGetField(tiff,33432,&count,&text); text != (char *) NULL; )
  {
    (void) SetImageProperty(image,"copyright",text,exception);
    break;
  }
for (text=TIFFGetField(tiff,TIFFTAG_OPIIMAGEID,&count,&text); text != (char *) NULL; )
  {
    if (count >= 0)
      {
        char
          id[MagickPathExtent];

        (void) FormatLocaleString(id,MagickPathExtent,"%.1024s",text);
        (void) SetImageProperty(image,"tiff:image-id",id,exception);
      }
    break;
  }
if (TIFFGetField(tiff,TIFFTAG_SUBFILETYPE,&count) == 1)
  {
    if ((count & FILETYPE_REDUCEDIMAGE) != 0)
      (void) SetImageProperty(image,"tiff:subfiletype","REDUCEDIMAGE",
        exception);
    if ((count & FILETYPE_PAGE) != 0)
      (void) SetImageProperty(image,"tiff:subfiletype","PAGE",exception);
    if ((count & FILETYPE_MASK) != 0)
      (void) SetImageProperty(image,"tiff:subfiletype","MASK",exception);
  }
