if (TIFFGetField(tiff, TIFFTAG_ARTIST, &text) && text)
  (void) SetImageProperty(img, "artist", text, exception);
if (TIFFGetField(tiff, TIFFTAG_COPYRIGHT, &text) && text)
  (void) SetImageProperty(img, "copyright", text, exception);
if (TIFFGetField(tiff, TIFFTAG_DOCUMENTNAME, &text) && text)
  (void) SetImageProperty(img, "document", text, exception);
if (TIFFGetField(tiff, TIFFTAG_OPIIMAGEID, &count, &length, &type, &text) && text && count == 1)
{
  text[length] = '\0'; // Ensure null termination
  (void) SetImageProperty(img, "opi:image-id", text, exception);
}
if (TIFFGetField(tiff, TIFFTAG_SUBFILETYPE, &type) && type)
{
  switch(type)
  {
    case FILETYPE_REDUCEDIMAGE:
      (void) SetImageProperty(img, "tiff:subfiletype", "REDUCEDIMAGE", exception);
      break;
    case FILETYPE_PAGE:
      (void) SetImageProperty(img, "tiff:subfiletype", "PAGE", exception);
      break;
    case FILETYPE_MASK:
      (void) SetImageProperty(img, "tiff:subfiletype", "MASK", exception);
      break;
    default:
      (void) FormatLocaleString(message, MagickPathExtent, "%u", type);
      (void) SetImageProperty(img, "tiff:subfiletype", message, exception);
      break;
  }
}
