      char
        *opi_image_id;
  
      opi_image_id=(char *) AcquireQuantumMemory(count,sizeof(*opi_image_id));
      if (opi_image_id != (char *) NULL)
        {
          (void) memcpy(opi_image_id,text,count);
          opi_image_id[count-1]='\0';
