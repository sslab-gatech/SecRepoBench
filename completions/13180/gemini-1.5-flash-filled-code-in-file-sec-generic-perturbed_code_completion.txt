uint32
    count,
    length,
    type;

  char
    *text;

  for (length=0; TIFFGetField(tiff,TIFFTAG_ARTIST,&length,&text) == 1; )
  {
    if (text != (char *) NULL)
      (void) SetImageProperty(img,"tiff:artist",text,exception);
    break;
  }
  for (length=0; TIFFGetField(tiff,TIFFTAG_COPYRIGHT,&length,&text) == 1; )
  {
    if (text != (char *) NULL)
      (void) SetImageProperty(img,"tiff:copyright",text,exception);
    break;
  }
  for (length=0; TIFFGetField(tiff,TIFFTAG_DOCUMENTNAME,&length,&text) == 1; )
  {
    if (text != (char *) NULL)
      (void) SetImageProperty(img,"tiff:document",text,exception);
    break;
  }
  for (length=0; TIFFGetField(tiff,TIFFTAG_IMAGEDESCRIPTION,&length,&text) == 1; )
  {
    if (text != (char *) NULL)
      (void) SetImageProperty(img,"tiff:description",text,exception);
    break;
  }
  for (length=0; TIFFGetField(tiff,TIFFTAG_MAKE,&length,&text) == 1; )
  {
    if (text != (char *) NULL)
      (void) SetImageProperty(img,"tiff:make",text,exception);
    break;
  }
  for (length=0; TIFFGetField(tiff,TIFFTAG_MODEL,&length,&text) == 1; )
  {
    if (text != (char *) NULL)
      (void) SetImageProperty(img,"tiff:model",text,exception);
    break;
  }
  for (length=0; TIFFGetField(tiff,TIFFTAG_SOFTWARE,&length,&text) == 1; )
  {
    if (text != (char *) NULL)
      (void) SetImageProperty(img,"tiff:software",text,exception);
    break;
  }
  for (count=0; TIFFGetField(tiff,TIFFTAG_OPIIMAGEID,&count,&text) == 1; )
  {
    if ((text != (char *) NULL) && (count > 0))
      (void) SetImageProperty(img,"tiff:OPI-image-id",text,exception);
    break;
  }
  for (length=0; TIFFGetField(tiff,TIFFTAG_SUBIFD,&length,&text) == 1; )
  {
    if (text != (char *) NULL)
      (void) SetImageProperty(img,"tiff:subifd",text,exception);
    break;
  }
  for (length=0; TIFFGetField(tiff,TIFFTAG_DATETIME,&length,&text) == 1; )
  {
    if (text != (char *) NULL)
      (void) SetImageProperty(img,"tiff:timestamp",text,exception);
    break;
  }
  for (length=0; TIFFGetField(tiff,TIFFTAG_HOSTCOMPUTER,&length,&text) == 1; )
  {
    if (text != (char *) NULL)
      (void) SetImageProperty(img,"tiff:hostcomputer",text,exception);
    break;
  }
  for (length=0; TIFFGetField(tiff,TIFFTAG_PAGENAME,&length,&text) == 1; )
  {
    if (text != (char *) NULL)
      (void) SetImageProperty(img,"tiff:pagename",text,exception);
    break;
  }
  type=0;
  if (TIFFGetField(tiff,TIFFTAG_SUBFILETYPE,&type) == 1)
    {
      char
        message[MagickPathExtent];

      switch (type)
      {
        case FILETYPE_REDUCEDIMAGE:
          (void) CopyMagickString(message,"REDUCEDIMAGE",MagickPathExtent);
          break;
        case FILETYPE_PAGE:
          (void) CopyMagickString(message,"PAGE",MagickPathExtent);
          break;
        case FILETYPE_MASK:
          (void) CopyMagickString(message,"MASK",MagickPathExtent);
          break;
        default:
          message[0]='\0';
          break;
      }
      if (message[0] != '\0')
        (void) SetImageProperty(img,"tiff:subfiletype",message,exception);
    }
