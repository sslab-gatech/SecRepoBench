void TIFFGetProperties(TIFF *tiff,Image *image,ExceptionInfo *exception)
{
  char
    message[MagickPathExtent],
    *text;

  uint32
    count,
    length,
    type;

  void
    *sans[2] = { NULL, NULL };

  if (TIFFGetField(tiff,TIFFTAG_ARTIST,&text,sans) == 1)
    (void) SetImageProperty(image,"tiff:artist",text,exception);
  if (TIFFGetField(tiff,TIFFTAG_COPYRIGHT,&text,sans) == 1)
    (void) SetImageProperty(image,"tiff:copyright",text,exception);
  if (TIFFGetField(tiff,TIFFTAG_DOCUMENTNAME,&text,sans) == 1)
    (void) SetImageProperty(image,"tiff:document",text,exception);
  if (TIFFGetField(tiff,TIFFTAG_HOSTCOMPUTER,&text,sans) == 1)
    (void) SetImageProperty(image,"tiff:hostcomputer",text,exception);
  if (TIFFGetField(tiff,TIFFTAG_IMAGEDESCRIPTION,&text,sans) == 1)
    (void) SetImageProperty(image,"comment",text,exception);
  if (TIFFGetField(tiff,TIFFTAG_MAKE,&text,sans) == 1)
    (void) SetImageProperty(image,"tiff:make",text,exception);
  if (TIFFGetField(tiff,TIFFTAG_MODEL,&text,sans) == 1)
    (void) SetImageProperty(image,"tiff:model",text,exception);
  if (TIFFGetField(tiff,TIFFTAG_OPIIMAGEID,&count,&text,sans) == 1)
  {
    if (count >= MagickPathExtent)
      count=MagickPathExtent-1;
    (void) CopyMagickString(message,text,count+1);
    (void) SetImageProperty(image,"tiff:image-id",message,exception);
  }
  if (TIFFGetField(tiff,TIFFTAG_PAGENAME,&text,sans) == 1)
    (void) SetImageProperty(image,"label",text,exception);
  if (TIFFGetField(tiff,TIFFTAG_SOFTWARE,&text,sans) == 1)
    (void) SetImageProperty(image,"tiff:software",text,exception);
  if (TIFFGetField(tiff,TIFFTAG_SUBFILETYPE,&type,sans) == 1)
  {
    if (type == FILETYPE_REDUCEDIMAGE)
      (void) SetImageProperty(image,"tiff:subfiletype","REDUCEDIMAGE",exception);
    else
      if (type == FILETYPE_PAGE)
        (void) SetImageProperty(image,"tiff:subfiletype","PAGE",exception);
      else
        if (type == FILETYPE_MASK)
          (void) SetImageProperty(image,"tiff:subfiletype","MASK",exception);
  }
}
