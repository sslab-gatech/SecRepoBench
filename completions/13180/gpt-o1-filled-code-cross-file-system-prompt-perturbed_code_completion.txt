/* Document name */
  text=(char *) NULL;
  if (TIFFGetField(tiff,TIFFTAG_DOCUMENTNAME,&text) == 1)
    if ((text != (char *) NULL) && (*text != '\0'))
      (void) SetImageArtifact(img,"tiff:document",text,exception);

  /* Host computer */
  text=(char *) NULL;
  if (TIFFGetField(tiff,TIFFTAG_HOSTCOMPUTER,&text) == 1)
    if ((text != (char *) NULL) && (*text != '\0'))
      (void) SetImageArtifact(img,"tiff:hostcomputer",text,exception);

  /* Artist */
  text=(char *) NULL;
  if (TIFFGetField(tiff,TIFFTAG_ARTIST,&text) == 1)
    if ((text != (char *) NULL) && (*text != '\0'))
      (void) SetImageArtifact(img,"tiff:artist",text,exception);

  /* Timestamp (DateTime) */
  text=(char *) NULL;
  if (TIFFGetField(tiff,TIFFTAG_DATETIME,&text) == 1)
    if ((text != (char *) NULL) && (*text != '\0'))
      (void) SetImageArtifact(img,"tiff:timestamp",text,exception);

  /* Make */
  text=(char *) NULL;
  if (TIFFGetField(tiff,TIFFTAG_MAKE,&text) == 1)
    if ((text != (char *) NULL) && (*text != '\0'))
      (void) SetImageArtifact(img,"tiff:make",text,exception);

  /* Model */
  text=(char *) NULL;
  if (TIFFGetField(tiff,TIFFTAG_MODEL,&text) == 1)
    if ((text != (char *) NULL) && (*text != '\0'))
      (void) SetImageArtifact(img,"tiff:model",text,exception);

  /* Software */
  text=(char *) NULL;
  if (TIFFGetField(tiff,TIFFTAG_SOFTWARE,&text) == 1)
    if ((text != (char *) NULL) && (*text != '\0'))
      (void) SetImageArtifact(img,"tiff:software",text,exception);

  /* Copyright */
  text=(char *) NULL;
  if (TIFFGetField(tiff,TIFFTAG_COPYRIGHT,&text) == 1)
    if ((text != (char *) NULL) && (*text != '\0'))
      (void) SetImageArtifact(img,"tiff:copyright",text,exception);

  /* Custom Kodak tags */
  text=(char *) NULL;
  if (TIFFGetField(tiff,33423,&text) == 1)
    if ((text != (char *) NULL) && (*text != '\0'))
      (void) SetImageArtifact(img,"kodak-33423",text,exception);

  text=(char *) NULL;
  if (TIFFGetField(tiff,36867,&text) == 1)
    if ((text != (char *) NULL) && (*text != '\0'))
      (void) SetImageArtifact(img,"kodak-36867",text,exception);

  /* Page name -> label */
  text=(char *) NULL;
  if (TIFFGetField(tiff,TIFFTAG_PAGENAME,&text) == 1)
    if ((text != (char *) NULL) && (*text != '\0'))
      (void) SetImageProperty(img,"label",text,exception);

  /* Description -> comment */
  text=(char *) NULL;
  if (TIFFGetField(tiff,TIFFTAG_IMAGEDESCRIPTION,&text) == 1)
    if ((text != (char *) NULL) && (*text != '\0'))
      (void) SetImageProperty(img,"comment",text,exception);

  /* Special handling for OPIIMAGEID */
#if defined(TIFFTAG_OPIIMAGEID)
  text=(char *) NULL;
  count=0;
  if (TIFFGetField(tiff,TIFFTAG_OPIIMAGEID,&count,&text) == 1)
    if ((text != (char *) NULL) && (count < MagickPathExtent))
    {
      (void) CopyMagickString(message,text,(size_t) count+1);
      (void) SetImageProperty(img,"tiff:opiimageid",message,exception);
    }
#endif

  /* Subfile type (REDUCEDIMAGE, PAGE, MASK, etc.) */
  type=0;
  if (TIFFGetField(tiff,TIFFTAG_SUBFILETYPE,&type) == 1)
    {
      if ((type & FILETYPE_REDUCEDIMAGE) != 0)
        (void) SetImageProperty(img,"tiff:subfiletype","REDUCEDIMAGE",exception);
      else if ((type & FILETYPE_PAGE) != 0)
        (void) SetImageProperty(img,"tiff:subfiletype","PAGE",exception);
      else if ((type & FILETYPE_MASK) != 0)
        (void) SetImageProperty(img,"tiff:subfiletype","MASK",exception);
      else
        (void) SetImageProperty(img,"tiff:subfiletype","none",exception);
    }
