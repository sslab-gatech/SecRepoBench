unsigned int lastOffset = 0;
for (unsigned int i = 0; i < fdCount; i++) {
  if (fdmap.includes(i)) {
    unsigned int serializedSize = FontDict::calculate_serialized_size(fontDicts[i], opszr);
    this->fds[i].set_offset(lastOffset, offsetSize);
    lastOffset += serializedSize;
  } else {
    this->fds[i].set_offset(0, offsetSize); // Set to zero if not included
  }
}
this->lastOffset.set(lastOffset);
