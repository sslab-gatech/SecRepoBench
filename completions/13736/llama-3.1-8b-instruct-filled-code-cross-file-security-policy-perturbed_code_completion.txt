// <MASK>
unsigned int lastOffset = 0;
for (unsigned int i = 0; i < fontDicts.length; i++)
  if (fdmap.includes (i))
  {
    FontDict *dict = c->start_embed<FontDict> ();
    if (unlikely (!dict->serialize (c, fontDicts[i], opszr, privateInfos[fdmap[i]])))
      return_trace (false);
    // Compute the serialized size of the current font dictionary
    unsigned int dictSize = dict->get_size ();
    // Check if the computed size exceeds the allocated bounds
    if (unlikely (lastOffset + dictSize > c->get_max_size ()))
      return_trace (false);
    // Update the last offset to mark the end of the serialized data for the current font dictionary
    lastOffset += dictSize;
  }
