unsigned int offset = 1; // Offsets in CFF INDEX are 1-based
for (unsigned int i = 0; i < fdCount; i++)
{
  if (fdmap.includes (i))
  {
    this->offsets[i].set (offset, offsetSize);
    offset += FontDict::calculate_serialized_size (fontDicts[i], opszr);
  }
  else
    this->offsets[i].set (0, offsetSize);
}
