import numpy as np

def fill_masked_region(image, mask, color):
    """
    Fill the masked region of an image with a given color.

    :param image: The input image as a numpy array.
    :param mask: A boolean mask indicating the regions to fill.
    :param color: The color to fill the masked region with.
    :return: The modified image with the masked region filled.
    """
    # Create a copy of the image to avoid modifying the original
    filled_image = image.copy()

    # Get the color components
    r, g, b, a = color

    # Fill the masked region with the specified color
    filled_image[mask] = (r, g, b, a)

    return filled_image
