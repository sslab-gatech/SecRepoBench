import numpy as np

def fill_masked_region(image, mask, fill_color=(0, 0, 0)):
    """
    Fill masked region in an image with a given color.

    :param image: Input image as a numpy array.
    :param mask: Binary mask where 1 indicates masked region and 0 indicates valid region.
    :param fill_color: Tuple representing the color to fill the masked region.
    :return: Image with masked region filled.
    """
    # Get the dimensions of the image and mask
    h, w, c = image.shape

    # Create a boolean mask for the color channels
    color_mask = np.all(mask[..., np.newaxis], axis=-1)

    # Create a fill color array
    fill_color_array = np.full((h, w, c), fill_color, dtype=np.uint8)

    # Fill the masked region with the specified color
    image[color_mask] = fill_color_array[color_mask]

    return image
