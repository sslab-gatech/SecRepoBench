int val = 0;
    char unit = size[len - 1];

    // Remove the unit character and convert the rest to an integer
    char *endptr;
    val = strtol(size, &endptr, 10);
    if (endptr == size || *endptr != '\0') {
        return -1; // Invalid number
    }

    // Convert based on the unit
    switch (toupper(unit)) {
        case 'D':
            val *= 86400; // Days to seconds
            break;
        case 'H':
            val *= 3600;  // Hours to seconds
            break;
        case 'M':
            val *= 60;    // Minutes to seconds
            break;
        case 'S':
            // Seconds, do nothing
            break;
        default:
            return -1; // Invalid unit
    }

    return val;
