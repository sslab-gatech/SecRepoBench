        return -1;
    }

    num = strtoll(timestr, &endptr, 10);
    if (endptr == timestr) {
        return -1;
    }

    if (*endptr != '\0') {
        unit = *endptr;
        if (unit != 'D' && unit != 'd' &&
            unit != 'H' && unit != 'h' &&
            unit != 'M' && unit != 'm') {
            return -1;
        }
    }

    val = num;
    if (unit == 'D' || unit == 'd') {
        val *= 24 * 60 * 60;
    } else if (unit == 'H' || unit == 'h') {
        val *= 60 * 60;
    } else if (unit == 'M' || unit == 'm') {
        val *= 60;
