long long int multiplier = 0;
int val = 0;
int num;

if (timestr == NULL || strlen(timestr) < 2) {
    return -1; // Invalid input
}

// Extract numeric value
char *endptr;
num = strtol(timestr, &endptr, 10);
if (endptr == timestr || *endptr == '\0') {
    return -1; // No valid number found or no unit specified
}

// Determine time unit and calculate multiplier
switch (*endptr) {
    case 'D':
    case 'd':
        multiplier = 86400; // Days to seconds
        break;
    case 'H':
    case 'h':
        multiplier = 3600; // Hours to seconds
        break;
    case 'M':
    case 'm':
        multiplier = 60; // Minutes to seconds
        break;
    default:
        return -1; // Unsupported unit
}

val = num * multiplier;

return val;
