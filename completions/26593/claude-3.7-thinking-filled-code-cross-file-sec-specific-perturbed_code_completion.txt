int val = 0;
    char *endptr;
    
    if (!timestr) {
        return 0;
    }
    
    /* Parse the numeric part of the time string */
    val = (int) strtol(timestr, &endptr, 10);
    
    /* Check for time unit suffix and convert accordingly */
    if (*endptr != '\0') {
        if (*endptr == 'D' || *endptr == 'd') {
            /* Check for potential overflow, 86400 = seconds in a day */
            if (val > 2147483647 / 86400) {
                return 0;  /* Overflow, return default */
            }
            val *= 86400;
        }
        else if (*endptr == 'H' || *endptr == 'h') {
            /* Check for potential overflow, 3600 = seconds in an hour */
            if (val > 2147483647 / 3600) {
                return 0;  /* Overflow, return default */
            }
            val *= 3600;
        }
        else if (*endptr == 'M' || *endptr == 'm') {
            /* Check for potential overflow, 60 = seconds in a minute */
            if (val > 2147483647 / 60) {
                return 0;  /* Overflow, return default */
            }
            val *= 60;
        }
    }
