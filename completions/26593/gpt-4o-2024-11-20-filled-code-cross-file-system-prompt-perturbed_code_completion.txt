int val = 0;
    int len = strlen(timestr);

    if (len == 0) {
        return -1; // Invalid input
    }

    char unit = timestr[len - 1];
    int multiplier = 1;

    if (unit == 'D' || unit == 'd') {
        multiplier = 86400; // 1 day = 86400 seconds
    }
    else if (unit == 'H' || unit == 'h') {
        multiplier = 3600; // 1 hour = 3600 seconds
    }
    else if (unit == 'M' || unit == 'm') {
        multiplier = 60; // 1 minute = 60 seconds
    }
    else if (isdigit(unit)) {
        // No unit specified, assume seconds
        multiplier = 1;
    }
    else {
        return -1; // Invalid unit
    }

    if (isdigit(unit)) {
        val = atoi(timestr);
    }
    else {
        char num_part[len];
        strncpy(num_part, timestr, len - 1);
        num_part[len - 1] = '\0';
        val = atoi(num_part) * multiplier;
    }
