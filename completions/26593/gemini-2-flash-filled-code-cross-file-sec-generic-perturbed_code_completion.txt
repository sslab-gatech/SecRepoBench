int val = 0;
    int len;
    int i;
    int d = 0;

    if (!timestr) {
        return 0;
    }

    len = strlen(timestr);
    if (len == 0) {
        return 0;
    }

    for (i = 0; i < len; i++) {
        if (isdigit(timestr[i])) {
            continue;
        }
        else {
            break;
        }
    }

    if (i == 0) {
        return 0;
    }

    /* convert to integer */
    char tmp[32];
    memset(tmp, '\0', sizeof(tmp));
    memcpy(tmp, timestr, i);
    val = atoi(tmp);

    if (i == len) {
        return val;
    }

    /* identify the unit */
    char unit = timestr[i];

    if (unit == 'D' || unit == 'd') {
        val *= (24 * 60 * 60);
    }
    else if (unit == 'H' || unit == 'h') {
        val *= (60 * 60);
    }
    else if (unit == 'M' || unit == 'm') {
        val *= 60;
    }
    else {
        /* invalid unit */
        return 0;
    }
