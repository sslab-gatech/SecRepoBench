int val = 0;
    char unit = size[len - 1];

    // Remove the unit character from the string
    size[len - 1] = '\0';

    val = atoi(size);

    switch (toupper(unit)) {
        case 'D':
            val *= 86400; // 24 * 60 * 60
            break;
        case 'H':
            val *= 3600;  // 60 * 60
            break;
        case 'M':
            val *= 60;    // 60
            break;
        default:
            // If no valid unit is found, return -1
            return -1;
    }

    return val;
