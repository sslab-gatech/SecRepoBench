for (i = xd - ydigits; i >= 0; i--) {
  mp_dbl_limb q = (x.p[i + ydigits] * RADIX + x.p[i + ydigits - 1]) / z;
  mp_dbl_limb r = (x.p[i + ydigits] * RADIX + x.p[i + ydigits - 1]) - q * y.p[ydigits - 1];
  while (r < RADIX && q > 0) {
    q--;
    r += z;
  }
  x.p[i + ydigits] = 0;
  for (size_t j = 0; j < ydigits; j++) {
    mp_dbl_limb p = x.p[i + j] - q * y.p[j];
    x.p[i + j] = (mp_limb_t)p;
    if (p >= RADIX) {
      x.p[i + j + 1]--;
    }
  }
  q.p[i] = (mp_limb_t)q;
}
