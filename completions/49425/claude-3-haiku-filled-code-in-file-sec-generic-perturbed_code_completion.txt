mp_limb2 z = y.p[yd-1];
if (z == 0) {
  mrb_raise(mrb, E_DIVISION_BY_ZERO_ERROR, "division by zero");
}
mp_limb2 q = x.p[xd-1] / z;
if (q > DIG_MASK) {
  q = DIG_MASK;
}
for (i=xd-1; i>=0; i--) {
  mp_limb2 r = x.p[i];
  for (j=0; j<yd; j++) {
    r -= (mp_limb2)y.p[j] * q;
  }
  x.p[i+yd] = LOW(r);
}
