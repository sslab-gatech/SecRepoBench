// Perform long division of the unsigned integers represented by `xx` and `yy`.
  mpz_t q, r;
  mpz_init(mrb, &q);
  mpz_init(mrb, &r);

  size_t xd = digits(&x);
  size_t yd = digits(&y);

  for (size_t i = xd; i > 0; --i) {
    ulshift(mrb, &r, &r, 1);
    r.p[0] = x.p[i - 1];
    while (ucmp(&r, &y) >= 0) {
      mpz_sub(mrb, &r, &r, &y);
      q.p[i - 1]++;
    }
  }

  trim(&q);
  mpz_set(mrb, qq, &q);
  mpz_set(mrb, rr, &r);

  mpz_clear(mrb, &q);
  mpz_clear(mrb, &r);
