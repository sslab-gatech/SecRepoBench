// Read the query name
    if (b->l_data < c->l_qname + 4) {
        if (realloc_bam_data(b, c->l_qname + 4) < 0) return -1;
    }
    if (bgzf_read(filepointer, b->data + b->l_data, c->l_qname) != c->l_qname) return -1;
    b->l_data += c->l_qname;

    // Pad the query name with extra nulls if necessary
    if (c->l_qname % 4 != 0) {
        int extra_nulls = 4 - (c->l_qname % 4);
        if (realloc_bam_data(b, b->l_data + extra_nulls) < 0) return -1;
        memset(b->data + b->l_data, 0, extra_nulls);
        b->l_data += extra_nulls;
    }
