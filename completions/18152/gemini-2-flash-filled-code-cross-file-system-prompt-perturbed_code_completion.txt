    if (new_l_data < c->l_qname) return -4;
    if (new_l_data > b->m_data) {
        b->m_data = new_l_data;
        kroundup32(b->m_data);
        b->data = (uint8_t*)realloc(b->data, b->m_data);

            return -4;
        }
    }

    if (bgzf_read(filepointer, b->data, c->l_qname - c->l_extranul) != c->l_qname - c->l_extranul) return -4;

        b->data[c->l_qname - c->l_extranul + i] = '\0';
