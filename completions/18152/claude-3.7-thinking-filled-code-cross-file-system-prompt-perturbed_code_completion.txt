c->isize = (int32_t)x[7]; // Set the insert size from the input data

// Calculate the new length for data allocation, considering extranul (padding)
new_l_data = block_len - 32 + c->l_extranul; // block_len includes core data

// Check for consistency errors in data lengths
if (new_l_data < c->l_qname + c->l_extranul) 
    return -4;

// Allocate or reallocate memory for the BAM data buffer
if (b->m_data < new_l_data) {
    uint8_t *new_data = realloc(b->data, new_l_data);
    if (!new_data)
        return -4;
    b->data = new_data;
    b->m_data = new_l_data;
}
b->l_data = new_l_data;

// Read the query name part of the BAM record into the buffer
if (bgzf_read(filepointer, b->data, c->l_qname) != c->l_qname)
    return -4;

// Pad the query name with extra nulls if necessary
if (c->l_extranul) {
    memset(b->data + c->l_qname, 0, c->l_extranul);
}
