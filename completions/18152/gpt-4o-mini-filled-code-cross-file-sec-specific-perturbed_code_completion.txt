new_l_data = c->l_qname + c->l_extranul + c->l_qseq;
    if (new_l_data < c->l_qname || new_l_data < c->l_extranul || new_l_data < c->l_qseq) {
        return -4; // overflow detected
    }
    b->l_data = new_l_data;
    b->data = realloc(b->data, b->l_data);
    if (!b->data) {
        return -1; // memory allocation failed
    }
    memset(b->data, 0, b->l_data); // initialize the buffer
    memcpy(b->data, b->data, c->l_qname); // copy the query name
