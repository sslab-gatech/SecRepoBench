c->isize = (int32_t)x[7]; // Set the insert size from the input data

// Calculate the new length for data allocation
new_l_data = block_len - 32; // block_len includes core data (32 bytes)

// Check for overflow and consistency errors
if (new_l_data < c->l_qname || new_l_data > INT32_MAX) 
    return -4;

// Allocate or reallocate memory for b->data
if (b->m_data < new_l_data) {
    uint8_t *new_data = realloc(b->data, new_l_data);
    if (!new_data) return -4; // Memory allocation failure
    b->data = new_data;
    b->m_data = new_l_data;
}

b->l_data = new_l_data;

// Read the query name part (without padding)
if (bgzf_read(filepointer, b->data, c->l_qname - c->l_extranul) != c->l_qname - c->l_extranul)
    return -4;

// Pad with extra nulls if necessary
if (c->l_extranul) 
    memset(b->data + c->l_qname - c->l_extranul, 0, c->l_extranul);
