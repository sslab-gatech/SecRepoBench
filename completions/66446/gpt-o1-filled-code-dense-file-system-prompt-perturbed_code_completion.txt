if (streaming) {
        /*
         * In streaming mode, the attribute might be temporary, so store the
         * attribute's name in the ID based on the document's dictionary or by
         * duplicating it if no dictionary is available.
         */
        if (doc->dict) {
            ret->name = xmlDictLookup(doc->dict, attr->name, -1);
        } else {
            ret->name = xmlStrdup(attr->name);
            if (ret->name == NULL) {
                xmlFreeID(ret);
                return(-1);
            }
        }
    } else {
        /*
         * Not in streaming mode, associate the attribute with the ID.
         */
        ret->attr = attr;
    }

    /*
     * Record the line number from the attribute's parent element if available.
     */
    if ((attr->parent != NULL) && (attr->parent->type == XML_ELEMENT_NODE)) {
        ret->lineno = attr->parent->line;
    }

    /*
     * Add the new ID to the document's ID table.
     */
    if (xmlHashAddEntry(table, ret->value, ret) < 0) {
        xmlFreeID(ret);
        return(-1);
    }

    /*
     * Set the attribute type to ID, and associate it with the ID if not streaming.
     */
    attr->atype = XML_ATTRIBUTE_ID;
    if (!streaming) {
        attr->id = ret;
    }

    /*
     * If an ID pointer was provided, assign the newly created ID to it.
     */
    if (id != NULL) {
        *id = ret;
    }

    /*
     * Return success.
     */
    return(1);
