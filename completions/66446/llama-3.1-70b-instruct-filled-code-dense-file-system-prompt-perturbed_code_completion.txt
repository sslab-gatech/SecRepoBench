if (streaming) {
    if (doc->dict!= NULL)
        ret->name = xmlDictLookup(doc->dict, attr->name, attr->nameLen);
    else
        ret->name = xmlStrdup(attr->name);
    if (ret->name == NULL) {
        xmlFreeID(ret);
        return(-1);
    }
} else {
    ret->attr = attr;
    attr->atype = XML_ATTRIBUTE_ID;
    attr->id = ret;
}
ret->line = attr->parent->line;
if (xmlHashAddEntry(table, ret->value, ret) < 0) {
    xmlFreeID(ret);
    return(-1);
}
if (id!= NULL)
    *id = ret;
return(0);
