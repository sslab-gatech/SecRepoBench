if (streaming) {
        ret->name = xmlStrdup(attr->name);
    } else {
        attr->id = ret;
        attr->atype = XML_ATTRIBUTE_ID;
    }

    ret->attr = attr;
    ret->line = attr->parent->line; // Record the line number from the attribute's parent element.

    if (xmlHashAddEntry(table, ret->value, ret) < 0) {
        xmlFreeID(ret);
        return(-1);
    }

    if (id != NULL)
        *id = ret;

    return(0);
