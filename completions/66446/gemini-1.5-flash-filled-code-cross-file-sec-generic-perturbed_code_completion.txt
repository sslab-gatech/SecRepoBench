if (streaming) {
        if (attr->atype == XML_ATTRIBUTE_ID) {
            ret->value = xmlDictLookup(doc->dict, attr->name,
                                       xmlStrlen(attr->name));
            if (ret->value == NULL) {
                ret->value = xmlStrdup(attr->name);
            }
        }
    } else {
        attr->atype = XML_ATTRIBUTE_ID;
        attr->id = ret;
    }
    ret->attr = attr;
    ret->lineno = xmlGetLineNo(attr->parent);
    if (xmlHashAddEntry(table, ret->value, ret) < 0) {
        xmlFreeID(ret);
        return(-1);
    }
    if (id != NULL)
        *id = ret;
    return(0);
