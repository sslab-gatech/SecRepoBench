if (streaming) {
        xmlDictPtr dict = (doc->dict != NULL) ? doc->dict : NULL;

        if (dict != NULL)
            ret->name = xmlDictLookup(dict, identifier, -1);
        else
            ret->name = xmlStrdup(identifier);
        ret->attr = NULL;
    } else {
        ret->name = xmlStrdup(attr->name);
        ret->attr = attr;
    }
    ret->lineno = xmlGetLineNo(attr->parent);

    if (xmlHashAdd(table, identifier, ret) <= 0) {
        xmlFreeID(ret);
        return(-1);
    }
    attr->atype = XML_ATTRIBUTE_ID;
    attr->id = ret;
    if (id != NULL)
        *id = ret;
    return(1);
