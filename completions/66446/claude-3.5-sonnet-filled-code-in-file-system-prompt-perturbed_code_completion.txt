if (xmlIsStreaming(ctxt)) {
        // In streaming mode, handle temporary attribute
        if (doc->dict != NULL)
            ret->name = xmlDictLookup(doc->dict, name, -1);
        else
            ret->name = xmlStrdup(name);
        if (ret->name == NULL)
            goto mem_error;
        ret->lineno = xmlGetLineNo(attr->parent);
    } else {
        // Not in streaming mode, associate attribute with ID
        ret->attr = attr;
        ret->lineno = xmlGetLineNo(attr->parent);
    }

    // Add the new ID to the document's ID table
    if (xmlHashAddEntry(table, ret->value, ret) < 0) {
        xmlFreeID(ret);
        return(-1);
    }

    // Set attribute type and associate with ID if not streaming
    attr->atype = XML_ATTRIBUTE_ID;
    if (!xmlIsStreaming(ctxt))
        attr->id = ret;

    // Assign newly created ID to output parameter if provided
    if (id != NULL)
        *id = ret;

    return(1);
