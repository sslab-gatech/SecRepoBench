if (streaming) {
    /*
     * In streaming mode, the attribute may be temporary, so we need
     * to handle this case differently.
     */
    if (doc->dict != NULL) {
        ret->name = xmlDictLookup(doc->dict, attr->name, -1);
    } else {
        ret->name = xmlStrdup(attr->name);
        if (ret->name == NULL) {
            xmlFreeID(ret);
            return(-1);
        }
    }
} else {
    /*
     * Associate the attribute with the ID if not in streaming mode
     */
    ret->attr = attr;
    attr->id = ret;
}

/* 
 * Record the line number from the attribute's parent element
 */
if ((attr->parent != NULL) && (attr->parent->type == XML_ELEMENT_NODE)) {
    ret->lineno = attr->parent->line;
}

/*
 * Add the new ID to the document's ID table
 */
if (xmlHashAddEntry(table, identifier, ret) < 0) {
    xmlFreeID(ret);
    return(-1);
}

/*
 * Set attribute properties
 */
attr->atype = XML_ATTRIBUTE_ID;
if (!streaming) {
    attr->id = ret;
}

/*
 * Return the new ID if requested
 */
if (id != NULL)
    *id = ret;

return(0); /* Success */
