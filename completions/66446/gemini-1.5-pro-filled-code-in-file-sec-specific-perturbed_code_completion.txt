if (streaming) {
        if (doc->dict != NULL)
            ret->name = xmlDictLookup(doc->dict, attr->name, -1);
        else
            ret->name = xmlStrdup(attr->name);
        if (ret->name == NULL) {
            xmlFreeID(ret);
            return(-1);
        }
	ret->attr = NULL;
    } else {
        ret->attr = attr;
        ret->name = NULL;
    }
    if (attr->parent != NULL)
        ret->lineno = xmlGetLineNo(attr->parent);

    if (xmlHashAdd(table, ret->value, ret) < 0) {
        xmlFreeID(ret);
        return(-1);
    }

    if (!streaming) {
	attr->atype = XML_ATTRIBUTE_ID;
        attr->id = ret;
    }
    if (id != NULL)
        *id = ret;
    return(1);
