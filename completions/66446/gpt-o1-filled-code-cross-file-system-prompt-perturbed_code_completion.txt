if (streaming) {
        /*
         * The attribute might be temporary in streaming mode, so don't
         * store this node pointer. Instead we store its name.
         */
        if (doc->dict) {
            ret->name = xmlDictLookup(doc->dict, attr->name, -1);
        } else {
            ret->name = xmlStrdup(attr->name);
        }
    } else {
        /*
         * Non-streaming mode: we can safely associate the attribute
         * with this ID.
         */
        ret->attr = attr;
    }

    /*
     * Record line number from the parent element if available.
     */
    if (attr->parent && (attr->parent->type == XML_ELEMENT_NODE)) {
        ret->lineno = attr->parent->line;
    }

    /*
     * Add the new ID to the document's ID table.
     */
    if (xmlHashAddEntry(table, ret->value, ret) != 0) {
        xmlFreeID(ret);
        return(-1);
    }

    /*
     * Set the attribute type and link it to the ID if not streaming.
     */
    if (!streaming) {
        attr->atype = XML_ATTRIBUTE_ID;
        attr->id = ret;
    }

    /*
     * If requested, return the newly created ID pointer.
     */
    if (id != NULL) {
        *id = ret;
    }

    return(0);
