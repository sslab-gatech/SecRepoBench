if (streaming) {
    /* Operating in streaming mode, attr is temporary and will disappear */
    if (doc->dict != NULL) {
        ret->name = xmlDictLookup(doc->dict, attr->name, -1);
    } else {
        ret->name = xmlStrdup(attr->name);
        if (ret->name == NULL)
            goto mem_error;
    }
    ret->attr = NULL;
} else {
    /* Not in streaming mode, associate attribute with the ID */
    ret->name = NULL;
    ret->attr = attr;
}

/* Record the line number from the attribute's parent element */
ret->lineno = (attr->parent != NULL) ? xmlGetLineNo(attr->parent) : 0;

/* Add the new ID to the ID table */
if (xmlHashAddEntry(table, ret->value, ret) < 0)
    goto mem_error;

/* Set attribute type to ID */
attr->atype = XML_ATTRIBUTE_ID;

/* Associate the attribute with the ID if not streaming */
if (!streaming)
    attr->id = ret;

/* If requested, return the newly created ID pointer */
if (id != NULL)
    *id = ret;

return(1);
