if (streaming) {
        /* In streaming mode, the attribute may disappear, so save its name */
        if ((attr->name != NULL) && (doc->dict != NULL))
            ret->name = (const xmlChar *)xmlDictLookup(doc->dict, attr->name, -1);
        else if (attr->name != NULL)
            ret->name = xmlStrdup(attr->name);
        else
            ret->name = NULL;
        ret->attr = NULL;
    } else {
        ret->attr = attr;
        attr->id = ret;
    }
    if ((attr->parent != NULL) && (attr->parent->type == XML_ELEMENT_NODE))
        ret->lineno = attr->parent->line;
    else
        ret->lineno = 0;

    if (xmlHashAddEntry(table, identifier, ret) < 0) {
        xmlFreeID(ret);
        return(-1);
    }
    attr->atype = XML_ATTRIBUTE_ID;
    if (!streaming)
        attr->id = ret;
    if (id != NULL)
        *id = ret;
    return(0);
