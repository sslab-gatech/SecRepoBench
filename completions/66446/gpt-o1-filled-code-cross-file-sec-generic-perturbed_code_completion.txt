if (streaming) {
        /*
         * In streaming mode, the attribute might not persist, so store the
         * attribute's name separately, either from the document's dictionary
         * or by duplicating it.
         */
        if (doc->dict) {
            ret->name = xmlDictLookup(doc->dict, attr->name, -1);
        } else {
            ret->name = xmlStrdup(attr->name);
            if (ret->name == NULL) {
                xmlFreeID(ret);
                return(-1);
            }
        }
    } else {
        /*
         * In non-streaming mode, associate the attribute with the ID.
         */
        ret->attr = attr;
    }

    /*
     * Record the line number from the parent element, if available.
     */
    if (attr->parent != NULL)
        ret->lineno = attr->parent->line;

    /*
     * Add the new ID to the document's ID table.
     */
    if (xmlHashAddEntry(table, ret->value, ret) < 0) {
        xmlFreeID(ret);
        return(-1);
    }

    /*
     * Set the attribute's type to ID, and if we are not in streaming mode,
     * link the attribute to the new ID.
     */
    attr->atype = XML_ATTRIBUTE_ID;
    if (!streaming)
        attr->id = ret;

    /*
     * If an ID output pointer is provided, assign the new ID to it.
     */
    if (id != NULL)
        *id = ret;

    return(0);
