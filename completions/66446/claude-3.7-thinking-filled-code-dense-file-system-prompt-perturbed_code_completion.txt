if (streaming) {
    /*
     * Operating in streaming mode, the attribute will disappear
     */
    if (attr->parent != NULL) {
        ret->lineno = attr->parent->line;
    }
    
    if (doc->dict != NULL) {
        ret->name = xmlDictLookup(doc->dict, attr->name, -1);
    } else {
        ret->name = xmlStrdup(attr->name);
        if (ret->name == NULL) {
            xmlFreeID(ret);
            return(-1);
        }
    }
    ret->attr = NULL; /* mark as streaming */
} else {
    /*
     * Normal operating mode, keep a pointer to the attribute
     */
    ret->attr = attr;
    if (attr->parent != NULL) {
        ret->lineno = attr->parent->line;
    }
}

/*
 * Add it to the hash table
 */
if (xmlHashAddEntry(table, identifier, ret) < 0) {
    xmlFreeID(ret);
    return(-1);
}

attr->atype = XML_ATTRIBUTE_ID;
if (!streaming) {
    attr->id = ret;
}

if (id != NULL)
    *id = ret;

return(1);
