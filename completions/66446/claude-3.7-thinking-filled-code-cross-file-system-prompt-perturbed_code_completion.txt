if (streaming) {
    /*
     * In streaming mode, the attribute might be temporary and will disappear,
     * so we need to save the name independently.
     */
    if (doc->dict != NULL) {
        ret->name = xmlDictLookup(doc->dict, attr->name, -1);
    } else {
        ret->name = xmlStrdup(attr->name);
        if (ret->name == NULL) {
            xmlFreeID(ret);
            return(-1);
        }
    }
    ret->attr = NULL;
} else {
    ret->attr = attr;
    attr->id = ret;
}

/* Record the line number from the attribute's parent element */
if ((attr->parent != NULL) && (attr->parent->type == XML_ELEMENT_NODE)) {
    ret->lineno = attr->parent->line;
}

/* Add the new ID to the document's ID table */
if (xmlHashAddEntry(table, identifier, ret) < 0) {
    xmlFreeID(ret);
    return(-1);
}

/* Set the attribute type */
attr->atype = XML_ATTRIBUTE_ID;

/* Set the return ID pointer if provided */
if (id != NULL)
    *id = ret;

return(0);
