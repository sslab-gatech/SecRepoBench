if (streaming) {
        // If in streaming mode, handle the case where the attribute is temporary and will disappear.
        // If the attribute's parent element has a name, use it to set the ID's name.
        // Otherwise, duplicate the attribute's name.
        if (attr->parent != NULL && attr->parent->name != NULL) {
            ret->value = xmlStrdup(attr->parent->name);
        } else {
            ret->value = xmlStrdup(attr->name);
        }
        if (ret->value == NULL) {
            xmlFreeID(ret);
            return(-1);
        }
    } else {
        // If not in streaming mode, associate the attribute with the ID.
        attr->atype = XML_ATTRIBUTE_ID;
        attr->id = ret;
    }

    // Add the new ID to the document's ID table.
    if (xmlHashAddEntry(table, ret->value, ret) < 0) {
        xmlFreeID(ret);
        return(-1);
    }

    // If an ID pointer is provided as an argument, assign the newly created ID to it.
    if (id != NULL) {
        *id = ret;
    }

    return(0);
}
