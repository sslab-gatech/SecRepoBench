if (streaming) {
    if (attr->parent != NULL) {
        if (doc->dict != NULL) {
            ret->name = xmlDictLookup(doc->dict, attr->name, -1);
        } else {
            ret->name = xmlStrdup(attr->name);
            if (ret->name == NULL) {
                xmlFreeID(ret);
                return(-1);
            }
        }
    }
} else {
    ret->attr = attr;
}

if (attr->parent != NULL)
    ret->lineno = attr->parent->line;

if (xmlHashAddEntry(table, identifier, ret) < 0) {
    xmlFreeID(ret);
    return(-1);
}

attr->atype = XML_ATTRIBUTE_ID;
if (!streaming)
    attr->id = ret;

if (id != NULL)
    *id = ret;

return(1);
