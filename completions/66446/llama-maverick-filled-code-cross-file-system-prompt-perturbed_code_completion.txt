if (streaming) {
    if (attr->parent == NULL) {
        ret->name = xmlDictLookup(doc->dict, attr->name, -1);
    } else {
        ret->name = xmlStrdup(attr->name);
        if (ret->name == NULL) {
            xmlFreeID(ret);
            return(-1);
        }
    }
    ret->lineno = xmlGetLineNo(attr->parent);
} else {
    ret->name = NULL;
    ret->lineno = 0;
    attr->id = ret;
}

if (xmlHashAddEntry(table, identifier, ret) < 0) {
    /*
     * The ID already exists in the table, this should not happen.
     * Free the duplicate and return an error.
     */
    xmlFreeID(ret);
    return(-1);
}

attr->atype = XML_ATTRIBUTE_ID;
if (!streaming)
    attr->id = ret;

if (id != NULL)
    *id = ret;

return(0);
