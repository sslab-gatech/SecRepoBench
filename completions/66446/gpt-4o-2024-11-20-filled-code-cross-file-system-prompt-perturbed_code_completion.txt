if (streaming) {
        if (attr->name != NULL) {
            if (doc->dict) {
                ret->name = (const xmlChar *)xmlDictLookup(doc->dict, attr->name, -1);
            } else {
                ret->name = xmlStrdup(attr->name);
            }
        }
        ret->lineno = (attr->parent != NULL) ? attr->parent->line : 0;
    } else {
        ret->attr = attr;
    }

    if (xmlHashAddEntry(table, identifier, ret) < 0) {
        xmlFreeID(ret);
        return (-1);
    }

    if (!streaming) {
        attr->atype = XML_ATTRIBUTE_ID;
        attr->id = ret;
    }

    if (id != NULL) {
        *id = ret;
    }

    return (0);
