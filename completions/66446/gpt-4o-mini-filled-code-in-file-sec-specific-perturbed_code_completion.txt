if (streaming) {
        ret->name = xmlStrdup(attr->name);
        if (ret->name == NULL) {
            xmlFreeID(ret);
            return(-1);
        }
        ret->attr = NULL;
    } else {
        ret->name = NULL;
        ret->attr = attr;
    }
    ret->lineno = xmlGetLineNo(attr->parent);

    /* Add the new ID to the document's ID table */
    if (xmlHashAdd(doc->ids, ret->value, ret) < 0) {
        xmlFreeID(ret);
        return(-1);
    }

    if (id != NULL) {
        *id = ret;
    }
    return(1);
