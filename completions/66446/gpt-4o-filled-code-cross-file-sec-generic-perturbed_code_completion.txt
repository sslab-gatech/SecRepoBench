if (streaming) {
        if (attr->name != NULL) {
            ret->name = xmlDictLookup(doc->dict, attr->name, -1);
        } else {
            ret->name = xmlStrdup(attr->name);
        }
        ret->lineno = (attr->parent != NULL) ? attr->parent->line : 0;
    } else {
        ret->attr = attr;
        ret->name = NULL;
        ret->lineno = (attr->parent != NULL) ? attr->parent->line : 0;
    }

    if (xmlHashAddEntry(table, identifier, ret) < 0) {
        xmlFreeID(ret);
        return(-1);
    }

    attr->atype = XML_ATTRIBUTE_ID;
    if (!streaming) {
        attr->id = ret;
    }

    if (id != NULL) {
        *id = ret;
    }

    return(0);
