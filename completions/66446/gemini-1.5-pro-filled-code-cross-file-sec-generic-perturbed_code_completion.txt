if (streaming) {
        if (attr->_private == NULL) {
            ret->name = xmlDictLookup(doc->dict, identifier, -1);
        } else {
            ret->name = xmlStrdup(identifier);
            if (ret->name == NULL) {
                xmlFree(ret->value);
                xmlFree(ret);
                return(-1);
            }
        }
    } else {
        ret->attr = attr;
    }
    if (attr->parent != NULL)
        ret->line = attr->parent->line;

    if (xmlHashAddEntry2(table, ret->name ? ret->name : ret->value, ret) < 0) {
        xmlFreeID(ret);
        return(-1);
    }

    attr->atype = XML_ATTRIBUTE_ID;
    if (!streaming)
        attr->id = ret;

    if (id != NULL) *id = ret;
    return(0);
