if (streaming) {
        if (doc->dict) {
            ret->name = xmlDictLookup(doc->dict, identifier, -1);
        } else {
            ret->name = xmlStrdup(attr->name);
        }
    } else {
        ret->attr = attr;
        ret->line = attr->parent->line;
        attr->atype = XML_ATTRIBUTE_ID;
        attr->id = ret;
    }

    if (xmlHashAddEntry(table, ret->value, ret) != 0) {
        xmlFreeID(ret);
        return(-1);
    }

    if (id != NULL)
        *id = ret;

    return(0);
