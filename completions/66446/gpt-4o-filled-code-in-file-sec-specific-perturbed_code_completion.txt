if (streaming) {
        // In streaming mode, the attribute is temporary and will disappear
        if (attr->name != NULL) {
            if (doc->dict) {
                ret->name = xmlDictLookup(doc->dict, attr->name, -1);
            } else {
                ret->name = xmlStrdup(attr->name);
            }
            if (ret->name == NULL) {
                xmlFreeID(ret);
                return(-1);
            }
        }
    } else {
        // Associate the attribute with the ID
        ret->attr = attr;
    }
    ret->lineno = xmlGetLineNo(attr->parent);

    // Add the new ID to the document's ID table
    if (xmlHashAddEntry(table, identifier, ret) < 0) {
        xmlFreeID(ret);
        return(-1);
    }

    attr->atype = XML_ATTRIBUTE_ID;
    if (!streaming) {
        attr->id = ret;
    }

    if (id != NULL) {
        *id = ret;
    }

    return(1);
