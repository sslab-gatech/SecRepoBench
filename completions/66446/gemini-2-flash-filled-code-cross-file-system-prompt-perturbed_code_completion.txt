if (streaming) {
        if (attr->name == NULL)
            ret->name = NULL;
        else if (doc->dict == NULL)
            ret->name = xmlStrdup(attr->name);
        else
            ret->name = xmlDictLookup(doc->dict, attr->name, -1);
        ret->attr = NULL;
    } else {
        ret->attr = attr;
    }
    if (attr->parent != NULL)
        ret->lineno = attr->parent->line;
    else
        ret->lineno = 0;

    if (xmlHashAddEntry(table, identifier, ret)) {
        xmlFreeID(ret);
        return(-1);
    }
    attr->atype = XML_ATTRIBUTE_ID;
    attr->id = ret;
    if (id != NULL)
        *id = ret;
    return(0);
