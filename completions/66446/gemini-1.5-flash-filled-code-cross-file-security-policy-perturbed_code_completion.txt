if (streaming) {
        if (doc->dict != NULL) {
            ret->value = xmlDictLookup(doc->dict, identifier, xmlStrlen(identifier));
        } else {
            ret->value = xmlStrdup(identifier);
        }
        if (ret->value == NULL) {
            xmlFreeID(ret);
            return(-1);
        }
    } else {
        ret->attr = attr;
    }
    if (attr->parent != NULL) {
        ret->line = attr->parent->line;
    }
    if (xmlHashAddEntry(table, identifier, ret) < 0) {
        xmlFreeID(ret);
        return(-1);
    }
    attr->atype = XML_ATTRIBUTE_ID;
    if (!streaming) {
        attr->id = ret;
    }
    if (id != NULL) {
        *id = ret;
    }
    return(0);
