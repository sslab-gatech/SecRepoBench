if (xmlIsStreaming(ctxt)) {
    if (attr->name!= NULL)
        ret->name = xmlStrdup(attr->name);
    else
        ret->name = xmlStrdup((const xmlChar *) "xmlns");
    if (ret->name == NULL)
        goto mem_error;
    ret->value = xmlStrdup(identifier);
    if (ret->value == NULL)
        goto mem_error;
    ret->lineno = xmlGetLineNo(attr->parent);
    ret->attr = NULL;
    ret->doc = doc;
    if (xmlHashAddEntry(table, ret->value, ret) < 0)
        goto mem_error;
    return(1);
} else {
    ret->name = NULL;
    ret->value = xmlStrdup(identifier);
    if (ret->value == NULL)
        goto mem_error;
    ret->lineno = xmlGetLineNo(attr->parent);
    ret->attr = attr;
    attr->id = ret;
    ret->doc = doc;
    if (xmlHashAddEntry(table, ret->value, ret) < 0)
        goto mem_error;
    return(1);
}
