// Ensure that we have enough data to read the header
    if (nCompressed < 20) {
        return EXR_ERR_CORRUPT_CHUNK;
    }

    // Read the header information
    im = readUInt(compressed);
    iM = readUInt(compressed + 4);
    nBits = readUInt(compressed + 12);

    // Validate the header values
    if (im >= HUF_ENCSIZE || iM >= HUF_ENCSIZE) {
        return EXR_ERR_CORRUPT_CHUNK;
    }

    // Calculate the number of bytes needed for the compressed data
    nBytes = (static_cast<uint64_t>(nBits) + 7) / 8;

    // Check if we have enough data for the compressed payload
    if (20 + nBytes > nCompressed) {
        return EXR_ERR_CORRUPT_CHUNK;
    }

    // Pointer to the compressed data
    ptr = compressed + 20;

    // Perform the Huffman decoding
    if (FastHufDecoder::enabled() && nBits > 128) {
        FastHufDecoder fhd(ptr, nCompressed - (ptr - compressed), im, iM, iM);
        fhd.decode((unsigned char*)ptr, nBits, raw, nRawSize);
    } else {
        AutoArray<uint64_t, HUF_ENCSIZE> freq;
        AutoArray<HufDec, HUF_DECSIZE> hdec;

        hufClearDecTable(hdec);
        hufUnpackEncTable(&ptr, nCompressed - (ptr - compressed), im, iM, freq);
        hufBuildDecTable(freq, im, iM, hdec);
        hufDecode(freq, hdec, ptr, nBits, iM, nRawSize, raw);
    }

    rv = EXR_ERR_SUCCESS;
