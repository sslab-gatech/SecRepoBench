if (dimensions == NULL) return 0;
    
    cmsUInt32Number i;
    cmsUInt32Number total = 1;
    
    for (i = 0; i < b; i++) {
        // Check if multiplication would overflow
        if ((total > 0) && (dimensions[i] > UINT32_MAX / total)) {
