cmsUInt32Number result = 1;
cmsUInt32Number max_value = 0x7FFFFFFF; // Assuming 32-bit signed integer maximum value

if (dimensions == NULL) {
    return 0; // Return 0 if dimensions array is null
}

for (cmsUInt32Number i = 0; i < b; i++) {
    if (dimensions[i] == 0) {
        return 0; // Return 0 if any dimension is zero
    }

    cmsUInt64Number temp = (cmsUInt64Number)result * dimensions[i];
    if (temp > max_value) {
        return 0; // Return 0 if product exceeds maximum value
    }

    result = (cmsUInt32Number)temp;
}

return result;
