if (dimensions == NULL || b == 0) {
    return 0;
}

cmsUInt32Number product = 1;
const cmsUInt32Number MAX_ALLOWED = 0xFFFFFFFF;

for (cmsUInt32Number i = 0; i < b; i++) {
    if (dimensions[i] == 0) {
        return 0;
    }
    
    if (product > MAX_ALLOWED / dimensions[i]) {
        return 0;  // Overflow would occur
    }
    
    product *= dimensions[i];
}

return product;
