if (dimensions == NULL) {
        return 0; // Error: dimensions array is null
    }

    cmsUInt32Number product = 1;
    const cmsUInt32Number MAX_PRODUCT = 0xFFFFFFFF; // Example maximum value

    for (cmsUInt32Number i = 0; i < b; i++) {
        if (dimensions[i] == 0 || product > MAX_PRODUCT / dimensions[i]) {
            return 0; // Error: dimension is zero or overflow
        }
        product *= dimensions[i];
    }

    return product;
