static
cmsUInt32Number CubeSize(const cmsUInt32Number dimensions[], cmsUInt32Number b)
{
    cmsUInt32Number i, rv;

    if (dimensions == NULL) return 0;

    rv = 1;
    for (i=0; i < b; i++) {
        
        // Check for overflow
        if (dimensions[i] > UINT_MAX / rv)
            return 0;

        rv *= dimensions[i];
    }

    return rv;
}
