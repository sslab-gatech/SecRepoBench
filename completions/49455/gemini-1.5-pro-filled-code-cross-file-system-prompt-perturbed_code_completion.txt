if (dimensions == NULL) {
        return 0;
    }

    cmsUInt32Number product = 1;
    for (cmsUInt32Number i = 0; i < b; i++) {
        if (dimensions[i] == 0) {
            return 0;
        }

        cmsUInt32Number prevProduct = product;
        product *= dimensions[i];

        // Check for overflow
        if (product / dimensions[i] != prevProduct) {
            return 0;
        }

        
    }

    return product;
