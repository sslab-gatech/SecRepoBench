cmsUInt32Number CubeSize(const cmsUInt32Number dimensions[], cmsUInt32Number b)
{
    cmsUInt32Number result = 1;
    for (cmsUInt32Number i = 0; i < b; i++) {
        if (dimensions[i] == 0) return 0;
        if (UINT_MAX / dimensions[i] < result) return 0;
        result *= dimensions[i];
    }
    return result;
}
