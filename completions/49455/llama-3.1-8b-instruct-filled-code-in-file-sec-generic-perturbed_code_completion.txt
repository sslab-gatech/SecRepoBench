static
cmsUInt32Number CubeSize(const cmsUInt32Number dimensions[], cmsUInt32Number b)
{
    cmsUInt32Number n = 1;
    cmsUInt32Number i;

    for (i=0; i < b; i++) {
        if (dimensions[i] == 0) return 0;
        if (n >= UINT_MAX / dimensions[i]) return 0;
        n *= dimensions[i];
    }

    return n;
}
