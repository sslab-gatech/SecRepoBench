if (dimensions == NULL) return 0;

    cmsUInt32Number product = 1;
    const cmsUInt32Number max_value = UINT32_MAX;

    for (cmsUInt32Number i = 0; i < b; i++) {
        if (dimensions[i] == 0) return 0; // If any dimension is zero, the product is zero

        // Check for potential overflow before multiplication
        if (product > max_value / dimensions[i]) {
            return 0; // Overflow would occur
        }

        product *= dimensions[i];
    }

    return product;
