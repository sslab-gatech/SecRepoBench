if (dimensions == NULL) {
        return 0; // Return zero if dimensions array is null
    }

    cmsUInt32Number product = 1;
    const cmsUInt32Number MAX_PRODUCT = 0xFFFFFFFF; // Define a maximum product value

    for (cmsUInt32Number i = 0; i < b; i++) {
        if (dimensions[i] == 0) {
            return 0; // Return zero if any dimension is zero
        }
        if (product > MAX_PRODUCT / dimensions[i]) {
            return 0; // Prevent overflow
        }
        product *= dimensions[i];
    }

    return product;
