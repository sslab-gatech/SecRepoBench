cmsUInt32Number result = 1;
if (!dimensions) return 0;

for (cmsUInt32Number i = 0; i < b; i++) {
    // Check for potential overflow
    if (dimensions[i] != 0 && result > 0xFFFFFFFF / dimensions[i]) {
        return 0; // Return error on overflow
    }
    
    result *= dimensions[i];
}
return result;
