if (dimensions == NULL || b == 0) {
        return 0;
    }

    cmsUInt32Number product = 1;
    for (cmsUInt32Number i = 0; i < b; i++) {
        if (dimensions[i] == 0) {
            return 0;
        }
        if (dimensions[i] != 1) {
            if (product > UINT_MAX / dimensions[i]) {
                return 0; // Overflow detected
            }
            product *= dimensions[i];
        }

    }
    return product;
