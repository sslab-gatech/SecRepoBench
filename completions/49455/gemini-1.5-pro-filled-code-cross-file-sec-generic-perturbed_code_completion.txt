if (dimensions == NULL || b == 0) {
        return 0;
    }

    cmsUInt32Number product = 1;
    for (cmsUInt32Number i = 0; i < b; i++) {
        if (dimensions[i] == 0) {
            return 0;
        }

        if (dimensions[i] > UINT32_MAX / product) {
            return 0; // Indicate an overflow error
        }
        product *= dimensions[i];
    }

    return product;
