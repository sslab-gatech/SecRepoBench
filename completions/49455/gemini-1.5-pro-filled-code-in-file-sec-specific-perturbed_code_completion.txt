cmsUInt32Number size = 1;
    for (cmsUInt32Number i = 0; i < b; i++) {
        if (dimensions == NULL || dimensions[i] == 0) {

        }

        // Check for potential overflow during multiplication
        if (size > UINT_MAX / dimensions[i]) {
