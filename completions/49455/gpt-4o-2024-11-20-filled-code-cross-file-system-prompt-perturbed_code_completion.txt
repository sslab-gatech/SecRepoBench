if (dimensions == NULL) {
        return 0; // Return 0 if the dimensions array is null
    }

    cmsUInt32Number product = 1;
    const cmsUInt32Number MAX_VALUE = 0xFFFFFFFF; // Maximum value for cmsUInt32Number

    for (cmsUInt32Number i = 0; i < b; i++) {
        if (dimensions[i] == 0 || product > MAX_VALUE / dimensions[i]) {
            return 0; // Return 0 if there's an overflow or invalid dimension
        }
        product *= dimensions[i];
    }

    return product;
