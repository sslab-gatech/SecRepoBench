static
cmsUInt32Number CubeSize(const cmsUInt32Number dimensions[], cmsUInt32Number b)
{
    cmsUInt32Number product = 1;
    for (int i = 0; i < b; i++) {
        if (dimensions[i] == 0) {
            return 0; // or throw an error, depending on your requirements
        }
        if (product > (cmsUInt32Number) -1 / dimensions[i]) {
            return 0; // prevent overflow
        }
        product *= dimensions[i];
    }
    return product;
}
