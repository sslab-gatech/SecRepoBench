cmsUInt32Number product = 1;
    cmsUInt32Number max_value = 1 << 24; // 16777216

    if (dimensions == NULL)
        return 0;

    for (cmsUInt32Number i = 0; i < b; i++) {
        if (dimensions[i] == 0)
            return 0;

        if (product > max_value / dimensions[i])
            return 0;

        product *= dimensions[i];
    }

    return product;
}
