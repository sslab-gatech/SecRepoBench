if (dimensions == NULL) {
        return 0; // Return 0 if the dimensions array is null
    }

    cmsUInt32Number product = 1;
    const cmsUInt32Number MAX_VALUE = UINT32_MAX; // Define the maximum allowable value

    for (cmsUInt32Number i = 0; i < b; i++) {
        if (dimensions[i] == 0) {
            return 0; // Return 0 if any dimension is zero
        }

        // Check for overflow before multiplying
        if (product > MAX_VALUE / dimensions[i]) {
            return 0; // Return 0 if the multiplication would overflow
        }

        product *= dimensions[i];
    }

    return product;
