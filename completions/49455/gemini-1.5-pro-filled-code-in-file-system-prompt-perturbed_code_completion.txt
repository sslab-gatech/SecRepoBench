cmsUInt32Number nTotalPoints = 1;
    if (dimensions == NULL || b == 0) {
        return 0;
    }

    for (cmsUInt32Number i = 0; i < b; i++) {
        if (dimensions[i] == 0) return 0; // Avoid multiplication by zero

        // Check for potential overflow during multiplication
        if (nTotalPoints > UINT_MAX / dimensions[i]) {
            return 0; // Indicate overflow error
        }
        nTotalPoints *= dimensions[i];
    }

    return nTotalPoints;
