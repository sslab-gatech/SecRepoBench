if (dimensions == NULL) {
        return 0;
    }

    const cmsUInt32Number MAX_VALUE = UINT32_MAX;
    cmsUInt32Number product = 1;

    for (cmsUInt32Number i = 0; i < b; i++) {
        if (dimensions[i] == 0 || product > MAX_VALUE / dimensions[i]) {
            return 0;
        }
        product *= dimensions[i];
    }

    return product;
