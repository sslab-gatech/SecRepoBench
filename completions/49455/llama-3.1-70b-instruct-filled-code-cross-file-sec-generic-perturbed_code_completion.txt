cmsUInt32Number product = 1;
cmsUInt32Number maxProduct = 0x7FFFFFFF; // Maximum value to prevent overflow

if (dimensions == NULL) {
    return 0; // Return 0 if dimensions array is null
}

for (cmsUInt32Number i = 0; i < b; i++) {
    if (dimensions[i] == 0) {
        return 0; // Return 0 if any dimension is 0
    }

    // Check for potential overflow
    if (product > maxProduct / dimensions[i]) {
        return 0; // Return 0 if overflow occurs
    }

    product *= dimensions[i];
}

return product;
