if (dimensions == NULL) return 0;

cmsUInt32Number product = 1;
cmsUInt32Number i;
for (i = 0; i < b; i++) {

    // Check for zero dimension
    if (dimensions[i] == 0)
        return 0;

    // Check for potential overflow: using the maximum value representable by cmsUInt32Number
    if (product > ((cmsUInt32Number)~0) / dimensions[i])
        return 0;

    product *= dimensions[i];
}

return product;
