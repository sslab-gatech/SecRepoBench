static
cmsUInt32Number CubeSize(const cmsUInt32Number dimensions[], cmsUInt32Number b)
{
    cmsUInt32Number product = 1;
    for (cmsUInt32Number i = 0; i < b; i++) {
        if (dimensions[i] == 0) {
            return 0; // error: zero dimension
        }
        if (product > MAX_CMS_CUBE_SIZE) {
            return 0; // error: product exceeds maximum value
        }
        product *= dimensions[i];
    }
    return product;
}
