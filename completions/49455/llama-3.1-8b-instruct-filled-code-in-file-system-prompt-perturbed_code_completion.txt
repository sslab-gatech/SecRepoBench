static
cmsUInt32Number CubeSize(const cmsUInt32Number dimensions[], cmsUInt32Number b)
{
    cmsUInt32Number n = 1;
    cmsUInt32Number i;

    for (i=0; i < b; i++) {
        n *= dimensions[i];
        if (n > UINT_MAX / dimensions[i]) {
            n = 0;
            break;
        }
    }

    return n;
}
