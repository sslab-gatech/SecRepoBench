{
    cmsUInt32Number i;
    cmsUInt32Number result = 1;

    if (dimensions == NULL) {
        return 0;
    }

    for (i = 0; i < b; i++) {
        if (dimensions[i] > (cmsUInt32Number) (0xFFFFFFFF / result)) {

        }

    }
