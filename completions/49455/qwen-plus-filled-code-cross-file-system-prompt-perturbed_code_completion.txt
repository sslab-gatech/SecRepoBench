if (dimensions == NULL) return 0;

cmsUInt32Number product = 1;
cmsUInt32Number maxAllowedValue = 0xFFFFFFFF; // Maximum allowed value for cmsUInt32Number

for(cmsUInt32Number i = 0; i < b; ++i){
    if(dimensions[i] > maxAllowedValue / product){
        return 0; // Return zero in case of overflow or error
    }
    product *= dimensions[i];
}

return product;
