if (dimensions == NULL) {
        return 0;
    }

    cmsUInt32Number product = 1;
    for (cmsUInt32Number i = 0; i < b; i++) {
        if (dimensions[i] == 0 || product > UINT32_MAX / dimensions[i]) {
            return 0; // Prevent overflow
        }
        product *= dimensions[i];
    }

    return product;
