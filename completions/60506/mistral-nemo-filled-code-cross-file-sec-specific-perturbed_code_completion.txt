def _dwarf_read_unaligned_ck_wrapper(dbg, value_ptr, line_ptr, size, line_ptr_end, error):
    # Check if the line_ptr is within the bounds of the section
    if line_ptr + size > line_ptr_end:
        _dwarf_error_string(dbg, error, DW_DLE_LINE_TABLE_BAD,
            "DW_DLE_LINE_TABLE_BAD: Reading unaligned data, "
            "off the end of the line table section")
        return DW_DLV_ERROR

    # Read the value from the line_ptr using the specified size
    value_ptr[0] = line_ptr[:size]

    # Return DW_DLV_OK if the read was successful
    return DW_DLV_OK
