if (version == EXPERIMENTAL_LINE_TABLES_VERSION) {
    /* The experimental two-level line table header is followed by a 4-byte magic sequence: "LINA" */
    static const char exp_magic[4] = { 'L', 'I', 'N', 'A' };
    if ((line_ptr_end - line_cursor) < 4) {
        _dwarf_error_string(dbg, err, DW_DLE_LINE_NUMBER_HEADER_ERROR,
            "DW_DLE_LINE_NUMBER_HEADER_ERROR: "
            "Experimental line table version: not enough bytes for magic 'LINA'");
        return DW_DLV_ERROR;
    }
    if (memcmp(line_cursor, exp_magic, 4) != 0) {
        _dwarf_error_string(dbg, err, DW_DLE_LINE_NUMBER_HEADER_ERROR,
            "DW_DLE_LINE_NUMBER_HEADER_ERROR: "
            "Experimental line table version: missing magic 'LINA'");
        return DW_DLV_ERROR;
    }
    line_cursor += 4;
    /* Read logicals_table_offset (ULEB128) */
    {
        Dwarf_Unsigned logicals_offset = 0;
        int resl = read_uword_de(&line_cursor, &logicals_offset,
            dbg, err, line_ptr_end);
        if (resl != DW_DLV_OK) {
            _dwarf_error_string(dbg, err, DW_DLE_LINE_NUMBER_HEADER_ERROR,
                "DW_DLE_LINE_NUMBER_HEADER_ERROR: "
                "Failed to read logicals_table_offset in experimental line table");
            return resl;
        }
        line_context->lc_logicals_table_offset = logicals_offset;
    }
    /* Read actuals_table_offset (ULEB128) */
    {
        Dwarf_Unsigned actuals_offset = 0;
        int resa = read_uword_de(&line_cursor, &actuals_offset,
            dbg, err, line_ptr_end);
        if (resa != DW_DLV_OK) {
            _dwarf_error_string(dbg, err, DW_DLE_LINE_NUMBER_HEADER_ERROR,
                "DW_DLE_LINE_NUMBER_HEADER_ERROR: "
                "Failed to read actuals_table_offset in experimental line table");
            return resa;
        }
        line_context->lc_actuals_table_offset = actuals_offset;
    }
    if (line_cursor > line_ptr_end) {
        _dwarf_error_string(dbg, err, DW_DLE_LINE_NUMBER_HEADER_ERROR,
            "DW_DLE_LINE_NUMBER_HEADER_ERROR: "
            "Experimental line table header: ran off end after reading offsets");
        return DW_DLV_ERROR;
    }
}
