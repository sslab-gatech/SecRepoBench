if (version == EXPERIMENTAL_LINE_TABLES_VERSION) {
    int dres = 0;
    /* Read the logicals_table_offset */
    dres = read_uword_de(&line_cursor, &line_context->lc_logicals_table_offset,
                         dbg, err, line_ptr_end);
    if (dres != DW_DLV_OK) {
        return dres;
    }
    if (line_cursor >= line_ptr_end) {
        _dwarf_error(dbg, err, DW_DLE_DEBUG_LINE_LENGTH_BAD);
        return DW_DLV_ERROR;
    }
    /* Read the actuals_table_offset */
    dres = read_uword_de(&line_cursor, &line_context->lc_actuals_table_offset,
                         dbg, err, line_ptr_end);
    if (dres != DW_DLV_OK) {
        return dres;
    }
    if (line_cursor > line_ptr_end) {
        _dwarf_error(dbg, err, DW_DLE_DEBUG_LINE_LENGTH_BAD);
        return DW_DLV_ERROR;
    }
}
