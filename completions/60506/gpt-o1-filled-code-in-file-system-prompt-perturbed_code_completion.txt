if (version == EXPERIMENTAL_LINE_TABLES_VERSION) {
        res = read_uword_de(&line_cursor,
            &line_context->lc_logicals_table_offset,
            dbg, err, line_ptr_end);
        if (res == DW_DLV_ERROR) {
            return res;
        }
        res = read_uword_de(&line_cursor,
            &line_context->lc_actuals_table_offset,
            dbg, err, line_ptr_end);
        if (res == DW_DLV_ERROR) {
            return res;
        }
        if (line_cursor > line_ptr_end) {
            _dwarf_error_string(dbg, err, DW_DLE_LINE_OFFSET_BAD,
                "DW_DLE_LINE_OFFSET_BAD: "
                "reading logicals/actuals table offsets "
                "exceeds line table end.");
            return DW_DLV_ERROR;
        }
    }
