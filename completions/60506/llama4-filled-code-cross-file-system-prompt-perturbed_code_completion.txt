if (version == EXPERIMENTAL_LINE_TABLES_VERSION) {
    Dwarf_Unsigned logicals_table_offset = 0;
    Dwarf_Unsigned actuals_table_offset = 0;

    res = read_uword_de(&line_cursor, &logicals_table_offset,
        dbg, err, line_ptr_end);
    if (res != DW_DLV_OK) {
        return res;
    }
    line_context->lc_logicals_table_offset =
        logicals_table_offset;
    res = read_uword_de(&line_cursor, &actuals_table_offset,
        dbg, err, line_ptr_end);
    if (res != DW_DLV_OK) {
        return res;
    }
    line_context->lc_actuals_table_offset =
        actuals_table_offset;
    if (line_cursor + 2 * sizeof(Dwarf_Unsigned) > line_ptr_end) {
        _dwarf_error(dbg, err, DW_DLE_LINE_OFFSET_BAD);
        return DW_DLV_ERROR;
    }
}
