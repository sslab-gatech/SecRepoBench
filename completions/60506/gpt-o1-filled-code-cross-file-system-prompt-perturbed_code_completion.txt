if (version == EXPERIMENTAL_LINE_TABLES_VERSION) {
    Dwarf_Unsigned ltoff = 0;
    Dwarf_Unsigned aoff = 0;
    int lres = 0;

    /* Read the logicals_table_offset */
    lres = read_uword_de(&line_cursor,&ltoff,dbg,err,line_ptr_end);
    if (lres != DW_DLV_OK) {
        return lres;
    }

    /* Read the actuals_table_offset */
    lres = read_uword_de(&line_cursor,&aoff,dbg,err,line_ptr_end);
    if (lres != DW_DLV_OK) {
        return lres;
    }

    /* Store these offsets in the line_context */
    line_context->lc_logicals_table_offset = ltoff;
    line_context->lc_actuals_table_offset = aoff;

    /* Ensure we have not exceeded the valid data range */
    if (line_cursor > line_ptr_end) {
        _dwarf_error(dbg, err, DW_DLE_DEBUG_LINE_LENGTH_BAD);
        return DW_DLV_ERROR;
    }
}
