if (version == EXPERIMENTAL_LINE_TABLES_VERSION) {
    int lres = 0;
    Dwarf_Unsigned temp_offset = 0;

    /* Verify and read logicals_table_offset */
    lres = read_uword_de(&line_cursor, &temp_offset, dbg, err, line_ptr_end);
    if (lres != DW_DLV_OK) {
        return lres;
    }
    line_context->lc_logicals_table_offset = temp_offset;

    /* Verify and read actuals_table_offset */
    lres = read_uword_de(&line_cursor, &temp_offset, dbg, err, line_ptr_end);
    if (lres != DW_DLV_OK) {
        return lres;
    }
    line_context->lc_actuals_table_offset = temp_offset;

    /* Ensure we do not advance past the end of the section data */
    if (line_cursor > line_ptr_end) {
        _dwarf_error(dbg, err, DW_DLE_DEBUG_LINE_LENGTH_BAD);
        return DW_DLV_ERROR;
    }
}
