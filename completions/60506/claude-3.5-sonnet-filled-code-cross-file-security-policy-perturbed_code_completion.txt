if (version == EXPERIMENTAL_LINE_TABLES_VERSION) {
    Dwarf_Small *expected_sequence = (Dwarf_Small *)"DWARF EXPERIMENT";
    size_t expected_length = strlen((char *)expected_sequence);
    
    if (line_ptr + expected_length > line_ptr_end) {
        _dwarf_error(dbg, error, DW_DLE_LINE_TABLE_BAD);
        return DW_DLV_ERROR;
    }
    
    if (memcmp(line_ptr, expected_sequence, expected_length) != 0) {
        _dwarf_error(dbg, error, DW_DLE_LINE_TABLE_BAD);
        return DW_DLV_ERROR;
    }
    
    line_ptr += expected_length;
    
    if (line_ptr + 2 * sizeof(Dwarf_Unsigned) > line_ptr_end) {
        _dwarf_error(dbg, error, DW_DLE_LINE_TABLE_BAD);
        return DW_DLV_ERROR;
    }
    
    READ_UNALIGNED_CK(dbg, line_context->lc_logicals_table_offset,
        Dwarf_Unsigned, line_ptr, sizeof(Dwarf_Unsigned), error, line_ptr_end);
    line_ptr += sizeof(Dwarf_Unsigned);
    
    READ_UNALIGNED_CK(dbg, line_context->lc_actuals_table_offset,
        Dwarf_Unsigned, line_ptr, sizeof(Dwarf_Unsigned), error, line_ptr_end);
    line_ptr += sizeof(Dwarf_Unsigned);
    
    if (line_ptr > line_ptr_end) {
        _dwarf_error(dbg, error, DW_DLE_LINE_TABLE_BAD);
        return DW_DLV_ERROR;
    }
}
