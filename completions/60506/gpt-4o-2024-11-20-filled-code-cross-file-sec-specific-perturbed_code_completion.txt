if (version == EXPERIMENTAL_LINE_TABLES_VERSION) {
    if (line_cursor + 2 * sizeof(Dwarf_Unsigned) > line_ptr_end) {
        _dwarf_error(dbg, err, DW_DLE_LINE_OFFSET_BAD);
        return DW_DLV_ERROR;
    }

    /* Read logicals_table_offset */
    res = read_uword_de(&line_cursor, &line_context->lc_logicals_table_offset, dbg, err, line_ptr_end);
    if (res != DW_DLV_OK) {
        return res;
    }

    /* Read actuals_table_offset */
    res = read_uword_de(&line_cursor, &line_context->lc_actuals_table_offset, dbg, err, line_ptr_end);
    if (res != DW_DLV_OK) {
        return res;
    }

    /* Ensure the line pointer does not exceed the end of the section */
    if (line_cursor > line_ptr_end) {
        _dwarf_error(dbg, err, DW_DLE_LINE_OFFSET_BAD);
        return DW_DLV_ERROR;
    }
}
