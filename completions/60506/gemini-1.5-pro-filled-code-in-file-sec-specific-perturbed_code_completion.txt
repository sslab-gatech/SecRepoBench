if (version == EXPERIMENTAL_LINE_TABLES_VERSION) {
        Dwarf_Unsigned log_offset = 0;
        Dwarf_Unsigned act_offset = 0;
        int resoff = 0;

        resoff = read_uword_de(&line_cursor, &log_offset,
            dbg, err, line_ptr_end);
        if (resoff != DW_DLV_OK) {
            return resoff;
        }
        if (log_offset > total_length) {
            IssueExpError(dbg, err,
                "Logicals offset too large to be real", log_offset);
            return DW_DLV_ERROR;
        }

        resoff = read_uword_de(&line_cursor, &act_offset,
            dbg, err, line_ptr_end);
        if (resoff != DW_DLV_OK) {
            return resoff;
        }
        if (act_offset > total_length) {
            IssueExpError(dbg, err,
                "Actuals offset too large to be real", act_offset);
            return DW_DLV_ERROR;
        }

        line_context->lc_logicals_table_offset = log_offset;
        line_context->lc_actuals_table_offset = act_offset;
    }
