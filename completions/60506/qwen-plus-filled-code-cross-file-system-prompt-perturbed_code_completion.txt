if (version == EXPERIMENTAL_LINE_TABLES_VERSION) {
    Dwarf_Small expected_sequence[] = {0x1, 0x2, 0x3, 0x4, 0x5};
    size_t expected_length = sizeof(expected_sequence);
    
    if (line_cursor + expected_length > line_ptr_end) {
        _dwarf_error(dbg, err, DW_DLE_LINE_OFFSET_BAD);
        return DW_DLV_ERROR;
    }

    if (memcmp(line_cursor, expected_sequence, expected_length) != 0) {
        dwarfstring m;

        dwarfstring_constructor(&m);
        dwarfstring_append_printf_u(&m,
            "DW_DLE_DEBUG_LINE_SEQUENCE_BAD "
            " the line table sequence does not match expected pattern.");
        _dwarf_error_string(dbg, err, DW_DLE_DEBUG_LINE_SEQUENCE_BAD,
            dwarfstring_string(&m));
        dwarfstring_destructor(&m);
        return DW_DLV_ERROR;
    }
    
    line_cursor += expected_length;

    READ_UNALIGNED_CK(dbg, line_context->lc_logicals_table_offset, Dwarf_Unsigned,
        line_cursor, local_length_size,
        err,line_ptr_end);

    line_cursor += local_length_size;

    READ_UNALIGNED_CK(dbg, line_context->lc_actuals_table_offset, Dwarf_Unsigned,
        line_cursor, local_length_size,
        err,line_ptr_end);

    line_cursor += local_length_size;

    if (line_cursor > line_ptr_end) {
        _dwarf_error(dbg, err, DW_DLE_LINE_OFFSET_BAD);
        return DW_DLV_ERROR;
    }
}
