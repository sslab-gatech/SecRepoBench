if (version == EXPERIMENTAL_LINE_TABLES_VERSION) {
    Dwarf_Unsigned logicals_table_offset = 0;
    Dwarf_Unsigned actuals_table_offset = 0;
    int res = 0;
    
    // Read logicals_table_offset
    res = read_uword_de(&line_ptr, &logicals_table_offset, dbg, err, line_ptr_end);
    if (res != DW_DLV_OK) {
        return res;
    }
    
    if (logicals_table_offset > total_length) {
        IssueExpError(dbg, err, "Logicals table offset too large to be real", logicals_table_offset);
        return DW_DLV_ERROR;
    }
    line_context->lc_logicals_table_offset = logicals_table_offset;
    
    // Read actuals_table_offset
    res = read_uword_de(&line_ptr, &actuals_table_offset, dbg, err, line_ptr_end);
    if (res != DW_DLV_OK) {
        return res;
    }
    
    if (actuals_table_offset > total_length) {
        IssueExpError(dbg, err, "Actuals table offset too large to be real", actuals_table_offset);
        return DW_DLV_ERROR;
    }
    line_context->lc_actuals_table_offset = actuals_table_offset;
    
    // Check for buffer overrun
    if (line_ptr >= line_ptr_end) {
        _dwarf_error(dbg, err, DW_DLE_LINE_NUMBER_HEADER_ERROR);
        return DW_DLV_ERROR;
    }
}
