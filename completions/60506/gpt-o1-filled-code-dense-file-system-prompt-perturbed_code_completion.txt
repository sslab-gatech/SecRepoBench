if (version == EXPERIMENTAL_LINE_TABLES_VERSION) {
    Dwarf_Unsigned logicals_off = 0;
    Dwarf_Unsigned actuals_off = 0;

    /* Check that we can read two offsets. The calls below
       also do boundary checks, so no extra check needed here. */
    res = read_uword_de(&line_cursor, &logicals_off,
        dbg, err, line_ptr_end);
    if (res != DW_DLV_OK) {
        return res;
    }
    line_context->lc_logicals_table_offset = logicals_off;

    res = read_uword_de(&line_cursor, &actuals_off,
        dbg, err, line_ptr_end);
    if (res != DW_DLV_OK) {
        return res;
    }
    line_context->lc_actuals_table_offset = actuals_off;

    /* Verify we are still within the line table. */
    if (line_cursor > line_ptr_end) {
        _dwarf_error_string(dbg, err, DW_DLE_DEBUG_LINE_LENGTH_BAD,
            "Reading EXPerimental line-table offsets goes beyond end of table");
        return DW_DLV_ERROR;
    }
} else {
    line_context->lc_logicals_table_offset = 0;
    line_context->lc_actuals_table_offset = 0;
}
