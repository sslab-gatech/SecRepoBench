        line_context->lc_logicals_table_offset =
            *((Dwarf_Unsigned *) line_cursor);

        /* Update the line pointer. */
        line_cursor += sizeof(Dwarf_Unsigned);

        /* Read the actuals_table_offset. */
        line_context->lc_actuals_table_offset =
            *((Dwarf_Unsigned *) line_cursor);

        /* Update the line pointer. */
        line_cursor += sizeof(Dwarf_Unsigned);

        /* Ensure that the line pointer does not exceed the end. */
        if (line_cursor > line_ptr_end) {
