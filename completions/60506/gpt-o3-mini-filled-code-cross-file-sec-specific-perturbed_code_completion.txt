if (version == EXPERIMENTAL_LINE_TABLES_VERSION) {
    /* Ensure there is enough room for two offsets */
    if (line_cursor + 2 * local_length_size > line_ptr_end) {
        _dwarf_error(dbg, err, DW_DLE_DEBUG_LINE_LENGTH_BAD);
        return DW_DLV_ERROR;
    }
    res = _dwarf_read_unaligned_ck_wrapper(dbg,
            &line_context->lc_logicals_table_offset,
            line_cursor,
            local_length_size,
            line_ptr_end,
            err);
    if (res == DW_DLV_ERROR) {
        return res;
    }
    line_cursor += local_length_size;

    res = _dwarf_read_unaligned_ck_wrapper(dbg,
            &line_context->lc_actuals_table_offset,
            line_cursor,
            local_length_size,
            line_ptr_end,
            err);
    if (res == DW_DLV_ERROR) {
        return res;
    }
    line_cursor += local_length_size;
}
