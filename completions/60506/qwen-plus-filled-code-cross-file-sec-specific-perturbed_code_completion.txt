if (version == EXPERIMENTAL_LINE_TABLES_VERSION) {
    Dwarf_Unsigned expected_logical_offset = line_cursor - section_start;
    Dwarf_Unsigned expected_actuals_offset = 0;

    /* Verify that the line pointer matches the expected sequence of bytes */
    if (*(line_cursor) != 'L' || *(line_cursor + 1) != 'O' ||
        *(line_cursor + 2) != 'G' || *(line_cursor + 3) != '_') {
        _dwarf_error_string(dbg, err, DW_DLE_LINE_OFFSET_BAD,
            "DW_DLE_LINE_OFFSET_BAD "
            "logicals table marker mismatch");
        return DW_DLV_ERROR;
    }
    line_cursor += 4;

    /* Read logicals_table_offset */
    READ_AREA_LENGTH_CK(dbg, line_context->lc_logicals_table_offset, Dwarf_Unsigned,
        line_cursor, local_length_size, local_extension_size,
        err, section_length, section_end);
    if (line_context->lc_logicals_table_offset != expected_logical_offset) {
        _dwarf_error_string(dbg, err, DW_DLE_LINE_OFFSET_BAD,
            "DW_DLE_LINE_OFFSET_BAD "
            "logicals table offset mismatch");
        return DW_DLV_ERROR;
    }

    /* Read actuals_table_offset */
    if (line_cursor >= line_ptr_end) {
        _dwarf_error(dbg, err, DW_DLE_LINE_OFFSET_BAD);
        return DW_DLV_ERROR;
    }
    READ_AREA_LENGTH_CK(dbg, expected_actuals_offset, Dwarf_Unsigned,
        line_cursor, local_length_size, local_extension_size,
        err, section_length, section_end);
    line_context->lc_actuals_table_offset = expected_actuals_offset;

    /* Update line pointer */
    line_cursor = line_cursor + local_length_size + local_extension_size;

    /* Ensure line pointer does not exceed line_ptr_end */
    if (line_cursor > line_ptr_end) {
        _dwarf_error(dbg, err, DW_DLE_LINE_OFFSET_BAD);
        return DW_DLV_ERROR;
    }
}
