if (version == EXPERIMENTAL_LINE_TABLES_VERSION) {
    /* For experimental two-level line tables, 
       read the offsets to the logical and actual tables */
    
    /* Check if we have enough bytes for the two offsets */
    if (line_cursor + (2 * local_length_size) > line_ptr_end) {
        _dwarf_error(dbg, err, DW_DLE_LINE_NUMBER_HEADER_ERROR);
        return DW_DLV_ERROR;
    }
    
    /* Read the logicals_table_offset */
    READ_UNALIGNED_CK(dbg, line_context->lc_logicals_table_offset, 
        Dwarf_Unsigned, line_cursor, local_length_size,
        err, line_ptr_end);
    line_cursor += local_length_size;
    
    /* Read the actuals_table_offset */
    READ_UNALIGNED_CK(dbg, line_context->lc_actuals_table_offset, 
        Dwarf_Unsigned, line_cursor, local_length_size,
        err, line_ptr_end);
    line_cursor += local_length_size;
    
    /* Ensure we haven't gone past the end of the buffer */
    if (line_cursor > line_ptr_end) {
        _dwarf_error(dbg, err, DW_DLE_LINE_NUMBER_HEADER_ERROR);
        return DW_DLV_ERROR;
    }
}
