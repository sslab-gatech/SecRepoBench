if (version == EXPERIMENTAL_LINE_TABLES_VERSION) {
        /* Verify that the line pointer matches the expected sequence of bytes for this version. */
        if ((line_cursor[0] != 0x01) || (line_cursor[1] != 0x02) ||
            (line_cursor[2] != 0x03) || (line_cursor[3] != 0x04)) {
            _dwarf_error(dbg, err, DW_DLE_LINE_NUMBER_HEADER_ERROR);
            return DW_DLV_ERROR;
        }
        line_cursor += 4;

        /* Read the logicals_table_offset and actuals_table_offset from the data. */
        READ_UNALIGNED_CK(dbg, line_context->lc_logicals_table_offset,
            Dwarf_Unsigned, line_cursor, local_length_size,
            err,line_ptr_end);
        line_cursor += local_length_size;

        READ_UNALIGNED_CK(dbg, line_context->lc_actuals_table_offset,
            Dwarf_Unsigned, line_cursor, local_length_size,
            err,line_ptr_end);
        line_cursor += local_length_size;

        /* Ensure that the line pointer does not exceed the line pointer end after the updates. */
        if (line_cursor > line_ptr_end) {
            _dwarf_error(dbg, err, DW_DLE_LINE_OFFSET_BAD);
            return DW_DLV_ERROR;
        }
    }
