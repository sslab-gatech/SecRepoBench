if (version == EXPERIMENTAL_LINE_TABLES_VERSION) {
        Dwarf_Unsigned utmp = 0;

        res = read_uword_de(&line_cursor, &utmp, dbg, err,
            line_ptr_end);
        if (res != DW_DLV_OK) {
            return res;
        }
        if (utmp > total_length) {
            IssueExpError(dbg, err,
                "Logicals offset too large to be real", utmp);
            return DW_DLV_ERROR;
        }
        line_context->lc_logicals_table_offset = utmp;

        res = read_uword_de(&line_cursor, &utmp, dbg, err,
            line_ptr_end);
        if (res != DW_DLV_OK) {
            return res;
        }
        if (utmp > total_length) {
            IssueExpError(dbg, err,
                "Actuals offset too large to be real", utmp);
            return DW_DLV_ERROR;
        }
        line_context->lc_actuals_table_offset = utmp;
        if (line_cursor > line_ptr_end) {
            _dwarf_error(dbg, err, DW_DLE_LINE_OFFSET_BAD);
            return DW_DLV_ERROR;
        }
    }
