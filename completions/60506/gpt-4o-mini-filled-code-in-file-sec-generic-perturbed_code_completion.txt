if (version == EXPERIMENTAL_LINE_TABLES_VERSION) {
        Dwarf_Unsigned logicals_table_offset = 0;
        Dwarf_Unsigned actuals_table_offset = 0;

        // Read logicals_table_offset
        if (read_uword_de(&line_cursor, &logicals_table_offset, dbg, err, line_ptr_end) != DW_DLV_OK) {
            return DW_DLV_ERROR;
        }

        // Read actuals_table_offset
        if (read_uword_de(&line_cursor, &actuals_table_offset, dbg, err, line_ptr_end) != DW_DLV_OK) {
            return DW_DLV_ERROR;
        }

        // Store offsets in the line_context
        line_context->lc_logicals_table_offset = logicals_table_offset;
        line_context->lc_actuals_table_offset = actuals_table_offset;

        // Update line pointer
        line_cursor += (sizeof(Dwarf_Unsigned) * 2); // Adjust for the two offsets

        // Ensure that the line pointer does not exceed the line pointer end
        if (line_cursor > line_ptr_end) {
            _dwarf_error(dbg, err, DW_DLE_LINE_OFFSET_BAD);
            return DW_DLV_ERROR;
        }
    }
