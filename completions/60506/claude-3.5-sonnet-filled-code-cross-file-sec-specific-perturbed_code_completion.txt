if (version == EXPERIMENTAL_LINE_TABLES_VERSION) {
    Dwarf_Small *expected_sequence = (Dwarf_Small *)"\1\0\0\0\2\0\0\0";
    if (memcmp(line_ptr, expected_sequence, 8) != 0) {
        _dwarf_error_string(dbg, error, DW_DLE_LINE_TABLE_BAD,
            "DW_DLE_LINE_TABLE_BAD: Unexpected sequence in "
            "EXPERIMENTAL_LINE_TABLES_VERSION");
        return DW_DLV_ERROR;
    }
    line_ptr += 8;

    if (line_ptr + 8 > line_ptr_end) {
        _dwarf_error_string(dbg, error, DW_DLE_LINE_TABLE_BAD,
            "DW_DLE_LINE_TABLE_BAD: Not enough space for "
            "logicals and actuals table offsets");
        return DW_DLV_ERROR;
    }

    READ_UNALIGNED_CK(dbg, line_context->lc_logicals_table_offset,
        Dwarf_Unsigned, line_ptr, 4, error, line_ptr_end);
    line_ptr += 4;

    READ_UNALIGNED_CK(dbg, line_context->lc_actuals_table_offset,
        Dwarf_Unsigned, line_ptr, 4, error, line_ptr_end);
    line_ptr += 4;

    if (line_ptr > line_ptr_end) {
        _dwarf_error_string(dbg, error, DW_DLE_LINE_TABLE_BAD,
            "DW_DLE_LINE_TABLE_BAD: Line pointer went past "
            "the end after reading table offsets");
        return DW_DLV_ERROR;
    }
}
