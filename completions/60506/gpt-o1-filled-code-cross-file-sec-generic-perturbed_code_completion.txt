if (version == EXPERIMENTAL_LINE_TABLES_VERSION) {
        Dwarf_Unsigned tmp_offset = 0;

        /* Read the logicals table offset. */
        res = read_uword_de(&line_cursor,&tmp_offset,dbg,err,line_ptr_end);
        if (res != DW_DLV_OK) {
            return res;
        }
        line_context->lc_logicals_table_offset = tmp_offset;

        /* Read the actuals table offset. */
        res = read_uword_de(&line_cursor,&tmp_offset,dbg,err,line_ptr_end);
        if (res != DW_DLV_OK) {
            return res;
        }
        line_context->lc_actuals_table_offset = tmp_offset;

        /* Ensure we have not exceeded the line table end. */
        if (line_cursor > line_ptr_end) {
            _dwarf_error(dbg, err, DW_DLE_DEBUG_LINE_LENGTH_BAD);
            return DW_DLV_ERROR;
        }
    }
