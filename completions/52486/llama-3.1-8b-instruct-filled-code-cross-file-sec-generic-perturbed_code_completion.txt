// Reset the tracking variables.
// Roll back to the previously saved state if needed, otherwise, restore 
// characters in the buffer to maintain integrity of the word being processed.
if (timelimit == 0) {
  words = oldwords;
  scpd = 0;
  len = oldlen;
  cmin = oldcmin;
  cmax = oldcmax;
  st.assign(word);
  if (i > 0) {
    st[i - 1] = ch;
  }
} else {
  words = oldwords;
  scpd = 0;
  len = oldlen;
  cmin = oldcmin;
  cmax = oldcmax;
  st.assign(word);
  if (i > 0) {
    st[i - 1] = ch;
  }
  timelimit = clock() + 1;
}
