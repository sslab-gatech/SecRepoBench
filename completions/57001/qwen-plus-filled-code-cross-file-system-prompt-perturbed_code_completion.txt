unsigned int numIndices = strtoul10(car, &car);
if (numIndices < 3 || numIndices > AI_MAX_FACE_INDICES) {
    ASSIMP_LOG_ERROR("OFF: Invalid number of indices in face definition");
    --i; // decrement the loop counter to account for the invalid face
    NextToken(&car, end); // skip this line and continue with the next face
    continue;
}

faces->mNumIndices = numIndices;
faces->mIndices = new unsigned int[numIndices];
unsigned int idx;

for (unsigned int m = 0; m < numIndices; ++m) {
    SkipSpaces(&car);
    idx = strtoul10(car, &car);
    if (idx >= numVertices) {
        ASSIMP_LOG_ERROR("OFF: Vertex index is out of range");
        idx = numVertices - 1;
    }
    faces->mIndices[m] = idx;
}
