unsigned int indexCount = strtoul10(car, &car);
if (indexCount < 3) {
    ASSIMP_LOG_ERROR("OFF: Invalid face with less than 3 indices");
    continue;
}
if (indexCount > AI_MAX_VERTICES_PER_FACE) {
    ASSIMP_LOG_ERROR("OFF: Face with more indices than supported");
    indexCount = AI_MAX_VERTICES_PER_FACE;
}
faces->mNumIndices = indexCount;
faces->mIndices = new unsigned int[indexCount];
unsigned int idx;
for (unsigned int m = 0; m < indexCount; ++m) {
    SkipSpaces(&vertex_data_ptr);
    idx = strtoul10(vertex_data_ptr, &vertex_data_ptr);
    if (idx >= numVertices) {
        ASSIMP_LOG_ERROR("OFF: Vertex index is out of range");
        idx = numVertices - 1;
    }
    faces->mIndices[m] = idx;
}
