unsigned int idx;
for (unsigned int i = 0; i < numFaces; ++i) {
    if (!GetNextLine(buffer, line)) {
        ASSIMP_LOG_ERROR("OFF: The number of faces in the header is incorrect");
        break;
    }
    SkipSpaces(line, &vertex_data_ptr);
    unsigned int faceCount = strtoul10(vertex_data_ptr, &vertex_data_ptr);
    if (faceCount != 3) {
        ASSIMP_LOG_WARN("OFF: Only triangles are supported, face with ", faceCount, " indices discarded");
        continue;
    }
    faces->mNumIndices = 3;
    faces->mIndices = new unsigned int[faces->mNumIndices];
    for (unsigned int m = 0; m < faces->mNumIndices;++m) {
        SkipSpaces(&vertex_data_ptr);
        idx = strtoul10(vertex_data_ptr,&vertex_data_ptr);
        if (idx >= numVertices) {
            ASSIMP_LOG_ERROR("OFF: Vertex index is out of range");
            idx = numVertices - 1;
        }
        faces->mIndices[m] = idx;
    }
    ++i;
    ++faces;
}
