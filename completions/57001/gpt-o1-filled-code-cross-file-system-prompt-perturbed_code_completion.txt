SkipSpaces(&vertex_data_ptr);
unsigned int faceIndexCount = strtoul10(vertex_data_ptr, &vertex_data_ptr);

// Check if face has at least 3 vertices
if (faceIndexCount < 3) {
    ASSIMP_LOG_ERROR("OFF: Invalid face with fewer than 3 vertices encountered, skipping face");
    --numFaces;
    --mesh->mNumFaces;
    ++i;
    ++faces;
    continue;
}

faces->mNumIndices = faceIndexCount;
