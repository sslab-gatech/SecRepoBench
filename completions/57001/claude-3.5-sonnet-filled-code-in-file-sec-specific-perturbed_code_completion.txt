if (!GetNextLine(buffer, line)) {
    ASSIMP_LOG_ERROR("OFF: The number of faces in the header is incorrect");
    break;
}
vertex_data_ptr = line;
SkipSpaces(&vertex_data_ptr);
unsigned int idx_count = strtoul10(vertex_data_ptr, &vertex_data_ptr);

if (idx_count < 3) {
    ASSIMP_LOG_ERROR("OFF: Invalid number of face indices");
    continue;
}

faces->mNumIndices = idx_count;
faces->mIndices = new unsigned int[idx_count];

for (unsigned int m = 0; m < idx_count; ++m) {
    SkipSpaces(&vertex_data_ptr);
    unsigned int idx = strtoul10(vertex_data_ptr, &vertex_data_ptr);
    if (idx >= numVertices) {
        ASSIMP_LOG_ERROR("OFF: Vertex index is out of range");
        idx = numVertices - 1;
    }
    faces->mIndices[m] = idx;
}
++i;
++faces;
