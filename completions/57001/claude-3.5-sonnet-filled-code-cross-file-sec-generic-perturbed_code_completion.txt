if (!GetNextLine(buffer, line)) {
    ASSIMP_LOG_ERROR("OFF: Unexpected end of file");
    break;
}
sz = line;
unsigned int idx_count = strtoul10(sz, &sz);
if (idx_count < 3) {
    ASSIMP_LOG_ERROR("OFF: Invalid face definition (less than 3 indices)");
    continue;
}
faces->mNumIndices = idx_count;
faces->mIndices = new unsigned int[idx_count];
for (unsigned int m = 0; m < idx_count; ++m) {
    SkipSpaces(&sz);
    unsigned int idx = strtoul10(sz, &sz);
    if (idx >= numVertices) {
        ASSIMP_LOG_ERROR("OFF: Vertex index out of range");
        idx = numVertices - 1;
    }
    faces->mIndices[m] = idx;
}
++i;
++faces;
