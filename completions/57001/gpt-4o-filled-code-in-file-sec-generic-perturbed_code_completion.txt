if (!GetNextLine(buffer, line)) {
        ASSIMP_LOG_ERROR("OFF: The number of faces in the header is incorrect");
        break;
    }
    vertex_data_ptr = line;
    SkipSpaces(&vertex_data_ptr);
    unsigned int numIndices = strtoul10(vertex_data_ptr, &vertex_data_ptr);

    if (numIndices < 3) {
        ASSIMP_LOG_ERROR("OFF: A face with less than 3 vertices is not valid");
        continue;
    }

    faces->mNumIndices = numIndices;
    faces->mIndices = new unsigned int[numIndices];

    for (unsigned int m = 0; m < faces->mNumIndices; ++m) {
        SkipSpaces(&vertex_data_ptr);
        unsigned int idx = strtoul10(vertex_data_ptr, &vertex_data_ptr);
        if (idx >= numVertices) {
            ASSIMP_LOG_ERROR("OFF: Vertex index is out of range");
            idx = numVertices - 1;
        }
        faces->mIndices[m] = idx;
    }
    ++i;
    ++faces;
