// Read the number of indices for the current face
        unsigned int numIndices = strtoul10(vertex_data_ptr, &vertex_data_ptr);
        if (numIndices < 3) {
            ASSIMP_LOG_ERROR("OFF: Faces with less than 3 vertices are not supported");
            continue;
        }

        // Allocate memory for the indices of the current face
        faces->mNumIndices = numIndices;
        faces->mIndices = new unsigned int[numIndices];

        // Read the indices for the current face
        for (unsigned int m = 0; m < numIndices; ++m) {
            SkipSpaces(&vertex_data_ptr);
            unsigned int idx = strtoul10(vertex_data_ptr, &vertex_data_ptr);
            if (idx >= numVertices) {
                ASSIMP_LOG_ERROR("OFF: Vertex index is out of range");
                idx = numVertices - 1; // Clamp to the last valid index
            }
            faces->mIndices[m] = idx;
        }

        // Move to the next face
        ++i;
        ++faces;
