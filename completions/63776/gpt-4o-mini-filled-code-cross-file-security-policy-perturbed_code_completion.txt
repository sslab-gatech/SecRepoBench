if (obj->size + obj->address > dat->size || 
                    obj->address < previous_byte_position || 
                    obj->size > (dat->size - (previous_byte_position - start))) 
                {
                    LOG_ERROR("Invalid object size or address for object at index %d", num);
                    error |= DWG_ERR_INVALIDDWG;
                    break;
                }
                if (dat->byte + obj->size > dat->size) 
                {
                    LOG_ERROR("Buffer overflow detected: current byte position %zu + object size %u exceeds total size %zu", dat->byte, obj->size, dat->size);
                    return DWG_ERR_INVALIDDWG;
                }
