if (bit_chain->version < R_11) {
                      obj->size = (bit_chain->byte - oldpos) & 0xFFFFFFFF;
                      LOG_TRACE(" size: %u,", obj->size);
                      if (obj->size * 8 < obj->common_size) {
                        LOG_ERROR("common_size overflow");
                        error |= DWG_ERR_VALUEOUTOFBOUNDS;
                      }
                      if (obj->size > 0 && obj->size < bit_chain->size && bit_chain->byte < bit_chain->size) {
                        size_t unknown_size = MIN(obj->size - (bit_chain->byte - oldpos), bit_chain->size - bit_chain->byte);
                        if (unknown_size > 0 && obj->size > unknown_size + (bit_chain->byte - oldpos)) {
                          LOG_ERROR("size mismatch: %u != %zu + trailing %zu", obj->size, bit_chain->byte - oldpos, unknown_size);
                          obj->unknown = bit_read_TF(bit_chain, unknown_size);
                          LOG_TRACE(" unknown: %zu,", unknown_size);
                          LOG_TRACE_TF(obj->unknown, unknown_size);
                        }
                      } else if (obj->size > bit_chain->size - (bit_chain->byte - oldpos)) {
                        LOG_ERROR("size overflow: %u > %zu", obj->size, bit_chain->size - bit_chain->byte);
                        error |= DWG_ERR_VALUEOUTOFBOUNDS;
                      }
                    } else {
                      if (obj->size != bit_chain->byte - oldpos) {
                        LOG_ERROR("size mismatch: %u != %" PRIuSIZE, obj->size, bit_chain->byte - oldpos);
                        error |= DWG_ERR_VALUEOUTOFBOUNDS;
                      }
                      if (bit_chain->byte < bit_chain->size && obj->address + obj->size > bit_chain->byte) {
                        size_t trailing_unknown_size = obj->address + obj->size - bit_chain->byte;
                        size_t unknown_offset = bit_chain->byte - obj->address;
                        if (trailing_unknown_size > 0 && trailing_unknown_size <= obj->size - unknown_offset) {
                          bit_chain->byte = obj->address + unknown_offset;
                          obj->unknown = bit_read_TF(bit_chain, trailing_unknown_size);
                          LOG_TRACE(" unknown[%zu]: %zu,", unknown_offset, trailing_unknown_size);
                          LOG_TRACE_TF(obj->unknown, trailing_unknown_size);
                          bit_chain->byte = obj->address + obj->size; // advance to next object
                        } else if (trailing_unknown_size > obj->size - unknown_offset) {
                          LOG_ERROR("trailing_unknown_size overflow: %zu > %u - %zu", trailing_unknown_size, obj->size, unknown_offset);
                          error |= DWG_ERR_VALUEOUTOFBOUNDS;
                        }
                      }
