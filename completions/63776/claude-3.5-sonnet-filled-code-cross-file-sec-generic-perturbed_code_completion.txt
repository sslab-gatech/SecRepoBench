PRE (R_11)
{
  obj->size = bit_read_RS (bit_chain);
  LOG_TRACE ("size: " FORMAT_RS " [RS]\n", obj->size);
  if (obj->size > size - (bit_chain->byte - oldpos))
    {
      LOG_ERROR ("Invalid object size " FORMAT_RS " > " FORMAT_RL, obj->size,
                 size - (bit_chain->byte - oldpos));
      obj->size = size - (bit_chain->byte - oldpos);
    }
}
LATER_VERSIONS
{
  obj->size = bit_read_RL (bit_chain);
  LOG_TRACE ("size: " FORMAT_RL " [RL]\n", obj->size);
  if (obj->size > size - (bit_chain->byte - oldpos))
    {
      LOG_ERROR ("Invalid object size " FORMAT_RL " > " FORMAT_RL, obj->size,
                 size - (bit_chain->byte - oldpos));
      obj->size = size - (bit_chain->byte - oldpos);
    }
}

if (obj->size + 2 < bit_chain->byte - oldpos)
  {
    BITCODE_RS crc;
    LOG_WARN ("Object size " FORMAT_RL " < " FORMAT_RL ", padding",
              obj->size, bit_chain->byte - oldpos);
    bit_chain->byte = oldpos + obj->size;
    crc = bit_read_RS (bit_chain);
    LOG_TRACE ("crc: " FORMAT_RS " [RS]\n", crc);
  }
else if (obj->size > bit_chain->byte - oldpos)
  {
    BITCODE_RL offset = obj->size - (bit_chain->byte - oldpos);
    LOG_WARN ("Object size " FORMAT_RL " > " FORMAT_RL " by " FORMAT_RL
              ", padding",
              obj->size, bit_chain->byte - oldpos, offset);
    obj->unknown_bits = (unsigned char *)calloc (offset, 1);
    if (obj->unknown_bits)
      {
        memcpy (obj->unknown_bits, &bit_chain->chain[bit_chain->byte], offset);
        LOG_TRACE_TF (obj->unknown_bits, offset);
        bit_chain->byte += offset;
      }
  }
