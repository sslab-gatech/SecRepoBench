if (obj->size > (bit_chain->byte - oldpos))
                {
                  LOG_WARN ("Object size %u > %" PRIuSIZE " bytes read",
                            obj->size, bit_chain->byte - oldpos);
                  obj->size = (bit_chain->byte - oldpos) & 0xFFFFFFFF;
                }
                else if (obj->size < (bit_chain->byte - oldpos))
                {
                  LOG_WARN ("Object size %u < %" PRIuSIZE " bytes read",
                            obj->size, bit_chain->byte - oldpos);
                  obj->size = (bit_chain->byte - oldpos) & 0xFFFFFFFF;
                }
                if (obj->size != (bit_chain->byte - oldpos))
                {
                  size_t offset = (bit_chain->byte - oldpos) - obj->size;
                  if (offset > 0)
                  {
                    obj->unknown_rest = (unsigned char *)malloc(offset);
                    if (!obj->unknown_rest)
                    {
                      LOG_ERROR ("Out of memory");
                      return DWG_ERR_OUTOFMEM;
                    }
                    memcpy(obj->unknown_rest, &bit_chain->chain[oldpos + obj->size], offset);
                    LOG_TRACE ("Unknown data: ");
                    LOG_TRACE_TF (obj->unknown_rest, offset);
                    LOG_TRACE ("\n");
                  }
                }
                if (bit_chain->byte != oldpos + obj->size)
                {
                  LOG_ERROR ("Object size mismatch: %u != %" PRIuSIZE,
                             obj->size, bit_chain->byte - oldpos);
                  return DWG_ERR_INVALIDDWG;
                }
                if (bit_chain->byte < oldpos + size)
                {
                  LOG_TRACE ("\n");
                }
                if (bit_chain->byte >= bit_chain->size)
                {
                  LOG_ERROR ("Buffer overflow %" PRIuSIZE " >= %" PRIuSIZE,
                             bit_chain->byte, bit_chain->size);
                  return DWG_ERR_INVALIDDWG;
