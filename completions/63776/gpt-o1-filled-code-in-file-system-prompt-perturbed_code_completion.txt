{
  size_t actual_size = dat->byte - previous_byte_position;
  if ((BITCODE_RL)actual_size < obj->size)
    {
      /* We didn't consume all bytes for this object; treat the remainder as unknown data. */
      size_t leftover = obj->size - actual_size;
      LOG_WARN ("Object type %d leftover: %zu bytes (pos %zu, expected %u)",
                obj->type, leftover, (size_t)dat->byte, obj->size);
      obj->unknown_bits = bit_read_TF (dat, leftover);
      obj->num_unknown_bits = leftover * 8;
    }
  else if ((BITCODE_RL)actual_size > obj->size)
    {
      /* We consumed more bytes than declared; a possible decode or DWG error. */
      LOG_ERROR ("Object type %d over-read: read %zu bytes, size was %u",
                 obj->type, actual_size, obj->size);
      /* We can't correct this over-read here, so just log the error. */
    }
  else
    {
      /* The consumed size matches exactly; no leftover data. */
      LOG_TRACE ("Object type %d read size matches exactly: %u bytes",
                 obj->type, obj->size);
    }
}
