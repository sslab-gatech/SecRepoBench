if (obj->size == 0)
  {
    obj->size = (dat->byte - previous_byte_position) & 0xFFFFFFFF;
    if (obj->size > size)
      {
        LOG_ERROR ("Invalid object size " FORMAT_RL " > " FORMAT_RL,
                   obj->size, size);
        obj->size = size;
      }
  }
else if (obj->size != (dat->byte - previous_byte_position))
  {
    BITCODE_RL offset = (dat->byte - previous_byte_position) & 0xFFFFFFFF;
    if (offset < obj->size)
      {
        BITCODE_RL remain = obj->size - offset;
        if (remain > 0 && remain < 0x100000)
          {
            LOG_TRACE ("unknown: remaining " FORMAT_RL " bytes\n", remain);
            obj->unknown_rest = (unsigned char *)calloc (remain, 1);
            if (obj->unknown_rest)
              {
                memcpy (obj->unknown_rest, &dat->chain[dat->byte], remain);
                obj->unknown_rest_size = remain;
                dat->byte += remain;
              }
          }
        else
          {
            LOG_ERROR ("Invalid object size " FORMAT_RL " - " FORMAT_RL,
                       obj->size, offset);
            obj->size = offset;
          }
      }
    else
      {
        LOG_ERROR ("Invalid object size " FORMAT_RL " < " FORMAT_RL,
                   obj->size, offset);
        obj->size = offset;
      }
  }
