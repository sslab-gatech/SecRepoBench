if (dat->byte >= dat->size)
        {
          LOG_ERROR ("Invalid object size %u > %" PRIuSIZE, obj->size, dat->size);
          return DWG_ERR_VALUEOUTOFBOUNDS;
        }

      // Read the object data
      obj->data = (unsigned char *)malloc(obj->size);
      if (!obj->data)
        {
          LOG_ERROR ("Out of memory");
          return DWG_ERR_OUTOFMEM;
        }
      memcpy(obj->data, &dat->chain[dat->byte], obj->size);
      dat->byte += obj->size;

      // Check for padding and CRC
      if (dat->byte % 2 != 0)
        {
          dat->byte++; // Skip padding byte
        }
      if (!bit_check_CRC(dat, obj->address, 0xC0C1))
        {
          LOG_ERROR ("CRC mismatch for object at address %" PRIuSIZE, obj->address);
          return DWG_ERR_WRONGCRC;
        }
