PRE (R_11)
{
  obj->size = bit_read_RS (dat);
  LOG_TRACE ("size: " FORMAT_RS " [RS]\n", obj->size);
  if (obj->size > dat->size - dat->byte)
    {
      LOG_ERROR ("Invalid object size " FORMAT_RS " > %" PRIuSIZE ", truncated",
                 obj->size, dat->size - dat->byte);
      obj->size = (BITCODE_RS)(dat->size - dat->byte);
    }
}
else
{
  obj->size = bit_read_RL (dat);
  LOG_TRACE ("size: " FORMAT_RL " [RL]\n", obj->size);
  if (obj->size > dat->size - dat->byte)
    {
      LOG_ERROR ("Invalid object size " FORMAT_RL " > %" PRIuSIZE ", truncated",
                 obj->size, dat->size - dat->byte);
      obj->size = (BITCODE_RL)(dat->size - dat->byte);
    }
}

if (obj->size + obj->address > dat->byte)
{
  BITCODE_RL offset = (obj->address + obj->size) - dat->byte;
  if (offset > obj->size)
    {
      LOG_ERROR ("Object size overflow: " FORMAT_RL " > " FORMAT_RL,
                 offset, obj->size);
      obj->size = (BITCODE_RL)(dat->byte - obj->address);
    }
  else if (offset)
    {
      LOG_TRACE ("UNKNOWN_BITS: " FORMAT_RL " [RL]\n", offset);
      obj->unknown_bits = bit_read_TF (dat, offset);
    }
}
else if (obj->address + obj->size < dat->byte)
{
  LOG_ERROR ("Object size underflow: %lu < %" PRIuSIZE,
             (unsigned long)(obj->address + obj->size), dat->byte);
  obj->size = (BITCODE_RL)(dat->byte - obj->address);
}
