BITCODE_RL obj_size = bit_read_RL (dat);
LOG_TRACE ("size: " FORMAT_RL " [RL]\n", obj_size);
if (obj_size <= 0 || obj_size > (BITCODE_RL)(dat->size - obj->address))
  {
    LOG_ERROR ("Invalid object size " FORMAT_RL " at 0x%zx\n", obj_size,
               obj->address);
    return DWG_ERR_INVALIDDWG;
  }
obj->size = obj_size;

// Check for unknown trailing data
if (obj->address + obj->size > dat->byte)
  {
    BITCODE_RL unknown_size = (obj->address + obj->size) - dat->byte;
    if (unknown_size > 0)
      {
        obj->unknown_data = (BITCODE_TF)calloc (1, unknown_size);
        if (!obj->unknown_data)
          {
            LOG_ERROR ("Out of memory for unknown data");
            return DWG_ERR_OUTOFMEM;
          }
        memcpy (obj->unknown_data, dat->chain + dat->byte, unknown_size);
        dat->byte += unknown_size;
        LOG_TRACE ("Unknown trailing data: %d bytes\n", unknown_size);
      }
  }
else if (obj->address + obj->size < dat->byte)
  {
    LOG_ERROR ("Object size mismatch at 0x%zx: expected %d, got %zu\n",
               obj->address, obj_size, dat->byte - obj->address);
    return DWG_ERR_INVALIDDWG;
  }
