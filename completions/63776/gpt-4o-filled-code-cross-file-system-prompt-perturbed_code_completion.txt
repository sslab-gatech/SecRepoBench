if (obj->address + obj->size != bit_chain->byte)
{
  BITCODE_RL offset = (bit_chain->byte - obj->address) & 0xFFFFFFFF;
  if (offset > obj->size)
  {
    obj->size = offset;
    if (obj->size > size)
    {
      LOG_ERROR("Object size exceeds section size");
      return DWG_ERR_INVALIDDWG;
    }
  }
  if (obj->size > 0)
  {
    obj->unknown_bits = (unsigned char *)calloc(1, obj->size);
    if (!obj->unknown_bits)
    {
      LOG_ERROR("Out of memory for unknown bits");
      return DWG_ERR_OUTOFMEM;
    }
    memcpy(obj->unknown_bits, bit_chain->chain + obj->address, obj->size);
    LOG_TRACE("Unknown data extracted for object at 0x%zx, size: %u\n", obj->address, obj->size);
  }
}
