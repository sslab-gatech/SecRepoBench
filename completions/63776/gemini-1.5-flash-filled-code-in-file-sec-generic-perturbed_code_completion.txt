if (dat->from_version < R_11)
                  {
                    if (obj->size > dat->size)
                      {
                        LOG_WARN ("Invalid object size %u > %" PRIuSIZE
                                  ". Adjust to %" PRIuSIZE,
                                  obj->size, dat->size, dat->size - 1);
                        obj->size = dat->size - 1;
                      }
                    if (obj->size * 8 != bit_position (dat))
                      {
                        LOG_WARN ("Invalid object size %u != %" PRIuSIZE,
                                  obj->size, bit_position (dat) / 8);
                        dwg_decode_unknown_rest (dat, obj);
                      }
                  }
                else
                  {
                    BITCODE_RS crc;
                    BITCODE_RL crc_size;
                    size_t crcpos;
                    if (obj->size * 8 != bit_position (dat))
                      {
                        crc_size = obj->size * 8 - bit_position (dat);
                        crcpos = obj->address + obj->size;
                        LOG_WARN ("Invalid object size %u != %" PRIuSIZE
                                  ", unknown data: %" PRIuSIZE " bytes",
                                  obj->size, bit_position (dat) / 8,
                                  crc_size / 8);
                        if (crc_size > 0)
                          {
                            obj->unknown_rest = bit_read_bits (dat, crc_size);
                            if (!obj->unknown_rest)
                              {
                                LOG_ERROR ("Failed to read unknown data");
                                return DWG_ERR_VALUEOUTOFBOUNDS;
                              }
                            obj->num_unknown_rest = crc_size;
                            LOG_TRACE ("unknown_rest [%ld %d TF]: ", crc_size,
                                       crc_size / 8);
                            LOG_TRACE_TF (obj->unknown_rest, crc_size / 8);
                            LOG_TRACE ("\n");
                          }
                        bit_set_position (dat, objpos + restartpos);
                      }
                    // check CRC-on
                    bit_set_position (dat, (obj->address + obj->size) * 8 - 2);
                    crc = bit_read_RS (dat);
                    LOG_TRACE ("crc: %04X [RSx] from %" PRIuSIZE "-%" PRIuSIZE
                               "=%ld\n",
                               crc, obj->address, dat->byte - 2,
                               (long)(dat->byte - 2 - obj->address));
                    if (!bit_check_CRC (dat, obj->address, 0xC0C1))
                      error |= DWG_ERR_WRONGCRC;
                  }
