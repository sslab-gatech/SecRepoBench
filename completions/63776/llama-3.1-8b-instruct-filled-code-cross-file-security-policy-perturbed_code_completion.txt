SINCE (R_2_0b)
{
  // This code block handles the size validation and unknown data extraction for DWG objects in pre-R11 and later versions.
  // For pre-R11 versions, the code checks if the object's size is valid and within the bounds of the data, adjusting it if necessary.
  // It extracts any unknown remaining data if the object size doesn't match the current data position and logs errors if issues are found.
  // For later versions, additional CRC checks are performed to ensure data integrity, and adjustments are made to account for any size discrepancies.
  // If the object's address and size don't align with the data position, the code calculates the offset for any trailing unknown data and allocates memory to store it, logging appropriate messages.
  if (obj->type!= DWG_TYPE_JUMP_r11)
    {
      // Check if the object's size is valid and within the bounds of the data
      if (obj->size > (bit_chain->byte - oldpos))
        {
          // If the object's size is larger than the remaining data, adjust it to the remaining data size
          obj->size = (bit_chain->byte - oldpos);
          LOG_WARN ("Object size adjusted to %d bytes\n", obj->size);
        }
      else if (obj->size < (bit_chain->byte - oldpos))
        {
          // If the object's size is smaller than the remaining data, extract the unknown remaining data
          size_t unknown_size = (bit_chain->byte - oldpos) - obj->size;
          if (unknown_size > 0)
            {
              // Allocate memory to store the unknown remaining data
              obj->unknown_bits = (BITCODE_RL)calloc(1, unknown_size);
              if (!obj->unknown_bits)
                {
                  LOG_ERROR ("Out of memory\n");
                  return DWG_ERR_OUTOFMEM;
                }
              // Extract the unknown remaining data
              bit_read_bits(bit_chain, unknown_size, obj->unknown_bits);
              LOG_TRACE ("Unknown remaining data: ");
              LOG_TRACE_TF(obj->unknown_bits, unknown_size);
              LOG_TRACE ("\n");
            }
        }
      // Check if the object's address and size align with the data position
      if ((BITCODE_RL)bit_chain->byte!= (obj->address + obj->size))
        {
          // Calculate the offset for any trailing unknown data
          size_t offset = (obj->address + obj->size) - (BITCODE_RL)bit_chain->byte;
          if (offset > 0)
            {
              // Allocate memory to store the trailing unknown data
              obj->unknown_bits = (BITCODE_RL)calloc(1, offset);
              if (!obj->unknown_bits)
                {
                  LOG_ERROR ("Out of memory\n");
                  return DWG_ERR_OUTOFMEM;
                }
              // Extract the trailing unknown data
              bit_read_bits(bit_chain, offset, obj->unknown_bits);
              LOG_TRACE ("Trailing unknown data: ");
              LOG_TRACE_TF(obj->unknown_bits, offset);
              LOG_TRACE ("\n");
            }
        }
    }
}
