PRE (R_11)
                  {
                    if (obj->size != (bit_chain->byte - oldpos))
                      {
                        if (obj->size > (bit_chain->byte - oldpos))
                          {
                            LOG_ERROR (
                                "Invalid object size " FORMAT_RL
                                " > %ld. Would overflow",
                                obj->size, (long)(bit_chain->byte - oldpos));
                            obj->size = (bit_chain->byte - oldpos);
                            error |= DWG_ERR_VALUEOUTOFBOUNDS;
                          }
                        else if (obj->size < (bit_chain->byte - oldpos))
                          {
                            size_t unknown_size = (bit_chain->byte - oldpos)
                                                  - obj->size;
                            LOG_ERROR ("Invalid object size " FORMAT_RL
                                       " < %ld. Trailing unknown data "
                                       "with size %ld",
                                       obj->size, (long)(bit_chain->byte - oldpos),
                                       (long)unknown_size);
                            obj->unknown = bit_read_TF (bit_chain, unknown_size);
                            error |= DWG_ERR_VALUEOUTOFBOUNDS;
                          }
                      }
                  }
                  else
                  {
                    BITCODE_RS crc, crcc;
                    if (obj->size != (bit_chain->byte - oldpos))
                      {
                        if (obj->size > (bit_chain->byte - oldpos))
                          {
                            LOG_ERROR (
                                "Invalid object size " FORMAT_RL
                                " > %ld. Would overflow",
                                obj->size, (long)(bit_chain->byte - oldpos));
                            obj->size = (bit_chain->byte - oldpos);
                            error |= DWG_ERR_VALUEOUTOFBOUNDS;
                          }
                        else
                          {
                            size_t unknown_size = (bit_chain->byte - oldpos)
                                                  - obj->size;
                            LOG_ERROR ("Invalid object size " FORMAT_RL
                                       " < %ld. Trailing unknown data "
                                       "with size %ld",
                                       obj->size, (long)(bit_chain->byte - oldpos),
                                       (long)unknown_size);
                            obj->unknown = bit_read_TF (bit_chain, unknown_size);
                            error |= DWG_ERR_VALUEOUTOFBOUNDS;
                          }
                      }

                    crcc = bit_calc_CRC (0xC0C1,
                                         &bit_chain->chain[oldpos],
                                         bit_chain->byte - oldpos);
                    crc = bit_read_RS (bit_chain);
                    LOG_TRACE ("crc: %04X [RSx] from 0x%zx-0x%zx\n", crc, oldpos,
                               bit_chain->byte - 2);
                    if (crc != crcc)
                      {
                        LOG_ERROR ("Entity CRC mismatch %04X <=> %04X", crc,
                                   crcc);
                        error |= DWG_ERR_WRONGCRC;
                      }
                  }
