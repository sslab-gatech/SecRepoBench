// Validate the size of the object
          size_t expected_size = (bit_chain->byte - oldpos) & 0xFFFFFFFF;
          if (obj->size != expected_size)
            {
              LOG_WARN ("Size mismatch for object at offset 0x%zx: expected %zu, got %zu\n",
                        oldpos, obj->size, expected_size);
              // Handle unknown data extraction
              if (expected_size > obj->size)
                {
                  size_t unknown_size = expected_size - obj->size;
                  obj->unknown_bits = (unsigned char *)malloc(unknown_size);
                  if (!obj->unknown_bits)
                    {
                      LOG_ERROR ("Out of memory for unknown bits\n");
                      return DWG_ERR_OUTOFMEM;
                    }
                  memcpy(obj->unknown_bits, bit_chain->chain + oldpos + obj->size, unknown_size);
                  obj->num_unknown_bits = unknown_size;
                  LOG_TRACE ("Extracted %zu unknown bytes\n", unknown_size);
                }
            }
