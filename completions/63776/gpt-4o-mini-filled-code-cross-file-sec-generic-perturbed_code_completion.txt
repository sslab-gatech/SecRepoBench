size_t remaining_size = (previous_byte_position + size) - dat->byte;
                if (remaining_size > 0)
                {
                    // Allocate memory for unknown data
                    obj->unknown_data = (unsigned char *)malloc(remaining_size);
                    if (!obj->unknown_data)
                    {
                        LOG_ERROR("Out of memory for unknown data");
                        return DWG_ERR_OUTOFMEM;
                    }
                    // Copy the unknown data into the allocated memory
                    memcpy(obj->unknown_data, &dat->chain[dat->byte], remaining_size);
                    obj->unknown_data_size = remaining_size;
                    LOG_TRACE("Extracted unknown data of size %zu", remaining_size);
                }
                else
                {
                    LOG_WARN("No unknown data to extract");
