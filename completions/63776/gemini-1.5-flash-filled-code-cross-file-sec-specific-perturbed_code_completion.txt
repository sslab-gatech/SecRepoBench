size_t unknown_data_size;
                if (obj->size > (bit_chain->byte - oldpos))
                  {
                    unknown_data_size = obj->size - (bit_chain->byte - oldpos);
                    obj->unknown_data = (unsigned char *)malloc (unknown_data_size);
                    if (!obj->unknown_data)
                      {
                        LOG_ERROR ("Out of memory");
                        return DWG_ERR_OUTOFMEM;
                      }
                    memcpy (obj->unknown_data, bit_chain->byte, unknown_data_size);
                    bit_chain->byte += unknown_data_size;
                    LOG_TRACE ("unknown data: %" PRIuSIZE " bytes\n",
                               unknown_data_size);
                  }
                else if (obj->size < (bit_chain->byte - oldpos))
                  {
                    LOG_WARN ("Object size mismatch: expected %" PRIuSIZE
                              ", got %" PRIuSIZE "\n",
                              obj->size, bit_chain->byte - oldpos);
                  }
