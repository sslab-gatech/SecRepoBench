size_t expected_end = obj->address + obj->size;
                if (bit_chain->byte > expected_end)
                  {
                    LOG_ERROR ("Object overflow: read %" PRIuSIZE " bytes, expected %u",
                               bit_chain->byte - obj->address, obj->size);
                    error |= DWG_ERR_INVALIDDWG;
                  }
                else
                  {
                    BITCODE_BL remaining = expected_end - bit_chain->byte;
                    if (remaining > 0)
                      {
                        if (bit_chain->byte + remaining > bit_chain->size)
                          {
                            LOG_WARN ("Truncating unknown data from %u to %" PRIuSIZE " bytes",
                                      remaining, bit_chain->size - bit_chain->byte);
                            remaining = bit_chain->size - bit_chain->byte;
                          }
                        if (remaining > 0)
                          {
                            obj->unknown_data_size = remaining;
                            obj->unknown_data = (BITCODE_TF)malloc (remaining);
                            if (!obj->unknown_data)
                              {
                                LOG_ERROR ("Out of memory for unknown data");
                                error |= DWG_ERR_OUTOFMEM;
                              }
                            else
                              {
                                memcpy (obj->unknown_data,
                                        bit_chain->chain + bit_chain->byte, remaining);
                                bit_chain->byte += remaining;
                                LOG_TRACE ("Extracted %u bytes of unknown data", remaining);
                              }
                          }
                      }
                  }
