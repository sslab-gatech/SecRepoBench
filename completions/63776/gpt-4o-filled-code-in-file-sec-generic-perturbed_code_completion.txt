if (obj->size > (bit_chain->byte - oldpos))
                {
                  LOG_WARN ("Unknown data in object %d: %u bytes",
                            obj->type, obj->size - (bit_chain->byte - oldpos));
                  obj->unknown_rest = (unsigned char *)malloc(obj->size - (bit_chain->byte - oldpos));
                  if (obj->unknown_rest)
                  {
                    memcpy(obj->unknown_rest, &bit_chain->chain[bit_chain->byte],
                           obj->size - (bit_chain->byte - oldpos));
                  }
                  bit_chain->byte += obj->size - (bit_chain->byte - oldpos);
                }
                else if (obj->size < (bit_chain->byte - oldpos))
                {
                  LOG_ERROR ("Object size mismatch: expected %u, got %zu",
                             obj->size, bit_chain->byte - oldpos);
                  error |= DWG_ERR_VALUEOUTOFBOUNDS;
                }
                else
                {
                  LOG_TRACE ("Object size matches: %u", obj->size);
                }
                if (bit_chain->byte < oldpos + size)
                {
                  LOG_TRACE ("Unknown trailing data: %zu bytes",
                             oldpos + size - bit_chain->byte);
                  obj->unknown_rest = (unsigned char *)malloc(oldpos + size - bit_chain->byte);
                  if (obj->unknown_rest)
                  {
                    memcpy(obj->unknown_rest, &bit_chain->chain[bit_chain->byte],
                           oldpos + size - bit_chain->byte);
                  }
                  bit_chain->byte = oldpos + size;
                }
                if (bit_chain->byte != oldpos + size)
                {
                  LOG_ERROR ("Object size mismatch after adjustment: expected %u, got %zu",
                             obj->size, bit_chain->byte - oldpos);
                  error |= DWG_ERR_VALUEOUTOFBOUNDS;
                }
