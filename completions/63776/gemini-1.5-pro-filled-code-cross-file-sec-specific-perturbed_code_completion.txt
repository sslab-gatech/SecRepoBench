                      {
                        LOG_ERROR ("Invalid object size, too big: %x > %" PRIuSIZE
                                   "\n",
                                   obj->size, dat->size - dat->byte);
                        obj->size = (BITCODE_RL)(dat->size - dat->byte);
                      }
                    if (obj->size > 0
                        && obj->size != (BITCODE_RL)(dat->byte - previous_byte_position))
                      {
                        size_t offset = dat->byte - previous_byte_position;
                        if (offset > obj->size)
                          offset = obj->size;
                        if (obj->size > offset)
                          {
                            size_t len = (obj->size - offset);
                            obj->unknown = (char *)calloc (1, len);
                            if (obj->unknown)
                              {
                                memcpy (obj->unknown, dat->chain + dat->byte, len);
                                LOG_TRACE ("unknown [%u]: ", (unsigned)len);
                                LOG_TRACE_N (obj->unknown, len);
                                LOG_TRACE ("\n");
                              }
                          }
                      }
                    dat->byte += obj->size;
                  }
                else
                  {
                    uint32_t crc;
                    if (obj->size != (BITCODE_RL)(dat->byte - previous_byte_position))
                      {
                        size_t offset = dat->byte - previous_byte_position;
                        if (offset > obj->size)
                          offset = obj->size;
                        if (obj->size > offset)
                          {
                            size_t len = (obj->size - offset);

                            if (len > dwg->max_unknown_size)
                              {
                                dwg->unknown = (unsigned char *)realloc (
                                    dwg->unknown, len);
                                if (!dwg->unknown)
                                  return DWG_ERR_OUTOFMEM;
                                dwg->max_unknown_size = len;
                              }
                            memcpy (dwg->unknown, dat->chain + dat->byte, len);
                            obj->unknown = dwg->unknown;
                            obj->unknown_size = len;
                            LOG_TRACE ("trailing unknown [%u]: ", (unsigned)len);
                            LOG_TRACE_N (obj->unknown, len);
                            LOG_TRACE ("\n");
                          }
                        else if (offset > obj->size)
                          {
                            LOG_ERROR ("Invalid object size, too small: %x < %" PRIuSIZE "\n", obj->size, offset);
                            return DWG_ERR_INVALIDDWG;
                          }
                      }
                    dat->byte += obj->size;
                    crc = dwg_crc32 (dat->chain + previous_byte_position, obj->size);
                    LOG_TRACE ("CRC: %x\n", crc);
