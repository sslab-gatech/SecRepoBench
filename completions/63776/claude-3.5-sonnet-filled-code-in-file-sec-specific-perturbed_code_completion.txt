BITCODE_RL size = bit_read_RL (bit_chain);
obj->size = size;
LOG_TRACE ("size: " FORMAT_RL " [RL]\n", size)
if (size > 0x1000000) // 16MB
  {
    LOG_ERROR ("Invalid object size " FORMAT_RL, size);
    obj->size = 0;
    error |= DWG_ERR_VALUEOUTOFBOUNDS;
  }
else if (obj->address + size > (BITCODE_RL)bit_chain->size)
  {
    LOG_ERROR ("Object size " FORMAT_RL " overflow", size);
    obj->size = (BITCODE_RL)bit_chain->size - obj->address;
    error |= DWG_ERR_VALUEOUTOFBOUNDS;
  }
if (obj->size < bit_chain->byte - oldpos)
  {
    LOG_ERROR ("Invalid object size " FORMAT_RL " < " FORMAT_RL, obj->size,
               (BITCODE_RL)(bit_chain->byte - oldpos));
    obj->size = (BITCODE_RL)(bit_chain->byte - oldpos);
    error |= DWG_ERR_VALUEOUTOFBOUNDS;
  }
else if (obj->size > bit_chain->byte - oldpos)
  {
    BITCODE_RL offset = obj->size - (bit_chain->byte - oldpos);
    LOG_TRACE (" unknown: " FORMAT_RL " [RL]", offset);
    bit_read_fixed (bit_chain, &obj->unknown_bits, offset);
    LOG_TRACE_TF (obj->unknown_bits, offset);
  }
SINCE (R_11)
{
  BITCODE_RS crc;
  crc = bit_read_RS (bit_chain);
  LOG_TRACE ("crc: %04X [RS]\n", crc);
  if (crc != bit_calc_CRC (0xC0C1, &bit_chain->chain[oldpos], size - 2))
    {
      LOG_ERROR ("Wrong CRC %04X", crc);
      error |= DWG_ERR_WRONGCRC;
    }
}
