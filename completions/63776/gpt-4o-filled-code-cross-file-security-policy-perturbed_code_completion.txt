if (obj->address + obj->size > dat->byte) {
  size_t unknown_size = (obj->address + obj->size) - dat->byte;
  if (unknown_size > dat->size - dat->byte) {
    LOG_ERROR("Object size exceeds available buffer size");
    return DWG_ERR_INVALIDDWG;
  }
  obj->unknown_bits = (unsigned char *)malloc(unknown_size);
  if (!obj->unknown_bits) {
    LOG_ERROR("Out of memory for unknown bits");
    return DWG_ERR_OUTOFMEM;
  }
  memcpy(obj->unknown_bits, dat->chain + dat->byte, unknown_size);
  dat->byte += unknown_size;
  LOG_TRACE("Extracted unknown data of size %zu\n", unknown_size);
} else if (obj->address + obj->size < dat->byte) {
  LOG_ERROR("Object size is less than current data position");
  return DWG_ERR_INVALIDDWG;
}
