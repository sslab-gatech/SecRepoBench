                  {
                    LOG_ERROR ("Invalid object size, too big: " FORMAT_RL
                               " > " FORMAT_RL "\n",
                               obj->size, bit_chain->size - oldpos);
                    obj->size = bit_chain->size - oldpos;
                  }
                if (obj->size > 0
                    && (BITCODE_RL)bit_chain->byte != oldpos + obj->size)
                  {
                    size_t unknown_size = oldpos + obj->size - bit_chain->byte;
                    LOG_ERROR ("Unknown remaining data " FORMAT_RL " bytes\n",
                               unknown_size);
                    obj->unknown = (char *)calloc (1, unknown_size + 1);
                    if (obj->unknown)
                      {
                        memcpy (obj->unknown, &bit_chain->chain[bit_chain->byte],
                                unknown_size);
                        bit_chain->byte += unknown_size;
                      }
                  }
                else if (obj->size > 0
                         && (BITCODE_RL)bit_chain->byte > oldpos + obj->size)
                  {
                    LOG_ERROR ("Invalid object size, too small: " FORMAT_RL
                               " < " FORMAT_RL "\n",
                               obj->size, bit_chain->byte - oldpos);
                    obj->size = bit_chain->byte - oldpos;
                  }
                if (bit_chain->version >= R_13)
                  {
                    BITCODE_U crc;
                    crc = bit_read_CRC32 (bit_chain);
                    LOG_TRACE ("CRC: " FORMAT_CRC "\n", crc);
                    if (obj->crc != crc)
                      {
                        LOG_ERROR ("CRC mismatch: " FORMAT_CRC " != " FORMAT_CRC
                                   "\n",
                                   obj->crc, crc);
                      }
                  }
                if (obj->size > 0
                    && (BITCODE_RL)bit_chain->byte != oldpos + obj->size)
                  {
                    size_t unknown_size = oldpos + obj->size - bit_chain->byte;
                    if (unknown_size > 0)
                      {
                        LOG_WARN ("Trailing unknown data " FORMAT_RL " bytes\n",
                                  unknown_size);
                        obj->trail_unknown = (char *)calloc (1, unknown_size + 1);
                        if (obj->trail_unknown)
                          {
                            memcpy (obj->trail_unknown,
                                    &bit_chain->chain[bit_chain->byte],
                                    unknown_size);
                            bit_chain->byte += unknown_size;
                          }
                      }
                    else if (unknown_size < 0)
                      {

                        obj->size = bit_chain->byte - oldpos;
