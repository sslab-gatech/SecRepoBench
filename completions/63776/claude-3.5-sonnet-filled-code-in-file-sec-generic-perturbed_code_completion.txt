PRE (R_11)
{
  obj->size = bit_read_RS (bit_chain);
  if (obj->size > size || obj->size < 2)
    {
      LOG_ERROR ("Invalid object size %d, resetting to %d", obj->size, size);
      obj->size = size;
    }
}
VERSIONS (R_11, R_13)
{
  obj->size = bit_read_RL (bit_chain);
  if (obj->size > size || obj->size < 4)
    {
      LOG_ERROR ("Invalid object size %d, resetting to %d", obj->size, size);
      obj->size = size;
    }
}

if (obj->size + obj->address != (unsigned long)bit_chain->byte)
  {
    unsigned long offset = obj->size + obj->address - bit_chain->byte;
    LOG_WARN ("Object size %lu + address %lu != file pos %lu",
              (unsigned long)obj->size, obj->address,
              (unsigned long)bit_chain->byte);
    if (offset > 0 && offset < 0xFFFFFFFF)
      {
        obj->unknown_bits = bit_read_TF (bit_chain, offset);
        LOG_TRACE ("unknown_bits: %lu [TF]\n", offset);
      }
  }

SINCE (R_11)
{
  BITCODE_RS crc, crcc;
  crc = bit_read_RS (bit_chain);
  LOG_TRACE ("crc: %04X [RSx]\n", crc);
  if (obj->address + obj->size == (unsigned long)bit_chain->byte - 2)
    {
      crcc = bit_calc_CRC (0xC0C1, &bit_chain->chain[obj->address],
                           obj->size - 2);
      if (crc != crcc)
        {
          LOG_ERROR ("Wrong CRC %04X, should be %04X", crc, crcc);
          error |= DWG_ERR_WRONGCRC;
        }
    }
  else
    {
      LOG_ERROR ("Wrong object size: %lu + %d != %lu",
                 obj->address, obj->size, bit_chain->byte - 2);
      error |= DWG_ERR_VALUEOUTOFBOUNDS;
    }
}
