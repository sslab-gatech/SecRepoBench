if (obj->size != (bit_chain->byte - obj->address))
                  {
                    BITCODE_RL unknown_size = obj->size - (bit_chain->byte - obj->address);
                    if (unknown_size > 0)
                      {
                        obj->unknown_rest = bit_read_bits (bit_chain,
                                                          unknown_size * 8);
                        LOG_WARN ("Trailing unknown data of size %u bytes",
                                  unknown_size);
                        LOG_TRACE_TF (obj->unknown_rest, unknown_size);
                      }
                    else if (obj->size < (bit_chain->byte - obj->address))
                      {
                        LOG_ERROR ("Object size underflow %u < %" PRIuSIZE,
                                   obj->size, bit_chain->byte - obj->address);
                        error |= DWG_ERR_VALUEOUTOFBOUNDS;
                      }
                  }
                else if (obj->size < (bit_chain->byte - obj->address))
                  {
                    LOG_ERROR ("Object size underflow %u < %" PRIuSIZE,
                               obj->size, bit_chain->byte - obj->address);
                    error |= DWG_ERR_VALUEOUTOFBOUNDS;
                  }
                if (obj->address + obj->size != bit_chain->byte)
                  {
                    BITCODE_RL unknown_size = bit_chain->byte - (obj->address + obj->size);
                    if (unknown_size > 0)
                      {
                        obj->unknown_rest = bit_read_bits (bit_chain,
                                                          unknown_size * 8);
                        LOG_WARN ("Trailing unknown data of size %u bytes",
                                  unknown_size);
                        LOG_TRACE_TF (obj->unknown_rest, unknown_size);
                      }
                    else if (obj->size > (bit_chain->byte - obj->address))
                      {
                        LOG_ERROR ("Object size overflow %u > %" PRIuSIZE,
                                   obj->size, bit_chain->byte - obj->address);
                        error |= DWG_ERR_VALUEOUTOFBOUNDS;
                      }
                  }
                else if (obj->size > (bit_chain->byte - obj->address))
                  {
                    LOG_ERROR ("Object size overflow %u > %" PRIuSIZE,
                               obj->size, bit_chain->byte - obj->address);
                    error |= DWG_ERR_VALUEOUTOFBOUNDS;
                  }
                if (error & DWG_ERR_VALUEOUTOFBOUNDS)
                  {
                    // obj->size = bit_chain->byte - obj->address;
                    // LOG_WARN ("Fixup object size to %u", obj->size);
                    // error &= ~DWG_ERR_VALUEOUTOFBOUNDS;
                    break;
                  }
                if (bit_chain->bit)
                  {
                    unsigned char r = 8 - bit_chain->bit;
                    LOG_HANDLE (" padding: %X/%X (%d bits)\n",
                                bit_chain->chain[bit_chain->byte],
                                bit_chain->chain[bit_chain->byte] & ((1 << r) - 1),
                                r);
                    bit_advance_position (bit_chain, r);
                  }
                bit_set_position (bit_chain, (obj->address + obj->size) * 8 - 2);
                if (!bit_check_CRC (bit_chain, obj->address, 0xC0C1))
                  error |= DWG_ERR_WRONGCRC;
              }
          }
