// Validate the size of the object
                if (obj->size != (bit_chain->byte - oldpos) && obj->size > 0) {
                    LOG_WARN("Size mismatch for object type %d: expected %u, got %u",
                             obj->type, obj->size, (unsigned)(bit_chain->byte - oldpos));
                    // Extract unknown data if size does not match
                    size_t unknown_size = (bit_chain->byte - oldpos) - obj->size;
                    if (unknown_size > 0) {
                        obj->unknown_bits = (unsigned char *)malloc(unknown_size);
                        if (!obj->unknown_bits) {
                            LOG_ERROR("Out of memory for unknown bits");
                            return DWG_ERR_OUTOFMEM;
                        }
                        memcpy(obj->unknown_bits, bit_chain->chain + oldpos + obj->size, unknown_size);
                        obj->num_unknown_bits = unknown_size;
                        LOG_TRACE("Extracted %zu unknown bytes for object type %d", unknown_size, obj->type);
                    }
                }
