return DWG_ERR_VALUEOUTOFBOUNDS;
    }
  if (dwg->header.section_infohdr.num_desc > 0)
    {
      dwg->header.section_info = (Dwg_Section_Info *)calloc (
          dwg->header.section_infohdr.num_desc, sizeof (Dwg_Section_Info));
      if (!dwg->header.section_info)
        {
          LOG_ERROR ("Out of memory");
          free (decomp);
          return DWG_ERR_OUTOFMEM;
        }
      decomp_end = decomp + decomp_data_size;
      for (i = 0; i < dwg->header.section_infohdr.num_desc; ++i)
        {
          bfr_read_32 (&dwg->header.section_info[i], &ptr, 16);
          if (ptr > decomp_end)
            {
              LOG_ERROR ("Invalid section_info");
              free (decomp);
              return DWG_ERR_VALUEOUTOFBOUNDS;
            }
          LOG_TRACE ("Section[%d]=%d,", i, dwg->header.section_info[i].number)
          LOG_TRACE (" size: %5u,", dwg->header.section_info[i].size)
          LOG_TRACE (" address: 0x%04lx\n",
                     (unsigned long)dwg->header.section_info[i].address)
          section_number = dwg->header.section_info[i].number;
          if (section_number > 0)
            {
              Dwg_Section *info = find_section (dwg, section_number);
              if (info)
                {
                  info->size = dwg->header.section_info[i].size;
                  info->address = dwg->header.section_info[i].address;
                }
              else
                {
                  LOG_WARN ("Section[%d]=%d not found", i, section_number);
                }
            }
          else
            {
              LOG_WARN ("Section[%d]=%d not found", i, section_number);
            }
        }
    }
  free (decomp);
