obj->size = bit_read_RL(dat);
                LOG_TRACE("size: " FORMAT_RL "\n", obj->size);
                if (obj->size > 0 && obj->size <= (dat->size - dat->byte)) {
                  if (obj->address + obj->size != dat->byte) {
                    size_t unknown_size = obj->address + obj->size - dat->byte;
                    if (unknown_size > 0) {
                      obj->unknown_bits = (unsigned char *)malloc(unknown_size);
                      if (!obj->unknown_bits) {
                        LOG_ERROR("Out of memory for unknown_bits");
                        return DWG_ERR_OUTOFMEM;
                      }
                      memcpy(obj->unknown_bits, &dat->chain[dat->byte], unknown_size);
                      dat->byte += unknown_size;
                      LOG_TRACE("Extracted unknown data of size %" PRIuSIZE "\n", unknown_size);
                    }
                  }
                } else {
                  LOG_ERROR("Invalid object size %" PRIuSIZE " at byte position %" PRIuSIZE, obj->size, dat->byte);
                  return DWG_ERR_INVALIDDWG;
                }
