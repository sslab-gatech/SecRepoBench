obj->size = (bit_chain->version >= R_11) ? bit_read_RL (bit_chain) : bit_read_RS (bit_chain);
                BITCODE_RL actual_size = (BITCODE_RL)(bit_chain->byte - oldpos);
                if (obj->size != actual_size)
                  {
                    LOG_ERROR ("Object size mismatch: read " FORMAT_RL " vs actual " FORMAT_RL,
                               obj->size, actual_size);
                    obj->size = actual_size; // Adjust to prevent overflow errors
                  }
                // Check for trailing unknown data
                if (bit_chain->byte < oldpos + obj->size)
                  {
                    BITCODE_RL remaining = (oldpos + obj->size) - bit_chain->byte;
                    obj->unknown_bits = bit_read_bytes (bit_chain, remaining);
                    LOG_WARN ("Extracted " FORMAT_RL " bytes of unknown data", remaining);
                  }
                else if (bit_chain->byte > oldpos + obj->size)
                  {
                    LOG_ERROR ("Overflow: Read past entity size by " FORMAT_RL " bytes",
                               bit_chain->byte - (oldpos + obj->size));
                    error |= DWG_ERR_INVALIDDWG;
                  }
                // Handle CRC checks for R11+
                SINCE (R_11)
                  {
                    BITCODE_BS crc = bit_read_BS (bit_chain);
                    BITCODE_BS computed_crc = compute_crc (bit_chain->chain + oldpos, obj->size);
                    if (crc != computed_crc)
                      {
                        LOG_ERROR ("CRC mismatch: expected 0x%04X, got 0x%04X", computed_crc, crc);
                        error |= DWG_ERR_CRC;
                      }
                  }
