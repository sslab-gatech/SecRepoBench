BITCODE_RL obj_size = (bit_chain->byte - oldpos) & 0xFFFFFFFF;
if (obj_size != obj->size)
  {
    LOG_WARN ("Object size mismatch: " FORMAT_RL " != " FORMAT_RL "\n",
              obj_size, obj->size);
    if (obj_size > obj->size)
      {
        LOG_ERROR ("Object size too large: " FORMAT_RL " > " FORMAT_RL "\n",
                   obj_size, obj->size);
        return DWG_ERR_INVALIDDWG;
      }
    else
      {
        BITCODE_RL remaining = obj->size - obj_size;
        if (remaining > 0)
          {
            LOG_TRACE ("Unknown remaining data: " FORMAT_RL " bytes\n", remaining);
            obj->unknown_data = bit_read_bytes(bit_chain, remaining);
            if (!obj->unknown_data)
              {
                LOG_ERROR ("Failed to read unknown data\n");
                return DWG_ERR_OUTOFMEM;
              }
          }
      }
  }
else
  {
    obj->size = obj_size;
  }

if (bit_chain->version >= R_13)
  {
    BITCODE_RL crc = bit_read_RL(bit_chain);
    BITCODE_RL calculated_crc = calculate_crc(bit_chain, oldpos, obj->size);
    if (crc != calculated_crc)
      {
        LOG_ERROR ("CRC mismatch: " FORMAT_RLx " != " FORMAT_RLx "\n",
                   crc, calculated_crc);
        return DWG_ERR_INVALIDDWG;
      }
  }
