BITCODE_RL obj_size = bit_read_RL (bit_chain);
              LOG_TRACE ("size: " FORMAT_RL " [RL]\n", obj_size);
              if (obj_size <= 0 || obj_size > (BITCODE_RL)(bit_chain->size - oldpos))
                {
                  LOG_ERROR ("Invalid object size " FORMAT_RL " at 0x%zx\n",
                             obj_size, bit_chain->byte - 4);
                  return DWG_ERR_INVALIDDWG;
                }
              obj->size = obj_size;
              if (bit_chain->byte + obj_size > oldpos + size)
                {
                  LOG_ERROR ("Object size " FORMAT_RL " exceeds section size " FORMAT_RL
                             " at 0x%zx\n",
                             obj_size, size, bit_chain->byte - 4);
                  return DWG_ERR_INVALIDDWG;
                }
              if (bit_chain->byte + obj_size < oldpos + size)
                {
                  BITCODE_RL unknown_size = (oldpos + size) - (bit_chain->byte + obj_size);
                  if (unknown_size > 0)
                    {
                      obj->unknown_data = (BITCODE_TF)calloc (1, unknown_size);
                      if (!obj->unknown_data)
                        {
                          LOG_ERROR ("Out of memory");
                          return DWG_ERR_OUTOFMEM;
                        }
                      memcpy (obj->unknown_data, bit_chain->chain + bit_chain->byte + obj_size, unknown_size);
                      obj->unknown_size = unknown_size;
                      LOG_TRACE ("Unknown trailing data: %d bytes\n", unknown_size);
                    }
                }
