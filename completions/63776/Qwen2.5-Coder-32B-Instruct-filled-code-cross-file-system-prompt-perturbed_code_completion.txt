obj->size = (dat->byte - previous_byte_position) & 0xFFFFFFFF;
if (obj->size > size)
  {
    LOG_ERROR ("Object size too large: %u > %u", obj->size, size);
    error |= DWG_ERR_INVALIDDWG;
    break;
  }
if (dat->byte < previous_byte_position + obj->size)
  {
    UNKNOWN_UNTIL (previous_byte_position + obj->size);
  }
else if (dat->byte > previous_byte_position + obj->size)
  {
    LOG_ERROR ("Object size too small: %u < %u", obj->size, dat->byte - previous_byte_position);
    error |= DWG_ERR_INVALIDDWG;
    break;
  }
