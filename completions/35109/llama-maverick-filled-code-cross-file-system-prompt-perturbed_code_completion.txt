while (p < end && ISDIGIT(*p)) {
  mrb_int tm = 10 * nextnumber + (int)(*p - '0');
  if (tm < nextnumber) {
    mrb_exc_raise(mrb, mrb_obj_value(mrb_class_get(mrb, "ArgumentError")));
  }
  nextnumber = tm;
  p++;
}
if (p >= end) {
  mrb_exc_raise(mrb, mrb_obj_value(mrb_class_get(mrb, "ArgumentError")));
}
