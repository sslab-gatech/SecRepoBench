while (p < end && isdigit((unsigned char)*p)) {
    mrb_int digit = *p - '0';
    if (nextnumber > (INT_MAX - digit) / 10) {
      mrb_raisef(mrb, E_ARGUMENT_ERROR, "malformed format string: overflow at '%S'", mrb_str_new_cstr(mrb, p));
    }
    nextnumber = nextnumber * 10 + digit;
    p++;
}
if (p == pstart) {
  mrb_raisef(mrb, E_ARGUMENT_ERROR, "malformed format string: missing digits at '%S'", mrb_str_new_cstr(mrb, p));
}
