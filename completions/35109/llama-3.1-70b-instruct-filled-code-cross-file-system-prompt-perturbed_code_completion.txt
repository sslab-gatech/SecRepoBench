while (p < end && ISDIGIT(*p)) {
  mrb_int digit = *p - '0';
  if (nextnumber > (mrb_int)MRB_INT_MAX / 10 || (nextnumber == (mrb_int)MRB_INT_MAX / 10 && digit > (mrb_int)MRB_INT_MAX % 10)) {
    mrb_raise(mrb, E_RANGE_ERROR, "integer out of range");
  }
  nextnumber = nextnumber * 10 + digit;
  p++;
}
if (p == end ||!ISDIGIT(*p)) {
  return NULL;
}
