const char *start = p;
while (p < end && ISDIGIT(*p)) {
  int digit = (*p - '0');
  
  // Check for overflow
  if (nextnumber > (MRB_INT_MAX - digit) / 10) {
    mrb_raise(mrb, E_ARGUMENT_ERROR, "too big argument number");
  }
  
  nextnumber = nextnumber * 10 + digit;
  p++;
}

// Check for a malformed format string
if (p == start) {
  mrb_raise(mrb, E_ARGUMENT_ERROR, "malformed format string - expected digits");
}
