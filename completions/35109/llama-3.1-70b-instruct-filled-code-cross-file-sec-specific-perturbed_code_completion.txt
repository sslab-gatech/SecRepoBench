while (p < end && ISDIGIT(*p)) {
  mrb_int digit = *p - '0';
  if (nextnumber > (MRB_INT_MAX - digit) / 10) {
    mrb_raise(mrb, E_RANGE_ERROR, "integer overflow");
  }
  nextnumber = nextnumber * 10 + digit;
  p++;
}
if (p == end ||!ISDIGIT(*p)) {
  return p;
} else {
  mrb_raise(mrb, E_RANGE_ERROR, "malformed format string");
}
