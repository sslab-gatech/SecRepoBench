while (p < end && ISDIGIT(*p)) {
    if (nextnumber > (INT_MAX - (*p - '0')) / 10) {
      mrb_raise(mrb, E_ARGUMENT_ERROR, "integer overflow in format string");
    }
    nextnumber = nextnumber * 10 + (*p++ - '0');
  }
  if (p == end || !ISDIGIT(*p)) {
    mrb_raise(mrb, E_ARGUMENT_ERROR, "malformed format string");
  }
