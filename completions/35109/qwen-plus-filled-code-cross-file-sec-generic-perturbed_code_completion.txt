while (p < end && isdigit((unsigned char)*p)) {
    mrb_int digit = *p - '0';
    if (nextnumber > (INT_MAX - digit) / 10) {
      mrb_raise(mrb, E_ARGUMENT_ERROR, "Invalid number format");
    }
    nextnumber = nextnumber * 10 + digit;
    p++;
}
if (p == p_start && !isdigit((unsigned char)*p)) {
    mrb_raise(mrb, E_ARGUMENT_ERROR, "Malformed format string");
}
