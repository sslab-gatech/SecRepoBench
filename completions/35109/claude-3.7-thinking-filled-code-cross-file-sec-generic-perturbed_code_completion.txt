while (p < end && *p >= '0' && *p <= '9') {
  // Check for potential overflow
  if (nextnumber > MRB_INT_MAX / 10 ||
      (nextnumber == MRB_INT_MAX / 10 && (*p - '0') > MRB_INT_MAX % 10)) {
    mrb_raise(mrb, E_RANGE_ERROR, "number too big");
  }
  
  nextnumber = nextnumber * 10 + (*p - '0');
  p++;
}

// Check for malformed format string
if (p >= end) {
  mrb_raise(mrb, E_ARGUMENT_ERROR, "malformed format string");
}
