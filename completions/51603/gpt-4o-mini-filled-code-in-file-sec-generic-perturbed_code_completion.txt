if (m->factor_op == FILE_FACTOR_OP_NONE) {
		m->factor_op = FILE_FACTOR_OP_PLUS;
		m->factor = 1;
	} else if (m->factor_op == FILE_FACTOR_OP_PLUS) {
		m->factor += 1;
	} else if (m->factor_op == FILE_FACTOR_OP_MINUS) {
		m->factor -= 1;
	} else if (m->factor_op == FILE_FACTOR_OP_TIMES) {
		m->factor *= 2;
	} else if (m->factor_op == FILE_FACTOR_OP_DIV) {
		if (m->factor > 1) {
			m->factor /= 2;
		} else {
			file_magwarn(ms, "Factor cannot be zero or negative");
			m->factor_op = FILE_FACTOR_OP_NONE;
			m->factor = 0;
			return -1;
		}
	}
