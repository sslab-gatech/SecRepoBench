if (m->type == FILE_NAME) {
	file_magwarn(ms, "Strength is not supported for file names");
	goto out;
}
if (m->factor_op != FILE_FACTOR_OP_NONE) {
	file_magwarn(ms, "Duplicate strength");
	goto out;
}
if (*l == '>') {
	l++;
	m->factor_op = FILE_FACTOR_OP_PLUS;
} else if (*l == '<') {
	l++;
	m->factor_op = FILE_FACTOR_OP_MINUS;
} else if (*l == '*') {
	l++;
	m->factor_op = FILE_FACTOR_OP_TIMES;
} else if (*l == '/') {
	l++;
	m->factor_op = FILE_FACTOR_OP_DIV;
}
