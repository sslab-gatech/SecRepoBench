/* Obtain the magic struct for this entry */
	struct magic *m = &myself->mp[myself->cont_count ? myself->cont_count - 1 : 0];

	/* Check if a strength operation is already set */
	if (m->factor_op != FILE_FACTOR_OP_NONE) {
		file_magwarn(ms, "Multiple strength types defined in one entry");
		goto out;
	}

	/* Check if type is FILE_NAME, which does not support strength settings */
	if (m->type == FILE_NAME) {
		file_magwarn(ms, "Strength settings are not supported for FILE_NAME type");
		goto out;
	}

	/* Parse the factor operation */
	switch (*l) {
	case '+':
		m->factor_op = FILE_FACTOR_OP_ADD;
		l++;
		break;
	case '-':
		m->factor_op = FILE_FACTOR_OP_SUB;
		l++;
		break;
	case '*':
		m->factor_op = FILE_FACTOR_OP_MUL;
		l++;
		break;
	case '/':
		m->factor_op = FILE_FACTOR_OP_DIV;
		l++;
		break;
	default:
		m->factor_op = FILE_FACTOR_OP_NONE;
		break;
	}
