if (m->type == FILE_NAME) {
	file_magwarn(ms, "strength settings not supported for FILE_NAME");
	goto out;
}
m->factor_op = *l++;
if (m->factor_op != FILE_FACTOR_OP_ADD && m->factor_op != FILE_FACTOR_OP_MUL &&
    m->factor_op != FILE_FACTOR_OP_DIV) {
	file_magwarn(ms, "unknown strength operator `%c'", m->factor_op);
	goto out;
}
while (isspace(CAST(unsigned char, *l)))
	l++;
