if (m->factor_op == FILE_FACTOR_OP_NONE) {
    if (*l == '+') {
        m->factor_op = FILE_FACTOR_OP_ADD;
        l++;
    } else if (*l == '-') {
        m->factor_op = FILE_FACTOR_OP_SUB;
        l++;
    } else if (*l == '*') {
        m->factor_op = FILE_FACTOR_OP_MUL;
        l++;
    } else if (*l == '/') {
        m->factor_op = FILE_FACTOR_OP_DIV;
        l++;
    } else {
        file_magwarn(ms, "Bad factor op `%c'", *l);
        goto out;
    }
    EATAB;
}
