struct magic *m = &myself->m;

	if (m->factor_op != FILE_FACTOR_OP_NONE) {
		file_magwarn(ms, "Multiple strength modifiers");
		goto out;
	}

	if (m->type == FILE_NAME) {
		file_magwarn(ms, "Strength settings are not supported for file names");
		goto out;
	}

	EATAB;
	switch (*l) {
	case '+':
		m->factor_op = FILE_FACTOR_OP_ADD;
		l++;
		break;
	case '/':
		m->factor_op = FILE_FACTOR_OP_DIV;
		l++;
		break;
	default:
		m->factor_op = FILE_FACTOR_OP_NONE;
		break;
	}
