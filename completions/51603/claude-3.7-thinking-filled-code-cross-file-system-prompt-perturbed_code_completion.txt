struct magic *m = &myself->m;
if (m->factor_op != FILE_FACTOR_OP_NONE) {
    file_magwarn(ms, "Current entry already has a strength type");
    return -1;
}

if (m->type == FILE_NAME) {
    file_magwarn(ms, "Strength setting is not supported for FILE_NAME");
    return -1;
}

/* Parse the operation type */
switch (*l) {
case '+':
case '-':
case '*':
case '/':
    m->factor_op = *l;
    l++;
    break;
default:
    file_magwarn(ms, "Unknown factor op `%c'", *l);
    goto out;
}
