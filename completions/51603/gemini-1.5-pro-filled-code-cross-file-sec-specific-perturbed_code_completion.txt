struct magic *m = myself;
	if (m->type != FILE_MAGIC) {
		file_magwarn(ms,
		    "Strength setting is not supported for `%s' magic entries",
		    file_magictype(m->type));
		return -1;
	}
	if (*l == '\0') {
		file_magwarn(ms, "Missing operator");
		goto out;
	}
	switch (*l++) {
	case '+':
		m->factor_op = FILE_FACTOR_OP_PLUS;
		break;
	case '-':
		m->factor_op = FILE_FACTOR_OP_MINUS;
		break;
	case '*':
		m->factor_op = FILE_FACTOR_OP_TIMES;
		break;
	case '/':
		m->factor_op = FILE_FACTOR_OP_DIV;
		break;
	default:
		file_magwarn(ms, "Unknown factor op `%c'", *--l);
		goto out;
	}
