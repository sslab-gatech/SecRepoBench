struct magic *m = &myself->mp[myself->cont_count == 0 ? 0 : myself->cont_count - 1];

if (m->factor_op != FILE_FACTOR_OP_NONE) {
    file_magwarn(ms, "Strength already defined for this entry");
    return -1;
}

if (m->type == FILE_NAME) {
    file_magwarn(ms, "Strength not supported for FILE_NAME entries");
    return -1;
}

switch (*l) {
    case '+':
        m->factor_op = FILE_FACTOR_OP_PLUS;
        break;
    case '-':
        m->factor_op = FILE_FACTOR_OP_MINUS;
        break;
    case '*':
        m->factor_op = FILE_FACTOR_OP_TIMES;
        break;
    case '/':
        m->factor_op = FILE_FACTOR_OP_DIV;
        break;
    default:
        file_magwarn(ms, "Unknown factor operation '%c'", *l);
        return -1;
}
l++;
