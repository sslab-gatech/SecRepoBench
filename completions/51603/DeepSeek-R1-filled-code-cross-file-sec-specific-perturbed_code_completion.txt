if (m->factor_op != FILE_FACTOR_OP_NONE) {
		file_magwarn(ms, "Multiple strength factors specified");
		goto out;
	}
	if (m->type == FILE_NAME) {
		file_magwarn(ms, "Strength settings are not supported for file names");
		goto out;
	}
	if (*l != '+' && *l != '-' && *l != '*' && *l != '/') {
		file_magwarn(ms, "Invalid strength factor operation '%c'", *l);
		goto out;
	}
	m->factor_op = *l++;
