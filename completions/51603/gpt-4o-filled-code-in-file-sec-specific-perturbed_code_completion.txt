if (me->mp == NULL) {
		file_magwarn(ms, "Current entry has no magic");
		return -1;
	}
	if (me->mp->type == FILE_NAME) {
		file_magwarn(ms, "Strength setting not supported for `name'");
		return -1;
	}

	switch (*l) {
	case FILE_FACTOR_OP_PLUS:
	case FILE_FACTOR_OP_MINUS:
	case FILE_FACTOR_OP_TIMES:
	case FILE_FACTOR_OP_DIV:
		me->mp->factor_op = *l++;
		break;
	default:
		file_magwarn(ms, "Unknown factor op `%c'", *l);
		return -1;
	}
