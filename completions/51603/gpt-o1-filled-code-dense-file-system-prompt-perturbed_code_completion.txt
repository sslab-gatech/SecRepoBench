struct magic *m = myself->mp;

	if (m->factor_op != FILE_FACTOR_OP_NONE) {
		file_magwarn(ms, "Strength type already set");
		goto out;
	}
	if (m->type == FILE_NAME) {
		file_magwarn(ms,
		    "Strength setting not supported for `name` file type");
		goto out;
	}

	switch (*l) {
	case '+':
		m->factor_op = FILE_FACTOR_OP_PLUS;
		l++;
		break;
	case '-':
		m->factor_op = FILE_FACTOR_OP_MINUS;
		l++;
		break;
	case '*':
		m->factor_op = FILE_FACTOR_OP_TIMES;
		l++;
		break;
	case '/':
		m->factor_op = FILE_FACTOR_OP_DIV;
		l++;
		break;
	default:
		m->factor_op = FILE_FACTOR_OP_NONE;
		break;
	}
