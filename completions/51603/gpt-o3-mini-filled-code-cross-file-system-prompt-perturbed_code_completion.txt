while (isspace((unsigned char)*l))
		l++;
	if (*l == '\0') {
		file_magwarn(ms, "Missing strength operator");
		goto out;
	}
	/* Check if strength operator is already defined */
	if (myself->type == FILE_NAME) {
		file_magwarn(ms, "Strength settings not supported for file name entries");
		goto out;
	}
	if (m->factor_op != FILE_FACTOR_OP_NONE) {
		file_magwarn(ms, "Multiple strength settings specified");
		goto out;
	}
	switch(*l) {
	case '*':
		m->factor_op = FILE_FACTOR_OP_MUL;
		break;
	case '/':
		m->factor_op = FILE_FACTOR_OP_DIV;
		break;
	default:
		file_magwarn(ms, "Unrecognized strength operator `%c'", *l);
		goto out;
	}
	l++;
	while (isspace((unsigned char)*l))
		l++;
