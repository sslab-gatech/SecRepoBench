if (m->factor_op != FILE_FACTOR_OP_NONE) {
	file_magwarn(ms, "Strength type already defined");
	goto out;
}

if (m->type == FILE_NAME) {
	file_magwarn(ms, "Strength settings are not supported for names");
	goto out;
}

while (isspace((unsigned char)*l))
	l++;

if (!*l) {
	file_magwarn(ms, "Missing strength operator");
	goto out;
}

switch (*l) {
case '*':
	m->factor_op = FILE_FACTOR_OP_MUL;
	break;
case '/':
	m->factor_op = FILE_FACTOR_OP_DIV;
	break;
default:
	file_magwarn(ms, "Unknown strength operator `%c'", *l);
	goto out;
}
l++;

while (isspace((unsigned char)*l))
	l++;
