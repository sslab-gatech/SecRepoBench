if (m->factor_op != FILE_FACTOR_OP_NONE) {
    file_magwarn(ms, "Strength already defined");
    return -1;
}

if (m->type == FILE_NAME) {
    file_magwarn(ms, "Strength setting is not supported for FILE_NAME type");
    return -1;
}

switch (*l++) {
    case '*':
        m->factor_op = FILE_FACTOR_OP_TIMES;
        break;
    case '/':
        m->factor_op = FILE_FACTOR_OP_DIV;
        break;
    case '+':
        m->factor_op = FILE_FACTOR_OP_PLUS;
        break;
    case '-':
        m->factor_op = FILE_FACTOR_OP_MINUS;
        break;
    default:
        file_magwarn(ms, "Unknown factor op `%c'", *--l);
        return -1;
}

if (*l == '\0') {
    file_magwarn(ms, "Missing factor");
    return -1;
}
