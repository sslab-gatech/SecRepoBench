if (m->factor_op != FILE_FACTOR_OP_NONE) {
    file_magwarn(ms, "Duplicate factor operation");
    goto out;
}

switch (*l) {
case '/':
    m->factor_op = FILE_FACTOR_OP_DIV;
    break;
case '*':
    m->factor_op = FILE_FACTOR_OP_MUL;
    break;
default:
    file_magwarn(ms, "Invalid factor operation `%c'", *l);
    goto out;
}
l++;
