if (m->factor_op != FILE_FACTOR_OP_NONE) {
    file_magwarn(ms, "Factor operation already set");
    goto out;
}

switch (*l) {
case '*':
    m->factor_op = FILE_FACTOR_OP_MULT;
    l++;
    break;
case '/':
    m->factor_op = FILE_FACTOR_OP_DIV;
    l++;
    break;
default:
    file_magwarn(ms, "Unknown factor operation `%c'", *l);
    goto out;
}
