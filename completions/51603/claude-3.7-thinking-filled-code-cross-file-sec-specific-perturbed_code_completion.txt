struct magic *m = myself->mp;

if (m->factor_op != FILE_FACTOR_OP_NONE) {
    file_magwarn(ms, "Current entry already has a strength type");
    return -1;
}

if (m->type == FILE_NAME) {
    file_magwarn(ms, "Strength setting is not supported for NAME type");
    return -1;
}

switch (*l) {
case '+':
    m->factor_op = FILE_FACTOR_OP_PLUS;
    break;
case '-':
    m->factor_op = FILE_FACTOR_OP_MINUS;
    break;
case '*':
    m->factor_op = FILE_FACTOR_OP_TIMES;
    break;
case '/':
    m->factor_op = FILE_FACTOR_OP_DIV;
    break;
default:
    file_magwarn(ms, "Unknown factor op `%c'", *l);
    goto out;
}
l++;
