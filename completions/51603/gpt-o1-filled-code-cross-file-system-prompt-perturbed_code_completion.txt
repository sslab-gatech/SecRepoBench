struct magic *m = myself->mp;
if (m->factor_op != FILE_FACTOR_OP_NONE) {
	file_magwarn(ms, "Strength setting already defined");
	goto out;
}
if (m->type == FILE_NAME) {
	file_magwarn(ms, "Strength setting not supported for file name rules");
}
EATAB;
switch (*l) {
case '*':
	m->factor_op = FILE_FACTOR_OP_MULT;
	l++;
	break;
case '/':
	m->factor_op = FILE_FACTOR_OP_DIV;
	l++;
	break;
case '+':
	m->factor_op = FILE_FACTOR_OP_ADD;
	l++;
	break;
case '-':
	m->factor_op = FILE_FACTOR_OP_SUB;
	l++;
	break;
default:
	file_magwarn(ms, "Invalid or missing factor operation '%c'", *l);
	goto out;
}
EATAB;
