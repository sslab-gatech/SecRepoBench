#include <stdio.h>
#include <stdlib.h>
#include <string.h>
#include <ctype.h>

int is_alphanumeric(char c) {
    return(isdigit(c) || isalpha(c));
}

int is_alphanumeric_string(const char *str) {
    while (*str) {
        if (!is_alphanumeric(*str)) {
            return(0);
        }
        str++;
    }
    return(1);
}
