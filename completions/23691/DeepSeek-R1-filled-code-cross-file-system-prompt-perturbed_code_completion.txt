static void ndpi_reset_packet_line_info(struct ndpi_packet_struct *pkt) {
  // Reset all the packet line information fields in the ndpi_packet_struct to their initial state.
  // This includes setting pointers to NULL, integer fields to 0, and ensuring that any
  // buffer lengths are also set to 0. This prepares the structure for new data processing.
  int i;
  
  for(i = 0; i < sizeof(pkt->line)/sizeof(pkt->line[0]); i++) {
    pkt->line[i].ptr = NULL;
    pkt->line[i].len = 0;
  }

  pkt->host_line.ptr = NULL;
  pkt->host_line.len = 0;
  pkt->content_line.ptr = NULL;
  pkt->content_line.len = 0;
  pkt->user_agent_line.ptr = NULL;
  pkt->user_agent_line.len = 0;
  pkt->referer_line.ptr = NULL;
  pkt->referer_line.len = 0;

  pkt->num_lines = 0;
  pkt->http_method = 0;
  pkt->http_status_code = 0;
  pkt->detected_http_packet = 0;
}
