  pkt->line[i].ptr = NULL;
  pkt->line[i].len = 0;
}

pkt->parsed_lines = 0;

// Reset content line information
pkt->content_line.ptr = NULL;
pkt->content_line.len = 0;

// Reset cookie line information
pkt->cookie_line.ptr = NULL;
pkt->cookie_line.len = 0;

// Reset user agent line information
pkt->user_agent_line.ptr = NULL;
pkt->user_agent_line.len = 0;
