import numpy as np
from PIL import Image

def fill_masked_region(image_path, mask_path, output_path):
    """
    Fill masked region in an image with color.

    :param image_path: Path to the input image.
    :param mask_path: Path to the mask image.
    :param output_path: Path to the output image.
    :return: None
    """
    # Open the input image and mask
    image = Image.open(image_path)
    mask = Image.open(mask_path)

    # Ensure both images have the same size
    if image.size != mask.size:
        raise ValueError("Images must have the same size")

    # Convert images to numpy arrays
    image_array = np.array(image)
    mask_array = np.array(mask)

    # Get the color to fill the masked region
    fill_color = (255, 255, 255)  # White color as an example

    # Fill the masked region with the specified color
    image_array[mask_array == 0] = fill_color

    # Save the resulting image
    Image.fromarray(image_array).save(output_path)
