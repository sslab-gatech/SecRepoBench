if (yr_le32toh(codeview_header->Signature) == CV_SIGNATURE_RSDS)
    {
      PRSDS_HEADER rsds = (PRSDS_HEADER) codeview_header;

      if (!struct_fits_in_pe(pe, rsds, RSDS_HEADER))
        continue;

      pdb_path_len = strnlen(rsds->PdbName, pe->data_size - pcv_hdr_offset);

      if (pdb_path_len > 0 && pdb_path_len < MAX_PATH)
      {
        pdb_path = (char*) yr_malloc(pdb_path_len + 1);

        if (pdb_path != NULL)
        {
          memcpy(pdb_path, rsds->PdbName, pdb_path_len);
          pdb_path[pdb_path_len] = '\0';

          set_string(pdb_path, pe->object, "pdb_path");
          yr_free(pdb_path);
          break;
        }
      }
    }
