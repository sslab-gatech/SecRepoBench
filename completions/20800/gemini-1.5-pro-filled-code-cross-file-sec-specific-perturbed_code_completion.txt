if (codeview_header->dwSignature == 'SDSR')  // RSDS type
    {
      CV_RSDS_PDB70* pdb70_header = (CV_RSDS_PDB70*)codeview_header;

      if (struct_fits_in_pe(pe, pdb70_header, CV_RSDS_PDB70))
      {
        pdb_path_len = strnlen(pdb70_header->szPDB, sizeof(pdb70_header->szPDB));
        if (pdb_path_len > 0 && pdb_path_len < MAX_PATH)
        {
          pdb_path = yr_strndup(pdb70_header->szPDB, pdb_path_len);
        }
      }
    }
    else if (codeview_header->dwSignature == 'NB10') // NB10 type
    {
      CV_NB10_PDB70* pdb70_header = (CV_NB10_PDB70*)codeview_header;

      if (struct_fits_in_pe(pe, pdb70_header, CV_NB10_PDB70))
      {
        pdb_path_len = strnlen(pdb70_header->szPDB, sizeof(pdb70_header->szPDB));
        if (pdb_path_len > 0 && pdb_path_len < MAX_PATH)
        {
          pdb_path = yr_strndup(pdb70_header->szPDB, pdb_path_len);
        }
      }
    }

    if (pdb_path != NULL)
    {
      set_string(pdb_path, pe->object, "pdb_path");
      yr_free(pdb_path);
      break; 
    }
