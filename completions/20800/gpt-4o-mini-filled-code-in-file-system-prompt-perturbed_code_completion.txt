if (yr_le32toh(debug_dir->AddressOfRawData) != 0)
    {
      // The CodeView header is located at the AddressOfRawData.
      pcv_hdr_offset = pe_rva_to_offset(pe, yr_le32toh(debug_dir->AddressOfRawData));

      if (pcv_hdr_offset < 0)
        continue;

      // Check if the CodeView header fits in the PE bounds.
      if (!struct_fits_in_pe(pe, (PIMAGE_DEBUG_DIRECTORY)(pe->data + pcv_hdr_offset), IMAGE_DEBUG_DIRECTORY))
        continue;

      // Extract the PDB path from the CodeView header.
      PDB_HEADER pdb_header = (PDB_HEADER)(pe->data + pcv_hdr_offset + sizeof(IMAGE_DEBUG_DIRECTORY));
      pdb_path_len = strnlen((char*)pdb_header->PdbFileName, sizeof(pdb_header->PdbFileName));

      if (pdb_path_len > 0 && wide_string_fits_in_pe(pe, (char*)pdb_header->PdbFileName))
      {
        pdb_path = (char*)yr_malloc(pdb_path_len + 1);
        if (pdb_path)
        {
          memcpy(pdb_path, pdb_header->PdbFileName, pdb_path_len);
          pdb_path[pdb_path_len] = '\0';
          set_string(pdb_path, pe->object, "pdb_path");
          yr_free(pdb_path);
        }
      }
    }
