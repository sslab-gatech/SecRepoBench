if (struct_fits_in_pe(pe, codeview_header, PCV_HEADER))
    {
      switch (codeview_header->signature)
      {
        case CV_PDB_SIGNATURE:
        {
          PDB_HEADER pdb_header = (PDB_HEADER) codeview_header;
          if (struct_fits_in_pe(pe, pdb_header, PDB_HEADER))
          {
            pdb_path_len = strnlen(pdb_header->pdb_path, MAX_PDB_PATH_LENGTH);
            if (pdb_path_len > 0 && pdb_path_len < MAX_PDB_PATH_LENGTH)
            {
              pdb_path = (char*) yr_malloc(pdb_path_len + 1);
              if (pdb_path != NULL)
              {
                strncpy(pdb_path, pdb_header->pdb_path, pdb_path_len + 1);
                set_string(pdb_path, pe->object, "pdb_path");
                yr_free(pdb_path);
              }
            }
          }
          break;
        }
        case CV_RSDS_SIGNATURE:
        case CV_INFO_SIGNATURE:
        case CV_NB10_SIGNATURE:
        case CV_NB11_SIGNATURE:
          break;
      }
    }
