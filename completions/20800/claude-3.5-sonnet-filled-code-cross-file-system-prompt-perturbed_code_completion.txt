if (!struct_fits_in_pe(pe, codeview_header, PCV_HEADER))
  continue;

if (codeview_header->Signature == CVINFO_PDB20_CODEVIEW_SIGNATURE)
{
  PCV_INFO_PDB20 cv_pdb20 = (PCV_INFO_PDB20) codeview_header;
  if (struct_fits_in_pe(pe, cv_pdb20, CV_INFO_PDB20))
  {
    pdb_path = (char*) cv_pdb20->PdbFileName;
    pdb_path_len = strnlen(pdb_path, PE_MAX_PDB_PATH_SIZE);
  }
}
else if (codeview_header->Signature == CVINFO_PDB70_CODEVIEW_SIGNATURE)
{
  PCV_INFO_PDB70 cv_pdb70 = (PCV_INFO_PDB70) codeview_header;
  if (struct_fits_in_pe(pe, cv_pdb70, CV_INFO_PDB70))
  {
    pdb_path = (char*) cv_pdb70->PdbFileName;
    pdb_path_len = strnlen(pdb_path, PE_MAX_PDB_PATH_SIZE);
  }
}

if (pdb_path != NULL && pdb_path_len > 0 && pdb_path_len < PE_MAX_PDB_PATH_SIZE)
{
  set_string(pdb_path, pe->object, "pdb_path");
  break;
}
