if (yr_le32toh(codeview_header->signature) == CV_SIGNATURE_RSDS)
    {
      PRSDS_HEADER rsds = (PRSDS_HEADER) codeview_header;

      if (!struct_fits_in_pe(pe, rsds, RSDS_HEADER))
        continue;

      pdb_path = (char*) (rsds->pdb_name);

      if (!fits_in_pe(pe, pdb_path, 1))
        continue;

      pdb_path_len = strnlen(pdb_path, pe->data_size - pcv_hdr_offset);

      if (pdb_path_len > 0 && pdb_path_len < MAX_PDB_PATH)
      {
        set_string(pdb_path, pe->object, "pdb_path");
        break;
      }
    }
    else if (yr_le32toh(codeview_header->signature) == CV_SIGNATURE_NB10)
    {
      PNB10_HEADER nb10 = (PNB10_HEADER) codeview_header;
