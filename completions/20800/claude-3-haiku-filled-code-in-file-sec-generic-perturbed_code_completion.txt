if (yr_le32toh(debug_dir->Type) == IMAGE_DEBUG_TYPE_CODEVIEW)
{
    if (yr_le32toh(debug_dir->AddressOfRawData) != 0)
    {
        PCV_HEADER codeview_header = (PCV_HEADER) (pe->data + pcv_hdr_offset);

        if (yr_le32toh(codeview_header->Signature) == CODEVIEW_PDB_70_SIGNATURE)
        {
            PPDB_70_INFO pdb_70_info = (PPDB_70_INFO) codeview_header;
            pdb_path = (char*) yr_strndup((char*) pdb_70_info->PdbFileName, yr_min(available_space(pe, pdb_70_info->PdbFileName), 1024));
            set_string(pdb_path, pe->object, "pdb_path");
            yr_free(pdb_path);
        }
        else if (yr_le32toh(codeview_header->Signature) == CODEVIEW_PDB_20_SIGNATURE)
        {
            PPDB_20_INFO pdb_20_info = (PPDB_20_INFO) codeview_header;
            pdb_path = (char*) yr_strndup((char*) pdb_20_info->PdbFileName, yr_min(available_space(pe, pdb_20_info->PdbFileName), 1024));
            set_string(pdb_path, pe->object, "pdb_path");
            yr_free(pdb_path);
        }
    }
}
