switch (yr_le32toh(codeview_header->Signature)) {
      case CV_SIGNATURE_RSDS: {
        PCV_RSDS_HEADER rsds_header = (PCV_RSDS_HEADER) codeview_header;
        if (struct_fits_in_pe(pe, rsds_header, CV_RSDS_HEADER) &&
            fits_in_pe(pe, rsds_header->PdbFileName,
                       rsds_header->PdbFileNameLength)) {
          pdb_path_len = rsds_header->PdbFileNameLength;
          pdb_path = (char*) yr_malloc(pdb_path_len + 1);
          if (pdb_path != NULL) {
            memcpy(pdb_path, rsds_header->PdbFileName, pdb_path_len);
            pdb_path[pdb_path_len] = '\0';
            set_string(pdb_path, pe->object, "pdb_path");
            yr_free(pdb_path);
          }
        }
        break;
      }
      case CV_SIGNATURE_NB10:
      case CV_SIGNATURE_NB11: {
        PCV_NB10_HEADER nb10_header = (PCV_NB10_HEADER) codeview_header;
        if (struct_fits_in_pe(pe, nb10_header, CV_NB10_HEADER) &&
            fits_in_pe(pe, nb10_header->PdbFileName,
                       nb10_header->PdbFileNameLength)) {

            pdb_path[pdb_path_len] = '\0';

        }
