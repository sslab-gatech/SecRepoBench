if (codeview_header->Signature == CV_SIGNATURE_PDB70)
    {
      PCV_HEADER_PDB70 pdb70_header = (PCV_HEADER_PDB70) codeview_header;

      if (!struct_fits_in_pe(pe, pdb70_header, CV_HEADER_PDB70))
        continue;

      pdb_path = pdb70_header->PdbFileName;
      pdb_path_len = sizeof(pdb70_header->PdbFileName);
    }
    else if (codeview_header->Signature == CV_SIGNATURE_PDB20)
    {
      PCV_HEADER_PDB20 pdb20_header = (PCV_HEADER_PDB20) codeview_header;

      if (!struct_fits_in_pe(pe, pdb20_header, CV_HEADER_PDB20))
        continue;

      pdb_path = pdb20_header->PdbFileName;
      pdb_path_len = sizeof(pdb20_header->PdbFileName);
    }
    else
    {
      continue;
    }

    // Ensure the PDB path fits within the PE file bounds.
    if (!fits_in_pe(pe, pdb_path, pdb_path_len))
      continue;

    // Compute the length of the PDB path.
    size_t len = strnlen(pdb_path, pdb_path_len);

    // Ensure the PDB path length is within reasonable limits.
    if (len <= 0 || len >= MAX_PDB_PATH)
      continue;

    // Store the PDB path in the object associated with the PE.
    set_string(pdb_path, pe->object, "pdb_path");
    
    // Stop processing after finding a valid PDB path.
    break;
