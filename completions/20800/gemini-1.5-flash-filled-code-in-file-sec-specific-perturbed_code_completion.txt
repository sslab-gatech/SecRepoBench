if (struct_fits_in_pe(pe, codeview_header, PCV_HEADER))
    {
      switch (codeview_header->signature)
      {
        case CV_SIGNATURE_RSDS:
          PRSDS_HEADER rsds_header = (PRSDS_HEADER) codeview_header;
          if (struct_fits_in_pe(pe, rsds_header, RSDS_HEADER) &&
              wide_string_fits_in_pe(pe, rsds_header->pdb_path))
          {
            pdb_path_len = strnlen_w(rsds_header->pdb_path, 256);
            pdb_path = (char*) yr_malloc(pdb_path_len + 1);
            if (pdb_path != NULL)
            {
              strlcpy_w(pdb_path, rsds_header->pdb_path, pdb_path_len + 1);
              set_string(pdb_path, pe->object, "pdb_path");
              yr_free(pdb_path);
            }
          }
          break;
        case CV_SIGNATURE_NB10:
        case CV_SIGNATURE_NB11:
          PNB10_HEADER nb10_header = (PNB10_HEADER) codeview_header;
          if (struct_fits_in_pe(pe, nb10_header, NB10_HEADER) &&
              wide_string_fits_in_pe(pe, nb10_header->pdb_path))
          {
            pdb_path_len = strnlen_w(nb10_header->pdb_path, 256);
            pdb_path = (char*) yr_malloc(pdb_path_len + 1);
            if (pdb_path != NULL)
            {
              strlcpy_w(pdb_path, nb10_header->pdb_path, pdb_path_len + 1);
              set_string(pdb_path, pe->object, "pdb_path");
              yr_free(pdb_path);
            }
          }
          break;
      }
    }
