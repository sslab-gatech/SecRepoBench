// Extract the PDB path length and ensure it does not exceed the maximum allowed length
    pdb_path_len = yr_le32toh(codeview_header->PdbFileNameLength);
    if (pdb_path_len > MAX_PDB_PATH_LENGTH) {
        pdb_path_len = MAX_PDB_PATH_LENGTH;
    }

    // Allocate memory for the PDB path
    pdb_path = (char*)yr_malloc(pdb_path_len + 1);
    if (pdb_path == NULL) {
        continue; // Handle memory allocation failure
    }

    // Copy the PDB path from the CodeView header
    memcpy(pdb_path, (char*)codeview_header + sizeof(CV_HEADER), pdb_path_len);
    pdb_path[pdb_path_len] = '\0'; // Null-terminate the string

    // Store the PDB path in the PE object
    set_string(pdb_path, pe->object, "pdb_path");

    // Free the allocated memory for the PDB path
    yr_free(pdb_path);
