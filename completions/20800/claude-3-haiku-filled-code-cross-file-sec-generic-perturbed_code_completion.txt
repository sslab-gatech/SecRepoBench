// Determine the type of CodeView debug information by checking the signature.
if (strncmp(codeview_header->Signature, "NB10", 4) == 0)
{
    PNB10_HEADER nb10_header = (PNB10_HEADER)codeview_header;
    pdb_path = (char*)(nb10_header + 1);
    pdb_path_len = strlen(pdb_path);
}
else if (strncmp(codeview_header->Signature, "RSDS", 4) == 0)
{
    PRSDS_HEADER rsds_header = (PRSDS_HEADER)codeview_header;
    pdb_path = (char*)(rsds_header + 1);
    pdb_path_len = strlen(pdb_path);
}
else
{
    return;
}

// Ensure the PDB path fits within the PE bounds
if (pdb_path_len > 0 && pdb_path_len < MAX_PATH && fits_in_pe(pe, pdb_path, pdb_path_len))
{
    set_string(pdb_path, pe->object, "pdb_path");
    return;
}
