// Determine the type of CodeView debug information by checking the signature.
    // Depending on the signature, cast the header to the appropriate PDB structure.
    // Extract the PDB file name from the structure if it fits within the PE bounds.
    // Compute the length of the PDB path and ensure it's within valid limits.
    // If a valid PDB path is found, store the path in the object associated with the PE.
    // Stop further processing after successfully storing a valid PDB path.
    switch (yr_le32toh(codeview_header->Signature)) {
        case CV_SIGNATURE_C13:
            // CV_HEADER_C13 is the structure for CV_SIGNATURE_C13
            PCV_HEADER_C13 cv_c13_header = (PCV_HEADER_C13)codeview_header;
            pdb_path_len = strlen(cv_c13_header->PdbFileName);
            break;
        case CV_SIGNATURE_C23:
            // CV_HEADER_C23 is the structure for CV_SIGNATURE_C23
            PCV_HEADER_C23 cv_c23_header = (PCV_HEADER_C23)codeview_header;
            pdb_path_len = strlen(cv_c23_header->PdbFileName);
            break;
        case CV_SIGNATURE_PDB20:
            // CV_HEADER_PDB20 is the structure for CV_SIGNATURE_PDB20
            PCV_HEADER_PDB20 cv_pdb20_header = (PCV_HEADER_PDB20)codeview_header;
            pdb_path_len = strlen(cv_pdb20_header->PdbFileName);
            break;
        default:
            continue;
    }

    if (pdb_path_len > MAX_PDB_PATH_LENGTH) {
        pdb_path_len = MAX_PDB_PATH_LENGTH;
    }

    pdb_path = (char*)malloc(pdb_path_len + 1);
    if (pdb_path == NULL) {
        continue;
    }

    switch (yr_le32toh(codeview_header->Signature)) {
        case CV_SIGNATURE_C13:
            PCV_HEADER_C13 cv_c13_header = (PCV_HEADER_C13)codeview_header;
            strncpy(pdb_path, cv_c13_header->PdbFileName, pdb_path_len);
            break;
        case CV_SIGNATURE_C23:
            PCV_HEADER_C23 cv_c23_header = (PCV_HEADER_C23)codeview_header;
            strncpy(pdb_path, cv_c23_header->PdbFileName, pdb_path_len);
            break;
        case CV_SIGNATURE_PDB20:
            PCV_HEADER_PDB20 cv_pdb20_header = (PCV_HEADER_PDB20)codeview_header;
            strncpy(pdb_path, cv_pdb20_header->PdbFileName, pdb_path_len);
            break;
    }

    pdb_path[pdb_path_len] = '\0';

    set_string(pdb_path, pe->object, "pdb_path");

    free(pdb_path);
    return;
