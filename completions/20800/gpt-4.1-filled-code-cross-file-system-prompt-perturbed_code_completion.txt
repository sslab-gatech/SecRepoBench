        // The PDB file name is after the GUID and age
        char* filename = (char*) (pdb70->PdbFileName);
        // Ensure the filename fits in the PE
        if (fits_in_pe(pe, filename, 1))
        {
          pdb_path_len = strnlen(filename, pe->data_size - (size_t)((uint8_t*)filename - pe->data));
          if (pdb_path_len > 0 && pdb_path_len < 512)
          {
            pdb_path = yr_strndup(filename, pdb_path_len);
            if (pdb_path != NULL)
            {
              set_string(pdb_path, pe->object, "pdb_path");
              yr_free(pdb_path);
              break;
            }
          }
        }
      }
    }
