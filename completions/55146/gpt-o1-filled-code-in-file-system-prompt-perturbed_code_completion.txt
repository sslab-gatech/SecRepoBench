if (parse_context->err || parse_context->pos >= parse_context->end || *parse_context->pos != '>') {
    PLIST_OSTEP_ERR("Missing '>' while parsing data at offset %ld\n", parse_context->pos - parse_context->start);
    parse_context->err++;
    byte_array_free(bytes);
    plist_free_data(data);
    break;
}
