if (parse_context->err) {
    byte_array_free(bytes);
    plist_free_data(data);
    PLIST_OSTEP_ERR("Error parsing data at offset %ld\n", parse_context->pos - parse_context->start);
    goto err_out;
}
if (*parse_context->pos != '>') {
    byte_array_free(bytes);
    plist_free_data(data);
    PLIST_OSTEP_ERR("Missing terminating '>' for data at offset %ld\n", parse_context->pos - parse_context->start);
    parse_context->err++;
    goto err_out;
}
