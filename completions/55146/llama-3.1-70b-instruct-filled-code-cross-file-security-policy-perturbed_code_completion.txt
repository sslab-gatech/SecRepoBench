if (parse_context->err) {
    plist_free_data(data);
    byte_array_free(bytes);
    return;
}
if (*parse_context->pos!= '>') {
    plist_free_data(data);
    byte_array_free(bytes);
    PLIST_OSTEP_ERR("Missing terminating '>' at offset %ld\n", parse_context->pos - parse_context->start);
    parse_context->err++;
    return;
}
