if (y == NULL || operand == NULL) {
  return -1; // or handle error appropriately
}

size_t y_digits = digits(y);
size_t operand_digits = digits(operand);

if (y_digits < operand_digits) {
  return -1;
}
if (y_digits > operand_digits) {
  return 1;
}

for (size_t i = y_digits; i > 0; i--) {
  if (y->p[i-1] < operand->p[i-1]) {
    return -1;
  }
  if (y->p[i-1] > operand->p[i-1]) {
    return 1;
  }
}

return 0;
