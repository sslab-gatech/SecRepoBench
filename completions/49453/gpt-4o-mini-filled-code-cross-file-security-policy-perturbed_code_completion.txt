if (y == NULL || operand == NULL) {
    return 0; // Handle NULL pointers
  }
  if (y->sz == 0 || operand->sz == 0) {
    return 0; // Handle zero size
  }

  // Compare signs
  if (y->sn < operand->sn) {
    return -1;
  }
  if (y->sn > operand->sn) {
    return 1;
  }

  // Compare absolute values
  size_t size_y = y->sz;
  size_t size_operand = operand->sz;
  if (size_y < size_operand) {
    return -1;
  }
  if (size_y > size_operand) {
    return 1;
  }

  // If sizes are equal, compare limb by limb
  for (size_t i = size_y; i > 0; i--) {
    if (y->limbs[i - 1] < operand->limbs[i - 1]) {
      return -1;
    }
    if (y->limbs[i - 1] > operand->limbs[i - 1]) {
      return 1;
    }
  }

  return 0; // They are equal
