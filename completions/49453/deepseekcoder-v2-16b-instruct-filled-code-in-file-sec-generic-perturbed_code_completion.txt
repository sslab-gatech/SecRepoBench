static int
ucmp(mpz_t *y, mpz_t *operand)
{
  size_t i;
  size_t y_digits = digits(y);
  size_t operand_digits = digits(operand);

  if (y_digits != operand_digits) {
    return (y_digits > operand_digits) ? 1 : -1;
  }

  for (i = y_digits; i > 0; i--) {
    if (y->p[i - 1] != operand->p[i - 1]) {
      return (y->p[i - 1] > operand->p[i - 1]) ? 1 : -1;
    }
  }

  return 0;
}
