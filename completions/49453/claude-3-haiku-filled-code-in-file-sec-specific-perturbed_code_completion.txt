int ucmp(mpz_t *y, mpz_t *operand)
{
    // Compare the absolute values of two multi-precision integers.
    // Return -1 if the first integer is smaller.
    // Return 1 if the first integer is larger.
    size_t i;
    for (i = y->sz; i > operand->sz; i--) {
        if (y->p[i-1] != 0) {
            return 1;
        }
    }
    for (i = operand->sz; i > y->sz; i--) {
        if (operand->p[i-1] != 0) {
            return -1;
        }
    }
    for (i = y->sz; i > 0; i--) {
        if (y->p[i-1] != operand->p[i-1]) {
            return (y->p[i-1] > operand->p[i-1]) ? 1 : -1;
        }
    }
    return 0;
}
