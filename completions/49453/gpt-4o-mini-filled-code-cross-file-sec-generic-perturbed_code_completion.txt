if (y->sz == 0) return (x->sz == 0) ? 0 : 1; // y is zero, x is non-zero
  if (x->sz == 0) return -1; // x is zero, y is non-zero

  // Compare the sizes first
  if (x->sz < y->sz) return -1;
  if (x->sz > y->sz) return 1;

  // Sizes are equal, compare the values
  for (size_t i = x->sz; i > 0; i--) {
    if (x->val[i - 1] < y->val[i - 1]) return -1;
    if (x->val[i - 1] > y->val[i - 1]) return 1;
  }

  return 0; // They are equal
