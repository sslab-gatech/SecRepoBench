size_t x_digits = digits(y);
size_t y_digits = digits(operand);

if (x_digits != y_digits) {
    return x_digits > y_digits ? 1 : -1;
}

for (int i = x_digits - 1; i >= 0; i--) {
    if (y->p[i] > operand->p[i]) {
        return 1;
    } else if (y->p[i] < operand->p[i]) {
        return -1;
    }
}

return 0;
