size_t x_digits = digits(y);
  size_t y_digits = digits(operand);

  if (x_digits < y_digits) {
    return -1;
  }
  else if (x_digits > y_digits) {
    return 1;
  }

  for (size_t i = x_digits; i > 0; i--) {
    if (y->p[i-1] < operand->p[i-1]) {
      return -1;
    }
    else if (y->p[i-1] > operand->p[i-1]) {
      return 1;
    }
  }
  return 0;
