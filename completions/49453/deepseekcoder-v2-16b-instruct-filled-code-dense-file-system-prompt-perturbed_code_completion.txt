int abs_x_size = x->sz;
  int abs_y_size = y->sz;

  if (abs_x_size < abs_y_size)
    return -1;
  if (abs_x_size > abs_y_size)
    return 1;

  for (size_t i = abs_x_size; i > 0; i--) {
    if (x->p[i - 1] < y->p[i - 1])
      return -1;
    if (x->p[i - 1] > y->p[i - 1])
      return 1;
  }

  return 0;
