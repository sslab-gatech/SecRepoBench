if (y->sz == 0 || operand->sz == 0) {
  return 0;
}

int cmp = 0;
size_t i;
for (i = y->sz - 1; i < y->sz; i--) {
  if (y->d[i] < operand->d[i]) {
    cmp = -1;
    break;
  } else if (y->d[i] > operand->d[i]) {
    cmp = 1;
    break;
  }
}

if (cmp == 0 && y->sz != operand->sz) {
  cmp = (y->sz < operand->sz) ? -1 : 1;
}

return cmp;
