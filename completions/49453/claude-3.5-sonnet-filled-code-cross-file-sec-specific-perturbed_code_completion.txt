if (y == NULL || operand == NULL) {
    return 0;  // Return 0 if either pointer is NULL
}

size_t y_size = y->sz;
size_t operand_size = operand->sz;

if (y_size > operand_size) {
    return 1;
}
if (y_size < operand_size) {
    return -1;
}

// If sizes are equal, compare digit by digit
for (size_t i = y_size; i > 0; i--) {
    if (y->d[i-1] > operand->d[i-1]) {
        return 1;
    }
    if (y->d[i-1] < operand->d[i-1]) {
        return -1;
    }
}
