// Check if both are zero
if (y->sz == 0 && operand->sz == 0) return 0;
// If one is zero, the other is larger
if (y->sz == 0) return -1;
if (operand->sz == 0) return 1;

// Compare by number of digits
size_t y_digits = digits(y);
size_t op_digits = digits(operand);
if (y_digits > op_digits) return 1;
if (y_digits < op_digits) return -1;

// Same number of digits, compare individual digits from most significant to least
for (int i = y_digits - 1; i >= 0; i--) {
    if (y->d[i] > operand->d[i]) return 1;
    if (y->d[i] < operand->d[i]) return -1;
}
