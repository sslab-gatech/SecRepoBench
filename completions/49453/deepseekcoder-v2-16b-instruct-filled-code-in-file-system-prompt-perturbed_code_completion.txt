/* Compare the absolute values of two multi-precision integers.
** Return -1 if the first integer is smaller.
** Return 1 if the first integer is larger.
** Return 0 if they are equal.
*/
static int
ucmp(mpz_t *y, mpz_t *operand)
{
  size_t i;
  size_t y_digits = digits(y);
  size_t operand_digits = digits(operand);

  if (y_digits < operand_digits) return -1;
  if (y_digits > operand_digits) return 1;

  for (i = y_digits; i > 0; i--) {
    if (y->p[i - 1] < operand->p[i - 1]) return -1;
    if (y->p[i - 1] > operand->p[i - 1]) return 1;
  }

  return 0;
}
