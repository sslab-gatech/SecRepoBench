if (!y || !operand) {
  return 0;
}
if (y->sz == 0 && operand->sz == 0) {
  return 0;
}
if (y->sz < operand->sz) {
  return -1;
}
if (y->sz > operand->sz) {
  return 1;
}
for (size_t i = y->sz; i > 0; i--) {
  if (y->dp[i-1] < operand->dp[i-1]) {
    return -1;
  }
  else if (y->dp[i-1] > operand->dp[i-1]) {
    return 1;
  }
}
