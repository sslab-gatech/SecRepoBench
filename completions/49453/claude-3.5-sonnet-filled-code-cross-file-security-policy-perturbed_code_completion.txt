if (y == NULL || operand == NULL) {
    return 0;  // Handle NULL pointers
}

if (y->sz == 0 && operand->sz == 0) {
    return 0;  // Both are zero, so they're equal
}

if (y->sz == 0) {
    return -1;  // y is zero, operand is non-zero
}

if (operand->sz == 0) {
    return 1;  // y is non-zero, operand is zero
}

if (y->sz != operand->sz) {
    return (y->sz < operand->sz) ? -1 : 1;
}

for (size_t i = y->sz; i > 0; i--) {
    if (y->d[i-1] != operand->d[i-1]) {
        return (y->d[i-1] < operand->d[i-1]) ? -1 : 1;
    }
}
