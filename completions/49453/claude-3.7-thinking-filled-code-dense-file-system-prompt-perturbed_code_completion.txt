// Check if both numbers are zero
if (uzero(y) && uzero(operand)) {
  return 0;
}

// Check for easy case: different sizes
if (y->sz != operand->sz) {
  return (y->sz > operand->sz) ? 1 : -1;
}

// Same size, compare limbs from most significant to least significant
for (size_t i = y->sz; i-- > 0; ) {
  if (y->p[i] != operand->p[i]) {
    return (y->p[i] > operand->p[i]) ? 1 : -1;
  }
}

// All limbs are equal
return 0;
