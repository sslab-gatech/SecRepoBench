static int
ucmp(mpz_t *y, mpz_t *operand)
{
  // Compare the absolute values of two multi-precision integers.
  // Return -1 if the first integer is smaller.
  // Return 1 if the first integer is larger.
  if (y->sz == 0 || operand->sz == 0) {
    // Handle the case where either operand is zero
    return 0;
  }

  size_t y_size = mpz_sizeinbase(y, 2);
  size_t operand_size = mpz_sizeinbase(operand, 2);

  if (y_size < operand_size) {
    return -1;
  } else if (y_size > operand_size) {
    return 1;
  } else {
    // Compare the most significant bits
    int cmp = lzb(y->d[y->sz - 1]) - lzb(operand->d[operand->sz - 1]);
    if (cmp!= 0) {
      return cmp;
    }

    // Compare the second most significant bits
    cmp = lzb(y->d[y->sz - 2]) - lzb(operand->d[operand->sz - 2]);
    if (cmp!= 0) {
      return cmp;
    }

    // If all bits are equal, the numbers are equal
    return 0;
  }
}
