if (y == NULL || operand == NULL) {
    // Handle the error case where one of the pointers is NULL
    return 0; // or an appropriate error code
  }

  if (y->sz == 0 && operand->sz == 0) {
    return 0; // Both are zero
  } else if (y->sz == 0) {
    return -1; // y is smaller (zero)
  } else if (operand->sz == 0) {
    return 1; // operand is smaller (zero)
  }

  // Compare the sizes of the two operands
  if (y->sz < operand->sz) {
    return -1; // y is smaller
  } else if (y->sz > operand->sz) {
    return 1; // y is larger
  }

  // If sizes are equal, compare the actual values
  for (size_t i = y->sz; i > 0; i--) {
    if (y->data[i - 1] < operand->data[i - 1]) {
      return -1; // y is smaller
    } else if (y->data[i - 1] > operand->data[i - 1]) {
      return 1; // y is larger
    }
  }

  return 0; // They are equal
