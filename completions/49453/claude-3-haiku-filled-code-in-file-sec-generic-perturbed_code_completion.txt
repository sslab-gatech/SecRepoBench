int ucmp(mpz_t *y, mpz_t *operand)
{
    // Compare the absolute values of two multi-precision integers.
    // Return -1 if the first integer is smaller.
    // Return 1 if the first integer is larger.
    size_t y_digits = digits(y);
    size_t operand_digits = digits(operand);

    if (y_digits < operand_digits) {
        return -1;
    } else if (y_digits > operand_digits) {
        return 1;
    } else {
        for (size_t i = y_digits - 1; i < y_digits; i--) {
            if (y->p[i] < operand->p[i]) {
                return -1;
            } else if (y->p[i] > operand->p[i]) {
                return 1;
            }
        }
        return 0;
    }
}
