if (ctxt->vstateNr >= ctxt->vstateMax) {
    xmlValidState *tmp;

    tmp = (xmlValidState *) xmlRealloc(ctxt->vstateTab,
                 2 * ctxt->vstateMax * sizeof(ctxt->vstateTab[0]));
    if (tmp == NULL) {
        xmlVErrMemory(ctxt, "realloc failed");
        return(-1);
    }
    ctxt->vstateMax *= 2;
    ctxt->vstateTab = tmp;
}
