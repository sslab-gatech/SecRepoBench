// <MASK>
switch (content->type) {
    case XML_ELEMENT_CONTENT_ELEMENT:
        if (content->name != NULL) {
            if (content->prefix != NULL) {
                char prefix[50];
                char *fullname = xmlBuildQName(content->name, content->prefix, prefix, 50);
                if (fullname != NULL) {
                    xmlBufferCat(buf, fullname);
                    if (fullname != prefix && fullname != content->name) {
                        xmlFree(fullname);
                    }
                }
            } else {
                xmlBufferCat(buf, content->name);
            }
        }
        break;
    case XML_ELEMENT_CONTENT_PCDATA:
        xmlBufferCat(buf, BAD_CAST "#PCDATA");
        break;
    case XML_ELEMENT_CONTENT_SEQ:
        xmlSnprintfElementContent(buf, capacity - strlen(buf), content->c1, 1);
        xmlBufferCat(buf, ", ");
        xmlSnprintfElementContent(buf, capacity - strlen(buf), content->c2, 1);
        break;
    case XML_ELEMENT_CONTENT_OR:
        xmlSnprintfElementContent(buf, capacity - strlen(buf), content->c1, 1);
        xmlBufferCat(buf, " | ");
        xmlSnprintfElementContent(buf, capacity - strlen(buf), content->c2, 1);
        break;
    default:
        break;
}
