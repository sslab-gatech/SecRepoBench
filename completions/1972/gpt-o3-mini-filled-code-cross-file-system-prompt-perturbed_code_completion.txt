switch (content->type) {
    case XML_ELEMENT_CONTENT_PCDATA:
        strcat(buf, "#PCDATA");
        break;
    case XML_ELEMENT_CONTENT_ELEMENT:
        if ((content->ns != NULL) && (content->ns->prefix != NULL)) {
            strcat(buf, (const char *)content->ns->prefix);
            strcat(buf, ":");
        }
        strcat(buf, (const char *)content->name);
        break;
    case XML_ELEMENT_CONTENT_SEQ:
        xmlSnprintfElementContent(buf, capacity, content->left, 1);
        strcat(buf, ", ");
        xmlSnprintfElementContent(buf, capacity, content->right, 1);
        break;
    case XML_ELEMENT_CONTENT_OR:
        xmlSnprintfElementContent(buf, capacity, content->left, 1);
        strcat(buf, " | ");
        xmlSnprintfElementContent(buf, capacity, content->right, 1);
        break;
    default:
        break;
}
