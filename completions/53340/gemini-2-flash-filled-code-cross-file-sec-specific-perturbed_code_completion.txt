if (maxaudiochannel >= MAX_CHANNELS) {
        av_log(m->avctx, AV_LOG_ERROR,
               "maxaudiochannel %d exceeds maximum channel count %d\n",
               maxaudiochannel, MAX_CHANNELS - 1);
        return AVERROR_INVALIDDATA;
    }

    if (max_matrix_channel >= std_max_matrix_channel) {
        av_log(m->avctx, AV_LOG_ERROR,
               "max_matrix_channel %d exceeds maximum matrix channel count %d\n",
               max_matrix_channel, std_max_matrix_channel - 1);
        avpriv_request_sample(m->avctx, "max_matrix_channel exceeds maximum");
        return AVERROR_PATCHWELCOME;
    }
