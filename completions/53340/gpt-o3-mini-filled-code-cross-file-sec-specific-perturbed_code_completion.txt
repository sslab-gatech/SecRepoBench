if (maxaudiochannel < min_channel) {
    av_log(m->avctx, AV_LOG_ERROR,
           "Invalid channel range: min_channel (%u) is greater than maxaudiochannel (%u)\n",
           min_channel, maxaudiochannel);
    avpriv_request_sample(m->avctx, "min_channel greater than maxaudiochannel");
    return AVERROR_INVALIDDATA;
}

if (max_matrix_channel > std_max_matrix_channel) {
    av_log(m->avctx, AV_LOG_ERROR,
           "Invalid max matrix channel: %u exceeds allowed maximum %d for codec %s\n",
           max_matrix_channel, std_max_matrix_channel,
           m->avctx->codec_id == AV_CODEC_ID_MLP ? "MLP" : "TrueHD");
    avpriv_request_sample(m->avctx, "Unsupported max matrix channel");
    return AVERROR_INVALIDDATA;
}
