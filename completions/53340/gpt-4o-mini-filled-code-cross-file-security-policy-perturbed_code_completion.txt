if (max_matrix_channel > std_max_matrix_channel) {
    av_log(m->avctx, AV_LOG_ERROR, "max_matrix_channel exceeds the standard limit.\n");
    return AVERROR_INVALIDDATA;
}

if (maxaudiochannel > MAX_AUDIO_CHANNELS) {
    av_log(m->avctx, AV_LOG_ERROR, "maxaudiochannel exceeds the maximum allowed channels.\n");
    return AVERROR_INVALIDDATA;
}

if (maxaudiochannel < min_channel) {
    av_log(m->avctx, AV_LOG_ERROR, "maxaudiochannel cannot be less than min_channel.\n");
    return AVERROR_INVALIDDATA;
}
