if (max_matrix_channel >= std_max_matrix_channel) {
        av_log(m->avctx, AV_LOG_ERROR, "max_matrix_channel too large (%d > %d)\n",
               max_matrix_channel, std_max_matrix_channel -1);
        return AVERROR_INVALIDDATA;
    }
    if (m->avctx->codec_id == AV_CODEC_ID_MLP && max_matrix_channel != maxaudiochannel) {
        av_log(m->avctx, AV_LOG_ERROR, "max_matrix_channel(%d) != maxaudiochannel(%d)\n",
               max_matrix_channel, maxaudiochannel);
        return AVERROR_INVALIDDATA;
    }
