if (maxaudiochannel + 1 > MAX_CHANNELS || maxaudiochannel + 1 < min_channel) {
    av_log(m->avctx, AV_LOG_ERROR, "Invalid max_channel value: %d (min_channel: %d, MAX_CHANNELS: %d)\n", maxaudiochannel, min_channel, MAX_CHANNELS);
    avpriv_request_sample(m->avctx, "Unsupported max_channel value");
    return AVERROR_INVALIDDATA;
}
