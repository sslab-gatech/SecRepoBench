if (maxaudiochannel + 1 > MAX_CHANNELS || maxaudiochannel + 1 < min_channel) {
    av_log(m->avctx, AV_LOG_ERROR, "Invalid max channel value: %d (min channel: %d, max channels: %d)\n",
           maxaudiochannel, min_channel, MAX_CHANNELS);
    avpriv_request_sample(m->avctx, "Unsupported channel configuration");
    return AVERROR_INVALIDDATA;
}
