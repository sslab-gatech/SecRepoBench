if (max_matrix_channel > std_max_matrix_channel) {
    av_log(m->avctx, AV_LOG_ERROR, "Invalid max_matrix_channel: %d (expected <= %d)\n", max_matrix_channel, std_max_matrix_channel);
    avpriv_request_sample(m->avctx, "max_matrix_channel > %d", std_max_matrix_channel);
    return AVERROR_INVALIDDATA;
}
