if (max_matrix_channel > std_max_matrix_channel || max_matrix_channel < min_channel ||
    maxaudiochannel > MAX_CHANNELS || maxaudiochannel < min_channel) {
    av_log(m->avctx, AV_LOG_ERROR, "Invalid max channel count.\n");
    if (m->avctx->err_recognition & AV_EF_CRCCHECK) {
        av_log(m->avctx, AV_LOG_ERROR, "Requesting sample for validation.\n");
        avpriv_request_sample(m->avctx, "Validation of max channel count.");
    }
    return AVERROR_INVALIDDATA;
}
