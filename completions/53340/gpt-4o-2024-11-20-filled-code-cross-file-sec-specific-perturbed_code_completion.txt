if (max_matrix_channel > std_max_matrix_channel) {
        av_log(m->avctx, AV_LOG_ERROR,
               "max_matrix_channel %d exceeds standard maximum %d.\n",
               max_matrix_channel, std_max_matrix_channel);
        return AVERROR_INVALIDDATA;
    }

    if (maxaudiochannel > MAX_CHANNELS - 1) {
        av_log(m->avctx, AV_LOG_ERROR,
               "maxaudiochannel %d exceeds maximum allowed channels %d.\n",
               maxaudiochannel, MAX_CHANNELS - 1);
        return AVERROR_INVALIDDATA;
    }

    if (min_channel > maxaudiochannel) {
        av_log(m->avctx, AV_LOG_ERROR,
               "min_channel %d is greater than maxaudiochannel %d.\n",
               min_channel, maxaudiochannel);
        return AVERROR_INVALIDDATA;
    }
