for (auto it = iterator; it; ++it)
    {
      hb_codepoint_t gid = (*it).first;
      unsigned klass = (*it).second;

      if (gid == prev_gid + 1 && klass == prev_klass)
      {
        // Continue current range
        record->last = gid;
      }
      else
      {
        // Finalize current range
        num_ranges++;

        // Start new range
        range_rec.first = gid;
        range_rec.last = gid;
        range_rec.value = klass;

        record = c->copy (range_rec);
        if (unlikely (!record)) return_trace (false);
      }

      prev_gid = gid;
      prev_klass = klass;
    }

    rangeRecord.len = num_ranges;

    // Sort ranges by first glyph ID
    hb_sort (&rangeRecord[0], num_ranges, RangeRecord<Types>::cmp);

    return_trace (true);
