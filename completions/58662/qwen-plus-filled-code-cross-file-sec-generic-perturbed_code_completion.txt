for (++iterator; *iterator; ++iterator)
{
  hb_codepoint_t gid = (*iterator).first;
  unsigned klass = (*iterator).second;

  if (klass == prev_klass && gid == range_rec.last + 1)
  {
    range_rec.last = gid;
    record->last = gid;
  }
  else
  {
    num_ranges++;
    range_rec.first = gid;
    range_rec.last = gid;
    range_rec.value = klass;

    RangeRecord<Types> new_range_rec = range_rec;
    auto *new_record = c->copy (new_range_rec);
    if (unlikely (!new_record)) return_trace (false);

    record = new_record;
    prev_gid = gid;
    prev_klass = klass;
  }
}

rangeRecord.len = num_ranges;
rangeRecord.sort ();

return_trace (true);
