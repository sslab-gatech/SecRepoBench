for (++iterator; iterator; ++iterator)
    {
      hb_codepoint_t gid = (*iterator).first;
      unsigned klass = (*iterator).second;

      if (gid == prev_gid + 1 && klass == prev_klass)
      {
        // Continuation of the current range
        range_rec.last = gid;
      }
      else
      {
        // Finalize the current range
        record->last = range_rec.last;
        record = c->copy (range_rec);
        if (unlikely (!record)) return_trace (false);

        // Start a new range
        range_rec.first = gid;
        range_rec.last = gid;
        range_rec.value = klass;
        num_ranges++;
      }

      prev_gid = gid;
      prev_klass = klass;
    }

    // Finalize the last range
    record->last = range_rec.last;

    rangeRecord.len = num_ranges;

    // Sort the range records if necessary
    c->sort_range_records();

    return_trace (true);
