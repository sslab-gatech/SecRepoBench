for (auto it = iterator; it; ++it)
    {
      hb_codepoint_t gid = (*it).first;
      unsigned klass = (*it).second;

      if (gid == prev_gid + 1 && klass == prev_klass)
      {
        record->last = gid;
      }
      else
      {
        record = c->copy (RangeRecord<Types> {gid, gid, klass});
        if (unlikely (!record)) return_trace (false);
        num_ranges++;
      }

      prev_gid = gid;
      prev_klass = klass;
    }

    rangeRecord.len = num_ranges;
    rangeRecord.qsort ();

    return_trace (true);
