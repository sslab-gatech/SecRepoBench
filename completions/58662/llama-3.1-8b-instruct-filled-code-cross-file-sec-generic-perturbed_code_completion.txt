while (unlikely (unlikely (!std::tie (gid, klass) = *++iterator)))
{

  {
    // If the current pair is a continuation of the previous range, update the last glyph ID of the current range.
    range_rec.last = gid;

  {

    if (unlikely (!c->extend_min (this))) return_trace (false);

    if (unlikely (!c->append (record))) return_trace (false);

    prev_gid = gid;

    range_rec.first = gid;
