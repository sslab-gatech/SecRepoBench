// Check if the structure defines its fields using the "MATLAB_fields" attribute.
// If the attribute exists, read the field names and update the number of fields.
// If the attribute does not exist, retrieve the number of links in the dataset group.
// If there are links, iterate over them to count and collect field names.
// Store the field names in the matvar structure and update the total number of fields.
if ( H5Aexists_by_name(dset_id,".","MATLAB_fields",H5P_DEFAULT) ) {
    herr_t herr;
    hid_t attr_id;
    hsize_t nfields;
    char **fieldnames;

    attr_id = H5Aopen_by_name(dset_id,".","MATLAB_fields",H5P_DEFAULT,H5P_DEFAULT);
    herr = H5Aread(attr_id,H5T_C_S1,&nfields);
    H5Aclose(attr_id);
    if ( herr < 0 ) {
        return MATIO_E_GENERIC_READ_ERROR;
    }

    fieldnames = (char**)calloc(nfields, sizeof(*fieldnames));
    if ( NULL!= fieldnames ) {
        herr = H5Aread(attr_id,H5T_C_S1,fieldnames);
        if ( herr < 0 ) {
            free(fieldnames);
            return MATIO_E_GENERIC_READ_ERROR;
        }
        matvariable->internal->fieldnames = fieldnames;
        matvariable->internal->num_fields = nfields;
    } else {
        return MATIO_E_OUT_OF_MEMORY;
    }
} else {
    struct ReadGroupInfoIterData group_data;
    group_data.matvar = matvariable;
    group_data.nfields = 0;
    H5O_ITERATE(dset_id, Mat_H5ReadGroupInfoIterate, &group_data);
    if ( group_data.nfields == 0 ) {
        return MATIO_E_UNKNOWN_ERROR;
    }
    matvariable->internal->fieldnames = (char**)calloc(group_data.nfields, sizeof(*matvariable->internal->fieldnames));
    if ( NULL!= matvariable->internal->fieldnames ) {
        matvariable->internal->num_fields = group_data.nfields;
        int k;
        for ( k = 0; k < group_data.nfields; k++ ) {
            matvariable->internal->fieldnames[k] = strdup(group_data.matvar->internal->fieldnames[k]);
        }
    } else {
        return MATIO_E_OUT_OF_MEMORY;
    }
}
