{
  if (!table_index)
    return false;

  unsigned int low = 0, high = tables.len - 1;
  while (low <= high)
  {
    unsigned int mid = (low + high) >> 1;
    int cmpval = tables[mid].cmp(hb_tag);
    if (cmpval == 0)
    {
      *table_index = mid;
      return true;
    }
    else if (cmpval < 0)
      low = mid + 1;
    else
      high = mid - 1;
  }
  return false;
}
