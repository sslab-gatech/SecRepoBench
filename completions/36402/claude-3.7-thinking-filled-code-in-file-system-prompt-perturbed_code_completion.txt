Tag t{hb_tag};
unsigned int left = 0;
unsigned int right = tables.len;
if (unlikely (!right))
  return false;
right--;

while (left <= right)
{
  unsigned int mid = (left + right) / 2;
  int c = tables.arrayZ[mid].cmp (t);
  if (c < 0)
    left = mid + 1;
  else if (c > 0)
    if (mid)
      right = mid - 1;
    else
      return false;
  else
  {
    *table_index = mid;
    return true;
  }
}
return false;
