if (tables.len == 0) {
  *table_index = UINT_MAX;
  return false;
}

// Binary search for the table entry with the matching tag
unsigned int low = 0, high = tables.len - 1;
while (low <= high) {
  unsigned int mid = low + (high - low) / 2;
  hb_tag_t mid_tag = tables[mid].tag;

  if (mid_tag == hb_tag) {
    *table_index = mid;
    return true;
  } else if (mid_tag < hb_tag) {
    low = mid + 1;
  } else {
    high = mid - 1;
  }
}

*table_index = UINT_MAX;
return false;
