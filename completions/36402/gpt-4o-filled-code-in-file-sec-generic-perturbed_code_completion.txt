    unsigned int low = 0, high = tables.len;
    while (low < high) {
      unsigned int mid = (low + high) / 2;
      int cmp = tables[mid].cmp(hb_tag);
      if (cmp < 0)
        low = mid + 1;
      else if (cmp > 0)
        high = mid;

    }
