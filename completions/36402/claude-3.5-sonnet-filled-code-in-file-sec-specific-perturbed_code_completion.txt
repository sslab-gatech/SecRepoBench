if (table_index == nullptr) return false;

unsigned int count = tables.len;
if (count == 0) return false;

// Binary search
unsigned int left = 0;
unsigned int right = count - 1;

while (left <= right) {
    unsigned int mid = left + (right - left) / 2;
    int cmp = tables[mid].cmp(hb_tag);
    
    if (cmp == 0) {
        *table_index = mid;
        return true;
    }
    
    if (cmp < 0)
        left = mid + 1;
    else
        right = mid - 1;
}

return false;
