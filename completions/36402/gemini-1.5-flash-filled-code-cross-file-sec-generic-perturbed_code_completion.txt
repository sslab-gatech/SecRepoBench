if (unlikely (tables.len == 0)) {
      *table_index = 0;
      return false;
    }
    unsigned int low = 0;
    unsigned int high = tables.len - 1;
    while (low <= high) {
      unsigned int mid = low + ((high - low) >> 1);
      hb_tag_t tag = tables.arrayZ[mid].tag;
      if (tag == hb_tag) {
	*table_index = mid;
	return true;
      } else if (tag < hb_tag) {
	low = mid + 1;
      } else {
	high = mid - 1;
      }
    }
    return false;
