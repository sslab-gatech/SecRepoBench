bool find_table_index (hb_tag_t hb_tag, unsigned int *table_index) const
{
  // Use binary search to find the index of the table entry with the given tag
  int left = 0;
  int right = tables.len - 1;
  while (left <= right)
  {
    int mid = left + (right - left) / 2;
    int cmp = tables[mid].cmp(hb_tag);
    if (cmp == 0)
    {
      *table_index = mid;
      return true;
    }
    else if (cmp < 0)
    {
      left = mid + 1;
    }
    else
    {
      right = mid - 1;
    }
  }
  return false;
}
