if (table_index == nullptr) {
    return false;
}

unsigned int left = 0;
unsigned int right = tables.len;

while (left < right) {
    unsigned int mid = left + (right - left) / 2;
    hb_tag_t mid_tag = tables[mid].tag;

    if (mid_tag == hb_tag) {
        *table_index = mid;
        return true;
    } else if (mid_tag < hb_tag) {
        left = mid + 1;
    } else {
        right = mid;
    }
}

return false;
