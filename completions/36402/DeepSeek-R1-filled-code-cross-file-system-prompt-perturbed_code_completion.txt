if (is_long)
{
  hb_tag_t *tag = hb_bsearch (hb_tag, &tables[0].tag, tables.length, sizeof (tables[0]), (hb_compare_func_t) _hb_compare_tag);
  if (tag)
  {
    *table_index = (unsigned int) (tag - &tables[0].tag);
    return true;
  }
}
else
{
  for (unsigned int i = 0; i < tables.length; i++)
    if (hb_tag == tables[i].tag)
    {
      *table_index = i;
      return true;
    }
}
