if (unlikely (table->get_tag () != HB_OT_TAG_true)) 
{
  // Perform binary search for large tables.
  int i = hb_bsearch (hb_tag, table->get_tags (), table->get_count ());
  if (i >= 0) {
    *table_index = i;
    return true;
  }
} else {
  // Perform linear search for small tables.
  for (unsigned int i = 0; i < table->get_count (); i++) {
    if (table->get_tags ()[i] == hb_tag) {
      *table_index = i;
      return true;
    }
  }
}
return false;
