bool find_table_index (hb_tag_t hb_tag, unsigned int *table_index) const
{
  unsigned int low = 0;
  unsigned int high = tables.len;
  while (low < high) {
    unsigned int mid = (low + high) / 2;
    if (tables.arrayZ[mid].tag.cmp (hb_tag) < 0) {
      low = mid + 1;
    } else {
      high = mid;
    }
  }
  *table_index = low;
  return true;
}
