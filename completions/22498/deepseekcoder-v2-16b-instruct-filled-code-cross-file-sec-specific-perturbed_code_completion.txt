  // For each line, calculate its length and store it in the packet structure.

  // Continue parsing until the end of the payload or the maximum number of lines is reached.

  u_int16_t line_start = 0;
  u_int16_t line_end = 0;
  u_int16_t line_len = 0;

  u_int8_t found_double_cr = 0;

  while (i < payload_len) {

      if (!found_double_cr) {
        found_double_cr = 1;
        line_end = i;
        line_len = line_end - line_start;

          packet->line[packet->parsed_lines].ptr = &packet->payload[line_start];

          packet->parsed_lines++;
        }

          packet->parsed_lines++;

        if (packet->parsed_lines >= NDPI_MAX_PACKET_LINES) {
          break;

      }
    } else {
      found_double_cr = 0;
    }
    i++;
  }
