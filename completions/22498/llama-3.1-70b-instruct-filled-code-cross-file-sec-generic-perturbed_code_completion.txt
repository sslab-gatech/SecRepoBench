u_int16_t line_len = 0;
u_int16_t line_start = 0;
u_int16_t line_end = 0;
u_int16_t i = 0;
u_int16_t http_status_code = 0;
u_int16_t http_status_code_len = 0;
u_int16_t http_status_code_pos = 0;
u_int16_t http_header_pos = 0;
u_int16_t http_header_len = 0;
u_int16_t http_header_count = 0;
u_int16_t empty_line_pos = 0;
u_int16_t empty_line_count = 0;

// Parse the payload of the packet to identify and extract lines separated by "\r\n" sequences.
for (i = 0; i < packet->payload_packet_len; i++) {
  if (packet->payload[i] == '\r' && i + 1 < packet->payload_packet_len && packet->payload[i + 1] == '\n') {
    line_end = i;
    line_len = line_end - line_start + 1;
    packet->line[packet->parsed_lines].len = line_len;
    packet->parsed_lines++;

    // If a sequence "\r\n\r\n" is encountered, extract subsequent bytes and store them in the data flow's initial binary bytes.
    if (packet->parsed_lines >= 2 && packet->line[packet->parsed_lines - 2].len == 2 && packet->line[packet->parsed_lines - 1].len == 2) {
      dataflow->initial_binary_bytes = &packet->payload[i + 2];
      dataflow->initial_binary_bytes_len = packet->payload_packet_len - (i + 2);
    }

    // Analyze the first line to detect and process HTTP response status codes.
    if (packet->parsed_lines == 1) {
      http_status_code_pos = ndpi_search_http_status_code(packet->payload, line_len);
      if (http_status_code_pos!= 0) {
        http_status_code_len = ndpi_get_http_status_code_len(packet->payload, line_len, http_status_code_pos);
        http_status_code = ndpi_get_http_status_code(packet->payload, line_len, http_status_code_pos, http_status_code_len);
      }
    }

    // Identify and extract specific HTTP headers such as "Server:", "Host:", "Content-Type:", etc., storing their positions and lengths.
    http_header_pos = ndpi_search_http_header(packet->payload, line_len, "Server:");
    if (http_header_pos!= 0) {
      http_header_len = ndpi_get_http_header_len(packet->payload, line_len, http_header_pos);
      packet->http_server_header_pos = http_header_pos;
      packet->http_server_header_len = http_header_len;
      http_header_count++;
    }

    http_header_pos = ndpi_search_http_header(packet->payload, line_len, "Host:");
    if (http_header_pos!= 0) {
      http_header_len = ndpi_get_http_header_len(packet->payload, line_len, http_header_pos);
      packet->http_host_header_pos = http_header_pos;
      packet->http_host_header_len = http_header_len;
      http_header_count++;
    }

    http_header_pos = ndpi_search_http_header(packet->payload, line_len, "Content-Type:");
    if (http_header_pos!= 0) {
      http_header_len = ndpi_get_http_header_len(packet->payload, line_len, http_header_pos);
      packet->http_content_type_header_pos = http_header_pos;
      packet->http_content_type_header_len = http_header_len;
      http_header_count++;
    }

    // Count the number of headers found in the packet.
    packet->http_header_count = http_header_count;

    // Mark the position of any empty lines, if encountered.
    if (line_len == 2) {
      empty_line_pos = i;
      empty_line_count++;
    }

    // Continue parsing until the end of the payload or the maximum number of lines is reached.
    if (packet->parsed_lines >= MAX_PACKET_LINES || i + 2 >= packet->payload_packet_len) {
      break;
    }

    line_start = i + 2;
  }
}
