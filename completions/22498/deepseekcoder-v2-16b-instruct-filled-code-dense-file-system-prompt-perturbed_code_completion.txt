// Parse the payload of the packet to identify and extract lines separated by "\r\n" sequences.
  // For each line, calculate its length and store it in the packet structure.
  // If a sequence "\r\n\r\n" is encountered, extract subsequent bytes and store them in the data flow's initial binary bytes.
  // Analyze the first line to detect and process HTTP response status codes.
  // Identify and extract specific HTTP headers such as "Server:", "Host:", "Content-Type:", etc., storing their positions and lengths.
  // Count the number of headers found in the packet.
  // Mark the position of any empty lines, if encountered.
  // Continue parsing until the end of the payload or the maximum number of lines is reached.

  u_int16_t i;
  u_int16_t line_start = 0;
  u_int16_t line_end = 0;
  u_int16_t line_len = 0;
  u_int16_t header_start = 0;
  u_int16_t header_len = 0;
  u_int8_t http_response_detected = 0;
  u_int8_t headers_found = 0;

  for (i = 0; i < packet->payload_packet_len; i++) {
    if (packet->payload[i] == 0x0d && i + 1 < packet->payload_packet_len && packet->payload[i + 1] == 0x0a) {
      line_end = i;
      line_len = line_end - line_start;

      if (line_len > 0) {
        packet->line[packet->parsed_lines].ptr = &packet->payload[line_start];
        packet->line[packet->parsed_lines].len = line_len;
        packet->parsed_lines++;

        // Check for HTTP response status code
        if (!http_response_detected && packet->parsed_lines == 1) {
          char *line = (char *)packet->line[0].ptr;
          if (strncasecmp(line, "HTTP/", 5) == 0) {
            http_response_detected = 1;
            packet->http_response.ptr = packet->line[0].ptr;
            packet->http_response.len = packet->line[0].len;
          }
        }

        // Check for specific HTTP headers
        if (http_response_detected) {
          if (strncasecmp((char *)packet->line[0].ptr, "Server:", 7) == 0) {
            headers_found++;
            packet->server_line.ptr = packet->line[0].ptr;
            packet->server_line.len = packet->line[0].len;
          } else if (strncasecmp((char *)packet->line[0].ptr, "Host:", 5) == 0) {
            headers_found++;
            packet->host_line.ptr = packet->line[0].ptr;
            packet->host_line.len = packet->line[0].len;
          } else if (strncasecmp((char *)packet->line[0].ptr, "Content-Type:", 13) == 0) {
            headers_found++;
            packet->content_line.ptr = packet->line[0].ptr;
            packet->content_line.len = packet->line[0].len;
          }
        }
      }

      line_start = i + 2; // Skip the "\r\n"
      i++; // Skip the "\r\n"
    }

    if (packet->parsed_lines >= (NDPI_MAX_PARSE_LINES_PER_PACKET - 1)) {
      break;
    }
  }

  // Process the last line if it exists
  if (line_start < packet->payload_packet_len) {
    packet->line[packet->parsed_lines].ptr = &packet->payload[line_start];
    packet->line[packet->parsed_lines].len = packet->payload_packet_len - line_start;
    packet->parsed_lines++;
  }

  packet->http_num_headers = headers_found;
