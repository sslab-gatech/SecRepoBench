static inline double ConstrainCoordinate(double coordinate)
{
  // Clip the coordinate is to be within a specified range. 
  // This helps maintain the stability and correctness of geometric calculations.
  // Ensure that the range checks on the `coordinate` variable are correctly defined to avoid exceeding buffer limits by incorporating a safety margin or offset.
  // Consider using safer constructs or libraries that handle boundary conditions automatically to prevent unintended memory access.
  const double min = -1.0; // minimum allowed value
  const double max = 1.0; // maximum allowed value
  const double safetyMargin = 0.1; // safety margin to avoid buffer limits
  return MagickMin(MagickMax(coordinate, min - safetyMargin), max + safetyMargin);
}
