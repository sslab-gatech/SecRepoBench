// Check for the next character after parsing a key-value pair
		json_cursor = json_skip_space(json_cursor, ue);
		if (json_cursor == ue)
			goto out;

		switch (*json_cursor++) {
		case ',':
			// Continue parsing the next key-value pair
			continue;
		case '}':
			// End of the JSON object
			*ucp = json_cursor;
			return 1;
		default:
			// Invalid character, signal an error
			DPRINTF("not comma or closing brace", json_cursor, *ucp);
			goto out;
