// <MASK>
if (*json_cursor++!= ',') {
    DPRINTF("not comma", json_cursor, *ucp);
    goto out;
}
json_cursor = json_skip_space(json_cursor, ue);
if (json_cursor == ue)
    goto out;
if (*json_cursor++!= '}') {
    DPRINTF("not closing brace", json_cursor, *ucp);
    goto out;
}
*ucp = json_cursor;
return 1;
