json_cursor = json_skip_space(json_cursor, ue);
if (json_cursor == ue || (*json_cursor != ',' && *json_cursor != '}')) {
	DPRINTF("expected comma or closing brace", json_cursor, *ucp);
	goto out;
}
if (*json_cursor == '}') {
	json_cursor++;
	goto out_success;
}
if (*json_cursor == ',') {
	json_cursor++;
	continue;
}
DPRINTF("invalid character after key-value pair", json_cursor, *ucp);
goto out;
out_success:
	*ucp = json_cursor;
	return 1;
