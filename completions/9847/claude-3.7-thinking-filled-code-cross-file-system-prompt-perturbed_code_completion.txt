json_cursor = json_skip_space(json_cursor, ue);
if (json_cursor == ue)
    goto out;
switch (*json_cursor) {
case ',':
    json_cursor++;
    continue; // Continue to parse another key-value pair
case '}':
    *ucp = json_cursor + 1; // Move past the closing brace
    return 1; // Successfully parsed object
default:
    DPRINTF("not comma or closing brace", json_cursor, *ucp);
    goto out; // Error
}
