json_cursor = json_skip_space(json_cursor, ue);
if (json_cursor == ue)
    goto out;

switch (*json_cursor++) {
case ',':
    /* Continue parsing the next key-value pair */
    break;
case '}':
    /* Object closed successfully */
    *ucp = json_cursor;
    return 1;
default:
    DPRINTF("not comma or brace", json_cursor, *ucp);
    goto out;
}
