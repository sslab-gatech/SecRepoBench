if (*json_cursor++ != ',') {
		if (*json_cursor == '}') {
			json_cursor++;
			return 1;
		}
		DPRINTF("not comma or closing brace", json_cursor, *ucp);
		goto out;
	}
	json_cursor = json_skip_space(json_cursor, ue);
	if (!json_parse(&json_cursor, ue, st, lvl + 1)) {
		DPRINTF("not json", json_cursor, *ucp);
		goto out;
	}
