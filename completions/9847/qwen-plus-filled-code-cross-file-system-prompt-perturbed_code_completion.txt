json_cursor = json_skip_space(json_cursor, ue);
if (json_cursor == ue || (*json_cursor != ',' && *json_cursor != '}')) {
	DPRINTF("Expected comma or closing brace in object", json_cursor, *ucp);
	goto out;
}
if (*json_cursor == '}') {
	json_cursor++;
	break;
}
if (*json_cursor++ != ',') {
	DPRINTF("not comma", json_cursor, *ucp);
	goto out;
}
