linesize = s->linesize;
    uvlinesize = s->uvlinesize;

    mx = motion_x;
    my = motion_y;
    sourcex = s->mb_x * block_s + (mx >> lowres);
    src_y = s->mb_y * block_s + (my >> lowres);

    if (s->quarter_sample) {
        mx /= 2;
        my /= 2;
    }

    sx = mx & s_mask;
    sy = my & s_mask;
    sourcex += mx >> lowres + 1;
    src_y += my >> lowres + 1;

    ptr_y  = ref_picture[0] + src_y * linesize + sourcex;
    ptr_cb = ref_picture[1] + (src_y >> s->chroma_y_shift) * uvlinesize + (sourcex >> s->chroma_x_shift);
    ptr_cr = ref_picture[2] + (src_y >> s->chroma_y_shift) * uvlinesize + (sourcex >> s->chroma_x_shift);

    if ((unsigned)sourcex > FFMAX(h_edge_pos - (!!sx) - 2 * block_s, 0) ||
        (unsigned)src_y > FFMAX(v_edge_pos - (!!sy) - 2 * block_s, 0)) {
        s->vdsp.emulated_edge_mc(s->sc.edge_emu_buffer, ptr_y,
                                 linesize, linesize,
                                 2 * block_s + 1, 2 * block_s + 1,
                                 sourcex, src_y, h_edge_pos, v_edge_pos);
        ptr_y = s->sc.edge_emu_buffer;
    }

    sx = (sx << 2) >> lowres;
    sy = (sy << 2) >> lowres;
    if (bottom_field)
        ptr_y += linesize;
    pix_op[op_index](dest_y, ptr_y, linesize, 2 * block_s, sx, sy);

    if (!CONFIG_GRAY || !(s->avctx->flags & AV_CODEC_FLAG_GRAY)) {
        uvsrc_x = sourcex >> s->chroma_x_shift;
        uvsrc_y = src_y >> s->chroma_y_shift;
        uvsx = sx >> s->chroma_x_shift;
        uvsy = sy >> s->chroma_y_shift;

        if ((unsigned)uvsrc_x > FFMAX(h_edge_pos - (!!uvsx) - block_s, 0) ||
            (unsigned)uvsrc_y > FFMAX(v_edge_pos - (!!uvsy) - block_s, 0)) {
            s->vdsp.emulated_edge_mc(s->sc.edge_emu_buffer, ptr_cb,
                                     uvlinesize, uvlinesize,
                                     block_s + 1, block_s + 1,
                                     uvsrc_x, uvsrc_y, h_edge_pos, v_edge_pos);
            ptr_cb = s->sc.edge_emu_buffer;
        }

        if ((unsigned)uvsrc_x > FFMAX(h_edge_pos - (!!uvsx) - block_s, 0) ||
            (unsigned)uvsrc_y > FFMAX(v_edge_pos - (!!uvsy) - block_s, 0)) {
            s->vdsp.emulated_edge_mc(s->sc.edge_emu_buffer, ptr_cr,
                                     uvlinesize, uvlinesize,
                                     block_s + 1, block_s + 1,
                                     uvsrc_x, uvsrc_y, h_edge_pos, v_edge_pos);
            ptr_cr = s->sc.edge_emu_buffer;
        }

        if (bottom_field) {
            ptr_cb += uvlinesize;
            ptr_cr += uvlinesize;
        }
        pix_op[op_index](dest_cb, ptr_cb, uvlinesize, block_s, uvsx, uvsy);
        pix_op[op_index](dest_cr, ptr_cr, uvlinesize, block_s, uvsx, uvsy);
    }
