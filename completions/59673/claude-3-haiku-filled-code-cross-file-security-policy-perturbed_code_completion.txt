int mx, my, sourcex, src_y, uvsrc_x, uvsrc_y, sx, sy, uvsx, uvsy;
    ptrdiff_t uvlinesize, linesize;

    mx = motion_x;
    my = motion_y;

    if (s->quarter_sample) {
        mx >>= 1;
        my >>= 1;
    }

    linesize   = s->linesize;
    uvlinesize = s->uvlinesize;

    sourcex = (2 * s->mb_x + (motion_x >> (lowres + 1))) * block_s;
    src_y   = (2 * mb_y + (motion_y >> (lowres + 1))) * block_s;

    uvsrc_x = (2 * s->mb_x + (mx >> (lowres + 2))) * (block_s >> s->chroma_x_shift);
    uvsrc_y = (2 * mb_y + (my >> (lowres + 2))) * (block_s >> s->chroma_y_shift);

    sx = (motion_x >> lowres) & s_mask;
    sy = (motion_y >> lowres) & s_mask;
    uvsx = (mx >> lowres) & s_mask;
    uvsy = (my >> lowres) & s_mask;

    ptr_y  = ref_picture[0] + sourcex + src_y * linesize;
    ptr_cb = ref_picture[1] + uvsrc_x + uvsrc_y * uvlinesize;
    ptr_cr = ref_picture[2] + uvsrc_x + uvsrc_y * uvlinesize;

    if ((unsigned)sourcex > FFMAX(h_edge_pos - (!!sx) - block_s, 0) ||
        (unsigned)src_y  > FFMAX(v_edge_pos - (!!sy) - block_s, 0)) {
        s->vdsp.emulated_edge_mc(s->sc.edge_emu_buffer, ptr_y,
                                 linesize, linesize,
                                 block_s, block_s,
                                 sourcex, src_y, h_edge_pos, v_edge_pos);
        ptr_y = s->sc.edge_emu_buffer;
        if ((unsigned)uvsrc_x > FFMAX(h_edge_pos / 2 - (!!uvsx) - (block_s >> s->chroma_x_shift), 0) ||
            (unsigned)uvsrc_y > FFMAX(v_edge_pos / 2 - (!!uvsy) - (block_s >> s->chroma_y_shift), 0)) {
            s->vdsp.emulated_edge_mc(s->sc.edge_emu_buffer + 16 * block_s,
                                     ptr_cb, uvlinesize, uvlinesize,
                                     block_s >> s->chroma_x_shift,
                                     block_s >> s->chroma_y_shift,
                                     uvsrc_x, uvsrc_y,
                                     h_edge_pos >> s->chroma_x_shift,
                                     v_edge_pos >> s->chroma_y_shift);
            ptr_cb = s->sc.edge_emu_buffer + 16 * block_s;
            s->vdsp.emulated_edge_mc(s->sc.edge_emu_buffer + 16 * block_s + 8 * block_s,
                                     ptr_cr, uvlinesize, uvlinesize,
                                     block_s >> s->chroma_x_shift,
                                     block_s >> s->chroma_y_shift,
                                     uvsrc_x, uvsrc_y,
                                     h_edge_pos >> s->chroma_x_shift,
                                     v_edge_pos >> s->chroma_y_shift);
            ptr_cr = s->sc.edge_emu_buffer + 16 * block_s + 8 * block_s;
        }
    }

    pix_op[op_index](dest_y, ptr_y, linesize, block_s, sx, sy);
    if (!CONFIG_GRAY || !(s->avctx->flags & AV_CODEC_FLAG_GRAY)) {
        pix_op[op_index](dest_cb, ptr_cb, uvlinesize, block_s >> s->chroma_x_shift, uvsx, uvsy);
        pix_op[op_index](dest_cr, ptr_cr, uvlinesize, block_s >> s->chroma_x_shift, uvsx, uvsy);
    }
