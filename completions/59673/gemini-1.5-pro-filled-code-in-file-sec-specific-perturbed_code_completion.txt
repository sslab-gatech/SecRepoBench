if (s->quarter_sample) {
        motion_x /= 2;
        motion_y /= 2;
    }

    mx = motion_x & s_mask;
    my = motion_y & s_mask;

    sourcex = mb_x * 2 * block_s + (motion_x >> lowres + 1);
    src_y = mb_y * 2 * block_s + (motion_y >> lowres + 1);

    linesize   = s->current_picture.f->linesize[0];

    ptr_y = ref_picture[0] + src_y * linesize + sourcex;

    if(bottom_field){
         ptr_y += s->linesize;
    }

    if ((unsigned)sourcex > FFMAX(h_edge_pos - (!!mx) - 2 * block_s, 0) ||
        (unsigned)src_y    > FFMAX((v_edge_pos >> field_based) - (!!my) - h, 0)) {
        s->vdsp.emulated_edge_mc(s->sc.edge_emu_buffer, ptr_y,
                                 linesize, linesize,
                                 2 * block_s + 1, (h + 1) << field_based,
                                 sourcex, src_y * (1 << field_based),
                                 h_edge_pos, v_edge_pos);
        ptr_y = s->sc.edge_emu_buffer;
    }

    mx = (mx << 2) >> lowres;
    my = (my << 2) >> lowres;

    if (field_select)
        ptr_y += linesize;

    pix_op[lowres - 1](dest_y, ptr_y, linesize, h, mx, my);


    if (!CONFIG_GRAY || !(s->avctx->flags & AV_CODEC_FLAG_GRAY)) {
        /* chroma MC */
        uvsrc_x = mb_x * block_s + (motion_x / (1 << (lowres + 1 + s->chroma_x_shift)));
        uvsrc_y = mb_y * block_s + (motion_y / (1 << (lowres + 1 + s->chroma_y_shift)));

        uvsx    = (motion_x / (1 << (lowres + 1 - s->chroma_x_shift))) & s_mask;
        uvsy    = (motion_y / (1 << (lowres + 1 - s->chroma_y_shift))) & s_mask;

        uvlinesize = s->current_picture.f->linesize[1];

        ptr_cb = ref_picture[1] + uvsrc_y * uvlinesize + uvsrc_x;
        ptr_cr = ref_picture[2] + uvsrc_y * uvlinesize + uvsrc_x;

        if(bottom_field){
            ptr_cb += s->uvlinesize;
            ptr_cr += s->uvlinesize;
        }

        if ((unsigned)uvsrc_x > FFMAX(h_edge_pos - (!!uvsx) - block_s, 0) ||
            (unsigned)uvsrc_y > FFMAX(v_edge_pos - (!!uvsy) - block_s, 0)) {
            s->vdsp.emulated_edge_mc(s->sc.edge_emu_buffer, ptr_cb,
                                     s->uvlinesize, s->uvlinesize,
                                     9, 9,
                                     uvsrc_x, uvsrc_y, h_edge_pos, v_edge_pos);
            ptr_cb = s->sc.edge_emu_buffer;

            s->vdsp.emulated_edge_mc(s->sc.edge_emu_buffer, ptr_cr,
                                     s->uvlinesize, s->uvlinesize,
                                     9, 9,
                                     uvsrc_x, uvsrc_y, h_edge_pos, v_edge_pos);
            ptr_cr = s->sc.edge_emu_buffer;
        }

        uvsx = (uvsx << 2) >> lowres;
        uvsy = (uvsy << 2) >> lowres;

        if (field_select) {
            ptr_cb += s->uvlinesize;
            ptr_cr += s->uvlinesize;
        }

        pix_op[op_index](dest_cb, ptr_cb, uvlinesize, block_s, uvsx, uvsy);
        pix_op[op_index](dest_cr, ptr_cr, uvlinesize, block_s, uvsx, uvsy);
    }
