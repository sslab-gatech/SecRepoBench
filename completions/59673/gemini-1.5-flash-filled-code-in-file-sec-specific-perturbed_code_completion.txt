linesize = s->linesize;
    uvlinesize = s->uvlinesize;

    ptr_y  = ref_picture[0] + (mb_y * 16 >> lowres) * linesize + (mb_x * 8 >> lowres);
    ptr_cb = ref_picture[1] + (mb_y * 8 >> (lowres + s->chroma_y_shift)) * uvlinesize + (mb_x * 4 >> (lowres + s->chroma_x_shift));
    ptr_cr = ref_picture[2] + (mb_y * 8 >> (lowres + s->chroma_y_shift)) * uvlinesize + (mb_x * 4 >> (lowres + s->chroma_x_shift));

    mx = motion_x & s_mask;
    my = motion_y & s_mask;
    sourcex = mb_x * block_s + (motion_x >> lowres + 1);
    src_y = mb_y * block_s + (motion_y >> lowres + 1);
    uvsrc_x = sourcex >> 1;
    uvsrc_y = src_y >> 1;

    if (bottom_field) {
        ptr_y += linesize;
        src_y++;
        uvsrc_y++;
    }

    if (field_select) {
        ptr_y += linesize;
        src_y++;
        uvsrc_y++;
    }

    if ((unsigned)sourcex > FFMAX(h_edge_pos - (!!mx) - 16 >> lowres, 0) ||
        (unsigned)src_y > FFMAX(v_edge_pos - (!!my) - 16 >> lowres, 0)) {
        s->vdsp.emulated_edge_mc(s->sc.edge_emu_buffer, ptr_y, linesize, linesize,
                                 17, 17, sourcex, src_y, h_edge_pos, v_edge_pos);
        ptr_y = s->sc.edge_emu_buffer;
    }

    if ((unsigned)uvsrc_x > FFMAX(h_edge_pos - (!!mx) - 8 >> lowres, 0) ||
        (unsigned)uvsrc_y > FFMAX(v_edge_pos - (!!my) - 8 >> lowres, 0)) {
        s->vdsp.emulated_edge_mc(s->sc.edge_emu_buffer, ptr_cb, uvlinesize, uvlinesize,
                                 9, 9, uvsrc_x, uvsrc_y, h_edge_pos, v_edge_pos);
        ptr_cb = s->sc.edge_emu_buffer;
        s->vdsp.emulated_edge_mc(s->sc.edge_emu_buffer, ptr_cr, uvlinesize, uvlinesize,
                                 9, 9, uvsrc_x, uvsrc_y, h_edge_pos, v_edge_pos);
        ptr_cr = s->sc.edge_emu_buffer;
    }

    sx = (mx << 2) >> lowres;
    sy = (my << 2) >> lowres;
    uvsx = (mx << 1) >> (lowres + 1);
    uvsy = (my << 1) >> (lowres + 1);

    pix_op[op_index](dest_y, ptr_y, linesize, h, sx, sy);
    pix_op[op_index](dest_cb, ptr_cb, uvlinesize, h >> 1, uvsx, uvsy);
    pix_op[op_index](dest_cr, ptr_cr, uvlinesize, h >> 1, uvsx, uvsy);
