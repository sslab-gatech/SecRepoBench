linesize   = s->current_picture.f->linesize[0] << field_based;
uvlinesize = s->current_picture.f->linesize[1] << field_based;

mx = motion_x >> (lowres + 1);
my = motion_y >> (lowres + 1);
sx = motion_x & s_mask;
sy = motion_y & s_mask;

src_y = (mb_y * 2 + 1 - field_based - bottom_field) << (3 - lowres);
uvsrc_y = src_y >> 1;

if (field_select)
    src_y += s->current_picture.f->linesize[0] >> lowres;

src_y += my * linesize;
uvsrc_y += (my >> 1) * uvlinesize;

ptr_y  = ref_picture[0] + src_y;
ptr_cb = ref_picture[1] + uvsrc_y;
ptr_cr = ref_picture[2] + uvsrc_y;

if ((unsigned) mx >= FFMAX(h_edge_pos - (!!sx) - block_s * 2, 0) ||
    (unsigned) my >= FFMAX(v_edge_pos - (!!sy) - h, 0)) {
    s->vdsp.emulated_edge_mc(s->sc.edge_emu_buffer, ptr_y,
                             linesize, linesize,
                             17, 17 + field_based,
                             mx, my, h_edge_pos, v_edge_pos);
    ptr_y = s->sc.edge_emu_buffer;
}
sx = (sx << 2) >> lowres;
sy = (sy << 2) >> lowres;
pix_op[lowres - 1](dest_y, ptr_y, linesize, h, sx, sy);

if (!CONFIG_GRAY || !(s->avctx->flags & AV_CODEC_FLAG_GRAY)) {
    mx >>= 1;
    my >>= 1;
    sx = mx & 7;
    sy = my & 7;
    sourcex = (mb_x * 2 + 1 - field_based) << (2 - lowres);
    sourcey = (mb_y * 2 + 1 - field_based) << (2 - lowres);
    uvsrc_x = sourcex >> 1;
    uvsrc_y = sourcey >> 1;
    uvsrc_x += mx >> 2;
    uvsrc_y += my >> 2;
    ptr_cb += uvsrc_x + uvsrc_y * uvlinesize;
    ptr_cr += uvsrc_x + uvsrc_y * uvlinesize;
    if ((unsigned) uvsrc_x >= FFMAX(h_edge_pos - (!!sx) - block_s, 0) ||
        (unsigned) uvsrc_y >= FFMAX(v_edge_pos - (!!sy) - h, 0)) {
        s->vdsp.emulated_edge_mc(s->sc.edge_emu_buffer, ptr_cb,
                                 uvlinesize, uvlinesize,
                                 9, 9 + field_based,
                                 uvsrc_x, uvsrc_y, h_edge_pos, v_edge_pos);
        ptr_cb = s->sc.edge_emu_buffer;
        s->vdsp.emulated_edge_mc(s->sc.edge_emu_buffer + 16, ptr_cr,
                                 uvlinesize, uvlinesize,
                                 9, 9 + field_based,
                                 uvsrc_x, uvsrc_y, h_edge_pos, v_edge_pos);
        ptr_cr = s->sc.edge_emu_buffer + 16;
    }
    pix_op[op_index](dest_cb, ptr_cb, uvlinesize, h >> 1, sx, sy);
    pix_op[op_index](dest_cr, ptr_cr, uvlinesize, h >> 1, sx, sy);
}
