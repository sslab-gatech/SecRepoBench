linesize   = s->linesize;
uvlinesize = s->uvlinesize;

// Adjust motion vector for lowres
if (s->quarter_sample) {
    motion_x >>= 1;
    motion_y >>= 1;
}

sourcex = s->mb_x * 2 * block_s;
src_y   = (mb_y * 2 * block_s) >> field_based;

if (field_based) {
    motion_y += bottom_field;
    src_y    += bottom_field;
}

sourcex += motion_x >> lowres;
src_y   += motion_y >> lowres;

sx = motion_x & s_mask;
sy = motion_y & s_mask;

if (!CONFIG_GRAY || !(s->avctx->flags & AV_CODEC_FLAG_GRAY)) {
    uvsrc_x = sourcex >> s->chroma_x_shift;
    uvsrc_y = src_y   >> s->chroma_y_shift;
    uvsx = (sx << (2 - s->chroma_x_shift)) >> lowres;
    uvsy = (sy << (2 - s->chroma_y_shift)) >> lowres;
}

ptr_y = ref_picture[0] + src_y * linesize + sourcex;
if ((unsigned)sourcex > FFMAX(h_edge_pos - (!!sx) - 2 * block_s, 0) ||
    (unsigned)src_y   > FFMAX(v_edge_pos - (!!sy) - h, 0)) {
    s->vdsp.emulated_edge_mc(s->sc.edge_emu_buffer, ptr_y, linesize, linesize,
                            2 * block_s + 1, h + 1,
                            sourcex, src_y, h_edge_pos, v_edge_pos);
    ptr_y = s->sc.edge_emu_buffer;
}

if (field_select)
    ptr_y += linesize;

sx = (sx << 2) >> lowres;
sy = (sy << 2) >> lowres;
s->hdsp.put_pixels_tab[lowres][((sy & 2) << 1) | (sx & 2)](dest_y, ptr_y, linesize, h);

if (!CONFIG_GRAY || !(s->avctx->flags & AV_CODEC_FLAG_GRAY)) {
    ptrdiff_t offset = uvsrc_y * uvlinesize + uvsrc_x;
    
    ptr_cb = ref_picture[1] + offset;
    if ((unsigned)uvsrc_x > FFMAX(h_edge_pos >> s->chroma_x_shift, 0) - (!!uvsx) - block_s ||
        (unsigned)uvsrc_y > FFMAX(v_edge_pos >> s->chroma_y_shift, 0) - (!!uvsy) - h) {
        s->vdsp.emulated_edge_mc(s->sc.edge_emu_buffer, ptr_cb, uvlinesize, uvlinesize,
                                block_s + 1, h + 1,
                                uvsrc_x, uvsrc_y,
                                h_edge_pos >> s->chroma_x_shift,
                                v_edge_pos >> s->chroma_y_shift);
        ptr_cb = s->sc.edge_emu_buffer;
    }
    if (field_select)
        ptr_cb += uvlinesize;
    pix_op[op_index](dest_cb, ptr_cb, uvlinesize, h, uvsx, uvsy);
    
    ptr_cr = ref_picture[2] + offset;
    if ((unsigned)uvsrc_x > FFMAX(h_edge_pos >> s->chroma_x_shift, 0) - (!!uvsx) - block_s ||
        (unsigned)uvsrc_y > FFMAX(v_edge_pos >> s->chroma_y_shift, 0) - (!!uvsy) - h) {
        s->vdsp.emulated_edge_mc(s->sc.edge_emu_buffer, ptr_cr, uvlinesize, uvlinesize,
                                block_s + 1, h + 1,
                                uvsrc_x, uvsrc_y,
                                h_edge_pos >> s->chroma_x_shift,
                                v_edge_pos >> s->chroma_y_shift);
        ptr_cr = s->sc.edge_emu_buffer;
    }
    if (field_select)
        ptr_cr += uvlinesize;
    pix_op[op_index](dest_cr, ptr_cr, uvlinesize, h, uvsx, uvsy);
}
