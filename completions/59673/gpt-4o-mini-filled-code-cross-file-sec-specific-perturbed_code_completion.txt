mx += motion_x;
    my += motion_y;

    sourcex = s->mb_x * block_s + (mx >> lowres);
    src_y = mb_y * block_s + (my >> lowres);
    
    if ((unsigned)sourcex >= h_edge_pos || (unsigned)src_y >= v_edge_pos) {
        s->vdsp.emulated_edge_mc(s->sc.edge_emu_buffer, ref_picture[0] + src_y * s->linesize + sourcex,
                                 s->linesize, s->linesize,
                                 9, 9,
                                 sourcex, src_y, h_edge_pos, v_edge_pos);
        ptr_y = s->sc.edge_emu_buffer;
    } else {
        ptr_y = ref_picture[0] + src_y * s->linesize + sourcex;
    }

    // Perform motion compensation for luma
    pix_op[op_index](dest_y, ptr_y, s->linesize, block_s, mx & s_mask, my & s_mask);

    // Handle chroma components
    uvsrc_x = (sourcex >> s->chroma_x_shift);
    uvsrc_y = (src_y >> s->chroma_y_shift);
    
    if ((unsigned)uvsrc_x >= h_edge_pos || (unsigned)uvsrc_y >= v_edge_pos) {
        s->vdsp.emulated_edge_mc(s->sc.edge_emu_buffer, ref_picture[1] + uvsrc_y * s->uvlinesize + uvsrc_x,
                                 s->uvlinesize, s->uvlinesize,
                                 9, 9,
                                 uvsrc_x, uvsrc_y, h_edge_pos, v_edge_pos);
        ptr_cb = s->sc.edge_emu_buffer;
        ptr_cr = s->sc.edge_emu_buffer;
    } else {
        ptr_cb = ref_picture[1] + uvsrc_y * s->uvlinesize + uvsrc_x;
        ptr_cr = ref_picture[2] + uvsrc_y * s->uvlinesize + uvsrc_x;
    }

    pix_op[op_index](dest_cb, ptr_cb, s->uvlinesize, block_s, mx & s_mask, my & s_mask);
    pix_op[op_index](dest_cr, ptr_cr, s->uvlinesize, block_s, mx & s_mask, my & s_mask);
