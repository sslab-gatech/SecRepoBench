mx = motion_x;
    my = motion_y;

    sx = mx & s_mask;
    sy = my & s_mask;
    sourcex = s->mb_x * block_s + (mx >> lowres);
    src_y   = s->mb_y * block_s + (my >> lowres);

    if (field_based) {
        src_y += bottom_field;
        src_y &= ~1;
    }

    uvsrc_x = s->mb_x * block_s + (mx >> (lowres + s->chroma_x_shift));
    uvsrc_y = s->mb_y * block_s + (my >> (lowres + s->chroma_y_shift));

    if (field_based) {
        uvsrc_y += bottom_field;
        uvsrc_y &= ~1;
    }

    linesize = s->linesize;
    uvlinesize = s->uvlinesize;

    ptr_y = ref_picture[0] + src_y * linesize + sourcex;

    if ((unsigned) sourcex > FFMAX(h_edge_pos - (!!sx) - block_s, 0) ||
        (unsigned) src_y   > FFMAX(v_edge_pos - (!!sy) - h, 0)) {
        s->vdsp.emulated_edge_mc(s->sc.edge_emu_buffer, ptr_y,
                                 linesize, linesize,
                                 9, h + 1,
                                 sourcex, src_y, h_edge_pos, v_edge_pos);
        ptr_y = s->sc.edge_emu_buffer;
    }

    if (!CONFIG_GRAY || !(s->avctx->flags & AV_CODEC_FLAG_GRAY)) {
        int op_index_uv = FFMIN(lowres - 1, 3);

        ptr_cb = ref_picture[1] + uvsrc_y * uvlinesize + uvsrc_x;
        ptr_cr = ref_picture[2] + uvsrc_y * uvlinesize + uvsrc_x;

        uvsx = (sx << 2) >> (lowres + s->chroma_x_shift);
        uvsy = (sy << 2) >> (lowres + s->chroma_y_shift);

        if ((unsigned) uvsrc_x > FFMAX((h_edge_pos >> s->chroma_x_shift) - (!!uvsx) - block_s, 0) ||
            (unsigned) uvsrc_y > FFMAX((v_edge_pos >> s->chroma_y_shift) - (!!uvsy) - block_s, 0)) {
            s->vdsp.emulated_edge_mc(s->sc.edge_emu_buffer, ptr_cb,
                                     uvlinesize, uvlinesize,
                                     9, 9,
                                     uvsrc_x, uvsrc_y, h_edge_pos >> s->chroma_x_shift, v_edge_pos >> s->chroma_y_shift);
            ptr_cb = s->sc.edge_emu_buffer;
            s->vdsp.emulated_edge_mc(s->sc.edge_emu_buffer, ptr_cr,
                                     uvlinesize, uvlinesize,
                                     9, 9,
                                     uvsrc_x, uvsrc_y, h_edge_pos >> s->chroma_x_shift, v_edge_pos >> s->chroma_y_shift);
            ptr_cr = s->sc.edge_emu_buffer;
        }

        if (s->quarter_sample) {
            uvsx = (mx & 3) << (2 - lowres - s->chroma_x_shift);
            uvsy = (my & 3) << (2 - lowres - s->chroma_y_shift);
        } else {
            uvsx = sx << (2 - lowres - s->chroma_x_shift);
            uvsy = sy << (2 - lowres - s->chroma_y_shift);
        }

        pix_op[op_index](dest_cb, ptr_cb, uvlinesize, block_s, uvsx, uvsy);
        pix_op[op_index](dest_cr, ptr_cr, uvlinesize, block_s, uvsx, uvsy);
    }

    sx <<= 2 - lowres;
    sy <<= 2 - lowres;

    if (s->quarter_sample) {
        sx = mx & 3;
        sy = my & 3;
    }

    s->dsp.put_pixels_tab[0](dest_y, ptr_y, linesize, sx, sy, h);
