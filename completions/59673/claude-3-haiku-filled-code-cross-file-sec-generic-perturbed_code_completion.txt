// <MASK>
int mx, my, sourcex, src_y, uvsrc_x, uvsrc_y, sx, sy, uvsx, uvsy;
ptrdiff_t uvlinesize, linesize;

// Calculate luma source positions
mx = motion_x;
my = motion_y;
sourcex = s->mb_x * block_s + (mx >> lowres + 1);
src_y = s->mb_y * block_s + (my >> lowres + 1);
sx = mx & s_mask;
sy = my & s_mask;

// Perform luma motion compensation
ptr_y = ref_picture[0] + src_y * s->linesize + sourcex;
if ((unsigned)sourcex > FFMAX(h_edge_pos - (!!sx) - block_s, 0) ||
    (unsigned)src_y > FFMAX(v_edge_pos - (!!sy) - block_s, 0)) {
    s->vdsp.emulated_edge_mc(s->sc.edge_emu_buffer, ptr_y,
                             s->linesize, s->linesize,
                             block_s, block_s,
                             sourcex, src_y, h_edge_pos, v_edge_pos);
    ptr_y = s->sc.edge_emu_buffer;
}
pix_op[op_index](dest_y, ptr_y, s->linesize, block_s, sx, sy);

// Calculate chroma source positions
uvsrc_x = s->mb_x * block_s + (mx >> lowres + s->chroma_x_shift + 1);
uvsrc_y = s->mb_y * block_s + (my >> lowres + s->chroma_y_shift + 1);
uvsx = mx & s_mask;
uvsy = my & s_mask;

// Perform chroma motion compensation
ptr_cb = ref_picture[1] + uvsrc_y * s->uvlinesize + uvsrc_x;
ptr_cr = ref_picture[2] + uvsrc_y * s->uvlinesize + uvsrc_x;
if ((unsigned)uvsrc_x > FFMAX(h_edge_pos - (!!uvsx) - block_s, 0) ||
    (unsigned)uvsrc_y > FFMAX(v_edge_pos - (!!uvsy) - block_s, 0)) {
    s->vdsp.emulated_edge_mc(s->sc.edge_emu_buffer, ptr_cb,
                             s->uvlinesize, s->uvlinesize,
                             block_s, block_s,
                             uvsrc_x, uvsrc_y, h_edge_pos, v_edge_pos);
    ptr_cb = s->sc.edge_emu_buffer;
    s->vdsp.emulated_edge_mc(s->sc.edge_emu_buffer, ptr_cr,
                             s->uvlinesize, s->uvlinesize,
                             block_s, block_s,
                             uvsrc_x, uvsrc_y, h_edge_pos, v_edge_pos);
    ptr_cr = s->sc.edge_emu_buffer;
}
pix_op[op_index](dest_cb, ptr_cb, s->uvlinesize, block_s, uvsx, uvsy);
pix_op[op_index](dest_cr, ptr_cr, s->uvlinesize, block_s, uvsx, uvsy);
