linesize = s->linesize;
    uvlinesize = s->uvlinesize;

    if (s->quarter_sample) {
        motion_x /= 2;
        motion_y /= 2;
    }

    mx = motion_x;
    my = motion_y;

    sx = mx & s_mask;
    sy = my & s_mask;
    sourcex = s->mb_x * block_s + (mx >> (lowres + 1));
    src_y = s->mb_y * block_s + (my >> (lowres + 1));

    ptr_y = ref_picture[0] + src_y * linesize + sourcex;
    if ((unsigned)sourcex > FFMAX(h_edge_pos - (!!sx) - block_s, 0) ||
        (unsigned)src_y > FFMAX(v_edge_pos - (!!sy) - h, 0)) {
        s->vdsp.emulated_edge_mc(s->sc.edge_emu_buffer, ptr_y, linesize, linesize,
                                 block_s + 1, h + 1, sourcex, src_y,
                                 h_edge_pos, v_edge_pos);
        ptr_y = s->sc.edge_emu_buffer;
        emu = 1;
    }
    sx = (sx << 2) >> lowres;
    sy = (sy << 2) >> lowres;
    pix_op[op_index](dest_y, ptr_y, linesize, h, sx, sy);

    if (!CONFIG_GRAY || !(s->avctx->flags & AV_CODEC_FLAG_GRAY)) {
        uvsx = ff_h263_round_chroma(mx);
        uvsy = ff_h263_round_chroma(my);
        uvsrc_x = s->mb_x * block_s + (uvsx >> (lowres + 1));
        uvsrc_y = s->mb_y * block_s + (uvsy >> (lowres + 1));

        ptr_cb = ref_picture[1] + uvsrc_y * uvlinesize + uvsrc_x;
        ptr_cr = ref_picture[2] + uvsrc_y * uvlinesize + uvsrc_x;

        if ((unsigned)uvsrc_x > FFMAX((h_edge_pos >> 1) - (!!uvsx) - (block_s >> 1), 0) ||
            (unsigned)uvsrc_y > FFMAX((v_edge_pos >> 1) - (!!uvsy) - (h >> 1), 0)) {
            s->vdsp.emulated_edge_mc(s->sc.edge_emu_buffer, ptr_cb, uvlinesize, uvlinesize,
                                     (block_s >> 1) + 1, (h >> 1) + 1, uvsrc_x, uvsrc_y,
                                     h_edge_pos >> 1, v_edge_pos >> 1);
            ptr_cb = s->sc.edge_emu_buffer;
            s->vdsp.emulated_edge_mc(s->sc.edge_emu_buffer, ptr_cr, uvlinesize, uvlinesize,
                                     (block_s >> 1) + 1, (h >> 1) + 1, uvsrc_x, uvsrc_y,
                                     h_edge_pos >> 1, v_edge_pos >> 1);
            ptr_cr = s->sc.edge_emu_buffer;
        }
        uvsx = (uvsx << 2) >> lowres;
        uvsy = (uvsy << 2) >> lowres;
        pix_op[op_index](dest_cb, ptr_cb, uvlinesize, h >> 1, uvsx, uvsy);
        pix_op[op_index](dest_cr, ptr_cr, uvlinesize, h >> 1, uvsx, uvsy);
    }
