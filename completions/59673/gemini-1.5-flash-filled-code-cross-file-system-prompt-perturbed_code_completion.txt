linesize  = s->linesize;
    uvlinesize = s->uvlinesize;
    mx = motion_x;
    my = motion_y;
    sx = mx & s_mask;
    sy = my & s_mask;
    sourcex = s->mb_x * block_s + (mx >> lowres);
    src_y   = mb_y * block_s + (my >> lowres);
    uvsrc_x = sourcex >> s->chroma_x_shift;
    uvsrc_y = src_y   >> s->chroma_y_shift;
    ptr_y   = ref_picture[0] + src_y * linesize + sourcex;
    ptr_cb  = ref_picture[1] + uvsrc_y * uvlinesize + uvsrc_x;
    ptr_cr  = ref_picture[2] + uvsrc_y * uvlinesize + uvsrc_x;
    if (field_based) {
        if (bottom_field)
            src_y += h;
        if (field_select) {
            ptr_y += h * linesize / 2;
            ptr_cb += h * uvlinesize / 2 >> s->chroma_y_shift;
            ptr_cr += h * uvlinesize / 2 >> s->chroma_y_shift;
        }
    }
    if ((unsigned)sourcex > h_edge_pos - (!!sx) - h ||
        (unsigned)src_y > v_edge_pos - (!!sy) - h) {
        s->vdsp.emulated_edge_mc(s->sc.edge_emu_buffer, ptr_y,
                                 linesize, linesize, h, h,
                                 sourcex, src_y, h_edge_pos, v_edge_pos);
        ptr_y = s->sc.edge_emu_buffer;
    }
    s->dsp.put_pixels_clamped(dest_y, ptr_y, linesize, h, sx, sy);
    if (!CONFIG_GRAY || !(s->avctx->flags & AV_CODEC_FLAG_GRAY)) {
        if ((unsigned)uvsrc_x > (s->h_edge_pos >> (lowres + s->chroma_x_shift)) - (!!sx) - (h >> s->chroma_x_shift) ||
            (unsigned)uvsrc_y > (s->v_edge_pos >> (lowres + s->chroma_y_shift)) - (!!sy) - (h >> s->chroma_y_shift)) {
            s->vdsp.emulated_edge_mc(s->sc.edge_emu_buffer, ptr_cb,
                                     uvlinesize, uvlinesize, h >> s->chroma_x_shift, h >> s->chroma_y_shift,
                                     uvsrc_x, uvsrc_y, s->h_edge_pos >> s->chroma_x_shift, s->v_edge_pos >> s->chroma_y_shift);
            ptr_cb = s->sc.edge_emu_buffer;
            s->vdsp.emulated_edge_mc(s->sc.edge_emu_buffer, ptr_cr,
                                     uvlinesize, uvlinesize, h >> s->chroma_x_shift, h >> s->chroma_y_shift,
                                     uvsrc_x, uvsrc_y, s->h_edge_pos >> s->chroma_x_shift, s->v_edge_pos >> s->chroma_y_shift);
            ptr_cr = s->sc.edge_emu_buffer;
        }
        pix_op[op_index](dest_cb, ptr_cb, uvlinesize, h >> s->chroma_x_shift, sx >> s->chroma_x_shift, sy >> s->chroma_y_shift);
        pix_op[op_index](dest_cr, ptr_cr, uvlinesize, h >> s->chroma_x_shift, sx >> s->chroma_x_shift, sy >> s->chroma_y_shift);
    }
