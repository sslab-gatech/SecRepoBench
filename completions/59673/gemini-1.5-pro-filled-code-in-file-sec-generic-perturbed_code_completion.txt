linesize = s->linesize << field_based;
    uvlinesize = s->uvlinesize << field_based;

    if (s->quarter_sample) {
        motion_x /= 2;
        motion_y /= 2;
    }

    sx = motion_x & s_mask;
    sy = motion_y & s_mask;
    sourcex = s->mb_x * 2 * block_s + (motion_x >> lowres + 1);
    src_y = mb_y * 2 * block_s + (motion_y >> lowres + 1);

    ptr_y  = ref_picture[0] + src_y * s->linesize + sourcex;

    if ((unsigned)sourcex > FFMAX(h_edge_pos - (!!sx) - 2 * block_s, 0) ||
        (unsigned)src_y    > FFMAX(v_edge_pos - (!!sy) - h, 0)) {
        s->vdsp.emulated_edge_mc(s->sc.edge_emu_buffer, ptr_y,
                                 s->linesize, s->linesize,
                                 2 * block_s + 1, (h + 1) << field_based,
                                 sourcex, src_y * (1 << field_based),
                                 h_edge_pos, v_edge_pos);
        ptr_y = s->sc.edge_emu_buffer;
    }

    sx = (sx << 2) >> lowres;
    sy = (sy << 2) >> lowres;
    if (bottom_field)
        ptr_y += s->linesize;
    if (field_select)
        ptr_y += s->linesize;
    pix_op[lowres](dest_y, ptr_y, linesize, h, sx, sy);


    if (!CONFIG_GRAY || !(s->avctx->flags & AV_CODEC_FLAG_GRAY)) {
        if (s->chroma_y_shift) {
            mx = motion_x / 2;
            my = motion_y / 2;
        } else if (s->quarter_sample) {
            mx = motion_x / 2;
            my = motion_y;
        } else {
            mx = motion_x;
            my = motion_y;
        }

        uvsx = (mx & s_mask) << 2;
        uvsy = (my & s_mask) << 2;

        uvsrc_x = s->mb_x * block_s + (mx >> lowres + 1);
        uvsrc_y = s->mb_y * block_s + (my >> lowres + 1);

        ptr_cb = ref_picture[1] + uvsrc_y * s->uvlinesize + uvsrc_x;
        ptr_cr = ref_picture[2] + uvsrc_y * s->uvlinesize + uvsrc_x;

        if ((unsigned)uvsrc_x > FFMAX(h_edge_pos - (!!(uvsx >> lowres)) - block_s, 0) ||
            (unsigned)uvsrc_y > FFMAX(v_edge_pos - (!!(uvsy >> lowres)) - (h >> s->chroma_y_shift), 0)) {
            s->vdsp.emulated_edge_mc(s->sc.edge_emu_buffer, ptr_cb,
                                     s->uvlinesize, s->uvlinesize,
                                     block_s + 1, (h >> s->chroma_y_shift) + 1,
                                     uvsrc_x, uvsrc_y,
                                     h_edge_pos, v_edge_pos);
            ptr_cb = s->sc.edge_emu_buffer;

            s->vdsp.emulated_edge_mc(s->sc.edge_emu_buffer, ptr_cr,
                                     s->uvlinesize, s->uvlinesize,
                                     block_s + 1, (h >> s->chroma_y_shift) + 1,
                                     uvsrc_x, uvsrc_y,
                                     h_edge_pos, v_edge_pos);
            ptr_cr = s->sc.edge_emu_buffer;
        }

        uvsx >>= lowres;
        uvsy >>= lowres;
        if (bottom_field) {
            ptr_cb += s->uvlinesize;
            ptr_cr += s->uvlinesize;
        }

        if (field_select) {
            ptr_cb += s->uvlinesize;
            ptr_cr += s->uvlinesize;
        }

        pix_op[op_index](dest_cb, ptr_cb, uvlinesize, h >> s->chroma_y_shift, uvsx, uvsy);
        pix_op[op_index](dest_cr, ptr_cr, uvlinesize, h >> s->chroma_y_shift, uvsx, uvsy);
    }
