linesize = s->linesize;
    uvlinesize = s->uvlinesize;

    if (s->quarter_sample) {
        motion_x /= 2;
        motion_y /= 2;
    }

    mx = motion_x;
    my = motion_y;

    sourcex = s->mb_x * 16 + (mx >> (1 + lowres));
    src_y = (mb_y * 16 >> field_based) + (my >> (1 + lowres));

    sx = mx & s_mask;
    sy = my & s_mask;

    ptr_y = ref_picture[0] + (src_y * linesize) + sourcex;
    ptr_cb = ref_picture[1] + ((src_y >> s->chroma_y_shift) * uvlinesize) + (sourcex >> s->chroma_x_shift);
    ptr_cr = ref_picture[2] + ((src_y >> s->chroma_y_shift) * uvlinesize) + (sourcex >> s->chroma_x_shift);

    if ((unsigned)sourcex > FFMAX(h_edge_pos - (!!sx) - w, 0) ||
        (unsigned)src_y > FFMAX((v_edge_pos >> field_based) - (!!sy) - h, 0)) {
        s->vdsp.emulated_edge_mc(s->sc.edge_emu_buffer, ptr_y, linesize, linesize,
                                 w + 1, (h + 1) << field_based,
                                 sourcex, src_y * (1 << field_based),
                                 h_edge_pos, v_edge_pos);
        ptr_y = s->sc.edge_emu_buffer;
        emu = 1;
    }

    sx = (sx << 2) >> lowres;
    sy = (sy << 2) >> lowres;
    if (field_select)
        ptr_y += linesize;
    pix_op[op_index](dest_y, ptr_y, linesize, h, sx, sy);

    if (!CONFIG_GRAY || !(s->avctx->flags & AV_CODEC_FLAG_GRAY)) {
        uvsx = (mx >> s->chroma_x_shift) & s_mask;
        uvsy = (my >> s->chroma_y_shift) & s_mask;

        if ((unsigned)(sourcex >> s->chroma_x_shift) > FFMAX((h_edge_pos >> s->chroma_x_shift) - (!!uvsx) - (w >> s->chroma_x_shift), 0) ||
            (unsigned)(src_y >> s->chroma_y_shift) > FFMAX((v_edge_pos >> (s->chroma_y_shift + field_based)) - (!!uvsy) - (h >> s->chroma_y_shift), 0)) {
            s->vdsp.emulated_edge_mc(s->sc.edge_emu_buffer, ptr_cb, uvlinesize, uvlinesize,
                                     (w >> s->chroma_x_shift) + 1, ((h >> s->chroma_y_shift) + 1) << field_based,
                                     sourcex >> s->chroma_x_shift, (src_y >> s->chroma_y_shift) * (1 << field_based),
                                     h_edge_pos >> s->chroma_x_shift, v_edge_pos >> s->chroma_y_shift);
            ptr_cb = s->sc.edge_emu_buffer;
            emu = 1;
        }

        uvsx = (uvsx << 2) >> lowres;
        uvsy = (uvsy << 2) >> lowres;
        if (field_select)
            ptr_cb += uvlinesize;
        pix_op[op_index](dest_cb, ptr_cb, uvlinesize, h >> s->chroma_y_shift, uvsx, uvsy);

        if (emu) {
            s->vdsp.emulated_edge_mc(s->sc.edge_emu_buffer, ptr_cr, uvlinesize, uvlinesize,
                                     (w >> s->chroma_x_shift) + 1, ((h >> s->chroma_y_shift) + 1) << field_based,
                                     sourcex >> s->chroma_x_shift, (src_y >> s->chroma_y_shift) * (1 << field_based),
                                     h_edge_pos >> s->chroma_x_shift, v_edge_pos >> s->chroma_y_shift);
            ptr_cr = s->sc.edge_emu_buffer;
        }
        if (field_select)
            ptr_cr += uvlinesize;
        pix_op[op_index](dest_cr, ptr_cr, uvlinesize, h >> s->chroma_y_shift, uvsx, uvsy);
    }
