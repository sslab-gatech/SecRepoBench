mx = motion_x;
    my = motion_y;

    sourcex = s->mb_x * block_s + (mx >> lowres);
    src_y = mb_y * block_s + (my >> lowres);
    uvsrc_x = sourcex >> s->chroma_x_shift;
    uvsrc_y = src_y >> s->chroma_y_shift;

    if ((unsigned)sourcex >= h_edge_pos || (unsigned)src_y >= v_edge_pos) {
        s->vdsp.emulated_edge_mc(s->sc.edge_emu_buffer, ref_picture[0] + src_y * linesize + sourcex,
                                 linesize, linesize,
                                 9, 9,
                                 sourcex, src_y, h_edge_pos, v_edge_pos);
        ptr_y = s->sc.edge_emu_buffer;
    } else {
        ptr_y = ref_picture[0] + src_y * linesize + sourcex;
    }

    // Process luma
    if (lowres) {
        // Handle low resolution specific processing
        pix_op[0](dest_y, ptr_y, linesize, block_s, mx & s_mask, my & s_mask);
    } else {
        // Normal processing
        pix_op[1](dest_y, ptr_y, linesize, block_s, mx & s_mask, my & s_mask);
    }

    // Process chroma
    if (s->chroma_x_shift || s->chroma_y_shift) {
        if ((unsigned)uvsrc_x >= h_edge_pos || (unsigned)uvsrc_y >= v_edge_pos) {
            s->vdsp.emulated_edge_mc(s->sc.edge_emu_buffer, ref_picture[1] + uvsrc_y * uvlinesize + uvsrc_x,
                                     uvlinesize, uvlinesize,
                                     9, 9,
                                     uvsrc_x, uvsrc_y, h_edge_pos, v_edge_pos);
            ptr_cb = s->sc.edge_emu_buffer;
            ptr_cr = s->sc.edge_emu_buffer;
        } else {
            ptr_cb = ref_picture[1] + uvsrc_y * uvlinesize + uvsrc_x;
            ptr_cr = ref_picture[2] + uvsrc_y * uvlinesize + uvsrc_x;
        }

        // Call chroma processing function
        pix_op[2](dest_cb, ptr_cb, uvlinesize, block_s, mx & s_mask, my & s_mask);
        pix_op[2](dest_cr, ptr_cr, uvlinesize, block_s, mx & s_mask, my & s_mask);
    }
