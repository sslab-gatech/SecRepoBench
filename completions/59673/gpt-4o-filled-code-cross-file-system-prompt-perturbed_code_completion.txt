if (s->quarter_sample) {
        motion_x /= 2;
        motion_y /= 2;
    }

    mx = motion_x >> lowres;
    my = motion_y >> lowres;
    sx = (motion_x & s_mask) << (3 - lowres);
    sy = (motion_y & s_mask) << (3 - lowres);

    sourcex = s->mb_x * block_s + mx;
    src_y   = mb_y * block_s + my;

    linesize = s->linesize >> field_based;
    uvlinesize = s->uvlinesize >> field_based;

    if (field_based) {
        src_y >>= 1;
        if (bottom_field) {
            dest_y += s->linesize;
            dest_cb += s->uvlinesize;
            dest_cr += s->uvlinesize;
        }
    }

    if (sourcex < 0 || sourcex + block_s > h_edge_pos ||
        src_y < 0 || src_y + h > v_edge_pos) {
        s->vdsp.emulated_edge_mc(s->sc.edge_emu_buffer, ref_picture[0] + src_y * linesize + sourcex,
                                 linesize, linesize, block_s + 1, h + 1,
                                 sourcex, src_y, h_edge_pos, v_edge_pos);
        ptr_y = s->sc.edge_emu_buffer;
    } else {
        ptr_y = ref_picture[0] + src_y * linesize + sourcex;
    }

    s->h264chroma.put_h264_chroma_pixels_tab[op_index](dest_y, ptr_y, linesize, block_s, sx, sy);

    if (!CONFIG_GRAY || !(s->avctx->flags & AV_CODEC_FLAG_GRAY)) {
        uvsx = (motion_x >> s->chroma_x_shift) & s_mask;
        uvsy = (motion_y >> s->chroma_y_shift) & s_mask;
        uvsrc_x = s->mb_x * (block_s >> s->chroma_x_shift) + (motion_x >> (lowres + s->chroma_x_shift));
        uvsrc_y = mb_y * (block_s >> s->chroma_y_shift) + (motion_y >> (lowres + s->chroma_y_shift));

        if (uvsrc_x < 0 || uvsrc_x + (block_s >> s->chroma_x_shift) > (h_edge_pos >> s->chroma_x_shift) ||
            uvsrc_y < 0 || uvsrc_y + (h >> s->chroma_y_shift) > (v_edge_pos >> s->chroma_y_shift)) {
            s->vdsp.emulated_edge_mc(s->sc.edge_emu_buffer, ref_picture[1] + uvsrc_y * uvlinesize + uvsrc_x,
                                     uvlinesize, uvlinesize, (block_s >> s->chroma_x_shift) + 1, (h >> s->chroma_y_shift) + 1,
                                     uvsrc_x, uvsrc_y, h_edge_pos >> s->chroma_x_shift, v_edge_pos >> s->chroma_y_shift);
            ptr_cb = s->sc.edge_emu_buffer;
        } else {
            ptr_cb = ref_picture[1] + uvsrc_y * uvlinesize + uvsrc_x;
        }

        pix_op[op_index](dest_cb, ptr_cb, uvlinesize, block_s >> s->chroma_x_shift, uvsx, uvsy);

        if (uvsrc_x < 0 || uvsrc_x + (block_s >> s->chroma_x_shift) > (h_edge_pos >> s->chroma_x_shift) ||
            uvsrc_y < 0 || uvsrc_y + (h >> s->chroma_y_shift) > (v_edge_pos >> s->chroma_y_shift)) {
            s->vdsp.emulated_edge_mc(s->sc.edge_emu_buffer, ref_picture[2] + uvsrc_y * uvlinesize + uvsrc_x,
                                     uvlinesize, uvlinesize, (block_s >> s->chroma_x_shift) + 1, (h >> s->chroma_y_shift) + 1,
                                     uvsrc_x, uvsrc_y, h_edge_pos >> s->chroma_x_shift, v_edge_pos >> s->chroma_y_shift);
            ptr_cr = s->sc.edge_emu_buffer;
        } else {
            ptr_cr = ref_picture[2] + uvsrc_y * uvlinesize + uvsrc_x;
        }

        pix_op[op_index](dest_cr, ptr_cr, uvlinesize, block_s >> s->chroma_x_shift, uvsx, uvsy);
    }
