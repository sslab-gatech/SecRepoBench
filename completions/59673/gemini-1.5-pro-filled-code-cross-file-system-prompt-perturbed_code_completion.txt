linesize   = s->linesize   << field_based;
    uvlinesize = s->uvlinesize << field_based;

    sx = motion_x & s_mask;
    sy = motion_y & s_mask;
    src_y = mb_y * block_s + (motion_y >> lowres + 1);
    sourcex = s->mb_x * block_s + (motion_x >> lowres + 1);

    ptr_y  = ref_picture[0] + src_y * linesize + sourcex;
    src_y += field_select * (linesize >> 1);

    if (field_based) {
        ptr_y += bottom_field * linesize;
        if (bottom_field != field_select)
            ref_picture = s->current_picture_ptr->f->data;
    }

    if ((unsigned) sourcex > FFMAX(h_edge_pos - (!!sx) - block_s, 0) ||
        (unsigned) src_y  > FFMAX(v_edge_pos - (!!sy) - h, 0)) {
        s->vdsp.emulated_edge_mc(s->sc.edge_emu_buffer, ptr_y,
                                linesize, linesize,
                                9, 9 + field_based,
                                sourcex, src_y,
                                h_edge_pos, v_edge_pos);
        ptr_y = s->sc.edge_emu_buffer;
    }

    sx = (sx << 2) >> lowres;
    sy = (sy << 2) >> lowres;

    if (!s->quarter_sample)
        s->hdsp.put_pixels_tab[0][op_index](dest_y, ptr_y, linesize, h);
    else
        s->hdsp.put_no_rnd_pixels_tab[0][op_index](dest_y, ptr_y, linesize, h);


    if (!CONFIG_GRAY || !(s->avctx->flags & AV_CODEC_FLAG_GRAY)) {
        mx = motion_x / 2;
        my = motion_y / 2;
        uvsx = (mx & s_mask) << 2;
        uvsy = (my & s_mask) << 2;

        uvsrc_x = s->mb_x * block_s / 2 + (mx >> lowres + 1);
        uvsrc_y = s->mb_y * block_s / 2 + (my >> lowres + 1);

        ptr_cb = ref_picture[1] + uvsrc_y * uvlinesize + uvsrc_x;
        ptr_cr = ref_picture[2] + uvsrc_y * uvlinesize + uvsrc_x;

        if (field_based) {
            ptr_cb += bottom_field * uvlinesize;
            ptr_cr += bottom_field * uvlinesize;
        }

        if ((unsigned) uvsrc_x > FFMAX(h_edge_pos - (!!uvsx / 4) - block_s / 2, 0) ||
            (unsigned) uvsrc_y > FFMAX(v_edge_pos - (!!uvsy / 4) - h / 2, 0)) {
            s->vdsp.emulated_edge_mc(s->sc.edge_emu_buffer, ptr_cb,
                                    uvlinesize, uvlinesize,
                                    9, 9 + field_based,
                                    uvsrc_x, uvsrc_y, h_edge_pos, v_edge_pos);
            ptr_cb = s->sc.edge_emu_buffer;

            s->vdsp.emulated_edge_mc(s->sc.edge_emu_buffer, ptr_cr,
                                     uvlinesize, uvlinesize,
                                     9, 9 + field_based,
                                     uvsrc_x, uvsrc_y, h_edge_pos, v_edge_pos);
            ptr_cr = s->sc.edge_emu_buffer;
        }

        uvsx >>= lowres;
        uvsy >>= lowres;

        pix_op[op_index](dest_cb, ptr_cb, uvlinesize, h >> s->chroma_y_shift, uvsx, uvsy);
        pix_op[op_index](dest_cr, ptr_cr, uvlinesize, h >> s->chroma_y_shift, uvsx, uvsy);
    }
