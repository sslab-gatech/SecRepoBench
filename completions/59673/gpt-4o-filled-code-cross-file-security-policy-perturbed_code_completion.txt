mx = motion_x >> lowres;
    my = motion_y >> lowres;
    sx = (motion_x & s_mask) << (3 - lowres);
    sy = (motion_y & s_mask) << (3 - lowres);

    sourcex = s->mb_x * block_s + (mx >> 1);
    src_y = (mb_y * block_s >> field_based) + (my >> 1);
    if (bottom_field)
        src_y += s->current_picture_ptr->f->linesize >> 1;

    if (sourcex < 0 || sourcex + block_s > h_edge_pos ||
        src_y < 0 || src_y + h > v_edge_pos) {
        s->vdsp.emulated_edge_mc(s->sc.edge_emu_buffer, ref_picture[0] + src_y * s->linesize + sourcex,
                                 s->linesize, s->linesize, block_s, h, sourcex, src_y, h_edge_pos, v_edge_pos);
        ptr_y = s->sc.edge_emu_buffer;
    } else {
        ptr_y = ref_picture[0] + src_y * s->linesize + sourcex;
    }

    s->hdsp.put_pixels_tab[0][0](dest_y, ptr_y, s->linesize, h);

    if (!CONFIG_GRAY || !(s->avctx->flags & AV_CODEC_FLAG_GRAY)) {
        uvsx = (motion_x >> s->chroma_x_shift) & s_mask;
        uvsy = (motion_y >> s->chroma_y_shift) & s_mask;
        uvsrc_x = s->mb_x * block_s / 2 + (mx >> (1 + s->chroma_x_shift));
        uvsrc_y = (mb_y * block_s >> (field_based + s->chroma_y_shift)) + (my >> (1 + s->chroma_y_shift));
        if (bottom_field)
            uvsrc_y += s->uvlinesize >> 1;

        uvlinesize = s->uvlinesize;
        if (uvsrc_x < 0 || uvsrc_x + (block_s >> 1) > (h_edge_pos >> s->chroma_x_shift) ||
            uvsrc_y < 0 || uvsrc_y + (h >> s->chroma_y_shift) > (v_edge_pos >> s->chroma_y_shift)) {
            s->vdsp.emulated_edge_mc(s->sc.edge_emu_buffer, ref_picture[1] + uvsrc_y * uvlinesize + uvsrc_x,
                                     uvlinesize, uvlinesize, block_s >> 1, h >> s->chroma_y_shift,
                                     uvsrc_x, uvsrc_y, h_edge_pos >> s->chroma_x_shift, v_edge_pos >> s->chroma_y_shift);
            ptr_cb = s->sc.edge_emu_buffer;
        } else {
            ptr_cb = ref_picture[1] + uvsrc_y * uvlinesize + uvsrc_x;
        }

        pix_op[op_index](dest_cb, ptr_cb, uvlinesize, block_s >> 1, uvsx, uvsy);

        if (uvsrc_x < 0 || uvsrc_x + (block_s >> 1) > (h_edge_pos >> s->chroma_x_shift) ||
            uvsrc_y < 0 || uvsrc_y + (h >> s->chroma_y_shift) > (v_edge_pos >> s->chroma_y_shift)) {
            s->vdsp.emulated_edge_mc(s->sc.edge_emu_buffer, ref_picture[2] + uvsrc_y * uvlinesize + uvsrc_x,
                                     uvlinesize, uvlinesize, block_s >> 1, h >> s->chroma_y_shift,
                                     uvsrc_x, uvsrc_y, h_edge_pos >> s->chroma_x_shift, v_edge_pos >> s->chroma_y_shift);
            ptr_cr = s->sc.edge_emu_buffer;
        } else {
            ptr_cr = ref_picture[2] + uvsrc_y * uvlinesize + uvsrc_x;
        }

        pix_op[op_index](dest_cr, ptr_cr, uvlinesize, block_s >> 1, uvsx, uvsy);
    }
