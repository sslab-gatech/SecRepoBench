linesize = s->linesize;
    uvlinesize = s->uvlinesize;
    ptr_y = ref_picture[0];
    ptr_cb = ref_picture[1];
    ptr_cr = ref_picture[2];

    mx = motion_x;
    my = motion_y;
    if (s->quarter_sample) {
        mx /= 2;
        my /= 2;
    }

    sourcex = s->mb_x * block_s + (mx >> lowres);
    src_y = (mb_y + (my >> lowres)) * (1 << field_based);
    uvsrc_x = sourcex >> s->chroma_x_shift;
    uvsrc_y = src_y >> s->chroma_y_shift;

    sx = mx & s_mask;
    sy = my & s_mask;
    uvsx = sx >> s->chroma_x_shift;
    uvsy = sy >> s->chroma_y_shift;

    ptr_y += src_y * linesize + sourcex;
    ptr_cb += uvsrc_y * uvlinesize + uvsrc_x;
    ptr_cr += uvsrc_y * uvlinesize + uvsrc_x;

    if ((unsigned)sourcex > FFMAX(h_edge_pos - (!!sx) - 2 * block_s, 0) ||
        (unsigned)src_y > FFMAX(v_edge_pos - (!!sy) - 2 * block_s, 0)) {
        s->vdsp.emulated_edge_mc(s->sc.edge_emu_buffer, ptr_y,
                                 linesize, linesize,
                                 2 * block_s + 1, (2 * block_s + 1) << field_based,
                                 sourcex, src_y,
                                 h_edge_pos, v_edge_pos);
        ptr_y = s->sc.edge_emu_buffer;
    }

    sx = (sx << 2) >> lowres;
    sy = (sy << 2) >> lowres;
    if (bottom_field)
        ptr_y += linesize;
    pix_op[op_index](dest_y, ptr_y, linesize, h, sx, sy);

    if ((unsigned)uvsrc_x > FFMAX(h_edge_pos - (!!uvsx) - block_s, 0) ||
        (unsigned)uvsrc_y > FFMAX(v_edge_pos - (!!uvsy) - block_s, 0)) {
        s->vdsp.emulated_edge_mc(s->sc.edge_emu_buffer, ptr_cb,
                                 uvlinesize, uvlinesize,
                                 block_s + 1, (block_s + 1) << field_based,
                                 uvsrc_x, uvsrc_y,
                                 h_edge_pos, v_edge_pos);
        ptr_cb = s->sc.edge_emu_buffer;
    }

    uvsx = (uvsx << 2) >> (lowres - s->chroma_x_shift);
    uvsy = (uvsy << 2) >> (lowres - s->chroma_y_shift);
    if (bottom_field)
        ptr_cb += uvlinesize;
    pix_op[op_index](dest_cb, ptr_cb, uvlinesize, h >> s->chroma_y_shift, uvsx, uvsy);

    if ((unsigned)uvsrc_x > FFMAX(h_edge_pos - (!!uvsx) - block_s, 0) ||
        (unsigned)uvsrc_y > FFMAX(v_edge_pos - (!!uvsy) - block_s, 0)) {
        s->vdsp.emulated_edge_mc(s->sc.edge_emu_buffer, ptr_cr,
                                 uvlinesize, uvlinesize,
                                 block_s + 1, (block_s + 1) << field_based,
                                 uvsrc_x, uvsrc_y,
                                 h_edge_pos, v_edge_pos);
        ptr_cr = s->sc.edge_emu_buffer;
    }
    if (bottom_field)
        ptr_cr += uvlinesize;
    pix_op[op_index](dest_cr, ptr_cr, uvlinesize, h >> s->chroma_y_shift, uvsx, uvsy);
