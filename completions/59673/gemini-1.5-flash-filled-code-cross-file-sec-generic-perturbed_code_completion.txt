mx = motion_x & s_mask;
    my = motion_y & s_mask;
    sourcex = s->mb_x * block_s + (motion_x >> lowres);
    src_y = mb_y * block_s + (motion_y >> lowres);
    uvsrc_x = sourcex >> s->chroma_x_shift;
    uvsrc_y = src_y >> s->chroma_y_shift;
    sx = mx << 2;
    sy = my << 2;
    linesize = s->linesize;
    uvlinesize = s->uvlinesize;
    ptr_y = ref_picture[0] + src_y * linesize + sourcex;
    ptr_cb = ref_picture[1] + uvsrc_y * uvlinesize + uvsrc_x;
    ptr_cr = ref_picture[2] + uvsrc_y * uvlinesize + uvsrc_x;
    if (sourcex < 0 || sourcex + block_s > h_edge_pos ||
        src_y < 0 || src_y + h > v_edge_pos) {
        s->vdsp.emulated_edge_mc(s->sc.edge_emu_buffer, ptr_y, linesize,
                                 linesize, h, block_s, sourcex, src_y,
                                 h_edge_pos, v_edge_pos);
        ptr_y = s->sc.edge_emu_buffer;
        if (s->avctx->bits_per_raw_sample == 8) {
            if (uvsrc_x < 0 || uvsrc_x + (block_s >> s->chroma_x_shift) > h_edge_pos >> s->chroma_x_shift ||
                uvsrc_y < 0 || uvsrc_y + (h >> s->chroma_y_shift) > v_edge_pos >> s->chroma_y_shift) {
                s->vdsp.emulated_edge_mc(s->sc.edge_emu_buffer, ptr_cb, uvlinesize,
                                         uvlinesize, h >> s->chroma_y_shift, block_s >> s->chroma_x_shift,
                                         uvsrc_x, uvsrc_y, h_edge_pos >> s->chroma_x_shift, v_edge_pos >> s->chroma_y_shift);
                ptr_cb = s->sc.edge_emu_buffer;
                s->vdsp.emulated_edge_mc(s->sc.edge_emu_buffer, ptr_cr, uvlinesize,
                                         uvlinesize, h >> s->chroma_y_shift, block_s >> s->chroma_x_shift,
                                         uvsrc_x, uvsrc_y, h_edge_pos >> s->chroma_x_shift, v_edge_pos >> s->chroma_y_shift);
                ptr_cr = s->sc.edge_emu_buffer;
            }
        }
    }
    s->dsp.put_pixels_clamped(dest_y, ptr_y, linesize, h, sx, sy);
    if (s->avctx->bits_per_raw_sample == 8)
        pix_op[op_index](dest_cb, ptr_cb, dest_cr, ptr_cr, uvlinesize, block_s >> s->chroma_x_shift, sx >> s->chroma_x_shift, sy >> s->chroma_y_shift);
