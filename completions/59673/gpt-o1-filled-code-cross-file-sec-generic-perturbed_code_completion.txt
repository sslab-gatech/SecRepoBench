int emu = 0;
    int src_x, src_y, dxy;
    int emu_w = block_s + 1;
    int emu_h = h + 1;

    // set line sizes based on whether this is field-based
    linesize   = field_based ? 2 * s->linesize   : s->linesize;
    uvlinesize = field_based ? 2 * s->uvlinesize : s->uvlinesize;

    // adjust destination pointers for bottom field
    if (field_based && bottom_field) {
        dest_y  += s->linesize;
        dest_cb += s->uvlinesize;
        dest_cr += s->uvlinesize;
    }

    // prepare luma motion vectors
    mx = motion_x;
    my = motion_y;
    if (s->quarter_sample) {
        mx >>= 1;
        my >>= 1;
    }
    dxy   = ((my & 1) << 1) + (mx & 1);
    mx   >>= 1;
    my   >>= 1;

    // compute luma source pointer
    src_x = mx + ((mx >= 0) ? 0 : -((mx & s_mask) != 0));
    src_y = mb_y * block_s + my;
    ptr_y = ref_picture[0] + src_y * linesize + src_x;

    // check for edge overlap and emulate if needed
    if ((unsigned)src_x > FFMAX(h_edge_pos - block_s, 0U) ||
        (unsigned)src_y > FFMAX(v_edge_pos - h,       0U)) {
        s->vdsp.emulated_edge_mc(s->sc.edge_emu_buffer, ptr_y, linesize, linesize,
                                 emu_w, emu_h, src_x, src_y,
                                 h_edge_pos, v_edge_pos);
        ptr_y = s->sc.edge_emu_buffer;
        emu   = 1;
    }

    // put the luma block using half-pel if needed
    s->dsp.put_no_rnd_pixels_tab[0][dxy](dest_y, ptr_y, linesize, h, 0);

    // handle chroma unless in gray mode
    if (!CONFIG_GRAY || !(s->avctx->flags & AV_CODEC_FLAG_GRAY)) {
        int cmx = motion_x, cmy = motion_y;

        // round chroma vectors for quarter sample
        if (s->quarter_sample) {
            cmx >>= 1;
            cmy >>= 1;
        }

        // shift and mask for chroma
        uvsx    = cmx & s_mask;
        uvsy    = cmy & s_mask;
        uvsrc_x = (cmx >> (lowres + s->chroma_x_shift));
        uvsrc_y = ((mb_y * block_s + (cmy >> lowres)) >> s->chroma_y_shift);

        ptr_cb = ref_picture[1] + uvsrc_y * uvlinesize + uvsrc_x;
        ptr_cr = ref_picture[2] + uvsrc_y * uvlinesize + uvsrc_x;

        // edge emu for chroma if needed
        if ((unsigned)uvsrc_x > FFMAX((s->h_edge_pos >> s->chroma_x_shift) - (block_s >> s->chroma_x_shift), 0U) ||
            (unsigned)uvsrc_y > FFMAX((s->v_edge_pos >> s->chroma_y_shift) - (h >> s->chroma_y_shift),       0U)) {
            // if not already emu'ed for luma, perform for chroma
            if (!emu) {
                // copy Cb
                s->vdsp.emulated_edge_mc(s->sc.edge_emu_buffer, ptr_cb, uvlinesize, uvlinesize,
                                         (block_s >> s->chroma_x_shift) + 1,
                                         (h        >> s->chroma_y_shift) + 1,
                                         uvsrc_x, uvsrc_y,
                                         s->h_edge_pos >> s->chroma_x_shift,
                                         s->v_edge_pos >> s->chroma_y_shift);
                ptr_cb = s->sc.edge_emu_buffer;
                // copy Cr right after Cb
                s->vdsp.emulated_edge_mc(s->sc.edge_emu_buffer + 16 * uvlinesize, ptr_cr,
                                         uvlinesize, uvlinesize,
                                         (block_s >> s->chroma_x_shift) + 1,
                                         (h        >> s->chroma_y_shift) + 1,
                                         uvsrc_x, uvsrc_y,
                                         s->h_edge_pos >> s->chroma_x_shift,
                                         s->v_edge_pos >> s->chroma_y_shift);
                ptr_cr = s->sc.edge_emu_buffer + 16 * uvlinesize;
            }
        }

        // apply chroma interpolation
        pix_op[op_index](dest_cb, ptr_cb, uvlinesize, h >> s->chroma_y_shift,
                         (uvsx << 2) >> (lowres + s->chroma_x_shift),
                         (uvsy << 2) >> (lowres + s->chroma_y_shift));
        pix_op[op_index](dest_cr, ptr_cr, uvlinesize, h >> s->chroma_y_shift,
                         (uvsx << 2) >> (lowres + s->chroma_x_shift),
                         (uvsy << 2) >> (lowres + s->chroma_y_shift));
    }
