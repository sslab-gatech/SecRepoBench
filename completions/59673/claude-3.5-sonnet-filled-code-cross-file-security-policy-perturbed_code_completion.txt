mx = motion_x >> lowres;
my = motion_y >> lowres;
sourcex = s->mb_x * 2 * block_s + (mx >> 2);
src_y   = (mb_y * 2 * block_s >> field_based) + (my >> 2);
uvsourcex = s->mb_x * block_s + (mx >> 3);
uvsrc_y   = (mb_y * block_s >> field_based) + (my >> 3);
sx = mx & 3;
sy = my & 3;
uvsx = (mx >> 1) & 1;
uvsy = (my >> 1) & 1;

if ((unsigned)sourcex >= h_edge_pos - (!!sx) - 2 * block_s ||
    (unsigned)src_y   >= v_edge_pos - (!!sy) - h) {
    s->vdsp.emulated_edge_mc(s->sc.edge_emu_buffer, ptr_y - 2 * block_s - 2,
                             linesize, linesize,
                             17, 17 + field_based,
                             sourcex - 2, src_y - 2,
                             h_edge_pos, v_edge_pos);
    ptr_y = s->sc.edge_emu_buffer + 2 * block_s + 2;
}
if (!CONFIG_GRAY || !(s->avctx->flags & AV_CODEC_FLAG_GRAY)) {
    if ((unsigned)uvsourcex >= h_edge_pos - (!!uvsx) - block_s ||
        (unsigned)uvsrc_y   >= v_edge_pos - (!!uvsy) - (h >> 1)) {
        s->vdsp.emulated_edge_mc(s->sc.edge_emu_buffer, ptr_cb,
                                 uvlinesize, uvlinesize,
                                 9, 9 + field_based,
                                 uvsourcex, uvsrc_y,
                                 h_edge_pos, v_edge_pos);
        ptr_cb = s->sc.edge_emu_buffer;
        s->vdsp.emulated_edge_mc(s->sc.edge_emu_buffer + 16, ptr_cr,
                                 uvlinesize, uvlinesize,
                                 9, 9 + field_based,
                                 uvsourcex, uvsrc_y,
                                 h_edge_pos, v_edge_pos);
        ptr_cr = s->sc.edge_emu_buffer + 16;
    }
}

pix_op[lowres - 1](dest_y, ptr_y, linesize, h, sx, sy);

if (!CONFIG_GRAY || !(s->avctx->flags & AV_CODEC_FLAG_GRAY)) {
    pix_op[op_index](dest_cb, ptr_cb, uvlinesize, h >> 1, uvsx, uvsy);
    pix_op[op_index](dest_cr, ptr_cr, uvlinesize, h >> 1, uvsx, uvsy);
}
