linesize = s->linesize;
    uvlinesize = s->uvlinesize;

    if (s->quarter_sample) {
        motion_x /= 2;
        motion_y /= 2;
    }

    mx = motion_x;
    my = motion_y;

    sourcex = s->mb_x * 16 + (mx >> (1 + lowres));
    src_y = (mb_y * 16 >> lowres) + (my >> (1 + lowres));

    uvsrc_x = s->mb_x * (8 >> s->chroma_x_shift) + (mx >> (2 + lowres + s->chroma_x_shift));
    uvsrc_y = (mb_y * (8 >> s->chroma_y_shift) >> lowres) + (my >> (2 + lowres + s->chroma_y_shift));

    sx = (mx & s_mask) << (2 - lowres);
    sy = (my & s_mask) << (2 - lowres);

    uvsx = (mx & s_mask) << (1 - lowres);
    uvsy = (my & s_mask) << (1 - lowres);

    ptr_y = ref_picture[0] + src_y * linesize + sourcex;
    ptr_cb = ref_picture[1] + uvsrc_y * uvlinesize + uvsrc_x;
    ptr_cr = ref_picture[2] + uvsrc_y * uvlinesize + uvsrc_x;

    if ((unsigned)sourcex > FFMAX(h_edge_pos - (!!sx) - 16, 0) ||
        (unsigned)src_y > FFMAX(v_edge_pos - (!!sy) - h, 0)) {
        s->vdsp.emulated_edge_mc(s->sc.edge_emu_buffer, ptr_y, linesize, linesize,
                                 17, h + 1, sourcex, src_y, h_edge_pos, v_edge_pos);
        ptr_y = s->sc.edge_emu_buffer;
        emu = 1;
    }

    if ((unsigned)uvsrc_x > FFMAX((h_edge_pos >> s->chroma_x_shift) - (!!uvsx) - (8 >> s->chroma_x_shift), 0) ||
        (unsigned)uvsrc_y > FFMAX((v_edge_pos >> s->chroma_y_shift) - (!!uvsy) - (h >> s->chroma_y_shift), 0)) {
        s->vdsp.emulated_edge_mc(s->sc.edge_emu_buffer, ptr_cb, uvlinesize, uvlinesize,
                                 9, (h >> s->chroma_y_shift) + 1, uvsrc_x, uvsrc_y,
                                 h_edge_pos >> s->chroma_x_shift, v_edge_pos >> s->chroma_y_shift);
        ptr_cb = s->sc.edge_emu_buffer;
        ptr_cr = s->sc.edge_emu_buffer + 256;
        emu = 1;
    }

    pix_op[op_index](dest_y, ptr_y, linesize, h, sx, sy);
    pix_op[op_index](dest_cb, ptr_cb, uvlinesize, h >> s->chroma_y_shift, uvsx, uvsy);
    pix_op[op_index](dest_cr, ptr_cr, uvlinesize, h >> s->chroma_y_shift, uvsx, uvsy);
