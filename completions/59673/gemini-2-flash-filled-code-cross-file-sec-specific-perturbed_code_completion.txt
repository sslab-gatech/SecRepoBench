mx = motion_x;
    my = motion_y;

    sx = mx & s_mask;
    sy = my & s_mask;
    sourcex = s->mb_x * block_s + (mx >> lowres);
    src_y   = s->mb_y * block_s + (my >> lowres);

    linesize = s->linesize;

    ptr_y = ref_picture[0] + src_y * linesize + sourcex;

    if ((unsigned) sourcex > FFMAX(h_edge_pos - (!!sx) - block_s, 0) ||
        (unsigned) src_y   > FFMAX(v_edge_pos - (!!sy) - h, 0)) {
        s->vdsp.emulated_edge_mc(s->sc.edge_emu_buffer, ptr_y,
                                 linesize, linesize,
                                 9, h + 1,
                                 sourcex, src_y, h_edge_pos, v_edge_pos);
        ptr_y = s->sc.edge_emu_buffer;
    }

    if (s->quarter_sample) {
        sx <<= 1;
        sy <<= 1;
    }

    s->dsp.put_pixels_tab[s->mpeg_enc_table_base + sy](dest_y, ptr_y,
                                                        linesize, linesize,
                                                        sx, h);

    if (!CONFIG_GRAY || !(s->avctx->flags & AV_CODEC_FLAG_GRAY)) {
        uvsx    = mx & (s_mask >> s->chroma_x_shift);
        uvsy    = my & (s_mask >> s->chroma_y_shift);
        uvsrc_x = s->mb_x * block_s + (mx >> (lowres + s->chroma_x_shift));
        uvsrc_y = s->mb_y * block_s + (my >> (lowres + s->chroma_y_shift));

        uvlinesize = s->uvlinesize;

        ptr_cb = ref_picture[1] + uvsrc_y * uvlinesize + uvsrc_x;
        ptr_cr = ref_picture[2] + uvsrc_y * uvlinesize + uvsrc_x;

        if ((unsigned) uvsrc_x > FFMAX((h_edge_pos >> s->chroma_x_shift) - (!!uvsx) - (block_s >> s->chroma_x_shift), 0) ||
            (unsigned) uvsrc_y > FFMAX((v_edge_pos >> s->chroma_y_shift) - (!!uvsy) - (h >> s->chroma_y_shift), 0)) {
            s->vdsp.emulated_edge_mc(s->sc.edge_emu_buffer, ptr_cb,
                                     uvlinesize, uvlinesize,
                                     9, (h >> s->chroma_y_shift) + 1,
                                     uvsrc_x, uvsrc_y,
                                     h_edge_pos >> s->chroma_x_shift,
                                     v_edge_pos >> s->chroma_y_shift);
            ptr_cb = s->sc.edge_emu_buffer;
            s->vdsp.emulated_edge_mc(s->sc.edge_emu_buffer, ptr_cr,
                                     uvlinesize, uvlinesize,
                                     9, (h >> s->chroma_y_shift) + 1,
                                     uvsrc_x, uvsrc_y,
                                     h_edge_pos >> s->chroma_x_shift,
                                     v_edge_pos >> s->chroma_y_shift);
            ptr_cr = s->sc.edge_emu_buffer;
        }

        if (s->quarter_sample) {
            uvsx <<= 1;
            uvsy <<= 1;
        }

        if (s->h263_aic) {
            pix_op[op_index](dest_cb, ptr_cb, uvlinesize, block_s >> s->chroma_x_shift, uvsx, uvsy);
            pix_op[op_index](dest_cr, ptr_cr, uvlinesize, block_s >> s->chroma_x_shift, uvsx, uvsy);
        } else {
            s->dsp.put_pixels_tab[s->mpeg_enc_table_base + 16 + uvsy](dest_cb, ptr_cb,
                                                                       uvlinesize, uvlinesize,
                                                                       uvsx, h >> s->chroma_y_shift);
            s->dsp.put_pixels_tab[s->mpeg_enc_table_base + 16 + uvsy](dest_cr, ptr_cr,
                                                                       uvlinesize, uvlinesize,
                                                                       uvsx, h >> s->chroma_y_shift);
        }
    }
