mx = motion_x;
    my = motion_y;

    sourcex = (mb_y * block_s) + (my >> lowres);
    src_y = (mb_y * block_s) + (my >> lowres);
    uvsrc_x = (mb_x * block_s) + (mx >> lowres);
    uvsrc_y = (mb_y * block_s) + (my >> lowres);

    if ((unsigned)sourcex >= h_edge_pos || (unsigned)src_y >= v_edge_pos) {
        s->vdsp.emulated_edge_mc(s->sc.edge_emu_buffer, ref_picture[0] + sourcex, 
                                 s->linesize, s->linesize, 
                                 9, 9, sourcex, src_y, h_edge_pos, v_edge_pos);
        ptr_y = s->sc.edge_emu_buffer;
    } else {
        ptr_y = ref_picture[0] + sourcex;
    }

    if ((unsigned)uvsrc_x >= (h_edge_pos >> s->chroma_x_shift) || 
        (unsigned)uvsrc_y >= (v_edge_pos >> s->chroma_y_shift)) {
        s->vdsp.emulated_edge_mc(s->sc.edge_emu_buffer, ref_picture[1] + uvsrc_x, 
                                 s->uvlinesize, s->uvlinesize, 
                                 9, 9, uvsrc_x, uvsrc_y, 
                                 h_edge_pos >> s->chroma_x_shift, 
                                 v_edge_pos >> s->chroma_y_shift);
        ptr_cb = s->sc.edge_emu_buffer;
        ptr_cr = s->sc.edge_emu_buffer;
    } else {
        ptr_cb = ref_picture[1] + uvsrc_x;
        ptr_cr = ref_picture[2] + uvsrc_x;
    }

    pix_op[op_index](dest_y, ptr_y, s->linesize, block_s, mx, my);
    pix_op[op_index](dest_cb, ptr_cb, s->uvlinesize, block_s, mx, my);
    pix_op[op_index](dest_cr, ptr_cr, s->uvlinesize, block_s, mx, my);
