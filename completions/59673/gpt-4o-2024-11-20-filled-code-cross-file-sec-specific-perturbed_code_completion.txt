mx = motion_x >> lowres;
    my = motion_y >> lowres;

    sx = (motion_x & s_mask) << (3 - lowres);
    sy = (motion_y & s_mask) << (3 - lowres);

    sourcex = (s->mb_x << (3 - lowres)) + mx;
    src_y = (mb_y << (3 - lowres)) + my;

    if (field_based) {
        src_y >>= 1;
        if (bottom_field)
            src_y += v_edge_pos;
    }

    ptr_y = ref_picture[0] + src_y * s->linesize + sourcex;

    if ((unsigned)sourcex > FFMAX(h_edge_pos - (!!sx) - block_s, 0) ||
        (unsigned)src_y > FFMAX(v_edge_pos - (!!sy) - h, 0)) {
        s->vdsp.emulated_edge_mc(s->sc.edge_emu_buffer, ptr_y, s->linesize, s->linesize,
                                 block_s + 1, h + 1, sourcex, src_y, h_edge_pos, v_edge_pos);
        ptr_y = s->sc.edge_emu_buffer;
    }

    s->hdsp.put_pixels_tab[1][lowres](dest_y, ptr_y, s->linesize, h, sx, sy);

    if (!CONFIG_GRAY || !(s->avctx->flags & AV_CODEC_FLAG_GRAY)) {
        uvsrc_x = sourcex >> s->chroma_x_shift;
        uvsrc_y = src_y >> s->chroma_y_shift;
        uvsx = sx >> s->chroma_x_shift;
        uvsy = sy >> s->chroma_y_shift;

        uvlinesize = s->uvlinesize;
        ptr_cb = ref_picture[1] + uvsrc_y * uvlinesize + uvsrc_x;
        ptr_cr = ref_picture[2] + uvsrc_y * uvlinesize + uvsrc_x;

        if ((unsigned)uvsrc_x > FFMAX((h_edge_pos >> s->chroma_x_shift) - (!!uvsx) - (block_s >> s->chroma_x_shift), 0) ||
            (unsigned)uvsrc_y > FFMAX((v_edge_pos >> s->chroma_y_shift) - (!!uvsy) - (h >> s->chroma_y_shift), 0)) {
            s->vdsp.emulated_edge_mc(s->sc.edge_emu_buffer, ptr_cb, uvlinesize, uvlinesize,
                                     (block_s >> s->chroma_x_shift) + 1, (h >> s->chroma_y_shift) + 1,
                                     uvsrc_x, uvsrc_y, h_edge_pos >> s->chroma_x_shift, v_edge_pos >> s->chroma_y_shift);
            ptr_cb = s->sc.edge_emu_buffer;

            s->vdsp.emulated_edge_mc(s->sc.edge_emu_buffer, ptr_cr, uvlinesize, uvlinesize,
                                     (block_s >> s->chroma_x_shift) + 1, (h >> s->chroma_y_shift) + 1,
                                     uvsrc_x, uvsrc_y, h_edge_pos >> s->chroma_x_shift, v_edge_pos >> s->chroma_y_shift);
            ptr_cr = s->sc.edge_emu_buffer;
        }

        pix_op[op_index](dest_cb, ptr_cb, uvlinesize, h >> s->chroma_y_shift, uvsx, uvsy);
        pix_op[op_index](dest_cr, ptr_cr, uvlinesize, h >> s->chroma_y_shift, uvsx, uvsy);
    }
