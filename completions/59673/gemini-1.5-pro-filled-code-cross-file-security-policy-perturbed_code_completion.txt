linesize   = s->linesize   << field_based;
    uvlinesize = s->uvlinesize << field_based;

    mx = motion_x & s_mask;
    my = motion_y & s_mask;
    sourcex = s->mb_x * block_s + (motion_x >> lowres + 1);
    src_y = mb_y * block_s + (motion_y >> lowres + 1) + bottom_field; // Added bottom_field

    sx = (mx << 2) >> lowres;
    sy = (my << 2) >> lowres;

    ptr_y  = ref_picture[0] + src_y * linesize + sourcex;
    ptr_cb = ref_picture[1] + uvsrc_y * uvlinesize + uvsx;
    ptr_cr = ref_picture[2] + uvsrc_y * uvlinesize + uvsx;

    if (field_select) {
         ptr_y  += linesize >> 1;
         ptr_cb += uvlinesize >> 1;
         ptr_cr += uvlinesize >> 1;
    }

    if ((unsigned)sourcex > FFMAX(h_edge_pos - (!!mx) - block_s * 2, 0) ||
        (unsigned)src_y   > FFMAX(v_edge_pos - (!!my) - h, 0)) { // Correct check
        s->vdsp.emulated_edge_mc(s->sc.edge_emu_buffer, ptr_y,
                                linesize, linesize,
                                17, 17 + field_based, // Correct sizes
                                sourcex, src_y, h_edge_pos, v_edge_pos);
        ptr_y = s->sc.edge_emu_buffer;
    }

    s->hdsp.put_pixels_tab[0][op_index](dest_y, ptr_y, linesize, h);


    if (!CONFIG_GRAY || !(s->avctx->flags & AV_CODEC_FLAG_GRAY)) {
        uvsx    = sourcex >> s->chroma_x_shift;
        uvsy    = src_y >> s->chroma_y_shift; // Correct shift

        uvsrc_x = s->mb_x * (block_s >> s->chroma_x_shift) + (motion_x >> (lowres + 1 + s->chroma_x_shift));
        uvsrc_y = mb_y * (block_s >> s->chroma_y_shift) + (motion_y >> (lowres + 1 + s->chroma_y_shift)) + (bottom_field >> s->chroma_y_shift); // Correct shift and add bottom_field

        uvsx = (uvsx << 2) >> lowres;
        uvsy = (uvsy << 2) >> lowres;

        if ((unsigned)uvsrc_x > FFMAX(h_edge_pos - (!!uvsx) - (block_s * 2 >> s->chroma_x_shift), 0) ||
            (unsigned)uvsrc_y > FFMAX(v_edge_pos - (!!uvsy) - (h >> s->chroma_y_shift), 0)) { // Correct check
            s->vdsp.emulated_edge_mc(s->sc.edge_emu_buffer, ptr_cb,
                                    uvlinesize, uvlinesize,
                                    9, 9 + field_based,
                                    uvsrc_x, uvsrc_y, h_edge_pos, v_edge_pos);
            ptr_cb = ptr_cr = s->sc.edge_emu_buffer;
        }

        pix_op[op_index](dest_cb, ptr_cb, uvlinesize, h >> s->chroma_y_shift, uvsx, uvsy);
        pix_op[op_index](dest_cr, ptr_cr, uvlinesize, h >> s->chroma_y_shift, uvsx, uvsy);
    }
