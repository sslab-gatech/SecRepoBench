size_t suffix_length = strnlen((const char*)curin, 128);
        if (suffix_length >= 128) return EXR_ERR_CORRUPT_CHUNK;

        mem = alloc_fn(suffix_length + 1); // +1 for null terminator
        if (!mem) return EXR_ERR_MEMORY_ALLOCATION;

        memcpy(mem, curin, suffix_length);
        mem[suffix_length] = '\0'; // Null-terminate the string

        out->_suffix = mem; // Store the pointer to the allocated string
        out->_isSuffixAllocated = DWA_CLASSIFIER_TRUE; // Mark as dynamically allocated
