const uint8_t* tmp = curin;
        while (len < *length && len < 128 && *tmp != '\0')
        {
            suffix[len++] = *tmp++;
        }

        if (len >= *length || len >= 128) return EXR_ERR_CORRUPT_CHUNK;

        curin += len + 1; // also skip the null terminator
        len++;             // include null terminator in total length

        mem = (char*) alloc_fn (len);
        if (!mem) return EXR_ERR_OUT_OF_MEMORY;

        memcpy (mem, suffix, len);

        out->_suffix     = mem;
        out->_alloc_flag = DWA_CLASSIFIER_ALLOC_SUFFIX;
