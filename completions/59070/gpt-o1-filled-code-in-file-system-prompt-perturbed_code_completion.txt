for (; consumed < 128; ++consumed)
{
    if (consumed >= *length) return EXR_ERR_CORRUPT_CHUNK;
    suffix[consumed] = (char)curin[consumed];
    if (suffix[consumed] == '\0')
    {
        consumed++;
        break;
    }
}

    return EXR_ERR_CORRUPT_CHUNK;
