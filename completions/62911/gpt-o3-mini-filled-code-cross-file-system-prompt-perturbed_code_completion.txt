if (prefix == NULL) {
    if ((entry->name[len] == 0) && (memcmp(entry->name, key_name, len) == 0)) {
        found = 1;
        break;
    }
} else {
    const xmlChar *colon = xmlStrchr(entry->name, ':');
    if (colon != NULL) {
        int prefixLen = colon - entry->name;
        if ((xmlStrlen(prefix) == (unsigned) prefixLen) &&
            (memcmp(entry->name, prefix, prefixLen) == 0)) {
            const xmlChar *local = colon + 1;
            if ((local[len] == 0) && (memcmp(local, key_name, len) == 0)) {
                found = 1;
                break;
            }
        }
    }
}
