((prefix == NULL) ?
 ((xmlStrncmp(entry->name, key_name, len) == 0) && (entry->name[len] == 0)) :
 ((colon = (const xmlChar *)xmlStrchr(entry->name, ':')) != NULL &&
  (xmlStrncmp(entry->name, prefix, colon - entry->name) == 0) &&
  (xmlStrcmp(colon + 1, key_name) == 0)))
{
    found = 1;
    break;
}
