if (prefix == NULL) {
    if (strcmp((const char *)entry->name, (const char *)name) == 0) {
        found = 1;
        break;
    }
} else {
    size_t prefixLen = strlen((const char *)prefix);
    size_t nameLen = strlen((const char *)name);
    if (strncmp((const char *)entry->name, (const char *)prefix, prefixLen) == 0 &&
        entry->name[prefixLen] == ':' &&
        strncmp((const char *)(entry->name + prefixLen + 1), (const char *)name, nameLen) == 0 &&
        entry->name[prefixLen + 1 + nameLen] == '\0') {
        found = 1;
        break;
    }
}
