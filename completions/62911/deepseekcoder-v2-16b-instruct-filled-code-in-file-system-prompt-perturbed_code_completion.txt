if (entry->hashValue == hashValue) {
    if ((prefix == NULL && len == klen && strncmp(entry->name, name, len) == 0) ||
        (prefix != NULL && plen + 1 + len == klen &&
         strncmp(entry->name, prefix, plen) == 0 && entry->name[plen] == ':' &&
         strncmp(entry->name + plen + 1, name, len) == 0)) {
        found = 1;
        break;
    }
