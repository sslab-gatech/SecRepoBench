((prefix == NULL) &&
 (xmlStrncmp(name, entry->name, len) == 0) &&
 (entry->name[len] == 0)) ||
((prefix != NULL) &&
 (xmlStrncmp(prefix, entry->name, plen) == 0) &&
 (entry->name[plen] == ':') &&
 (xmlStrncmp(name, entry->name + plen + 1, len) == 0) &&
 (entry->name[plen + 1 + len] == 0))
{
    found = 1;
    break;
}
