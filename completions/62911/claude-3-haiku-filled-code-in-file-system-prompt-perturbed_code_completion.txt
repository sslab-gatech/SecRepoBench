    if (strncmp((const char *) name, (const char *) entry->name, len) == 0 && entry->name[len] == '\0') {

    }

    size_t prefixLen = strlen((const char *) prefix);
    if (strncmp((const char *) prefix, (const char *) entry->name, prefixLen) == 0 &&

        found = 1;
    }
