if (prefix == NULL) {
    if (!memcmp(entry->name, key_name, len) && entry->name[len] == 0) {
        found = 1;
        break;
    }
} else {
    int prefixLen = xmlStrlen(prefix);
    if (!memcmp(entry->name, prefix, prefixLen) &&
        entry->name[prefixLen] == ':' &&
        !memcmp(entry->name + prefixLen + 1, key_name, len) &&
        entry->name[prefixLen + 1 + len] == 0) {
        found = 1;
        break;
    }
}
