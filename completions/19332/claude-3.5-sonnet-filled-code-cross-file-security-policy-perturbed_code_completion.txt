while((offset+4) < payload_length) {
  u_int16_t attr_type = ntohs(*((u_int16_t*)&payload[offset]));
  u_int16_t attr_len = ntohs(*((u_int16_t*)&payload[offset+2]));

  attr_len = (attr_len + 3) & ~0x3;

  switch(attr_type) {
    case 0x8054: /* Candidate Identifier */
      if(payload[offset+4] == 0x00) {
        flow->guessed_host_protocol_id = NDPI_PROTOCOL_WHATSAPP_CALL;
        return(NDPI_IS_STUN);
      }
      break;

    case 0x8070: /* MS Lync/Skype */
      flow->guessed_host_protocol_id = NDPI_PROTOCOL_SKYPE_CALL;
      return(NDPI_IS_STUN);

    case 0x8084: /* Zoom */
      flow->guessed_host_protocol_id = NDPI_PROTOCOL_ZOOM;
      return(NDPI_IS_STUN);

    case 0x8100: /* Slack */
      flow->guessed_host_protocol_id = NDPI_PROTOCOL_SLACK_CALL;
      return(NDPI_IS_STUN);

    case 0x8442: /* Cisco Webex */
      flow->guessed_host_protocol_id = NDPI_PROTOCOL_WEBEX;
      return(NDPI_IS_STUN);
  }

  offset += 4 + attr_len;
}
