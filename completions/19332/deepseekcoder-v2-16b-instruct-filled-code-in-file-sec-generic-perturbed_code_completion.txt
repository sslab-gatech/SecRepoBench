u_int offset = 20;

  /*
   This can either be the standard RTCP or Ms Lync RTCP that
   later will become Ms Lync RTP. In this case we need to
   be careful before deciding about the protocol before dissecting the packet

   MS Lync = Skype
   https://en.wikipedia.org/wiki/Skype_for_Business
   */

  while (offset < payload_length) {
    u_int8_t attribute_type = payload[offset];
    u_int16_t attribute_length = payload[offset + 1];

    if (attribute_length % 4 != 0) {
      attribute_length = (attribute_length & 0xFFFC) + 4; // Align to 4-byte boundaries
    }

    if (attribute_type == 0x0001) { // Software attribute
      // Extract the software name and determine the application protocol
      char software[256];
      memcpy(software, &payload[offset + 2], attribute_length - 2);
      software[attribute_length - 2] = '\0';

      if (strstr(software, "WhatsApp") != NULL) {
        flow->guessed_host_protocol_id = NDPI_PROTOCOL_WHATSAPP_CALL;
        return NDPI_IS_STUN;
      } else if (strstr(software, "Skype") != NULL) {
        flow->guessed_host_protocol_id = NDPI_PROTOCOL_SKYPE;
        return NDPI_IS_STUN;
      } else if (strstr(software, "Zoom") != NULL) {
        flow->guessed_host_protocol_id = NDPI_PROTOCOL_ZOOM;
        return NDPI_IS_STUN;
      }
    }

    offset += attribute_length;
  }
