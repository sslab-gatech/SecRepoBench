int run_length = token & 0x7F; // Extract run length from token
      uint8_t value = token >> 7;     // Extract value for the run
      if (run_length > neblock) {
        // Ensure run length does not exceed the block size
        return BLOSC2_ERROR_DATA;
      }
      memset(comp_buffer, value, run_length); // Fill the destination buffer with the run value
      nbytes = run_length; // Set the number of decompressed bytes
