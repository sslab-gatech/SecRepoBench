import blosc2

# Create a chunk made of zeros
chunk_zeros = blosc2.Blosc2Chunk.zeros(1024, 4)
print(chunk_zeros)

# Create a chunk made of NaNs
chunk_nans = blosc2.Blosc2Chunk.nans(1024, 4)
print(chunk_nans)

# Create a chunk made of repeated values
chunk_repeatval = blosc2.Blosc2Chunk.repeatval(1024, 4, b'x')
print(chunk_repeatval)
