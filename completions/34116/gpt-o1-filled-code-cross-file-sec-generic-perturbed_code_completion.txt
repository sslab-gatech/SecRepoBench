/* check if the key is a string */
if (k->type != MSGPACK_OBJECT_STR) {
    continue;
}

/* ensure the key's pointer is not NULL */
if (!k->via.str.ptr) {
    flb_free(tmp_out_buf);
    *out_buf = NULL;
    msgpack_unpacked_destroy(&unpacked_data);
    return -1;
}

/* check if the key's size matches the expected time key size */
if (k->via.str.size != slen) {
    continue;
}
