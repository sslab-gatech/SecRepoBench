if (strncmp(k->via.str.ptr, time_key, k->via.str.size) == 0) {
    /* We found the key, break the loop and keep the index */
    if (parser->time_keep == FLB_FALSE) {
        skip = i;
        break;
    }
    else {
        skip = -1;
    }
    break;
}
else {
    if (k->via.str.ptr == NULL) {
        flb_free(mp_buf);
        msgpack_unpacked_destroy(&unpacked_data);
        *out_buf = NULL;
        return -1;
    }
    k = NULL;
    v = NULL;
}
