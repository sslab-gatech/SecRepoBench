for (tagIndex = 0; tagIndex < TagCount; tagIndex++) {

        if (io -> Read(io, &Tag, sizeof(cmsTagEntry), 1) != 1) {
            return FALSE;
        }

        // Adjust endianness
        Tag.sig    = _cmsAdjustEndianess32(Tag.sig);
        Tag.offset = _cmsAdjustEndianess32(Tag.offset);
        Tag.size   = _cmsAdjustEndianess32(Tag.size);

        // Check for invalid offset or size
        if (Tag.offset + Tag.size > HeaderSize) {
            cmsSignalError(Icc ->ContextID, cmsERROR_BAD_TAG, "Invalid offset/size for tag %x", Tag.sig);
            return FALSE;
        }

        Icc -> TagSigs[Icc->TagCount] = Tag.sig;
        Icc -> TagOffsets[Icc->TagCount] = Tag.offset;
        Icc -> TagSizes[Icc->TagCount] = Tag.size;
        Icc ->TagLinked[Icc->TagCount] = (cmsTagSignature) 0;

        // Check if the tag is already present. If so, link it.
        for (j=0; j < Icc->TagCount; j++) {
            if (Icc->TagOffsets[j] == Tag.offset &&
                Icc->TagSizes[j] == Tag.size &&
                Icc->TagSigs[j] != Tag.sig)
            {
                cmsTagDescriptor* TagDescriptor = _cmsGetTagDescriptor(Icc ->ContextID, Tag.sig);
                cmsTagDescriptor* LinkedTagDescriptor = _cmsGetTagDescriptor(Icc ->ContextID, Icc->TagSigs[j]);

                // Only link tags if their types are compatible
                if (TagDescriptor != NULL && LinkedTagDescriptor != NULL) {
                    if (IsTypeSupported(TagDescriptor, LinkedTagDescriptor->Signature)) {
                        Icc ->TagLinked[Icc->TagCount] = Icc->TagSigs[j];
                    } else {
                        cmsSignalError(Icc ->ContextID, cmsERROR_BAD_TAG_TYPE, "Incompatible tag types for linking: %x and %x", Tag.sig, Icc->TagSigs[j]);
                        return FALSE;
                    }
                }
                break;
            }
        }

        Icc -> TagCount++;
    }
