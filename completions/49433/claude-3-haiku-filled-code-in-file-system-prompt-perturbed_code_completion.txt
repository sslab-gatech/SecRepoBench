// Read tag directory
if (!_cmsReadUInt32Number(io, &TagCount)) return FALSE;
if (TagCount > MAX_TABLE_TAG) {

    cmsSignalError(Icc ->ContextID, cmsERROR_RANGE, "Too many tags (%d)", TagCount);
    return FALSE;
}

Icc -> TagCount = 0;
for (i = 0; i < TagCount; i++) {
    cmsTagEntry Tag;

    if (!io->Read(io, &Tag, sizeof(cmsTagEntry), 1)) {
        return FALSE;
    }

    // Adjust endianness
    Tag.sig    = (cmsTagSignature) _cmsAdjustEndianess32((cmsUInt32Number) Tag.sig);
    Tag.offset = _cmsAdjustEndianess32(Tag.offset);
    Tag.size   = _cmsAdjustEndianess32(Tag.size);

    // Check for valid offset and size
    if (Tag.offset + Tag.size > HeaderSize) {
        cmsSignalError(Icc->ContextID, cmsERROR_CORRUPTION_DETECTED, "Tag at offset %d has size %d, which exceeds header size %d", Tag.offset, Tag.size, HeaderSize);
        return FALSE;
    }

    // Search for existing tag with matching offset and size
    int j;
    for (j = 0; j < Icc->TagCount; j++) {
        if (Icc->TagOffsets[j] == Tag.offset && Icc->TagSizes[j] == Tag.size) {
            Icc->TagLinked[i] = Icc->TagNames[j];
            break;
        }
    }

    // If no matching tag found, add new tag
    if (j == Icc->TagCount) {
        Icc->TagNames[Icc->TagCount] = Tag.sig;
        Icc->TagOffsets[Icc->TagCount] = Tag.offset;
        Icc->TagSizes[Icc->TagCount] = Tag.size;
        Icc->TagLinked[Icc->TagCount] = (cmsTagSignature)0;
        Icc->TagCount++;
    }
}
