for (tagIndex = 0; tagIndex < TagCount; tagIndex++) {

    if (!io->Read(io, &Tag, sizeof(cmsTagEntry), 1)) return FALSE;

    // Adjust endianness of tag
    Tag.sig    = (cmsTagSignature) _cmsAdjustEndianess32((cmsUInt32Number) Tag.sig);
    Tag.offset = _cmsAdjustEndianess32((cmsUInt32Number) Tag.offset);
    Tag.size   = _cmsAdjustEndianess32((cmsUInt32Number) Tag.size);

    // Validate offset and size
    if (Tag.offset + Tag.size > HeaderSize) {
        cmsSignalError(Icc->ContextID, cmsERROR_RANGE, "Tag offset %d + size %d goes beyond profile size %d", 
                       Tag.offset, Tag.size, HeaderSize);
        continue;
    }

    // Check if there's already a tag with the same offset and size
    cmsTagSignature LinkedSig = (cmsTagSignature) 0;
    for (j = 0; j < Icc->TagCount; j++) {
        if (Icc->TagOffsets[j] == Tag.offset && Icc->TagSizes[j] == Tag.size) {
            LinkedSig = Icc->TagNames[j];
            break;
        }
    }

    // Now add the tag to the directory
    if (Icc->TagCount >= MAX_TABLE_TAG) {
        cmsSignalError(Icc->ContextID, cmsERROR_RANGE, "Too many tags (%d)", Icc->TagCount);
        return FALSE;
    }

    Icc->TagNames[Icc->TagCount] = Tag.sig;
    Icc->TagOffsets[Icc->TagCount] = Tag.offset;
    Icc->TagSizes[Icc->TagCount] = Tag.size;
    Icc->TagSaveAsRaw[Icc->TagCount] = TRUE;
    Icc->TagPtrs[Icc->TagCount] = NULL;
    Icc->TagLinked[Icc->TagCount] = LinkedSig;  // Set the link if any

    Icc->TagCount++;
}
