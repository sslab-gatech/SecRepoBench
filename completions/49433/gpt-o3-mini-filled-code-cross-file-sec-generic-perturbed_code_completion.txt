for (tagIndex = 0; tagIndex < TagCount; tagIndex++) {

    if (!_cmsReadUInt32Number(io, &Tag.signature) ||
        !_cmsReadUInt32Number(io, &Tag.offset) ||
        !_cmsReadUInt32Number(io, &Tag.size)) {
        cmsSignalError(Icc->ContextID, cmsERROR_READ, "Error reading tag directory entry");
        return FALSE;
    }

    Tag.signature = _cmsAdjustEndianess32(Tag.signature);
    Tag.offset    = _cmsAdjustEndianess32(Tag.offset);
    Tag.size      = _cmsAdjustEndianess32(Tag.size);

    if (Tag.offset > HeaderSize ||
        Tag.size > HeaderSize ||
        (Tag.offset + Tag.size) > HeaderSize) {
        cmsSignalError(Icc->ContextID, cmsERROR_CORRUPTION, "Invalid tag directory entry (offset/size out of bounds)");
        return FALSE;
    }

    cmsTagSignature linkedSig = 0;
    for (j = 0; j < Icc->TagCount; j++) {
        if ((Icc->TagOffsets[j] == Tag.offset) && (Icc->TagSizes[j] == Tag.size)) {
            linkedSig = Icc->TagSignatures[j];
            break;
        }
    }

    Icc->TagSignatures[Icc->TagCount] = Tag.signature;
    Icc->TagOffsets[Icc->TagCount]     = Tag.offset;
    Icc->TagSizes[Icc->TagCount]       = Tag.size;
    Icc->TagLinked[Icc->TagCount]      = linkedSig;
    Icc->TagCount++;
}
