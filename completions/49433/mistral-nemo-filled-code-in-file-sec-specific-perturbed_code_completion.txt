static
cmsBool _cmsReadUInt32Number(cmsIOHANDLER* iohandler, cmsUInt32Number* Number)
{
    cmsUInt8Number Buffer[4];

    if (iohandler -> Read(iohandler, Buffer, 4, 1) != 1) return FALSE;

    *Number = (cmsUInt32Number) _cmsAdjustEndianess32(*((cmsUInt32Number*) Buffer));
    return TRUE;
}

static
cmsBool _cmsWriteUInt32Number(cmsIOHANDLER* iohandler, cmsUInt32Number Number)
{
    cmsUInt8Number Buffer[4];

    Buffer[0] = (cmsUInt8Number) _cmsAdjustEndianess32(Number);
    Buffer[1] = (cmsUInt8Number) _cmsAdjustEndianess32(Number >> 8);
    Buffer[2] = (cmsUInt8Number) _cmsAdjustEndianess24(Number >> 16);
    Buffer[3] = (cmsUInt8Number) _cmsAdjustEndianess24(Number >> 24);

    return iohandler -> Write(iohandler, Buffer, 4, 1) == 1;
}
