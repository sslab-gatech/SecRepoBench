for (tagIndex = 0; tagIndex < TagCount; tagIndex++) {

        if (io -> Read(io, &Tag, sizeof(cmsTagEntry), 1) != 1) return FALSE;

        // Adjust endianness
        Tag.sig    = _cmsAdjustEndianess32(Tag.sig);
        Tag.offset = _cmsAdjustEndianess32(Tag.offset);
        Tag.size   = _cmsAdjustEndianess32(Tag.size);

        // Check offsets and sizes
        if (Tag.offset + Tag.size > HeaderSize) {
            cmsSignalError(Icc ->ContextID, cmsERROR_BAD_TAG, "Tag %x has invalid offset/size", Tag.sig);
            return FALSE;
        }

        Icc -> TagSigs[Icc -> TagCount] = Tag.sig;
        Icc -> TagOffsets[Icc -> TagCount] = Tag.offset;
        Icc -> TagSizes[Icc -> TagCount] = Tag.size;
        Icc ->TagLinked[Icc -> TagCount] = (cmsTagSignature) 0;

        // Check for duplicated tags. In this case, link to the existing tag to avoid multiple allocation of memory
        for (j=0; j < Icc -> TagCount; j++) {
            if (Icc -> TagOffsets[j] == Tag.offset &&
                Icc -> TagSizes[j] == Tag.size) {

                Icc ->TagLinked[Icc -> TagCount] = Icc ->TagSigs[j];
                break;
            }
        }

        Icc -> TagCount++;
    }

    SetLinks(Icc);
