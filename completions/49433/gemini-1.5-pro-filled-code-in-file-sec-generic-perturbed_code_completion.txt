for (j = 0; j < TagCount; j++) {

        if (!io->Read(io, &Tag, sizeof(cmsTagEntry), 1)) return FALSE;

        Tag.sig    = (cmsTagSignature) _cmsAdjustEndianess32((cmsUInt32Number) Tag.sig);
        Tag.offset = _cmsAdjustEndianess32((cmsUInt32Number) Tag.offset);
        Tag.size   = _cmsAdjustEndianess32((cmsUInt32Number) Tag.size);

        // Check for invalid offset and size
        if (Tag.offset + Tag.size > HeaderSize) {
            cmsSignalError(Icc->ContextID, cmsERROR_BAD_TAG, "Offset and size of tag %u extend beyond profile size", j);
            return FALSE;
        }

        if (Tag.size == 0) {
            cmsSignalError(Icc->ContextID, cmsERROR_BAD_TAG, "Size of tag %u is zero", j);
            return FALSE;
        }

        Icc->TagNames[Icc->TagCount]  = Tag.sig;
        Icc->TagOffsets[Icc->TagCount] = Tag.offset;
        Icc->TagSizes[Icc->TagCount]   = Tag.size;
        Icc->TagPtrs[Icc->TagCount] = NULL;  // Mark as not yet read
        Icc->TagTypeHandlers[Icc->TagCount] = NULL;
        Icc->TagSaveAsRaw[Icc->TagCount] = FALSE;

        // Check for duplicated tags. Many profiles does duplicate
        // tags, so check for duplicated offsets and mark the second
        // as a link.
        for (tagIndex = 0; tagIndex < Icc->TagCount; tagIndex++) {
            if (Icc->TagOffsets[tagIndex] == Tag.offset &&
                Icc->TagSizes[tagIndex] == Tag.size) {
                Icc->TagLinked[Icc->TagCount] = Icc->TagNames[tagIndex];
                break;
            }
        }

        Icc->TagCount++;
    }
