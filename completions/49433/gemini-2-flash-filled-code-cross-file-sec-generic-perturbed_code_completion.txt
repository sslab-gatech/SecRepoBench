for (tagIndex = 0; tagIndex < TagCount; tagIndex++) {

        if (io -> Read(io, &Tag, sizeof(cmsTagEntry), 1) != 1) return FALSE;

        Tag.sig    = (cmsTagSignature) _cmsAdjustEndianess32(Tag.sig);
        Tag.offset = _cmsAdjustEndianess32(Tag.offset);
        Tag.size   = _cmsAdjustEndianess32(Tag.size);

        // Zero-sized tags are invalid
        if (Tag.size == 0) {
            cmsSignalError(Icc ->ContextID, cmsERROR_RANGE, "Tag '%#x' has invalid size (zero)", (unsigned int) Tag.sig);
            continue;
        }

        // Check for out-of-bounds offsets -- Do not allow anything outside the file
        if (Tag.offset > Icc ->IOhandler ->ReportedSize) {
            cmsSignalError(Icc->ContextID, cmsERROR_RANGE, "Tag '%#x' has invalid offset (%d)", (unsigned int)Tag.sig, (int)Tag.offset);
            continue;
        }

        // Check for out-of-bounds sizes -- Do not allow anything outside the file
        if ((Tag.offset + Tag.size) > Icc ->IOhandler ->ReportedSize) {

            cmsSignalError(Icc ->ContextID, cmsERROR_RANGE, "Tag '%#x' is out of profile (offset=%d, size=%d, profile=%d)",
                (unsigned int) Tag.sig, (int) Tag.offset, (int) Tag.size, (int) Icc ->IOhandler ->ReportedSize);
            continue;
        }

        // Store the tag
        Icc ->TagSignatures[Icc ->TagCount] = Tag.sig;
        Icc ->TagOffsets[Icc ->TagCount]    = Tag.offset;
        Icc ->TagSizes[Icc ->TagCount]      = Tag.size;
        Icc ->TagLinked  [Icc ->TagCount]    = (cmsTagSignature) 0;

        // Check for links
        for (j=0; j < Icc ->TagCount; j++) {

            if ((Icc ->TagOffsets[j] == Tag.offset) && (Icc ->TagSizes[j] == Tag.size)) {

                Icc ->TagLinked[Icc ->TagCount] = Icc ->TagSignatures[j];
                break;
            }
        }

        Icc ->TagCount++;
    }

    // Fix the links
    SetLinks(Icc);
