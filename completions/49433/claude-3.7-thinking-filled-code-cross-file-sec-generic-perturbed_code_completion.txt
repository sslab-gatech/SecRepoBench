for (tagIndex = 0; tagIndex < TagCount; tagIndex++) {
    
    // Read the tag signature, offset, and size
    if (!_cmsReadUInt32Number(io, &Tag.sig)) return FALSE;
    if (!_cmsReadUInt32Number(io, &Tag.offset)) return FALSE;
    if (!_cmsReadUInt32Number(io, &Tag.size)) return FALSE;
    
    // Adjust endianness
    Tag.sig = _cmsAdjustEndianess32(Tag.sig);
    Tag.offset = _cmsAdjustEndianess32(Tag.offset);
    Tag.size = _cmsAdjustEndianess32(Tag.size);
    
    // Validate offset and size
    if (Tag.offset >= HeaderSize || Tag.size > HeaderSize - Tag.offset) {
        cmsSignalError(Icc->ContextID, cmsERROR_CORRUPTION, "Tag offset %u + size %u goes outside the profile size %u", Tag.offset, Tag.size, HeaderSize);
        continue;  // Skip this tag
    }
    
    // Check for zero size
    if (Tag.size == 0) {
        cmsSignalError(Icc->ContextID, cmsERROR_CORRUPTION, "Tag size is zero");
        continue;  // Skip this tag
    }
    
    // Add this tag to the profile
    Icc->Tags[Icc->TagCount] = (cmsTagSignature) Tag.sig;
    Icc->TagOffsets[Icc->TagCount] = Tag.offset;
    Icc->TagSizes[Icc->TagCount] = Tag.size;
    Icc->TagLinked[Icc->TagCount] = (cmsTagSignature) 0;  // No link yet
    
    // Check for existing tags with matching offset and size
    for (j = 0; j < Icc->TagCount; j++) {
        if (Icc->TagOffsets[j] == Tag.offset && Icc->TagSizes[j] == Tag.size) {
            // We found a link
            Icc->TagLinked[Icc->TagCount] = Icc->Tags[j];
            break;
        }
    }
    
    Icc->TagCount++;
}
