for (tagIndex = 0; tagIndex < TagCount; tagIndex++) {

        // Read tag entry
        if (io->Read(io, &Tag, sizeof(cmsTagEntry), 1) != 1) {
            return FALSE;
        }

        // Adjust endianness
        Tag.sig    = _cmsAdjustEndianess32(Tag.sig);
        Tag.offset = _cmsAdjustEndianess32(Tag.offset);
        Tag.size   = _cmsAdjustEndianess32(Tag.size);

        // Check that tag offset and size are within the profile's header size
        if (Tag.offset + Tag.size > HeaderSize || Tag.offset < sizeof(cmsICCHeader)) {
            cmsSignalError(Icc->ContextID, cmsERROR_RANGE, "Tag %08x out of profile bounds", Tag.sig);
            return FALSE;
        }

        // Check for duplicate tags (links)
        for (j = 0; j < Icc->TagCount; j++) {
            if (Icc->TagOffsets[j] == Tag.offset && Icc->TagSizes[j] == Tag.size) {
                // This is a link to an existing tag
                Icc->TagLinked[Icc->TagCount] = Icc->TagNames[j];
                break;
            }
        }
        if (j == Icc->TagCount) {
            // Not a link, mark as not linked
            Icc->TagLinked[Icc->TagCount] = 0;
        }

        // Store tag info
        Icc->TagNames[Icc->TagCount]   = Tag.sig;
        Icc->TagOffsets[Icc->TagCount] = Tag.offset;
        Icc->TagSizes[Icc->TagCount]   = Tag.size;

        Icc->TagCount++;
    }
