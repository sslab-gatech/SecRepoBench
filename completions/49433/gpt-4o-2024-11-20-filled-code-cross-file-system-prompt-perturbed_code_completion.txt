for (tagIndex = 0; tagIndex < TagCount; tagIndex++) {
        if (!_cmsReadUInt32Number(io, &Tag.sig) ||  // Read tag signature
            !_cmsReadUInt32Number(io, &Tag.offset) ||  // Read tag offset
            !_cmsReadUInt32Number(io, &Tag.size)) {  // Read tag size
            return FALSE;
        }

        // Adjust endianness
        Tag.offset = _cmsAdjustEndianess32(Tag.offset);
        Tag.size = _cmsAdjustEndianess32(Tag.size);

        // Validate tag offset and size
        if (Tag.offset + Tag.size > HeaderSize || Tag.size == 0) {
            cmsSignalError(Icc->ContextID, cmsERROR_CORRUPTION_DETECTED, 
                           "Tag %u has invalid offset/size", tagIndex);
            return FALSE;
        }

        // Check for duplicate tags
        for (j = 0; j < Icc->TagCount; j++) {
            if (Icc->TagOffsets[j] == Tag.offset && Icc->TagSizes[j] == Tag.size) {
                // Found a link, store the linked tag signature
                Icc->TagLinked[j] = Tag.sig;
                break;
            }
        }

        // If no link was found, store the tag normally
        if (j == Icc->TagCount) {
            if (Icc->TagCount >= MAX_TABLE_TAG) {
                cmsSignalError(Icc->ContextID, cmsERROR_RANGE, "Too many tags");
                return FALSE;
            }

            Icc->TagNames[Icc->TagCount] = Tag.sig;
            Icc->TagOffsets[Icc->TagCount] = Tag.offset;
            Icc->TagSizes[Icc->TagCount] = Tag.size;
            Icc->TagLinked[Icc->TagCount] = 0;  // No link
            Icc->TagCount++;
        }
    }

    // Establish links between tags
    if (!SetLinks(Icc)) {
        return FALSE;
    }
