for (tagIndex = 0; tagIndex < TagCount; tagIndex++) {

    if (io -> Read(io, &Tag, sizeof(cmsTagEntry), 1)!= 1) {
        return FALSE;
    }

    // Adjust endianness of the tag's signature, offset, and size
    Tag.sig    = (cmsTagSignature) _cmsAdjustEndianess32((cmsUInt32Number) Tag.sig);
    Tag.offset = _cmsAdjustEndianess32(Tag.offset);
    Tag.size   = _cmsAdjustEndianess32(Tag.size);

    // Check if the tag's offset and size are valid and within the profile's header size
    if (Tag.offset + Tag.size > HeaderSize) {
        cmsSignalError(Icc ->ContextID, cmsERROR_CORRUPTION_DETECTED, "Invalid tag offset or size");
        return FALSE;
    }

    // Store the tag's signature, offset, and size in the profile structure
    Icc -> TagNames[j] = Tag.sig;
    Icc -> TagOffsets[j] = Tag.offset;
    Icc -> TagSizes[j] = Tag.size;

    // Check for existing tags with matching offsets and sizes to establish links between them
    for (j = 0; j < Icc -> TagCount; j++) {
        if (Icc -> TagOffsets[j] == Tag.offset && Icc -> TagSizes[j] == Tag.size) {
            Icc -> TagLinked[j] = Tag.sig;
            break;
        }
    }

    // Increment the tag count in the profile structure
    Icc -> TagCount++;
    j++;
}
