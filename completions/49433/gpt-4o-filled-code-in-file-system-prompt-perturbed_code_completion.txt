for (tagIndex = 0; tagIndex < TagCount; tagIndex++) {

        if (!_cmsReadTagEntry(io, &Tag)) return FALSE;

        // Check for valid offset and size
        if (Tag.offset + Tag.size > HeaderSize) {
            cmsSignalError(Icc->ContextID, cmsERROR_RANGE, "Tag offset and size exceed profile size");
            return FALSE;
        }

        // Store tag information
        Icc->TagNames[Icc->TagCount] = Tag.sig;
        Icc->TagOffsets[Icc->TagCount] = Tag.offset;
        Icc->TagSizes[Icc->TagCount] = Tag.size;

        // Check for existing tags with the same offset and size to establish links
        for (j = 0; j < Icc->TagCount; j++) {
            if (Icc->TagOffsets[j] == Tag.offset && Icc->TagSizes[j] == Tag.size) {
                Icc->TagLinked[Icc->TagCount] = Icc->TagNames[j];
                break;
            }
        }

        Icc->TagCount++;
    }
