for (j = 0; j < TagCount; j++) {

        if (io -> Read(io, &Tag, sizeof(cmsTagEntry), 1) != 1) {
            return FALSE;
        }

        // validate the tag offset and size
        if (_cmsAdjustEndianess32(Tag.offset) + _cmsAdjustEndianess32(Tag.size) > HeaderSize) {
            cmsSignalError(Icc->ContextID, cmsERROR_BAD_TAG, "Tag %d has invalid offset/size", j);
            return FALSE;
        }

        Icc -> TagNames[Icc->TagCount]  = (cmsTagSignature) _cmsAdjustEndianess32(Tag.sig);
        Icc -> TagOffsets[Icc->TagCount] = _cmsAdjustEndianess32(Tag.offset);
        Icc -> TagSizes[Icc->TagCount]   = _cmsAdjustEndianess32(Tag.size);
        Icc -> TagPtrs[Icc->TagCount] = NULL; // No data yet
        Icc -> TagTypeHandlers[Icc->TagCount] = NULL;
        Icc -> TagSaveAsRaw[Icc->TagCount] = FALSE;

        // Check for duplicated tags. Some buggy profiles does duplicate
        // tags, so if found, mark as linked.
        for (tagIndex = 0; tagIndex < Icc->TagCount; tagIndex++) {
            if (Icc->TagOffsets[tagIndex] == Icc->TagOffsets[Icc->TagCount] &&
                Icc->TagSizes[tagIndex] == Icc->TagSizes[Icc->TagCount]) {

                Icc->TagLinked[Icc->TagCount] = Icc->TagNames[tagIndex];
                break;
            }
        }

        Icc -> TagCount++;
    }
