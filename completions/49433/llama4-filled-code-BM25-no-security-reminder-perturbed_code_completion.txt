for (tagIndex = 0; tagIndex < TagCount; tagIndex++) {
    if (!_cmsReadTagEntry(io, &Tag)) return FALSE;

    // Adjust endianness
    Tag.sig   = _cmsAdjustEndianess32(Tag.sig);
    Tag.offset = _cmsAdjustEndianess32(Tag.offset);
    Tag.size   = _cmsAdjustEndianess32(Tag.size);

    // Check if the tag is within the profile's bounds
    if (Tag.offset >= HeaderSize || Tag.size > HeaderSize - Tag.offset) {
        cmsSignalError(Icc->ContextID, cmsERROR_RANGE, "Tag %d has invalid offset or size", tagIndex);
        return FALSE;
    }

    // Check for existing tags with matching offsets and sizes
    for (j = 0; j < Icc->TagCount; j++) {
        if (Icc->TagOffsets[j] == Tag.offset && Icc->TagSizes[j] == Tag.size) {
            // If a link is found, store the linked tag signature in the profile structure
            Icc->TagLinked[Icc->TagCount] = Icc->TagSignatures[j];
            break;
        }
    }

    // Store the tag's signature, offset, and size in the profile structure
    Icc->TagSignatures[Icc->TagCount] = Tag.sig;
    Icc->TagOffsets[Icc->TagCount]    = Tag.offset;
    Icc->TagSizes[Icc->TagCount]      = Tag.size;

    // Increment the tag count in the profile structure for each successfully processed tag
    Icc->TagCount++;
}
