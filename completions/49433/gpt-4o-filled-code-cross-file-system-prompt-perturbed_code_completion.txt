for (tagIndex = 0; tagIndex < TagCount; tagIndex++) {
    if (!_cmsReadUInt32Number(io, &Tag.sig) ||
        !_cmsReadUInt32Number(io, &Tag.offset) ||
        !_cmsReadUInt32Number(io, &Tag.size)) {
        return FALSE;
    }

    // Adjust endianness
    Tag.offset = _cmsAdjustEndianess32(Tag.offset);
    Tag.size = _cmsAdjustEndianess32(Tag.size);

    // Check if the tag is within the profile's header size
    if (Tag.offset + Tag.size > HeaderSize) {
        cmsSignalError(Icc->ContextID, cmsERROR_RANGE, "Tag out of range");
        return FALSE;
    }

    // Check for existing tags with matching offsets and sizes
    for (j = 0; j < Icc->TagCount; j++) {
        if (Icc->TagOffsets[j] == Tag.offset && Icc->TagSizes[j] == Tag.size) {
            Icc->TagLinked[j] = Tag.sig;
            break;
        }
    }

    // If no link was found, store the tag
    if (j == Icc->TagCount) {
        Icc->TagNames[Icc->TagCount] = Tag.sig;
        Icc->TagOffsets[Icc->TagCount] = Tag.offset;
        Icc->TagSizes[Icc->TagCount] = Tag.size;
        Icc->TagLinked[Icc->TagCount] = 0; // No link
        Icc->TagCount++;
    }
}
