for (tagIndex = 0; tagIndex < TagCount; tagIndex++) {
    
    // Read the tag entry
    if (!io->Read(io, &Tag, sizeof(cmsTagEntry), 1)) return FALSE;
    
    // Adjust endianness
    Tag.sig    = (cmsTagSignature) _cmsAdjustEndianess32(Tag.sig);
    Tag.offset = _cmsAdjustEndianess32(Tag.offset);
    Tag.size   = _cmsAdjustEndianess32(Tag.size);
    
    // Validate the tag
    if (Tag.offset >= HeaderSize || Tag.size >= HeaderSize || 
        Tag.offset + Tag.size > HeaderSize) {
        cmsSignalError(Icc->ContextID, cmsERROR_CORRUPTION, 
                       "Tag %d invalid (offset=%d, size=%d, header size=%d)", 
                       tagIndex, Tag.offset, Tag.size, HeaderSize);
        continue;  // Skip this tag
    }
    
    // Store the tag
    Icc->TagNames[Icc->TagCount]   = Tag.sig;
    Icc->TagOffsets[Icc->TagCount] = Tag.offset;
    Icc->TagSizes[Icc->TagCount]   = Tag.size;
    Icc->TagLinked[Icc->TagCount]  = (cmsTagSignature)0;  // No link by default
    
    // Check for links with already processed tags
    for (j = 0; j < Icc->TagCount; j++) {
        
        // Check if this is a tag with the same offset and size of another tag
        if (Icc->TagOffsets[j] == Tag.offset && 
            Icc->TagSizes[j] == Tag.size) {
            
            // This tag is linked to the one I found
            Icc->TagLinked[Icc->TagCount] = Icc->TagNames[j];
            break;
        }
    }
    
    Icc->TagCount++;
}

// Establish the links
SetLinks(Icc);
