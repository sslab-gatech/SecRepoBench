for (tagIndex = 0; tagIndex < TagCount; tagIndex++) {
    if (!_cmsReadTagEntry(io, &Tag)) return FALSE;

    // Ensure that the offset and size are within the profile's header size
    if (_cmsAdjustEndianess32(Tag.offset) >= HeaderSize || _cmsAdjustEndianess32(Tag.size) > HeaderSize - _cmsAdjustEndianess32(Tag.offset)) {
        cmsSignalError(Icc ->ContextID, cmsERROR_RANGE, "Invalid tag offset or size");
        return FALSE;
    }

    // Store the tag's signature, offset, and size in the profile structure
    Icc -> Tags[Icc -> TagCount].sig = _cmsAdjustEndianess32(Tag.sig);
    Icc -> Tags[Icc -> TagCount].offset = _cmsAdjustEndianess32(Tag.offset);
    Icc -> Tags[Icc -> TagCount].size = _cmsAdjustEndianess32(Tag.size);

    // Check for existing tags with matching offsets and sizes to establish links between them
    for (j = 0; j < Icc -> TagCount; j++) {
        if (Icc -> Tags[j].offset == Icc -> Tags[Icc -> TagCount].offset && 
            Icc -> Tags[j].size == Icc -> Tags[Icc -> TagCount].size) {
            Icc -> Tags[Icc -> TagCount].linkedSig = Icc -> Tags[j].sig;
            break;
        }
    }

    // Increment the tag count in the profile structure for each successfully processed tag
    Icc -> TagCount++;
}

// Set the links after all tags have been processed
if (!SetLinks(Icc)) return FALSE;
