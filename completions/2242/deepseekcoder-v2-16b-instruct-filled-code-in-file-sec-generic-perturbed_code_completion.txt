if (ms->flags & MAGIC_APPLE) {
	if (!firstline && file_printf(ms, "\n- ") == -1)
		return -1;
	if (file_printf(ms, "%.8s", m->apple) == -1)
		return -1;
}
if (ms->flags & MAGIC_EXTENSION) {
	if (!firstline && file_printf(ms, "\n- ") == -1)
		return -1;
	if (file_printf(ms, "%s", m->ext) == -1)
		return -1;
}
if (ms->flags & MAGIC_MIME_TYPE) {
	if (!firstline && file_printf(ms, "\n- ") == -1)
		return -1;
	if (file_printf(ms, "%s", m->mimetype) == -1)
		return -1;
}
return 0;
