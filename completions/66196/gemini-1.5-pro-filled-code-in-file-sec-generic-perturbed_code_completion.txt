if (bitrateIdx == 15) {
		GF_LOG(GF_LOG_ERROR, GF_LOG_CODING, ("[MPEG-1/2 Audio] Bitrate index not valid\n"));
		return 0;
	}
	/*MPEG-1*/
	if (version & 1) lidx = 0;
	/*MPEG-2/2.5*/
	else lidx = (version ? 3 : 4);

	if (lidx == 4) {
		if (layer == 1) {
			GF_LOG(GF_LOG_ERROR, GF_LOG_CODING, ("[MPEG-1/2 Audio] Layer index not valid\n"));
			return 0;
		}
		lidx = 4;
	}
	else {
		lidx = layer - 1;
	}
	if (lidx >= 5 || bitrateIdx >= 15) {
		GF_LOG(GF_LOG_ERROR, GF_LOG_CODING, ("[MPEG-1/2 Audio] Invalid bitrate/layer index %d/%d\n", lidx, bitrateIdx));
		return 0;
	}
	return bitrate_table[lidx][bitrateIdx];
