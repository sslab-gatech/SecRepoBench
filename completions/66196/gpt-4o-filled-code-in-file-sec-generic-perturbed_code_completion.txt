if (version == 3) {
		switch (layer) {
		case 1:
			lidx = 0;
			break;
		case 2:
			lidx = 1;
			break;
		case 3:
			lidx = 2;
			break;
		default:
			GF_LOG(GF_LOG_ERROR, GF_LOG_CODING, ("[MPEG-1/2 Audio] Layer not valid\n"));
			return 0;
		}
	} else {
		switch (layer) {
		case 1:
			lidx = 3;
			break;
		case 2:
		case 3:
			lidx = 4;
			break;
		default:
			GF_LOG(GF_LOG_ERROR, GF_LOG_CODING, ("[MPEG-1/2 Audio] Layer not valid\n"));
			return 0;
		}
	}

	if (bitrateIdx == 0 || bitrateIdx == 15) {
		GF_LOG(GF_LOG_ERROR, GF_LOG_CODING, ("[MPEG-1/2 Audio] Bitrate index not valid\n"));
		return 0;
	}

	return bitrate_table[lidx][bitrateIdx];
