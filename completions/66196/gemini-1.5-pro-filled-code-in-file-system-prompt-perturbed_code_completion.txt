if (bitrateIdx == 15) {
		GF_LOG(GF_LOG_ERROR, GF_LOG_CODING, ("[MPEG-1/2 Audio] Bitrate index not valid\n"));
		return 0;
	}
	if (version == 3) lidx = 0;	/*MPEG 1*/
	else if (layer == 1) lidx = 3; /*MPEG-2/2.5 Layer 1*/
	else lidx = 4; /*MPEG-2/2.5 Layer 2/3*/

	if ((lidx>4) || (bitrateIdx>14)) {
		GF_LOG(GF_LOG_ERROR, GF_LOG_CODING, ("[MPEG-1/2 Audio] Layer/bitrate index not valid\n"));
		return 0;
	}
	return bitrate_table[lidx][bitrateIdx];
