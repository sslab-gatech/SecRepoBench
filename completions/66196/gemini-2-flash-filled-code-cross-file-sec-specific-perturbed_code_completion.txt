if (version == 3) { /*MPEG-1*/
		if (layer == 3) lidx = 0;
		else if (layer == 2) lidx = 1;
		else if (layer == 1) lidx = 2;
		else {
			GF_LOG(GF_LOG_ERROR, GF_LOG_PARSER, ("Invalid MP3 layer %d\n", layer));
			return 0;
		}
	} else { /*MPEG-2/2.5*/
		if (layer == 3) lidx = 3;
		else if (layer == 2) lidx = 4;
		else if (layer == 1) lidx = 5;
		else {
			GF_LOG(GF_LOG_ERROR, GF_LOG_PARSER, ("Invalid MP3 layer %d\n", layer));
			return 0;
		}
	}

	if (lidx > 5) {
		GF_LOG(GF_LOG_ERROR, GF_LOG_PARSER, ("Invalid MP3 layer index %d\n", lidx));
		return 0;
	}

	if (bitrateIdx > 14) {
		GF_LOG(GF_LOG_ERROR, GF_LOG_PARSER, ("Invalid MP3 bitrate index %d\n", bitrateIdx));
		return 0;
	}

	return gf_mp3_bitrate_table[lidx][bitrateIdx];
