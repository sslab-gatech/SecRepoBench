if (version == 3) {
    lidx = 3 - layer;
} else {
    lidx = 6 - layer;
}

if (lidx > 3) {
    GF_LOG(GF_LOG_ERROR, GF_LOG_CODING, ("Invalid MP3 layer index %d\n", lidx));
    return 0;
}

if (bitrateIdx > 15) {
    GF_LOG(GF_LOG_ERROR, GF_LOG_CODING, ("Invalid MP3 bitrate index %d\n", bitrateIdx));
    return 0;
}

static const u16 mp3_bitrates[5][16] = {
    {0, 32, 64, 96, 128, 160, 192, 224, 256, 288, 320, 352, 384, 416, 448, 0},
    {0, 32, 48, 56, 64, 80, 96, 112, 128, 160, 192, 224, 256, 320, 384, 0},

    {0, 8, 16, 24, 32, 40, 48, 56, 64, 80, 96, 112, 128, 144, 160, 0}
};

return mp3_bitrates[lidx][bitrateIdx];
