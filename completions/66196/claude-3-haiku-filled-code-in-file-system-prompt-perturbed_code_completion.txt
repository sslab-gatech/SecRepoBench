u32 gf_mp3_bit_rate(u32 hdr)
{
	u8 version = gf_mp3_version(hdr);
	u8 layer = gf_mp3_layer(hdr);
	u8 bitrateIdx = (hdr >> 12) & 0xF;
	u32 lidx;
	/*MPEG-1*/
	if (version == 3) {
		if (layer == 1) {
			lidx = 0;
		}
		else if (layer == 2) {
			lidx = 1;
		}
		else {
			lidx = 2;
		}
	}
	/*MPEG-2 or MPEG-2.5*/
	else {
		if (layer == 1) {
			lidx = 3;
		}
		else {
			lidx = 4;
		}
	}
	if (bitrateIdx == 0 || bitrateIdx == 15) {
		return 0;
	}
	return bitrate_table[lidx][bitrateIdx];
}
