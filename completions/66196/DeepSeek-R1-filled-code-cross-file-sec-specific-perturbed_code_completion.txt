if (version == 3) {
		lidx = 3 - layer;
	} else {
		lidx = (layer == 3) ? 3 : 4;
	}

	if (lidx >= 5) {
		GF_LOG(GF_LOG_ERROR, GF_LOG_CODEC, ("Invalid layer index %u\n", lidx));
		return 0;
	}

	if (bitrateIdx >= 16) {
		GF_LOG(GF_LOG_ERROR, GF_LOG_CODEC, ("Invalid bitrate index %u\n", bitrateIdx));
		return 0;
	}

	return mpeg_bitrate[lidx][bitrateIdx];
