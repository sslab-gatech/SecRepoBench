if (version == 3) { /*MPEG 1*/
		if (layer == 3) lidx = 0; /*layer 1*/
		else if (layer == 2) lidx = 1; /*layer 2*/
		else if (layer == 1) lidx = 2; /*layer 3*/
		else {
			GF_LOG(GF_LOG_ERROR, GF_LOG_CONTAINER, ("[MPEG-1] Invalid layer index: %d\n", layer));
			return 0;
		}
		if (bitrateIdx == 0 || bitrateIdx == 15) return 0; /*free or reserved*/
		return gf_mp3_bitrate_tbl[lidx][bitrateIdx - 1];
	} else if ((version == 2) || (version == 0)) {/*MPEG-2 LSF/MPEG-2.5 LSF*/
		if (layer == 3) lidx = 3; /*layer 1*/
		else if (layer == 2) lidx = 4; /*layer 2*/
		else if (layer == 1) lidx = 5; /*layer 3*/
		else {
			GF_LOG(GF_LOG_ERROR, GF_LOG_CONTAINER, ("[MPEG-2/2.5] Invalid layer index: %d\n", layer));
			return 0;
		}
		if (bitrateIdx == 0 || bitrateIdx == 15) return 0; /*free or reserved*/
		return gf_mp3_bitrate_tbl[lidx][bitrateIdx - 1];
	}
	return 0;
