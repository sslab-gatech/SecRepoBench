if (lookupType == LookupType::Extension)
{
  if (edit_count == 0)
  {
    unsigned int prevType = 0;
    bool sameType = true;
    for (unsigned int i = 0; i < get_subtable_count(); i++)
    {
      unsigned int currType = get_subtable<TSubTable>(i).get_type();
      if (i == 0)
        prevType = currType;
      else if (currType != prevType)
      {
        sameType = false;
        break;
      }
    }
    if (!sameType)
      return_trace (false);
  }
}
