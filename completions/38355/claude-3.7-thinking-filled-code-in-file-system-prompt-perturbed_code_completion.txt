// For Extension Lookups (type = 7 for GSUB, type = 9 for GPOS), make sure all subtables
// have the same Extension lookup type.
if ((get_type() == 7 || get_type() == 9) && c->get_num_edits() == 0)
{
  unsigned int type = 0;
  unsigned count = get_subtable_count();
  for (unsigned int i = 0; i < count; i++)
  {
    const auto &subtable = get_subtable<TSubTable>(i);
    unsigned int subtable_type = subtable.u.extension.get_type();
    if (i == 0)
      type = subtable_type;
    else if (type != subtable_type)
      return_trace (false);
  }
}
