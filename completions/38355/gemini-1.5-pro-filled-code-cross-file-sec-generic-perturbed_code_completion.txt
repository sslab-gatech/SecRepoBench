      if (edit_count == 0)
      {
        hb_vector_t<TSubTable*> subtables = get_subtables<TSubTable> ();
        if (!subtables.empty ())
        {
          LookupType first_subtable_type = subtables[0]->get_type ();
          for (unsigned i = 1; i < subtables.length; i++)
          {
            if (subtables[i]->get_type () != first_subtable_type)
              return_trace (false);
          }
        }
