if (get_type () == TSubTable::Extension && c->edit_count == 0)
{
  unsigned subtable_type = 0;
  for (unsigned i = 0; i < subtablecount; i++)
  {
    const auto& subtable = get_subtables<TSubTable> ()[i];
    unsigned current_type = subtable.u.extension.get_type ();
    if (i == 0)
      subtable_type = current_type;
    else if (subtable_type != current_type)
      return_trace (false);
  }
}
