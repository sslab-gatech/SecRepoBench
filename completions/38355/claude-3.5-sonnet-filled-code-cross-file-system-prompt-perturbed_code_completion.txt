if (get_type () == ExtensionType::EXTENSION && c->edit_count == 0)
{
  unsigned first_type = 0;
  for (unsigned i = 0; i < subtablecount; i++)
  {
    const ExtensionSubst& extension = get_subtable<ExtensionSubst> (i);
    unsigned current_type = extension.get_type ();
    if (i == 0)
      first_type = current_type;
    else if (current_type != first_type)
      return_trace (false);
  }
}
