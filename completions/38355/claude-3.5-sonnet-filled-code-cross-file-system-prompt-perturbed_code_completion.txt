if (get_type() == ExtensionType && c->edit_count == 0)
{
  unsigned int first_type = 0;
  for (const auto &subtable : get_subtables<TSubTable>())
  {
    unsigned int type = subtable.u.extension.get_type();
    if (first_type == 0)
      first_type = type;
    else if (type != first_type)
      return_trace (false);
  }
}
