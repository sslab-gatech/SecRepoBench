if (get_type () == Extension && c->edit_count == 0)
{
  auto subtables = get_subtables<TSubTable> ();
  unsigned int count = get_subtable_count ();
  
  if (count > 1)
  {
    unsigned int first_type = subtables[0].get_extension_type ();
    
    for (unsigned int i = 1; i < count; i++)
    {
      if (subtables[i].get_extension_type () != first_type)
        return_trace (false);
    }
  }
}
