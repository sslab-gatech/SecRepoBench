if (get_type () == LookupType::Extension && c->edit_count == 0)
{
  unsigned int subtableCount = get_subtable_count ();
  LookupType firstType = LookupType::Unsupported;
  for (unsigned int i = 0; i < subtableCount; i++)
  {
    const TSubTable& subTable = get_subtables<TSubTable>()[i];
    LookupType currentType = subTable.get_type ();
    if (firstType == LookupType::Unsupported)
      firstType = currentType;
    else if (currentType != firstType)
      return_trace (false);
  }
}
return_trace (true);
