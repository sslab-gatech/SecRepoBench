if (get_type () == LookupType::Extension)
{
  unsigned int expected_type = 0xFFFF; // Initialize with invalid value
  for (unsigned i = 0; i < subtablecount; i++)
  {
    unsigned int sub_type = get_subtable<TSubTable> (i).get_type ();
    if (i == 0)
      expected_type = sub_type;
    else if (unlikely (sub_type != expected_type))
      return_trace (false);
  }
}
