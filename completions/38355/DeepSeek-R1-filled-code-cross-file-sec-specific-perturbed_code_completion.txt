if (get_type () == 7 || get_type () == 8) // Check Extension lookup types (GSUB7/GPOS8)
{
    unsigned int extension_type = get_type ();
    for (unsigned int i = 0; i < subtablecount; i++)
    {
        const TSubTable& subtable = this + get_subtable<TSubTable> (i);
        if (subtable.get_type () != extension_type)
            return_trace (false);
    }
