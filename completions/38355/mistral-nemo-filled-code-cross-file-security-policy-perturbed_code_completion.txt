if (get_type () == ExtensionType) {
  unsigned subtablecount = get_subtable_count ();
  if (unlikely (subtablecount == 0)) return_trace (false);
  TSubTable *subtables = get_subtables<TSubTable> ();
  HBUINT16 type = subtables[0].get_type ();
  for (unsigned i = 1; i < subtablecount; ++i) {
    if (unlikely (subtables[i].get_type () != type)) return_trace (false);
  }
}
