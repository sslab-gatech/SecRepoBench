if (get_type() == LookupType::Extension && c->edit_count == 0)
    {
      unsigned int extension_type = 0;
      bool first_subtable = true;

      for (unsigned int i = 0; i < subtablecount; i++)
      {
        const TSubTable &subtable = get_subtable<TSubTable>(i);
        unsigned int subtable_type = subtable.get_type();

        if (first_subtable)
        {
          extension_type = subtable_type;
          first_subtable = false;
        }
        else if (subtable_type != extension_type)
        {
          return_trace(false);
        }
      }
    }
