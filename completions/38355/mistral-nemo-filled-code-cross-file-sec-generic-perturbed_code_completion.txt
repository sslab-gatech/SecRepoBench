if (get_type() == ExtensionType) {
  const auto& subtables = get_subtables<TSubTable>();
  if (subtables.empty()) {
    return_trace(false);
  }
  const auto first_subtable_type = subtables[0].get_type();
  if (std::any_of(subtables.begin() + 1, subtables.end(), [first_subtable_type](const auto& subtable) {
    return subtable.get_type() != first_subtable_type;
  })) {
    return_trace(false);
  }
}
