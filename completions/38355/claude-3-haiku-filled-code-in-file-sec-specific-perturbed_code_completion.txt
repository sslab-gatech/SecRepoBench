// Check if the type is an Extension and edit_count is zero.
// If so, ensure that all subtables of the Extension lookup have the same type.
// If any subtable has a different type, return false, indicating a failure.
unsigned int edit_count = 0;
hb_tag_t lookup_type = get_type ();
if (lookup_type == HB_OT_LAYOUT_LOOKUP_TYPE_EXTENSION)
{
  for (unsigned int i = 0; i < get_subtable_count (); i++)
  {
    hb_tag_t subtable_type = get_subtable<Extension> (i).lookup_type;
    if (subtable_type != lookup_type)
      return_trace (false);
    edit_count++;
  }
  if (edit_count == 0)
    return_trace (false);
}
return_trace (true);
