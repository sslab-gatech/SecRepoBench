if (get_type() == ExtensionType && c->edit_count == 0)
{
  unsigned int first_type = 0;
  for (const auto &subtable : get_subtables<TSubTable>())
  {
    unsigned int current_type = subtable.u.extension.get_type();
    if (first_type == 0)
    {
      first_type = current_type;
    }
    else if (current_type != first_type)
    {
      return_trace(false);
    }
  }
}
