dataspace_id = H5Aget_space(attr_id);
    if ( dataspace_id < 0 ) {
        H5Aclose(attr_id);
        return MATIO_E_GENERIC_READ_ERROR;
    }
    *nfields = H5Sget_simple_extent_npoints(dataspace_id);
    if ( *nfields < 0 ) {
        H5Sclose(dataspace_id);
        H5Aclose(attr_id);
        return MATIO_E_GENERIC_READ_ERROR;
    }
    fieldnames_vl = (hvl_t *)calloc((size_t)(*nfields), sizeof(hvl_t));
    if ( fieldnames_vl == NULL ) {
        H5Sclose(dataspace_id);
        H5Aclose(attr_id);
        return MATIO_E_OUT_OF_MEMORY;
    }
    herr = H5Aread(attr_id, field_id, fieldnames_vl);
    if ( herr < 0 ) {
        H5Sclose(dataspace_id);
        H5Aclose(attr_id);
        return MATIO_E_GENERIC_READ_ERROR;
    }
    H5Sclose(dataspace_id);
