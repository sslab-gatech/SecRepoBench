if ( nfields > 0 ) {
    fieldnames_vl = (hvl_t *)calloc((size_t)nfields, sizeof(*fieldnames_vl));
    if ( fieldnames_vl == NULL ) {
        H5Sclose(dataspace_id);
        H5Aclose(attr_id);
        return MATIO_E_OUT_OF_MEMORY;
    }
} else {
    nfields = 1;
    fieldnames_vl = (hvl_t *)calloc((size_t)nfields, sizeof(*fieldnames_vl));
    if ( fieldnames_vl == NULL ) {
        H5Sclose(dataspace_id);
        H5Aclose(attr_id);
        return MATIO_E_OUT_OF_MEMORY;
    }
}
