    space_id = H5Aget_space(attr_id);
    herr = H5Sget_simple_extent_npoints(space_id);
    if (herr < 0) {
        H5Sclose(space_id);
        H5Aclose(attr_id);
        return MATIO_E_GENERIC_READ_ERROR;
    }
    *nfields = (hsize_t)herr;
    fieldnames_vl = (hvl_t *)calloc((size_t)(*nfields), sizeof(hvl_t));
    if (fieldnames_vl != NULL) {
        herr = H5Aread(attr_id, H5T_NATIVE_HVL, fieldnames_vl);
        if (herr < 0) {
            free(fieldnames_vl);
            fieldnames_vl = NULL;
        }
