dataspace_id = H5Aget_space(attr_id);
    if ( dataspace_id < 0 ) {
        H5Aclose(attr_id);
        return MATIO_E_GENERIC_READ_ERROR;
    }

    /* Check that the attribute is 1-dimensional */
    if ( H5Sget_simple_extent_ndims(dataspace_id) != 1 ) {
        H5Sclose(dataspace_id);
        H5Aclose(attr_id);
        return MATIO_E_GENERIC_READ_ERROR;
    }

    /* Retrieve the number of fields from the attribute dataspace */
    {
        hsize_t dims[1] = {0};
        if ( H5Sget_simple_extent_dims(dataspace_id, dims, NULL) < 0 || dims[0] < 1 ) {
            H5Sclose(dataspace_id);
            H5Aclose(attr_id);
            return MATIO_E_GENERIC_READ_ERROR;
        }
        *nfields = dims[0];
    }

    /* Allocate memory for the variable-length field names */
    fieldnames_vl = (hvl_t *)calloc((size_t)(*nfields), sizeof(*fieldnames_vl));
