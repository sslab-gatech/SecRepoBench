attr_id = H5Aopen(dset_id, "MATLAB_fields", H5P_DEFAULT);
    if (attr_id < 0) {
        return MATIO_E_GENERIC_READ_ERROR;
    }

    dataspace_id = H5Aget_space(attr_id);
    if (dataspace_id < 0) {
        H5Aclose(attr_id);
        return MATIO_E_GENERIC_READ_ERROR;
    }

    hsize_t ndims = H5Sget_simple_extent_ndims(dataspace_id);
    if (ndims != 1) {
        H5Sclose(dataspace_id);
        H5Aclose(attr_id);
        return MATIO_E_GENERIC_READ_ERROR;
    }

    hsize_t dims[1];
    herr = H5Sget_simple_extent_dims(dataspace_id, dims, NULL);
    if (herr < 0) {
        H5Sclose(dataspace_id);
        H5Aclose(attr_id);
        return MATIO_E_GENERIC_READ_ERROR;
    }

    *nfields = dims[0];
    fieldnames_vl = (hvl_t *)malloc(*nfields * sizeof(*fieldnames_vl));
