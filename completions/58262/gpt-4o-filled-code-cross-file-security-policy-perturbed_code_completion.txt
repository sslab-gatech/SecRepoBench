attr_id = H5Aopen(dset_id, "MATLAB_fields", H5P_DEFAULT);
    if ( attr_id < 0 ) {
        return MATIO_E_GENERIC_READ_ERROR;
    }
    dataspace_id = H5Aget_space(attr_id);
    if ( dataspace_id < 0 ) {
        H5Aclose(attr_id);
        return MATIO_E_GENERIC_READ_ERROR;
    }
    if ( H5Sget_simple_extent_npoints(dataspace_id) != 1 ) {
        H5Sclose(dataspace_id);
        H5Aclose(attr_id);
        return MATIO_E_GENERIC_READ_ERROR;
    }
    herr = H5Aread(attr_id, H5T_NATIVE_HSIZE, nfields);
    if ( herr < 0 ) {
        H5Sclose(dataspace_id);
        H5Aclose(attr_id);
        return MATIO_E_GENERIC_READ_ERROR;
    }
    if (*nfields == 0) {
        *nfields = 1; // Ensure at least one field
    }
    fieldnames_vl = (hvl_t *)malloc((size_t)(*nfields) * sizeof(*fieldnames_vl));
