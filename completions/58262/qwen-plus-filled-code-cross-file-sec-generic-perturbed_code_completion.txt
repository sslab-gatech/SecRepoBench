attr_id = H5Aopen(dset_id, "MATLAB_fields", H5P_DEFAULT);
if ( attr_id < 0 ) {
    return MATIO_E_ATTRIBUTE_OPEN_ERROR;
}
dataspace_id = H5Aget_space(attr_id);
if ( dataspace_id < 0 ) {
    H5Aclose(attr_id);
    return MATIO_E_DATASPACE_GET_ERROR;
}
*hsize_t_nfields = H5Sget_simple_extent_npoints(dataspace_id);
if ( *hsize_t_nfields <= 0 ) {
    H5Sclose(dataspace_id);
    H5Aclose(attr_id);
    return MATIO_E_INVALID_FIELD_COUNT;
}
fieldnames_vl = (hvl_t *)calloc((size_t)(*hsize_t_nfields), sizeof(hvl_t));
if ( fieldnames_vl == NULL ) {
    H5Sclose(dataspace_id);
    H5Aclose(attr_id);
    return MATIO_E_OUT_OF_MEMORY;
}
