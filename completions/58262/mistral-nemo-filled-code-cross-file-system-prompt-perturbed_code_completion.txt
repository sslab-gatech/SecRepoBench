// Allocate memory to store the field names as variable-length structures.
    fieldnames_vl = (hvl_t *)calloc((size_t)*nfields, sizeof(hvl_t));
    if ( fieldnames_vl == NULL ) {
        H5Sclose(space_id);
        H5Aclose(attr_id);
        return MATIO_E_OUT_OF_MEMORY;
    }
    // Read the field names from the attribute.
    herr = H5Aread(attr_id, H5T_NATIVE_CHAR, fieldnames_vl);
    if ( herr < 0 ) {
        H5Sclose(space_id);
        H5Aclose(attr_id);
        free(fieldnames_vl);
        return MATIO_E_GENERIC_READ_ERROR;
    }
