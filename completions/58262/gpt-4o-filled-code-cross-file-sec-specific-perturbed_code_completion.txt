    if ( attr_id < 0 ) {
        return MATIO_E_GENERIC_READ_ERROR;
    }
    dataspace_id = H5Aget_space(attr_id);
    if ( dataspace_id < 0 ) {
        H5Aclose(attr_id);
        return MATIO_E_GENERIC_READ_ERROR;
    }
    *nfields = (hsize_t)H5Sget_simple_extent_npoints(dataspace_id);
    if ( *nfields == 0 ) {
        H5Sclose(dataspace_id);
        H5Aclose(attr_id);
        return MATIO_E_GENERIC_READ_ERROR;
    }
