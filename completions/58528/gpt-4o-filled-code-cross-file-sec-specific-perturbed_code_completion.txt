for (i = 0; i < 3; i++) {
            int nb_refs = s->rps[cand_lists[i]].nb_refs;
            for (refindex = 0; refindex < nb_refs; refindex++) {
                if (rpl_tmp.nb_refs >= HEVC_MAX_REFS)
                    break;

                rpl_tmp.list[rpl_tmp.nb_refs] = s->rps[cand_lists[i]].list[refindex];
                rpl_tmp.ref[rpl_tmp.nb_refs] = s->rps[cand_lists[i]].ref[refindex];
                rpl_tmp.isLongTerm[rpl_tmp.nb_refs] = (cand_lists[i] == LT_CURR);
                rpl_tmp.nb_refs++;
            }
        }

        if (s->ps.pps->pps_curr_pic_ref_enabled_flag && rpl_tmp.nb_refs < HEVC_MAX_REFS) {
            rpl_tmp.list[rpl_tmp.nb_refs] = s->ref->poc;
            rpl_tmp.ref[rpl_tmp.nb_refs] = s->ref;
            rpl_tmp.isLongTerm[rpl_tmp.nb_refs] = 0;
            rpl_tmp.nb_refs++;
        }
