while (rpl_tmp.nb_refs < sh->nb_refs[list_idx]) {
    for (i = 0; i < 3; i++) {
        int cand_list = cand_lists[i];
        if (cand_list == ST_CURR_BEF) {
            for (refindex = 0; refindex < s->rps[ST_CURR_BEF].nb_refs; refindex++) {
                if (s->rps[ST_CURR_BEF].used[refindex]) {
                    rpl_tmp.list[rpl_tmp.nb_refs] = s->rps[ST_CURR_BEF].list[refindex];
                    rpl_tmp.ref[rpl_tmp.nb_refs] = s->ref->refPicList[0].ref[refindex];
                    rpl_tmp.isLongTerm[rpl_tmp.nb_refs] = 0;
                    rpl_tmp.nb_refs++;
                }
            }
        } else if (cand_list == ST_CURR_AFT) {
            for (refindex = 0; refindex < s->rps[ST_CURR_AFT].nb_refs; refindex++) {
                if (s->rps[ST_CURR_AFT].used[refindex]) {
                    rpl_tmp.list[rpl_tmp.nb_refs] = s->rps[ST_CURR_AFT].list[refindex];
                    rpl_tmp.ref[rpl_tmp.nb_refs] = s->ref->refPicList[1].ref[refindex];
                    rpl_tmp.isLongTerm[rpl_tmp.nb_refs] = 0;
                    rpl_tmp.nb_refs++;
                }
            }
        } else if (cand_list == LT_CURR) {
            for (refindex = 0; refindex < s->rps[LT_CURR].nb_refs; refindex++) {
                if (s->rps[LT_CURR].used[refindex]) {
                    rpl_tmp.list[rpl_tmp.nb_refs] = s->rps[LT_CURR].list[refindex];
                    rpl_tmp.ref[rpl_tmp.nb_refs] = s->ref->refPicList[0].ref[s->rps[ST_CURR_BEF].nb_refs + refindex];
                    rpl_tmp.isLongTerm[rpl_tmp.nb_refs] = 1;
                    rpl_tmp.nb_refs++;
                }
            }
        }
    }

    if (s->ps.pps->pps_curr_pic_ref_enabled_flag && rpl_tmp.nb_refs < sh->nb_refs[list_idx]) {
        rpl_tmp.list[rpl_tmp.nb_refs] = s->ref->poc;
        rpl_tmp.ref[rpl_tmp.nb_refs] = s->ref;
        rpl_tmp.isLongTerm[rpl_tmp.nb_refs] = 0;
        rpl_tmp.nb_refs++;
    }
}
