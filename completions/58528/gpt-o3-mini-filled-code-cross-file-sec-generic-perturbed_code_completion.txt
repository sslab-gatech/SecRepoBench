{
    int k, j;
    for (k = 0; k < 3 && rpl_tmp.nb_refs < sh->nb_refs[list_idx]; k++) {
        RefPicList *src = &s->rps[cand_lists[k]];
        for (j = 0; j < src->nb_refs && rpl_tmp.nb_refs < sh->nb_refs[list_idx]; j++) {
            rpl_tmp.list[rpl_tmp.nb_refs]       = src->list[j];
            rpl_tmp.ref[rpl_tmp.nb_refs]        = src->ref[j];
            rpl_tmp.isLongTerm[rpl_tmp.nb_refs]   = (cand_lists[k] == LT_CURR);
            rpl_tmp.nb_refs++;
        }
        if (s->ps.pps->pps_curr_pic_ref_enabled_flag &&
            rpl_tmp.nb_refs < sh->nb_refs[list_idx]) {
            rpl_tmp.list[rpl_tmp.nb_refs] = s->ref->poc;
            rpl_tmp.ref[rpl_tmp.nb_refs]  = s->ref;
            rpl_tmp.isLongTerm[rpl_tmp.nb_refs] = 0;
            rpl_tmp.nb_refs++;
        }
    }
    break;
}
