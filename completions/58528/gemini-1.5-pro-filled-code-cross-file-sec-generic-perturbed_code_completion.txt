for (i = 0; i < 3 && rpl_tmp.nb_refs < sh->nb_refs[list_idx]; i++) {
    int cand_list = cand_lists[i];
    int j;

    for (j = 0; j < s->rps[cand_list].nb_refs && rpl_tmp.nb_refs < sh->nb_refs[list_idx]; j++) {
        if (s->rps[cand_list].used[j]) {
            rpl_tmp.list[rpl_tmp.nb_refs]       = s->rps[cand_list].poc[j];
            rpl_tmp.ref[rpl_tmp.nb_refs]        = s->rps[cand_list].ref[j];
            rpl_tmp.isLongTerm[rpl_tmp.nb_refs] = s->rps[cand_list].is_long_term[j];
            rpl_tmp.nb_refs++;
        }
    }
}

if (s->ps.pps->pps_curr_pic_ref_enabled_flag && rpl_tmp.nb_refs < sh->nb_refs[list_idx]) {
    rpl_tmp.list[rpl_tmp.nb_refs]       = s->poc;
    rpl_tmp.ref[rpl_tmp.nb_refs]        = s->ref;
    rpl_tmp.isLongTerm[rpl_tmp.nb_refs] = 0;
    rpl_tmp.nb_refs++;
}
