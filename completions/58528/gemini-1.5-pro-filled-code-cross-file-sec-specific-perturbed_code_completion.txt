for (i = 0; i < 3 && rpl_tmp.nb_refs < sh->nb_refs[list_idx]; i++) {
    int j, list = cand_lists[i];
    for (j = 0; j < s->rps[list].nb_refs && rpl_tmp.nb_refs < sh->nb_refs[list_idx]; j++) {
        if (s->rps[list].used[j]) {
            rpl_tmp.list[rpl_tmp.nb_refs]       = s->rps[list].poc[j];
            rpl_tmp.ref[rpl_tmp.nb_refs]        = s->rps[list].ref[j];
            rpl_tmp.isLongTerm[rpl_tmp.nb_refs] = s->rps[list].isLongTerm[j];
            rpl_tmp.nb_refs++;
        }
    }
}

if (s->ps.pps->pps_curr_pic_ref_enabled_flag && rpl_tmp.nb_refs < sh->nb_refs[list_idx]) {
    rpl_tmp.list[rpl_tmp.nb_refs] = s->poc;
    rpl_tmp.ref[rpl_tmp.nb_refs]  = s->ref;
    rpl_tmp.nb_refs++;
}
