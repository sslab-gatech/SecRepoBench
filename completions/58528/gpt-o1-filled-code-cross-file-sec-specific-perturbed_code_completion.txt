for (i = 0; i < 3 && rpl_tmp.nb_refs < sh->nb_refs[list_idx]; i++) {
    int cand_idx = cand_lists[i];
    for (refindex = 0; refindex < s->rps[cand_idx].nb_refs && rpl_tmp.nb_refs < sh->nb_refs[list_idx]; refindex++) {
        rpl_tmp.list[rpl_tmp.nb_refs]       = s->rps[cand_idx].list[refindex];
        rpl_tmp.ref[rpl_tmp.nb_refs]        = s->rps[cand_idx].ref[refindex];
        rpl_tmp.isLongTerm[rpl_tmp.nb_refs] = s->rps[cand_idx].isLongTerm[refindex];
        rpl_tmp.nb_refs++;
    }
}

if (s->ps.pps->pps_curr_pic_ref_enabled_flag && rpl_tmp.nb_refs < sh->nb_refs[list_idx]) {
    rpl_tmp.list[rpl_tmp.nb_refs]       = s->ref->poc;
    rpl_tmp.ref[rpl_tmp.nb_refs]        = s->ref;
    rpl_tmp.isLongTerm[rpl_tmp.nb_refs] = 0;
    rpl_tmp.nb_refs++;
}

break;
