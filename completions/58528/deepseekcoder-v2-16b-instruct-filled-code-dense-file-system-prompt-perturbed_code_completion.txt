// Iterate over the candidate reference picture lists (ST_CURR_BEF, ST_CURR_AFT, LT_CURR).
            // For each candidate list, append its references to a temporary reference picture list.
            // If the current picture reference is enabled by the PPS, add the current picture as a reference.
            for (i = 0; i < 3; i++) {
                int list = cand_lists[i];
                for (int j = 0; j < s->rps[list].nb_refs; j++) {
                    int poc = s->rps[list].poc[j];
                    if (s->ps.pps->pps_curr_pic_ref_enabled_flag && j == 0) {
                        poc = s->ref->poc;
                    }
                    if (rpl_tmp.nb_refs < sh->nb_refs[list]) {
                        rpl_tmp.list[rpl_tmp.nb_refs] = poc;
                        rpl_tmp.ref[rpl_tmp.nb_refs] = s->rps[list].ref[j];
                        rpl_tmp.isLongTerm[rpl_tmp.nb_refs] = list == LT_CURR;
                        rpl_tmp.nb_refs++;
                    }
                }
            }
