for (i = 0; i < 3 && rpl_tmp.nb_refs < sh->nb_refs[list_idx]; i++) {
    for (refindex = 0; refindex < s->rps[cand_lists[i]].nb_refs && rpl_tmp.nb_refs < sh->nb_refs[list_idx]; refindex++) {
        rpl_tmp.list[rpl_tmp.nb_refs] = s->rps[cand_lists[i]].list[refindex];
        rpl_tmp.ref[rpl_tmp.nb_refs] = s->rps[cand_lists[i]].ref[refindex];
        rpl_tmp.isLongTerm[rpl_tmp.nb_refs] = (cand_lists[i] == LT_CURR);
        rpl_tmp.nb_refs++;
    }
}

// If enabled by PPS, add the current picture as a reference
if (s->ps.pps->pps_curr_pic_ref_enabled_flag && rpl_tmp.nb_refs < sh->nb_refs[list_idx]) {
    rpl_tmp.list[rpl_tmp.nb_refs] = s->ref->poc;
    rpl_tmp.ref[rpl_tmp.nb_refs] = s->ref;
    rpl_tmp.isLongTerm[rpl_tmp.nb_refs] = 0; // Current picture is not a long-term reference
    rpl_tmp.nb_refs++;
}
