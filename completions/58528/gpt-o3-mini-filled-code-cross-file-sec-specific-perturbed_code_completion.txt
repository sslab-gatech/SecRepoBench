{
    int refs_before = rpl_tmp.nb_refs;
    for (int j = 0; j < 3 && rpl_tmp.nb_refs < sh->nb_refs[list_idx]; j++) {
        int cand = cand_lists[j];
        for (int k = 0; k < s->rps[cand].nb_refs && rpl_tmp.nb_refs < sh->nb_refs[list_idx]; k++) {
            /* Append reference from candidate list cand */
            rpl_tmp.list[rpl_tmp.nb_refs]       = s->rps[cand].poc[k];
            rpl_tmp.ref[rpl_tmp.nb_refs]        = s->rps[cand].ref[k];
            rpl_tmp.isLongTerm[rpl_tmp.nb_refs]   = (cand == LT_CURR);
            rpl_tmp.nb_refs++;
        }
    }
    if (s->ps.pps->pps_curr_pic_ref_enabled_flag &&
        rpl_tmp.nb_refs < sh->nb_refs[list_idx]) {
        /* Append current picture as a reference */
        rpl_tmp.list[rpl_tmp.nb_refs]       = s->ref->poc;
        rpl_tmp.ref[rpl_tmp.nb_refs]        = s->ref;
        rpl_tmp.isLongTerm[rpl_tmp.nb_refs]   = 0;
        rpl_tmp.nb_refs++;
    }
    if (refs_before == rpl_tmp.nb_refs)
        break;
}
