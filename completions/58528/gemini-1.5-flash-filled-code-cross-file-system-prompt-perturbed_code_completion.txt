                RefPicList *cand_rpl = &s->rps[cand_lists[cand_list_idx]];
                for (refindex = 0; refindex < cand_rpl->nb_refs; refindex++) {
                    if (rpl_tmp.nb_refs >= sh->nb_refs[list_idx])
                        break;
                    rpl_tmp.list[rpl_tmp.nb_refs]       = cand_rpl->list[refindex];
                    rpl_tmp.ref[rpl_tmp.nb_refs]        = cand_rpl->ref[refindex];
                    rpl_tmp.isLongTerm[rpl_tmp.nb_refs] = cand_rpl->isLongTerm[refindex];
                    rpl_tmp.nb_refs++;
                }
            }
            if (s->ps.pps->pps_curr_pic_ref_enabled_flag) {
                if (rpl_tmp.nb_refs < sh->nb_refs[list_idx]) {
                    rpl_tmp.list[rpl_tmp.nb_refs]       = s->ref->poc;
                    rpl_tmp.ref[rpl_tmp.nb_refs]        = s->ref;
                    rpl_tmp.isLongTerm[rpl_tmp.nb_refs] = 0;
                    rpl_tmp.nb_refs++;
                }
