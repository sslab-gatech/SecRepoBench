{
    int appended = 0;
    for (i = 0; i < 3 && rpl_tmp.nb_refs < sh->nb_refs[list_idx]; i++) {
        int num_refs = s->rps[cand_lists[i]].nb_refs;
        for (refindex = 0; refindex < num_refs && rpl_tmp.nb_refs < sh->nb_refs[list_idx]; refindex++) {
            rpl_tmp.list[rpl_tmp.nb_refs]       = s->rps[cand_lists[i]].poc[refindex];
            rpl_tmp.ref[rpl_tmp.nb_refs]        = s->rps[cand_lists[i]].ref[refindex];
            rpl_tmp.isLongTerm[rpl_tmp.nb_refs] = (cand_lists[i] == LT_CURR);
            rpl_tmp.nb_refs++;
            appended = 1;
        }
    }
    if (!appended && s->ps.pps->pps_curr_pic_ref_enabled_flag &&
        rpl_tmp.nb_refs < sh->nb_refs[list_idx]) {
        rpl_tmp.list[rpl_tmp.nb_refs]       = s->ref->poc;
        rpl_tmp.ref[rpl_tmp.nb_refs]        = s->ref;
        rpl_tmp.isLongTerm[rpl_tmp.nb_refs] = 0;
        rpl_tmp.nb_refs++;
    }
    if (!appended)
        break;
}
