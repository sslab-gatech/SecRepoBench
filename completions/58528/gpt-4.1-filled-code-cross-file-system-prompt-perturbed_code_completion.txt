for (i = 0; i < 3 && rpl_tmp.nb_refs < sh->nb_refs[list_idx]; i++) {
    int cand_list = cand_lists[i];
    for (refindex = 0; refindex < s->rps[cand_list].nb_refs && rpl_tmp.nb_refs < sh->nb_refs[list_idx]; refindex++) {
        rpl_tmp.list[rpl_tmp.nb_refs]       = s->rps[cand_list].poc[refindex];
        rpl_tmp.ref[rpl_tmp.nb_refs]        = s->DPB[s->rps[cand_list].ref[refindex]];
        rpl_tmp.isLongTerm[rpl_tmp.nb_refs] = (cand_list == LT_CURR);
        rpl_tmp.nb_refs++;
    }
    // If PPS enables current picture as reference, add it as last entry (only once)
    if (s->ps.pps->pps_curr_pic_ref_enabled_flag &&
        rpl_tmp.nb_refs < sh->nb_refs[list_idx] &&
        i == 2) {
        rpl_tmp.list[rpl_tmp.nb_refs]       = s->ref->poc;
        rpl_tmp.ref[rpl_tmp.nb_refs]        = s->ref;
        rpl_tmp.isLongTerm[rpl_tmp.nb_refs] = 0;
        rpl_tmp.nb_refs++;
    }
}
