for (i = 0; i < 3 && rpl_tmp.nb_refs < sh->nb_refs[list_idx]; i++) {
    int list_idx_rps = cand_lists[i];
    RefPicList *rpl_rps = &s->rps[list_idx_rps];

    for (refindex = 0; refindex < rpl_rps->nb_refs && rpl_tmp.nb_refs < sh->nb_refs[list_idx]; refindex++) {
        rpl_tmp.list[rpl_tmp.nb_refs]       = rpl_rps->poc[refindex];
        rpl_tmp.ref[rpl_tmp.nb_refs]        = rpl_rps->ref[refindex];
        rpl_tmp.isLongTerm[rpl_tmp.nb_refs] = rpl_rps->isLongTerm[refindex];
        rpl_tmp.nb_refs++;
    }
}

if (s->ps.pps->pps_curr_pic_ref_enabled_flag && rpl_tmp.nb_refs < sh->nb_refs[list_idx]) {
    rpl_tmp.list[rpl_tmp.nb_refs] = s->poc;
    rpl_tmp.ref[rpl_tmp.nb_refs]  = s->ref;
    rpl_tmp.nb_refs++;
}
