for (int j = 0; j < 3; j++) {
                int list_idx_cur = cand_lists[j];
                RefPicList *rps = &s->rps[list_idx_cur];
                for (i = 0; i < rps->nb_refs; i++) {
                    if (rpl_tmp.nb_refs < HEVC_MAX_REFS) {
                        rpl_tmp.list[rpl_tmp.nb_refs] = rps->list[i];
                        rpl_tmp.ref[rpl_tmp.nb_refs] = rps->ref[i];
                        rpl_tmp.isLongTerm[rpl_tmp.nb_refs] = rps->isLongTerm[i];
                        rpl_tmp.nb_refs++;
                    }
                }
            }
            if (s->ps.pps->pps_curr_pic_ref_enabled_flag && rpl_tmp.nb_refs < HEVC_MAX_REFS) {
                rpl_tmp.list[rpl_tmp.nb_refs] = s->ref->poc;
                rpl_tmp.ref[rpl_tmp.nb_refs] = s->ref;
                rpl_tmp.isLongTerm[rpl_tmp.nb_refs] = 0;
                rpl_tmp.nb_refs++;
            }
