for (int cand_list_idx = 0; cand_list_idx < 3; cand_list_idx++) {
                int list_idx_to_use = cand_lists[cand_list_idx];
                const RefPicSet *rps = &s->rps[list_idx_to_use];
                for (i = 0; i < rps->nb_refs; i++) {
                    if (rps->used[i]) {
                        rpl_tmp.list[rpl_tmp.nb_refs] = rps->poc[i];
                        rpl_tmp.ref[rpl_tmp.nb_refs]  = rps->ref[i];
                        rpl_tmp.isLongTerm[rpl_tmp.nb_refs] = rps->isLongTerm[i];
                        rpl_tmp.nb_refs++;
                    }
                }
            }
            if (s->ps.pps->pps_curr_pic_ref_enabled_flag) {
                rpl_tmp.list[rpl_tmp.nb_refs] = s->ref->poc;
                rpl_tmp.ref[rpl_tmp.nb_refs]  = s->ref;
                rpl_tmp.isLongTerm[rpl_tmp.nb_refs] = 0;
                rpl_tmp.nb_refs++;
            }
