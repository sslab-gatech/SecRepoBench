for (i = 0; i < 3; i++) {
                int idx = s->rps[cand_lists[i]].nb_refs;
                if (idx > 0) {
                    for (refindex = 0; refindex < idx; refindex++) {
                        if (rpl_tmp.nb_refs < sh->nb_refs[list_idx]) {
                            rpl_tmp.list[rpl_tmp.nb_refs] = s->rps[cand_lists[i]].list[refindex];
                            rpl_tmp.ref[rpl_tmp.nb_refs] = s->rps[cand_lists[i]].ref[refindex];
                            rpl_tmp.isLongTerm[rpl_tmp.nb_refs] = s->rps[cand_lists[i]].isLongTerm[refindex];
                            rpl_tmp.nb_refs++;
                        }
                    }
                }
            }

            if (s->ps.pps->pps_curr_pic_ref_enabled_flag) {
                if (rpl_tmp.nb_refs < sh->nb_refs[list_idx]) {
                    rpl_tmp.list[rpl_tmp.nb_refs] = s->ref->poc;
                    rpl_tmp.ref[rpl_tmp.nb_refs] = s->ref;
                    rpl_tmp.isLongTerm[rpl_tmp.nb_refs] = 0; // Assuming current picture is not long term
