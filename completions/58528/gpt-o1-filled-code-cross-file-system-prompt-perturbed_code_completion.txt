for (i = 0; i < 3; i++) {
    int cand = cand_lists[i];
    int j;
    for (j = 0; j < s->rps[cand].nb_refs && rpl_tmp.nb_refs < sh->nb_refs[list_idx]; j++) {
        rpl_tmp.list[rpl_tmp.nb_refs]       = s->rps[cand].poc[j];
        rpl_tmp.ref[rpl_tmp.nb_refs]        = s->rps[cand].ref[j];
        rpl_tmp.isLongTerm[rpl_tmp.nb_refs] = (cand == LT_CURR);
        rpl_tmp.nb_refs++;
    }
    if (rpl_tmp.nb_refs >= sh->nb_refs[list_idx])
        break;
}

if (s->ps.pps->pps_curr_pic_ref_enabled_flag &&
    rpl_tmp.nb_refs < sh->nb_refs[list_idx]) {
    rpl_tmp.list[rpl_tmp.nb_refs]       = s->ref->poc;
    rpl_tmp.ref[rpl_tmp.nb_refs]        = s->ref;
    rpl_tmp.isLongTerm[rpl_tmp.nb_refs] = 0;
    rpl_tmp.nb_refs++;
}
