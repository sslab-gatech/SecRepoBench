while (rpl_tmp.nb_refs < sh->nb_refs[list_idx]) {
    for (refindex = 0; refindex < 3; refindex++) {
        int cand_list = cand_lists[refindex];
        if (s->rps[cand_list].nb_refs > 0) {
            rpl_tmp.list[rpl_tmp.nb_refs] = s->ref->poc[s->rps[cand_list].list[0]];
            rpl_tmp.ref[rpl_tmp.nb_refs]  = s->ref;

            rpl_tmp.nb_refs++;
        }
    }
    if (s->ps.pps->pps_curr_pic_ref_enabled_flag && rpl_tmp.nb_refs < sh->nb_refs[list_idx]) {
        rpl_tmp.list[rpl_tmp.nb_refs] = s->ref->poc;
        rpl_tmp.ref[rpl_tmp.nb_refs]  = s->ref;
        rpl_tmp.isLongTerm[rpl_tmp.nb_refs] = 0;
        rpl_tmp.nb_refs++;
    }
