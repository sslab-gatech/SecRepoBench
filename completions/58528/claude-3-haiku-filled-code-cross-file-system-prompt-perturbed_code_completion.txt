while (rpl_tmp.nb_refs < sh->nb_refs[list_idx]) {
    for (i = 0; i < 3; i++) {
        int cand_list = cand_lists[i];
        int nb_refs = s->rps[cand_list].nb_refs;
        for (refindex = 0; refindex < nb_refs; refindex++) {
            if (s->rps[cand_list].used[refindex]) {
                rpl_tmp.list[rpl_tmp.nb_refs] = s->poc + s->rps[cand_list].delta_poc[refindex];
                rpl_tmp.ref[rpl_tmp.nb_refs] = s->ref->refPicList[cand_list].ref[refindex];
                rpl_tmp.isLongTerm[rpl_tmp.nb_refs] = s->rps[cand_list].is_long_term[refindex];
                rpl_tmp.nb_refs++;
            }
        }
    }
    if (s->ps.pps->pps_curr_pic_ref_enabled_flag) {
        rpl_tmp.list[rpl_tmp.nb_refs] = s->poc;
        rpl_tmp.ref[rpl_tmp.nb_refs] = s->ref;
        rpl_tmp.isLongTerm[rpl_tmp.nb_refs] = 0;
        rpl_tmp.nb_refs++;
    }
}
