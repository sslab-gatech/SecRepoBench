{
    int cand, j;

    /* Iterate over candidate reference picture lists (ST_CURR_BEF, ST_CURR_AFT, LT_CURR) */
    for (cand = 0; cand < 3 && rpl_tmp.nb_refs < sh->nb_refs[list_idx]; cand++) {
        int cList = cand_lists[cand];
        for (j = 0; j < s->rps[cList].nb_refs && rpl_tmp.nb_refs < sh->nb_refs[list_idx]; j++) {
            rpl_tmp.list[rpl_tmp.nb_refs]       = s->rps[cList].list[j];
            rpl_tmp.ref[rpl_tmp.nb_refs]        = s->rps[cList].ref[j];
            rpl_tmp.isLongTerm[rpl_tmp.nb_refs] = (cList == LT_CURR);
            rpl_tmp.nb_refs++;
        }
    }

    /* If the current picture can be used as a reference, add it if we still haven't reached the needed count */
    if (s->ps.pps->pps_curr_pic_ref_enabled_flag && rpl_tmp.nb_refs < sh->nb_refs[list_idx]) {
        rpl_tmp.list[rpl_tmp.nb_refs]       = s->ref->poc;
        rpl_tmp.ref[rpl_tmp.nb_refs]        = s->ref;
        rpl_tmp.isLongTerm[rpl_tmp.nb_refs] = 0;
        rpl_tmp.nb_refs++;
    }
}
