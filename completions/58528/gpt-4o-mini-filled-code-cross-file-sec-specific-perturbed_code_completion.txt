for (i = 0; i < 3; i++) {
                refindex = s->rps[cand_lists[i]].nb_refs;
                if (refindex > 0) {
                    for (int j = 0; j < refindex; j++) {
                        if (rpl_tmp.nb_refs < sh->nb_refs[list_idx]) {
                            rpl_tmp.list[rpl_tmp.nb_refs] = s->rps[cand_lists[i]].list[j];
                            rpl_tmp.ref[rpl_tmp.nb_refs] = s->rps[cand_lists[i]].ref[j];
                            rpl_tmp.isLongTerm[rpl_tmp.nb_refs] = s->rps[cand_lists[i]].isLongTerm[j];
                            rpl_tmp.nb_refs++;
                        }
                    }
                }
            }
            if (s->ps.pps->pps_curr_pic_ref_enabled_flag && rpl_tmp.nb_refs < sh->nb_refs[list_idx]) {
                rpl_tmp.list[rpl_tmp.nb_refs] = s->ref->poc;
                rpl_tmp.ref[rpl_tmp.nb_refs] = s->ref;
                rpl_tmp.isLongTerm[rpl_tmp.nb_refs] = 0; // Assuming current picture is not long term
