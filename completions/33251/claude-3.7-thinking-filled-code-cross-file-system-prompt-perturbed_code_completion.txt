// Check if the offset is valid and we have enough space to read the marker
if (offset + 1 > header_len) {
  return BLOSC2_ERROR_READ_BUFFER;
}

// Check if the content marker has the expected value (msgpack binary marker)
if (*content_marker != 0xc6) {
  return BLOSC2_ERROR_DATA;
}

// Extract the content length from the bytes following the marker
int32_t content_len;
if (offset + 1 + 4 > header_len) {
  return BLOSC2_ERROR_READ_BUFFER;
}
from_big(&content_len, content_marker + 1, sizeof(content_len));

// Check if the content length is valid and doesn't exceed the header
if (content_len < 0 || offset + 1 + 4 + content_len > header_len) {
  return BLOSC2_ERROR_DATA;
}

// Store the length in the metalayer structure
current_metalayer->content_len = content_len;
