if ((header_len - offset) < (1 + 4)) {
  return BLOSC2_ERROR_READ_BUFFER;
}
if (content_marker[0] != 0xd3u) {  // expected marker for metalayer content
  return BLOSC2_ERROR_DATA;
}
int32_t content_len;
from_big(&content_len, content_marker + 1, sizeof(content_len));
if (content_len < 0 || header_len < offset + 1 + 4 + content_len) {
  return BLOSC2_ERROR_DATA;
}
current_metalayer->content_len = content_len;
