// Check the content marker
header_pos += 1;
if (header_len < header_pos) {
  return BLOSC2_ERROR_READ_BUFFER;
}
if (*content_marker != 0xc6) {   // sanity check
  return BLOSC2_ERROR_DATA;
}
// Retrieve the length of the serialized content that follows the marker
int32_t content_len;
header_pos += sizeof(content_len);
if (header_len < header_pos) {
  return BLOSC2_ERROR_READ_BUFFER;
}
from_big(&content_len, content_marker + 1, sizeof(content_len));
// Check content_len is valid
if (content_len < 0) {
  return BLOSC2_ERROR_DATA;
}
// Check if the content fits within the header
int64_t content_offset = (content_marker + 1 + 4) - header;
if (content_offset < 0 || content_offset + content_len > header_len) {
  return BLOSC2_ERROR_DATA;
}
current_metalayer->content_len = content_len;
