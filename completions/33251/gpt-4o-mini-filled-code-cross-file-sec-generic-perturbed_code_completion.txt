    return BLOSC2_ERROR_READ_BUFFER;
}
int32_t content_len;
from_big(&content_len, content_marker + 1, sizeof(content_len));
if (content_len < 0 || content_len > header_len - offset) {
    return BLOSC2_ERROR_DATA;
}

if (content == NULL) {
    return BLOSC2_ERROR_MEMORY; // Check for allocation failure
}
