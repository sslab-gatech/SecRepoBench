if (*content_marker != 0xc6) {  // sanity check
  return BLOSC2_ERROR_DATA;
}
int32_t content_len;
if (offset + 1 + sizeof(content_len) > header_len) {
  return BLOSC2_ERROR_READ_BUFFER;
}
from_big(&content_len, content_marker + 1, sizeof(content_len));
if (content_len < 0) {
  return BLOSC2_ERROR_DATA;
}
// Make sure the content is within bounds of the header
if (offset + 1 + sizeof(content_len) + content_len > header_len) {
  return BLOSC2_ERROR_READ_BUFFER;
}
current_metalayer->content_len = content_len;
