// Handle DWARF attribute forms that indicate offsets to other entries within
    // the same compilation unit. Specifically, handle DW_FORM_ref1, DW_FORM_ref2,
    // and DW_FORM_ref4.
    case DW_FORM_ref1:
        READ_UNALIGNED_CK(dbg, offset, Dwarf_Unsigned,
            attribute->ar_debug_ptr, DWARF_8BIT_SIZE,
            error,section_end);
        goto fixoffset;

    case DW_FORM_ref2:
        READ_UNALIGNED_CK(dbg, offset, Dwarf_Unsigned,
            attribute->ar_debug_ptr, DWARF_16BIT_SIZE,
            error,section_end);
        goto fixoffset;

    case DW_FORM_ref4:
        READ_UNALIGNED_CK(dbg, offset, Dwarf_Unsigned,
            attribute->ar_debug_ptr, DWARF_32BIT_SIZE,
            error,section_end);
        goto fixoffset;
