// <MASK>
if (*p == '.') {
  if (!pos) {  // dots are not metacharacters in groups: [.]
    p = nextchar(p);
    // skip the next character
    ++st;
    while ((opts & aeUTF8) && st < str.size() && (str[st] & 0xc0) == 0x80)
      ++st;
    if (st == str.size() && p)
      return 0;  // word <= condition
    break;
  }
}
