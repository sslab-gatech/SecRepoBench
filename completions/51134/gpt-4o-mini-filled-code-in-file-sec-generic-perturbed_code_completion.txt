if (*st == *p) {
          p = nextchar(p);
          st--;
        } else if (pos) {  // group
          p = nextchar(p);
        } else {
          return 0;
        }
        if (st < beg && p && *p != ']')
          return 0;      // word <= condition
        break;
      }
    }
    if (!p)
      return 1;
