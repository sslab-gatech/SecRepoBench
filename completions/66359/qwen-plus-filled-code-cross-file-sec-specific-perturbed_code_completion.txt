u8 *frame;
frame = memchr(decoderconfig + i, 0x00, vosh_size - i);
if (frame && (frame - decoderconfig > 3)) {
    if (*(frame-1) == 0x00 && *(frame-2) == 0x00 && *(frame+1) == 0x01) {
        u8 *divx_str = memchr(frame+2, 'D', vosh_size - (frame - decoderconfig));
        if (divx_str && divx_str[1] == 'i' && divx_str[2] == 'v' && divx_str[3] == 'X') {
            ctx->is_packed = GF_TRUE;
        }
    }
}
