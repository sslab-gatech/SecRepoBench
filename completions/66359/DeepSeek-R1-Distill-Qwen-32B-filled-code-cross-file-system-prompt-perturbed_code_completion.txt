while (i < vosh_size - 7) {
    // Search for MPEG start code (0x00 0x00 0x01) followed by "DivX"
    if (decoderconfig[i] == 0x00 && decoderconfig[i+1] == 0x00 && decoderconfig[i+2] == 0x01 &&
        decoderconfig[i+3] == 'D' && decoderconfig[i+4] == 'i' && decoderconfig[i+5] == 'v' && decoderconfig[i+6] == 'X') {
        ctx->is_packed = GF_TRUE;
        break;
    }
    i++;
}
