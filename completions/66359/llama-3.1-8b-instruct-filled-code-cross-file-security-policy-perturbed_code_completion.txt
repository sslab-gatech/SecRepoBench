while (1) {
	// The code iterates through a buffer to search for the occurrence of
	// a specific byte pattern. It checks for the MPEG start code pattern (0x00 0x00 0x01)
	// followed by the string "DivX". The loop continues until either the end of the buffer
	// is reached or the pattern is found.
	char *start_code = memchr(decoderconfig + i + 4, 0x00, vosh_size - i - 4);
	if (start_code) {
		char *divx = memchr(start_code + 1, 'D', vosh_size - (start_code - decoderconfig) - 4);
		if (divx && memchr(divx + 1, 'i', vosh_size - (divx - start_code) - 1) && memchr(divx + 2, 'v', vosh_size - (divx - start_code) - 2) && memchr(divx + 3, 'X', vosh_size - (divx - start_code) - 3)) {
			ctx->forced_packed = GF_TRUE;
			start_code[0] = 'n';
		}
	}
	break;
}
