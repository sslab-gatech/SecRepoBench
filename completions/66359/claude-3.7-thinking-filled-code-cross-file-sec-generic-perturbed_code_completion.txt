while (i <= vosh_size - 7) { // Need at least 7 bytes: 3 for start code + 4 for "DivX"
    // Check for start code pattern (0x00 0x00 0x01)
    if (decoderconfig[i] == 0 && decoderconfig[i+1] == 0 && decoderconfig[i+2] == 1) {
        // Check if followed by "DivX"
        if (decoderconfig[i+3] == 'D' && 
            decoderconfig[i+4] == 'i' && 
            decoderconfig[i+5] == 'v' && 
            decoderconfig[i+6] == 'X') {
            // Found the pattern, i is already pointing to the start code
            break;
        }
    }
    i++;
}
