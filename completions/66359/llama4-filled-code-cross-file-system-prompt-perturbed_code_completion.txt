char *frame = NULL;
while (i < vosh_size - 4) {
    frame = memchr(decoderconfig + i, 0x00, vosh_size - i - 4);
    if (frame) {
        if (*(frame + 1) == 0x00 && *(frame + 2) == 0x01 && strncmp((char *)frame + 3, "DivX", 4) == 0) {
            ctx->forced_packed = GF_TRUE;
            break;
        }
        i = (u32)(frame - decoderconfig) + 1;
    } else {
        break;
    }
}
