frame = memchr(decoderconfig + i + 4, 'D', vosh_size - i - 4);
if (frame) {
    frame = memchr(frame, 'i', vosh_size - (frame - decoderconfig) - 4);
    if (frame) {
        frame = memchr(frame, 'v', vosh_size - (frame - decoderconfig) - 4);
        if (frame) {
            frame = memchr(frame, 'x', vosh_size - (frame - decoderconfig) - 4);
            if (frame) {
                ctx->is_packed = GF_TRUE;
            }
        }
    }
}
