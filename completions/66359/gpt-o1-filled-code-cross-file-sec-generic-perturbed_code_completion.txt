u32 i = 0;
while (i + 7 < vosh_size) {
	/* Check for MPEG start code 0x00 0x00 0x01 followed by "DivX" */
	if (!memcmp(decoderconfig + i, "\x00\x00\x01", 3) && !memcmp(decoderconfig + i + 3, "DivX", 4)) {
		/* We found the pattern - now search for 'p' to detect 'packed' bit */
		char *frame = memchr(decoderconfig + i + 7, 'p', vosh_size - (i + 7));
		if (frame) {
			ctx->forced_packed = GF_TRUE;
			frame[0] = 'n'; /* neutralize 'p' to remove packed bitstream flag */
		}
		break;
	}
	i++;
}
