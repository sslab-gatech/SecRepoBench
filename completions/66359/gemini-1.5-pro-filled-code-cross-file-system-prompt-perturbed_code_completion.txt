if (i+4 >= vosh_size) break;
			if ((decoderconfig[i] != 0) || (decoderconfig[i+1] != 0) || (decoderconfig[i+2] != 1)) {
				i++;
				continue;
			}
			if (i+8 >= vosh_size) break;

			if (!strncmp(decoderconfig + i + 4, "DivX", 4)) {
				char *frame;
				ctx->is_packed = GF_TRUE;
				frame = memchr(decoderconfig + i + 4, 'p', vosh_size - i - 4);
				if (frame) {
					ctx->forced_packed = GF_TRUE;
					frame[0] = 'n';
				}
				if (strstr(decoderconfig+i+4, "fps"))
					ctx->is_vfr = GF_TRUE;

				break;
			}
			i++;
