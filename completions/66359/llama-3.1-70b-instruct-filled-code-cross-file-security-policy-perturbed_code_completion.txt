while (i < vosh_size - 3) {
    if (decoderconfig[i] == 0x00 && decoderconfig[i + 1] == 0x00 && decoderconfig[i + 2] == 0x01) {
        if (memchr(decoderconfig + i + 3, 'D', vosh_size - i - 3) == NULL) {
            break;
        }
        frame = memchr(decoderconfig + i + 4, 'p', vosh_size - i - 4);
        if (frame) {
            ctx->forced_packed = GF_TRUE;
            frame[0] = 'n';
        }
        break;
    }
    i++;
}
