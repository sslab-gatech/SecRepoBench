    u8 *found = NULL;
    while (i + 7 <= vosh_size) {
        if (decoderconfig[i]     == 0x00 &&
            decoderconfig[i + 1] == 0x00 &&
            decoderconfig[i + 2] == 0x01 &&
            decoderconfig[i + 3] == 'D'  &&
            decoderconfig[i + 4] == 'i'  &&
            decoderconfig[i + 5] == 'v'  &&
            decoderconfig[i + 6] == 'X') {
            found = decoderconfig + i + 3; /* points to the start of "DivX" */
            break;
        }
        i++;
    }
    if (found) {
        ctx->forced_packed = GF_TRUE;
        /* change "DivX" to "Divn": remove the packed flag */
        found[3] = 'n';
    }
}
