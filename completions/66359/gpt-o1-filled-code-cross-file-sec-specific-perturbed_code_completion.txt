u32 i = 0;
while ((i + 7) <= vosh_size) {
    /* check for sequence 0x00 0x00 0x01 'D' 'i' 'v' 'X' */
    if (!memcmp(decoderconfig + i, "\x00\x00\x01DivX", 7)) {
        ctx->forced_packed = GF_TRUE;
        /* replace 'D' with 'n' (or any safe override) */
        decoderconfig[i + 3] = 'n';
        break;
    }
    i++;
}
