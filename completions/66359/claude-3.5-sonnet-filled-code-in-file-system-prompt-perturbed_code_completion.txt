while (i < vosh_size - 4) {
    if (decoderconfig[i] == 0 && decoderconfig[i+1] == 0 && decoderconfig[i+2] == 1 && decoderconfig[i+3] == 0xB2) {
        char *frame = memchr(decoderconfig + i + 4, 'D', vosh_size - i - 4);
        if (frame && !strncmp(frame, "DivX", 4)) {
            frame = memchr(frame + 4, 'p', vosh_size - (frame - decoderconfig) - 4);
            if (frame) {
                ctx->forced_packed = GF_TRUE;
                frame[0] = 'n';
            }
            break;
        }
    }
    i++;
}
