case 0x07: /* Open Connection Request 2 */
      if (packet->payload_packet_len < 22 || packet->payload[1] != 0 || packet->payload[21] > 10)
      {
        NDPI_EXCLUDE_PROTO(ndpi_struct, flow);
        return;
      }

      {
        u_int16_t mtu_size = ntohs(get_u_int16_t(packet->payload, 2));

        if (mtu_size > 1500)
        {
          NDPI_EXCLUDE_PROTO(ndpi_struct, flow);
          return;
        }
      }
      required_packets = 4;
      break;

      if (packet->payload_packet_len != 12 || packet->payload[11] > 10)
      {
        NDPI_EXCLUDE_PROTO(ndpi_struct, flow);
        return;
      }
