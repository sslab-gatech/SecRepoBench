case 0x07: /* Open Connection Request 2 */
      if (packet->payload_packet_len < 18)
      {
        NDPI_EXCLUDE_PROTO(ndpi_struct, flow);
        return;
      }
      required_packets = 6;
      break;

    case 0x08: /* Open Connection Reply 2 */
      if (packet->payload_packet_len != 30 ||
          packet->payload[25] > 0x01 /* connection uses encryption: bool -> 0x00 or 0x01 */)
      {
        NDPI_EXCLUDE_PROTO(ndpi_struct, flow);
        return;
      }

      {
        u_int16_t server_port = ntohs(get_u_int16_t(packet->payload, 28));
        if (server_port == 0)
        {
          NDPI_EXCLUDE_PROTO(ndpi_struct, flow);
          return;
        }
      }
      required_packets = 4;
      break;
