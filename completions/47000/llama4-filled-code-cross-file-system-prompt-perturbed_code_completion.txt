case 0x07: /* Open Connection Request 2 */
      if (packet->payload_packet_len < 32)
      {
        NDPI_EXCLUDE_PROTO(ndpi_struct, flow);
        return;
      }
      ip_addr_offset = 24 + raknet_dissect_ip(packet, 0);
      if (ip_addr_offset != packet->payload_packet_len)
      {
        NDPI_EXCLUDE_PROTO(ndpi_struct, flow);
        return;
      }
      required_packets = 4;
      break;

    case 0x08: /* Open Connection Reply 2 */
      if (packet->payload_packet_len != 40)
      {
        NDPI_EXCLUDE_PROTO(ndpi_struct, flow);
        return;
      }

      {
        u_int16_t max_mtu_size = ntohs(get_u_int16_t(packet->payload, 38));
        if (max_mtu_size > 1500 /* Max. supported MTU, see: http://www.jenkinssoftware.com/raknet/manual/programmingtips.html */)
        {
          NDPI_EXCLUDE_PROTO(ndpi_struct, flow);
          return;
        }
      }
      required_packets = 4;
      break;
