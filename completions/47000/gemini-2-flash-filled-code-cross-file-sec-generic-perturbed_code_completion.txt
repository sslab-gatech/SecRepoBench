case 0x07: /* Open Connection Request 2 */
      if (packet->payload_packet_len != 20)
      {
        NDPI_EXCLUDE_PROTO(ndpi_struct, flow);
        return;
      }
      required_packets = 6;
      break;

    case 0x08: /* Open Connection Reply 2 */
      ip_addr_offset = 1;
      if (packet->payload_packet_len < 20 + ip_addr_offset)
      {
        NDPI_EXCLUDE_PROTO(ndpi_struct, flow);
        return;
      }

      {
        u_int16_t server_port = ntohs(get_u_int16_t(packet->payload, 18));
        u_int16_t client_mtu = ntohs(get_u_int16_t(packet->payload, 20));

        if (client_mtu > 1500)
        {
          NDPI_EXCLUDE_PROTO(ndpi_struct, flow);
          return;
        }
      }
      required_packets = 4;
      break;
