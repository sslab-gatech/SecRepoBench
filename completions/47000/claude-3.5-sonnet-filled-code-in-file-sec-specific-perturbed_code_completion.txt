case 0x07: /* Open Connection Request 2 */
  ip_addr_offset = 21 + raknet_dissect_ip(packet, 21);
  if (packet->payload_packet_len != ip_addr_offset + 2)
  {
    NDPI_EXCLUDE_PROTO(ndpi_struct, flow);
    return;
  }
  required_packets = 6;
  break;

case 0x08: /* Open Connection Reply 2 */
  ip_addr_offset = 1 + raknet_dissect_ip(packet, 1);
  if (packet->payload_packet_len != ip_addr_offset + 12)
  {
    NDPI_EXCLUDE_PROTO(ndpi_struct, flow);
    return;
  }
  {
    u_int16_t mtu_size = ntohs(get_u_int16_t(packet->payload, ip_addr_offset + 2));
    if (mtu_size > 1500 /* Max. supported MTU */)
    {
      NDPI_EXCLUDE_PROTO(ndpi_struct, flow);
      return;
    }
  }
  required_packets = 4;
  break;
