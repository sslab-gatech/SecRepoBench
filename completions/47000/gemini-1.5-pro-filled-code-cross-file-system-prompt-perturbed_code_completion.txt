case 0x07: /* Open Connection Request 2 */
      if (packet->payload_packet_len < 23 || packet->payload[22] > 10)
      {
        NDPI_EXCLUDE_PROTO(ndpi_struct, flow);
        return;
      }
      required_packets = 4;
      break;

    case 0x08: /* Open Connection Reply 2 */
      ip_addr_offset = 16;
      if (packet->payload_packet_len < ip_addr_offset + 6)
      {
        NDPI_EXCLUDE_PROTO(ndpi_struct, flow);
        return;
      }

      {
        u_int16_t mtu_size = ntohs(get_u_int16_t(packet->payload, ip_addr_offset + 4));

        if (mtu_size > 1500)
        {
          NDPI_EXCLUDE_PROTO(ndpi_struct, flow);
          return;
        }
      }
      required_packets = 4;
      break;
