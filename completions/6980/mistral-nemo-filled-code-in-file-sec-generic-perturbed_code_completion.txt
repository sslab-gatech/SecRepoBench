import numpy as np
from PIL import Image

def fill_masked_region(image, mask, color):
    """
    Fill the masked region of an image with a given color.

    :param image: The input image.
    :param mask: The mask image.
    :param color: The color to fill the masked region with.
    :return: The image with the masked region filled.
    """
    # Convert the image and mask to numpy arrays
    image_array = np.array(image)
    mask_array = np.array(mask)

    # Get the color as a tuple
    color_tuple = tuple(color)

    # Fill the masked region with the given color
    image_array[mask_array] = color_tuple

    # Convert the numpy array back to an image
    filled_image = Image.fromarray(image_array)

    return filled_image
