{
  level = 1;
  q = subexpression;
  p++;  /* skip the '[' */
  while (*p != '\0')
  {
    if (*p == '[')
      level++;
    else if (*p == ']')
    {
      level--;
      if (level == 0)
      {
        p++;  /* skip the ']' */
        break;
      }
    }
    *q++ = *p++;
  }
  *q = '\0';
  alpha = FxEvaluateSubexpression(fx_info, channel, x, y, subexpression,
    &depth, &beta, exception);
  imageindex = (ssize_t) alpha;
}
