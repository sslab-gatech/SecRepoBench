if (codeLength < (uint64_t) SHORT_ZEROCODE_RUN)
{
    // Literal code length (0-58)
    if (codeLength > 0)
    {
        if ((uint64_t) fhd->_minCodeLength > codeLength)
            fhd->_minCodeLength = codeLength;
        
        if ((uint64_t) fhd->_maxCodeLength < codeLength)
            fhd->_maxCodeLength = codeLength;
        
        codeCount[codeLength]++;
    }
}
else if (codeLength == (uint64_t) LONG_ZEROCODE_RUN)
{
    // Long run of zeros (63)
    // Read next 8 bits for run length
    uint64_t runLength = fasthuf_read_bits(8, &currBits, &currBitCount, &currByte);
    runLength += SHORTEST_LONG_RUN;
    symbol += runLength - 1; // -1 because the loop will increment
}
else
{
    // Short run of zeros (59-62)
    uint64_t runLength = codeLength - SHORT_ZEROCODE_RUN + 1;
    symbol += runLength - 1; // -1 because the loop will increment
}
