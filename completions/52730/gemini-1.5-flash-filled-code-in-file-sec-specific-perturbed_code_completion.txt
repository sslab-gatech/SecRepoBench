if (codeLength < (uint64_t) SHORT_ZEROCODE_RUN)
        {
            if (codeLength == 0) continue;
            codeCount[codeLength]++;
            if (base[codeLength] == 0xffffffffffffffffULL)
                base[codeLength] = symbol;
            fhd->_minCodeLength =
                (unsigned char) std::min ((int) fhd->_minCodeLength,
                                          (int) codeLength);
            fhd->_maxCodeLength =
                (unsigned char) std::max ((int) fhd->_maxCodeLength,
                                          (int) codeLength);
        }
        else if (codeLength == (uint64_t) LONG_ZEROCODE_RUN)
        {
            uint64_t runLength =
                fasthuf_read_bits (8, &currBits, &currBitCount, &currByte) +
                SHORTEST_LONG_RUN - 1;
            codeCount[0] += runLength;
            symbol += runLength;
        }
        else
        {
            uint64_t runLength = codeLength - SHORT_ZEROCODE_RUN + 1;
            codeCount[0] += runLength;
            symbol += runLength;
        }
