if (codeLength < SHORT_ZEROCODE_RUN)
        {
            /* literal code length */
            if (codeLength != 0)
            {
                if (codeLength < fhd->_minCodeLength)
                    fhd->_minCodeLength = (uint8_t) codeLength;
                if (codeLength > fhd->_maxCodeLength)
                    fhd->_maxCodeLength = (uint8_t) codeLength;
                codeCount[codeLength]++;
            }
        }
        else if (codeLength < LONG_ZEROCODE_RUN)
        {
            /* short run of zeros */
            uint64_t runLength = codeLength - SHORT_ZEROCODE_RUN + 2;
            symbol += (runLength - 1);
        }
        else
        {
            /* long run of zeros */
            if ((currByte + 1) > topByte)
            {
                if (pctxt)
                    pctxt->print_error (
                        pctxt,
                        EXR_ERR_CORRUPT_CHUNK,
                        "Error decoding Huffman table (Truncated run data).");
                return EXR_ERR_CORRUPT_CHUNK;
            }
            uint64_t runLength =
                fasthuf_read_bits (8, &currBits, &currBitCount, &currByte) +
                SHORTEST_LONG_RUN;
            symbol += (runLength - 1);
        }
