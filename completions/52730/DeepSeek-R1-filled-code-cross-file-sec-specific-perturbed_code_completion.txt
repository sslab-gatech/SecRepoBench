if (codeLength < (uint64_t) SHORT_ZEROCODE_RUN)
        {
            if (codeLength != 0)
            {
                if (codeLength < fhd->_minCodeLength)
                    fhd->_minCodeLength = (uint8_t)codeLength;
                if (codeLength > fhd->_maxCodeLength)
                    fhd->_maxCodeLength = (uint8_t)codeLength;
                codeCount[codeLength]++;
            }
            else
            {
                codeCount[0]++;
            }
        }
        else if (codeLength < (uint64_t) LONG_ZEROCODE_RUN)
        {
            int runLength = codeLength - SHORT_ZEROCODE_RUN + 1;
            codeCount[0] += runLength;
            symbol += runLength - 1;
        }
        else
        {
            uint64_t runLength = fasthuf_read_bits(8, &currBits, &currBitCount, &currByte) + SHORTEST_LONG_RUN - 1;
            codeCount[0] += runLength;
            symbol += runLength - 1;
        }
