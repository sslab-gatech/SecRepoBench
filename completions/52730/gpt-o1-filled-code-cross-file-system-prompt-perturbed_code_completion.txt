// Handle codeLength and update code counts and min/max lengths.
        if (codeLength < SHORT_ZEROCODE_RUN)
        {
            // A literal code length in the range [1..58], or 0 meaning no code
            if (codeLength != 0)
            {
                if ((uint8_t) codeLength < fhd->_minCodeLength)
                    fhd->_minCodeLength = (uint8_t) codeLength;
                if ((uint8_t) codeLength > fhd->_maxCodeLength)
                    fhd->_maxCodeLength = (uint8_t) codeLength;

                codeCount[codeLength]++;
            }
        }
        else if (codeLength == LONG_ZEROCODE_RUN)
        {
            // A long run of zeros; the run length is in the next 8 bits (+ SHORTEST_LONG_RUN)
            uint64_t run =
                fasthuf_read_bits(8, &currBits, &currBitCount, &currByte) + SHORTEST_LONG_RUN;
            symbol += run - 1; // Adjust the symbol index to account for the zero-length run
        }
        else
        {
            // A short run of zeros (codeLength in [59..62])
            symbol += (codeLength - SHORT_ZEROCODE_RUN + 1);
