if (codeLength < (uint64_t) SHORT_ZEROCODE_RUN)
        {
            if (codeLength != 0)
            {
                if (codeLength > (uint64_t) MAX_CODE_LEN)
                {
                    if (pctxt)
                        pctxt->print_error (
                            pctxt,
                            EXR_ERR_CORRUPT_CHUNK,
                            "Error decoding Huffman table (Invalid code length).");
                    return EXR_ERR_CORRUPT_CHUNK;
                }

                if (fhd->_minCodeLength > codeLength)
                    fhd->_minCodeLength = (unsigned char) codeLength;

                if (fhd->_maxCodeLength < codeLength)
                    fhd->_maxCodeLength = (unsigned char) codeLength;

                codeCount[codeLength]++;
            }
        }
        else if (codeLength == (uint64_t) LONG_ZEROCODE_RUN)
        {
            uint64_t zeroRunLength =
                fasthuf_read_bits (8, &currBits, &currBitCount, &currByte) +
                SHORTEST_LONG_RUN - 1;

            if (symbol + zeroRunLength > (uint64_t) maxSymbol + 1)
            {
                if (pctxt)
                    pctxt->print_error (
                        pctxt,
                        EXR_ERR_CORRUPT_CHUNK,
                        "Error decoding Huffman table (Zero run beyond end of table).");
                return EXR_ERR_CORRUPT_CHUNK;
            }

            symbol += zeroRunLength;
        }
        else
        {
            symbol += codeLength - SHORT_ZEROCODE_RUN + 1;
        }
