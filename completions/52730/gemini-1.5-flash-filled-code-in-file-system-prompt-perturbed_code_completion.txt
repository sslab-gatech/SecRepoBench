if (codeLength < (uint64_t) SHORT_ZEROCODE_RUN)
        {
            if (codeLength > 0)
            {
                codeCount[codeLength]++;
                fhd->_minCodeLength =
                    (unsigned char) std::min ((int) fhd->_minCodeLength,
                                              (int) codeLength);
                fhd->_maxCodeLength =
                    (unsigned char) std::max ((int) fhd->_maxCodeLength,
                                              (int) codeLength);
            }
        }
        else if (codeLength == (uint64_t) LONG_ZEROCODE_RUN)
        {
            uint64_t runLength =
                fasthuf_read_bits (8, &currBits, &currBitCount, &currByte) +
                SHORTEST_LONG_RUN - 1;
            symbol += runLength;
        }
        else
        {
            symbol += codeLength - SHORT_ZEROCODE_RUN + 1;
        }
