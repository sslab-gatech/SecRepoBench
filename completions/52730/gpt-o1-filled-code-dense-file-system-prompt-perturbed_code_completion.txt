const int SHORT_ZEROCODE_RUN  = 59;
        const int LONG_ZEROCODE_RUN   = 63;
        const int SHORTEST_LONG_RUN   = 2 + (LONG_ZEROCODE_RUN - SHORT_ZEROCODE_RUN);

        if (codeLength == (uint64_t) LONG_ZEROCODE_RUN)
        {
            if (currByte >= topByte)
            {
                if (pctxt)
                    pctxt->print_error (
                        pctxt,
                        EXR_ERR_CORRUPT_CHUNK,
                        "Error decoding Huffman table (Truncated table data).");
                return EXR_ERR_CORRUPT_CHUNK;
            }
            uint64_t runLen =
                fasthuf_read_bits (8, &currBits, &currBitCount, &currByte) +
                SHORTEST_LONG_RUN;
            symbol += (runLen - 1);
        }
        else if (codeLength >= (uint64_t) SHORT_ZEROCODE_RUN)
        {
            uint64_t runLen = codeLength - SHORT_ZEROCODE_RUN + 2;
            symbol += (runLen - 1);
        }
        else if (codeLength != 0)
        {
            codeCount[codeLength]++;
            if (codeLength < fhd->_minCodeLength)
                fhd->_minCodeLength = (uint8_t) codeLength;
            if (codeLength > fhd->_maxCodeLength)
                fhd->_maxCodeLength = (uint8_t) codeLength;
        }
