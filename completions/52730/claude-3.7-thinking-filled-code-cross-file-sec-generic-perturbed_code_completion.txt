if (codeLength < (uint64_t) SHORT_ZEROCODE_RUN)
{
    if (codeLength > 0)
    {
        if (codeLength > MAX_CODE_LEN)
        {
            if (pctxt)
                pctxt->print_error (
                    pctxt,
                    EXR_ERR_CORRUPT_CHUNK,
                    "Error decoding Huffman table (Code length too large).");
            return EXR_ERR_CORRUPT_CHUNK;
        }

        if ((unsigned char) codeLength < fhd->_minCodeLength)
            fhd->_minCodeLength = (unsigned char) codeLength;

        if ((unsigned char) codeLength > fhd->_maxCodeLength)
            fhd->_maxCodeLength = (unsigned char) codeLength;

        codeCount[codeLength]++;
    }
}
else if (codeLength == (uint64_t) LONG_ZEROCODE_RUN)
{
    symbol += fasthuf_read_bits (8, &currBits, &currBitCount, &currByte) +
              SHORTEST_LONG_RUN - 1;
}
else
{
    symbol += codeLength - SHORT_ZEROCODE_RUN + 1;
}
