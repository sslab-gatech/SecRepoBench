if (codeLength < (uint64_t) SHORT_ZEROCODE_RUN)
        {
            if (codeLength > 0)
            {
                if (codeLength < fhd->_minCodeLength)
                    fhd->_minCodeLength = (unsigned char) codeLength;
                if (codeLength > fhd->_maxCodeLength)
                    fhd->_maxCodeLength = (unsigned char) codeLength;

                codeCount[codeLength]++;
            }
        }
        else if (codeLength == (uint64_t) LONG_ZEROCODE_RUN)
        {
            if (currByte >= topByte)
            {
                if (pctxt)
                    pctxt->print_error(
                        pctxt,
                        EXR_ERR_CORRUPT_CHUNK,
                        "Error decoding Huffman table (Truncated long zero-code run).");
                return EXR_ERR_CORRUPT_CHUNK;
            }

            uint64_t runLength = fasthuf_read_bits(8, &currBits, &currBitCount, &currByte);
            symbol += runLength + SHORTEST_LONG_RUN - 1;
        }
        else
        {
            uint64_t runLength = codeLength - SHORT_ZEROCODE_RUN + 1;
            symbol += runLength - 1;
        }
