if (codeLength > (uint64_t) MAX_CODE_LEN)
        {
            if (pctxt)
                pctxt->print_error (
                    pctxt,
                    EXR_ERR_CORRUPT_CHUNK,
                    "Error decoding Huffman table (Invalid code length).");
            return EXR_ERR_CORRUPT_CHUNK;
        }

        if (codeLength > 0)
        {
            fhd->_minCodeLength = fhd->_minCodeLength < codeLength ? fhd->_minCodeLength : codeLength;
            fhd->_maxCodeLength = fhd->_maxCodeLength > codeLength ? fhd->_maxCodeLength : codeLength;
            codeCount[codeLength]++;
        }
