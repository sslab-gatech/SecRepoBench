if (codeLength < SHORT_ZEROCODE_RUN) {
    if (codeLength != 0) {
        codeCount[codeLength]++;
        if (codeLength < fhd->_minCodeLength)
            fhd->_minCodeLength = codeLength;
        if (codeLength > fhd->_maxCodeLength)
            fhd->_maxCodeLength = codeLength;
    } else {
        codeCount[0]++;
    }
} else if (codeLength == LONG_ZEROCODE_RUN) {
    uint64_t runLength = fasthuf_read_bits(8, &currBits, &currBitCount, &currByte) + SHORTEST_LONG_RUN;
    codeCount[0] += runLength;
    symbol += runLength - 1;
} else {
    uint64_t runLength = codeLength - SHORT_ZEROCODE_RUN + 1;
    codeCount[0] += runLength;
    symbol += runLength - 1;
}
