if (codeLength < (uint64_t) SHORT_ZEROCODE_RUN)
{
    if (codeLength > 0 && codeLength <= MAX_CODE_LEN)
    {
        fhd->_minCodeLength = (codeLength < fhd->_minCodeLength) ? codeLength : fhd->_minCodeLength;
        fhd->_maxCodeLength = (codeLength > fhd->_maxCodeLength) ? codeLength : fhd->_maxCodeLength;
        codeCount[codeLength]++;
    }
}
else if (codeLength == (uint64_t) LONG_ZEROCODE_RUN)
{
    uint64_t runLength = fasthuf_read_bits (8, &currBits, &currBitCount, &currByte);
    symbol += runLength + SHORTEST_LONG_RUN - 1;
}
else
{
    uint64_t runLength = codeLength - SHORT_ZEROCODE_RUN + 1;
    symbol += runLength;
}
