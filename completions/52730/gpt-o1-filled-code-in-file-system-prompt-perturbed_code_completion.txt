if (codeLength < SHORT_ZEROCODE_RUN)
{
    /* codeLength in [0..58], either 0 means no code, or >0 means a real code length */
    if (codeLength != 0)
    {
        if (codeLength < fhd->_minCodeLength) fhd->_minCodeLength = (unsigned char) codeLength;
        if (codeLength > fhd->_maxCodeLength) fhd->_maxCodeLength = (unsigned char) codeLength;
        codeCount[codeLength]++;
    }
}
else if (codeLength == LONG_ZEROCODE_RUN)
{
    /* run of zeros, length is next 8 bits plus SHORTEST_LONG_RUN (6) */
    uint64_t runLength = fasthuf_read_bits(8, &currBits, &currBitCount, &currByte) + SHORTEST_LONG_RUN;
    symbol += runLength - 1;
}
else
{
    /* short run of zeros, length is codeLength - SHORT_ZEROCODE_RUN + 2 */
    uint64_t runLength = codeLength - SHORT_ZEROCODE_RUN + 2;
    symbol += runLength - 1;
}
