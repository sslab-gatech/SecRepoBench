if (codeLength < 59)
{
    if (codeLength > 0)
    {
        if (codeLength < fhd->_minCodeLength)
            fhd->_minCodeLength = (unsigned char)codeLength;
        if (codeLength > fhd->_maxCodeLength)
            fhd->_maxCodeLength = (unsigned char)codeLength;
        codeCount[codeLength]++;
    }
}
else if (codeLength == 63)
{
    uint64_t zeroes = fasthuf_read_bits(8, &currBits, &currBitCount, &currByte) + SHORTEST_LONG_RUN;
    symbol += zeroes - 1;
}
else
{
    uint64_t zeroes = codeLength - SHORT_ZEROCODE_RUN + 2;
    symbol += zeroes - 1;
}
