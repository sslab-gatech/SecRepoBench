if (codeLength < (uint64_t) SHORT_ZEROCODE_RUN)
        {
            if (codeLength > 0)
            {
                if (codeLength > (uint64_t) fhd->_maxCodeLength)
                {
                    if (pctxt)
                        pctxt->print_error (
                            pctxt,
                            EXR_ERR_CORRUPT_CHUNK,
                            "Error decoding Huffman table (Code length too long).");
                    return EXR_ERR_CORRUPT_CHUNK;
                }

                codeCount[codeLength]++;
                fhd->_minCodeLength =
                    (int) std::min ((uint64_t) fhd->_minCodeLength, codeLength);
                fhd->_maxCodeLength =
                    (int) std::max ((uint64_t) fhd->_maxCodeLength, codeLength);
            }
        }
        else if (codeLength == (uint64_t) LONG_ZEROCODE_RUN)
        {
            uint64_t runLength =
                fasthuf_read_bits (8, &currBits, &currBitCount, &currByte) +
                SHORTEST_LONG_RUN - 1;
            symbol += runLength;
        }
        else
        {
            symbol += codeLength - SHORT_ZEROCODE_RUN + 1;
        }
