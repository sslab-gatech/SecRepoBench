	  char *quote = strchr((char*)&http_packet_struct->content_disposition_line.ptr[attachment_len+1], '\"');

	  if(quote) filename_len = quote - (char*)&http_packet_struct->content_disposition_line.ptr[attachment_len+1];

	  flow->http.filename = ndpi_malloc(filename_len+1);
	  if(flow->http.filename) {
	    memcpy(flow->http.filename, &http_packet_struct->content_disposition_line.ptr[attachment_len+1], filename_len);
	    flow->http.filename[filename_len] = '\0';
	  }
	} else {
	  flow->http.filename = ndpi_strdup((char*)&http_packet_struct->content_disposition_line.ptr[attachment_len]);
	}
