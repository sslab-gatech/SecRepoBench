control_points=quantum/2+2; // Number of control points for Bezier curve computation
coefficients=(double *) AcquireQuantumMemory((size_t) number_coordinates,
  sizeof(*coefficients));
points=(PointInfo *) AcquireQuantumMemory((size_t) control_points,
  sizeof(*points));
if ((coefficients == (double *) NULL) || (points == (PointInfo *) NULL))
  {
    (void) ThrowMagickException(bezierinfo->exception,GetMagickModule(),
      ResourceLimitError,"MemoryAllocationFailed","`%s'","");
    coefficients=(double *) RelinquishMagickMemory(coefficients);
    points=(PointInfo *) RelinquishMagickMemory(points);
    return(MagickFalse);
  }
if (CheckPrimitiveExtent(bezierinfo,PrimitiveExtentPad) == MagickFalse)
  {
    coefficients=(double *) RelinquishMagickMemory(coefficients);
    points=(PointInfo *) RelinquishMagickMemory(points);
    return(MagickFalse);
  }
