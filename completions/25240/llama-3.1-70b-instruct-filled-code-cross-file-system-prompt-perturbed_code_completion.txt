        } break;

            if (getFloatValue(fValue)) {

            }
        } break;

            std::string strMaterialName;
            if (getStringValue(strMaterialName)) {
                m_pModel->m_pCurrentMaterial->MaterialName = strMaterialName;

        } break;
        case'm': { // Texture map

            }
        } break;

            int iValue;
            if (getIntValue(iValue)) {
                m_pModel->m_pCurrentMaterial->m_IlluminationModel = iValue;
            }
        } break;
