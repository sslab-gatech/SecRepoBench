case 'K': { // Assuming 'K' indicates a new material property
                std::string propertyName;
                m_DataIt = getNextToken<DataArrayIt>(m_DataIt, m_DataItEnd, propertyName);
                if (m_DataIt == m_DataItEnd) {
                    return; // Prevent buffer overflow
                }

                if (propertyName == "ambient") {
                    aiColor3D ambientColor;
                    getColorRGBA(ambientColor);
                    m_pCurrentMaterial->Ambient = ambientColor;
                } else if (propertyName == "diffuse") {
                    aiColor3D diffuseColor;
                    getColorRGBA(diffuseColor);
                    m_pCurrentMaterial->Diffuse = diffuseColor;
                } else if (propertyName == "specular") {
                    aiColor3D specularColor;
                    getColorRGBA(specularColor);
                    m_pCurrentMaterial->Specular = specularColor;
                } else if (propertyName == "emissive") {
                    aiColor3D emissiveColor;
                    getColorRGBA(emissiveColor);
                    m_pCurrentMaterial->Emissive = emissiveColor;
                } else if (propertyName == "transmission") {
                    float transmission;
                    m_DataIt = getNextToken<DataArrayIt>(m_DataIt, m_DataItEnd, transmission);
                    m_pCurrentMaterial->Transmission = transmission;
                } else if (propertyName == "alpha") {
                    float alpha;
                    m_DataIt = getNextToken<DataArrayIt>(m_DataIt, m_DataItEnd, alpha);
                    m_pCurrentMaterial->Alpha = alpha;
                } else if (propertyName == "specular_exponent") {
                    float specExponent;
                    m_DataIt = getNextToken<DataArrayIt>(m_DataIt, m_DataItEnd, specExponent);
                    m_pCurrentMaterial->SpecularExponent = specExponent;
                } else if (propertyName == "ior") {
                    float ior;
                    m_DataIt = getNextToken<DataArrayIt>(m_DataIt, m_DataItEnd, ior);
                    m_pCurrentMaterial->IndexOfRefraction = ior;
                } else {
                    // Handle unknown property
                    ASSIMP_LOG_WARN("Unknown material property: ", propertyName);
                }
            } break;
