case 'n': {
                if (*(m_DataIt + 1) == 's') {
                    // Specular exponent
                    m_DataIt += 2; // Skip 'ns'
                    getFloatValue(m_pModel->m_pCurrentMaterial->shininess);
                } else if (*(m_DataIt + 1) == 'i') {
                    // Index of Refraction
                    m_DataIt += 2; // Skip 'ni'
                    getFloatValue(m_pModel->m_pCurrentMaterial->ior);
                }
            } break;

            case 'K': {
                if (*(m_DataIt + 1) == 'a') {
                    // Ambient color
                    m_DataIt += 2; // Skip 'Ka'
                    getColorRGBA(m_pModel->m_pCurrentMaterial->ambient);
                } else if (*(m_DataIt + 1) == 'd') {
                    // Diffuse color
                    m_DataIt += 2; // Skip 'Kd'
                    getColorRGBA(m_pModel->m_pCurrentMaterial->diffuse);
                } else if (*(m_DataIt + 1) == 's') {
                    // Specular color
                    m_DataIt += 2; // Skip 'Ks'
                    getColorRGBA(m_pModel->m_pCurrentMaterial->specular);
                }
            } break;

            case 'd': {
                // Transparency
                m_DataIt++; // Skip 'd'
                getFloatValue(m_pModel->m_pCurrentMaterial->transparency);
            } break;

            case 'newmtl': {
                // New material
                m_DataIt += 6; // Skip 'newmtl'
                createMaterial();
            } break;

            case 'map': {
                // Texture mapping
                m_DataIt++;
                getTexture();
            } break;

            default: {
                m_DataIt = skipLine<DataArrayIt>(m_DataIt, m_DataItEnd, m_uiLine);
            } break;
        }
