                // Ensure we have enough data to process
                if (std::distance(m_DataIt, m_DataItEnd) < 1) {
                    m_DataIt = skipLine<DataArrayIt>(m_DataIt, m_DataItEnd, m_uiLine);
                    break;
                }

                // Parse material properties
                ++m_DataIt; // Move past 'k'
                if (m_DataIt == m_DataItEnd) {
                    break;
                }

                switch (*m_DataIt) {
                    case 'a': { // Ambient color
                        ++m_DataIt;
                        aiColor3D color;
                        getColorRGBA(&color);
                        m_pModel->m_pCurrentMaterial->Ambient = color;
                    } break;

                    case 'd': { // Diffuse color
                        ++m_DataIt;
                        aiColor3D color;
                        getColorRGBA(&color);
                        m_pModel->m_pCurrentMaterial->Diffuse = color;
                    } break;

                    case 's': { // Specular color
                        ++m_DataIt;
                        aiColor3D color;
                        getColorRGBA(&color);
                        m_pModel->m_pCurrentMaterial->Specular = color;
                    } break;

                    case 'e': { // Emissive color
                        ++m_DataIt;
                        aiColor3D color;
                        getColorRGBA(&color);
                        m_pModel->m_pCurrentMaterial->Emissive = color;
                    } break;

                    case 't': { // Transmission color
                        ++m_DataIt;
                        aiColor3D color;

                        m_pModel->m_pCurrentMaterial->Transmission = color;
                    } break;

                    default: {
                        m_DataIt = skipLine<DataArrayIt>(m_DataIt, m_DataItEnd, m_uiLine);
                    } break;
                }
