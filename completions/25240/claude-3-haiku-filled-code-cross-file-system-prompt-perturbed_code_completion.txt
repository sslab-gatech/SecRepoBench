case 'k':
{
    // Ambient color
    if (*m_DataIt == 'a') {
        ++m_DataIt;
        getColorRGBA(m_pModel->m_pCurrentMaterial->ambient);
    }
    // Diffuse color
    else if (*m_DataIt == 'd') {
        ++m_DataIt;
        getColorRGBA(m_pModel->m_pCurrentMaterial->diffuse);
    }
    // Specular color
    else if (*m_DataIt == 's') {
        ++m_DataIt;
        getColorRGBA(m_pModel->m_pCurrentMaterial->specular);
    }
    // Emissive color
    else if (*m_DataIt == 'e') {
        ++m_DataIt;
        getColorRGBA(m_pModel->m_pCurrentMaterial->emissive);
    }
    // Transmission color
    else if (*m_DataIt == 'T') {
        ++m_DataIt;
        getColorRGBA(m_pModel->m_pCurrentMaterial->transparent);
    }
    // Transparency
    else if (*m_DataIt == 'n') {
        ++m_DataIt;
        if (*m_DataIt == 's') {
            ++m_DataIt;
            getFloat(m_pModel->m_pCurrentMaterial->shineness);
        } else if (*m_DataIt == 'i') {
            ++m_DataIt;
            getFloat(m_pModel->m_pCurrentMaterial->ior);
        }
    }
    // Texture
    else if (*m_DataIt == 'm') {
        ++m_DataIt;
        if (*m_DataIt == 'a') {
            ++m_DataIt;
            getTexture(m_pModel->m_pCurrentMaterial->textureAmbient);
        } else if (*m_DataIt == 'd') {
            ++m_DataIt;
            getTexture(m_pModel->m_pCurrentMaterial->textureDiffuse);
        } else if (*m_DataIt == 's') {
            ++m_DataIt;
            getTexture(m_pModel->m_pCurrentMaterial->textureSpecular);
        } else if (*m_DataIt == 'b') {
            ++m_DataIt;
            getTexture(m_pModel->m_pCurrentMaterial->textureBump);
        } else if (*m_DataIt == 'l') {
            ++m_DataIt;
            getTexture(m_pModel->m_pCurrentMaterial->textureLight);
        } else if (*m_DataIt == 'o') {
            ++m_DataIt;
            getTexture(m_pModel->m_pCurrentMaterial->textureOpacity);
        } else if (*m_DataIt == 'r') {
            ++m_DataIt;
            getTexture(m_pModel->m_pCurrentMaterial->textureReflection);
        } else if (*m_DataIt == 'T') {
            ++m_DataIt;
            getTexture(m_pModel->m_pCurrentMaterial->textureTransparency);
        }
    }
    // Illumination model
    else if (*m_DataIt == 'i') {
        ++m_DataIt;
        getInteger(m_pModel->m_pCurrentMaterial->illumination_model);
    }
    // Anisotropy
    else if (*m_DataIt == 'o') {
        ++m_DataIt;
        if (*m_DataIt == 'a') {
            ++m_DataIt;
            getFloat(m_pModel->m_pCurrentMaterial->anisotropy);
        } else if (*m_DataIt == 'r') {
            ++m_DataIt;
            getFloat(m_pModel->m_pCurrentMaterial->anisotropy_rotation);
        }
    }
    // New material
    else if (*m_DataIt == 'n') {
        ++m_DataIt;
        if (*m_DataIt == 'e') {
            ++m_DataIt;
            if (*m_DataIt == 'w') {
                ++m_DataIt;
                skipSpaces();
                std::string name;
                while (m_DataIt != m_DataItEnd && !IsLineEnd(*m_DataIt)) {
                    name += *m_DataIt;
                    ++m_DataIt;
                }
                createMaterial(name);
            }
        }
    }
    else {
        m_DataIt = skipLine<DataArrayIt>(m_DataIt, m_DataItEnd, m_uiLine);
    }
} break;
