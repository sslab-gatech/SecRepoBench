case 'k': {
    // We'll read one more character to see if this is Ka, Kd, Ks, Ke, or Tf
    ++m_DataIt;
    if (m_DataIt == m_DataItEnd) {
        break;
    }
    char c2 = static_cast<char>(std::tolower(static_cast<unsigned char>(*m_DataIt)));

    // Move iterator so we can parse color data
    ++m_DataIt;

    // Prepare a color variable
    Maybe<aiColor3D> colorData;
    getColorRGBA(colorData);
    aiColor3D defaultColor(0.0f, 0.0f, 0.0f);

    // Assign to the correct material property
    if (m_pCurrentMaterial) {
        switch (c2) {
            case 'a': // Ka: Ambient color
                m_pCurrentMaterial->Ambient = colorData.value_or(defaultColor);
                break;
            case 'd': // Kd: Diffuse color
                m_pCurrentMaterial->Diffuse = colorData.value_or(defaultColor);
                break;
            case 's': // Ks: Specular color
                m_pCurrentMaterial->Specular = colorData.value_or(defaultColor);
                break;
            case 'e': // Ke: Emissive color
                m_pCurrentMaterial->Emissive = colorData.value_or(defaultColor);
                break;
            case 'f': // Tf: Transmission filter
                m_pCurrentMaterial->TransmissionFilter = colorData.value_or(defaultColor);
                break;
            default:
                // Not recognized, do nothing
                break;
        }
    }

    // Move to the end of this line
    m_DataIt = skipLine<DataArrayIt>(m_DataIt, m_DataItEnd, m_uiLine);
} break;
