case 'K':
    {
        ++m_DataIt;
        if (m_DataIt == m_DataItEnd) break;
        switch (*m_DataIt) {
            case 'a': // Ambient color
                if (m_pModel && m_pModel->m_pCurrentMaterial) {
                    ++m_DataIt;
                    getColorRGBA(m_pModel->m_pCurrentMaterial->ambient);
                }
                break;
            case 'd': // Diffuse color
                if (m_pModel && m_pModel->m_pCurrentMaterial) {
                    ++m_DataIt;
                    getColorRGBA(m_pModel->m_pCurrentMaterial->diffuse);
                }
                break;
            case 's': // Specular color
                if (m_pModel && m_pModel->m_pCurrentMaterial) {
                    ++m_DataIt;
                    getColorRGBA(m_pModel->m_pCurrentMaterial->specular);
                }
                break;
            case 'e': // Emissive color
                if (m_pModel && m_pModel->m_pCurrentMaterial) {
                    ++m_DataIt;
                    getColorRGBA(m_pModel->m_pCurrentMaterial->emissive);
                }
                break;
            case 't': // Transmission color
                if (m_pModel && m_pModel->m_pCurrentMaterial) {
                    ++m_DataIt;
                    getColorRGBA(m_pModel->m_pCurrentMaterial->transparent);
                }
                break;
            default:
                m_DataIt = skipLine<DataArrayIt>(m_DataIt, m_DataItEnd, m_uiLine);
                break;
        }
    }
    break;
case 'd': // Alpha value
    if (m_pModel && m_pModel->m_pCurrentMaterial) {
        ++m_DataIt;
        getFloatValue(m_pModel->m_pCurrentMaterial->alpha);
    }
    break;
case 'N':
    ++m_DataIt;
    if (m_DataIt != m_DataItEnd) {
        switch (*m_DataIt) {
            case 's': // Specular exponent
                if (m_pModel && m_pModel->m_pCurrentMaterial) {
                    ++m_DataIt;
                    getFloatValue(m_pModel->m_pCurrentMaterial->shineness);
                }
                break;
            case 'i': // Index of refraction
                if (m_pModel && m_pModel->m_pCurrentMaterial) {
                    ++m_DataIt;
                    getFloatValue(m_pModel->m_pCurrentMaterial->ior);
                }
                break;
            default:
                m_DataIt = skipLine<DataArrayIt>(m_DataIt, m_DataItEnd, m_uiLine);
                break;
        }
    }
    break;
case 'm': // Texture
    {
        ++m_DataIt;
        if (m_DataIt != m_DataItEnd && m_pModel && m_pModel->m_pCurrentMaterial) {
            getMaterialTexture();
        }
    }
    break;
case 'i': // Illumination model
    if (m_pModel && m_pModel->m_pCurrentMaterial) {
        ++m_DataIt;
        getIlluminationModel(m_pModel->m_pCurrentMaterial->illumination_model);
    }
    break;
case 'a': // Anisotropy
    if (m_pModel && m_pModel->m_pCurrentMaterial) {
        ++m_DataIt;
        getFloatValue(m_pModel->m_pCurrentMaterial->anisotropy);
    }
    break;
