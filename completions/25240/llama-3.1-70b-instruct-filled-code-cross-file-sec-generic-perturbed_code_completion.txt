case 'k': {
    // Parse material properties
    switch (*(++m_DataIt)) {
        case 'a': { // Ambient color
            getColorRGBA(m_pModel->m_pCurrentMaterial->m_Ambient);
        } break;
        case 'd': { // Diffuse color
            getColorRGBA(m_pModel->m_pCurrentMaterial->m_Diffuse);
        } break;
        case's': { // Specular color
            getColorRGBA(m_pModel->m_pCurrentMaterial->m_Specular);
        } break;
        case 'e': { // Emissive color
            getColorRGBA(m_pModel->m_pCurrentMaterial->m_Emissive);
        } break;
        case 't': { // Transmission filter
            getColorRGBA(m_pModel->m_pCurrentMaterial->m_Transparency);
        } break;
        case 'T': { // Transparency
            float fValue;
            if (getFloatValue(fValue)) {
                m_pModel->m_pCurrentMaterial->m_Opacity = 1.0f - fValue;
            }
        } break;
        case 'i': { // Illumination model
            int iValue;
            if (getIntValue(iValue)) {
                m_pModel->m_pCurrentMaterial->m_ShadingModel = (aiShadingMode)iValue;
            }
        } break;
        case 'N': { // Shininess
            float fValue;
            if (getFloatValue(fValue)) {
                m_pModel->m_pCurrentMaterial->m_Shininess = fValue;
            }
        } break;
        case 'n': { // Shininess strength
            float fValue;
            if (getFloatValue(fValue)) {
                m_pModel->m_pCurrentMaterial->m_ShininessStrength = fValue;
            }
        } break;
        case 'N': { // Normal mapping
            char *pStart = &(*m_DataIt);
            while (m_DataIt!= m_DataItEnd &&!isDataDefinitionEnd(m_DataIt)) {
                ++m_DataIt;
            }
            std::string strTextureName(pStart, &(*m_DataIt));
            m_pModel->m_pCurrentMaterial->m_NormalTexture = strTextureName;
        } break;
        case'm': { // Material name
            char *pStart = &(*m_DataIt);
            while (m_DataIt!= m_DataItEnd &&!isDataDefinitionEnd(m_DataIt)) {
                ++m_DataIt;
            }
            std::string strMaterialName(pStart, &(*m_DataIt));
            m_pModel->m_pCurrentMaterial->MaterialName = strMaterialName;
        } break;
        case 'n': { // Material name
            char *pStart = &(*m_DataIt);
            while (m_DataIt!= m_DataItEnd &&!isDataDefinitionEnd(m_DataIt)) {
                ++m_DataIt;
            }
            std::string strMaterialName(pStart, &(*m_DataIt));
            m_pModel->m_pCurrentMaterial->MaterialName = strMaterialName;
        } break;
        case 'b': { // Bump mapping
            char *pStart = &(*m_DataIt);
            while (m_DataIt!= m_DataItEnd &&!isDataDefinitionEnd(m_DataIt)) {
                ++m_DataIt;
            }
            std::string strTextureName(pStart, &(*m_DataIt));
            m_pModel->m_pCurrentMaterial->m_BumpTexture = strTextureName;
        } break;
        case 'd': { // Diffuse texture
            char *pStart = &(*m_DataIt);
            while (m_DataIt!= m_DataItEnd &&!isDataDefinitionEnd(m_DataIt)) {
                ++m_DataIt;
            }
            std::string strTextureName(pStart, &(*m_DataIt));
            m_pModel->m_pCurrentMaterial->m_DiffuseTexture = strTextureName;
        } break;
        default: {
            m_DataIt = skipLine<DataArrayIt>(m_DataIt, m_DataItEnd, m_uiLine);
        } break;
    }
} break;
