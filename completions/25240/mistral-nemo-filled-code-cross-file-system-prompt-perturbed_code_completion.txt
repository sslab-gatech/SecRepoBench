case 'k':
    // Parse material properties
    m_DataIt++;
    if (*m_DataIt != 'd') {
        m_DataIt = skipLine<DataArrayIt>(m_DataIt, m_DataItEnd, m_uiLine);
        break;
    }
    m_DataIt++;

    // Parse material name
    std::string matName;
    while (m_DataIt != m_DataItEnd && !isDataDefinitionEnd(m_DataIt)) {
        matName += *m_DataIt;
        m_DataIt++;
    }

    // Import material library
    getMaterialLib();

    // Create new material
    m_pModel->m_MaterialLib.push_back(matName);
    m_pModel->m_pCurrentMaterial = new aiMaterial();
    m_pModel->m_pCurrentMaterial->AddProperty(&aiMaterialProperty::New(aiMaterialProperty::COLOR, 0, aiColor3D(0.0f, 0.0f, 0.0f)));
    break;
