if (m_pModel->m_pCurrentMaterial) {
    switch (*m_DataIt) {
        case 'Ka':
            getColorRGBA(m_pModel->m_pCurrentMaterial->Ambient);
            break;
        case 'Kd':
            getColorRGBA(m_pModel->m_pCurrentMaterial->Diffuse);
            break;
        case 'Ks':
            getColorRGBA(m_pModel->m_pCurrentMaterial->Specular);
            break;
        case 'Ke':
            getColorRGBA(m_pModel->m_pCurrentMaterial->Emissive);
            break;
        case 'Kt':
            getColorRGBA(m_pModel->m_pCurrentMaterial->Transmission);
            break;
        case 'Tr':
            m_pModel->m_pCurrentMaterial->Transparency = (float)readFloat();
            break;
        case 'd':
            m_pModel->m_pCurrentMaterial->Alpha = (float)readFloat();
            break;
        case 'Ns':
            m_pModel->m_pCurrentMaterial->SpecularExponent = (float)readFloat();
            break;
        case 'Ni':
            m_pModel->m_pCurrentMaterial->IndexOfRefraction = (float)readFloat();
            break;

            m_pModel->m_pCurrentMaterial->IlluminationModel = (aiShadingMode)readInt();
            break;
        case 'anisot':
            m_pModel->m_pCurrentMaterial->Anisotropy = (float)readFloat();
            break;
        case 'map_Kd':
            m_pModel->m_pCurrentMaterial->DiffuseTexture = readString();
            break;
        // Add more cases for other material properties as needed
        default:
            break;
    }
    m_DataIt = skipLine<DataArrayIt>(m_DataIt, m_DataItEnd, m_uiLine);
}
