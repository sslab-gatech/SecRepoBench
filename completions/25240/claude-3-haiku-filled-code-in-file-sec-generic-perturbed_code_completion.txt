case 'k':
    // Ambient color
    if (*m_DataIt == 'a') {
        ++m_DataIt;
        getColorRGBA(&m_pModel->m_pCurrentMaterial->ambient);
    }
    // Diffuse color
    else if (*m_DataIt == 'd') {
        ++m_DataIt;
        getColorRGBA(&m_pModel->m_pCurrentMaterial->diffuse);
    }
    // Specular color
    else if (*m_DataIt == 's') {
        ++m_DataIt;
        getColorRGBA(&m_pModel->m_pCurrentMaterial->specular);
    }
    // Emissive color
    else if (*m_DataIt == 'e') {
        ++m_DataIt;
        getColorRGBA(&m_pModel->m_pCurrentMaterial->emissive);
    }
    // Transmission color
    else if (*m_DataIt == 'T') {
        ++m_DataIt;
        getColorRGBA(&m_pModel->m_pCurrentMaterial->transparent);
    }
    else {
        m_DataIt = skipLine<DataArrayIt>(m_DataIt, m_DataItEnd, m_uiLine);
    }
    break;
case 'N':
    // Specular exponent (shininess)
    if (*m_DataIt == 's') {
        ++m_DataIt;
        getFloatValue(m_pModel->m_pCurrentMaterial->shininess);
    }
    // Index of refraction
    else if (*m_DataIt == 'i') {
        ++m_DataIt;
        getFloatValue(m_pModel->m_pCurrentMaterial->ior);
    }
    else {
        m_DataIt = skipLine<DataArrayIt>(m_DataIt, m_DataItEnd, m_uiLine);
    }
    break;
case 'd':
    // Dissolve (transparency)
    getFloatValue(m_pModel->m_pCurrentMaterial->alpha);
    break;
case 'i':
    // Illumination model
    if (*m_DataIt == 'l') {
        ++m_DataIt;
        getIlluminationModel(m_pModel->m_pCurrentMaterial->illumination_model);
    }
    else {
        m_DataIt = skipLine<DataArrayIt>(m_DataIt, m_DataItEnd, m_uiLine);
    }
    break;
case 'm':
    // New material
    if (*m_DataIt == 't' || *m_DataIt == 'l') {
        createMaterial();
    }
    else {
        m_DataIt = skipLine<DataArrayIt>(m_DataIt, m_DataItEnd, m_uiLine);
    }
    break;
case 'T':
    // Anisotropy
    if (*m_DataIt == 'r') {
        ++m_DataIt;
        getFloatValue(m_pModel->m_pCurrentMaterial->anisotropy);
    }
    else {
        m_DataIt = skipLine<DataArrayIt>(m_DataIt, m_DataItEnd, m_uiLine);
    }
    break;
case 'M':
    // PBR Metallic
    if (*m_DataIt == 'e') {
        ++m_DataIt;
        getFloatValue(m_pModel->m_pCurrentMaterial->metallic);
    }
    else {
        m_DataIt = skipLine<DataArrayIt>(m_DataIt, m_DataItEnd, m_uiLine);
    }
    break;
case 'R':
    // PBR Roughness
    if (*m_DataIt == 'o') {
        ++m_DataIt;
        getFloatValue(m_pModel->m_pCurrentMaterial->roughness);
    }
    else {
        m_DataIt = skipLine<DataArrayIt>(m_DataIt, m_DataItEnd, m_uiLine);
    }
    break;
case 'S':
    // PBR Sheen
    if (*m_DataIt == 'h') {
        ++m_DataIt;
        getFloatValue(m_pModel->m_pCurrentMaterial->sheen);
    }
    else {
        m_DataIt = skipLine<DataArrayIt>(m_DataIt, m_DataItEnd, m_uiLine);
    }
    break;
case 'm':
    // Texture
    if (*m_DataIt == 'a') {
        getTexture();
    }
    else {
        m_DataIt = skipLine<DataArrayIt>(m_DataIt, m_DataItEnd, m_uiLine);
    }
    break;
