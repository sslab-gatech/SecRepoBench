{
                ++m_DataIt;
                if (m_DataIt == m_DataItEnd) {
                    return;
                }

                switch (*m_DataIt) {
                    case 'a': // Ambient color
                        ++m_DataIt;
                        getColorRGBA(m_pCurrentMaterial->m_Ambient);
                        break;
                    case 'd': // Diffuse color
                        ++m_DataIt;
                        getColorRGBA(m_pCurrentMaterial->m_Diffuse);
                        break;
                    case 's': // Specular color
                        ++m_DataIt;
                        getColorRGBA(m_pCurrentMaterial->m_Specular);
                        break;
                    case 'e': // Emissive color
                        ++m_DataIt;
                        getColorRGBA(m_pCurrentMaterial->m_Emissive);
                        break;
                    case 't': // Transmission filter
                        ++m_DataIt;
                        getColorRGBA(m_pCurrentMaterial->m_TransmissionFilter);
                        break;
                    default:
                        m_DataIt = skipLine<DataArrayIt>(m_DataIt, m_DataItEnd, m_uiLine);
                        break;
                }
            }
