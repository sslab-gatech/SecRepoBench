if (m_pCurrentMaterial == nullptr) {
    m_pCurrentMaterial = new aiMaterial();
}
m_DataIt++;
switch (*m_DataIt) {
    case 'a': // Ambient color
        getColorRGBA(m_pCurrentMaterial->mAmbient);
        break;
    case 'd': // Diffuse color
        getColorRGBA(m_pCurrentMaterial->mDiffuse);
        break;
    case 's': // Specular color
        getColorRGBA(m_pCurrentMaterial->mSpecular);
        break;
    case 'e': // Emissive color
        getColorRGBA(m_pCurrentMaterial->mEmissive);
        break;
    case 't': // Transmission color
        getColorRGBA(m_pCurrentMaterial->mTransparency);
        break;
    default:
        m_DataIt = skipLine<DataArrayIt>(m_DataIt, m_DataItEnd, m_uiLine);
        break;
}
