case 'k':
    // Ambient color
    if (*m_DataIt == 'a') {
        ++m_DataIt;
        getColorRGBA(&m_pModel->m_pCurrentMaterial->ambient);
    }
    // Diffuse color
    else if (*m_DataIt == 'd') {
        ++m_DataIt;
        getColorRGBA(&m_pModel->m_pCurrentMaterial->diffuse);
    }
    // Specular color
    else if (*m_DataIt == 's') {
        ++m_DataIt;
        getColorRGBA(&m_pModel->m_pCurrentMaterial->specular);
    }
    // Emissive color
    else if (*m_DataIt == 'e') {
        ++m_DataIt;
        getColorRGBA(&m_pModel->m_pCurrentMaterial->emissive);
    }
    // Transmission filter color
    else if (*m_DataIt == 'f') {
        ++m_DataIt;
        getColorRGBA(&m_pModel->m_pCurrentMaterial->transparent);
        m_pModel->m_pCurrentMaterial->alpha = 1.0f - m_pModel->m_pCurrentMaterial->transparent.r;
    }
    else {
        m_DataIt = skipLine<DataArrayIt>(m_DataIt, m_DataItEnd, m_uiLine);
    }
    break;
case 'N':
    // Specular exponent (shininess)
    if (*m_DataIt == 's') {
        ++m_DataIt;
        getFloatValue(m_pModel->m_pCurrentMaterial->shininess);
    }
    // Index of refraction
    else if (*m_DataIt == 'i') {
        ++m_DataIt;
        getFloatValue(m_pModel->m_pCurrentMaterial->ior);
    }
    else {
        m_DataIt = skipLine<DataArrayIt>(m_DataIt, m_DataItEnd, m_uiLine);
    }
    break;
case 'd':
    // Dissolve (transparency)
    getFloatValue(m_pModel->m_pCurrentMaterial->alpha);
    break;
case 'i':
    // Illumination model
    if (*m_DataIt == 'l') {
        ++m_DataIt;
        getIlluminationModel(m_pModel->m_pCurrentMaterial->illumination_model);
    }
    else {
        m_DataIt = skipLine<DataArrayIt>(m_DataIt, m_DataItEnd, m_uiLine);
    }
    break;
case 'm':
    // New material
    createMaterial();
    break;
case 'T':
    // Texture
    if (*m_DataIt == 'r') {
        ++m_DataIt;
        getTexture();
    }
    else {
        m_DataIt = skipLine<DataArrayIt>(m_DataIt, m_DataItEnd, m_uiLine);
    }
    break;
case 'a':
    // Anisotropy
    if (*m_DataIt == 's') {
        ++m_DataIt;
        getFloatValue(m_pModel->m_pCurrentMaterial->anisotropy);
    }
    else {
        m_DataIt = skipLine<DataArrayIt>(m_DataIt, m_DataItEnd, m_uiLine);
    }
    break;
