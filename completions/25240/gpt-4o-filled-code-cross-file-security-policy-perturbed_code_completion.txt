                char nextChar = *(m_DataIt + 1);
                if (nextChar == 'a' || nextChar == 'd' || nextChar == 's' || nextChar == 'e' || nextChar == 't') {
                    // Ensure current material is not null
                    if (m_pModel->m_pCurrentMaterial) {
                        Maybe<aiColor3D> color;
                        getColorRGBA(color);
                        switch (nextChar) {
                            case 'a':
                                m_pModel->m_pCurrentMaterial->m_Ambient = color;
                                break;
                            case 'd':
                                m_pModel->m_pCurrentMaterial->m_Diffuse = color;
                                break;
                            case 's':
                                m_pModel->m_pCurrentMaterial->m_Specular = color;
                                break;
                            case 'e':
                                m_pModel->m_pCurrentMaterial->m_Emissive = color;
                                break;
                            case 't':
                                m_pModel->m_pCurrentMaterial->m_Transmission = color;
                                break;
                        }
                    }
                    m_DataIt = skipLine<DataArrayIt>(m_DataIt, m_DataItEnd, m_uiLine);
