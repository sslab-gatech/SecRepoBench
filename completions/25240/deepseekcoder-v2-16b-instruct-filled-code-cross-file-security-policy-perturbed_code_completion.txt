case 'k':
        // Parses material properties from a buffer when loading material files. The code processes
        // different material properties based on the current character in the buffer. It handles:
        // 1. Colors: Ambient, Diffuse, Specular, Emissive, and Transmission.
        // 2. Transparency and Alpha values.
        // 3. Specular exponent and Index of Refraction (IOR).
        // 4. Materials: Creates a new material when needed.
        // 5. Textures: Handles various texture types like diffuse, bump, etc.
        // 6. Illumination models and anisotropy.
        // The parsing is done by moving through the buffer character by character, determining
        // which property is being specified, and calling the appropriate function to parse
        // and assign the value to the current material being processed.

        // Check if the current material is null
        if (!m_pModel->m_pCurrentMaterial) {
            m_DataIt = skipLine<DataArrayIt>(m_DataIt, m_DataItEnd, m_uiLine);
            break;
        }

        // Move to the next character
        ++m_DataIt;

        // Parse the material property
        switch (*m_DataIt) {
            case 'a': // Ambient color
                getColorRGBA(m_pModel->m_pCurrentMaterial->AmbientColor);
                break;
            case 'd': // Diffuse color
                getColorRGBA(m_pModel->m_pCurrentMaterial->DiffuseColor);
                break;
            case 's': // Specular color
                getColorRGBA(m_pModel->m_pCurrentMaterial->SpecularColor);
                break;
            case 'e': // Emissive color
                getColorRGBA(m_pModel->m_pCurrentMaterial->EmissiveColor);
                break;
            case 't': // Transmission color
                getColorRGBA(m_pModel->m_pCurrentMaterial->TransmissionColor);
                break;
            case 'A': // Alpha value
                m_pModel->m_pCurrentMaterial->m_fAlpha = getFloat();
                break;
            case 'N': // Specular exponent (shininess)
                m_pModel->m_pCurrentMaterial->m_fShininess = getFloat();
                break;
            case 'I': // Index of Refraction (IOR)
                m_pModel->m_pCurrentMaterial->m_fIor = getFloat();
                break;
            case 'm': // Material name
                m_pModel->m_pCurrentMaterial->MaterialName = getNextToken<DataArrayIt>(m_DataIt, m_DataItEnd);
                break;
            case 'n': // Texture file name
                m_pModel->m_pCurrentMaterial->map_Kd = getNextToken<DataArrayIt>(m_DataIt, m_DataItEnd);
                break;
            case 'b': // Bump map file name
                m_pModel->m_pCurrentMaterial->map_Bump = getNextToken<DataArrayIt>(m_DataIt, m_DataItEnd);
                break;
            case 'l': // Illumination model
                m_pModel->m_pCurrentMaterial->Illum = getInt();
                break;
            case 'a': // Anisotropy
                m_pModel->m_pCurrentMaterial->m_fAnisotropy = getFloat();
                break;
            default:
                m_DataIt = skipLine<DataArrayIt>(m_DataIt, m_DataItEnd, m_uiLine);
                break;
        }
        break;
