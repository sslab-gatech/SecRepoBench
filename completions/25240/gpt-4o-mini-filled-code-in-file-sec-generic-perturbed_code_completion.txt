case 'K': {
                if (m_DataIt + 1 < m_DataItEnd) {
                    if (*(m_DataIt + 1) == 'd') {
                        // Diffuse color
                        m_DataIt += 2; // Skip 'Kd'
                        aiColor3D color;
                        getColorRGBA(&color);
                        m_pModel->m_pCurrentMaterial->colorDiffuse = color;
                    } else if (*(m_DataIt + 1) == 'a') {
                        // Ambient color
                        m_DataIt += 2; // Skip 'Ka'
                        aiColor3D color;
                        getColorRGBA(&color);
                        m_pModel->m_pCurrentMaterial->colorAmbient = color;
                    } else if (*(m_DataIt + 1) == 's') {
                        // Specular color
                        m_DataIt += 2; // Skip 'Ks'
                        aiColor3D color;
                        getColorRGBA(&color);
                        m_pModel->m_pCurrentMaterial->colorSpecular = color;
                    } else if (*(m_DataIt + 1) == 'e') {
                        // Emissive color
                        m_DataIt += 2; // Skip 'Ke'
                        aiColor3D color;
                        getColorRGBA(&color);
                        m_pModel->m_pCurrentMaterial->colorEmissive = color;
                    }
                }
            } break;

            case 'N': {
                if (*(m_DataIt + 1) == 's') {
                    // Specular exponent
                    m_DataIt += 2; // Skip 'Ns'
                    getFloatValue(m_pModel->m_pCurrentMaterial->shininess);
                }
            } break;

            case 'd': {
                // Transparency
                m_DataIt += 1; // Skip 'd'
                getFloatValue(m_pModel->m_pCurrentMaterial->transparency);
            } break;

            case 'i': {
                if (*(m_DataIt + 1) == 'o') {
                    // Index of Refraction
                    m_DataIt += 2; // Skip 'ior'
                    getFloatValue(m_pModel->m_pCurrentMaterial->indexOfRefraction);
                }
            } break;

            case 'm': {
                if (*(m_DataIt + 1) == 'a') {
                    // New material
                    m_DataIt += 2; // Skip 'newmtl'
                    createMaterial();
                }
            } break;

            case 't': {
                if (*(m_DataIt + 1) == 'r') {
                    // Transparency
                    m_DataIt += 2; // Skip 'tr'
                    getFloatValue(m_pModel->m_pCurrentMaterial->transparency);
                }
            } break;

            case 'b': {
                if (*(m_DataIt + 1) == 'u') {
                    // Bump mapping
                    m_DataIt += 2; // Skip 'bu'
                    getTexture();
                }
            } break;

            case 'm': {
                if (*(m_DataIt + 1) == 'a') {
                    // Material name
                    m_DataIt += 2; // Skip 'ma'
                    createMaterial();
                }
            } break;

            default: {
                m_DataIt = skipLine<DataArrayIt>(m_DataIt, m_DataItEnd, m_uiLine);
            } break;
