case 'k': {
    if (m_pModel->m_pCurrentMaterial == nullptr) {
        m_pModel->m_pCurrentMaterial = new ObjFile::Material();
        m_pModel->m_pCurrentMaterial->MaterialName = "DefaultMaterial";
    }

    switch (*++m_DataIt) {
        case 'a': {
            // Ambient color
            if (m_pModel->m_pCurrentMaterial->m_AmbientColor.empty()) {
                getColorRGBA(m_pModel->m_pCurrentMaterial->m_AmbientColor);
            }
            break;
        }
        case 'd': {
            // Diffuse color
            if (m_pModel->m_pCurrentMaterial->m_DiffuseColor.empty()) {
                getColorRGBA(m_pModel->m_pCurrentMaterial->m_DiffuseColor);
            }
            break;
        }
        case's': {
            // Specular color
            if (m_pModel->m_pCurrentMaterial->m_SpecularColor.empty()) {
                getColorRGBA(m_pModel->m_pCurrentMaterial->m_SpecularColor);
            }
            break;
        }
        case 'e': {
            // Emissive color
            if (m_pModel->m_pCurrentMaterial->m_EmissiveColor.empty()) {
                getColorRGBA(m_pModel->m_pCurrentMaterial->m_EmissiveColor);
            }
            break;
        }
        case 't': {
            // Transmission color
            if (m_pModel->m_pCurrentMaterial->m_TransmissionColor.empty()) {
                getColorRGBA(m_pModel->m_pCurrentMaterial->m_TransmissionColor);
            }
            break;
        }
        case 'N': {
            // Specular exponent
            if (m_pModel->m_pCurrentMaterial->m_fSpecularExponent == 0.0f) {
                m_pModel->m_pCurrentMaterial->m_fSpecularExponent = getNextFloat(m_DataIt, m_DataItEnd);
            }
            break;
        }
        case 'i': {
            // Index of Refraction (IOR)
            if (m_pModel->m_pCurrentMaterial->m_fIOR == 0.0f) {
                m_pModel->m_pCurrentMaterial->m_fIOR = getNextFloat(m_DataIt, m_DataItEnd);
            }
            break;
        }
        case'm': {
            // Material name
            if (m_pModel->m_pCurrentMaterial->MaterialName.empty()) {
                m_pModel->m_pCurrentMaterial->MaterialName = getNextToken<std::string>(m_DataIt, m_DataItEnd);
            }
            break;
        }
        case 'T': {
            // Texture name
            if (m_pModel->m_pCurrentMaterial->m_strTextureName.empty()) {
                m_pModel->m_pCurrentMaterial->m_strTextureName = getNextToken<std::string>(m_DataIt, m_DataItEnd);
            }
            break;
        }
        case 'I': {
            // Illumination model
            if (m_pModel->m_pCurrentMaterial->m_strIlluminationModel.empty()) {
                m_pModel->m_pCurrentMaterial->m_strIlluminationModel = getNextToken<std::string>(m_DataIt, m_DataItEnd);
            }
            break;
        }
        case 'a': {
            // Anisotropy
            if (m_pModel->m_pCurrentMaterial->m_fAnisotropy == 0.0f) {
                m_pModel->m_pCurrentMaterial->m_fAnisotropy = getNextFloat(m_DataIt, m_DataItEnd);
            }
            break;
        }
        default: {
            m_DataIt = skipLine<DataArrayIt>(m_DataIt, m_DataItEnd, m_uiLine);
        } break;
    }
} break;
