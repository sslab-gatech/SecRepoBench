// <MASK>
{
    std::string strPropertyName;
    getNextToken<DataArrayIt>(m_DataIt, m_DataItEnd, strPropertyName);

    if (strPropertyName == "Ka") {
        getColorRGBA(m_pModel->m_pCurrentMaterial->ambient);
    } else if (strPropertyName == "Kd") {
        getColorRGBA(m_pModel->m_pCurrentMaterial->diffuse);
    } else if (strPropertyName == "Ks") {
        getColorRGBA(m_pModel->m_pCurrentMaterial->specular);
    } else if (strPropertyName == "Ke") {
        getColorRGBA(m_pModel->m_pCurrentMaterial->emissive);
    } else if (strPropertyName == "Kt") {
        getColorRGBA(m_pModel->m_pCurrentMaterial->transparent);
    } else if (strPropertyName == "d" || strPropertyName == "Tr") {
        float d;
        getFloat(d);
        m_pModel->m_pCurrentMaterial->alpha = d;
    } else if (strPropertyName == "Ns") {
        float ns;
        getFloat(ns);
        m_pModel->m_pCurrentMaterial->shininess = ns;
    } else if (strPropertyName == "Ni") {
        float ni;
        getFloat(ni);
        m_pModel->m_pCurrentMaterial->ior = ni;
    } else if (strPropertyName == "newmtl") {
        std::string strName;
        getNextToken<DataArrayIt>(m_DataIt, m_DataItEnd, strName);
        createMaterial(strName);
    } else if (strPropertyName == "map_Kd") {
        std::string strTexture;
        getNextToken<DataArrayIt>(m_DataIt, m_DataItEnd, strTexture);
        m_pModel->m_pCurrentMaterial->textureDiffuse.file = strTexture;
    } else if (strPropertyName == "map_Ks") {
        std::string strTexture;
        getNextToken<DataArrayIt>(m_DataIt, m_DataItEnd, strTexture);
        m_pModel->m_pCurrentMaterial->textureSpecular.file = strTexture;
    } else if (strPropertyName == "map_Bump" || strPropertyName == "bump") {
        std::string strTexture;
        getNextToken<DataArrayIt>(m_DataIt, m_DataItEnd, strTexture);
        m_pModel->m_pCurrentMaterial->textureBump.file = strTexture;
    } else if (strPropertyName == "map_Disp" || strPropertyName == "disp") {
        std::string strTexture;
        getNextToken<DataArrayIt>(m_DataIt, m_DataItEnd, strTexture);
        m_pModel->m_pCurrentMaterial->textureDisplacement.file = strTexture;
    } else if (strPropertyName == "illum") {
        int iIllum;
        getInt(iIllum);
        m_pModel->m_pCurrentMaterial->illumination = iIllum;
    } else if (strPropertyName == "aniso") {
        float aniso;
        getFloat(aniso);
        m_pModel->m_pCurrentMaterial->anisotropy = aniso;
    } else {
        ASSIMP_LOG_WARN("OBJ: Unknown property: ", strPropertyName);
        m_DataIt = skipLine<DataArrayIt>(m_DataIt, m_DataItEnd, m_uiLine);
    }
}
