case 'K': {
                ++m_DataIt;
                if (m_DataIt == m_DataItEnd) {
                    break;
                }

                switch (*m_DataIt) {
                    case 'a': {
                        ++m_DataIt;
                        aiColor3D color;
                        getColorRGBA(&color);
                        m_pModel->m_pCurrentMaterial->ambient = color;
                    } break;

                    case 'd': {
                        ++m_DataIt;
                        aiColor3D color;
                        getColorRGBA(&color);
                        m_pModel->m_pCurrentMaterial->diffuse = color;
                    } break;

                    case 's': {
                        ++m_DataIt;
                        aiColor3D color;
                        getColorRGBA(&color);
                        m_pModel->m_pCurrentMaterial->specular = color;
                    } break;

                    case 'e': {
                        ++m_DataIt;
                        aiColor3D color;
                        getColorRGBA(&color);
                        m_pModel->m_pCurrentMaterial->emissive = color;
                    } break;

                    case 't': {
                        ++m_DataIt;
                        aiColor3D color;
                        getColorRGBA(&color);
                        m_pModel->m_pCurrentMaterial->transmission = color;
                    } break;

                    default:
                        m_DataIt = skipLine<DataArrayIt>(m_DataIt, m_DataItEnd, m_uiLine);
                        break;
                }
            } break;

            case 'N': {
                ++m_DataIt;
                if (m_DataIt == m_DataItEnd) {
                    break;
                }

                switch (*m_DataIt) {
                    case 's': {
                        ++m_DataIt;
                        ai_real value;
                        getFloatValue(value);
                        m_pModel->m_pCurrentMaterial->shininess = value;
                    } break;

                    case 'i': {
                        ++m_DataIt;
                        ai_real value;
                        getFloatValue(value);
                        m_pModel->m_pCurrentMaterial->ior = value;
                    } break;

                    default:
                        m_DataIt = skipLine<DataArrayIt>(m_DataIt, m_DataItEnd, m_uiLine);
                        break;
                }
            } break;

            case 'd': {
                ++m_DataIt;
                ai_real value;
                getFloatValue(value);
                m_pModel->m_pCurrentMaterial->alpha = value;
            } break;

            case 'T': {
                ++m_DataIt;
                if (m_DataIt == m_DataItEnd) {
                    break;
                }

                switch (*m_DataIt) {
                    case 'r': {
                        ++m_DataIt;
                        ai_real value;
                        getFloatValue(value);
                        m_pModel->m_pCurrentMaterial->alpha = 1.0f - value;
                    } break;

                    default:
                        m_DataIt = skipLine<DataArrayIt>(m_DataIt, m_DataItEnd, m_uiLine);
                        break;
                }
            } break;

            case 'i': {
                ++m_DataIt;
                int illum_model;
                getIlluminationModel(illum_model);
                m_pModel->m_pCurrentMaterial->illumination_model = illum_model;
            } break;

            case 'm': {
                getTexture();
            } break;

            case 'n': {
                createMaterial();
            } break;
