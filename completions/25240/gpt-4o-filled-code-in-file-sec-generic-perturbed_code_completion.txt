case 'n': {
                // newmtl <name>
                if (!ASSIMP_strincmp(&(*m_DataIt), "newmtl", 6)) {
                    createMaterial();
                }
                m_DataIt = skipLine<DataArrayIt>(m_DataIt, m_DataItEnd, m_uiLine);
            } break;

            case 'K': {
                // Kd, Ka, Ks, Ke, Tf
                aiColor3D color;
                switch (*(m_DataIt + 1)) {
                    case 'd': // Kd
                        getColorRGBA(&color);
                        m_pModel->m_pCurrentMaterial->diffuse = color;
                        break;
                    case 'a': // Ka
                        getColorRGBA(&color);
                        m_pModel->m_pCurrentMaterial->ambient = color;
                        break;
                    case 's': // Ks
                        getColorRGBA(&color);
                        m_pModel->m_pCurrentMaterial->specular = color;
                        break;
                    case 'e': // Ke
                        getColorRGBA(&color);
                        m_pModel->m_pCurrentMaterial->emissive = color;
                        break;
                    case 'f': // Tf
                        getColorRGBA(&color);
                        m_pModel->m_pCurrentMaterial->transmissionFilter = color;
                        break;
                    default:
                        break;
                }
                m_DataIt = skipLine<DataArrayIt>(m_DataIt, m_DataItEnd, m_uiLine);
            } break;

            case 'T': {
                // Tr
                if (*(m_DataIt + 1) == 'r') {
                    ai_real value;
                    getFloatValue(value);
                    m_pModel->m_pCurrentMaterial->alpha = 1.0f - value;
                }
                m_DataIt = skipLine<DataArrayIt>(m_DataIt, m_DataItEnd, m_uiLine);
            } break;

            case 'd': {
                // d
                ai_real value;
                getFloatValue(value);
                m_pModel->m_pCurrentMaterial->alpha = value;
                m_DataIt = skipLine<DataArrayIt>(m_DataIt, m_DataItEnd, m_uiLine);
            } break;

            case 'N': {
                // Ns, Ni
                switch (*(m_DataIt + 1)) {
                    case 's': // Ns
                        ai_real value;
                        getFloatValue(value);
                        m_pModel->m_pCurrentMaterial->shineness = value;
                        break;
                    case 'i': // Ni
                        ai_real value;
                        getFloatValue(value);
                        m_pModel->m_pCurrentMaterial->ior = value;
                        break;
                    default:
                        break;
                }
                m_DataIt = skipLine<DataArrayIt>(m_DataIt, m_DataItEnd, m_uiLine);
            } break;

            case 'i': {
                // illum
                if (!ASSIMP_strincmp(&(*m_DataIt), "illum", 5)) {
                    int illum_model;
                    getIlluminationModel(illum_model);
                    m_pModel->m_pCurrentMaterial->illumination_model = illum_model;
                }
                m_DataIt = skipLine<DataArrayIt>(m_DataIt, m_DataItEnd, m_uiLine);
            } break;

            case 'm': {
                // map_Kd, map_Ka, map_Ks, map_d, map_bump, bump, map_disp, disp, map_ns, map_Pr, map_Pm, map_Ps
                getTexture();
                m_DataIt = skipLine<DataArrayIt>(m_DataIt, m_DataItEnd, m_uiLine);
            } break;
