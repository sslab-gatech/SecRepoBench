case 'k':
{
    // Extract the token following 'k' to determine the material property type
    m_DataIt = CopyNextWord<DataArrayIt>(m_DataIt, m_DataItEnd, &m_buffer[0], BUFFERSIZE);

    if (!ASSIMP_strincmp(&m_buffer[0], "Ka", 2)) {
        // Ambient color
        getColorRGBA(&m_pModel->m_pCurrentMaterial->ambient);
    } else if (!ASSIMP_strincmp(&m_buffer[0], "Kd", 2)) {
        // Diffuse color
        getColorRGBA(&m_pModel->m_pCurrentMaterial->diffuse);
    } else if (!ASSIMP_strincmp(&m_buffer[0], "Ks", 2)) {
        // Specular color
        getColorRGBA(&m_pModel->m_pCurrentMaterial->specular);
    } else if (!ASSIMP_strincmp(&m_buffer[0], "Ke", 2)) {
        // Emissive color
        getColorRGBA(&m_pModel->m_pCurrentMaterial->emissive);
    } else if (!ASSIMP_strincmp(&m_buffer[0], "Kt", 2)) {
        // Transmission color (non-standard, but used by some exporters)
        getColorRGBA(&m_pModel->m_pCurrentMaterial->transmissionFilter);
    } else {
        // Unrecognized 'k' property, skip the entire line
        m_DataIt = skipLine<DataArrayIt>(m_DataIt, m_DataItEnd, m_uiLine);
    }
}
break;
