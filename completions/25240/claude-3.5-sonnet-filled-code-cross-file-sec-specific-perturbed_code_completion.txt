case 'K':

        ++m_DataIt;

        getColorRGBA(m_pModel->m_pCurrentMaterial->specular);
    } else if (*(m_DataIt + 1) == 'e') {
        ++m_DataIt;
        ++m_DataIt;
        getColorRGBA(m_pModel->m_pCurrentMaterial->emissive);
    } else {
        m_DataIt = skipLine<DataArrayIt>(m_DataIt, m_DataItEnd, m_uiLine);
    }

case 'd':

case 'T':

case 'n':
    if (*(m_DataIt + 1) == 'e' && *(m_DataIt + 2) == 'w') {
        m_DataIt = getNextToken<DataArrayIt>(m_DataIt, m_DataItEnd);
        if (m_DataIt == m_DataItEnd) {
            return;
        }
        m_pModel->m_pCurrentMaterial = new ObjFile::Material();
        m_pModel->m_pCurrentMaterial->MaterialName.Set(m_DataIt, static_cast<unsigned int>(m_DataItEnd - m_DataIt));
        m_pModel->m_MaterialLib.push_back(std::string(m_DataIt, m_DataItEnd));
        m_DataIt = skipLine<DataArrayIt>(m_DataIt, m_DataItEnd, m_uiLine);
    }
    break;
case 'm':
    if (*(m_DataIt + 1) == 'a' && *(m_DataIt + 2) == 'p') {
        m_DataIt += 3;
        if (*m_DataIt == '_') {
            ++m_DataIt;
            if (*m_DataIt == 'K') {
                ++m_DataIt;
                switch (*m_DataIt) {
                    case 'd':
                        getTexture(m_pModel->m_pCurrentMaterial->texture);
                        break;
                    case 's':
                        getTexture(m_pModel->m_pCurrentMaterial->specularTexture);
                        break;
                    case 'a':
                        getTexture(m_pModel->m_pCurrentMaterial->ambientTexture);
                        break;
                    case 'e':
                        getTexture(m_pModel->m_pCurrentMaterial->emissiveTexture);
                        break;
                    default:
                        m_DataIt = skipLine<DataArrayIt>(m_DataIt, m_DataItEnd, m_uiLine);
                        break;
                }
            } else if (*m_DataIt == 'b' && *(m_DataIt + 1) == 'u' && *(m_DataIt + 2) == 'm' && *(m_DataIt + 3) == 'p') {
                m_DataIt += 4;
                getTexture(m_pModel->m_pCurrentMaterial->bumpTexture);
            } else {
                m_DataIt = skipLine<DataArrayIt>(m_DataIt, m_DataItEnd, m_uiLine);
            }
        } else {
            m_DataIt = skipLine<DataArrayIt>(m_DataIt, m_DataItEnd, m_uiLine);
        }
    } else {
        m_DataIt = skipLine<DataArrayIt>(m_DataIt, m_DataItEnd, m_uiLine);
    }
    break;
