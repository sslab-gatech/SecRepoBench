++m_DataIt;
if(m_DataIt == m_DataItEnd)
    break; // unexpected end of data

// Read the property identifier (e.g., 'a' for ambient, 'd' for diffuse, etc.)
char typeChar = *m_DataIt;
++m_DataIt; // consume the identifier

// Skip any whitespace
while(m_DataIt != m_DataItEnd && isspace(*m_DataIt)) {
    ++m_DataIt;
}

Maybe<aiColor3D> colorValue;
getColorRGBA(colorValue);

switch(typeChar) {
    case 'a': // Ambient color
        m_pCurrentMaterial->ColorAmbient = colorValue;
        break;
    case 'd': // Diffuse color
        m_pCurrentMaterial->ColorDiffuse = colorValue;
        break;
    case 's': // Specular color
        m_pCurrentMaterial->ColorSpecular = colorValue;
        break;
    case 'e': // Emissive color
        m_pCurrentMaterial->ColorEmissive = colorValue;
        break;
    default:
        m_DataIt = skipLine(m_DataIt, m_DataItEnd, m_uiLine);
        break;
}
