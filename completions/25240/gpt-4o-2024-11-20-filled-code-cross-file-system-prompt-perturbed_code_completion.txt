{
                ++m_DataIt;
                if (m_DataIt == m_DataItEnd) {
                    return;
                }

                switch (*m_DataIt) {
                    case 'a': { // Ambient color
                        ++m_DataIt;
                        Maybe<aiColor3D> color;
                        getColorRGBA(color);
                        if (color) {
                            m_pCurrentMaterial->Ambient = *color;
                        }
                    } break;

                    case 'd': { // Diffuse color
                        ++m_DataIt;
                        Maybe<aiColor3D> color;
                        getColorRGBA(color);
                        if (color) {
                            m_pCurrentMaterial->Diffuse = *color;
                        }
                    } break;

                    case 's': { // Specular color
                        ++m_DataIt;
                        Maybe<aiColor3D> color;
                        getColorRGBA(color);
                        if (color) {
                            m_pCurrentMaterial->Specular = *color;
                        }
                    } break;

                    case 'e': { // Emissive color
                        ++m_DataIt;
                        Maybe<aiColor3D> color;
                        getColorRGBA(color);
                        if (color) {
                            m_pCurrentMaterial->Emissive = *color;
                        }
                    } break;

                    case 't': { // Transmission filter
                        ++m_DataIt;
                        Maybe<aiColor3D> color;
                        getColorRGBA(color);
                        if (color) {
                            m_pCurrentMaterial->TransmissionFilter = *color;
                        }
                    } break;

                    default: {
                        m_DataIt = skipLine<DataArrayIt>(m_DataIt, m_DataItEnd, m_uiLine);
                    } break;
                }
            }
