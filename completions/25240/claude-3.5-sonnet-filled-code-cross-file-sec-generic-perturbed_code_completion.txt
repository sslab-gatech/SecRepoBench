case 'K':
    {
        ++m_DataIt;
        if (*m_DataIt == 'a') {
            ++m_DataIt;
            getColorRGBA(m_pModel->m_pCurrentMaterial->ambient);
        } else if (*m_DataIt == 'd') {
            ++m_DataIt;
            getColorRGBA(m_pModel->m_pCurrentMaterial->diffuse);
        } else if (*m_DataIt == 's') {
            ++m_DataIt;
            getColorRGBA(m_pModel->m_pCurrentMaterial->specular);
        } else if (*m_DataIt == 'e') {
            ++m_DataIt;
            getColorRGBA(m_pModel->m_pCurrentMaterial->emissive);
        } else if (*m_DataIt == 't') {
            ++m_DataIt;
            getColorRGBA(m_pModel->m_pCurrentMaterial->transparent);
        } else {
            m_DataIt = skipLine<DataArrayIt>(m_DataIt, m_DataItEnd, m_uiLine);
        }
    }
    break;
case 'd':
    {
        ++m_DataIt;
        getFloatValue(m_pModel->m_pCurrentMaterial->alpha);
    }
    break;
case 'N':
    {
        ++m_DataIt;
        switch(*m_DataIt) {
            case 's':
                ++m_DataIt;
                getFloatValue(m_pModel->m_pCurrentMaterial->shininess);
                break;
            case 'i':
                ++m_DataIt;
                getFloatValue(m_pModel->m_pCurrentMaterial->ior);
                break;
            default:
                m_DataIt = skipLine<DataArrayIt>(m_DataIt, m_DataItEnd, m_uiLine);
                break;
        }
    }
    break;
case 'n':
    {
        ++m_DataIt;
        if (*m_DataIt == 'e') {
            ++m_DataIt;
            if (*m_DataIt == 'w') {
                ++m_DataIt;
                createMaterial();
            }
        }
        m_DataIt = skipLine<DataArrayIt>(m_DataIt, m_DataItEnd, m_uiLine);
    }
    break;
case 'm':
    {
        ++m_DataIt;
        if (strncmp(m_DataIt, "ap_", 3) == 0) {
            m_DataIt += 3;
            if (strncmp(m_DataIt, "Kd", 2) == 0) {
                m_DataIt = getTexture(m_DataIt, m_DataItEnd, "diffuse", m_pModel->m_pCurrentMaterial);
            } else if (strncmp(m_DataIt, "Ks", 2) == 0) {
                m_DataIt = getTexture(m_DataIt, m_DataItEnd, "specular", m_pModel->m_pCurrentMaterial);
            } else if (strncmp(m_DataIt, "Bump", 4) == 0) {
                m_DataIt = getTexture(m_DataIt, m_DataItEnd, "bump", m_pModel->m_pCurrentMaterial);
            } else {
                m_DataIt = skipLine<DataArrayIt>(m_DataIt, m_DataItEnd, m_uiLine);
            }
        } else {
            m_DataIt = skipLine<DataArrayIt>(m_DataIt, m_DataItEnd, m_uiLine);
        }
    }
    break;
case 'i':
    {
        ++m_DataIt;
        if (*m_DataIt == 'l') {
            ++m_DataIt;
            int illuminationModel;
            getIntValue(illuminationModel);
            m_pModel->m_pCurrentMaterial->illumination_model = illuminationModel;
        }
    }
    break;
case 'a':
    {
        ++m_DataIt;
        if (*m_DataIt == 'n') {
            ++m_DataIt;
            if (*m_DataIt == 'i') {
                ++m_DataIt;
                getFloatValue(m_pModel->m_pCurrentMaterial->anisotropy);
            }
        }
    }
    break;
