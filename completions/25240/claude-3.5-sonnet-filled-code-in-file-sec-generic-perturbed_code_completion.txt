case 'K':
    // Ambient color
    if ('a' == m_DataIt[1]) {
        ++m_DataIt;
        getColorRGBA(&m_pModel->m_pCurrentMaterial->ambient);
    }
    // Diffuse color
    else if ('d' == m_DataIt[1]) {
        ++m_DataIt;
        getColorRGBA(&m_pModel->m_pCurrentMaterial->diffuse);
    }
    // Specular color
    else if ('s' == m_DataIt[1]) {
        ++m_DataIt;
        getColorRGBA(&m_pModel->m_pCurrentMaterial->specular);
    }
    // Emissive color
    else if ('e' == m_DataIt[1]) {
        ++m_DataIt;
        getColorRGBA(&m_pModel->m_pCurrentMaterial->emissive);
    }
    m_DataIt = skipLine<DataArrayIt>(m_DataIt, m_DataItEnd, m_uiLine);
    break;
case 'd':
    // Alpha value
    getFloatValue(m_pModel->m_pCurrentMaterial->alpha);
    m_DataIt = skipLine<DataArrayIt>(m_DataIt, m_DataItEnd, m_uiLine);
    break;
case 'N':
    // Shineness
    if ('s' == m_DataIt[1]) {
        ++m_DataIt;
        getFloatValue(m_pModel->m_pCurrentMaterial->shineness);
    }
    m_DataIt = skipLine<DataArrayIt>(m_DataIt, m_DataItEnd, m_uiLine);
    break;
case 'n':
    // New material
    if ('e' == m_DataIt[1] && 'w' == m_DataIt[2] && 'm' == m_DataIt[3] &&
        't' == m_DataIt[4] && 'l' == m_DataIt[5]) {
        createMaterial();
    }
    m_DataIt = skipLine<DataArrayIt>(m_DataIt, m_DataItEnd, m_uiLine);
    break;
case 'i':
    // Illumination model
    if ('l' == m_DataIt[1] && 'l' == m_DataIt[2] && 'u' == m_DataIt[3] &&
        'm' == m_DataIt[4]) {
        getIlluminationModel(m_pModel->m_pCurrentMaterial->illumination_model);
    }
    m_DataIt = skipLine<DataArrayIt>(m_DataIt, m_DataItEnd, m_uiLine);
    break;
case 'm':
    // Texture
    if ('a' == m_DataIt[1] && 'p' == m_DataIt[2] && '_' == m_DataIt[3]) {
        getTexture();
    }
    m_DataIt = skipLine<DataArrayIt>(m_DataIt, m_DataItEnd, m_uiLine);
    break;
case 'b':
    // Bump map
    if ('u' == m_DataIt[1] && 'm' == m_DataIt[2] && 'p' == m_DataIt[3]) {
        getTexture();
    }
    m_DataIt = skipLine<DataArrayIt>(m_DataIt, m_DataItEnd, m_uiLine);
    break;
case 'r':
    // Reflection map
    if ('e' == m_DataIt[1] && 'f' == m_DataIt[2] && 'l' == m_DataIt[3]) {
        getTexture();
    }
    m_DataIt = skipLine<DataArrayIt>(m_DataIt, m_DataItEnd, m_uiLine);
    break;
