{
    const char *pStart = &(*m_DataIt);
    m_DataIt = getNextWord<DataArrayIt>(m_DataIt, m_DataItEnd);
    if (m_DataIt == m_DataItEnd) {
        return;
    }
    const size_t len = std::distance(pStart, &(*m_DataIt));

    switch (*pStart) {
        case 'K':
            // Diffuse color
            if (pStart[1] == 'd') {
                getColorRGBA(m_pCurrentMaterial->Diffuse);
            }
            // Ambient color
            else if (pStart[1] == 'a') {
                getColorRGBA(m_pCurrentMaterial->Ambient);
            }
            // Specular color
            else if (pStart[1] == 's') {
                getColorRGBA(m_pCurrentMaterial->Specular);
            }
            // Emissive color
            else if (pStart[1] == 'e') {
                getColorRGBA(m_pCurrentMaterial->Emissive);
            }
            // Transmission filter
            else if (pStart[1] == 't') {
                getColorRGBA(m_pCurrentMaterial->TransmissionFilter);
            } else {
                goto pf_default;
            }
            break;

        case 'T':
            if (pStart[1] == 'f') {
                // Transmission filter
                getColorRGBA(m_pCurrentMaterial->TransmissionFilter);
            } else if (pStart[1] == 'r') {
                // Transparency value
                getFloatValue(m_pCurrentMaterial->Transparency);
                if (!m_pCurrentMaterial->Transparency.has_value()) {
                    m_pCurrentMaterial->Transparency = Maybe<float>(1.f - m_pCurrentMaterial->Alpha.value_or(1.f));
                }
            } else {
                goto pf_default;
            }
            break;

        case 'd':
            // Alpha value
            getFloatValue(m_pCurrentMaterial->Alpha);
            break;

        case 'N':
            if (pStart[1] == 's') {
                // Specular exponent
                getFloatValue(m_pCurrentMaterial->Shininess);
            } else if (pStart[1] == 'i') {
                // Index of refraction
                getFloatValue(m_pCurrentMaterial->IOR);
            } else {
                goto pf_default;
            }
            break;

        case 'n':
            // Newmtl statement
            if (len >= 6 && strncmp(pStart, "newmtl", 6) == 0) {
                createMaterial();
            } else {
                goto pf_default;
            }
            break;

        case 'm':
            if (len >= 3) {
                if (strncmp(pStart, "map_Kd", 6) == 0) {
                    getTexture(aiTextureType_DIFFUSE);
                } else if (strncmp(pStart, "map_Ks", 6) == 0) {
                    getTexture(aiTextureType_SPECULAR);
                } else if (strncmp(pStart, "map_Ka", 6) == 0) {
                    getTexture(aiTextureType_AMBIENT);
                } else if (strncmp(pStart, "map_bump", 8) == 0 || strncmp(pStart, "bump", 4) == 0) {
                    getTexture(aiTextureType_HEIGHT);
                } else if (strncmp(pStart, "map_d", 5) == 0) {
                    getTexture(aiTextureType_OPACITY);
                } else if (strncmp(pStart, "map_refl", 8) == 0) {
                    getTexture(aiTextureType_REFLECTION);
                } else if (strncmp(pStart, "map_ns", 6) == 0) {
                    getTexture(aiTextureType_SHININESS);
                } else if (strncmp(pStart, "map_norm", 8) == 0 || strncmp(pStart, "norm", 4) == 0) {
                    getTexture(aiTextureType_NORMALS);
                } else if (strncmp(pStart, "map_disp", 8) == 0 || strncmp(pStart, "disp", 4) == 0) {
                    getTexture(aiTextureType_DISPLACEMENT);
                } else if (strncmp(pStart, "map_emis", 8) == 0) {
                    getTexture(aiTextureType_EMISSIVE);
                } else if (strncmp(pStart, "map_opacity", 11) == 0) {
                    getTexture(aiTextureType_OPACITY);
                } else if (strncmp(pStart, "map_trans", 9) == 0) {
                    getTexture(aiTextureType_OPACITY);
                } else {
                    goto pf_default;
                }
            } else {
                goto pf_default;
            }
            break;


        case 'i':
            if (strncmp(pStart, "illum", 5) == 0) {
                // Illumination model
                getIlluminationModel();
            } else {
                goto pf_default;
            }
            break;

        case 'a':
            if (strncmp(pStart, "anisotropy", 10) == 0) {
                getFloatValue(m_pCurrentMaterial->Anisotropy);
            } else {
                goto pf_default;
            }
            break;

        default:
        pf_default: {
            m_DataIt = skipLine<DataArrayIt>(m_DataIt, m_DataItEnd, m_uiLine);
        } break;
    }
} break;
