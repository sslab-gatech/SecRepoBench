++m_DataIt;
if(m_DataIt == m_DataItEnd) break;
if(nullptr == m_pCurrentMaterial) {
    m_DataIt = skipLine<DataArrayIt>(m_DataIt, m_DataItEnd, m_uiLine);
    break;
}

switch(*m_DataIt)
{
    case 'a': // Ambient color (Ka)
    {
        m_DataIt = getNextToken<DataArrayIt>(m_DataIt, m_DataItEnd);
        getColorRGBA( m_pCurrentMaterial->Ambient );
        break;
    }
    case 'd': // Diffuse color (Kd)
    {
        m_DataIt = getNextToken<DataArrayIt>(m_DataIt, m_DataItEnd);
        getColorRGBA( m_pCurrentMaterial->Diffuse );
        break;
    }
    case 's': // Specular color (Ks)
    {
        m_DataIt = getNextToken<DataArrayIt>(m_DataIt, m_DataItEnd);
        getColorRGBA( m_pCurrentMaterial->Specular );
        break;
    }
    case 'e': // Emissive color (Ke)
    {
        m_DataIt = getNextToken<DataArrayIt>(m_DataIt, m_DataItEnd);
        getColorRGBA( m_pCurrentMaterial->Emissive );
        break;
    }
    default:
    {
        m_DataIt = skipLine<DataArrayIt>(m_DataIt, m_DataItEnd, m_uiLine);
        break;
    }
}
