case 'K': {
    // Parse color definitions (Ka, Kd, Ks)
    char type = *(m_DataIt + 1);
    if (type == 'a' || type == 'd' || type == 's') {
        m_DataIt += 2; // Skip past 'K' and the specific type character
        aiColor3D color;
        if (!parseColor(&color)) {
            ASSIMP_LOG_ERROR("OBJ: Failed to parse color value at line ", m_uiLine);
            m_DataIt = skipLine<DataArrayIt>(m_DataIt, m_DataItEnd, m_uiLine);
            break;
        }
        switch (type) {
            case 'a': { // Ambient color
                if (m_pModel->m_pCurrentMaterial) {
                    m_pModel->m_pCurrentMaterial->MatAmbient = color;
                }
            } break;
            case 'd': { // Diffuse color
                if (m_pModel->m_pCurrentMaterial) {
                    m_pModel->m_pCurrentMaterial->MatDiffuse = color;
                }
            } break;
            case 's': { // Specular color
                if (m_pModel->m_pCurrentMaterial) {
                    m_pModel->m_pCurrentMaterial->MatSpecular = color;
                }
            } break;
        }
    } else {
        ASSIMP_LOG_WARN("OBJ: Invalid color type specified at line ", m_uiLine);
        m_DataIt = skipLine<DataArrayIt>(m_DataIt, m_DataItEnd, m_uiLine);
    }
} break;
