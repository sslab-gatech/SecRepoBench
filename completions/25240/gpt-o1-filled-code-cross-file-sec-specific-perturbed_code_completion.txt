{
    // Consume the 'k'
    ++m_DataIt;
    if (m_DataIt == m_DataItEnd) {
        m_DataIt = skipLine<DataArrayIt>(m_DataIt, m_DataItEnd, m_uiLine);
        break;
    }

    // Peek the next character to determine which color we're dealing with
    char nextChar = static_cast<char>(std::tolower(*m_DataIt));

    // Ensure a current material exists
    if (!m_pCurrentMaterial) {
        m_pCurrentMaterial = new ObjFile::Material();
        m_MaterialCache.push_back(m_pCurrentMaterial);
    }

    // Parse the color
    Maybe<aiColor3D> colorValue;
    getColorRGBA(colorValue);
    if (colorValue.IsSet()) {
        switch (nextChar) {
            case 'a': // Ka -> Ambient
                m_pCurrentMaterial->Ambient = colorValue.Get();
                break;
            case 'd': // Kd -> Diffuse
                m_pCurrentMaterial->Diffuse = colorValue.Get();
                break;
            case 's': // Ks -> Specular
                m_pCurrentMaterial->Specular = colorValue.Get();
                break;
            case 'e': // Ke -> Emissive
                m_pCurrentMaterial->Emissive = colorValue.Get();
                break;
            case 'f': // Tf -> Transmission
                m_pCurrentMaterial->Transmittance = colorValue.Get();
                break;
            default:
                // Unrecognized 'k' property
                break;
        }
    }

    // Move to the next line after processing
    m_DataIt = skipLine<DataArrayIt>(m_DataIt, m_DataItEnd, m_uiLine);
}
break;
