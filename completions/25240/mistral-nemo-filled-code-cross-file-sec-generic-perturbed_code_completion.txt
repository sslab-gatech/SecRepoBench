    // Parse material properties
    m_DataIt++;
    std::string materialName;
    while (m_DataIt != m_DataItEnd && !isDataDefinitionEnd(m_DataIt)) {
        materialName += *m_DataIt;
        m_DataIt++;
    }
    int matIndex = getMaterialIndex(materialName);
    if (matIndex != -1) {
        m_pModel->m_pCurrentMaterial = &m_pModel->m_MaterialLib[matIndex];
    } else {
        createObject(materialName);
    }
