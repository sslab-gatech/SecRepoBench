case 'K': {
    // Check for Ka, Kd, Ks, Ke or other material color properties
    m_DataIt = getNextToken<DataArrayIt>(m_DataIt, m_DataItEnd);
    if (m_DataIt == m_DataItEnd || IsLineEnd(*m_DataIt)) {
        ASSIMP_LOG_WARN("OBJ: Unexpected end of line while parsing material color.");
        break;
    }

    std::string propType(m_DataIt, m_DataIt + 2);  // Extract property type (Ka, Kd, etc.)
    m_DataIt += 2;

    if (propType == "a" || propType == "d" || propType == "s" || propType == "e") {
        aiColor3D color;
        if (!parseColor(&color)) {  // Assume parseColor is a helper function to read RGB values
            ASSIMP_LOG_WARN("OBJ: Failed to parse material color values.");
            break;
        }

        switch (propType[0]) {
            case 'a': m_pModel->m_pCurrentMaterial->m_ColorAmbient = color; break;
            case 'd': m_pModel->m_pCurrentMaterial->m_ColorDiffuse = color; break;
            case 's': m_pModel->m_pCurrentMaterial->m_ColorSpecular = color; break;
            case 'e': m_pModel->m_pCurrentMaterial->m_ColorEmissive = color; break;
            default: break;
        }
    } else {
        ASSIMP_LOG_WARN("OBJ: Unknown material color property encountered.");
    }
} break;
