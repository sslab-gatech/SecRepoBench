{
    ++m_DataIt;
    if (m_DataIt == m_DataItEnd) {
        break;
    }
    char c = *m_DataIt;
    switch (c) {
        case 'a': // ka - ambient color
            ++m_DataIt;
            if (m_pCurrentMaterial) {
                getColorRGBA(m_pCurrentMaterial->ambient);
            }
            break;
        case 'd': // kd - diffuse color
            ++m_DataIt;
            if (m_pCurrentMaterial) {
                getColorRGBA(m_pCurrentMaterial->diffuse);
            }
            break;
        case 's': // ks - specular color
            ++m_DataIt;
            if (m_pCurrentMaterial) {
                getColorRGBA(m_pCurrentMaterial->specular);
            }
            break;
        case 'e': // ke - emissive color
            ++m_DataIt;
            if (m_pCurrentMaterial) {
                getColorRGBA(m_pCurrentMaterial->emissive);
            }
            break;
        case 't': // kt - transmission filter
            ++m_DataIt;
            if (m_pCurrentMaterial) {
                getColorRGBA(m_pCurrentMaterial->transmission);
            }
            break;
        default:
            m_DataIt = skipLine<DataArrayIt>(m_DataIt, m_DataItEnd, m_uiLine);
            break;
    }
    break;
}
