switch (*(++m_DataIt)) {
                case 'a': {
                    ++m_DataIt;
                    if (*m_DataIt == 'o') {
                        // Ambient color
                        m_DataIt = getNextToken<DataArrayIt>(m_DataIt, m_DataItEnd);
                        getColorRGBA(m_pModel->m_pCurrentMaterial->Ambient);
                    } else if (*m_DataIt == 'd') {
                        // Alpha value
                        m_DataIt = getNextToken<DataArrayIt>(m_DataIt, m_DataItEnd);
                        getFloat(m_pModel->m_pCurrentMaterial->Alpha);
                    }
                } break;

                case 'd': {
                    // Diffuse color
                    m_DataIt = getNextToken<DataArrayIt>(m_DataIt, m_DataItEnd);
                    getColorRGBA(m_pModel->m_pCurrentMaterial->Diffuse);
                } break;

                case 's': {
                    ++m_DataIt;
                    if (*m_DataIt == 'p') {
                        // Specular color
                        m_DataIt = getNextToken<DataArrayIt>(m_DataIt, m_DataItEnd);
                        getColorRGBA(m_pModel->m_pCurrentMaterial->Specular);
                    } else {
                        // Specular exponent
                        m_DataIt = getNextToken<DataArrayIt>(m_DataIt, m_DataItEnd);
                        getFloat(m_pModel->m_pCurrentMaterial->Shininess);
                    }
                } break;

                case 'e': {
                    // Emissive color
                    m_DataIt = getNextToken<DataArrayIt>(m_DataIt, m_DataItEnd);
                    getColorRGBA(m_pModel->m_pCurrentMaterial->Emissive);
                } break;

                case 'T': {
                    ++m_DataIt;
                    if (*m_DataIt == 'f') {
                        // Transmission filter
                        m_DataIt = getNextToken<DataArrayIt>(m_DataIt, m_DataItEnd);
                        getColorRGBA(m_pModel->m_pCurrentMaterial->Transparent);
                    } else {
                        // Transparency
                        m_DataIt = getNextToken<DataArrayIt>(m_DataIt, m_DataItEnd);
                        getFloat(m_pModel->m_pCurrentMaterial->Opacity);
                        m_pModel->m_pCurrentMaterial->Alpha = Maybe<float>(1.0f - m_pModel->m_pCurrentMaterial->Opacity.value);
                    }
                } break;

                case 'r': {
                    // Index of refraction
                    m_DataIt = getNextToken<DataArrayIt>(m_DataIt, m_DataItEnd);
                    getFloat(m_pModel->m_pCurrentMaterial->Refracti);
                } break;

                case 'i': {
                    // Illumination model
                    m_DataIt = getNextToken<DataArrayIt>(m_DataIt, m_DataItEnd);
                    getInteger(m_pModel->m_pCurrentMaterial->illumination_model);
                } break;

                case 'N': {
                    ++m_DataIt;
                    if (*m_DataIt == 's') {
                        // Specular exponent
                        m_DataIt = getNextToken<DataArrayIt>(m_DataIt, m_DataItEnd);
                        getFloat(m_pModel->m_pCurrentMaterial->Shininess);
                    } else if (*m_DataIt == 'i') {
                        // Optical density (index of refraction)
                        m_DataIt = getNextToken<DataArrayIt>(m_DataIt, m_DataItEnd);
                        getFloat(m_pModel->m_pCurrentMaterial->Refracti);
                    }
                } break;

                case 'm': {
                    // Texture map
                    ++m_DataIt;
                    if (*m_DataIt == 'a') {
                        ++m_DataIt;
                        if (*m_DataIt == 'p') {
                            ++m_DataIt;
                            if (*m_DataIt == '_') {
                                ++m_DataIt;
                                if (*m_DataIt == 'K') {
                                    ++m_DataIt;
                                    if (*m_DataIt == 'd') {
                                        m_DataIt = getNextToken<DataArrayIt>(m_DataIt, m_DataItEnd);
                                        getString(m_pModel->m_pCurrentMaterial->TextureDiffuse.data, MAX_TEXTUREPATH);
                                    } else if (*m_DataIt == 'a') {
                                        m_DataIt = getNextToken<DataArrayIt>(m_DataIt, m_DataItEnd);
                                        getString(m_pModel->m_pCurrentMaterial->TextureAmbient.data, MAX_TEXTUREPATH);
                                    } else if (*m_DataIt == 's') {
                                        m_DataIt = getNextToken<DataArrayIt>(m_DataIt, m_DataItEnd);
                                        getString(m_pModel->m_pCurrentMaterial->TextureSpecular.data, MAX_TEXTUREPATH);
                                    }
                                } else if (*m_DataIt == 'b') {
                                    m_DataIt = getNextToken<DataArrayIt>(m_DataIt, m_DataItEnd);
                                    getString(m_pModel->m_pCurrentMaterial->TextureBump.data, MAX_TEXTUREPATH);
                                }
                            }
                        }
                    }
                } break;

                case 'n': {
                    // New material
                    m_DataIt = getNextToken<DataArrayIt>(m_DataIt, m_DataItEnd);
                    m_pModel->m_pCurrentMaterial = new ObjFile::Material;
                    getString(m_pModel->m_pCurrentMaterial->MaterialName.data, MAX_MATERIAL_NAME);
                    m_pModel->m_MaterialLib.push_back(m_pModel->m_pCurrentMaterial->MaterialName.data);
                    m_pModel->m_Materials.push_back(m_pModel->m_pCurrentMaterial);
                } break;
