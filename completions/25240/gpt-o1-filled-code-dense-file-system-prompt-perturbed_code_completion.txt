case 'k':
{
    ++m_DataIt;
    if (m_DataIt == m_DataItEnd) {
        break;
    }

    char c = static_cast<char>(::tolower(*m_DataIt));
    ++m_DataIt;  // Skip the second character (e.g., 'a' in "Ka")
    switch (c) {
        case 'a':  // Ka = Ambient
            getColorRGBA(&m_pModel->m_pCurrentMaterial->Ambient);
            break;
        case 'd':  // Kd = Diffuse
            getColorRGBA(&m_pModel->m_pCurrentMaterial->Diffuse);
            break;
        case 's':  // Ks = Specular
            getColorRGBA(&m_pModel->m_pCurrentMaterial->Specular);
            break;
        case 'e':  // Ke = Emissive
            getColorRGBA(&m_pModel->m_pCurrentMaterial->Emissive);
            break;
        case 'f':  // Kf/Tf = Transmission
            getColorRGBA(&m_pModel->m_pCurrentMaterial->Tf);
            break;
        default:
            // Unrecognized property after 'k'
            break;
    }

    m_DataIt = skipLine<DataArrayIt>(m_DataIt, m_DataItEnd, m_uiLine);
} break;
