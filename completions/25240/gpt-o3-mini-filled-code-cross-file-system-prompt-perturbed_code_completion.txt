++m_DataIt;
if (m_DataIt == m_DataItEnd)
    break;

switch (*m_DataIt) {
    case 'a': // Ambient color (Ka)
        ++m_DataIt;
        getColorRGBA(m_pCurrentMaterial->m_ColorAmbient);
        break;
    case 'd': // Diffuse color (Kd)
        ++m_DataIt;
        getColorRGBA(m_pCurrentMaterial->m_ColorDiffuse);
        break;
    case 's': // Specular color (Ks)
        ++m_DataIt;
        getColorRGBA(m_pCurrentMaterial->m_ColorSpecular);
        break;
    case 'e': // Emissive color (Ke)
        ++m_DataIt;
        getColorRGBA(m_pCurrentMaterial->m_ColorEmissive);
        break;
    case 't': // Transmission color (Kt)
        ++m_DataIt;
        getColorRGBA(m_pCurrentMaterial->m_ColorTransmission);
        break;
    default:
        m_DataIt = skipLine<DataArrayIt>(m_DataIt, m_DataItEnd, m_uiLine);
        break;
}
