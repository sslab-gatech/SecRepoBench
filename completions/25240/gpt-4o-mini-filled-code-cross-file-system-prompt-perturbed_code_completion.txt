case 'K': // New material
                m_DataIt = getNextToken<DataArrayIt>(m_DataIt, m_DataItEnd);
                if (m_DataIt != m_DataItEnd) {
                    std::string materialName(&(*m_DataIt), &(*m_DataItEnd));
                    m_pModel->m_pCurrentMaterial = new Material(materialName);
                    m_pModel->m_Materials.push_back(m_pModel->m_pCurrentMaterial);
                }
                break;

            case 'a': // Ambient color
                getColorRGBA(m_pModel->m_pCurrentMaterial->AmbientColor);
                break;

            case 'd': // Diffuse color
                getColorRGBA(m_pModel->m_pCurrentMaterial->DiffuseColor);
                break;

            case 's': // Specular color
                getColorRGBA(m_pModel->m_pCurrentMaterial->SpecularColor);
                break;

            case 'e': // Emissive color
                getColorRGBA(m_pModel->m_pCurrentMaterial->EmissiveColor);
                break;

            case 't': // Transparency
                m_DataIt = getNextToken<DataArrayIt>(m_DataIt, m_DataItEnd);
                if (m_DataIt != m_DataItEnd) {
                    m_pModel->m_pCurrentMaterial->Transparency = std::stof(std::string(&(*m_DataIt), &(*m_DataItEnd)));
                }
                break;

            case 'n': // Specular exponent
                m_DataIt = getNextToken<DataArrayIt>(m_DataIt, m_DataItEnd);
                if (m_DataIt != m_DataItEnd) {
                    m_pModel->m_pCurrentMaterial->SpecularExponent = std::stof(std::string(&(*m_DataIt), &(*m_DataItEnd)));
                }
                break;

            case 'i': // Index of Refraction
                m_DataIt = getNextToken<DataArrayIt>(m_DataIt, m_DataItEnd);
                if (m_DataIt != m_DataItEnd) {
                    m_pModel->m_pCurrentMaterial->IndexOfRefraction = std::stof(std::string(&(*m_DataIt), &(*m_DataItEnd)));
                }
                break;

            case 'm': // Material library
                getMaterialLib();
                break;

            case 't': // Texture
                // Handle texture loading here
                break;
