case 'K': {
                ++m_DataIt;
                if (m_DataIt == m_DataItEnd) {
                    break;
                }

                aiColor3D color;
                switch (*m_DataIt) {
                    case 'a': // Ambient color
                        ++m_DataIt;
                        getColorRGBA(&color);
                        m_pModel->m_pCurrentMaterial->ambient = color;
                        break;

                    case 'd': // Diffuse color
                        ++m_DataIt;
                        getColorRGBA(&color);
                        m_pModel->m_pCurrentMaterial->diffuse = color;
                        break;

                    case 's': // Specular color
                        ++m_DataIt;
                        getColorRGBA(&color);
                        m_pModel->m_pCurrentMaterial->specular = color;
                        break;

                    case 'e': // Emissive color
                        ++m_DataIt;
                        getColorRGBA(&color);
                        m_pModel->m_pCurrentMaterial->emissive = color;
                        break;

                    case 't': // Transmission filter
                        ++m_DataIt;
                        getColorRGBA(&color);
                        m_pModel->m_pCurrentMaterial->transmissionFilter = color;
                        break;

                    default:
                        m_DataIt = skipLine<DataArrayIt>(m_DataIt, m_DataItEnd, m_uiLine);
                        break;
                }
            } break;

            case 'N': {
                ++m_DataIt;
                if (m_DataIt == m_DataItEnd) {
                    break;
                }

                ai_real value;
                switch (*m_DataIt) {
                    case 's': // Specular exponent
                        ++m_DataIt;
                        getFloatValue(value);
                        m_pModel->m_pCurrentMaterial->shininess = value;
                        break;

                    case 'i': // Index of refraction
                        ++m_DataIt;
                        getFloatValue(value);
                        m_pModel->m_pCurrentMaterial->ior = value;
                        break;

                    default:
                        m_DataIt = skipLine<DataArrayIt>(m_DataIt, m_DataItEnd, m_uiLine);
                        break;
                }
            } break;

            case 'd': // Transparency
            case 'T': {
                ++m_DataIt;
                if (m_DataIt == m_DataItEnd) {
                    break;
                }

                ai_real value;
                if (*m_DataIt == 'r') {
                    ++m_DataIt;
                    getFloatValue(value);
                    m_pModel->m_pCurrentMaterial->transparency = value;
                } else if (*m_DataIt == 'f') {
                    ++m_DataIt;
                    getFloatValue(value);
                    m_pModel->m_pCurrentMaterial->transparency = 1.0f - value;
                } else {
                    m_DataIt = skipLine<DataArrayIt>(m_DataIt, m_DataItEnd, m_uiLine);
                }
            } break;

            case 'i': // Illumination model
                ++m_DataIt;
                int illum_model;
                getIlluminationModel(illum_model);
                m_pModel->m_pCurrentMaterial->illumination_model = illum_model;
                break;

            case 'n': // New material
                ++m_DataIt;
                createMaterial();
                break;

            case 'm': // Texture map
                getTexture();
                break;
