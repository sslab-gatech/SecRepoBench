case 'k': {
    ++m_DataIt;
    if (m_DataIt == m_DataItEnd) break;

    Maybe<aiColor3D> *target = nullptr;
    switch (*m_DataIt) {
        case 'a': target = &m_pCurrentMaterial->ambient; break;
        case 'd': target = &m_pCurrentMaterial->diffuse; break;
        case 's': target = &m_pCurrentMaterial->specular; break;
        case 'e': target = &m_pCurrentMaterial->emissive; break;
        case 't': target = &m_pCurrentMaterial->transparent; break;
    }

    if (target) {
        ++m_DataIt;
        getColorRGBA(*target);
    }
    m_DataIt = skipLine<DataArrayIt>(m_DataIt, m_DataItEnd, m_uiLine);
} break;
