{
    // Advance past 'k'
    ++m_DataIt;
    if (m_DataIt == m_DataItEnd) {
        break;
    }

    // Decide which color property to parse
    char c = static_cast<char>(std::tolower(*m_DataIt));
    ++m_DataIt;
    switch (c) {
        case 'a': {
            // "Ka" -> Ambient color
            getColorRGBA(m_pCurrentMaterial->ColAmbient);
        } break;
        case 'd': {
            // "Kd" -> Diffuse color
            getColorRGBA(m_pCurrentMaterial->ColDiffuse);
        } break;
        case 's': {
            // "Ks" -> Specular color
            getColorRGBA(m_pCurrentMaterial->ColSpecular);
        } break;
        case 'e': {
            // "Ke" -> Emissive color
            getColorRGBA(m_pCurrentMaterial->ColEmissive);
        } break;
        case 't': {
            // Some exporters use "Kt" for transmission, though more common is "Tf"
            // Still handle it here if present
            getColorRGBA(m_pCurrentMaterial->ColTransmittance);
        } break;
        default: {
            // Unrecognized property after 'k'; skip the entire line
            m_DataIt = skipLine<DataArrayIt>(m_DataIt, m_DataItEnd, m_uiLine);
            break;
        }
    }

    // After handling the color, skip rest of the line
    m_DataIt = skipLine<DataArrayIt>(m_DataIt, m_DataItEnd, m_uiLine);
}
