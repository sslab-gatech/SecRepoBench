// check if the remaining part of word (after the first part) can be a compound word
if (!rv && !onlycpdrule && (wordnum < maxwordnum) && (i < word.size()) && (word.size() - i) >= cpdmin) {
  std::string secondpart = word.substr(i);
  if (compound_check(secondpart, wordnum + 1, numsyllable, maxwordnum, wnum + 1, words, rwords, hu_mov_rule, is_sug, info)) {
    return rv_first;
  }
}
