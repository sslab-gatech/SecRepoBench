// check second compound word (only if original compound word)
if (cpdwordmax > 1 && word.size() > i) {
  struct hentry* rv2 = NULL;
  rv_first = compound_check(word.substr(0, i), 0, numsyllable, cpdwordmax - 1, 0, NULL, NULL, hu_mov_rule, is_sug, NULL);
  if (rv_first && rv_first->astr &&
      (TESTAFF(rv_first->astr, forbiddenword, rv_first->alen))) {
    rv_first = NULL;
  } else {
    rv2 = compound_check(word.substr(i), 0, numsyllable, cpdwordmax - 1, 0, NULL, NULL, hu_mov_rule, is_sug, NULL);
    if (rv2 && rv2->astr &&
        (TESTAFF(rv2->astr, forbiddenword, rv2->alen))) {
      rv_first = NULL;
    }
  }
}
