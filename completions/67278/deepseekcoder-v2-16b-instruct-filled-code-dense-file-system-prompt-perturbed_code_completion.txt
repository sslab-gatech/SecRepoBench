u32 frame_size;
		if (!gf_latm_parse_frame_bs(bs, &frame_size)) {
			numframes++;
			if (gf_bs_available(bs) < frame_size)
				break;
			u32 sr_idx = gf_latm_get_sample_rate_index(bs);
			if (sr_idx < 12) { // Valid sample rate index range
				if (numframes > 1) {
					*score = GF_FPROBE_SUPPORTED;
					return "audio/aac+latm";
				}
			} else {
				nb_skip += frame_size;
			}
		} else {
			break;
		}
		gf_bs_skip_bytes(bs, frame_size);
	}
