u32 frame_size;
        u8 sample_rate_index;
        
        // Read the frame size from the bitstream
        frame_size = gf_bs_read_int(bs, 13); // Assuming 13 bits for frame size
        if (frame_size == 0 || frame_size > size) break;

        // Read the sample rate index from the bitstream
        sample_rate_index = gf_bs_read_int(bs, 4); // Assuming 4 bits for sample rate index
        if (sample_rate_index >= 12) break; // Invalid sample rate index

        // Skip the rest of the frame
        if (gf_bs_skip_bytes(bs, frame_size - 2) != GF_OK) break;

        numframes++;
        nb_skip += frame_size;
        
        // If we have read enough frames, exit the loop
        if (numframes >= 2) break;
