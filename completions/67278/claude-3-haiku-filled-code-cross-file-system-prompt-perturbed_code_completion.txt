while (1) {
	if (gf_m4a_get_config(bs, &acfg) != GF_OK) {
		nb_skip += gf_bs_get_bit_offset(bs) / 8;
		if (!gf_bs_available(bs)) break;
		continue;
	}
	if (acfg.base_sr_index >= 0 && acfg.base_sr_index < 13) {
		numframes++;
	} else {
		nb_skip += gf_bs_get_bit_offset(bs) / 8;
	}
	if (!gf_bs_available(bs)) break;
}
