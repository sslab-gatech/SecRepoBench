while (gf_bs_available(bs) >= 8) {
    u8 sr_byte = (u8) gf_bs_read_int(bs, 8);
    /* In this simple check, the top 4 bits are assumed to hold the sample rate index. 
       Adjust shifting/masking as needed for actual bit layout. */
    u8 sr_idx = (sr_byte >> 4) & 0x0F;

    /* Validate index to avoid out-of-bounds usage (CWE-129). */
    if (sr_idx >= 13) {
        nb_skip++;
    } else {
        numframes++;
    }

    /* Stop if we have successfully parsed enough frames. */
    if (numframes >= 2) {
        break;
    }
}
