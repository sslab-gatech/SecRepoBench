GF_Err e = gf_m4a_parse_config(bs, &acfg, 0);
		if (e) {
			if (e==GF_EOS) break;
			nb_skip += gf_bs_get_bit_offset(bs) - (numframes ? acfg.configSize*8 : 0);
			gf_bs_seek(bs, 0);
			gf_m4a_parse_config(bs, &acfg, 1);
		}
		if (acfg.audioPL) {
			u32 i;
			for (i=0; i<gf_list_count(acfg.audioPL); i++) {
				GF_M4ADecSpecInfo *dsi = gf_list_get(acfg.audioPL, i);
				if (dsi->sampling_rate_index>12) break;
			}
			if (i<gf_list_count(acfg.audioPL)) break;
		} else {
			if (acfg.sampling_rate_index>12) break;
		}

		numframes++;
		if (numframes>5) break;
		acfg.configSize = gf_bs_get_bit_offset(bs)/8;
		gf_bs_seek(bs, acfg.configSize);
