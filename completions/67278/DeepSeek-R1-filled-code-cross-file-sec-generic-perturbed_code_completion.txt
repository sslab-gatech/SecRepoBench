u64 pos = gf_bs_get_position(bs);
if (gf_bs_available(bs) < 11) break;
u32 sync = gf_bs_read_int(bs, 11);
if (sync != 0x2B7) {
    gf_bs_seek(bs, pos + 8);
    nb_skip++;
    continue;
}
memset(&acfg, 0, sizeof(acfg));
GF_Err e = gf_m4a_parse_audio_mux_element(bs, &acfg, 0);
if (e != GF_OK) break;
if (acfg.base_sr_index >= 0xC) break;
numframes++;
if (numframes >= 2) break;
