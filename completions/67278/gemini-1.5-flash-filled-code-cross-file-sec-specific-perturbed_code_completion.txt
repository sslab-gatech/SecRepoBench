u32 au_size;
		u32 ret = gf_bs_read_u32(bs);
		if (bs->error) break;
		au_size = ret;
		if (gf_bs_read_data(bs, (u8 *)&acfg, sizeof(acfg)) != sizeof(acfg)) break;
		if (acfg.index >= GF_M4A_NB_SAMPLE_RATES) break;
		nb_skip += gf_bs_get_byte_count(bs) - (sizeof(acfg) + 4);
		gf_bs_seek(bs, au_size, SEEK_CUR);
		numframes++;
