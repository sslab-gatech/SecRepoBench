while (gf_bs_available(bs)) {
			u32 start_pos = (u32)gf_bs_get_position(bs);
			memset(&acfg, 0, sizeof(acfg));
			/* Attempt to parse one LATM frame */
			if (!gf_m4a_latm_parse_frame(bs, &acfg)) {
				/* If parsing failed but the bitstream didn't advance, break out */
				if (gf_bs_get_position(bs) == start_pos) {
					break;
				}
				/* Otherwise, skip a byte and count it as a skip */
				gf_bs_skip_bytes(bs, 1);
				nb_skip++;
				continue;
			}
			/* Check if sample rate index is valid */
			if (acfg.sr_idx >= 13) {
				break;
			}
			numframes++;
			/* If we've parsed enough valid frames, stop */
			if (numframes >= 2) {
				break;
