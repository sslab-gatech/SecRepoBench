u32 au_size;
		u32 err;
		if (gf_bs_read_u32(bs, &au_size)) break;
		if (au_size > size) break;
		if (gf_bs_read_data(bs, (u8*)&acfg, sizeof(acfg)) != sizeof(acfg)) break;
		if (acfg.index > 15) {
			nb_skip += au_size;
			gf_bs_seek(bs, au_size * 8, SEEK_CUR);
			continue;
		}
		err = gf_bs_read_data(bs, NULL, au_size - sizeof(acfg));
		if (err) break;
		numframes++;
