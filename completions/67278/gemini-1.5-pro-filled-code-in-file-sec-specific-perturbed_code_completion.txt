memset(&acfg, 0, sizeof(GF_M4ADecSpecInfo));
if (!latm_dmx_sync_frame_bs(bs, &acfg, NULL, NULL, &nb_skip)) break;

numframes++;
if (acfg.base_sr_index >= GF_ARRAY_LENGTH(GF_M4ASampleRates)) {
    GF_LOG(GF_LOG_WARNING, GF_LOG_CONTAINER, ("[LATM probe] Invalid sample rate index %u\n", acfg.base_sr_index));
    break;
}
if (!GF_M4ASampleRates[acfg.base_sr_index]) {
    GF_LOG(GF_LOG_WARNING, GF_LOG_CONTAINER, ("[LATM probe] Sample rate not found for index %u\n", acfg.base_sr_index));
    break;
}
