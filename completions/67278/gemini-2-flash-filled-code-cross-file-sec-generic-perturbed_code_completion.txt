u32 frame_start = gf_bs_get_pos(bs);
		if (gf_m4a_read_config(bs, &acfg, 0)) {
			break;
		}
		if (acfg.samplingRateIndex >= GF_M4A_NUM_SAMPLING_RATES) {
			gf_bs_seek(bs, frame_start);
			break;
		}
		numframes++;
		if (numframes>10) break;

		/*skip frame data*/
		if (acfg.frameLength) {
			gf_bs_seek(bs, frame_start + acfg.frameLength);
		} else {
			/*we don't know the frame size, skip one byte*/
			nb_skip++;
			gf_bs_read(bs, NULL, 8);
		}
		if (gf_bs_eos(bs))
			break;
