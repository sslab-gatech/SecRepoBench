u32 frame_size;
		s32 sr_idx;
		GF_Err e = gf_m4a_parse_config_bs(bs, &acfg, &frame_size);
		if (e) break;
		sr_idx = acfg.base_object_info.sampling_frequency_index;

		if (sr_idx>11) {
			if (!acfg.base_object_info.extension_sampling_frequency_index)
				break;
			if (acfg.base_object_info.extension_sampling_frequency_index>11)
				break;
		}
		if (frame_size>gf_bs_available(bs)) {
			nb_skip += frame_size;
			break;
		}
		gf_bs_skip_bytes(bs, frame_size);
		numframes++;
		if (numframes==4) break; //arbitrary limit
