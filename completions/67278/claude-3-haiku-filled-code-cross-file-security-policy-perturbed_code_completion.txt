// <MASK>
u32 sr_idx = gf_bs_read_int(bs, 4);
if (sr_idx >= 13) {
    gf_bs_skip_bytes(bs, 2);
    nb_skip += 2;
    continue;
}
gf_m4a_get_config_from_sample_index(&acfg, sr_idx);
if (acfg.base_object_type == 0) {
    gf_bs_skip_bytes(bs, 2);
    nb_skip += 2;
    continue;
}
numframes++;
gf_bs_skip_bytes(bs, 2);
nb_skip += 2;
