GF_LATMFrameHeader latm_hdr;
	u32 skip_bytes = 0;
	while (gf_bs_available(bs) >= 16) {
		gf_bs_read_int(bs, &latm_hdr, 16);
		if (latm_hdr.sr_index >= 15) {
			skip_bytes += 16;
			continue;
		}
		numframes++;
		skip_bytes += 16;
		u32 frame_size = gf_bs_read_int(bs, 24);
		if (frame_size > (size - skip_bytes)) break;
		skip_bytes += frame_size;
	}
	nb_skip = skip_bytes;
