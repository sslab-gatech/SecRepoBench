		if (sample_rate_index >= 12) {
			nb_skip++;

		}

		if (frame_size == 0 || frame_size > gf_bs_available(bs)) {
			nb_skip++;
			continue; // Skip invalid frame
		}

		gf_bs_skip_bytes(bs, frame_size);
