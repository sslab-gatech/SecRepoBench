size_t ref_len = b->uncomp_size;
            if (slice->hdr->ref_seq_span > ref_len) {
                hts_log_error("Embedded reference is too small: %zu < %"PRIu64,
                              ref_len, slice->hdr->ref_seq_span);
                return -1;
            }
            slice->ref = b->data;
            slice->ref_start = slice->hdr->ref_seq_start;
            slice->ref_end = slice->hdr->ref_seq_start + slice->hdr->ref_seq_span - 1;
