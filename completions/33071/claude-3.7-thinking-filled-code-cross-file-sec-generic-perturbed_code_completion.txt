if (*content_marker != 0xc6) {   // sanity check for content marker
  return BLOSC2_ERROR_DATA;
}
int32_t content_len;
from_big(&content_len, content_marker + 1, sizeof(content_len));
if (content_len < 0 || (int64_t)offset + 1 + 4 + (int64_t)content_len > trailer_len) {
  // Content length is negative or would exceed the trailer bounds
  return BLOSC2_ERROR_DATA;
}
layer_metadata->content_len = content_len;
