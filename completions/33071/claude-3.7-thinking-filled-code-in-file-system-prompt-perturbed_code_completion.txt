if (*content_marker != 0xc6) {  // sanity check for bin32
  return BLOSC2_ERROR_DATA;
}
// Read the size of the content
int32_t content_len;
trailer_pos += 1 + sizeof(content_len);  // 1 for marker, 4 for int32
if (trailer_len < trailer_pos) {
  return BLOSC2_ERROR_READ_BUFFER;
}
from_big(&content_len, content_marker + 1, sizeof(content_len));
if (content_len < 0) {
  return BLOSC2_ERROR_DATA;
}
layer_metadata->content_len = content_len;

// Validate that there's enough space for the content
trailer_pos += content_len;
if (trailer_len < trailer_pos) {
  return BLOSC2_ERROR_READ_BUFFER;
}
