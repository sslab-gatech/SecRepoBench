// Read the content length from the content marker
    int32_t content_len;
    from_big(&content_len, content_marker + 1, sizeof(content_len));
    
    // Validate the content length
    if (content_len < 0 || content_len > trailer_len - offset) {
        return BLOSC2_ERROR_DATA;
    }

    // Allocate memory for the content
    uint8_t* content = malloc((size_t)content_len);
    if (content == NULL) {
        return BLOSC2_ERROR_MEMORY;
    }

    // Copy the content data into the metalayer
    memcpy(content, content_marker + 1 + sizeof(content_len), (size_t)content_len);
    layer_metadata->content = content;
