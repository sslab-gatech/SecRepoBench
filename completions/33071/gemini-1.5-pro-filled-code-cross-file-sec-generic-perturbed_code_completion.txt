if (*content_marker != 0xd1) {   // sanity check
  return BLOSC2_ERROR_DATA;
}
content_marker++;
int32_t content_len;
trailer_pos += sizeof(content_len) + 1;
if (trailer_len < trailer_pos) {
  return BLOSC2_ERROR_READ_BUFFER;
}

from_big(&content_len, content_marker, sizeof(content_len));
content_marker += sizeof(content_len);
trailer_pos += content_len;
if (trailer_len < trailer_pos) {
  return BLOSC2_ERROR_READ_BUFFER;
}
layer_metadata->content_len = content_len;
