if (LocaleCompare(label, "id") == 0)
{
  // Copy the value from options to the id variable securely
  (void) CopyMagickString(id, options, MagickPathExtent);
}
else if (LocaleCompare(label, "iterations") == 0)
{
  // Convert the value from options to an unsigned long and assign it to image->iterations
  image->iterations = StringToUnsignedLong(options);
}
else
{
  // For all other labels, set the image property using the label and options values
  (void) SetImageProperty(image, label, options, exception);
}
