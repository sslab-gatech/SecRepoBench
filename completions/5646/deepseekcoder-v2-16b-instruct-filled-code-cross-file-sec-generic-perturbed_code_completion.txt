if (strncasecmp(next_token, "url(", 4) == 0) {
    char *url_start = next_token + 4;
    if (*url_start == '#') {
      url_start++;
    }
    size_t url_length = strlen(url_start);
    for (size_t j = 0; j < url_length; j++) {
      if (next_token[j + 4] == ')') {
        url_length = j;
        break;
      }
    }
    memmove(next_token, url_start, url_length);
    next_token[url_length] = '\0';
  }
