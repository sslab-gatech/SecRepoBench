// Check if the token starts with "url(".
if (strncmp(next_token, "url(", 4) == 0) {
    // Extract the actual URL by skipping the "url(" prefix and optional '#' character.
    const char* url = next_token + 4;
    if (*url == '#') {
        url++;
    }
    size_t url_len = strlen(url) - 1;
    if (url[url_len] == ')')
        url_len--;
    memmove(next_token, url, url_len + 1);
    next_token[url_len] = '\0';
}
