if (strncmp(next_token,"url(",4) == 0)
{
  char
    *closing;

  closing=strchr(next_token+4,')');
  if (closing != (char *) NULL)
  {
    size_t
      length;

    char
      *url;

    url=next_token+4;
    length=(size_t) (closing-url);
    if ((length > 0) && (*url == '#'))
      {
        url++;
        length--;
      }
    (void) memmove(next_token,url,length);
    next_token[length]='\0';
  }
}
