next_token[i]='\0';
if ((i >= 4) && (strncmp(next_token,"url(",4) == 0))
{
  char
    *end_url;

  end_url=strchr(next_token+4,')');
  if (end_url != (char *) NULL)
  {
    char
      *url;

    size_t
      url_length;

    url=next_token+4;
    url_length=(size_t)(end_url-url);
    if ((url_length > 0) && (*url == '#'))
      {
        url++;
        url_length--;
      }
    (void) memmove(next_token,url,url_length);
    next_token[url_length]='\0';
    i=(ssize_t) url_length;
  }
}
