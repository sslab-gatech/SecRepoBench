next_token[i]='\0';
if (strncmp(next_token, "url(", 4) == 0)
{
  char *url = next_token + 4;
  if (*url == '#')
    url++;
  size_t url_length = strlen(url);
  if (url_length > 0 && url[url_length - 1] == ')')
    url[url_length - 1] = '\0';
  memmove(next_token, url, url_length);
  next_token[url_length] = '\0';
  i = (ssize_t)url_length;
}
