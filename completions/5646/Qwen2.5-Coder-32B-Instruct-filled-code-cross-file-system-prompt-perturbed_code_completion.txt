next_token[i] = '\0';
if (strncmp(next_token, "url(", 4) == 0) {
  char *url_start = next_token + 4;
  char *url_end = strchr(url_start, ')');
  if (url_end != NULL) {
    size_t url_length = url_end - url_start;
    if (*url_start == '#') {
      url_start++;
      url_length--;
    }
    if (url_length < extent) {
      memmove(next_token, url_start, url_length);
      next_token[url_length] = '\0';
    }
  }
