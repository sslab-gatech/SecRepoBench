next_token[i]='\0';
if (strncmp(next_token, "url(", 4) == 0)
{
  char *url = next_token + 4;
  size_t url_length = strlen(url);
  if (url_length > 0 && url[url_length - 1] == ')')
  {
    url[url_length - 1] = '\0';
    url_length--;
  }
  if (url_length > 0 && *url == '#')
  {
    url++;
    url_length--;
  }
  if (url_length > 0)
  {
    memmove(next_token, url, url_length + 1);
    i = url_length;
  }
}
