next_token[i] = '\0';
if ((strlen(next_token) >= 4) && (strncmp(next_token, "url(", 4) == 0))
{
  char *url_start = next_token + 4;
  if (*url_start == '#')
    url_start++;
  {
    size_t url_len = strlen(url_start);
    if ((url_len > 0) && (url_start[url_len - 1] == ')'))
      url_start[url_len - 1] = '\0';
  }
  memmove(next_token, url_start, strlen(url_start) + 1);
}
