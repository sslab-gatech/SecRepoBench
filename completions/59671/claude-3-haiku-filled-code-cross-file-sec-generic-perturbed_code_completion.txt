default:
    switch (s->mv_type) {
    case MV_TYPE_FIELD:
        mpeg_motion_lowres(s, dest_y, dest_cb, dest_cr,
                           1, s->field_select[dir][0], s->field_select[dir][0],
                           ref_picture, pix_op,
                           s->mv[dir][0][0], s->mv[dir][0][1],
                           s->mb_height >> 1, s->mb_y);
        break;
    case MV_TYPE_16X8:
        mpeg_motion_lowres(s, dest_y, dest_cb, dest_cr,
                           1, 0, 0,
                           ref_picture, pix_op,
                           s->mv[dir][0][0], s->mv[dir][0][1],
                           8, s->mb_y * 2);
        mpeg_motion_lowres(s, dest_y, dest_cb, dest_cr,
                           1, 1, 1,
                           ref_picture, pix_op,
                           s->mv[dir][1][0], s->mv[dir][1][1],
                           8, s->mb_y * 2 + 1);
        break;
    case MV_TYPE_DUAL_PRIME:
        mpeg_motion_lowres(s, dest_y, dest_cb, dest_cr,
                           1, 0, 0,
                           ref_picture, pix_op,
                           s->mv[dir][0][0], s->mv[dir][0][1],
                           s->mb_height >> 1, s->mb_y);
        mpeg_motion_lowres(s, dest_y, dest_cb, dest_cr,
                           1, 1, 1,
                           ref_picture, pix_op,
                           s->mv[dir][1][0], s->mv[dir][1][1],
                           s->mb_height >> 1, s->mb_y);
        break;
    default:
        av_assert2(0);
    }
