case MV_TYPE_FIELD:
        for (int i = 0; i < 2; i++) {
            int field_select = s->field_select[dir][i];
            int motion_x = s->mv[dir][i][0];
            int motion_y = s->mv[dir][i][1];
            mpeg_motion_lowres(s, dest_y + i * s->linesize, dest_cb + i * s->uvlinesize,
                               dest_cr + i * s->uvlinesize, 1, i, field_select,
                               ref_picture, pix_op, motion_x, motion_y,
                               block_s, macroblocky);
        }
        break;
    case MV_TYPE_16X8:
        for (int i = 0; i < 2; i++) {
            int motion_x = s->mv[dir][i][0];
            int motion_y = s->mv[dir][i][1];
            mpeg_motion_lowres(s, dest_y + i * block_s * s->linesize, dest_cb + i * (block_s >> 1) * s->uvlinesize,
                               dest_cr + i * (block_s >> 1) * s->uvlinesize, 0, 0, 0,
                               ref_picture, pix_op, motion_x, motion_y,
                               block_s, macroblocky * 2 + i);
        }
        break;
    case MV_TYPE_DMV:
        for (int i = 0; i < 2; i++) {
            int motion_x = s->mv[dir][0][0] + s->dmv[dir][i][0];
            int motion_y = s->mv[dir][0][1] + s->dmv[dir][i][1];
            mpeg_motion_lowres(s, dest_y + i * s->linesize, dest_cb + i * s->uvlinesize,
                               dest_cr + i * s->uvlinesize, 1, i, i,
                               ref_picture, pix_op, motion_x, motion_y,
                               block_s, macroblocky);
        }
        break;
