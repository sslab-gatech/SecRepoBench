case MV_TYPE_FIELD:
    if (s->picture_structure == PICT_FRAME) {
        /* top field */
        mpeg_motion_lowres(s, dest_y, dest_cb, dest_cr,
                           1, 0, s->field_select[dir][0],
                           ref_picture, pix_op,
                           s->mv[dir][0][0], s->mv[dir][0][1],
                           block_s, macroblocky);
        /* bottom field */
        mpeg_motion_lowres(s, dest_y + s->linesize,
                           dest_cb + s->uvlinesize,
                           dest_cr + s->uvlinesize,
                           1, 1, s->field_select[dir][1],
                           ref_picture, pix_op,
                           s->mv[dir][1][0], s->mv[dir][1][1],
                           block_s, macroblocky);
    } else {
        mpeg_motion_lowres(s, dest_y, dest_cb, dest_cr,
                           0, 0, s->field_select[dir][0],
                           ref_picture, pix_op,
                           s->mv[dir][0][0], s->mv[dir][0][1],
                           2 * block_s, macroblocky);
    }
    break;
case MV_TYPE_16X8:
    for (int i = 0; i < 2; i++) {
        uint8_t *const dest_y_base = dest_y + i * 8 * s->linesize;
        mpeg_motion_lowres(s, dest_y_base, dest_cb + i * 4 * s->uvlinesize,
                           dest_cr + i * 4 * s->uvlinesize,
                           0, 0, s->field_select[dir][i],
                           ref_picture, pix_op,
                           s->mv[dir][i][0], s->mv[dir][i][1],
                           block_s, macroblocky + i);
    }
    break;
case MV_TYPE_DMV:
    {
        int dmv_x[2], dmv_y[2];
        for (int i = 0; i < 2; i++) {
            int opposite = s->picture_structure != i + 1;
            if (s->picture_structure == PICT_FRAME) {
                dmv_x[i] = s->mv[dir][2 * i][0];
                dmv_y[i] = s->mv[dir][2 * i][1];
                if (opposite) {
                    dmv_x[i] = (dmv_x[i] + 1) >> 1;
                    dmv_y[i] = (dmv_y[i] + 1) >> 1;
                }
            } else {
                dmv_x[i] = s->mv[dir][i][0];
                dmv_y[i] = s->mv[dir][i][1];
                if (opposite) {
                    dmv_x[i] = (dmv_x[i] + 1) >> 1;
                    dmv_y[i] = (dmv_y[i] + 1) >> 1;
                }
            }
        }
        mpeg_motion_lowres(s, dest_y, dest_cb, dest_cr,
                           0, 0, s->picture_structure != PICT_BOTTOM_FIELD,
                           ref_picture, pix_op,
                           dmv_x[0], dmv_y[0],
                           block_s, macroblocky);
        mpeg_motion_lowres(s, dest_y, dest_cb, dest_cr,
                           0, 0, s->picture_structure != PICT_TOP_FIELD,
                           ref_picture, pix_op,
                           dmv_x[1], dmv_y[1],
                           block_s, macroblocky);
        if (!s->quarter_sample) {
            uint8_t *dest_y_base = dest_y;
            for (int i = 0; i < 2; i++) {
                for (int j = 0; j < block_s; j++) {
                    for (int k = 0; k < block_s * 2; k++) {
                        dest_y_base[k] = (dest_y_base[k] + dest_y_base[k + s->linesize] + 1) >> 1;
                    }
                    dest_y_base += s->linesize;
                }
                dest_y_base = dest_y + s->linesize * block_s;
            }
        }
    }
    break;
