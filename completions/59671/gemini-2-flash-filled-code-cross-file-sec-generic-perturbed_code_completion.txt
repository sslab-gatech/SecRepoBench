case MV_TYPE_FIELD: {
        const int field_select = s->field_select[dir];
        const int bottom_field = s->picture_structure == PICT_BOTTOM_FIELD;

        mpeg_motion_lowres(s, dest_y, dest_cb, dest_cr,
                           1, bottom_field, field_select,
                           ref_picture, pix_op,
                           s->mv[dir][0][0], s->mv[dir][0][1],
                           1 * block_s, macroblocky);
        break;
    }
    case MV_TYPE_16X8: {
        mpeg_motion_lowres(s, dest_y, dest_cb, dest_cr,
                           0, 0, 0,
                           ref_picture, pix_op,
                           s->mv[dir][0][0], s->mv[dir][0][1],
                           1 * block_s, macroblocky + 0);
        mpeg_motion_lowres(s, dest_y + block_s * s->linesize,
                           dest_cb, dest_cr,
                           0, 0, 0,
                           ref_picture, pix_op,
                           s->mv[dir][1][0], s->mv[dir][1][1],
                           1 * block_s, macroblocky + 1);
        break;
    }
    case MV_TYPE_DMV: {
        int mx1 = s->mv[0][0][0];
        int my1 = s->mv[0][0][1];
        int mx2 = s->mv[1][0][0];
        int my2 = s->mv[1][0][1];
        uint8_t *tmp = s->edge_emu_buffer;

        if (s->picture_structure == PICT_FRAME) {
            mx1 += mx2;
            my1 += my2;
            mx1 /= 2;
            my1 /= 2;
        }

        mpeg_motion_lowres(s, dest_y, dest_cb, dest_cr,
                           0, 0, 0,
                           ref_picture, pix_op,
                           mx1, my1,
                           2 * block_s, macroblocky);
        break;
    }
