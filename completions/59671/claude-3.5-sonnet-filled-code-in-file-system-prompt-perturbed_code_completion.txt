case MV_TYPE_FIELD:
    if (s->picture_structure == PICT_FRAME) {
        /* top field */
        mpeg_motion_lowres(s, dest_y, dest_cb, dest_cr,
                           1, 0, s->field_select[dir][0],
                           ref_picture, pix_op,
                           s->mv[dir][0][0], s->mv[dir][0][1],
                           block_s, macroblocky);
        /* bottom field */
        mpeg_motion_lowres(s, dest_y, dest_cb, dest_cr,
                           1, 1, s->field_select[dir][1],
                           ref_picture, pix_op,
                           s->mv[dir][1][0], s->mv[dir][1][1],
                           block_s, macroblocky);
    } else {
        mpeg_motion_lowres(s, dest_y, dest_cb, dest_cr,
                           0, 0, s->field_select[dir][0],
                           ref_picture, pix_op,
                           s->mv[dir][0][0], s->mv[dir][0][1],
                           2 * block_s, macroblocky);
    }
    break;
case MV_TYPE_16X8:
    for (int i = 0; i < 2; i++) {
        uint8_t *const dest_y_base = dest_y + i * 8 * s->linesize;
        mpeg_motion_lowres(s, dest_y_base, dest_cb, dest_cr,
                           0, 0, s->field_select[dir][i],
                           ref_picture, pix_op,
                           s->mv[dir][i][0], s->mv[dir][i][1] + 2 * i * block_s,
                           block_s, macroblocky + i);
    }
    break;
case MV_TYPE_DMV:
    {
        int dmv_x[2], dmv_y[2];
        for (int i = 0; i < 2; i++) {
            int opposite = s->picture_structure != i + 1;
            if (s->picture_structure == PICT_FRAME) {
                dmv_x[i] = s->mv[dir][2 * i][0];
                dmv_y[i] = s->mv[dir][2 * i][1];
                if (opposite) {
                    dmv_x[i] = (dmv_x[i] + 1) >> 1;
                    dmv_y[i] = (dmv_y[i] + 1) >> 1;
                }
                mpeg_motion_lowres(s, dest_y, dest_cb, dest_cr,
                                   opposite, opposite, 0,
                                   ref_picture, pix_op,
                                   dmv_x[i], dmv_y[i],
                                   block_s, macroblocky);
            } else {
                dmv_x[i] = s->mv[dir][i][0];
                dmv_y[i] = s->mv[dir][i][1];
                if (opposite) {
                    dmv_x[i] = (dmv_x[i] + 1) >> 1;
                    dmv_y[i] = (dmv_y[i] + 1) >> 1;
                }
                /* top/bottom field */
                mpeg_motion_lowres(s, dest_y, dest_cb, dest_cr,
                                   0, 0, 0,
                                   ref_picture, pix_op,
                                   dmv_x[i], dmv_y[i],
                                   2 * block_s, macroblocky);
            }
        }
    }
    break;
