case MV_TYPE_FIELD:
        mpeg_motion_lowres(s, dest_y, dest_cb, dest_cr,
                           1, 0, s->field_select[dir][0],
                           ref_picture, pix_op,
                           s->mv[dir][0][0], s->mv[dir][0][1],
                           2 * block_s, macroblocky);
        if (s->picture_structure == PICT_FRAME) {
            mpeg_motion_lowres(s, dest_y, dest_cb, dest_cr,
                               1, 1, s->field_select[dir][1],
                               ref_picture, pix_op,
                               s->mv[dir][1][0], s->mv[dir][1][1],
                               2 * block_s, macroblocky);
        }
        break;
    case MV_TYPE_16X8:
        mpeg_motion_lowres(s, dest_y, dest_cb, dest_cr,
                           0, 0, 0,
                           ref_picture, pix_op,
                           s->mv[dir][0][0], s->mv[dir][0][1],
                           block_s, macroblocky * 2);
        mpeg_motion_lowres(s, dest_y + block_s * s->linesize, dest_cb, dest_cr,
                           0, 0, 0,
                           ref_picture, pix_op,
                           s->mv[dir][1][0], s->mv[dir][1][1],
                           block_s, macroblocky * 2 + 1);

        break;
    case MV_TYPE_DMV:
        mpeg_motion_lowres(s, dest_y, dest_cb, dest_cr,
                           1, 0, 0,
                           ref_picture, pix_op,
                           s->mv[dir][0][0], s->mv[dir][0][1],
                           2 * block_s, macroblocky);
        if (s->picture_structure == PICT_FRAME) {
            /* yes, the fancy field select stuff is needed here */
            mpeg_motion_lowres(s, dest_y, dest_cb, dest_cr,
                               1, 1, 1,
                               ref_picture, pix_op,
                               s->mv[dir][1][0], s->mv[dir][1][1],
                               2 * block_s, macroblocky);

            s->mpv_dmv_compensate_lowres(s, dest_y, dest_cb, dest_cr, dir);
        }
        break;
