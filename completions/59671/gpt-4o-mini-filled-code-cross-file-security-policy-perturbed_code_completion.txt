case MV_TYPE_16X8:
        for (int i = 0; i < 2; i++) {
            if (ref_picture[0]) {
                mpeg_motion_lowres(s, dest_y + (i * block_s * s->linesize),
                                   dest_cb, dest_cr,
                                   0, 0, 0,
                                   ref_picture, pix_op,
                                   s->mv[dir][i][0], s->mv[dir][i][1],
                                   block_s, macroblocky);
            }
        }
        break;
    case MV_TYPE_FIELD:
        if (ref_picture[0] && ref_picture[1]) {
            mpeg_motion_lowres(s, dest_y, dest_cb, dest_cr,
                               1, 0, 0,
                               ref_picture, pix_op,
                               s->mv[dir][0][0], s->mv[dir][0][1],
                               2 * block_s, macroblocky);
            mpeg_motion_lowres(s, dest_y + s->linesize, dest_cb, dest_cr,
                               1, 1, 0,
                               ref_picture, pix_op,
                               s->mv[dir][1][0], s->mv[dir][1][1],
                               2 * block_s, macroblocky);
        }
        break;
    case MV_TYPE_DMV:
        if (ref_picture[0] && ref_picture[1]) {
            mpeg_motion_lowres(s, dest_y, dest_cb, dest_cr,
                               0, 0, 0,
                               ref_picture, pix_op,
                               (s->mv[dir][0][0] + s->mv[dir][1][0]) / 2,
                               (s->mv[dir][0][1] + s->mv[dir][1][1]) / 2,
                               2 * block_s, macroblocky);
        }
        break;
