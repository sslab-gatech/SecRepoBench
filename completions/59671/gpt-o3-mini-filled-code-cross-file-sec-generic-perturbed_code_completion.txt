case MV_TYPE_FIELD:
        mpeg_motion_lowres(s, dest_y, dest_cb, dest_cr,
                           1, 0, 0,
                           ref_picture, pix_op,
                           s->mv[dir][0][0], s->mv[dir][0][1],
                           2 * block_s, macroblocky);
        break;
    case MV_TYPE_16X8:
        /* Process top half */
        mpeg_motion_lowres(s, dest_y, dest_cb, dest_cr,
                           0, 0, 0,
                           ref_picture, pix_op,
                           s->mv[dir][0][0], s->mv[dir][0][1],
                           block_s, macroblocky);
        /* Process bottom half */
        mpeg_motion_lowres(s, dest_y + block_s * s->linesize,
                           dest_cb, dest_cr,
                           0, 0, 0,
                           ref_picture, pix_op,
                           s->mv[dir][1][0], s->mv[dir][1][1],
                           block_s, macroblocky);
        if (!CONFIG_GRAY || !(s->avctx->flags & AV_CODEC_FLAG_GRAY))
            chroma_4mv_motion_lowres(s, dest_cb, dest_cr, ref_picture, pix_op,
                                     s->mv[dir][0][0] + s->mv[dir][1][0],
                                     s->mv[dir][0][1] + s->mv[dir][1][1]);
        break;
    case MV_TYPE_DMV:
    {
        int avg_mv_x = (s->mv[dir][0][0] + s->mv[dir][1][0] + 1) >> 1;
        int avg_mv_y = (s->mv[dir][0][1] + s->mv[dir][1][1] + 1) >> 1;
        mpeg_motion_lowres(s, dest_y, dest_cb, dest_cr,
                           0, 0, 0,
                           ref_picture, pix_op,
                           avg_mv_x, avg_mv_y,
                           2 * block_s, macroblocky);
        break;
    }
