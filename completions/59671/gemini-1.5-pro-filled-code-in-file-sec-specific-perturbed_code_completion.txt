case MV_TYPE_FIELD:
        if (s->picture_structure == PICT_FRAME) {
            mpeg_motion_lowres(s, dest_y, dest_cb, dest_cr,
                               1, 0, 0,
                               ref_picture, pix_op,
                               s->mv[dir][0][0], s->mv[dir][0][1],
                               block_s, macroblocky);
            // in case of field motion vectors,
            // we construct a single chroma motion vector
            // with a special rounding
            mpeg_motion_lowres(s, dest_y + s->linesize, dest_cb, dest_cr,
                               1, 1, 1,
                               ref_picture, pix_op,
                               s->mv[dir][1][0], s->mv[dir][1][1],
                               block_s, macroblocky);

        } else {
            if (s->picture_structure != s->field_select)
                ref_picture = s->next_picture.f->data;

            mpeg_motion_lowres(s, dest_y, dest_cb, dest_cr,
                               0, 0, s->field_select,
                               ref_picture, pix_op,
                               s->mv[dir][0][0], s->mv[dir][0][1],
                               2 * block_s, macroblocky >> 1);
        }
        break;
    case MV_TYPE_16X8:
        mpeg_motion_lowres(s, dest_y, dest_cb, dest_cr,
                           0, 0, 0,
                           ref_picture, pix_op,
                           s->mv[dir][0][0], s->mv[dir][0][1],
                           block_s, macroblocky);
        mpeg_motion_lowres(s, dest_y + block_s*s->linesize, dest_cb, dest_cr,
                           0, 0, 0,
                           ref_picture, pix_op,
                           s->mv[dir][1][0], s->mv[dir][1][1],
                           block_s, macroblocky);

        break;
    case MV_TYPE_DMV:
        mpeg_motion_lowres(s, dest_y, dest_cb, dest_cr,
                           0, 0, 0,
                           ref_picture, pix_op,
                           s->mv[dir][0][0], s->mv[dir][0][1],
                           2 * block_s, macroblocky);
        if (s->picture_structure == PICT_FRAME) {
            for (int i=0; i<2*block_s; i++) {
                for (int j=0; j<2*block_s; j++) {
                    dest_y[i*s->linesize + j] = (dest_y[i*s->linesize + j] +
                        s->last_picture.f->data[0][(macroblocky * 2 * block_s + i)*s->linesize + mb_x * 2 * block_s + j] + 1) >> 1;
                }
            }
        }
        break;
