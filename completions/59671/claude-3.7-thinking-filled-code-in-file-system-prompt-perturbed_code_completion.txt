case MV_TYPE_FIELD:
    if (s->picture_structure == PICT_FRAME) {
        /* top field */
        mpeg_motion_lowres(s, dest_y, dest_cb, dest_cr,
                           1, 0, s->field_select[dir][0],
                           ref_picture, pix_op,
                           s->mv[dir][0][0], s->mv[dir][0][1],
                           block_s, macroblocky);
        /* bottom field */
        mpeg_motion_lowres(s, dest_y, dest_cb, dest_cr,
                           1, 1, s->field_select[dir][1],
                           ref_picture, pix_op,
                           s->mv[dir][1][0], s->mv[dir][1][1],
                           block_s, macroblocky);
    } else {
        mpeg_motion_lowres(s, dest_y, dest_cb, dest_cr,
                           0, 0, s->field_select[dir][0],
                           ref_picture, pix_op,
                           s->mv[dir][0][0], s->mv[dir][0][1],
                           2 * block_s, macroblocky);
    }
    break;
case MV_TYPE_16X8:
    for (int i = 0; i < 2; i++) {
        uint8_t *dest_y_base = dest_y + i * block_s * s->linesize;
        uint8_t *dest_cb_base = dest_cb + (i * block_s >> s->chroma_y_shift) * s->uvlinesize;
        uint8_t *dest_cr_base = dest_cr + (i * block_s >> s->chroma_y_shift) * s->uvlinesize;
        
        mpeg_motion_lowres(s, dest_y_base, dest_cb_base, dest_cr_base,
                           0, 0, s->field_select[dir][i],
                           ref_picture, pix_op,
                           s->mv[dir][i][0], s->mv[dir][i][1],
                           block_s, macroblocky + i);
    }
    break;
case MV_TYPE_DMV:
    {
        int field_select = s->picture_structure != PICT_BOTTOM_FIELD;
        
        if (s->picture_structure == PICT_FRAME) {
            /* top field */
            mpeg_motion_lowres(s, dest_y, dest_cb, dest_cr,
                               1, 0, 0,
                               ref_picture, pix_op,
                               s->mv[dir][0][0], s->mv[dir][0][1],
                               block_s, macroblocky);
            /* bottom field */
            mpeg_motion_lowres(s, dest_y, dest_cb, dest_cr,
                               1, 1, 0,
                               ref_picture, pix_op,
                               s->mv[dir][0][0], s->mv[dir][0][1],
                               block_s, macroblocky);
        } else {
            int opposite_field_select = !field_select;
            int bottom_field = (s->picture_structure == PICT_BOTTOM_FIELD);
            
            mpeg_motion_lowres(s, dest_y, dest_cb, dest_cr,
                               0, 0, field_select,
                               ref_picture, pix_op,
                               s->mv[dir][0][0], s->mv[dir][0][1],
                               2 * block_s, macroblocky);
            
            /* Compute the second motion vector */
            int dmx = s->mv[dir][0][0];
            int dmy = s->mv[dir][0][1];
            
            if (bottom_field) {
                dmx = (dmx * lowres) >> (lowres + 1);
                dmy = ((dmy * lowres) >> (lowres + 1)) - 1;
            } else {
                dmx = (dmx * lowres) >> (lowres + 1);
                dmy = ((dmy * lowres) >> (lowres + 1)) + 1;
            }
            
            mpeg_motion_lowres(s, dest_y, dest_cb, dest_cr,
                               0, 0, opposite_field_select,
                               ref_picture, pix_op,
                               dmx, dmy,
                               2 * block_s, macroblocky);
        }
    }
    break;
