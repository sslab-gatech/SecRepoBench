case MV_TYPE_FIELD:
        if (s->picture_structure == PICT_FRAME) {
            mpeg_motion_lowres(s, dest_y, dest_cb, dest_cr,
                               1, 0, s->field_select[dir][0],
                               ref_picture, pix_op,
                               s->mv[dir][0][0], s->mv[dir][0][1],
                               2 * block_s, macroblocky);
        } else {
            for (int i = 0; i < 2; i++) {
                mpeg_motion_lowres(s, dest_y, dest_cb, dest_cr,
                                   1, i, s->field_select[dir][i],
                                   ref_picture, pix_op,
                                   s->mv[dir][i][0], s->mv[dir][i][1],
                                   block_s, macroblocky);

                dest_y += s->linesize << (1 - lowres);
            }
        }
        break;
    case MV_TYPE_16X8:
        for (int i = 0; i < 2; i++) {
            mpeg_motion_lowres(s, dest_y, dest_cb, dest_cr,
                               0, 0, 0,
                               ref_picture, pix_op,
                               s->mv[dir][i][0], s->mv[dir][i][1],
                               2 * block_s, macroblocky + i);

            dest_y += (s->linesize * block_s);
        }
        break;
    case MV_TYPE_DMV:
        if (s->picture_structure == PICT_FRAME) {
            for (int i = 0; i < 2; i++) {
                mpeg_motion_lowres(s, dest_y, dest_cb, dest_cr,
                                   1, i, s->field_select[dir][i],
                                   ref_picture, pix_op,
                                   s->mv[dir][0][0], s->mv[dir][0][1],
                                   block_s, macroblocky);
                // opposite parity is always in the same frame if this is DMV
                // Note, we are using the same motion vectors for both fields,
                // as we are only using DMV intra frame coding
                mpeg_motion_lowres(s, dest_y, dest_cb, dest_cr,
                                   1, 1 - i, s->field_select[dir][i],
                                   ref_picture, pix_op,
                                   s->mv[dir][0][0], s->mv[dir][0][1],
                                   block_s, macroblocky);

                s->mpv_adjust_luma(s, dest_y, s->linesize,
                                   s->mv[dir][0][0], s->mv[dir][0][1], 1,
                                   s->field_select[dir][i],
                                   s->field_select[dir][1 - i]);

                dest_y += s->linesize << (1 - lowres);
            }
        } else {
            for (int i = 0; i < 2; i++) {
                mpeg_motion_lowres(s, dest_y, dest_cb, dest_cr,
                                   1, i, s->field_select[dir][i],
                                   ref_picture, pix_op,
                                   s->mv[dir][i][0], s->mv[dir][i][1],
                                   block_s, macroblocky);

                dest_y += s->linesize << (1 - lowres);
            }
        }
        break;
