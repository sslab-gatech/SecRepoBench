case MV_TYPE_FIELD:
        for (int i = 0; i < 2; i++) {
            int field_select = s->field_select[dir][i];
            int motion_x = s->mv[dir][i][0];
            int motion_y = s->mv[dir][i][1];
            mpeg_motion_lowres(s, dest_y, dest_cb, dest_cr,
                               1, i, field_select,
                               ref_picture, pix_op,
                               motion_x, motion_y,
                               block_s, macroblocky);
        }
        break;
    case MV_TYPE_16X8:
        for (int i = 0; i < 2; i++) {
            int motion_x = s->mv[dir][i][0];
            int motion_y = s->mv[dir][i][1];
            mpeg_motion_lowres(s, dest_y + i * block_s * s->linesize,
                               dest_cb + i * (block_s >> s->chroma_y_shift) * s->uvlinesize,
                               dest_cr + i * (block_s >> s->chroma_y_shift) * s->uvlinesize,
                               0, 0, 0,
                               ref_picture, pix_op,
                               motion_x, motion_y,
                               block_s, macroblocky * 2 + i);
        }
        break;
    case MV_TYPE_DMV:
        for (int i = 0; i < 2; i++) {
            int motion_x = s->mv[dir][i][0];
            int motion_y = s->mv[dir][i][1];
            mpeg_motion_lowres(s, dest_y, dest_cb, dest_cr,
                               1, i, 0,
                               ref_picture, pix_op,
                               motion_x, motion_y,
                               block_s, macroblocky);
        }
        break;
