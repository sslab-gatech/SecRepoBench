case MV_TYPE_FIELD:
        for (int i = 0; i < 2; i++) {
            mpeg_motion_lowres(s,
                               dest_y, dest_cb, dest_cr,
                               1, // field_based
                               s->picture_structure == PICT_BOTTOM_FIELD,
                               s->field_select[dir][i],
                               ref_picture,
                               pix_op,
                               s->mv[dir][i][0],
                               s->mv[dir][i][1],
                               2 * block_s,
                               macroblocky);
        }
        break;

    case MV_TYPE_16X8:
        mpeg_motion_lowres(s,
                           dest_y, dest_cb, dest_cr,
                           0, 0, 0,
                           ref_picture,
                           pix_op,
                           s->mv[dir][0][0],
                           s->mv[dir][0][1],
                           block_s,
                           macroblocky * 2);
        mpeg_motion_lowres(s,
                           dest_y + block_s * s->linesize,
                           dest_cb + (block_s >> s->chroma_y_shift) * s->uvlinesize,
                           dest_cr + (block_s >> s->chroma_y_shift) * s->uvlinesize,
                           0, 0, 0,
                           ref_picture,
                           pix_op,
                           s->mv[dir][1][0],
                           s->mv[dir][1][1],
                           block_s,
                           macroblocky * 2 + 1);
        break;

    case MV_TYPE_DMV: {
        // Dual-prime motion: average original motion vector with the two extra vectors
        int mx, my;
        // First field
        mx = (s->mv[dir][0][0] + s->mv[dir][1][0]) >> 1;
        my = (s->mv[dir][0][1] + s->mv[dir][1][1]) >> 1;
        mpeg_motion_lowres(s,
                           dest_y, dest_cb, dest_cr,
                           1,
                           s->picture_structure == PICT_BOTTOM_FIELD,
                           s->field_select[dir][0],
                           ref_picture,
                           pix_op,
                           mx,
                           my,
                           2 * block_s,
                           macroblocky);
        // Opposite field
        mx = (s->mv[dir][0][0] + s->mv[dir][2][0]) >> 1;
        my = (s->mv[dir][0][1] + s->mv[dir][2][1]) >> 1;
        mpeg_motion_lowres(s,
                           dest_y, dest_cb, dest_cr,
                           1,
                           s->picture_structure == PICT_BOTTOM_FIELD,
                           s->field_select[dir][1],
                           ref_picture,
                           pix_op,
                           mx,
                           my,
                           2 * block_s,
                           macroblocky);
    }
        break;
