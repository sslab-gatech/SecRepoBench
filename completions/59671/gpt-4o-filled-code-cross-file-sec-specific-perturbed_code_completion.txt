case MV_TYPE_16X8:
        for (int i = 0; i < 2; i++) {
            mpeg_motion_lowres(s, dest_y + i * block_s * s->linesize, dest_cb, dest_cr,
                               0, 0, 0,
                               ref_picture, pix_op,
                               s->mv[dir][i][0], s->mv[dir][i][1] + i * block_s,
                               block_s, macroblocky + i);
        }
        break;
    case MV_TYPE_FIELD:
        for (int i = 0; i < 2; i++) {
            mpeg_motion_lowres(s, dest_y, dest_cb, dest_cr,
                               1, i, s->field_select[dir][i],
                               ref_picture, pix_op,
                               s->mv[dir][i][0], s->mv[dir][i][1],
                               2 * block_s, macroblocky);
        }
        break;
    case MV_TYPE_DMV:
        for (int i = 0; i < 2; i++) {
            for (int j = 0; j < 2; j++) {
                mpeg_motion_lowres(s, dest_y, dest_cb, dest_cr,
                                   1, j, s->field_select[dir][i],
                                   ref_picture, pix_op,
                                   s->mv[dir][i][0] + s->dmv[dir][i][j][0],
                                   s->mv[dir][i][1] + s->dmv[dir][i][j][1],
                                   2 * block_s, macroblocky);
            }
        }
        break;
