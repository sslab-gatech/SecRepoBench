case MV_TYPE_FIELD:
        for (int i = 0; i < 2; i++) {
            mpeg_motion_lowres(s, dest_y, dest_cb, dest_cr,
                               1, // field_based
                               s->field_select[dir][i], // bottom_field
                               s->field_select[dir][i], // field_select
                               ref_picture, pix_op,
                               s->mv[dir][i][0], s->mv[dir][i][1],
                               2 * block_s, macroblocky * 2 + i);
        }
        break;
    case MV_TYPE_16X8:
        for (int i = 0; i < 2; i++) {
            uint8_t *dy  = dest_y  + i * (block_s * s->linesize);
            uint8_t *dcb = dest_cb + i * ((block_s >> s->chroma_y_shift) * s->uvlinesize);
            uint8_t *dcr = dest_cr + i * ((block_s >> s->chroma_y_shift) * s->uvlinesize);

            mpeg_motion_lowres(s, dy, dcb, dcr,
                               0, 0, 0,
                               ref_picture, pix_op,
                               s->mv[dir][i][0], s->mv[dir][i][1],
                               block_s, macroblocky * 2 + i);
        }
        break;
    case MV_TYPE_DMV:
        /* For dual-prime motion vectors, predictions from both fields are
           typically averaged. This snippet shows two calls for each field;
           the final combination step (averaging) can be implemented as needed. */
        for (int i = 0; i < 2; i++) {
            mpeg_motion_lowres(s, dest_y, dest_cb, dest_cr,
                               1, // field_based
                               s->field_select[dir][i], // bottom_field
                               s->field_select[dir][i], // field_select
                               ref_picture, pix_op,
                               s->mv[dir][i][0], s->mv[dir][i][1],
                               2 * block_s, macroblocky * 2);
        }
        break;
