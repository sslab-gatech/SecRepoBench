case MV_TYPE_FIELD:
    if (s->picture_structure == PICT_FRAME) {
        // Frame with field MV
        for (int i = 0; i < 2; i++) {
            int field_select = s->field_select[dir][i];
            mpeg_motion_lowres(s, dest_y, dest_cb, dest_cr,
                              1, i, field_select,
                              ref_picture, pix_op,
                              s->mv[dir][i][0], s->mv[dir][i][1],
                              block_s, macroblocky);
        }
    } else {
        // Field picture
        mpeg_motion_lowres(s, dest_y, dest_cb, dest_cr,
                          0, 0, s->field_select[dir][0],
                          ref_picture, pix_op,
                          s->mv[dir][0][0], s->mv[dir][0][1],
                          2 * block_s, macroblocky);
    }
    break;
case MV_TYPE_16X8:
    for (int i = 0; i < 2; i++) {
        int field_select = s->field_select[dir][i];
        mpeg_motion_lowres(s, dest_y + i * block_s * s->linesize,
                          dest_cb + (i * block_s >> s->chroma_y_shift) * s->uvlinesize,
                          dest_cr + (i * block_s >> s->chroma_y_shift) * s->uvlinesize,
                          0, 0, field_select,
                          ref_picture, pix_op,
                          s->mv[dir][i][0], s->mv[dir][i][1],
                          block_s, macroblocky + i);
    }
    break;
case MV_TYPE_DMV:
    // For DMV, we simplify to a single motion compensation for lowres
    mpeg_motion_lowres(s, dest_y, dest_cb, dest_cr,
                      s->picture_structure != PICT_FRAME,
                      s->picture_structure == PICT_BOTTOM_FIELD,
                      s->picture_structure != PICT_BOTTOM_FIELD,
                      ref_picture, pix_op,
                      s->mv[dir][0][0], s->mv[dir][0][1],
                      2 * block_s, macroblocky);
    break;
