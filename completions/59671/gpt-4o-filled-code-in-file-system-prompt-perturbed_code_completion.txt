case MV_TYPE_FIELD:
        for (int i = 0; i < 2; i++) {
            int field_select = s->field_select[dir][i];
            int field_based = 1;
            int bottom_field = i;
            mpeg_motion_lowres(s, dest_y, dest_cb, dest_cr,
                               field_based, bottom_field, field_select,
                               ref_picture, pix_op,
                               s->mv[dir][i][0], s->mv[dir][i][1],
                               block_s, macroblocky);
        }
        break;
    case MV_TYPE_16X8:
        for (int i = 0; i < 2; i++) {
            mpeg_motion_lowres(s, dest_y + i * block_s * s->linesize,
                               dest_cb, dest_cr,
                               0, 0, 0,
                               ref_picture, pix_op,
                               s->mv[dir][i][0], s->mv[dir][i][1],
                               block_s, macroblocky + i);
        }
        break;
    case MV_TYPE_DMV:
        for (int i = 0; i < 2; i++) {
            int field_select = s->field_select[dir][i];
            int field_based = 1;
            int bottom_field = i;
            mpeg_motion_lowres(s, dest_y, dest_cb, dest_cr,
                               field_based, bottom_field, field_select,
                               ref_picture, pix_op,
                               s->mv[dir][i][0], s->mv[dir][i][1],
                               block_s, macroblocky);
            mpeg_motion_lowres(s, dest_y, dest_cb, dest_cr,
                               field_based, bottom_field, !field_select,
                               ref_picture, pix_op,
                               s->mv[dir][i][0] / 2, s->mv[dir][i][1] / 2,
                               block_s, macroblocky);
        }
        break;
