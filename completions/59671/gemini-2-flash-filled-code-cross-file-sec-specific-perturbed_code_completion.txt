case MV_TYPE_FIELD: {
        const int field_select = s->field_select[dir][0];
        uint8_t *ref_picture_ptr[3];

        ref_picture_ptr[0] = ref_picture[0];
        ref_picture_ptr[1] = ref_picture[1];
        ref_picture_ptr[2] = ref_picture[2];

        mpeg_motion_lowres(s, dest_y, dest_cb, dest_cr,
                           1, 0, field_select,
                           ref_picture_ptr, pix_op,
                           s->mv[dir][0][0], s->mv[dir][0][1],
                           block_s * 2, macroblocky);
        break;
    }
    case MV_TYPE_16X8: {
        uint8_t *ref_picture_ptr[3];

        ref_picture_ptr[0] = ref_picture[0];
        ref_picture_ptr[1] = ref_picture[1];
        ref_picture_ptr[2] = ref_picture[2];

        mpeg_motion_lowres(s, dest_y, dest_cb, dest_cr,
                           0, 0, 0,
                           ref_picture_ptr, pix_op,
                           s->mv[dir][0][0], s->mv[dir][0][1],
                           block_s * 2, macroblocky);

        mpeg_motion_lowres(s, dest_y + 8 * s->linesize,
                           dest_cb + 4 * s->uvlinesize,
                           dest_cr + 4 * s->uvlinesize,
                           0, 0, 0,
                           ref_picture_ptr, pix_op,
                           s->mv[dir][1][0], s->mv[dir][1][1],
                           block_s * 2, macroblocky + 1);
        break;
    }
    case MV_TYPE_DMV: {
        int mx1 = s->mv[0][0][0];
        int my1 = s->mv[0][0][1];
        int mx2 = s->mv[1][0][0];
        int my2 = s->mv[1][0][1];
        uint8_t *ref_picture_ptr[3];

        ref_picture_ptr[0] = ref_picture[0];
        ref_picture_ptr[1] = ref_picture[1];
        ref_picture_ptr[2] = ref_picture[2];

        mx1 = (mx1 + mx2 + 1) >> 1;
        my1 = (my1 + my2 + 1) >> 1;

        mpeg_motion_lowres(s, dest_y, dest_cb, dest_cr,
                           1, 0, 0,
                           ref_picture_ptr, pix_op,
                           mx1, my1,
                           block_s * 2, macroblocky);
        break;
    }
