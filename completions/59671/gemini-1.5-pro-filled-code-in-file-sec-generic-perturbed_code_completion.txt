case MV_TYPE_FIELD:
        if (s->picture_structure == PICT_FRAME) {
            mpeg_motion_lowres(s, dest_y, dest_cb, dest_cr, dir,
                               0, 0, ref_picture, pix_op,
                               s->mv[dir][0][0], s->mv[dir][0][1],
                               block_s, macroblocky);

            mpeg_motion_lowres(s, dest_y + s->linesize, dest_cb, dest_cr,
                               dir, 1, 1, ref_picture, pix_op,
                               s->mv[dir][1][0], s->mv[dir][1][1],
                               block_s, macroblocky);

        } else {
            if(s->picture_structure != s->field_select){
                ref_picture = s->next_picture.f->data;
            }

            mpeg_motion_lowres(s, dest_y, dest_cb, dest_cr, dir,
                               1, s->field_select, ref_picture, pix_op,
                               s->mv[dir][0][0], s->mv[dir][0][1],
                               2 * block_s, macroblocky);
        }
        break;
    case MV_TYPE_16X8:
        mpeg_motion_lowres(s, dest_y, dest_cb, dest_cr,
                           0, 0, 0, ref_picture, pix_op,
                           s->mv[dir][0][0], s->mv[dir][0][1],
                           block_s, macroblocky);
        mpeg_motion_lowres(s, dest_y + block_s*s->linesize, dest_cb, dest_cr,
                           0, 0, 0, ref_picture, pix_op,
                           s->mv[dir][1][0], s->mv[dir][1][1],
                           block_s, macroblocky + 1);

        break;
    case MV_TYPE_DMV:
        mx = 0;
        my = 0;
        if (s->picture_structure == PICT_FRAME) {
            for (int i=0; i<2; i++) {
                mpeg_motion_lowres(s, dest_y + i*block_s, dest_cb, dest_cr,
                                   dir, 0, 0, ref_picture, pix_op,
                                   s->mv[dir][i][0], s->mv[dir][i][1],
                                   block_s, macroblocky);
                mx += s->mv[dir][i][0];
                my += s->mv[dir][i][1];
            }

            s->vdsp.avg_pixels(dest_y, dest_y + s->linesize, s->linesize, 16 >> lowres);

            if (!CONFIG_GRAY || !(s->avctx->flags & AV_CODEC_FLAG_GRAY))
                chroma_4mv_motion_lowres(s, dest_cb, dest_cr, ref_picture,
                                         pix_op, mx, my);

        } else {
            mpeg_motion_lowres(s, dest_y, dest_cb, dest_cr, dir,
                               1, s->field_select, ref_picture, pix_op,
                               s->mv[dir][0][0], s->mv[dir][0][1],
                               2 * block_s, macroblocky);
        }
        break;
