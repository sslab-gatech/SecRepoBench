case MV_TYPE_FIELD:
        if (s->picture_structure == PICT_FRAME) {
            mpeg_motion_lowres(s, dest_y, dest_cb, dest_cr,
                               1, 0, 0,
                               ref_picture, pix_op,
                               s->mv[dir][0][0], s->mv[dir][0][1],
                               block_s, macroblocky);
            // in case of field motion vectors, we interpolate the second field
            // from the other field of the same frame.
            // this is probably wrong but I doubt it matters much.
            mpeg_motion_lowres(s, dest_y + s->linesize, dest_cb + s->uvlinesize, dest_cr + s->uvlinesize,
                               1, 1, 1,
                               ref_picture, pix_op,
                               s->mv[dir][1][0], s->mv[dir][1][1],
                               block_s, macroblocky);
        } else {
            if(s->picture_structure != s->field_select){
                ref_picture = s->current_picture_ptr->f->data;
            }

            mpeg_motion_lowres(s, dest_y, dest_cb, dest_cr,
                               1, s->picture_structure == PICT_BOTTOM_FIELD, s->field_select,
                               ref_picture, pix_op,
                               s->mv[dir][0][0], s->mv[dir][0][1],
                               2 * block_s, macroblocky);
        }
        break;
    case MV_TYPE_16X8:
        mpeg_motion_lowres(s, dest_y, dest_cb, dest_cr,
                           0, 0, 0,
                           ref_picture, pix_op,
                           s->mv[dir][0][0], s->mv[dir][0][1],
                           block_s, macroblocky);
        mpeg_motion_lowres(s, dest_y + block_s * s->linesize, dest_cb + block_s*s->uvlinesize, dest_cr + block_s*s->uvlinesize,
                           0, 0, 0,
                           ref_picture, pix_op,
                           s->mv[dir][1][0], s->mv[dir][1][1],
                           block_s, macroblocky);

        break;
    case MV_TYPE_DMV:
        mpeg_motion_lowres(s, dest_y, dest_cb, dest_cr,
                           0, 0, 0,
                           ref_picture, pix_op,
                           s->mv[dir][0][0], s->mv[dir][0][1],
                           2 * block_s, macroblocky);
        if (s->picture_structure == PICT_FRAME) {
            for (int i=0; i<2*block_s*s->linesize; i++)
                dest_y[i] = (dest_y[i] + dest_y[i+s->linesize])>>1;
            for (int i=0; i<block_s*s->uvlinesize; i++) {
                dest_cb[i] = (dest_cb[i] + dest_cb[i+s->uvlinesize])>>1;
                dest_cr[i] = (dest_cr[i] + dest_cr[i+s->uvlinesize])>>1;
            }
        }
        break;
