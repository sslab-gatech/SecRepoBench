case MV_TYPE_FIELD:
        if (s->picture_structure == PICT_FRAME) {
            mpeg_motion_lowres(s, dest_y, dest_cb, dest_cr,
                               1, 0, s->field_select[dir][0],
                               ref_picture, pix_op,
                               s->mv[dir][0][0], s->mv[dir][0][1],
                               2 * block_s, macroblocky);
        } else {
            for (int i = 0; i < 2; i++) {
                mpeg_motion_lowres(s, dest_y, dest_cb, dest_cr,
                                   1, i, s->field_select[dir][i],
                                   ref_picture, pix_op,
                                   s->mv[dir][i][0], s->mv[dir][i][1],
                                   block_s, macroblocky);

                dest_y  += s->linesize << (lowres + 1);
                dest_cb += s->uvlinesize << (lowres + 1);
                dest_cr += s->uvlinesize << (lowres + 1);
            }
        }
        break;
    case MV_TYPE_16X8:
        for (int i = 0; i < 2; i++) {
            if (s->picture_structure == PICT_FRAME) {
                mpeg_motion_lowres(s, dest_y, dest_cb, dest_cr,
                                   0, 0, 0,
                                   ref_picture, pix_op,
                                   s->mv[dir][i][0], s->mv[dir][i][1],
                                   block_s, macroblocky + i);
            } else {
                mpeg_motion_lowres(s, dest_y, dest_cb, dest_cr,
                                   1, i, s->field_select[dir][i],
                                   ref_picture, pix_op,
                                   s->mv[dir][i][0], s->mv[dir][i][1],
                                   block_s, macroblocky >> 1);
            }
            dest_y  += (s->linesize << lowres) * (2 * block_s - 1 + !!lowres); //FIXME optimize
            dest_cb += s->uvlinesize << lowres;
            dest_cr += s->uvlinesize << lowres;
        }
        break;
    case MV_TYPE_DMV:
        if (s->picture_structure == PICT_FRAME) {
            for (int i = 0; i < 2; i++) {
                mpeg_motion_lowres(s, dest_y, dest_cb, dest_cr,
                                   0, 0, 0,
                                   ref_picture, pix_op,
                                   s->mv[dir][0][0], s->mv[dir][0][1],
                                   block_s, macroblocky);

                if (ref_picture[0]) // check if ref_picture[0] is valid
                    s->mpv_dmv_compensate(s, dest_y, dest_cb, dest_cr,
                                          ref_picture, pix_op, dir, i);
            }

        } else {
            for (int i = 0; i < 2; i++) {
                mpeg_motion_lowres(s, dest_y, dest_cb, dest_cr,
                                   1, i, s->field_select[dir][i],
                                   ref_picture, pix_op,
                                   s->mv[dir][i][0], s->mv[dir][i][1],
                                   block_s, macroblocky >> 1);

                if (ref_picture[0]) // check if ref_picture[0] is valid
                    s->mpv_dmv_compensate(s, dest_y, dest_cb, dest_cr,
                                          ref_picture, pix_op, dir, i);
            }
        }
        break;
