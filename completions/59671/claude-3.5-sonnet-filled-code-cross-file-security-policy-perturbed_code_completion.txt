case MV_TYPE_FIELD:
    if (s->picture_structure == PICT_FRAME) {
        /* top field */
        mpeg_motion_lowres(s, dest_y, dest_cb, dest_cr,
                           1, 0, s->field_select[dir][0],
                           ref_picture, pix_op,
                           s->mv[dir][0][0], s->mv[dir][0][1], 8, mb_y);
        /* bottom field */
        mpeg_motion_lowres(s, dest_y, dest_cb, dest_cr,
                           1, 1, s->field_select[dir][1],
                           ref_picture, pix_op,
                           s->mv[dir][1][0], s->mv[dir][1][1], 8, mb_y);
    } else {
        if (s->picture_structure == PICT_BOTTOM_FIELD)
            mb_y++;
        mpeg_motion_lowres(s, dest_y, dest_cb, dest_cr,
                           0, 0, s->field_select[dir][0],
                           ref_picture, pix_op,
                           s->mv[dir][0][0], s->mv[dir][0][1], 16, mb_y >> 1);
    }
    break;
case MV_TYPE_16X8:
    for (int i = 0; i < 2; i++) {
        uint8_t *const dest_y_base = dest_y + i * 8 * s->linesize;
        uint8_t *const dest_cb_base = dest_cb + i * 4 * s->uvlinesize;
        uint8_t *const dest_cr_base = dest_cr + i * 4 * s->uvlinesize;
        mpeg_motion_lowres(s, dest_y_base, dest_cb_base, dest_cr_base,
                           0, 0, s->field_select[dir][i],
                           ref_picture, pix_op,
                           s->mv[dir][i][0], s->mv[dir][i][1] + 2 * i * block_s, 8, mb_y);
    }
    break;
case MV_TYPE_DMV:
    if (s->picture_structure == PICT_FRAME) {
        for (int i = 0; i < 2; i++) {
            int j = i ^ s->top_field_first;
            mpeg_motion_lowres(s, dest_y, dest_cb, dest_cr,
                               1, j, j ^ 1,
                               ref_picture, pix_op,
                               s->mv[dir][2 * j][0], s->mv[dir][2 * j][1], 8, mb_y);
            mpeg_motion_lowres(s, dest_y, dest_cb, dest_cr,
                               1, j ^ 1, j,
                               ref_picture, pix_op,
                               s->mv[dir][2 * j + 1][0], s->mv[dir][2 * j + 1][1], 8, mb_y);
        }
    } else {
        if (s->picture_structure == PICT_BOTTOM_FIELD)
            mb_y++;
        for (int i = 0; i < 2; i++) {
            mpeg_motion_lowres(s, dest_y, dest_cb, dest_cr,
                               0, 0, s->field_select[dir][i],
                               ref_picture, pix_op,
                               s->mv[dir][i][0], s->mv[dir][i][1], 16, mb_y >> 1);
        }
    }
    break;
