case MV_TYPE_FIELD:
    if (s->picture_structure == PICT_FRAME) {
        /* top field */
        mpeg_motion_lowres(s, dest_y, dest_cb, dest_cr,
                           1, 0, s->field_select[dir][0],
                           ref_picture, pix_op,
                           s->mv[dir][0][0], s->mv[dir][0][1],
                           block_s, macroblocky);
        /* bottom field */
        mpeg_motion_lowres(s, dest_y + s->linesize,
                           dest_cb + s->uvlinesize,
                           dest_cr + s->uvlinesize,
                           1, 1, s->field_select[dir][1],
                           ref_picture, pix_op,
                           s->mv[dir][1][0], s->mv[dir][1][1],
                           block_s, macroblocky);
    } else {
        mpeg_motion_lowres(s, dest_y, dest_cb, dest_cr,
                           0, 0, s->field_select[dir][0],
                           ref_picture, pix_op,
                           s->mv[dir][0][0], s->mv[dir][0][1],
                           2 * block_s, macroblocky);
    }
    break;
case MV_TYPE_16X8:
    for (int i = 0; i < 2; i++) {
        uint8_t *const dest_y_base = dest_y + i * 8 * s->linesize;
        mpeg_motion_lowres(s, dest_y_base, dest_cb + i * 4 * s->uvlinesize,
                           dest_cr + i * 4 * s->uvlinesize,
                           0, 0, s->field_select[dir][i],
                           ref_picture, pix_op,
                           s->mv[dir][i][0], s->mv[dir][i][1],
                           block_s, macroblocky + i);
    }
    break;
case MV_TYPE_DMV:
    {
        int dmx, dmy;
        mx = s->mv[dir][0][0];
        my = s->mv[dir][0][1];
        dmx = s->mv[dir][1][0];
        dmy = s->mv[dir][1][1];
        mpeg_motion_lowres(s, dest_y, dest_cb, dest_cr,
                           0, 0, s->picture_structure == PICT_BOTTOM_FIELD,
                           ref_picture, pix_op, mx, my, block_s, macroblocky);
        mpeg_motion_lowres(s, dest_y, dest_cb, dest_cr,
                           0, 0, s->picture_structure == PICT_TOP_FIELD,
                           ref_picture, pix_op,
                           dmx, dmy, block_s, macroblocky);
        mx = (mx * s->field_select[dir][0] + dmx * (1 - s->field_select[dir][0])) / 2;
        my = (my * s->field_select[dir][0] + dmy * (1 - s->field_select[dir][0])) / 2;
        mpeg_motion_lowres(s, dest_y, dest_cb, dest_cr,
                           0, 0, 0, ref_picture, pix_op,
                           mx, my, block_s, macroblocky);
    }
    break;
