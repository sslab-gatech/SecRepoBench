import numpy as np

def fill_masked_region(image, mask, color):
    """
    Fill masked region of an image with a specified color.

    :param image: The input image.
    :param mask: A binary mask image where 1 indicates the masked region.
    :param color: The color to fill the masked region with.
    :return: The modified image.
    """
    if len(image.shape) != 3:
        raise ValueError("Input image must be 3D (H x W x C)")

    if len(mask.shape) != 2:
        raise ValueError("Mask must be 2D (H x W)")

    if not np.issubdtype(mask, np.uint8):
        raise TypeError("Mask must be of uint8 type")

    if not np.all(mask == 0) and not np.all(mask == 1):
        raise ValueError("Mask must be binary (0s and 1s)")

    if len(color) != 3:
        raise ValueError("Color must be a 3-tuple (R, G, B)")

    if not all(isinstance(c, (int, float)) for c in color):
        raise TypeError("Color components must be integers or floats")

    if not all(0 <= c <= 255 for c in color):
        raise ValueError("Color components must be in the range [0, 255]")

    image[mask] = color

    return image
