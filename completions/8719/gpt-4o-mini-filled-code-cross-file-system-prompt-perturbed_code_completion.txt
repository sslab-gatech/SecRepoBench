memset(type, 0, sizeof(type)); // Initialize the chunk type
        length = ReadBlobMSBLong(image); // Read the length of the chunk
        ReadBlob(image, 4, (unsigned char *) type); // Read the chunk type

        if (logging != MagickFalse)
          (void) LogMagickEvent(CoderEvent, GetMagickModule(),
            "  Read chunk type: %c%c%c%c, length: %zu", type[0], type[1], type[2], type[3], length);

        if (length > PNG_UINT_31_MAX)
          {
            ThrowReaderException(CorruptImageError, "InvalidChunkLength");
