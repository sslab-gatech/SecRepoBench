{
  /*
    Read EXIF properties from the current TIFF directory, unless explicitly
    disabled through the "tiff:exif-properties" option.
  */
  const char
    *exif_option;

  exif_option=GetImageOption(image_info,"tiff:exif-properties");
  if ((exif_option == (const char *) NULL) ||
      (IsStringFalse(exif_option) == MagickFalse))
    (void) TIFFGetEXIFProperties(tiff,image,exception);
}
