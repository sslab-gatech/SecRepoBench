import Image

# Open the image file
image = Image.open('input.jpg')

# Create a mask image with the same size as the input image
mask = Image.new('L', image.size)

# Define the region to be masked (x, y, width, height)
x, y, width, height = 100, 100, 200, 200

# Fill the mask with black (0) for the region to be masked and white (255) for the rest
for i in range(image.size[1]):
    for j in range(image.size[0]):
        if (j >= x and j < x + width and i >= y and i < y + height):
            mask.putpixel((j, i), 0)
        else:
            mask.putpixel((j, i), 255)

# Paste the mask onto the input image
image.paste(mask, mask)

# Save the resulting image
image.save('output.jpg')
