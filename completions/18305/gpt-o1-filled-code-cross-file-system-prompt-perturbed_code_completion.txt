/*
    Retrieve EXIF properties unless explicitly disabled by the "tiff:exif-properties" option.
  */
  option=GetImageOption(image_info,"tiff:exif-properties");
  if ((option == (const char *) NULL) || (IsStringFalse(option) == MagickFalse))
    (void) TIFFGetEXIFProperties(tiff,image,exception);

  /*
    Fetch the resolution properties (X/Y) from the TIFF tags.
  */
  if (TIFFGetFieldDefaulted(tiff,TIFFTAG_XRESOLUTION,&hor_res) == 1)
    image->resolution.x=(double) hor_res;
  if (TIFFGetFieldDefaulted(tiff,TIFFTAG_YRESOLUTION,&y_resolution) == 1)
    image->resolution.y=(double) y_resolution;
