              case 'W':
              {

                    image->chromaticity.white_point.y=geometry_info.sigma;
                    if ((flags & SigmaValue) == 0)
                      image->chromaticity.white_point.y=

                    break;
                  }
                (void) SetImageProperty(image,keyword,options,exception);
                break;
              }
              case 'x':
              case 'X':
              {
                if (LocaleCompare(keyword,"x-resolution") == 0)
                  {
                    image->resolution.x=StringToDouble(options,(char **) NULL);
                    break;
                  }
                (void) SetImageProperty(image,keyword,options,exception);
                break;

              case 'y':
              case 'Y':
              {
                if (LocaleCompare(keyword,"y-resolution") == 0)
                  {
                    image->resolution.y=StringToDouble(options,(char **) NULL);
                    break;
                  }
                (void) SetImageProperty(image,keyword,options,exception);
                break;
              }
              default:
                (void) SetImageProperty(image,keyword,options,exception);
                break;
            }
            (void) memset(keyword,0,sizeof(keyword));
            (void) memset(options,0,sizeof(options));
          }

          {
            if (c == EOF)
              break;
            c=ReadBlobByte(image);
          }
