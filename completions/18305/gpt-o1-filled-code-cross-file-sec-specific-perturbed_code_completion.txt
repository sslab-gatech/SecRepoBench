{
    const char *option = GetImageOption(image_info,"tiff:exif-properties");
    /*
      Exif properties are enabled by default unless explicitly disabled.
      If disabled, remove any previously read EXIF profile from the image.
    */
    if ((option != (const char *) NULL) && (IsStringFalse(option) != MagickFalse))
      (void) DeleteImageProfile(image,"exif");

    /*
      Fetch the X and Y resolution tags and store them in the image object
      if they are available.
    */
    {
      float x_resolution=0.0f, y_resolution=0.0f;
      if (TIFFGetFieldDefaulted(tiff,TIFFTAG_XRESOLUTION,&x_resolution) == 1)
        image->resolution.x = (double) x_resolution;
      if (TIFFGetFieldDefaulted(tiff,TIFFTAG_YRESOLUTION,&y_resolution) == 1)
        image->resolution.y = (double) y_resolution;
    }
  }
