  /*
    Read EXIF properties unless explicitly disabled through the
    "tiff:exif-properties" option
  */
  const char *option = GetImageOption(image_info, "tiff:exif-properties");
  if ((option == (const char *) NULL) || (IsStringFalse(option) == MagickFalse))

}
