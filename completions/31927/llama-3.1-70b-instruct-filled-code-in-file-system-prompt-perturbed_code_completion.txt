for (i=0; i<ptr->sampleCount; i++) {
    u32 len = ptr->sizes[i];
    if (len <= 0xF) continue;
    //switch to 8-bit table
    else if (len <= 0xFF) {
        ptr->sampleSize = 8;
    }
    //switch to 16-bit table
    else if (len <= 0xFFFF) {
        ptr->sampleSize = 16;
    }
    //switch to 32-bit table
    else {
        ptr->sampleSize = 32;
    }
    break;
}
