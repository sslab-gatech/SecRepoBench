if (ptr->sampleSize == 4) {
    while (index < ptr->sampleCount) {
        u8 val = gf_bs_read_u8(bs);
        ptr->sizes[index] = (val >> 4) & 0x0F;
        if (ptr->sizes[index] > ptr->max_size)
            ptr->max_size = ptr->sizes[index];
        ptr->total_size += ptr->sizes[index];
        ptr->total_samples++;
        index++;
        if (index < ptr->sampleCount) {
            ptr->sizes[index] = val & 0x0F;
            if (ptr->sizes[index] > ptr->max_size)
                ptr->max_size = ptr->sizes[index];
            ptr->total_size += ptr->sizes[index];
            ptr->total_samples++;
            index++;
        }
    }
} else {
    for (; index < ptr->sampleCount; index++) {
        u32 sizeVal = 0;
        if (ptr->sampleSize == 8)
            sizeVal = gf_bs_read_u8(bs);
        else if (ptr->sampleSize == 16)
            sizeVal = gf_bs_read_u16(bs);
        else if (ptr->sampleSize == 32)
            sizeVal = gf_bs_read_u32(bs);
        else
            return GF_ISOM_INVALID_FILE;
        ptr->sizes[index] = sizeVal;
        if (sizeVal > ptr->max_size)
            ptr->max_size = sizeVal;
        ptr->total_size += sizeVal;
        ptr->total_samples++;
    }
}
