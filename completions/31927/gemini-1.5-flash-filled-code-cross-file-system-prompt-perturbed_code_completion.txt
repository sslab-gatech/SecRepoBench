u32 sampleSize = ptr->sampleSize;
			u32 count = 0;
			if (sampleSize == 4) {
				u8 val;
				for (; index < ptr->sampleCount; index += 2) {
					val = gf_bs_read_u8(bs);
					ptr->sizes[index] = (val >> 4) & 0xF;
					if (index + 1 < ptr->sampleCount) {
						ptr->sizes[index + 1] = val & 0xF;
					}
					count++;
				}
				if (ptr->sampleCount % 2) {
					gf_bs_read_int(bs, 4);
				}
			} else {
				for (; index < ptr->sampleCount; index++) {
					ptr->sizes[index] = gf_bs_read_u32(bs);
					count++;
				}
			}
			for (u32 i = 0; i < count; i++) {
				if (ptr->max_size < ptr->sizes[i])
					ptr->max_size = ptr->sizes[i];
				ptr->total_size += ptr->sizes[i];
				ptr->total_samples++;
			}
