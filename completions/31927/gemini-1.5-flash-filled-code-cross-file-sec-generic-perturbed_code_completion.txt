u32 sampleSize = ptr->sampleSize;
			if (sampleSize == 4) {
				for (; index < ptr->sampleCount; index += 2) {
					u32 val = gf_bs_read_u8(bs);
					ptr->sizes[index] = (val >> 4) & 0xF;
					if (index + 1 < ptr->sampleCount) {
						ptr->sizes[index + 1] = val & 0xF;
					}
					if (ptr->max_size < ptr->sizes[index]) ptr->max_size = ptr->sizes[index];
					ptr->total_size += ptr->sizes[index];
					ptr->total_samples++;
					if (index + 1 < ptr->sampleCount) {
						if (ptr->max_size < ptr->sizes[index + 1]) ptr->max_size = ptr->sizes[index + 1];
						ptr->total_size += ptr->sizes[index + 1];
						ptr->total_samples++;
					}
				}
			} else {
				for (; index < ptr->sampleCount; index++) {
					ptr->sizes[index] = gf_bs_read_u32(bs);
					if (ptr->max_size < ptr->sizes[index]) ptr->max_size = ptr->sizes[index];
					ptr->total_size += ptr->sizes[index];
					ptr->total_samples++;
				}
