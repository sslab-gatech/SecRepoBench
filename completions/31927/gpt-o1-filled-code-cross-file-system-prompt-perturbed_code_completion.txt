if (ptr->sampleSize == 4) {
    while (index < ptr->sampleCount) {
        u8 val = gf_bs_read_u8(bs);

        /* Upper nibble */
        ptr->sizes[index] = (val >> 4) & 0x0F;
        if (ptr->max_size < ptr->sizes[index]) ptr->max_size = ptr->sizes[index];
        ptr->total_size += ptr->sizes[index];
        ptr->total_samples++;
        index++;

        /* Lower nibble (if more samples remain) */
        if (index < ptr->sampleCount) {
            ptr->sizes[index] = val & 0x0F;
            if (ptr->max_size < ptr->sizes[index]) ptr->max_size = ptr->sizes[index];
            ptr->total_size += ptr->sizes[index];
            ptr->total_samples++;
            index++;
        }
    }
} else if (ptr->sampleSize == 8) {
    while (index < ptr->sampleCount) {
        u8 val = gf_bs_read_u8(bs);
        ptr->sizes[index] = val;
        if (ptr->max_size < ptr->sizes[index]) ptr->max_size = ptr->sizes[index];
        ptr->total_size += ptr->sizes[index];
        ptr->total_samples++;
        index++;
    }
} else if (ptr->sampleSize == 16) {
    while (index < ptr->sampleCount) {
        u16 val = gf_bs_read_u16(bs);
        ptr->sizes[index] = val;
        if (ptr->max_size < ptr->sizes[index]) ptr->max_size = ptr->sizes[index];
        ptr->total_size += ptr->sizes[index];
        ptr->total_samples++;
        index++;
    }
}
