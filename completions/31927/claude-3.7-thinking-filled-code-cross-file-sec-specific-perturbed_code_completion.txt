if (ptr->sampleSize == 4) {
    // For 4-bit sample sizes, we read two samples per byte
    for (; index < ptr->sampleCount; index += 2) {
        // Read one byte containing two 4-bit values
        u8 val = gf_bs_read_u8(bs);
        // First sample (higher 4 bits)
        ptr->sizes[index] = (val >> 4) & 0x0F;
        if (ptr->max_size < ptr->sizes[index])
            ptr->max_size = ptr->sizes[index];
        ptr->total_size += ptr->sizes[index];
        ptr->total_samples++;
        
        // Second sample (lower 4 bits) - only if we haven't reached the end
        if (index + 1 < ptr->sampleCount) {
            ptr->sizes[index + 1] = val & 0x0F;
            if (ptr->max_size < ptr->sizes[index + 1])
                ptr->max_size = ptr->sizes[index + 1];
            ptr->total_size += ptr->sizes[index + 1];
            ptr->total_samples++;
        }
    }
} else if (ptr->sampleSize == 8) {
    // For 8-bit sample sizes
    for (; index < ptr->sampleCount; index++) {
        ptr->sizes[index] = gf_bs_read_u8(bs);
        if (ptr->max_size < ptr->sizes[index])
            ptr->max_size = ptr->sizes[index];
        ptr->total_size += ptr->sizes[index];
        ptr->total_samples++;
    }
} else if (ptr->sampleSize == 16) {
    // For 16-bit sample sizes
    for (; index < ptr->sampleCount; index++) {
        ptr->sizes[index] = gf_bs_read_u16(bs);
        if (ptr->max_size < ptr->sizes[index])
            ptr->max_size = ptr->sizes[index];
        ptr->total_size += ptr->sizes[index];
        ptr->total_samples++;
    }
}
