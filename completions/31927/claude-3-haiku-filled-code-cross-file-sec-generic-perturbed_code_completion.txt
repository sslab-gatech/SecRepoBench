for (index = 0; index < ptr->sampleCount; ) {
    if (ptr->sampleSize == 4) {
        // Read two 4-bit sample sizes per byte
        u8 sample_size_byte = gf_bs_read_u8(bs);
        ptr->sizes[index++] = (sample_size_byte >> 4) & 0x0F;
        if (index < ptr->sampleCount) {
            ptr->sizes[index++] = sample_size_byte & 0x0F;
        }
    } else {
        // Read sample sizes of the specified size (8-bit, 16-bit, or 32-bit)
        ptr->sizes[index++] = gf_bs_read_int(bs, ptr->sampleSize);
    }

    // Update the maximum size, total size, and total sample count
    if (ptr->max_size < ptr->sizes[index - 1])
        ptr->max_size = ptr->sizes[index - 1];
    ptr->total_size += ptr->sizes[index - 1];
    ptr->total_samples++;
}
