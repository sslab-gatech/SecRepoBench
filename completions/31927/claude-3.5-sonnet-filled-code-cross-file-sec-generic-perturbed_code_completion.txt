if (ptr->sampleSize == 4) {
    u8 tmp;
    for (index = 0; index < ptr->sampleCount; index += 2) {
        tmp = gf_bs_read_u8(bs);
        ptr->sizes[index] = (tmp >> 4) & 0x0F;
        if (index + 1 < ptr->sampleCount) {
            ptr->sizes[index + 1] = tmp & 0x0F;
        }
        ptr->max_size = GF_MAX(ptr->max_size, ptr->sizes[index]);
        ptr->total_size += ptr->sizes[index];
        ptr->total_samples++;
        if (index + 1 < ptr->sampleCount) {
            ptr->max_size = GF_MAX(ptr->max_size, ptr->sizes[index + 1]);
            ptr->total_size += ptr->sizes[index + 1];
            ptr->total_samples++;
        }
    }
} else {
    for (index = 0; index < ptr->sampleCount; index++) {
        switch (ptr->sampleSize) {
        case 8:
            ptr->sizes[index] = gf_bs_read_u8(bs);
            break;
        case 16:
            ptr->sizes[index] = gf_bs_read_u16(bs);
            break;
        default:
            return GF_ISOM_INVALID_FILE;
        }
        ptr->max_size = GF_MAX(ptr->max_size, ptr->sizes[index]);
        ptr->total_size += ptr->sizes[index];
        ptr->total_samples++;
    }
}
