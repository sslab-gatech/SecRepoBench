if (ptr->sampleSize == 4) {
    u8 val;
    while (index < ptr->sampleCount) {
        val = gf_bs_read_u8(bs);
        /* high nibble for first sample */
        ptr->sizes[index] = val >> 4;
        if (ptr->max_size < ptr->sizes[index])
            ptr->max_size = ptr->sizes[index];
        ptr->total_size += ptr->sizes[index];
        ptr->total_samples++;
        index++;
        /* low nibble for second sample if exists */
        if (index < ptr->sampleCount) {
            ptr->sizes[index] = val & 0x0F;
            if (ptr->max_size < ptr->sizes[index])
                ptr->max_size = ptr->sizes[index];
            ptr->total_size += ptr->sizes[index];
            ptr->total_samples++;
            index++;
        }
    }
} else if (ptr->sampleSize == 8) {
    for (; index < ptr->sampleCount; index++) {
        ptr->sizes[index] = gf_bs_read_u8(bs);
        if (ptr->max_size < ptr->sizes[index])
            ptr->max_size = ptr->sizes[index];
        ptr->total_size += ptr->sizes[index];
        ptr->total_samples++;
    }
} else if (ptr->sampleSize == 16) {
    for (; index < ptr->sampleCount; index++) {
        ptr->sizes[index] = gf_bs_read_int(bs, 16);
        if (ptr->max_size < ptr->sizes[index])
            ptr->max_size = ptr->sizes[index];
        ptr->total_size += ptr->sizes[index];
        ptr->total_samples++;
    }
}
