for (index = 0; index < ptr->sampleCount; ) {
    if (ptr->sampleSize == 4) {
        u8 byte = gf_bs_read_u8(bs);
        if (index < ptr->sampleCount) {
            ptr->sizes[index] = (byte >> 4) & 0x0F;
            if (ptr->max_size < ptr->sizes[index])
                ptr->max_size = ptr->sizes[index];
            ptr->total_size += ptr->sizes[index];
            ptr->total_samples++;
            index++;
        }
        if (index < ptr->sampleCount) {
            ptr->sizes[index] = byte & 0x0F;
            if (ptr->max_size < ptr->sizes[index])
                ptr->max_size = ptr->sizes[index];
            ptr->total_size += ptr->sizes[index];
            ptr->total_samples++;
            index++;
        }
    } else {
        ptr->sizes[index] = gf_bs_read_int(bs, ptr->sampleSize);
        if (ptr->max_size < ptr->sizes[index])
            ptr->max_size = ptr->sizes[index];
        ptr->total_size += ptr->sizes[index];
        ptr->total_samples++;
        index++;
    }
}
