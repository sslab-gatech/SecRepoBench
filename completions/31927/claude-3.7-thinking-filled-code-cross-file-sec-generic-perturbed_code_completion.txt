if (ptr->sampleSize == 4) {
    u8 val;
    // Process two samples at a time for 4-bit sizes
    for (index = 0; index < ptr->sampleCount; index += 2) {
        val = gf_bs_read_u8(bs);
        ptr->sizes[index] = val >> 4;
        if (ptr->max_size < ptr->sizes[index])
            ptr->max_size = ptr->sizes[index];
        ptr->total_size += ptr->sizes[index];
        ptr->total_samples++;
        
        // Handle the second sample if we're not at the end (for odd counts)
        if (index + 1 < ptr->sampleCount) {
            ptr->sizes[index+1] = val & 0x0F;
            if (ptr->max_size < ptr->sizes[index+1])
                ptr->max_size = ptr->sizes[index+1];
            ptr->total_size += ptr->sizes[index+1];
            ptr->total_samples++;
        }
    }
} else if (ptr->sampleSize == 8) {
    // Process one byte at a time for 8-bit sizes
    for (index = 0; index < ptr->sampleCount; index++) {
        ptr->sizes[index] = gf_bs_read_u8(bs);
        if (ptr->max_size < ptr->sizes[index])
            ptr->max_size = ptr->sizes[index];
        ptr->total_size += ptr->sizes[index];
        ptr->total_samples++;
    }
} else if (ptr->sampleSize == 16) {
    // Process two bytes at a time for 16-bit sizes
    for (index = 0; index < ptr->sampleCount; index++) {
        ptr->sizes[index] = gf_bs_read_u16(bs);
        if (ptr->max_size < ptr->sizes[index])
            ptr->max_size = ptr->sizes[index];
        ptr->total_size += ptr->sizes[index];
        ptr->total_samples++;
    }
}
