if(!quic_pp_cipher_init(&ciphers.pp_cipher, GCRY_MD_SHA256,
			   client_secret, version)) {
    NDPI_LOG_DBG(ndpi_struct, "Error quic_pp_cipher_init\n");
    return NULL;
  }

  if(!quic_hp_cipher_init(&ciphers.hp_cipher, GCRY_MD_SHA256,
			   16, client_secret, version)) {
    NDPI_LOG_DBG(ndpi_struct, "Error quic_hp_cipher_init\n");
    quic_ciphers_reset(&ciphers);
    return NULL;
  }

  pn_offset = 5; /* Skip version + first byte */

  token_length = packet->payload[pn_offset];
  pn_offset++;
  pn_offset += token_length;

  payload_length = ndpi_read_be16(packet->payload + pn_offset);
  pn_offset += 2;

  first_byte = packet->payload[0];

  NDPI_LOG_DBG2(ndpi_struct, "pn_offset %u payload_length %u\n",
               pn_offset, (unsigned int)payload_length);
