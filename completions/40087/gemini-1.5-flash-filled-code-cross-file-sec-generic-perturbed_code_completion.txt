pn_offset = 0;
  token_length = 0;
  if (!quic_hp_cipher_init(&ciphers.hp_cipher, GCRY_MD_SHA256, 16, client_secret, version)) {
    NDPI_LOG_DBG(ndpi_struct, "Error quic_hp_cipher_init\n");
    return NULL;
  }
  token_length = quic_get_token_length(packet->payload, packet->payload_packet_len, version);
  pn_offset += token_length;
  payload_length = packet->payload_packet_len - pn_offset;
  NDPI_LOG_DBG2(ndpi_struct, "pn_offset %d token_length %d payload_length %d\n",
		pn_offset, token_length, payload_length);
