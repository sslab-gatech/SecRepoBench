pn_offset = 1;
  token_length = 0;
  if (!quic_hp_cipher_init(&ciphers.hp_cipher, GCRY_MD_SHA256, 16, client_secret, version)) {
    NDPI_LOG_DBG(ndpi_struct, "Error quic_hp_cipher_init\n");
    return NULL;
  }
  if (!quic_pp_cipher_init(&ciphers.pp_cipher, GCRY_MD_SHA256, 16, client_secret, version)) {
    NDPI_LOG_DBG(ndpi_struct, "Error quic_pp_cipher_init\n");
    quic_ciphers_reset(&ciphers);
    return NULL;
  }
  pn_offset += token_length;
  payload_length = packet->payload_packet_len - pn_offset;
  NDPI_LOG_DBG2(ndpi_struct, "pn_offset %d payload_length %d\n", pn_offset, payload_length);
