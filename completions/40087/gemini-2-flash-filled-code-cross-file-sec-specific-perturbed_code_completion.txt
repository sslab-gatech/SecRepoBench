			     client_secret, version, 1) ||
     !quic_hp_cipher_init(&ciphers.hp_cipher, GCRY_MD_SHA256, 16,
			   client_secret, version)) {
    NDPI_LOG_DBG(ndpi_struct, "Error initializing ciphers\n");
    quic_ciphers_reset(&ciphers);
    return NULL;

  pn_offset += 1 + token_length;

  payload_length = ndpi_read_be16(packet->payload + pn_offset);

  NDPI_LOG_DBG2(ndpi_struct, "pn_offset %u\n", pn_offset);
