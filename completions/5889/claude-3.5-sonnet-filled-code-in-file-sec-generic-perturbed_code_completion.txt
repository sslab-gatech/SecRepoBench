case CirclePrimitive:
{
  double
    radius;

  PointInfo
    center;

  center.x = (primitive_info[0].point.x + primitive_info[1].point.x) / 2.0;
  center.y = (primitive_info[0].point.y + primitive_info[1].point.y) / 2.0;
  radius = hypot(primitive_info[1].point.x - center.x,
                 primitive_info[1].point.y - center.y);
  points_extent = 2.0 * MagickPI * radius;
  break;
}
case EllipsePrimitive:
{
  double
    rx,
    ry;

  rx = fabs(primitive_info[1].point.x - primitive_info[0].point.x);
  ry = fabs(primitive_info[1].point.y - primitive_info[0].point.y);
  points_extent = 2.0 * MagickPI * sqrt((rx * rx + ry * ry) / 2.0);
  break;
}
