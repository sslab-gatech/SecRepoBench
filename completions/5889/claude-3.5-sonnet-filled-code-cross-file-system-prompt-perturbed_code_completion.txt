case CirclePrimitive:
case EllipsePrimitive:
{
  double
    rx,
    ry;

  if (primitive_type == CirclePrimitive)
  {
    if (primitive_info[j].coordinates != 2)
    {
      status=MagickFalse;
      break;
    }
    rx = ry = hypot(primitive_info[j+1].point.x - primitive_info[j].point.x,
                    primitive_info[j+1].point.y - primitive_info[j].point.y);
  }
  else // EllipsePrimitive
  {
    if (primitive_info[j].coordinates != 3)
    {
      status=MagickFalse;
      break;
    }
    rx = primitive_info[j+1].point.x;
    ry = primitive_info[j+1].point.y;
  }
  
  points_extent = 2 * MagickPI * MagickMax(rx, ry) + 1;
  break;
}
