case CirclePrimitive:
      case EllipsePrimitive:
      case ArcPrimitive:
      {
        /*
          Estimate the number of points required for circular or elliptical primitives.
          Calculate the bounding width and height.
          Then compute a radius based on the diagonal.
          Finally, add an approximate number of points needed to render the shape.
        */
        double
          width = (bounds.x2 - bounds.x1),
          height = (bounds.y2 - bounds.y1),
          radius = hypot(width, height) / 2.0;

        /*
          A rough estimate of how many control points might be used to draw large
          circles or ellipses.  The logic here follows a similar approach to the
          RoundRectanglePrimitive logic, adding extra space for Bezier segments.
        */
        points_extent += 2.0 * ceil(MagickPI * radius) + 6.0 * BezierQuantum + 360.0;
        break;
      }
