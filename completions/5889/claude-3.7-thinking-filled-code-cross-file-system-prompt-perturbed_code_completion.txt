case CirclePrimitive:
{
  double
    alpha,
    beta,
    radius;

  if (primitive_info[j].coordinates != 2)
    {
      status=MagickFalse;
      break;
    }
  alpha=primitive_info[j+1].point.x-primitive_info[j].point.x;
  beta=primitive_info[j+1].point.y-primitive_info[j].point.y;
  radius=hypot(alpha,beta);
  points_extent=2.0*MagickPI*radius+6*BezierQuantum+360;
  break;
}
case EllipsePrimitive:
{
  double
    major_axis,
    minor_axis;

  if (primitive_info[j].coordinates != 3)
    {
      status=MagickFalse;
      break;
    }
  major_axis=primitive_info[j+1].point.x;
  minor_axis=primitive_info[j+1].point.y;
  points_extent=2.0*MagickPI*sqrt((major_axis*major_axis+minor_axis*minor_axis)/2.0)+
    6*BezierQuantum+360;
  break;
}
