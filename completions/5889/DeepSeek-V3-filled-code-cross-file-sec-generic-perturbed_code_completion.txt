case CirclePrimitive:
{
    double
        width,
        height,
        radius;

    width = bounds.x2 - bounds.x1;
    height = bounds.y2 - bounds.y1;
    radius = hypot(width, height);
    points_extent = 2 * ceil(MagickPI * radius) + 360;
    break;
}
case EllipsePrimitive:
{
    double
        width,
        height,
        radius;

    width = bounds.x2 - bounds.x1;
    height = bounds.y2 - bounds.y1;
    radius = hypot(width, height);
    points_extent = 2 * ceil(MagickPI * radius) + 360;
    break;
}
