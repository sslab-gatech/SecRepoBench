case ArcPrimitive:
{
  double
    alpha,
    beta,
    radius;

  alpha=bounds.x2-bounds.x1;
  beta=bounds.y2-bounds.y1;
  radius=hypot((double) alpha,(double) beta);
  points_extent*=5;
  points_extent+=2*ceil((double) MagickPI*radius)+6*BezierQuantum+360;
  break;
}
