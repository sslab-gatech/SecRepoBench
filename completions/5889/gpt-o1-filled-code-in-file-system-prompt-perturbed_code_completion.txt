case ArcPrimitive:
case CirclePrimitive:
case EllipsePrimitive:
{
  /*
    Estimate how many points are needed to render arcs, circles and ellipses.
    First compute the width and height of the bounding box, then compute a
    radius.  Finally, estimate the number of segments based on a combination
    of the radius, some allowance for curve complexity (BezierQuantum), and
    some extra margin for small arcs or very large arcs.
  */
  double alphaprimitive, beta, radius;
  alphaprimitive = bounds.x2 - bounds.x1;
  beta = bounds.y2 - bounds.y1;
  radius = hypot(alphaprimitive, beta) / 2.0;
  points_extent += 2.0 * ceil(MagickPI * radius) + 6.0 * BezierQuantum + 360.0;
  break;
}
