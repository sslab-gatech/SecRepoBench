case ArcPrimitive:
      {
        /*
          Estimate additional points needed to render an arc.
        */
        double dx=(bounds.x2 - bounds.x1)/2.0;
        double dy=(bounds.y2 - bounds.y1)/2.0;
        double radius=hypot(dx,dy);
        /*
          Add an approximation based on circumference plus extra for bezier
          handling and any possible transforms.
        */
        points_extent += 2.0*ceil(MagickPI*radius) + 6*BezierQuantum + 360;
        break;
      }
      case CirclePrimitive:
      {
        /*
          Estimate additional points needed to render a circle.
        */
        double dx=(bounds.x2 - bounds.x1)/2.0;
        double dy=(bounds.y2 - bounds.y1)/2.0;
        double radius=hypot(dx,dy);
        points_extent += 2.0*ceil(MagickPI*radius) + 6*BezierQuantum + 360;
        break;
      }
      case EllipsePrimitive:
      {
        /*
          Estimate additional points needed to render an ellipse.
        */
        double rx=(bounds.x2 - bounds.x1)/2.0;
        double ry=(bounds.y2 - bounds.y1)/2.0;
        double perimeter=2.0*MagickPI*sqrt((rx*rx + ry*ry)/2.0);
        points_extent += perimeter + 6*BezierQuantum + 360;
        break;
      }
