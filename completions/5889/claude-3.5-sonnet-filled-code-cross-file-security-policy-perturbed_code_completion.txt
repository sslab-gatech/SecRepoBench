case CirclePrimitive:
case EllipsePrimitive:
{
  double
    rx,
    ry;

  rx = fabs(primitive_info[j+1].point.x - primitive_info[j].point.x);
  ry = fabs(primitive_info[j+1].point.y - primitive_info[j].point.y);
  points_extent = MagickMax(rx, ry) * MagickPI;
  if (points_extent > (double) SSIZE_MAX)
    points_extent = (double) SSIZE_MAX;
  points_extent = MagickMin(points_extent, (double) SSIZE_MAX - i);
  break;
}
