case CirclePrimitive:
{
  double
    radius;

  if (primitive_info[j].coordinates != 2)
    {
      status=MagickFalse;
      break;
    }
  radius=hypot(primitive_info[j+1].point.x-primitive_info[j].point.x,
               primitive_info[j+1].point.y-primitive_info[j].point.y);
  points_extent=ceil(2.0*MagickPI*radius);
  break;
}
case EllipsePrimitive:
{
  double
    rx,
    ry;

  if (primitive_info[j].coordinates != 3)
    {
      status=MagickFalse;
      break;
    }
  rx=primitive_info[j+1].point.x;
  ry=primitive_info[j+1].point.y;
  points_extent=ceil(2.0*MagickPI*sqrt((rx*rx+ry*ry)/2.0));
  break;
}
