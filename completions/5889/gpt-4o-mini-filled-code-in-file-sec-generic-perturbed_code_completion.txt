case CirclePrimitive:
      {
        double
          radius;

        radius = hypot((end.x - start.x), (end.y - start.y)) / 2.0;
        TraceEllipse(primitive_info, start, (PointInfo){radius, radius}, (PointInfo){0.0, 360.0});
        break;
      }
      case EllipsePrimitive:
      {
        double
          radius_x,
          radius_y;

        radius_x = fabs(end.x - start.x) / 2.0;
        radius_y = fabs(end.y - start.y) / 2.0;
        TraceEllipse(primitive_info, start, (PointInfo){radius_x, radius_y}, (PointInfo){0.0, 360.0});
        break;
      }
