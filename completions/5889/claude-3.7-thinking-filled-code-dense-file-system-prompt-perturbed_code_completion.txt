case CirclePrimitive:
{
  if (primitive_info[j].coordinates != 2)
    {
      status=MagickFalse;
      break;
    }
  double
    alpha,
    beta,
    radius;

  alpha=primitive_info[j+1].point.x-primitive_info[j].point.x;
  beta=primitive_info[j+1].point.y-primitive_info[j].point.y;
  radius=hypot((double) alpha,(double) beta);
  points_extent=(double) (2.0*MagickPI*radius+6*BezierQuantum+360);
  break;
}
case EllipsePrimitive:
{
  if (primitive_info[j].coordinates != 3)
    {
      status=MagickFalse;
      break;
    }
  double
    major,
    minor,
    radius;

  major=primitive_info[j+1].point.x;
  minor=primitive_info[j+1].point.y;
  radius=hypot((double) major,(double) minor);
  points_extent=(double) (2.0*MagickPI*radius+6*BezierQuantum+360);
  break;
}
