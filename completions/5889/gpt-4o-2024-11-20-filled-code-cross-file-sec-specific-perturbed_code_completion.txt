case CirclePrimitive:
{
  double radius;

  // Calculate the radius using the bounding box dimensions
  radius = hypot(bounds.x2 - bounds.x1, bounds.y2 - bounds.y1) / 2.0;

  // Estimate the number of points required for rendering the circle
  points_extent += ceil(2 * MagickPI * radius) + 6 * BezierQuantum + 360;
  break;
}
case EllipsePrimitive:
{
  double radius_x, radius_y;

  // Calculate the radii for the ellipse
  radius_x = (bounds.x2 - bounds.x1) / 2.0;
  radius_y = (bounds.y2 - bounds.y1) / 2.0;

  // Estimate the number of points required for rendering the ellipse
  points_extent += ceil(2 * MagickPI * sqrt((radius_x * radius_x + radius_y * radius_y) / 2.0)) + 6 * BezierQuantum + 360;
  break;
}
