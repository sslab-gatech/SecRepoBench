case CirclePrimitive:
{
  double
    alpha,
    beta,
    radius;

  alpha = primitive_info[j+1].point.x - primitive_info[j].point.x;
  beta = primitive_info[j+1].point.y - primitive_info[j].point.y;
  radius = hypot(alpha, beta);
  points_extent = 2 * MagickPI * radius + 6 * BezierQuantum + 360;
  break;
}
case EllipsePrimitive:
{
  double
    alpha,
    beta,
    radius;

  alpha = (primitive_info[j+1].point.x - primitive_info[j].point.x) / 2.0;
  beta = (primitive_info[j+1].point.y - primitive_info[j].point.y) / 2.0;
  radius = hypot(alpha, beta);
  points_extent = 2 * MagickPI * radius + 6 * BezierQuantum + 360;
  break;
}
