case CirclePrimitive:
      {
        double
          radius;

        radius=fabs(end.x-start.x)/2.0;
        points_extent=(size_t) (2.0*MagickPI*radius/BezierQuantum);
        break;
      }
      case EllipsePrimitive:
      {
        double
          radius_x,
          radius_y;

        radius_x=fabs(end.x-start.x)/2.0;
        radius_y=fabs(end.y-start.y)/2.0;
        points_extent=(size_t) (2.0*MagickPI*sqrt(radius_x*radius_y)/BezierQuantum);
        break;
      }
