case CirclePrimitive:
{
  double
    alpha,
    beta,
    radius;

  if (primitive_info[j].coordinates != 2)
    {
      status=MagickFalse;
      break;
    }
  alpha=bounds.x2-bounds.x1;
  beta=bounds.y2-bounds.y1;
  radius=hypot((double) alpha,(double) beta);
  points_extent+=2.0*ceil((double) MagickPI*radius)+360.0;
  break;
}
case EllipsePrimitive:
{
  double
    alpha,
    beta,
    perimeter;
    
  if (primitive_info[j].coordinates != 3)
    {
      status=MagickFalse;
      break;
    }
  alpha=bounds.x2-bounds.x1;
  beta=bounds.y2-bounds.y1;
  perimeter=2.0*MagickPI*sqrt((alpha*alpha+beta*beta)/2.0);
  points_extent+=2.0*ceil(perimeter)+360.0;
  break;
}
default:
{
  points_extent+=primitive_info[j].coordinates;
  break;
}
