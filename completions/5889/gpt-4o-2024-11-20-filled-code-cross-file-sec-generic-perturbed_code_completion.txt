case CirclePrimitive:
{
  double radius;

  // Calculate the radius using the bounding box dimensions
  radius = hypot(bounds.x2 - bounds.x1, bounds.y2 - bounds.y1) / 2.0;

  // Estimate the number of points required for rendering the circle
  points_extent += 2 * ceil(MagickPI * radius) + 6 * BezierQuantum + 360;
  break;
}
case EllipsePrimitive:
{
  double major_axis, minor_axis;

  // Calculate the major and minor axes using the bounding box dimensions
  major_axis = fabs(bounds.x2 - bounds.x1) / 2.0;
  minor_axis = fabs(bounds.y2 - bounds.y1) / 2.0;

  // Estimate the number of points required for rendering the ellipse
  points_extent += 2 * ceil(MagickPI * sqrt((major_axis * major_axis + minor_axis * minor_axis) / 2.0)) + 6 * BezierQuantum + 360;
  break;
}
