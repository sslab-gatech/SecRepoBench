// Estimate the number of points required for circular or elliptical primitives.
case EllipsePrimitive:
{
  double
    alpha,
    beta,
    radius;

  alpha=primitive_info[j].point.x-primitive_info[j+1].point.x;
  beta=primitive_info[j].point.y-primitive_info[j+1].point.y;
  radius=hypot((double) alpha,(double) beta);
  points_extent*=360.0;
  points_extent+=2*ceil((double) MagickPI*radius)+6*BezierQuantum+360;
  break;
}
case CirclePrimitive:
{
  double
    radius;

  radius=hypot(primitive_info[j].point.x-primitive_info[j+1].point.x,
    primitive_info[j].point.y-primitive_info[j+1].point.y);
  points_extent*=360.0;
  points_extent+=2*ceil((double) MagickPI*radius)+6*BezierQuantum+360;
  break;
}
