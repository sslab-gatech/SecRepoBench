case CirclePrimitive:
      {
        double
          alpha,
          beta,
          radius;

        if (primitive_info[j].coordinates != 2)
          {
            status=MagickFalse;
            break;
          }
        alpha=primitive_info[j+1].point.x-primitive_info[j].point.x;
        beta=primitive_info[j+1].point.y-primitive_info[j].point.y;
        radius=hypot(alpha,beta);
        points_extent=ceil(MagickPI*radius);
        break;
      }
      case EllipsePrimitive:
      {
        if (primitive_info[j].coordinates != 3)
          {
            status = MagickFalse;
            break;
          }
        double alpha = primitive_info[j + 1].point.x;
        double beta = primitive_info[j + 1].point.y;
        double radius = hypot(alpha, beta);
        points_extent = ceil(MagickPI * radius);

        break;
      }
