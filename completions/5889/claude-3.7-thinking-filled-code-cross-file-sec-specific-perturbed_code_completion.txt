case CirclePrimitive:
{
  double
    radius;

  if (primitive_info[j].coordinates != 2)
    {
      status=MagickFalse;
      break;
    }
  
  // Calculate radius from the two points (center and edge)
  radius=hypot(primitive_info[j+1].point.x-primitive_info[j].point.x,
              primitive_info[j+1].point.y-primitive_info[j].point.y);
  
  // Estimate points needed for a smooth circle
  points_extent=2.0*ceil((double) MagickPI*radius)+6.0*BezierQuantum+360.0;
  break;
}
case EllipsePrimitive:
{
  double
    major,
    minor;
  
  if (primitive_info[j].coordinates != 3)
    {
      status=MagickFalse;
      break;
    }
  
  // Extract the major and minor axes
  major=primitive_info[j+1].point.x;
  minor=primitive_info[j+1].point.y;
  
  // Estimate points based on approximate ellipse circumference
  points_extent=2.0*ceil((double) MagickPI*sqrt((major*major+minor*minor)/2.0))+
    6.0*BezierQuantum+360.0;
  break;
}
case ArcPrimitive:
{
  double
    radius;
  
  if (primitive_info[j].coordinates != 3)
    {
      status=MagickFalse;
      break;
    }
  
  // For an arc, we need to compute the radius from the points
  radius=hypot(primitive_info[j+1].point.x-primitive_info[j].point.x,
              primitive_info[j+1].point.y-primitive_info[j].point.y);
  
  // Estimate points needed for a smooth arc
  points_extent=2.0*ceil((double) MagickPI*radius)+6.0*BezierQuantum+360.0;
  break;
}
