case CirclePrimitive:

        double width = bounds.x2 - bounds.x1;
        double height = bounds.y2 - bounds.y1;
        double radius = hypot(width, height);
        points_extent = 2 * ceil(MagickPI * radius) + 6 * BezierQuantum + 360;
        break;

        break;
