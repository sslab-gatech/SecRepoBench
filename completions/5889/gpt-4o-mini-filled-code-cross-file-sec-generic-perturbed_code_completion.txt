case EllipsePrimitive:
      {
        double radius_x, radius_y;

        radius_x = fabs(bounds.x2 - bounds.x1) / 2.0;
        radius_y = fabs(bounds.y2 - bounds.y1) / 2.0;
        points_extent = BezierQuantum * (radius_x + radius_y); // Estimate points based on the average of the radii
        break;
      }
      case CirclePrimitive:
      {
        double radius;

        radius = hypot(bounds.x2 - bounds.x1, bounds.y2 - bounds.y1) / 2.0;
        points_extent = BezierQuantum * radius; // Estimate points for a circle
        break;
      }
