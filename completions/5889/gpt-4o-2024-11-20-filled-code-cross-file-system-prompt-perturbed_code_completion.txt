case CirclePrimitive:
{
  double
    alpha,
    beta,
    radius;

  alpha = bounds.x2 - bounds.x1;
  beta = bounds.y2 - bounds.y1;
  radius = hypot(alpha, beta) / 2.0;
  points_extent += ceil(2.0 * MagickPI * radius) + 6 * BezierQuantum + 360;
  break;
}
case EllipsePrimitive:
{
  double
    alpha,
    beta,
    radius;

  alpha = bounds.x2 - bounds.x1;
  beta = bounds.y2 - bounds.y1;
  radius = hypot(alpha, beta);
  points_extent += ceil(2.0 * MagickPI * radius) + 6 * BezierQuantum + 360;
  break;
}
