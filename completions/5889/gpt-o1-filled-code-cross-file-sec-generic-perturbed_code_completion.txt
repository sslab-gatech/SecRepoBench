case ArcPrimitive:
      case EllipsePrimitive:
      case CirclePrimitive:
      {
        /*
          Estimate number of points for circular or elliptical primitives.
          Calculate the width/height of the bounding box, then compute a radius
          and use it to determine the approximate number of points needed.
        */
        double alpha = bounds.x2 - bounds.x1;
        double beta = bounds.y2 - bounds.y1;
        double radius = hypot(alpha, beta) / 2.0;
        points_extent += 2 * ceil((double) MagickPI * radius) +
                         6.0 * BezierQuantum + 360.0;
        break;
      }
