case ArcPrimitive:
case CirclePrimitive:
case EllipsePrimitive:
{
  double
    alphaprimitive,
    beta,
    radius;

  /*
    Calculate width and height of the bounding box.
  */
  alphaprimitive=bounds.x2-bounds.x1;
  beta=bounds.y2-bounds.y1;

  /*
    Compute the radius.
  */
  radius=hypot((double) alphaprimitive,(double) beta);

  /*
    Use the radius to estimate how many points will be needed to render
    these circular or elliptical primitives.
  */
  points_extent+=2*ceil((double) MagickPI*radius)+6*BezierQuantum+360;
  break;
}
