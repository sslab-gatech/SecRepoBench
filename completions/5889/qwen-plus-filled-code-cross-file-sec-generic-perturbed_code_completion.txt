      width,
      height;

    if (primitive_info[j].coordinates != 3)
    {

    height = fabs(primitive_info[j+1].point.y - primitive_info[j].point.y);

    double radius = hypot(width, height) / 2.0;

    points_extent = ceil(2 * MagickPI * radius) + BezierQuantum;
