case 'X':
    case 'x':
      if((packet->line[i].len > 5) && (!strncmp(packet->line[i].ptr, "X-UIDH", 6))) {
	ndpi_set_risk(ndpi_struct, ndpiflow, NDPI_PROTOCOL_HTTP, NDPI_RISK_X_UIDH,
		      "Detected X-UIDH http header");
	return;
      }
      break;
    case 'T':
    case 't':
      if((packet->line[i].len > 10) && (!strncmp(packet->line[i].ptr, "True-Client", 11))) {
	ndpi_set_risk(ndpi_struct, ndpiflow, NDPI_PROTOCOL_HTTP, NDPI_RISK_TRUE_CLIENT,
		      "Detected True-Client-* http header");
	return;
      }
      break;
    case 'C':
    case 'c':
      if((packet->line[i].len > 7) && (!strncmp(packet->line[i].ptr, "Cookie:", 7))) {
          if (strstr((char*)packet->line[i].ptr, "PHPSESSID")) {
              ndpi_set_risk(ndpi_struct, ndpiflow, NDPI_PROTOCOL_HTTP, NDPI_RISK_PHP_SESSID,
                            "Detected PHP Session ID in Cookie");
              return;
          }
      }
      break;
    default:
      break;
