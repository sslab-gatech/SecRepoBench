if (strncmp(packet->line[i].ptr, "X-", 2) == 0) {
  if (ndpi_check_suspicious_header(ndpi_struct, ndpiflow, packet->line[i].ptr, packet->line[i].len)) {
    char msg[256];
    snprintf(msg, sizeof(msg), "Suspicious HTTP header detected: %.*s", packet->line[i].len, packet->line[i].ptr);

    return;
