static void ndpi_check_http_header(struct ndpi_detection_module_struct *ndpi_struct,
				   struct ndpi_flow_struct *flow) {
  u_int32_t i;
  struct ndpi_packet_struct *packet = &ndpi_struct->packet;

  for(i=0; (i < packet->parsed_lines)
	&& (packet->line[i].ptr != NULL)
	&& (packet->line[i].len > 0); i++) {
    if(is_a_suspicious_header(suspicious_http_header_keys_A, packet->line[i])) {
      ndpi_set_risk(ndpi_struct, flow, NDPI_SUSPICIOUS_HEADER, "Found suspicious header 'Arch'");
    } else if(is_a_suspicious_header(suspicious_http_header_keys_C, packet->line[i])) {
      ndpi_set_risk(ndpi_struct, flow, NDPI_SUSPICIOUS_HEADER, "Found suspicious header 'Cores'");
    } else if(is_a_suspicious_header(suspicious_http_header_keys_M, packet->line[i])) {
      ndpi_set_risk(ndpi_struct, flow, NDPI_SUSPICIOUS_HEADER, "Found suspicious header 'Mem'");
    } else if(is_a_suspicious_header(suspicious_http_header_keys_O, packet->line[i])) {
      ndpi_set_risk(ndpi_struct, flow, NDPI_SUSPICIOUS_HEADER, "Found suspicious header 'Os' or 'Osname' or 'Osversion'");
    } else if(is_a_suspicious_header(suspicious_http_header_keys_R, packet->line[i])) {
      ndpi_set_risk(ndpi_struct, flow, NDPI_SUSPICIOUS_HEADER, "Found suspicious header 'Root'");
    } else if(is_a_suspicious_header(suspicious_http_header_keys_S, packet->line[i])) {
      ndpi_set_risk(ndpi_struct, flow, NDPI_SUSPICIOUS_HEADER, "Found suspicious header 'S'");
    } else if(is_a_suspicious_header(suspicious_http_header_keys_T, packet->line[i])) {
      ndpi_set_risk(ndpi_struct, flow, NDPI_SUSPICIOUS_HEADER, "Found suspicious header 'TLS_version'");
    } else if(is_a_suspicious_header(suspicious_http_header_keys_U, packet->line[i])) {
      ndpi_set_risk(ndpi_struct, flow, NDPI_SUSPICIOUS_HEADER, "Found suspicious header 'Uuid'");
    } else if(is_a_suspicious_header(suspicious_http_header_keys_X, packet->line[i])) {
      ndpi_set_risk(ndpi_struct, flow, NDPI_SUSPICIOUS_HEADER, "Found suspicious header 'X-Hire-Me'");
    }
  }
}
