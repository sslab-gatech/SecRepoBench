case 'X':
      case 'x':
	if(ndpi_match_string(packet->line[i].ptr, "X-ATT-DeviceId", packet->line[i].len)) {
	  ndpi_set_detected_protocol(ndpiflow, NDPI_PROTOCOL_X_ATT_DEVICEID, NDPI_PROTOCOL_X_ATT_DEVICEID, NDPI_CONFIDENCE_CERTAIN, ndpi_struct->detected_protocol_stack);
	  NDPI_LOG_INFO(ndpi_struct, "HTTP", "Detected suspicious header: X-ATT-DeviceId");
	  snprintf(ndpiflow->risk.risk_string, sizeof(ndpiflow->risk.risk_string),
		   "Detected suspicious header: X-ATT-DeviceId");
	  ndpiflow->risk.app_risk = 5;
	  return;
	}
	break;
      case 'U':
      case 'u':
	if(ndpi_match_string(packet->line[i].ptr, "User-Agent", packet->line[i].len)) {
	  if (ndpi_check_ua_header(ndpi_struct, ndpiflow, packet->line[i].ptr, packet->line[i].len))
	    return;
	}
	break;
      case 'C':
      case 'c':
	if(ndpi_match_string(packet->line[i].ptr, "Cookie", packet->line[i].len)) {
