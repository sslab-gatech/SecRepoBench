case 'X':
      // Check for headers like "X-Forwarded-For" or "X-Real-IP"
      if(packet->line[i].len >= 15 &&
         (!strncasecmp(packet->line[i].ptr, "X-Forwarded-For", 15) ||
          (packet->line[i].len >= 9 && !strncasecmp(packet->line[i].ptr, "X-Real-IP", 9)))) {
        snprintf(ndpiflow->risk_str, sizeof(ndpiflow->risk_str),
                 "Suspicious HTTP header detected: %.*s", packet->line[i].len, packet->line[i].ptr);
        ndpiflow->risk = 1;
        return;
      }
      break;
    case 'P':
      // Check for headers like "Proxy-Connection"
      if(packet->line[i].len >= 16 &&
         !strncasecmp(packet->line[i].ptr, "Proxy-Connection", 16)) {
        snprintf(ndpiflow->risk_str, sizeof(ndpiflow->risk_str),
                 "Suspicious HTTP header detected: %.*s", packet->line[i].len, packet->line[i].ptr);
        ndpiflow->risk = 1;
        return;
      }
      break;
    case 'F':
      // Check for headers like "Forwarded"
      if(packet->line[i].len >= 9 &&
         !strncasecmp(packet->line[i].ptr, "Forwarded", 9)) {
        snprintf(ndpiflow->risk_str, sizeof(ndpiflow->risk_str),
                 "Suspicious HTTP header detected: %.*s", packet->line[i].len, packet->line[i].ptr);
        ndpiflow->risk = 1;
        return;
      }
      break;
    default:
      // For other header starting characters, no specific check.
      break;
