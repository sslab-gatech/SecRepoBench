      if((packet->line[i].len >= 15)

	char msg[256];
	snprintf(msg, sizeof(msg), "Suspicious HTTP header found: %.*s",
		 packet->line[i].len,
		 packet->line[i].ptr);
	/* This function sets a risk flag and associates a message with the flow */
	ndpi_set_risk(ndpiflow, NDPI_LOW_RISK, msg);
	return;
      }
      break;
    }

      /* Check for StrangeHeader (example suspicious header) */
