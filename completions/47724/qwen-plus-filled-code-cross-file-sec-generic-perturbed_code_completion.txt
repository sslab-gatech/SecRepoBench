case 'X':
case 'x': {
  const char **suspicious_headers = ndpi_struct->SuspiciousHTTPHeaderKeys;
  while (*suspicious_headers != NULL) {
    if (packet->line[i].len >= strlen(*suspicious_headers) && 
        strncasecmp(packet->line[i].ptr, *suspicious_headers, strlen(*suspicious_headers)) == 0) {
      char msg[256];
      snprintf(msg, sizeof(msg), "Suspicious HTTP Header Detected: %s", *suspicious_headers);
      ndpi_set_risk(ndpi_struct, ndpiflow, NDPI_HTTP_SUSPICIOUS_HEADER, msg);
      return;
    }
    suspicious_headers++;
  }
  break;
}
