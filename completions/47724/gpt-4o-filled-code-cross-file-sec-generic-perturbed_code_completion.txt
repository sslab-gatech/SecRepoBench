case 'U': // Check for User-Agent
      if(packet->line[i].len >= 10 && strncasecmp(packet->line[i].ptr, "User-Agent", 10) == 0) {

          snprintf(ndpiflow->risk_message, sizeof(ndpiflow->risk_message), "Suspicious User-Agent: %.*s", packet->line[i].len, packet->line[i].ptr);

        }
      }
