case 'A':
    case 'a':
      if(packet->line[i].len >= 14 && 
         strncasecmp((const char *)packet->line[i].ptr, "Authorization:", 14) == 0) {
        char msg[256];
        
        snprintf(msg, sizeof(msg), "Found suspicious header: Authorization");
        ndpi_set_risk(ndpi_struct, ndpiflow, NDPI_SUSPICIOUS_HEADER, msg);
        return;
      }
      break;
      
    case 'C':
    case 'c':
      if(packet->line[i].len >= 13 && 
         strncasecmp((const char *)packet->line[i].ptr, "Content-Type:", 13) == 0) {
        char msg[256];
        
        snprintf(msg, sizeof(msg), "Found suspicious header: Content-Type");
        ndpi_set_risk(ndpi_struct, ndpiflow, NDPI_SUSPICIOUS_HEADER, msg);
        return;
      }
      break;
      
    case 'U':
    case 'u':
      if(packet->line[i].len >= 11 && 
         strncasecmp((const char *)packet->line[i].ptr, "User-Agent:", 11) == 0) {
        char msg[256];
        
        snprintf(msg, sizeof(msg), "Found suspicious header: User-Agent");
        ndpi_set_risk(ndpi_struct, ndpiflow, NDPI_SUSPICIOUS_HEADER, msg);
        return;
      }
      break;
      
    case 'X':
    case 'x':
      if((packet->line[i].len >= 16 && 
          strncasecmp((const char *)packet->line[i].ptr, "X-Forwarded-For:", 16) == 0) ||
         (packet->line[i].len >= 10 && 
          strncasecmp((const char *)packet->line[i].ptr, "X-Real-IP:", 10) == 0)) {
        char msg[256];
        
        snprintf(msg, sizeof(msg), "Found suspicious header: X-header");
        ndpi_set_risk(ndpi_struct, ndpiflow, NDPI_SUSPICIOUS_HEADER, msg);
        return;
      }
      break;

    default:
      break;
