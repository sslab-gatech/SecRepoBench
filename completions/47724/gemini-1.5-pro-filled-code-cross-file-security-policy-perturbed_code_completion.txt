case 'c':
    case 'C':

          ndpi_match_string("Content-Type", packet->line[i].ptr, packet->line[i].len) == 0 ||
          ndpi_match_string("Connection", packet->line[i].ptr, packet->line[i].len) == 0) {

        char http_suspicious_header[256];

        snprintf(http_suspicious_header, sizeof(http_suspicious_header) - 1,
                 "Suspicious HTTP header (Content-Encoding, Content-Type or Connection)");
        http_suspicious_header[sizeof(http_suspicious_header) - 1] = '\0';

        ndpi_flow_add_risk(ndpi_struct, ndpiflow, NDPI_RISK_HTTP_SUSPICIOUS_HEADER, http_suspicious_header);
        return;
      }
      break;
