case 'X':
    case 'x': {
        const char *hdr = packet->line[i].ptr;
        u_int hdr_len = packet->line[i].len;
        const char *colon = memchr(hdr, ':', hdr_len);
        
        if(colon) {
            size_t key_len = colon - hdr;
            const char *suspicious_hdr = "X-Forwarded-For";
            size_t check_len = strlen(suspicious_hdr);
            
            if(key_len >= check_len && strncasecmp(hdr, suspicious_hdr, check_len) == 0) {
                char risk_msg[64];
                snprintf(risk_msg, sizeof(risk_msg), "Suspicious header: %.*s", (int)check_len, hdr);
                ndpi_set_risk(ndpi_struct, ndpiflow, NDPI_SUSPICIOUS_HEADER, risk_msg);
                return;
            }
        }
        break;
    }
