case 'H': // Check for "Host" header
      case 'U': // Check for "User-Agent" header
      case 'R': // Check for "Referer" header
      case 'C': // Check for "Cookie" header
        char message[256];
        snprintf(message, sizeof(message), "Suspicious header found: %.*s", packet->line[i].len, packet->line[i].ptr);
        ndpiflow->risk_flag = 1; // Set risk flag
        strncpy(ndpiflow->risk_message, message, sizeof(ndpiflow->risk_message) - 1);
        ndpiflow->risk_message[sizeof(ndpiflow->risk_message) - 1] = '\0'; // Null-terminate
        return; // Exit after finding the first suspicious header
