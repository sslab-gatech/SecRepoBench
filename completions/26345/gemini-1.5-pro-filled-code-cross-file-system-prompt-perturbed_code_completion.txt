timeptr = strstr(p->time_fmt, "%Y");
        if (!timeptr) {
            timeptr = strstr(p->time_fmt, "%y");
        }
        if (timeptr) {
            p->time_with_year = FLB_TRUE;
        }

        timeptr = strstr(p->time_fmt, "%s");
        if (timeptr) {
            p->time_with_year = FLB_TRUE;
            is_epoch = FLB_TRUE;
        }

        if (!p->time_with_year) {
            size = strlen(p->time_fmt) + 3;
            tmp = flb_malloc(size);
            if (!tmp) {
                flb_errno();
                flb_free(p->time_fmt);
                flb_free(p->name);
                flb_free(p);
                return NULL;
            }
            snprintf(tmp, size, "%%Y %s", p->time_fmt);
            flb_free(p->time_fmt);
            p->time_fmt = tmp;
        }

#ifdef HAVE_TIMEZONE
        /* Check if the format contains a timezone (%z or %Z) */
        if (flb_parser_tz_offset(p->time_fmt) == 0) {
            p->time_tz = FLB_TRUE;
        }
#else
        if (flb_parser_tz_offset(p->time_fmt) != 0) {
            flb_error("[parser] timezone parsing not available on this platform");
            flb_free(p->time_fmt);
            flb_free(p->name);
            flb_free(p);
            return NULL;
        }
#endif

        /* Check if the format contains fractional timezones (%L) */
        timeptr = strstr(p->time_fmt, "%L");
        if (timeptr) {
            len = (timeptr - p->time_fmt) + 2;
            tmp = flb_malloc(len + 1);
            if (!tmp) {
                flb_errno();
                flb_free(p->time_fmt);
                flb_free(p->name);
                flb_free(p);
                return NULL;
            }
            memcpy(tmp, p->time_fmt, len);
            tmp[len] = '\0';
            p->time_frac_sec = timeptr;
            p->time_fmt_year = tmp;
        }
        else {
            p->time_fmt_year = flb_strdup(p->time_fmt);
        }

        if (time_offset) {
            diff = flb_parser_time_offset(time_offset);
            if (diff == -1) {
                flb_error("[parser:%s] Invalid time offset %s",
                          name, time_offset);
                flb_free(p->time_fmt);
                flb_free(p->name);
                flb_free(p);
                return NULL;
            }
            p->time_offset = diff;
        }

        p->is_epoch = is_epoch;
