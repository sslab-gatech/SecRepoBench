if (strstr(time_fmt, "%Y") || strstr(time_fmt, "%y")) {
            p->time_with_year = FLB_TRUE;
        }
        else if (strstr(time_fmt, "%s")) {
            p->time_with_year = FLB_TRUE;
            is_epoch = FLB_TRUE;
        }
        else {
            len = strlen(time_fmt) + 3; // 3 for "%Y" and null terminator
            tmp = flb_malloc(len);
            if (!tmp) {
                flb_errno();
                flb_free(p->time_fmt);
                flb_free(p->time_fmt_full);
                flb_free(p);
                return NULL;
            }
            snprintf(tmp, len, "%%Y %s", time_fmt);
            p->time_fmt_year = tmp;
            p->time_with_year = FLB_FALSE;
        }

        if (strstr(time_fmt, "%z") || strstr(time_fmt, "%Z")) {
            p->time_with_tz = FLB_TRUE;
        }
        else {
            p->time_with_tz = FLB_FALSE;
        }

        if (strstr(time_fmt, "%L")) {
            timeptr = strstr(p->time_fmt, "%L");
            *timeptr = '\0';
            p->time_frac_secs = timeptr + 2;
        }
        else {
            p->time_frac_secs = NULL;
        }

        if (time_offset) {
            diff = atoi(time_offset);
        }
        p->time_offset = diff;
