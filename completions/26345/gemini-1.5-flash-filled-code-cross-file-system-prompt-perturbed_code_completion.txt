if (strstr(p->time_fmt, "%s") != NULL) {
            is_epoch = FLB_TRUE;
            p->time_fmt_year = NULL;
        }
        else if (strstr(p->time_fmt, "%Y") != NULL || strstr(p->time_fmt, "%y") != NULL) {
            p->time_fmt_year = NULL;
        }
        else {
            len = strlen(p->time_fmt) + 3;
            tmp = flb_malloc(len);
            if (!tmp) {
                flb_errno();
                flb_free(p->time_fmt_full);
                flb_free(p->time_fmt);
                flb_free(p);
                return NULL;
            }
            snprintf(tmp, len, "%Y%s", p->time_fmt);
            p->time_fmt_year = tmp;
        }

        if (flb_tz_support() == FLB_FALSE) {
            if (strstr(p->time_fmt, "%z") != NULL ||
                strstr(p->time_fmt, "%Z") != NULL) {
                flb_error("[parser:%s] timezone is not supported in this platform", name);
                flb_free(p->time_fmt_full);
                flb_free(p->time_fmt);
                if (p->time_fmt_year) {
                    flb_free(p->time_fmt_year);
                }
                flb_free(p);
                return NULL;
            }
        }
        else {
            if (strstr(p->time_fmt, "%z") != NULL ||
                strstr(p->time_fmt, "%Z") != NULL) {
                p->has_tz = FLB_TRUE;
            }
        }

        timeptr = strstr(p->time_fmt, "%L");
        if (timeptr != NULL) {
            p->time_fmt_frac = timeptr;
            *timeptr = '\0';
        }

        if (time_offset) {
            diff = flb_utils_time_to_seconds(time_offset);
            if (diff == -1) {
                flb_error("[parser:%s] Invalid time offset '%s'", name, time_offset);
                flb_free(p->time_fmt_full);
                flb_free(p->time_fmt);
                if (p->time_fmt_year) {
                    flb_free(p->time_fmt_year);
                }
                flb_free(p);
                return NULL;
            }
        }
        p->time_offset = diff;
