/* Check if the format is considering the year */
        if (strstr(time_fmt, "%Y") || strstr(time_fmt, "%y")) {
            p->time_with_year = FLB_TRUE;
        }
        else if (strstr(time_fmt, "%s")) {
            p->time_with_year = FLB_TRUE;
            is_epoch = FLB_TRUE;
        }
        else {
            len = strlen(time_fmt) + 3; /* 3 for "%Y" and null terminator */
            tmp = flb_malloc(len);
            if (!tmp) {
                flb_errno();
                flb_free(p->time_fmt);
                flb_free(p->time_fmt_full);
                flb_free(p);
                return NULL;
            }
            snprintf(tmp, len, "%%Y %s", time_fmt);
            p->time_fmt_year = tmp;
        }

        /* Check for timezone support */
        if (strstr(time_fmt, "%z") || strstr(time_fmt, "%Z")) {
#ifdef FLB_HAVE_TIMESPEC_GET
            p->time_with_tz = FLB_TRUE;
#else
            flb_error("[parser:%s] time format includes timezone but platform "
                      "does not support it", name);
            flb_free(p->time_fmt);
            flb_free(p->time_fmt_full);
            if (p->time_fmt_year) {
                flb_free(p->time_fmt_year);
            }
            flb_free(p);
            return NULL;
#endif
        }

        /* Handle fractional seconds */
        timeptr = strstr(time_fmt, "%L");
        if (timeptr) {
            diff = timeptr - time_fmt;
            size = diff + 1; /* +1 for null terminator */
            tmp = flb_malloc(size);
            if (!tmp) {
                flb_errno();
                flb_free(p->time_fmt);
                flb_free(p->time_fmt_full);
                if (p->time_fmt_year) {
                    flb_free(p->time_fmt_year);
                }
                flb_free(p);
                return NULL;
            }
            memcpy(tmp, time_fmt, diff);
            tmp[diff] = '\0';
            flb_free(p->time_fmt);
            p->time_fmt = tmp;
            p->time_frac_secs = FLB_TRUE;
        }

        /* Handle fixed timezone offset */
        if (time_offset) {
            p->time_offset = atoi(time_offset);
        }
