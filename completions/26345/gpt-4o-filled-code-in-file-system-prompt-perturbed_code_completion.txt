if (strstr(p->time_fmt, "%Y") || strstr(p->time_fmt, "%y")) {
            p->time_with_year = FLB_TRUE;
        }
        else if (strstr(p->time_fmt, "%s")) {
            p->time_with_year = FLB_TRUE;
            is_epoch = FLB_TRUE;
        }
        else {
            len = strlen(p->time_fmt) + 3;
            tmp = flb_malloc(len);
            if (!tmp) {
                flb_errno();
                flb_free(p->time_fmt);
                flb_free(p->time_fmt_full);
                flb_free(p);
                return NULL;
            }
            snprintf(tmp, len, "%%Y %s", p->time_fmt);
            p->time_fmt_year = tmp;
        }

        if (strstr(p->time_fmt, "%z") || strstr(p->time_fmt, "%Z")) {
#ifdef FLB_HAVE_GMTOFF
            p->time_with_tz = FLB_TRUE;
#else
            flb_error("[parser:%s] time format contains timezone but " \
                      "your platform does not support it", name);
            flb_free(p->time_fmt);
            flb_free(p->time_fmt_full);
            flb_free(p);
            return NULL;
#endif
        }

        timeptr = strstr(p->time_fmt, "%L");
        if (timeptr) {
            size = (timeptr - p->time_fmt);
            p->time_frac_secs = flb_strdup(timeptr + 2);
            tmp = flb_malloc(size + 1);
            if (!tmp) {
                flb_errno();
                flb_free(p->time_fmt);
                flb_free(p->time_fmt_full);
                flb_free(p->time_frac_secs);
                flb_free(p);
                return NULL;
            }
            memcpy(tmp, p->time_fmt, size);
            tmp[size] = '\0';
            flb_free(p->time_fmt);
            p->time_fmt = tmp;
        }

        if (time_offset) {
            status = flb_parser_tzone_offset(time_offset, strlen(time_offset), &diff);
            if (status == -1) {
                flb_error("[parser:%s] Invalid time offset %s", name, time_offset);
                flb_free(p->time_fmt);
                flb_free(p->time_fmt_full);
                flb_free(p);
                return NULL;
            }
            p->time_offset = diff;
        }
