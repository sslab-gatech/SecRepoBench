if (strstr(time_fmt, "%Y") || strstr(time_fmt, "%y")) {
            p->time_year = FLB_TRUE;
        }
        else if (strstr(time_fmt, "%s")) {
            is_epoch = FLB_TRUE;
            p->time_year = FLB_TRUE;
        }
        else {
            /* Prepend the year */
            len = strlen(time_fmt);
            size = len + 4;
            tmp = flb_malloc(size);
            if (!tmp) {
                flb_errno();
                flb_free(p);
                return NULL;
            }
            memset(tmp, 0, size);
            memcpy(tmp, "%Y ", 3);
            memcpy(tmp + 3, time_fmt, len);
#ifdef FLB_HAVE_TIMEZONE
            p->time_fmt_year = flb_strdup(tmp);
#endif
            flb_free(tmp);
            p->time_fmt = flb_strdup(time_fmt);
        }

#ifdef FLB_HAVE_TIMEZONE
        if (strstr(time_fmt, "%z") || strstr(time_fmt, "%Z")) {
            p->time_timezone = FLB_TRUE;
        }
#else
        if (strstr(time_fmt, "%z") || strstr(time_fmt, "%Z")) {
            flb_error("[parser:%s] timezone is not supported on this platform",
                      name);
            flb_free(p->time_fmt);
            flb_free(p->time_fmt_full);
            flb_free(p);
            return NULL;
        }
#endif

        /* Fractional seconds */
        timeptr = strstr(time_fmt, "%L");
        if (timeptr) {
            p->time_frac_fmt = timeptr;
            p->time_frac_len = strlen(timeptr);
            len = timeptr - time_fmt;
            flb_free(p->time_fmt);
            p->time_fmt = flb_strndup(time_fmt, len);
        }

        /* Fixed timezone offset */
        if (time_offset) {
#ifdef FLB_HAVE_TIMEZONE
            diff = atoi(time_offset);
            p->time_offset_sec = diff * 60;
#else
            flb_error("[parser:%s] timezone offset is not supported on this platform",
                      name);
            flb_free(p->time_fmt);
            flb_free(p->time_fmt_full);
            flb_free(p);
            return NULL;
#endif
        }
