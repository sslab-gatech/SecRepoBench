if (strstr(time_fmt, "%Y") || strstr(time_fmt, "%y")) {
            p->time_with_year = FLB_TRUE;
        }
        else if (strstr(time_fmt, "%s")) {
            p->time_with_year = FLB_TRUE;
            is_epoch = FLB_TRUE;
        }
        else {
            len = strlen(time_fmt) + 3;
            tmp = flb_malloc(len);
            if (!tmp) {
                flb_errno();
                flb_free(p->time_fmt);
                flb_free(p->time_fmt_full);
                flb_free(p);
                return NULL;
            }
            snprintf(tmp, len, "%%Y %s", time_fmt);
            flb_free(p->time_fmt);
            p->time_fmt = tmp;
            p->time_with_year = FLB_FALSE;
        }

        if (strstr(p->time_fmt, "%L")) {
            timeptr = strstr(p->time_fmt, "%L");
            diff = timeptr - p->time_fmt;
            size = strlen(p->time_fmt) - diff;
            p->time_fmt_year = flb_malloc(size + 1);
            if (!p->time_fmt_year) {
                flb_errno();
                flb_free(p->time_fmt);
                flb_free(p->time_fmt_full);
                flb_free(p);
                return NULL;
            }
            strncpy(p->time_fmt_year, p->time_fmt, diff);
            p->time_fmt_year[diff] = '\0';
        }

        if (time_offset) {
            p->time_offset = atoi(time_offset);
        }
