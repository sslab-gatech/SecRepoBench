if (strstr(p->time_fmt, "%s") != NULL) {
            is_epoch = FLB_TRUE;
            p->has_year = FLB_TRUE;
        }
        else if (strstr(p->time_fmt, "%Y") != NULL || strstr(p->time_fmt, "%y") != NULL) {
            p->has_year = FLB_TRUE;
        }
        else {
            len = strlen(p->time_fmt) + 3;
            tmp = flb_malloc(len);
            if (!tmp) {
                flb_errno();
                flb_free(p->time_fmt_full);
                flb_free(p->time_fmt);
                flb_free(p);
                return NULL;
            }
            snprintf(tmp, len, "%Y%s", p->time_fmt);
            flb_free(p->time_fmt);
            p->time_fmt = tmp;
            p->time_fmt_year = flb_strdup(p->time_fmt);
            p->has_year = FLB_TRUE;
        }

        if (flb_tz_support() == FLB_FALSE) {
            if (strstr(p->time_fmt, "%z") != NULL || strstr(p->time_fmt, "%Z") != NULL) {
                flb_error("[parser:%s] timezone not supported by this platform", name);
                flb_free(p->time_fmt_full);
                flb_free(p->time_fmt);
                if (p->time_fmt_year) {
                    flb_free(p->time_fmt_year);
                }
                flb_free(p);
                return NULL;
            }
        }
        else {
            if (strstr(p->time_fmt, "%z") != NULL || strstr(p->time_fmt, "%Z") != NULL) {
                p->has_tz = FLB_TRUE;
            }
        }

        if (strstr(p->time_fmt, "%L") != NULL) {
            timeptr = strstr(p->time_fmt, "%L");
            p->time_fmt_frac = timeptr;
            size = timeptr - p->time_fmt;
            tmp = flb_malloc(size + 1);
            if (!tmp) {
                flb_errno();
                flb_free(p->time_fmt_full);
                flb_free(p->time_fmt);
                if (p->time_fmt_year) {
                    flb_free(p->time_fmt_year);
                }
                flb_free(p);
                return NULL;
            }
            strncpy(tmp, p->time_fmt, size);
            tmp[size] = '\0';
            flb_free(p->time_fmt);
            p->time_fmt = tmp;
        }

        if (time_offset) {
            diff = atoi(time_offset);
        }
