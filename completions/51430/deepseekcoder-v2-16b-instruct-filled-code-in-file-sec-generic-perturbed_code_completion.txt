// timeout
      return NULL;
  }

  if (cpdwordmax > 0 && wordnum >= cpdwordmax)
    return NULL;

  if (cpdmin > 0 && len < cpdmin)
    return NULL;

  if (cpdmaxsyllable > 0 && numsyllable > cpdmaxsyllable)
    return NULL;

  if (maxwordnum > 0 && wnum > maxwordnum)
    return NULL;

  if (cpdmin > 0) {
    setcminmax(&cmin, &cmax, compound_word.c_str(), len);
    if (cmin > 0 && len < cmin)
      return NULL;
    if (cmax > 0 && len < cmax)
      return NULL;
  }

  if (checkcompoundcase && cpdmin > 0) {
    for (int i = cmin; i < cmax; i++) {
      if (isupper(compound_word[i]) && islower(compound_word[i - 1]))
        return NULL;
      if (islower(compound_word[i]) && isupper(compound_word[i - 1]))
        return NULL;
    }
  }

  if (checkcompoundtriple && cpdmin > 0) {
    for (int i = cmin; i < cmax - 1; i++) {
      if (compound_word[i] == compound_word[i + 1] &&
          compound_word[i] == compound_word[i - 1]) {
        if (!simplifiedtriple)
          return NULL;
        striple = 1;
      }
    }
  }

  if (checkcompounddup && cpdmin > 0) {
    for (int i = cmin; i < cmax - 1; i++) {
      if (compound_word[i] == compound_word[i + 1])
        return NULL;
    }
  }

  if (checkcompoundrep && cpdmin > 0) {
    if (cpdrep_check(compound_word.c_str(), len))
      return NULL;
  }

  if (checkcompoundpat && cpdmin > 0) {
    for (int i = 0; i < wordnum; i++) {
      if (cpdpat_check(compound_word.c_str(), words[i]->alen, words[i], NULL,
                       affixed))
        return NULL;
    }
  }

  if (checknum && cpdmin > 0) {
    for (int i = 0; i < wordnum; i++) {
      if (words[i]->alen && TESTAFF(words[i]->astr, "0", words[i]->alen))
        return NULL;
    }
  }

  if (cpdwordpair && cpdmin > 0) {
    if (cpdwordpair_check(compound_word.c_str(), len))
      return NULL;
  }

  if (cpdsyllablenum.size() > 0 && cpdmin > 0) {
    oldnumsyllable = numsyllable;
    numsyllable = get_syllable(compound_word);
    if (numsyllable < 0)
      return NULL;
    if (numsyllable < cpdmin)
      return NULL;
    if (cpdmaxsyllable > 0 && numsyllable > cpdmaxsyllable)
      return NULL;
  }

  if (cpdmin > 0) {
    oldwordnum = wordnum;
    wordnum++;
    if (wordnum > cpdmin)
      return NULL;
  }

  if (cpdmin > 0) {
    oldcmin = cmin;

  }

  if (cpdmin > 0) {

        return NULL;

  if (cpdmin > 0) {
    if (pfxappnd) {
      oldcmin = cmin;
      cmin = pfxappnd->getMinLen(cmin);
      if (cmin < 0)
        return NULL;
    }
    if (sfxappnd) {
      oldcmin = cmin;

      if (cmin < 0)
        return NULL;
    }

  if (cpdmin > 0) {
    if (pfxappnd) {
      oldcmax = cmax;
      cmax = pfxappnd->getMaxLen(cmax);
      if (cmax < 0)
        return NULL;
    }
    if (sfxappnd) {
      oldcmax = cmax;
      cmax = sfxappnd->getMaxLen(cmax);
      if (cmax < 0)
        return NULL;
    }
  }

  if (cpdmin > 0) {
    if (pfxappnd) {
      oldlen = len;
      len = pfxappnd->getLen(len);
      if (len < 0)
        return NULL;
    }
    if (sfxappnd) {
      oldlen = len;
      len = sfxappnd->getLen(len);
      if (len < 0)
        return NULL;
    }
  }

  if (cpdmin > 0) {
    if (pfxappnd) {
      checkedstriple = striple;
      striple = pfxappnd->getTriple(striple);
      if (striple < 0)
        return NULL;
    }
    if (sfxappnd) {
      checkedstriple = striple;
      striple = sfxappnd->getTriple(striple);
      if (striple < 0)
        return NULL;
    }
  }

  if (cpdmin > 0) {
    if (pfxappnd) {
      soldi = 1;
      checked_prefix = 1;
      checked_prefix = pfxappnd->getCheckPrefix(checked_prefix);
      if (checked_prefix < 0)
        return NULL;
    }
    if (sfxappnd) {
      soldi = 1;
      checked_prefix = 1;
      checked_prefix = sfxappnd->getCheckPrefix(checked_prefix);
      if (checked_prefix < 0)
        return NULL;
    }
  }

  if (cpdmin > 0) {
    if (pfxappnd) {
      scpd = 1;
      scpd = pfxappnd->getCPD(scpd);
      if (scpd < 0)
        return NULL;
    }
    if (sfxappnd) {
      scpd = 1;
      scpd = sfxappnd->getCPD(scpd);
      if (scpd < 0)
        return NULL;
    }
  }

  if (cpdmin > 0) {
    if (pfxappnd) {
      rv_first = pfxappnd->getCompound(compound_word, words, rwords, hu_mov_rule,
                                       is_sug, info);
      if (rv_first)
        return rv_first;
    }
    if (sfxappnd) {
      rv_first = sfxappnd->getCompound(compound_word, words, rwords, hu_mov_rule,
                                       is_sug, info);
      if (rv_first)
        return rv_first;
    }
  }

    if (pfxappnd) {
      pfxappnd = NULL;
    }
    if (sfxappnd) {
      sfxappnd = NULL;
    }
  }

    numsyllable = oldnumsyllable2;
    wordnum = oldwordnum2;
    cmin = oldcmin;
    cmax = oldcmax;
    len = oldlen;
    striple = checkedstriple;
    checked_prefix = soldi;
    scpd = soldi;
