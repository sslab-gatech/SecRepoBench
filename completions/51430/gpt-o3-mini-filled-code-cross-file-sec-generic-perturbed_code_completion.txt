{
  char orig = st[i];
  st[i] = '\0';
  std::string firstPart = st.substr(0, i);
  // Check if the beginning of the remaining part repeats the first part.
  if (st.size() - i >= firstPart.size() &&
      st.compare(i, firstPart.size(), firstPart) == 0) {
    st[i] = orig;
    return NULL;
  }
  st[i] = orig;
}
