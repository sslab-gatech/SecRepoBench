if (checkcompoundrep && cpdrep_check(compound_word.c_str(), len)) {
                  return NULL;
                }

                size_t first_part_len = i;
                if (first_part_len > 0 && first_part_len <= len) {
                  std::string first_part = compound_word.substr(0, first_part_len);
                  std::string second_part = compound_word.substr(i);

                  if (first_part == second_part) {
                    return NULL;
                  }

                  if (checkcompoundrep) {
                    if (cpdrep_check(second_part.c_str(), second_part.size())) {
                      return NULL;
                    }
                  }
