{
  std::string prefix = compound_word.substr(0, i);
  if (!prefix.empty() && (i + prefix.size() <= compound_word.size()) &&
      compound_word.compare(i, prefix.size(), prefix) == 0) {
    // Temporary modification: replace the first character of the repeated segment
    char original = st[i];
    st[i] = '*';
    struct hentry* tmp = lookup(st.c_str(), st.size());
    st[i] = original;  // restore the original character
    if (!tmp)
      return NULL;
  }
