if (ret == NULL) {
        xsltTransformError(tctxt, NULL, tctxt->inst,
            "exsltCryptoRc4DecryptFunction: Failed to allocate decrypted string\n");
        tctxt->state = XSLT_STATE_STOPPED;
        goto done;
    }

    PLATFORM_RC4_DECRYPT(parsercontext, padkey, bin, bin_len, ret, ret_len);

    ret[ret_len] = '\0'; // Null-terminate the decrypted string
    xmlXPathReturnString(parsercontext, ret);

done:
    if (key != NULL)
        xmlFree(key);
    if (str != NULL)
        xmlFree(str);
    if (padkey != NULL)
        xmlFree(padkey);
    if (bin != NULL)
        xmlFree(bin);
    if (ret != NULL)
        xmlFree(ret);
