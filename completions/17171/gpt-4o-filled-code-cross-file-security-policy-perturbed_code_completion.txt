if (ret == NULL) {
        xsltTransformError(tctxt, NULL, tctxt->inst,
            "exsltCryptoRc4DecryptFunction: Failed to allocate memory for decrypted data\n");
        tctxt->state = XSLT_STATE_STOPPED;
        xmlXPathReturnEmptyString(parsercontext);
        goto done;
    }

    RC4_KEY rc4_key;
    RC4_set_key(&rc4_key, RC4_KEY_LENGTH, padkey);
    RC4(&rc4_key, ret_len, bin, ret);

    ret[ret_len] = '\0'; // Null-terminate the decrypted string

    // Validate UTF-8 encoding of the decrypted result
    if (xmlUTF8Strlen(ret) < 0) {
        xsltTransformError(tctxt, NULL, tctxt->inst,
            "exsltCryptoRc4DecryptFunction: Decrypted data is not valid UTF-8\n");
        tctxt->state = XSLT_STATE_STOPPED;
        xmlXPathReturnEmptyString(parsercontext);
        xmlFree(ret);
        goto done;
    }

    xmlXPathReturnString(parsercontext, ret);

done:
    if (key != NULL)
        xmlFree(key);
    if (ret != NULL)
        xmlFree(ret);
