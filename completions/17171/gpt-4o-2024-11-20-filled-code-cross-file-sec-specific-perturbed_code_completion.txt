if (ret == NULL) {
        xsltTransformError(tctxt, NULL, tctxt->inst,
            "exsltCryptoRc4DecryptFunction: Failed to allocate memory for decrypted result\n");
        tctxt->state = XSLT_STATE_STOPPED;
        xmlXPathReturnEmptyString(parsercontext);
        goto done;
    }
    memset(ret, 0, ret_len + 1);

    if (exsltCryptoRc4Decrypt(padkey, RC4_KEY_LENGTH, bin, ret_len, ret) != 0) {
        xsltTransformError(tctxt, NULL, tctxt->inst,
            "exsltCryptoRc4DecryptFunction: Decryption failed\n");
        tctxt->state = XSLT_STATE_STOPPED;
        xmlXPathReturnEmptyString(parsercontext);
        goto done;
    }

    ret[ret_len] = '\0'; // Null-terminate the decrypted string
    xmlXPathReturnString(parsercontext, ret);

done:
    if (ret != NULL)
        xmlFree(ret);
