static MagickBooleanType FillMaskedImage(Image *image,const DrawInfo *draw_info,
  const char *id,ExceptionInfo *exception)
{
  Image
    *mask;

  MagickBooleanType
    status;

  mask=DrawClippingMask(image,draw_info,draw_info->clip_mask,id,exception);
  if (mask == (Image *) NULL)
    return(MagickFalse);
  status=SetImageMask(image,WritePixelMask,mask,exception);
  mask=DestroyImage(mask);
  return(status);
}
