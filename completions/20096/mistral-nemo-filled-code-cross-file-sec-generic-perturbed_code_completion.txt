#define CheckPathExtent(needed) \
  if (path_p + needed > path_p_end) \
    path_p = (PointInfo *) ResizeQuantumMemory(path_p, (size_t) (path_p_end - path_p) * sizeof(*path_p), needed); \
  if (path_q + needed > path_q_end) \
    path_q = (PointInfo *) ResizeQuantumMemory(path_q, (size_t) (path_q_end - path_q) * sizeof(*path_q), needed);
