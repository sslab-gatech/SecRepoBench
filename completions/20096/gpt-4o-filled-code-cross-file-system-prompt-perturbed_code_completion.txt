#define CheckPathExtent(pad) \
{ \
  if ((p+pad) >= (ssize_t) stroke_limit) \
    { \
      PointInfo \
        *reserve_p, \
        *reserve_q; \
\
      stroke_limit+=MaxStrokePad; \
      reserve_p=(PointInfo *) ResizeQuantumMemory(path_p,(size_t) stroke_limit+MaxStrokePad,sizeof(*reserve_p)); \
      reserve_q=(PointInfo *) ResizeQuantumMemory(path_q,(size_t) stroke_limit+MaxStrokePad,sizeof(*reserve_q)); \
      if ((reserve_p == (PointInfo *) NULL) || (reserve_q == (PointInfo *) NULL)) \
        { \
          if (reserve_p != (PointInfo *) NULL) \
            path_p=(PointInfo *) RelinquishMagickMemory(reserve_p); \
          if (reserve_q != (PointInfo *) NULL) \
            path_q=(PointInfo *) RelinquishMagickMemory(reserve_q); \
          polygon_primitive=(PrimitiveInfo *) RelinquishMagickMemory(polygon_primitive); \
          return((PrimitiveInfo *) NULL); \
        } \
      path_p=reserve_p; \
      path_q=reserve_q; \
    } \
}
