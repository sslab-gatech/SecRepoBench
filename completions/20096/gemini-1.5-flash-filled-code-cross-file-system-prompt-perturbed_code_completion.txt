#define CheckPathExtent(add_amount) \
  if ((p+add_amount) >= (ssize_t) (stroke_limit+MaxStrokePad)) \
    { \
      size_t extent; \
      \
      extent=stroke_limit+MaxStrokePad; \
      extent*=2; \
      path_p=(PointInfo *) ResizeQuantumMemory(path_p,extent, \
        sizeof(*path_p)); \
      if (path_p == (PointInfo *) NULL) \
        { \
          path_q=(PointInfo *) RelinquishMagickMemory(path_q); \
          polygon_primitive=(PrimitiveInfo *) RelinquishMagickMemory( \
            polygon_primitive); \
          return((PrimitiveInfo *) NULL); \
        } \
      path_q=(PointInfo *) ResizeQuantumMemory(path_q,extent, \
        sizeof(*path_q)); \
      if (path_q == (PointInfo *) NULL) \
        { \
          path_p=(PointInfo *) RelinquishMagickMemory(path_p); \
          polygon_primitive=(PrimitiveInfo *) RelinquishMagickMemory( \
            polygon_primitive); \
          return((PrimitiveInfo *) NULL); \
        } \
      stroke_limit=extent; \
    }
