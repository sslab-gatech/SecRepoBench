#define CheckPathExtent(extent) \
do { \
  if ((size_t) (p+extent) > stroke_limit) \
    { \
      stroke_limit=(size_t) (p+extent); \
      path_p=(PointInfo *) ResizeQuantumMemory(path_p,stroke_limit+ \
        MaxStrokePad,sizeof(*path_p)); \
      if (path_p == (PointInfo *) NULL) \
        { \
          path_q=(PointInfo *) RelinquishMagickMemory(path_q); \
          polygon_primitive=(PrimitiveInfo *) RelinquishMagickMemory( \
            polygon_primitive); \
          return((PrimitiveInfo *) NULL); \
        } \
      path_q=(PointInfo *) ResizeQuantumMemory(path_q,stroke_limit+ \
        MaxStrokePad,sizeof(*path_q)); \
      if (path_q == (PointInfo *) NULL) \
        { \
          path_p=(PointInfo *) RelinquishMagickMemory(path_p); \
          polygon_primitive=(PrimitiveInfo *) RelinquishMagickMemory( \
            polygon_primitive); \
          return((PrimitiveInfo *) NULL); \
        } \
    } \
  if ((size_t) (q+extent) > stroke_limit) \
    { \
      stroke_limit=(size_t) (q+extent); \
      path_p=(PointInfo *) ResizeQuantumMemory(path_p,stroke_limit+ \
        MaxStrokePad,sizeof(*path_p)); \
      if (path_p == (PointInfo *) NULL) \
        { \
          path_q=(PointInfo *) RelinquishMagickMemory(path_q); \
          polygon_primitive=(PrimitiveInfo *) RelinquishMagickMemory( \
            polygon_primitive); \
          return((PrimitiveInfo *) NULL); \
        } \
      path_q=(PointInfo *) ResizeQuantumMemory(path_q,stroke_limit+ \
        MaxStrokePad,sizeof(*path_q)); \
      if (path_q == (PointInfo *) NULL) \
        { \
          path_p=(PointInfo *) RelinquishMagickMemory(path_p); \
          polygon_primitive=(PrimitiveInfo *) RelinquishMagickMemory( \
            polygon_primitive); \
          return((PrimitiveInfo *) NULL); \
        } \
    } \
} while (0)
