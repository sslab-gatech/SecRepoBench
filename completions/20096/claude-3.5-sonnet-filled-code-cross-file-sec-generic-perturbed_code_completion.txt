#define CheckPathExtent(delta) \
{ \
  if ((ssize_t) (p+delta) >= (ssize_t) stroke_limit) \
    { \
      stroke_limit+=delta; \
      path_p=(PointInfo *) ResizeQuantumMemory(path_p,(size_t) stroke_limit+ \
        MaxStrokePad,sizeof(*path_p)); \
      if (path_p == (PointInfo *) NULL) \
        { \
          path_q=(PointInfo *) RelinquishMagickMemory(path_q); \
          polygon_primitive=(PrimitiveInfo *) \
            RelinquishMagickMemory(polygon_primitive); \
          return((PrimitiveInfo *) NULL); \
        } \
    } \
  if ((ssize_t) (q+delta) >= (ssize_t) stroke_limit) \
    { \
      stroke_limit+=delta; \
      path_q=(PointInfo *) ResizeQuantumMemory(path_q,(size_t) stroke_limit+ \
        MaxStrokePad,sizeof(*path_q)); \
      if (path_q == (PointInfo *) NULL) \
        { \
          path_p=(PointInfo *) RelinquishMagickMemory(path_p); \
          polygon_primitive=(PrimitiveInfo *) \
            RelinquishMagickMemory(polygon_primitive); \
          return((PrimitiveInfo *) NULL); \
        } \
    } \
}
