#define CheckPathExtent(extent) \
  if ((p+extent) >= (ssize_t) (stroke_limit+MaxStrokePad)) \
    { \
      size_t \
        new_limit; \
      \
      new_limit=stroke_limit+MaxStrokePad+extent; \
      path_p=(PointInfo *) ResizeQuantumMemory(path_p,(size_t) \
        stroke_limit+MaxStrokePad,new_limit*sizeof(*path_p)); \
      if (path_p == (PointInfo *) NULL) \
        { \
          path_q=(PointInfo *) RelinquishMagickMemory(path_q); \
          polygon_primitive=(PrimitiveInfo *) RelinquishMagickMemory( \
            polygon_primitive); \
          return((PrimitiveInfo *) NULL); \
        } \
      path_q=(PointInfo *) ResizeQuantumMemory(path_q,(size_t) \
        stroke_limit+MaxStrokePad,new_limit*sizeof(*path_q)); \
      if (path_q == (PointInfo *) NULL) \
        { \
          path_p=(PointInfo *) RelinquishMagickMemory(path_p); \
          polygon_primitive=(PrimitiveInfo *) RelinquishMagickMemory( \
            polygon_primitive); \
          return((PrimitiveInfo *) NULL); \
        } \
      stroke_limit=new_limit; \
    }
