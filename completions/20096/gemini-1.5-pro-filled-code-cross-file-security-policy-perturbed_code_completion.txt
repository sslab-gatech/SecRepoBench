#define CheckPathExtent(pad) \
{ \
  if ((p+pad) >= stroke_limit) \
    { \
      stroke_limit+=pad; \
      PointInfo \
        *stroke_p; \
      stroke_p=(PointInfo *) ResizeQuantumMemory(path_p,(size_t) \
        stroke_limit+MaxStrokePad,sizeof(*path_p)); \
      if (stroke_p == (PointInfo *) NULL) \
        { \
          path_q=(PointInfo *) RelinquishMagickMemory(path_q); \
          path_p=(PointInfo *) RelinquishMagickMemory(path_p); \
          polygon_primitive=(PrimitiveInfo *) RelinquishMagickMemory( \
            polygon_primitive); \
          return((PrimitiveInfo *) NULL); \
        } \
      path_p=stroke_p; \
    } \
  if ((q+pad) >= stroke_limit) \
    { \
      stroke_limit+=pad; \
      PointInfo \
        *stroke_q; \
      stroke_q=(PointInfo *) ResizeQuantumMemory(path_q,(size_t) \
        stroke_limit+MaxStrokePad,sizeof(*path_q)); \
      if (stroke_q == (PointInfo *) NULL) \
        { \
          path_p=(PointInfo *) RelinquishMagickMemory(path_p); \
          path_q=(PointInfo *) RelinquishMagickMemory(path_q); \
          polygon_primitive=(PrimitiveInfo *) RelinquishMagickMemory( \
            polygon_primitive); \
          return((PrimitiveInfo *) NULL); \
        } \
      path_q=stroke_q; \
    } \
}
