#define CheckPathExtent(additional) \
  if ((p + q + additional) >= (ssize_t) stroke_limit) { \
    size_t new_size = (stroke_limit + additional + MaxStrokePad) * 2; \
    path_p = (PointInfo *) ResizeQuantumMemory(path_p, new_size, sizeof(*path_p)); \
    path_q = (PointInfo *) ResizeQuantumMemory(path_q, new_size, sizeof(*path_q)); \
    if ((path_p == (PointInfo *) NULL) || (path_q == (PointInfo *) NULL)) { \
      path_p = (PointInfo *) RelinquishMagickMemory(path_p); \
      path_q = (PointInfo *) RelinquishMagickMemory(path_q); \
      polygon_primitive = (PrimitiveInfo *) RelinquishMagickMemory(polygon_primitive); \
      return((PrimitiveInfo *) NULL); \
    } \
    stroke_limit = new_size - MaxStrokePad; \
  }
