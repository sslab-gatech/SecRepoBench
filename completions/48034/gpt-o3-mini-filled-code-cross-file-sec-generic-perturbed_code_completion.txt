if ((x + 6) > packet->payload_packet_len) {
  break;
}
responsetype = get16(&x, packet->payload);
rsp_ttl = get32(&x, packet->payload);
if (rsp_ttl == 0) {
  ndpi_set_risk(ndpi_struct, flow, NDPI_MALFORMED_PACKET, "DNS response with zero TTL");
}
#ifdef DNS_DEBUG
NDPI_LOG_DBG2(ndpi_struct, "DNS reply: response type=%u, TTL=%u\n", responsetype, rsp_ttl);
printf("[DNS] [reply] response type=%u, TTL=%u\n", responsetype, rsp_ttl);
#endif
