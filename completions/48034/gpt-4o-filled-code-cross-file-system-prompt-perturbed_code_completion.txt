responsetype = get16(&x, packet->payload);
	  rsp_ttl = get32(&x, packet->payload);

	  if(rsp_ttl == 0) {
	    ndpi_set_risk(ndpi_struct, flow, NDPI_SUSPICIOUS_DNS_TRAFFIC, "Zero TTL in DNS Response");
	  }

#ifdef DNS_DEBUG
	  NDPI_LOG_DBG2(ndpi_struct, "response_type=%2d ttl=%u\n", responsetype, rsp_ttl);
	  printf("[DNS] [response] response_type=%d ttl=%u\n", responsetype, rsp_ttl);
#endif

	  if(responsetype == 0x05 /* CNAME */) {
	    x += 2; /* Skip class */
	    x += 4; /* Skip TTL */
	    data_len = get16(&x, packet->payload);
	    x += data_len; /* Skip CNAME data */
	    continue;
	  }
