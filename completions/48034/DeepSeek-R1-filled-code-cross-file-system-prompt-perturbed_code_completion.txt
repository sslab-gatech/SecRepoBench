responsetype = get16(&x, packet->payload);
x += 2; /* Skip class */
rsp_ttl = get32(&x, packet->payload);

if(rsp_ttl == 0) {
    ndpi_set_risk(ndpi_struct, flow, NDPI_SUSPICIOUS_TTL, "Zero TTL in DNS response");
}
