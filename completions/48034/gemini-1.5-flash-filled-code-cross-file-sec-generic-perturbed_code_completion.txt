responsetype = get16(&x, packet->payload);
	rsp_ttl = get32(&x, packet->payload);

	if(rsp_ttl == 0) {
	  ndpi_set_risk(ndpi_struct, flow, NDPI_SUSPICIOUS_DNS_TTL, "DNS response with TTL 0");
	}
