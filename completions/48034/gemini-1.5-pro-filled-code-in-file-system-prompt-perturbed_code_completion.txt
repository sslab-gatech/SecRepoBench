responsetype = get16(&x, packet->payload);
	  rsp_ttl = get16(&x, packet->payload) * 1000; /* TTL is in seconds, convert to milliseconds */

	  if(rsp_ttl == 0) {
	    ndpi_set_risk(ndpi_struct, flow, NDPI_DNS_SUSPICIOUS_TRAFFIC, "Zero TTL");
	  }

#endif

	  ndpi_check_dns_type(ndpi_struct, flow, responsetype);
