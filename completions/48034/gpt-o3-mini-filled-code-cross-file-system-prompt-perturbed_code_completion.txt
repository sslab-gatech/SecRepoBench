responsetype = get16(&x, packet->payload);
if ((x + 6) > packet->payload_packet_len)
  break;
u_int32_t rsp_ttl = ntohl(*(u_int32_t *)(packet->payload + x + 2));
if (rsp_ttl == 0) {
  ndpi_set_risk(ndpi_struct, flow, NDPI_MALFORMED_PACKET, "DNS TTL Zero");
}
#ifdef DNS_DEBUG
NDPI_LOG_DBG2(ndpi_struct, "DNS response: type=%u, ttl=%u\n", responsetype, rsp_ttl);
#endif
