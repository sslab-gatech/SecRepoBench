responsetype = get16(&x, packet->payload);
	  rsp_ttl = ntohl(*(u_int32_t*)&packet->payload[x+2]);

	  if(rsp_ttl == 0) {
	    ndpi_set_risk(ndpi_struct, flow, NDPI_DNS_SUSPICIOUS_TRAFFIC, "Zero TTL in DNS response");
	  }

	  NDPI_LOG_DBG2(ndpi_struct, "TTL=%u, responsetype=%u\n", rsp_ttl, responsetype);
#endif

	  ndpi_check_dns_type(ndpi_struct, flow, responsetype);
