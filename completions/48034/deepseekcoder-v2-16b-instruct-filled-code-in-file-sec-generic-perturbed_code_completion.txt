  x += data_len;
  continue; /* Skip CNAME */
}

    || ((responsetype == 0x1c) && (data_len == 16)) /* AAAA */
    ) {
  memcpy(&flow->protos.dns.rsp_addr, packet->payload + x, data_len);
}
