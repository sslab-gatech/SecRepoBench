responsetype = get16(&x, packet->payload);
	  rsp_ttl = get32(&x, packet->payload);

	  if (rsp_ttl == 0) {
	    ndpi_set_risk(ndpi_struct, flow, NDPI_DNS_SUSPICIOUS_TTL, "Suspicious DNS response with TTL of zero");
	  }

	  NDPI_LOG_DBG2(ndpi_struct, "rsp_type=%u, ttl=%u\n", responsetype, rsp_ttl);
