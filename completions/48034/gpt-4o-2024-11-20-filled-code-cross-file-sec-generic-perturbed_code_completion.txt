// Extract the response type and TTL (Time-To-Live) from the DNS response section
          responsetype = get16(&x, packet->payload); // Extract response type
          rsp_ttl = get32(&x, packet->payload);      // Extract TTL

          // Set a risk flag if the TTL is zero, indicating suspicious DNS traffic
          if(rsp_ttl == 0) {
            ndpi_set_risk(ndpi_struct, flow, NDPI_SUSPICIOUS_TTL, "DNS Response with TTL=0");
          }

#ifdef DNS_DEBUG
          // Output debug information about TTL and response type if DNS debugging is enabled
          NDPI_LOG_DBG2(ndpi_struct, "response_type=%u, ttl=%u\n", responsetype, rsp_ttl);
          printf("[DNS] [response] response_type=%u, ttl=%u\n", responsetype, rsp_ttl);
#endif

          // Perform a check on the DNS response type to further analyze the response
          if((responsetype != 0x1 /* A */) && (responsetype != 0x1c /* AAAA */) &&
             (responsetype != 0x5 /* CNAME */) && (responsetype != 0x2 /* NS */)) {
            ndpi_set_risk(ndpi_struct, flow, NDPI_UNKNOWN_DNS_RESPONSE_TYPE, "Unknown DNS Response Type");
          }
