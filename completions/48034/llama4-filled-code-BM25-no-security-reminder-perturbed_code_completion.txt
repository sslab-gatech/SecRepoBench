responsetype = get16(&x, packet->payload);
rsp_ttl = get32(&x, packet->payload);

if(rsp_ttl == 0) {
  ndpi_set_risk(ndpi_struct, flow, NDPI_SUSPICIOUS_DGA_DOMAIN, "Suspicious DNS TTL");
}

#ifdef DNS_DEBUG
NDPI_LOG_DBG2(ndpi_struct, "rsp_ttl=%u\n", rsp_ttl);
NDPI_LOG_DBG2(ndpi_struct, "responsetype=%u\n", responsetype);
#endif

if((responsetype == 0x01 /* A */) || (responsetype == 0x1c /* AAAA */)) {
  ;
}
