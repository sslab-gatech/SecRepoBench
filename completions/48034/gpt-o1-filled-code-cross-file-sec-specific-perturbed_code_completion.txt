if((x + 8) <= packet->payload_packet_len) {
  /* Read DNS response type */
  responsetype = get16(&x, packet->payload);

  /* Skip the class field (2 bytes) */
  (void)get16(&x, packet->payload);

  /* Read the TTL (4 bytes) */
  rsp_ttl = get32(&x, packet->payload);

  /* Check if TTL is zero - mark as suspicious */
  if(rsp_ttl == 0) {
    ndpi_set_risk(ndpi_struct, flow, NDPI_SUSPICIOUS_DGA_DOMAIN, "DNS TTL is zero");
  }

#ifdef DNS_DEBUG
  NDPI_LOG_DBG2(ndpi_struct, "DNS ans: type=%u TTL=%u\n", responsetype, rsp_ttl);
  printf("[DNS] [answer] type=%u, TTL=%u\n", responsetype, rsp_ttl);
#endif
}
