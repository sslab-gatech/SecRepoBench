responsetype = get16(&x, packet->payload);
rsp_ttl = get32(&x, packet->payload);

if (rsp_ttl == 0) {
    ndpi_set_risk(ndpi_struct, flow, NDPI_DNS_ZERO_TTL, "Suspicious DNS response with zero TTL");
}

#ifdef DNS_DEBUG
NDPI_LOG_DBG2(ndpi_struct, "TTL=%u, response_type=%u\n", rsp_ttl, responsetype);
#endif
