// <MASK>
int i;
for (i = 0; i < FF_ARRAY_ELEMS(mjpegdecodecontext->vlcs[0]) + FF_ARRAY_ELEMS(mjpegdecodecontext->vlcs[1]); i++) {
    int class = (i < FF_ARRAY_ELEMS(mjpegdecodecontext->vlcs[0])) ? 0 : 1;
    int index = (i < FF_ARRAY_ELEMS(mjpegdecodecontext->vlcs[0])) ? i : i - FF_ARRAY_ELEMS(mjpegdecodecontext->vlcs[0]);
    int bits = ht[i].bits;
    uint8_t *val = ht[i].values;

    build_vlc(&mjpegdecodecontext->vlcs[class][index], bits, val, 1, 0, 0);

    if (class < 2) {
        mjpegdecodecontext->raw_huffman_lengths[class][index] = bits;
        mjpegdecodecontext->raw_huffman_values[class][index] = val;
    }
}
