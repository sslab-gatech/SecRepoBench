static const struct {
        int class, index, bits[17], values[256];
    } ht[] = {
        { 0, 0, ff_mjpeg_bits_dc_luminance,   ff_mjpeg_val_dc },
        { 0, 1, ff_mjpeg_bits_dc_chrominance, ff_mjpeg_val_dc },
        { 1, 0, ff_mjpeg_bits_ac_luminance,   ff_mjpeg_val_ac },
        { 1, 1, ff_mjpeg_bits_ac_chrominance, ff_mjpeg_val_ac },
    };
    int i, j, ret;

    for (i = 0; i < FF_ARRAY_ELEMS(ht); i++) {
        if ((ret = build_vlc(&mjpegdecodecontext->vlcs[ht[i].class][ht[i].index],
                             ht[i].bits, ht[i].values, 1 << 9, 1, ht[i].class > 0)) < 0)
            return ret;
        if (ht[i].class < 2) {
            for (j = 0; j < 16; j++)
                mjpegdecodecontext->raw_huffman_lengths[ht[i].class][ht[i].index][j] = ht[i].bits[1 + j];
            for (j = 0; j < 256; j++)
                mjpegdecodecontext->raw_huffman_values[ht[i].class][ht[i].index][j] = ht[i].values[j];
        }
    }
