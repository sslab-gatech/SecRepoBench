int i;
    static const HuffmanTable ht[12] = {
        { 0, 0, 0x0001, 0x00000000 },
        { 0, 1, 0x0001, 0x00000000 },
        { 1, 0, 0x0001, 0x00000000 },
        { 1, 1, 0x0001, 0x00000000 },
        { 2, 0, 0x0001, 0x00000000 },
        { 2, 1, 0x0001, 0x00000000 },
        { 0, 0, 0x0001, 0x00000000 },
        { 0, 1, 0x0001, 0x00000000 },
        { 1, 0, 0x0001, 0x00000000 },
        { 1, 1, 0x0001, 0x00000000 },
        { 2, 0, 0x0001, 0x00000000 },
        { 2, 1, 0x0001, 0x00000000 }
    };
    for (i = 0; i < 12; i++) {
        int ret = build_vlc(&mjpegdecodecontext->vlcs[0][i].table, 16, &ht[i].bits, 1, &ht[i].values, 4, 1);
        if (ret < 0)
            return ret;
        if (ht[i].class < 2) {
            mjpegdecodecontext->vlcs[0][i].length = ht[i].bits;
            mjpegdecodecontext->vlcs[0][i].value = ht[i].values;
        }
    }
