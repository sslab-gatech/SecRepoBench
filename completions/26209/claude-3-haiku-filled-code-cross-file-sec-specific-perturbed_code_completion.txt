// <MASK>
int i;
for (i = 0; i < 4; i++) {
    int class = ht[i].class;
    int index = ht[i].index;
    int bits  = ht[i].bits;
    int huffval = ht[i].huffval;

    if (class < 2) {
        s->huff_size_dc[class][index] = bits;
        s->huff_code_dc[class][index] = build_vlc(&s->vlcs[class][index], bits, huffval);
    } else {
        s->huff_size_ac[class-2][index] = bits;
        s->huff_code_ac[class-2][index] = build_vlc(&s->vlcs[class][index], bits, huffval);
    }
}
return 0;
