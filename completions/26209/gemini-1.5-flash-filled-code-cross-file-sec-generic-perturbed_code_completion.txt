int i;
    static const HuffmanTable ht[12] = {
        { 0, 0, 12, 0x0000000000000000ULL },
        { 1, 0, 12, 0x0000000000000000ULL },
        { 0, 1, 12, 0x0000000000000000ULL },
        { 1, 1, 12, 0x0000000000000000ULL },
        { 0, 0, 16, 0x000102030405060708090a0b0c0d0e0fULL },
        { 1, 0, 16, 0x000102030405060708090a0b0c0d0e0fULL },
        { 0, 1, 16, 0x000102030405060708090a0b0c0d0e0fULL },
        { 1, 1, 16, 0x000102030405060708090a0b0c0d0e0fULL },
        { 0, 0, 16, 0x000102030405060708090a0b0c0d0e0fULL },
        { 1, 0, 16, 0x000102030405060708090a0b0c0d0e0fULL },
        { 0, 1, 16, 0x000102030405060708090a0b0c0d0e0fULL },
        { 1, 1, 16, 0x000102030405060708090a0b0c0d0e0fULL }
    };
    for (i = 0; i < 12; i++) {
        int ret = build_vlc(&mjpegdecodecontext->vlcs[0][i].table, ht[i].bits, 1 << ht[i].bits,
                            ht[i].values, 1, 1, 0);
        if (ret < 0)
            return ret;
        if (ht[i].class < 2) {
            mjpegdecodecontext->vlcs[0][i].length = ht[i].bits;
            mjpegdecodecontext->vlcs[0][i].value = ht[i].values;
        }
    }
