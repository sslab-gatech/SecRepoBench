{
    static const struct {
        int class;
        int index;
        const uint8_t *bits;
        const uint8_t *vals;
    } ht[4] = {
        { 0, 0, default_dc_luminance_bits,    default_dc_luminance_vals },
        { 0, 1, default_dc_chrominance_bits,    default_dc_chrominance_vals },
        { 1, 0, default_ac_luminance_bits,      default_ac_luminance_vals   },
        { 1, 1, default_ac_chrominance_bits,      default_ac_chrominance_vals }
    };
    int i, ret;
    for (i = 0; i < 4; i++) {
        ret = build_vlc(&mjpegdecodecontext->vlcs[ht[i].class][ht[i].index],
                        VLC_BITS, 257,
                        ht[i].bits, 1, 1,
                        ht[i].vals, 1, 1,
                        0, 0);
        if (ret < 0)
            return ret;
        if (ht[i].class < 2) {
            memcpy(mjpegdecodecontext->huff_raw_bits[ht[i].index],
                   ht[i].bits, sizeof(mjpegdecodecontext->huff_raw_bits[ht[i].index]));
            memcpy(mjpegdecodecontext->huff_raw_val[ht[i].index],
                   ht[i].vals, sizeof(mjpegdecodecontext->huff_raw_val[ht[i].index]));
        }
    }
}
