static int init_default_huffman_tables(MJpegDecodeContext *mjpegdecodecontext)
{
    int i;

    for (i = 0; i < 4; i++) {
        build_vlc(&mjpegdecodecontext->vlcs[0][i], ht[i].bits, 16,
                  ht[i].huffval, sizeof(ht[i].huffval), 1, 0, 0);
        if (ht[i].class < 2) {
            mjpegdecodecontext->huff_size_dc[i] = ht[i].bits;
            mjpegdecodecontext->huff_code_dc[i] = ht[i].huffval;
        }
    }

    for (i = 0; i < 4; i++) {
        build_vlc(&mjpegdecodecontext->vlcs[1][i], ht[i+4].bits, 16,
                  ht[i+4].huffval, sizeof(ht[i+4].huffval), 1, 0, 0);
        if (ht[i+4].class < 2) {
            mjpegdecodecontext->huff_size_ac[i] = ht[i+4].bits;
            mjpegdecodecontext->huff_code_ac[i] = ht[i+4].huffval;
        }
    }

    return 0;
}
