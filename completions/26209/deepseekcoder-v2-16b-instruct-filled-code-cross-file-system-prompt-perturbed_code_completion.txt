// Initialize default Huffman tables for MJPEG decoding. This involves setting
    // up a static structure `ht` which contains class, index, bits, and values
    // for both DC and AC coefficients of luminance and chrominance. The function
    // iterates over the elements of the `ht` array, building VLC tables using
    // these parameters. The tables are built using the `build_vlc` function. If
    // the class is less than 2, it also populates the raw Huffman length and
    // value arrays with the corresponding bits and values from the `ht` entries.
    for (int i = 0; i < 4; i++) {
        for (int j = 0; j < 16; j++) {
            mjpegdecodecontext->ht[i][j].class = get_bits(&s->gb, 4);
            mjpegdecodecontext->ht[i][j].index = get_bits(&s->gb, 4);
            mjpegdecodecontext->ht[i][j].bits = get_bits(&s->gb, 4);
            mjpegdecodecontext->ht[i][j].values = get_bits(&s->gb, 8);

            if (mjpegdecodecontext->ht[i][j].class < 2) {
                mjpegdecodecontext->raw_huff_len[i][j] = mjpegdecodecontext->ht[i][j].bits;
                mjpegdecodecontext->raw_huff_val[i][j] = mjpegdecodecontext->ht[i][j].values;
            }
        }
    }

    // Build VLC tables for DC and AC coefficients
    for (int i = 0; i < 4; i++) {
        build_vlc(&mjpegdecodecontext->ht[i], &mjpegdecodecontext->vlc[i], 0);
    }
