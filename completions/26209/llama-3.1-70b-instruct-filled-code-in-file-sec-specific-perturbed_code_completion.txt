for (i = 0; i < 4; i++) {
    for (j = 0; j < 16; j++) {
        s->raw_huffman_lengths[0][i][j] = ff_mjpeg_default_luminance_lengths[j];
        s->raw_huffman_values[0][i][j] = ff_mjpeg_default_luminance_values[j];
    }
    for (j = 0; j < 16; j++) {
        s->raw_huffman_lengths[1][i][j] = ff_mjpeg_default_chrominance_lengths[j];
        s->raw_huffman_values[1][i][j] = ff_mjpeg_default_chrominance_values[j];
    }
    if (build_vlc(&s->vlcs[0][i], ff_mjpeg_default_luminance_lengths,
                  ff_mjpeg_default_luminance_values, 162, 1, 0) < 0)
        return -1;
    if (build_vlc(&s->vlcs[1][i], ff_mjpeg_default_chrominance_lengths,
                  ff_mjpeg_default_chrominance_values, 162, 1, 0) < 0)
        return -1;
}
