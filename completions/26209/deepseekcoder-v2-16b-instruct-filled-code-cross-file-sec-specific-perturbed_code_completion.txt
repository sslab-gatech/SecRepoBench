// Initialize default Huffman tables for MJPEG decoding. This involves setting
    // up a static structure `ht` which contains class, index, bits, and values
    // for both DC and AC coefficients of luminance and chrominance. The function
    // iterates over the elements of the `ht` array, building VLC tables using
    // these parameters. The tables are built using the `build_vlc` function. If
    // the class is less than 2, it also populates the raw Huffman length and
    // value arrays with the corresponding bits and values from the `ht` entries.
    // <MASK>

    // Initialize Huffman tables
    for (int i = 0; i < FF_ARRAY_ELEMS(mjpeg_huff_class_info); i++) {
        int class = mjpeg_huff_class_info[i].class;
        if (class < 2) {
            int index = mjpeg_huff_class_info[i].index;
            int bits = mjpeg_huff_class_info[i].bits;
            int values = mjpeg_huff_class_info[i].values;

            // Populate raw Huffman length and value arrays
            mjpeg_huff_length[index] = bits;
            mjpeg_huff_value[index] = values;
        }

        // Build VLC tables
        build_vlc(&s->vlcs[0][i], mjpeg_huff_class_info[i].bits, mjpeg_huff_class_info[i].values, mjpeg_huff_class_info[i].class);
        build_vlc(&s->vlcs[1][i], mjpeg_huff_class_info[i].bits, mjpeg_huff_class_info[i].values, mjpeg_huff_class_info[i].class);
    }

    // Initialize VLC tables for AC coefficients
    for (int i = 0; i < FF_ARRAY_ELEMS(mjpeg_huff_ac_info); i++) {
        build_vlc(&s->vlcs[2][i], mjpeg_huff_ac_info[i].bits, mjpeg_huff_ac_info[i].values, mjpeg_huff_ac_info[i].class);
