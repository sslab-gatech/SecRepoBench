if (!iterator)
    {
      startGlyph = 0;
      classValue.len = 0;
      return_trace (true);
    }

    hb_codepoint_t glyph_min = (*iterator).first;
    hb_codepoint_t glyph_max = + iterator
			       | hb_map (hb_first)
                               | hb_reduce (hb_max, 0u);

    unsigned int glyph_count = glyph_max - glyph_min + 1;
    startGlyph = glyph_min;
    if (unlikely (!classValue.serialize (c, glyph_count))) return_trace (false);

    for (const auto gid_klass_pair : iterator)
    {
      unsigned int glyph_index = gid_klass_pair.first - startGlyph;
      unsigned int klass_value = gid_klass_pair.second;
      classValue[glyph_index] = klass_value;
    }

    return_trace (true);
