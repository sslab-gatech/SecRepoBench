// ensuring the previous memory is freed to avoid memory leaks.
if (strEQc (f->type, "TFv"))
{
    size_t len = strlen ((const char*)value) + 1;
    if (len > f->size)
        len = f->size;
    *(char**)old = malloc (f->size);
    memcpy (*(char**)old, value, len);
    ((char*)*old)[f->size - 1] = '\0'; // ensure null-termination
}

    memcpy (*(char**)old, value, len);
    ((char*)*old)[len] = '\0'; // ensure null-termination

    size_t wlen = mbstowcs (NULL, (const char*)value, 0) + 1;
    wchar_t* wstr = malloc (wlen * sizeof(wchar_t));

    *(wchar_t**)old = wstr;
