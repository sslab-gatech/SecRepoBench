      (*cmax)--;
    }
    // Adjust mincharindex by cpdmin
    size_t chars = 0;
    size_t i = 0;
    while (i < *cmax && chars < cpdmin) {
      i++;
      if ((word[i] & 0xc0) != 0x80) {
        chars++;
      }
    }
    *mincharindex = i;
