void AffixMgr::setcminmax(size_t* mincharindex, size_t* cmax, const char* word, size_t len) {
  if (utf8) {
    *mincharindex = 0;
    *cmax = len;
    for (size_t i = 0; i < cpdmin; ++i) {
      if (*mincharindex < len) {
        do {
          ++(*mincharindex);
        } while ((*mincharindex < len) && ((word[*mincharindex] & 0xc0) == 0x80));
      }
    }
    while ((*cmax > 0) && ((word[*cmax - 1] & 0xc0) == 0x80)) {
      --(*cmax);
    }
  } else {
    *mincharindex = cpdmin;
    *cmax = len - cpdmin + 1;
  }
}
