import blosc2

# Create a new (empty) frame
frame = blosc2.frame_new("path/to/frame.b2frame")

# Free memory from a frame.
blosc2.frame_free(frame)

# Get a super-chunk out of a frame
schunk = blosc2.frame_to_schunk(frame, copy=True)

# Free memory from a super-chunk.
blosc2.schunk_free(schunk)

# Append an existing chunk into a frame.
blosc2.frame_append_chunk(frame, chunk, schunk)

# Insert a chunk into a frame at a specific position.
blosc2.frame_insert_chunk(frame, nchunk, chunk, schunk)

# Update a chunk in a frame at a specific position.
blosc2.frame_update_chunk(frame, nchunk, chunk, schunk)

# Reorder the offsets of chunks in a frame.
blosc2.frame_reorder_offsets(frame, offsets_order, schunk)

# Decompress and return a chunk that is part of a frame.
blosc2.frame_decompress_chunk(dctx, frame, nchunk, dest, nbytes)
