// Retrieve the offsets chunk for this frame
  int32_t coffsets_cbytes = 0;
  uint8_t *coffsets = get_coffsets(frame, headerlength, cbytes, &coffsets_cbytes);
  if (coffsets == NULL) {
    BLOSC_TRACE_ERROR("Cannot get the offsets for the frame.");
    return BLOSC2_ERROR_DATA;
  }
  if (coffsets_cbytes == 0) {
    coffsets_cbytes = (int32_t)cbytes;
  }

  // Retrieve the number of chunks from the frame
  int32_t chunksize, nchunks;
  int64_t frame_len, nbytes_, cbytes_;
  int rc = get_header_info(frame, NULL, &frame_len, &nbytes_, &cbytes_, &chunksize, &nchunks,
                           NULL, NULL, NULL, NULL, NULL);
  if (rc < 0) {
    BLOSC_TRACE_ERROR("Unable to get meta info from frame.");
    return rc;
  }
  if (nchunk >= nchunks) {
    BLOSC_TRACE_ERROR("nchunk ('%d') exceeds the number of chunks ('%d') in frame.", nchunk, nchunks);
    return BLOSC2_ERROR_INVALID_PARAM;
  }

  // Decompress and obtain the chunk offset
  int64_t *offsets_buf = (int64_t *)malloc((size_t)nchunks * sizeof(int64_t));
  if (offsets_buf == NULL) {
    BLOSC_TRACE_ERROR("Cannot allocate memory for offsets buffer.");
    return BLOSC2_ERROR_MEMORY_ALLOC;
  }

  blosc2_dparams dparams = BLOSC2_DPARAMS_DEFAULTS;
  blosc2_context *dctx = blosc2_create_dctx(dparams);
  int32_t dsize = blosc2_decompress_ctx(dctx, coffsets, coffsets_cbytes,
                                        offsets_buf, (size_t)nchunks * sizeof(int64_t));
  blosc2_free_ctx(dctx);
  if (dsize < 0) {
    free(offsets_buf);
    BLOSC_TRACE_ERROR("Cannot decompress offsets chunk.");
    return BLOSC2_ERROR_DATA;
  }

  // Store the offset for the requested chunk
  *offset = offsets_buf[nchunk];
  free(offsets_buf);

  return 0;
