int get_coffset(blosc2_frame_s* frame, int32_t headerlength, int64_t cbytes, int32_t nchunk, int64_t *offset) {
  // Get the offset to nchunk
  int32_t coffsets_cbytes = 0;
  uint8_t *coffsets = get_coffsets(frame, headerlength, cbytes, &coffsets_cbytes);
  if (coffsets == NULL) {
    BLOSC_TRACE_ERROR("Unable to get the offsets for the frame.");
    return BLOSC2_ERROR_DATA;
  }

  // Decompress offsets
  blosc2_dparams off_dparams = BLOSC2_DPARAMS_DEFAULTS;
  blosc2_context *dctx = blosc2_create_dctx(off_dparams);
  int64_t* offsets = malloc(nchunk * sizeof(int64_t) + sizeof(int64_t));
  if (offsets == NULL) {
    BLOSC_TRACE_ERROR("Unable to allocate memory for offsets.");
    return BLOSC2_ERROR_MEMORY_ALLOC;
  }
  int32_t off_nbytes = blosc2_decompress_ctx(dctx, coffsets, coffsets_cbytes, offsets, (nchunk + 1) * sizeof(int64_t));
  blosc2_free_ctx(dctx);
  if (off_nbytes < 0) {
    BLOSC_TRACE_ERROR("Unable to decompress the offsets chunk.");
    free(offsets);
    return off_nbytes;
  }

  *offset = offsets[nchunk];
  free(offsets);

  return 0;
}
