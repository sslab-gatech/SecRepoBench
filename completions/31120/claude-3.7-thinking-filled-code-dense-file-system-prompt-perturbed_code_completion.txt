int get_coffset(blosc2_frame_s* frame, int32_t headerlength, int64_t cbytes, int32_t nchunk, int64_t *offset) {
  // Get the offset to nchunk
  int32_t coffsets_cbytes = 0;
  uint8_t *coffsets = get_coffsets(frame, headerlength, cbytes, &coffsets_cbytes);
  if (coffsets == NULL) {
    BLOSC_TRACE_ERROR("Cannot get the offsets for the frame.");
    return BLOSC2_ERROR_DATA;
  }

  // We need to decompress at least enough to get the nchunk offset
  int32_t min_nbytes = (nchunk + 1) * sizeof(int64_t);
  
  // Allocate memory for the offsets
  int64_t* offsets = (int64_t *) malloc((size_t)min_nbytes);
  if (offsets == NULL) {
    BLOSC_TRACE_ERROR("Cannot allocate memory for the offsets.");
    return BLOSC2_ERROR_MEMORY_ALLOC;
  }

  // Decompress offsets
  blosc2_dparams off_dparams = BLOSC2_DPARAMS_DEFAULTS;
  blosc2_context *dctx = blosc2_create_dctx(off_dparams);
  int32_t prev_nbytes = blosc2_decompress_ctx(dctx, coffsets, coffsets_cbytes, offsets, min_nbytes);
  blosc2_free_ctx(dctx);
  
  if (prev_nbytes < 0) {
    free(offsets);
    BLOSC_TRACE_ERROR("Cannot decompress the offsets chunk.");
    return prev_nbytes;
  }

  // Check if we got enough data
  if (prev_nbytes < min_nbytes) {
    free(offsets);
    BLOSC_TRACE_ERROR("Not enough offset data decompressed.");
    return BLOSC2_ERROR_DATA;
  }

  // Get the specific offset for the requested chunk
  *offset = offsets[nchunk];
  free(offsets);

  return 0;
}
