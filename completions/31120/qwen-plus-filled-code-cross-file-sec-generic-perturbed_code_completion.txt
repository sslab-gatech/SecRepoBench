int32_t coffsets_cbytes = 0;
uint8_t *coffsets = get_coffsets(frame, headerlength, cbytes, &coffsets_cbytes);
if (coffsets == NULL) {
  BLOSC_TRACE_ERROR("Unable to retrieve offsets for chunks in the frame.");
  return BLOSC2_ERROR_DATA;
}

// Decompress offsets
blosc2_dparams off_dparams = BLOSC2_DPARAMS_DEFAULTS;
blosc2_context *dctx = blosc2_create_dctx(off_dparams);
int32_t prev_nbytes = blosc2_decompress_ctx(dctx, coffsets, coffsets_cbytes, NULL, nchunk * sizeof(int64_t));
blosc2_free_ctx(dctx);
if (prev_nbytes < 0) {
  BLOSC_TRACE_ERROR("Unable to decompress offsets chunk.");
  return BLOSC2_ERROR_DECOMPRESSION;
}

// Obtain the offset for the specified chunk index
*offset = ((int64_t *)coffsets)[nchunk];
if (*offset < 0) {
  BLOSC_TRACE_ERROR("Invalid offset retrieved for the specified chunk.");
  return BLOSC2_ERROR_INVALID_OFFSET;
}

return 0;
