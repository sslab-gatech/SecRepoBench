int get_coffset(blosc2_frame_s* frame, int32_t headerlength, int64_t cbytes, int32_t nchunk, int64_t *offset) {
  int32_t off_nbytes = (nchunk + 1) * sizeof(int64_t);
  int64_t* offsets = malloc((size_t)off_nbytes);
  if (offsets == NULL) {
    BLOSC_TRACE_ERROR("Cannot allocate memory for offsets.");
    return BLOSC2_ERROR_MEMORY_ALLOC;
  }

  int32_t coffsets_cbytes = 0;
  uint8_t *coffsets = get_coffsets(frame, headerlength, cbytes, &coffsets_cbytes);
  if (coffsets == NULL) {
    BLOSC_TRACE_ERROR("Cannot get the offsets for the frame.");
    free(offsets);
    return BLOSC2_ERROR_DATA;
  }

  blosc2_dparams off_dparams = BLOSC2_DPARAMS_DEFAULTS;
  blosc2_context *dctx = blosc2_create_dctx(off_dparams);
  if (dctx == NULL) {
    BLOSC_TRACE_ERROR("Cannot create decompression context.");
    free(offsets);
    return BLOSC2_ERROR_CONTEXT;
  }

  int32_t off_nbytes_actual = blosc2_decompress_ctx(dctx, coffsets, coffsets_cbytes, offsets, off_nbytes);
  blosc2_free_ctx(dctx);

  if (off_nbytes_actual < 0) {
    BLOSC_TRACE_ERROR("Cannot decompress the offsets chunk.");
    free(offsets);
    return off_nbytes_actual;
  }

  if (nchunk >= off_nbytes_actual / sizeof(int64_t)) {
    BLOSC_TRACE_ERROR("nchunk is out of bounds.");
    free(offsets);
    return BLOSC2_ERROR_INVALID_PARAM;
  }

  *offset = offsets[nchunk];
  free(offsets);

  return 0;
}
