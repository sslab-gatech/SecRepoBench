// Multiply the matrices to get the result
                     _cmsMAT3per(&res, (cmsMAT3*)Matrix2->Double, (cmsMAT3*)Matrix1->Double);

                     // Check if the resulting matrix is an identity matrix
                     if (_cmsMAT3isIdentity(&res)) {
                         isIdentityMatrix = TRUE; // We can optimize away the matrix
                     }
