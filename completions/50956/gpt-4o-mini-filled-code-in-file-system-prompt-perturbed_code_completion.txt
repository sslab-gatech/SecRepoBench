_cmsStageMatrixData* Data = (_cmsStageMatrixData*)cmsStageData(Matrix1);

                     // Copy the matrix to our result
                     memcpy(&res, Data->Double, sizeof(res));

                     // Preserve the Offset (may be NULL as a zero offset)
                     Offset = Data->Offset;

                     if (_cmsMAT3isIdentity(&res) && Offset == NULL) {

                            // We can get rid of full matrix
                            isIdentityMatrix = TRUE;
                     }
                     else {
                            // Multiply the matrices to combine them into a single result matrix
                            _cmsStageMatrixData* Data2 = (_cmsStageMatrixData*)cmsStageData(Matrix2);
                            cmsMAT3 res2;
                            memcpy(&res2, Data2->Double, sizeof(res2));
                            _cmsMAT3mul(&res, &res, &res2);
                            Offset = Data->Offset; // Use the offset from the first matrix
                     }
