{
    _cmsStageMatrixData* Data1 = (_cmsStageMatrixData*)cmsStageData(Matrix1);
    _cmsStageMatrixData* Data2 = (_cmsStageMatrixData*)cmsStageData(Matrix2);

    // Ensure the input offset of the first matrix is zero
    if (Data1->Offset != NULL) return FALSE;

    // Multiply both matrices to get the result
    _cmsMAT3per(&res, (cmsMAT3*)Data2->Double, (cmsMAT3*)Data1->Double);

    // Check if the resulting matrix is an identity matrix and if the offset is null
    if (_cmsMAT3isIdentity(&res) && Data2->Offset == NULL) {
        // We can get rid of the full matrix
        isIdentityMatrix = TRUE;
    }

    // Preserve the offset from the second matrix
    Offset = Data2->Offset;
}
