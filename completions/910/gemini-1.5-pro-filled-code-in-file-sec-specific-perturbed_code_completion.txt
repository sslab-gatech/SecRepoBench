if (!io ->Seek(io, Offset)) goto Error;

// Get tag descriptor
TagDescriptor = _cmsGetTagDescriptor(Icc->ContextID, tagSignature);
if (TagDescriptor == NULL) goto Error;

// Read base tag type, usually is the tag type
if (!_cmsReadTypeBase(io, &BaseType)) goto Error;

// Does the tag support this type?
if (!IsTypeSupported(TagDescriptor, BaseType)) {

    char String[5];
    _cmsTagSignature2String(String, tagSignature);
    cmsSignalError(Icc ->ContextID, cmsERROR_UNKNOWN_EXTENSION, "Unsupported tag type '%x' for tag '%s'", BaseType, String);
    goto Error;
}

TagSize -= sizeof(cmsTagTypeSignature);       // Discount type base already read
