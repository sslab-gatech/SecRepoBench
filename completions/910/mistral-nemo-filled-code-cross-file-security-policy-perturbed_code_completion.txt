// Check if the tag size is at least 8 bytes
    if (TagSize < 8) {
        cmsSignalError(Icc ->ContextID, cmsERROR_INVALID_TAG_SIZE, "Tag size is too small (%d bytes)", TagSize);
        goto Error;
    }

    // Seek to the tag's location in the IO handler.
    if (!Icc ->IOhandler ->Seek(Icc ->IOhandler, Offset)) goto Error;

    // Check if the tag is supported.
    if (!IsTypeSupported(TagDescriptor, BaseType)) goto Error;

    // Change the tag size to account for the tag's base type header.
    TagSize -= 8;  // Subtract the size of the base type header
