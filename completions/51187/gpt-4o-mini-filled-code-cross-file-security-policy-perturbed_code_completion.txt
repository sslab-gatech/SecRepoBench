std::string* word_copy = new std::string(in_word);
std::string* desc_copy = nullptr;

if (inputdesc) {
    desc_copy = new std::string(*inputdesc);
}

if (word_copy == nullptr || (inputdesc && desc_copy == nullptr)) {
    delete desc_copy;
    delete word_copy;
    return 1;
}

// Remove ignored characters if necessary
if (contains_ignored_characters(*word_copy)) {
    *word_copy = remove_ignored_characters(*word_copy);
}

// Reverse the word if complex prefixes are enabled
if (complex_prefixes_enabled) {
    std::reverse(word_copy->begin(), word_copy->end());
    if (desc_copy) {
        std::reverse(desc_copy->begin(), desc_copy->end());
    }
}
