const std::string* desc = inputdesc;
std::string* word_copy = nullptr;
std::string* desc_copy = nullptr;

if (has_ignchars || complex_prefix) {
    word_copy = new std::string(in_word);
    if (has_ignchars) {
        remove_ignored_chars(*word_copy);
    }
    if (complex_prefix) {
        reverse_string(*word_copy);
    }
}

if (desc && !use_alias) {
    desc_copy = new std::string(*desc);
    if (complex_prefix) {
        reverse_string(*desc_copy);
    }
}

const std::string* word = word_copy ? word_copy : &in_word;
