std::string* word_copy = nullptr;
std::string* desc_copy = nullptr;

if (!ignorechars.empty() || complexprefixes) {
  word_copy = new std::string(in_word);
  if (!ignorechars.empty()) {
    if (utf8) {
      remove_ignored_chars_utf(*word_copy, ignorechars_utf16);
    } else {
      remove_ignored_chars(*word_copy, ignorechars);
    }
  }
  if (complexprefixes) {
    if (utf8) {
      reverseword_utf(*word_copy);
    } else {
      reverseword(*word_copy);
    }
  }
  word = word_copy;
}

if (desc && complexprefixes && !aliasm.empty()) {
  desc_copy = new std::string(*desc);
  if (utf8) {
    reverseword_utf(*desc_copy);
  } else {
    reverseword(*desc_copy);
  }
  desc = desc_copy;
}
