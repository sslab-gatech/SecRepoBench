std::string* word_copy = new std::string(in_word);
std::string* desc_copy = nullptr;

if (onlyupcase) {
    // If onlyupcase is true, we should not modify the original word.
    // Just create a copy of the original word.
    word_copy = new std::string(in_word);
} else {
    // Remove ignored characters if necessary
    std::string cleaned_word = remove_ignored_characters(*word_copy);
    if (complex_prefixes_enabled) {
        std::reverse(cleaned_word.begin(), cleaned_word.end());
    }
    word_copy = new std::string(cleaned_word);
}

if (inputdesc) {
    desc_copy = new std::string(*inputdesc);
    if (complex_prefixes_enabled) {
        std::reverse(*desc_copy);
    }
}
