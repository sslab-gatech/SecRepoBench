std::string* word_copy = NULL;
std::string* desc_copy = NULL;

if (ignorechars) {
  std::string word(in_word);
  clean_ignore(word, in_word);
  if (word.empty()) {
    HUNSPELL_WARNING(stderr, "error: word contains only ignored characters\n");
    return 1;
  }
  word_copy = new std::string(word);
}

if (complexprefixes) {
  if (utf8)
    reverseword_utf(*word_copy);
  else
    reverseword(*word_copy);
}

if (inputdesc &&!aliasm.empty() && complexprefixes) {
  desc_copy = new std::string(*inputdesc);
  if (utf8)
    reverseword_utf(*desc_copy);
  else
    reverseword(*desc_copy);
}

std::string* word = (word_copy)? word_copy : const_cast<std::string*>(&in_word);
const std::string* desc = (desc_copy)? desc_copy : inputdesc;
