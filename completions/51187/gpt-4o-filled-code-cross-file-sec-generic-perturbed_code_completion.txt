std::string* word_copy = nullptr;
  std::string* desc_copy = nullptr;
  const std::string* word = &in_word;
  const std::string* desc = inputdesc;

  // Check if the word contains ignored characters or if complex prefixes are enabled
  if (!ignorechars.empty() || complexprefixes) {
    word_copy = new std::string(in_word);
    if (!ignorechars.empty()) {
      word_copy->erase(std::remove_if(word_copy->begin(), word_copy->end(),
                                      [this](char c) { return ignorechars.find(c) != std::string::npos; }),
                       word_copy->end());
    }
    if (complexprefixes) {
      std::reverse(word_copy->begin(), word_copy->end());
    }
    word = word_copy;
  }

  // Handle description if provided
  if (desc && !aliasm.empty()) {
    desc_copy = new std::string(*desc);
    if (complexprefixes) {
      std::reverse(desc_copy->begin(), desc_copy->end());
    }
    desc = desc_copy;
