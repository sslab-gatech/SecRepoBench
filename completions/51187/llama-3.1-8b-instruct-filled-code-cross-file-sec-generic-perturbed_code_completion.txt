// If the word contains ignored characters or complex prefixes are being handled:
//   - Create a new copy of the word and remove ignored characters if necessary.
//   - Reverse the word if complex prefixes are enabled.
//   - If a description is provided and aliasing is not used, create a copy and reverse it if complex prefixes are enabled.
// Update the word and description pointers to point to the new copies.
std::string word_copy = *word;
std::string desc_copy;
if (desc) {
  desc_copy = *desc;
}
if (ignore_chars.size() > 0) {
  std::string new_word;
  for (char c : word_copy) {
    if (ignore_chars.find(c) == std::string::npos) {
      new_word += c;
    }
  }
  word_copy = new_word;
}
if (complex_prefixes) {
  std::string new_word = word_copy;
  std::reverse(new_word.begin(), new_word.end());
  word_copy = new_word;
}
if (desc_copy.size() > 0 &&!aliasing) {
  if (complex_prefixes) {
    std::string new_desc;
    for (char c : desc_copy) {
      new_desc += c;
    }
    std::reverse(new_desc.begin(), new_desc.end());
    desc_copy = new_desc;
  }
}
word = &word_copy;
if (desc) {
  desc = &desc_copy;
}
