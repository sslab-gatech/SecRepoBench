std::string* word_copy = nullptr;
  const std::string* word = &in_word;
  std::string* desc_copy = nullptr;
  const std::string* desc = inputdesc;

  if (!ignored_chars.empty()) {
    std::string filtered_word;
    filtered_word.reserve(in_word.size());
    for (char c : in_word) {
      if (ignored_chars.find(c) == std::string::npos)
        filtered_word += c;
    }
    if (filtered_word.size() != in_word.size()) {
      word_copy = new std::string(std::move(filtered_word));
      word = word_copy;
    }
  }

  if (complexprefixes) {
    if (!word_copy) {
      word_copy = new std::string(in_word);
      word = word_copy;
    }
    std::reverse(word_copy->begin(), word_copy->end());
  }

  if (inputdesc && aliasm.empty()) {
    desc_copy = new std::string(*inputdesc);
    if (complexprefixes)
      std::reverse(desc_copy->begin(), desc_copy->end());
    desc = desc_copy;
