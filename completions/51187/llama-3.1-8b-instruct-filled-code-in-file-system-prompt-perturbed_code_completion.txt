    std::vector<w_char> w;

    reverseword_utf(w);
    u16_u8(word_copy, w);

    reverseword(word_copy);

  if (desc_copy.size() > 0) {

      reverseword_utf(w);
      u16_u8(desc_copy, w);
    } else {
      reverseword(desc_copy);
    }
  }
