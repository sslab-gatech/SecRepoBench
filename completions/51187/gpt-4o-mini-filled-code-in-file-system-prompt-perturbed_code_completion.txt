std::string* word_copy = new std::string(in_word);
  std::string* desc_copy = inputdesc ? new std::string(*inputdesc) : nullptr;

  if (onlyupcase && (captype == INITCAP)) {
    if (utf8) {
      std::vector<w_char> w;
      u8_u16(w, *word_copy);
      mkinitcap_utf(w, langnum);
      u16_u8(*word_copy, w);
    } else {
      mkinitcap(*word_copy, csconv);
    }
  }

  if (complexprefixes) {
    if (utf8) {
      reverseword_utf(*word_copy);
      if (desc_copy) reverseword_utf(*desc_copy);
    } else {
      reverseword(*word_copy);
      if (desc_copy) reverseword(*desc_copy);
    }
