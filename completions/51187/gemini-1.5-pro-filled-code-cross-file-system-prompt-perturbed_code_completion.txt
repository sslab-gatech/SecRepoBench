  std::string* desc = const_cast<std::string*>(inputdesc);
  std::string* word_copy = NULL;
  std::string* desc_copy = NULL;

  if (ignore.size() || complexprefixes) {
    word_copy = new std::string(in_word);
    word = word_copy;
    if (ignore.size()) {
      for (size_t i = 0; i < ignore.size(); ++i) {
        word->erase(std::remove(word->begin(), word->end(), ignore[i]), word->end());
      }
    }

    if (complexprefixes)
      reverseword(word);

    if (desc && aliasm.empty()) {
      desc_copy = new std::string(*inputdesc);
      desc = desc_copy;
      if (complexprefixes)
        reverseword(desc);
