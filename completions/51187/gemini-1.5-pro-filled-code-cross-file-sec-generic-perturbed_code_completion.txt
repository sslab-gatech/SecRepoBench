  std::string* desc_copy = nullptr;

        word->erase(std::remove(word->begin(), word->end(), ignore[i]), word->end());
      }
    }

    if (complexprefixes)
      reverse_utf8(*word);

    if (desc && aliasm.empty()) {
      desc_copy = new std::string(*desc);
      desc = desc_copy;
      if (complexprefixes)
