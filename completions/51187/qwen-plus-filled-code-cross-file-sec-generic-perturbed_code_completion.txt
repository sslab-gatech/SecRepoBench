std::string* word_copy = nullptr;
std::string* desc_copy = nullptr;

if (!ignored.empty() || complexprefixes) {
    std::string new_word = in_word;
    for (char c : ignored) {
        new_word.erase(std::remove(new_word.begin(), new_word.end(), c), new_word.end());
    }
    if (complexprefixes) {
        std::reverse(new_word.begin(), new_word.end());
    }
    word_copy = new std::string(new_word);
}

if (inputdesc && !aliasm.empty()) {
    std::string new_desc = *inputdesc;
    if (complexprefixes) {
        std::reverse(new_desc.begin(), new_desc.end());
    }
    desc_copy = new std::string(new_desc);
}

const std::string* word = word_copy ? word_copy : &in_word;
