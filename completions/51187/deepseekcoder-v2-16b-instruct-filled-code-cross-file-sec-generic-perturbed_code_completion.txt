  //   - Create a new copy of the word and remove ignored characters if necessary.
  //   - Reverse the word if complex prefixes are enabled.
  //   - If a description is provided and aliasing is not used, create a copy and reverse it if complex prefixes are enabled.
  // Update the word and description pointers to point to the new copies.
  std::string word_copy = in_word;
  std::string* desc_copy = inputdesc ? new std::string(*inputdesc) : nullptr;

  if (onlyupcase) {
    // Convert the word to uppercase if onlyupcase is true
    std::transform(word_copy.begin(), word_copy.end(), word_copy.begin(), ::toupper);
    if (desc_copy) {
      std::transform(desc_copy->begin(), desc_copy->end(), desc_copy->begin(), ::toupper);
    }
  }

  // Update the word and description pointers to point to the new copies
  word = &word_copy;
  if (desc_copy) {
    desc = desc_copy;
  } else {
    desc = nullptr;
