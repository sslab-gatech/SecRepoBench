std::string* word_copy = NULL;
std::string* desc_copy = NULL;

if (pAMgr->get_complexprefixes()) {
  std::string word_reversed = in_word;
  std::reverse(word_reversed.begin(), word_reversed.end());
  word_copy = new std::string(word_reversed);
  word = word_copy;
}

if (inputdesc &&!aliasm.empty() && pAMgr->get_complexprefixes()) {
  std::string desc_reversed = *inputdesc;
  std::reverse(desc_reversed.begin(), desc_reversed.end());
  desc_copy = new std::string(desc_reversed);
  inputdesc = desc_copy;
}

if (pAMgr->get_ignore()) {
  std::string word_filtered = in_word;
  for (int i = 0; i < word_filtered.size(); i++) {
    if (TESTCHAR(pAMgr->get_ignore(), word_filtered[i]))
      word_filtered.erase(i, 1);
  }
  word_copy = new std::string(word_filtered);
  word = word_copy;
}
