  // Create dynamic copies for in-place modifications.
  word_copy = new std::string(in_word);
  word = word_copy;

  // If there are any ignored characters defined, remove them.
  // (Assuming 'ignore' is a member std::string containing the characters to ignore.)
  if (!ignore.empty()) {
    word->erase(std::remove_if(word->begin(), word->end(),
                 [this](char c) { return (ignore.find(c) != std::string::npos); }),
                word->end());
  }

  // If complex prefixes are enabled, reverse the word.
  if (complex_prefix) {
    std::reverse(word->begin(), word->end());
  }

  // Handle the description if provided.
  if (inputdesc) {
    desc_copy = new std::string(*inputdesc);
    desc = desc_copy;
    // Reverse the description if complex prefixes are enabled and aliasing is not used.
    if (complex_prefix && aliasm.empty()) {
      std::reverse(desc->begin(), desc->end());
    }
  } else {
    desc = nullptr;
  }
