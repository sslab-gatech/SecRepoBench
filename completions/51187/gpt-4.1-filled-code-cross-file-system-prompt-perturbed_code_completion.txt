  std::string* desc_copy = nullptr;
  const std::string* word = &in_word;
  const std::string* desc = inputdesc;

  // Remove ignored characters if needed
  if (!ignoredchars.empty()) {
    size_t pos = word->find_first_of(ignoredchars);
    if (pos != std::string::npos) {
      word_copy = new std::string(*word);
      word_copy->erase(std::remove_if(word_copy->begin(), word_copy->end(),
                                      [this](char c) { return ignoredchars.find(c) != std::string::npos; }),
                       word_copy->end());
      word = word_copy;
    }
  }

  if (complexprefixes) {
    if (!word_copy) {
      word_copy = new std::string(*word);
    }
    std::reverse(word_copy->begin(), word_copy->end());
