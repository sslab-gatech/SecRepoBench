if (head < 0) {
  head += arraylength;
  if (head < 0) {
    mrb_raisef(mrb, E_INDEX_ERROR, "index %i out of array", head - arraylength);
  }
}
tail = head + len;
if (tail < head) goto out_of_range;  /* check for integer overflow */
if (head > ARY_MAX_SIZE || tail > ARY_MAX_SIZE) goto out_of_range;
if (tail > arraylength) {
  tail = arraylength;
  len = tail - head;
}
