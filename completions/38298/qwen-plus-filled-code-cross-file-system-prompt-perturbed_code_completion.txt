if (head < 0) {
  head += arraylength;
  if (head < 0) {
    mrb_raisef(mrb, E_INDEX_ERROR, "index %i out of array", head - arraylength);
  }
}

tail = head + len;

if (head > ARY_MAX_SIZE - argc || tail > ARY_MAX_SIZE - argc) {
  goto out_of_range;
}

if (arraylength < len) {
  len = arraylength - head;
}
if (arraylength < tail) {
  tail = arraylength;
}
