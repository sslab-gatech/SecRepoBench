  head += arraylength;

    mrb_raisef(mrb, E_INDEX_ERROR, "index %i out of array", head - arraylength);
  }
}

  mrb_raisef(mrb, E_INDEX_ERROR, "index out of bounds");
}

if (tail > arraylength) {
  len = arraylength - head;
  tail = arraylength;
