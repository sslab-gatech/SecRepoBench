if (head < 0) {
  head += arraylength;
  if (head < 0) {
    mrb_raisef(mrb, E_INDEX_ERROR, "index %i out of array", head - arraylength);
  }
}
tail = head + len;
if (head > ARY_MAX_SIZE - len || tail > ARY_MAX_SIZE) {
  goto out_of_range;
}
if (arraylength < len) {
  len = arraylength;
}
if (tail > arraylength) {
  tail = arraylength;
}
