    head += arraylength;
    if (head < 0) {
      mrb_raisef(mrb, E_INDEX_ERROR, "index %i out of array", head - arraylength);
    }
  }
  tail = head + len;
  if (head > ARY_MAX_SIZE - len || tail < 0) {
    goto out_of_range;
  }
  if (tail > arraylength) {
    len = arraylength - head;
    tail = arraylength;
