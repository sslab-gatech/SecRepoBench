if (zend_parse_parameters(ZEND_NUM_ARGS(), "s", &serialized_data_buffer, &buf_len) == FAILURE) {
		RETURN_THROWS();
	}

	if (buf_len == 0) {
		return;
	}

	p = (const unsigned char*)serialized_data_buffer;
	s = p;

	PHP_VAR_UNSERIALIZE_INIT(var_hash);

	/* storage */
	if (*p != 'x' || *++p != ':') {
		goto outexcept;
	}
	++p;

	if (!php_var_unserialize(&flags, &p, s + buf_len, &var_hash) || Z_TYPE(flags) != IS_LONG) {
		goto outexcept;
	}

	while (*p == ';') {
		zval obj, inf;
		spl_SplObjectStorageElement *pelement, element;

		++p;
		if (!php_var_unserialize(&obj, &p, s + buf_len, &var_hash) || Z_TYPE(obj) != IS_OBJECT) {
			goto outexcept;
		}

		if (*p != ',') {
			goto outexcept;
		}
		++p;

		if (!php_var_unserialize(&inf, &p, s + buf_len, &var_hash)) {
			goto outexcept;
		}

		element.obj = Z_OBJ(obj);
		GC_ADDREF(element.obj);
		ZVAL_COPY_VALUE(&element.inf, &inf);

		if (spl_object_storage_get_hash(&key, intern, element.obj) == FAILURE) {
			zval_ptr_dtor(&inf);
			zend_object_release(element.obj);
			goto outexcept;
		}

		if (key.key) {
			pelement = zend_hash_update_mem(&intern->storage, key.key, &element, sizeof(spl_SplObjectStorageElement));
		} else {
			pelement = zend_hash_index_update_mem(&intern->storage, key.h, &element, sizeof(spl_SplObjectStorageElement));
		}
		spl_object_storage_free_hash(intern, &key);

		if (pelement) {
			ZVAL_UNDEF(&inf);
		}
	}
