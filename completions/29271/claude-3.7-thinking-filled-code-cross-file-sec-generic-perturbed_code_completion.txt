if (zend_parse_parameters(ZEND_NUM_ARGS(), "s", &serialized_data_buffer, &buf_len) == FAILURE) {
    RETURN_THROWS();
}

if (buf_len == 0) {
    return;
}

s = p = (const unsigned char*)serialized_data_buffer;
PHP_VAR_UNSERIALIZE_INIT(var_hash);

if (*p != 'x' || *++p != ':') {
    goto outexcept;
}
++p;

zval *pcount = var_tmp_var(&var_hash);
if (!php_var_unserialize(pcount, &p, s + buf_len, &var_hash) || Z_TYPE_P(pcount) != IS_LONG) {
    goto outexcept;
}

if (*p != ':') {
    goto outexcept;
}
++p;

zend_long count = Z_LVAL_P(pcount);

for (zend_long i = 0; i < count; ++i) {
    zval *obj, *inf;
    
    obj = var_tmp_var(&var_hash);
    if (!php_var_unserialize(obj, &p, s + buf_len, &var_hash) || Z_TYPE_P(obj) != IS_OBJECT) {
        goto outexcept;
    }
    
    if (*p == ';') {
        if (spl_object_storage_attach(intern, obj, NULL) == NULL) {
            goto outexcept;
        }
        ++p;
        continue;
    } else if (*p != ':') {
        goto outexcept;
    }
    ++p;
    
    inf = var_tmp_var(&var_hash);
    if (!php_var_unserialize(inf, &p, s + buf_len, &var_hash)) {
        goto outexcept;
    }
    
    if (*p != ';') {
        goto outexcept;
    }
    ++p;
    
    if (spl_object_storage_attach(intern, obj, inf) == NULL) {
        goto outexcept;
    }
}
