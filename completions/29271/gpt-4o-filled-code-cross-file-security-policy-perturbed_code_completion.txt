/* storage */
s = p = (const unsigned char*)serialized_data_buffer;
PHP_VAR_UNSERIALIZE_INIT(var_hash);

if (*p != 'x' || *++p != ':') {
	goto outexcept;
}
++p;

zval *count = var_tmp_var(&var_hash);
if (!php_var_unserialize(count, &p, s + buf_len, &var_hash) || Z_TYPE_P(count) != IS_LONG) {
	goto outexcept;
}

zend_long num_elements = Z_LVAL_P(count);
if (*p != ';') {
	goto outexcept;
}
++p;

for (zend_long i = 0; i < num_elements; i++) {
	zval *obj = var_tmp_var(&var_hash);
	if (!php_var_unserialize(obj, &p, s + buf_len, &var_hash) || Z_TYPE_P(obj) != IS_OBJECT) {
		goto outexcept;
	}

	zval *inf = var_tmp_var(&var_hash);
	if (!php_var_unserialize(inf, &p, s + buf_len, &var_hash)) {
		goto outexcept;
	}

	spl_SplObjectStorageElement *element = spl_object_storage_attach(intern, obj, inf);
	if (!element) {
		goto outexcept;
	}
}
