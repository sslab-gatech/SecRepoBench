		RETURN_THROWS();
	}

	if (buf_len == 0) {
		return;
	}

	s = p = (const unsigned char*)serialized_data_buffer;
	PHP_VAR_UNSERIALIZE_INIT(var_hash);

	if (*p != 'c' || *++p != ':') {
		goto outexcept;
	}
	++p;

	if (!php_var_unserialize(&intern->std, &p, s + buf_len, &var_hash) || Z_TYPE(intern->std) != IS_OBJECT) {
		goto outexcept;
	}

	if (*p != ';') {
		goto outexcept;
	}
	++p;

	if (*p!= 's' || *++p != ':') {
		goto outexcept;
	}
	++p;

	zend_long count;
	if (!php_var_unserialize(&count, &p, s + buf_len, &var_hash) || Z_TYPE(count) != IS_LONG) {
		goto outexcept;
	}

	if (*p != ';') {
		goto outexcept;
	}
	++p;

	for (zend_long i = 0; i < count; ++i) {
		zval *obj, *inf;
		obj = var_tmp_var(&var_hash);
		inf = var_tmp_var(&var_hash);

		if (!php_var_unserialize(obj, &p, s + buf_len, &var_hash) || Z_TYPE_P(obj) != IS_OBJECT) {
			goto outexcept;
		}

		if (*p != ';') {
			goto outexcept;
		}
		++p;

		if (!php_var_unserialize(inf, &p, s + buf_len, &var_hash)) {
			goto outexcept;
		}

		if (*p != ';') {
			goto outexcept;
		}
		++p;

		zend_hash_index_update(&intern->storage, Z_OBJ_HANDLE_P(obj), inf);
