zval *pmembers;

	if (zend_parse_parameters(ZEND_NUM_ARGS(), "s", &serialized_data_buffer, &buf_len) == FAILURE) {
		RETURN_THROWS();
	}

	if (buf_len == 0) {
		return;
	}

	s = p = (const unsigned char*)serialized_data_buffer;
	PHP_VAR_UNSERIALIZE_INIT(var_hash);

	if (*p!= 'x' || *++p != ':') {
		goto outexcept;
	}
	++p;

	intern->pos.dtor = NULL; /* reset */

	if (*p != ';') {
		zval *zv = var_tmp_var(&var_hash);
		if (!php_var_unserialize(zv, &p, s + buf_len, &var_hash)) {
			goto outexcept;
		}
		if (Z_TYPE_P(zv) != IS_LONG) {
			goto outexcept;
		}
		--p; /* for ';' */
	}
