zend_long count;
	spl_SplObjectStorageElement *element;
	zval *pmembers;
	const unsigned char *pmembers_orig;

	if (zend_parse_parameters(ZEND_NUM_ARGS(), "s", &serialized_data_buffer, &buf_len) == FAILURE) {
		RETURN_THROWS();
	}

	s = (const unsigned char*)serialized_data_buffer;
	p = s;

	PHP_VAR_UNSERIALIZE_INIT(var_hash);

	/* storage */
	if (*p != 'x' || *++p != ':') {
		goto outexcept;
	}
	++p;

	if (!php_var_unserialize(&count, &p, s + buf_len, &var_hash) || Z_TYPE(count) != IS_LONG) {
		goto outexcept;
	}

	zend_hash_init(&intern->storage, Z_LVAL(count), NULL, spl_object_storage_dtor, 0);

	while (Z_LVAL(count)--) {
		zval obj, inf;
		if (*p != 'O' || *++p != ':') {
			goto outexcept;
		}
		++p;

		if (!php_var_unserialize(&obj, &p, s + buf_len, &var_hash) || Z_TYPE(obj) != IS_OBJECT) {
			goto outexcept;
		}

		if (*p != ',') {
			goto outexcept;
		}
		++p;

		if (!php_var_unserialize(&inf, &p, s + buf_len, &var_hash)) {
			goto outexcept;
		}

		spl_object_storage_attach(intern, Z_OBJ_P(obj), &inf);
		zval_ptr_dtor(&obj);
		zval_ptr_dtor(&inf);
