/* storage */
	s = p = (const unsigned char*)serialized_data_buffer;
	PHP_VAR_UNSERIALIZE_INIT(var_hash);

	if (*p != 'x' || *++p != ':') {
		goto outexcept;
	}
	++p;

	zval *count = var_tmp_var(&var_hash);
	if (!php_var_unserialize(count, &p, s + buf_len, &var_hash) || Z_TYPE_P(count) != IS_LONG) {
		goto outexcept;
	}

	zend_long num_elements = Z_LVAL_P(count);
	zval *array, *obj, *inf;
	while (num_elements-- > 0) {
		if (*p != 'O' && *p != 'C' && *p != 'r') {
			goto outexcept;
		}

		array = var_tmp_var(&var_hash);
		if (!php_var_unserialize(array, &p, s + buf_len, &var_hash) || Z_TYPE_P(array) != IS_OBJECT) {
			goto outexcept;
		}

		obj = var_tmp_var(&var_hash);
		if (!php_var_unserialize(obj, &p, s + buf_len, &var_hash)) {
			goto outexcept;
		}

		inf = var_tmp_var(&var_hash);
		if (!php_var_unserialize(inf, &p, s + buf_len, &var_hash)) {
			goto outexcept;
		}

		spl_SplObjectStorageElement *element = spl_object_storage_get(intern, obj);
		if (!element) {
			element = spl_object_storage_attach(intern, obj, inf);
		} else {
			zval_ptr_dtor(&element->inf);
			ZVAL_COPY(&element->inf, inf);
		}
	}
