zval *pmembers, *pcount;
	zend_long count;

	if (zend_parse_parameters(ZEND_NUM_ARGS(), "s", &serialized_data_buffer, &buf_len) == FAILURE) {
		RETURN_THROWS();
	}

	s = (const unsigned char*)serialized_data_buffer;
	p = s;

	PHP_VAR_UNSERIALIZE_INIT(var_hash);

	/* storage */
	if (*p!= 'x' || *++p != ':') {
		goto outexcept;
	}
	++p;

	pcount = var_tmp_var(&var_hash);
	if (!php_var_unserialize(pcount, &p, s + buf_len, &var_hash) || Z_TYPE_P(pcount) != IS_LONG) {
		goto outexcept;
	}

	count = Z_LVAL_P(pcount);

	while (count-- > 0) {
		zval *pinf;
		zval *pobj = var_tmp_var(&var_hash);

		if (!php_var_unserialize(pobj, &p, s + buf_len, &var_hash)) {
			goto outexcept;
		}

		if (Z_TYPE_P(pobj) != IS_OBJECT) {
			goto outexcept;
		}

		pinf = var_tmp_var(&var_hash);

		if (*p == ',') { /* new version has inf */
			++p;
			if (!php_var_unserialize(pinf, &p, s + buf_len, &var_hash)) {
				zval_ptr_dtor(pobj);
				goto outexcept;
			}
		} else {
			ZVAL_NULL(pinf);
		}

		spl_object_storage_attach(intern, Z_OBJ_P(pobj), pinf);
		zval_ptr_dtor(pobj);
	}
