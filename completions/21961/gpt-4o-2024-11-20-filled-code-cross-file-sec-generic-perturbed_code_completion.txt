case ReadStripMethod:
      {
        /*
          Read TIFF image using strip-based method.
        */
        for (row_index=0; row_index < (ssize_t) image->rows; row_index += rows_per_strip)
        {
          register Quantum
            *magick_restrict q;

          register ssize_t
            x,
            strip_rows;

          unsigned char
            *p;

          strip_rows=(ssize_t) MagickMin((size_t) rows_per_strip, image->rows-row_index);
          tiff_status=TIFFReadEncodedStrip(tiff,TIFFComputeStrip(tiff,row_index,0),
            pixels,TIFFStripSize(tiff));
          if (tiff_status == -1)
            break;
          p=pixels;
          for (ssize_t strip_row=0; strip_row < strip_rows; strip_row++)
          {
            q=QueueAuthenticPixels(image,0,row_index+strip_row,image->columns,1,exception);
            if (q == (Quantum *) NULL)
              break;
            for (x=0; x < (ssize_t) image->columns; x++)
            {
              ImportQuantumPixels(image,quantum_info,quantum_type,p,q,exception);
              p+=GetQuantumExtent(image,quantum_info,quantum_type);
              q+=GetPixelChannels(image);
            }
            if (SyncAuthenticPixels(image,exception) == MagickFalse)
              break;
          }
          if (image->previous == (Image *) NULL)
            {
              status=SetImageProgress(image,LoadImageTag,(MagickOffsetType) row_index,
                image->rows);
              if (status == MagickFalse)
                break;
            }
        }
        break;
      }
      case ReadTileMethod:
      {
        /*
          Read TIFF image using tile-based method.
        */
        uint32
          tile_width,
          tile_length;

        TIFFGetField(tiff,TIFFTAG_TILEWIDTH,&tile_width);
        TIFFGetField(tiff,TIFFTAG_TILELENGTH,&tile_length);
        for (row_index=0; row_index < (ssize_t) image->rows; row_index += tile_length)
        {
          for (ssize_t col_index=0; col_index < (ssize_t) image->columns; col_index += tile_width)
          {
            register Quantum
              *magick_restrict q;

            register ssize_t
              x,
              tile_row,
              tile_col;

            unsigned char
              *p;

            tiff_status=TIFFReadTile(tiff,pixels,col_index,row_index,0,0);
            if (tiff_status == -1)
              break;
            p=pixels;
            for (tile_row=0; tile_row < (ssize_t) tile_length; tile_row++)
            {
              if ((row_index+tile_row) >= (ssize_t) image->rows)
                break;
              q=QueueAuthenticPixels(image,col_index,row_index+tile_row,
                MagickMin(tile_width,image->columns-col_index),1,exception);
              if (q == (Quantum *) NULL)
                break;
              for (tile_col=0; tile_col < (ssize_t) tile_width; tile_col++)
              {
                if ((col_index+tile_col) >= (ssize_t) image->columns)
                  break;
                ImportQuantumPixels(image,quantum_info,quantum_type,p,q,exception);
                p+=GetQuantumExtent(image,quantum_info,quantum_type);
                q+=GetPixelChannels(image);
              }
              if (SyncAuthenticPixels(image,exception) == MagickFalse)
                break;
            }
          }
          if (image->previous == (Image *) NULL)
            {
              status=SetImageProgress(image,LoadImageTag,(MagickOffsetType) row_index,
                image->rows);
              if (status == MagickFalse)
                break;
            }
        }
        break;
      }
