case ReadStripMethod:
      {
        /*
          Convert strip TIFF image.
        */
        for (row_index=0; row_index < (ssize_t) image->rows; row_index+=rows_per_strip)
        {
          register Quantum
            *magick_restrict q;

          register ssize_t
            x,
            y;

          unsigned char
            *p;

          tiff_status=TIFFReadEncodedStrip(tiff,(tstrip_t) (row_index/rows_per_strip),
            pixels,(tsize_t) -1);
          if (tiff_status == -1)
            break;
          p=pixels;
          for (y=0; y < (ssize_t) rows_per_strip; y++)
          {
            if ((row_index+y) >= (ssize_t) image->rows)
              break;
            q=QueueAuthenticPixels(image,0,row_index+y,image->columns,1,exception);
            if (q == (Quantum *) NULL)
              break;
            for (x=0; x < (ssize_t) image->columns; x++)
            {
              SetPixelRed(image,ScaleCharToQuantum(*p++),q);
              SetPixelGreen(image,ScaleCharToQuantum(*p++),q);
              SetPixelBlue(image,ScaleCharToQuantum(*p++),q);
              if (image->alpha_trait != UndefinedPixelTrait)
                SetPixelAlpha(image,ScaleCharToQuantum(*p++),q);
              q+=GetPixelChannels(image);
            }
            if (SyncAuthenticPixels(image,exception) == MagickFalse)
              break;
          }
          if (image->previous == (Image *) NULL)
            {
              status=SetImageProgress(image,LoadImageTag,(MagickOffsetType) row_index,
                image->rows);
              if (status == MagickFalse)
                break;
            }
        }
        break;
      }
      case ReadTileMethod:
      {
        /*
          Convert tile TIFF image.
        */
        uint32
          tile_width,
          tile_height;

        TIFFGetField(tiff,TIFFTAG_TILEWIDTH,&tile_width);
        TIFFGetField(tiff,TIFFTAG_TILELENGTH,&tile_height);
        for (row_index=0; row_index < (ssize_t) image->rows; row_index+=tile_height)
        {
          register ssize_t
            x,
            y;

          for (x=0; x < (ssize_t) image->columns; x+=tile_width)
          {
            tiff_status=TIFFReadTile(tiff,pixels,(uint32) x,(uint32) row_index,0,0);
            if (tiff_status == -1)
              break;
            for (y=0; y < (ssize_t) tile_height; y++)
            {
              register Quantum
                *magick_restrict q;

              register unsigned char
                *p;

              if ((row_index+y) >= (ssize_t) image->rows)
                break;
              q=QueueAuthenticPixels(image,x,row_index+y,tile_width,1,exception);
              if (q == (Quantum *) NULL)
                break;
              p=pixels+y*tile_width*samples_per_pixel;
              for (x=0; x < (ssize_t) tile_width; x++)
              {
                SetPixelRed(image,ScaleCharToQuantum(*p++),q);
                SetPixelGreen(image,ScaleCharToQuantum(*p++),q);
                SetPixelBlue(image,ScaleCharToQuantum(*p++),q);
                if (image->alpha_trait != UndefinedPixelTrait)
                  SetPixelAlpha(image,ScaleCharToQuantum(*p++),q);
                q+=GetPixelChannels(image);
              }
              if (SyncAuthenticPixels(image,exception) == MagickFalse)
                break;
            }
          }
          if (image->previous == (Image *) NULL)
            {
              status=SetImageProgress(image,LoadImageTag,(MagickOffsetType) row_index,
                image->rows);
              if (status == MagickFalse)
                break;
            }
        }
        break;
      }
