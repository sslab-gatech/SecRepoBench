case ReadStripMethod:
      {
        /*
          Convert strip TIFF image.
        */
        for (row_index=0; row_index < (ssize_t) image->rows; row_index+=rows_per_strip)
        {
          register Quantum
            *magick_restrict q;

          register ssize_t
            x;

          unsigned char
            *p;

          tiff_status=TIFFReadEncodedStrip(tiff,TIFFComputeStrip(tiff,row_index,0),
            pixels,TIFFStripSize(tiff));
          if (tiff_status == -1)
            break;
          p=pixels;
          for (x=0; x < (ssize_t) image->columns; x++)
          {
            q=QueueAuthenticPixels(image,0,row_index,image->columns,1,exception);
            if (q == (Quantum *) NULL)
              break;
            ImportQuantumPixels(image,(CacheView *) NULL,quantum_info,quantum_type,
              p,exception);
            p+=GetQuantumExtent(image,quantum_info,quantum_type);
            if (SyncAuthenticPixels(image,exception) == MagickFalse)
              break;
          }
          if (image->previous == (Image *) NULL)
            {
              status=SetImageProgress(image,LoadImageTag,(MagickOffsetType) row_index,
                image->rows);
              if (status == MagickFalse)
                break;
            }
        }
        break;
      }
      case ReadTileMethod:
      {
        /*
          Convert tile TIFF image.
        */
        uint32
          tile_width,
          tile_length;

        TIFFGetField(tiff,TIFFTAG_TILEWIDTH,&tile_width);
        TIFFGetField(tiff,TIFFTAG_TILELENGTH,&tile_length);
        for (row_index=0; row_index < (ssize_t) image->rows; row_index+=tile_length)
        {
          register Quantum
            *magick_restrict q;

          register ssize_t
            x,
            y;

          unsigned char
            *p;

          for (x=0; x < (ssize_t) image->columns; x+=tile_width)
          {
            tiff_status=TIFFReadTile(tiff,pixels,x,row_index,0,0);
            if (tiff_status == -1)
              break;
            p=pixels;
            for (y=0; y < (ssize_t) tile_length; y++)
            {
              q=QueueAuthenticPixels(image,x,row_index+y,tile_width,1,exception);
              if (q == (Quantum *) NULL)
                break;
              ImportQuantumPixels(image,(CacheView *) NULL,quantum_info,quantum_type,
                p,exception);
              p+=GetQuantumExtent(image,quantum_info,quantum_type);
              if (SyncAuthenticPixels(image,exception) == MagickFalse)
                break;
            }
          }
          if (image->previous == (Image *) NULL)
            {
              status=SetImageProgress(image,LoadImageTag,(MagickOffsetType) row_index,
                image->rows);
              if (status == MagickFalse)
                break;
            }
        }
        break;
      }
