case ReadStripMethod:
      {
        size_t
          number_pixels;

        /*
          Convert strip TIFF image.
        */
        if (HeapOverflowSanityCheck(rows_per_strip,sizeof(*pixels)) != MagickFalse)
          ThrowTIFFException(ResourceLimitError,"MemoryAllocationFailed");
        number_pixels=rows_per_strip*image->columns;
        for (row_index=0; row_index < (ssize_t) image->rows; row_index+=rows_per_strip)
        {
          size_t
            rows=rows_per_strip;

          if ((row_index+rows) > image->rows)
            rows=image->rows-row_index;

          if (TIFFReadEncodedStrip(tiff,TIFFComputeStrip(tiff,row_index,0),
            pixels,number_pixels) == -1)
            break;

          ImportQuantumPixels(image,(CacheView *) NULL,quantum_info,quantum_type,
            pixels,exception);

          if (image->previous == (Image *) NULL)
            {
              status=SetImageProgress(image,LoadImageTag,(MagickOffsetType) row_index,image->rows);
              if (status == MagickFalse)
                break;
            }
        }
        break;
      }
      case ReadTileMethod:
      {
        uint32
          columns,
          rows;

        size_t
          number_pixels;

        /*
          Convert tile TIFF image.
        */
        TIFFGetField(tiff,TIFFTAG_TILEWIDTH,&columns);
        TIFFGetField(tiff,TIFFTAG_TILELENGTH,&rows);
        if (HeapOverflowSanityCheck(rows,sizeof(*pixels)) != MagickFalse)
          ThrowTIFFException(ResourceLimitError,"MemoryAllocationFailed");

        number_pixels = rows * columns;
        for (row_index=0; row_index < (ssize_t) image->rows; row_index+=rows)
        {
          register ssize_t y;
          size_t
            rows_remaining = rows;

          if ((row_index+rows_remaining) > image->rows)
            rows_remaining=image->rows-row_index;

          for (y=0; y < (ssize_t) rows_remaining; y++)
          {
            if (TIFFReadEncodedTile(tiff,TIFFComputeTile(tiff,0,y+row_index,0,0),pixels,number_pixels) == -1)
              break;

            ImportQuantumPixels(image,(CacheView *) NULL,quantum_info,quantum_type,pixels,exception);
          }

          if (image->previous == (Image *) NULL)
            {
              status=SetImageProgress(image,LoadImageTag,(MagickOffsetType) row_index,image->rows);
              if (status == MagickFalse)
                break;
            }
        }
        break;
      }
