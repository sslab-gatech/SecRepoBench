case ReadStripMethod:
      {
        /*
          Convert strip TIFF image.
        */
        for (row_index=0; row_index < (ssize_t) image->rows; row_index+=rows_per_strip)
        {
          register Quantum
            *magick_restrict q;

          register ssize_t
            x;

          ssize_t
            row_count;

          unsigned char
            *p;

          tiff_status=TIFFReadEncodedStrip(tiff,TIFFComputeStrip(tiff,row_index,0),
            pixels,(tsize_t) -1);
          if (tiff_status == -1)
            break;
          p=pixels;
          row_count=(ssize_t) MagickMin((size_t) rows_per_strip,image->rows-row_index);
          for (i=0; i < row_count; i++)
          {
            q=QueueAuthenticPixels(image,0,row_index+i,image->columns,1,exception);
            if (q == (Quantum *) NULL)
              break;
            for (x=0; x < (ssize_t) image->columns; x++)
            {
              SetPixelRed(image,ScaleCharToQuantum(*p++),q);
              SetPixelGreen(image,ScaleCharToQuantum(*p++),q);
              SetPixelBlue(image,ScaleCharToQuantum(*p++),q);
              if (image->alpha_trait != UndefinedPixelTrait)
                SetPixelAlpha(image,ScaleCharToQuantum(*p++),q);
              q+=GetPixelChannels(image);
            }
            if (SyncAuthenticPixels(image,exception) == MagickFalse)
              break;
          }
          if (image->previous == (Image *) NULL)
            {
              status=SetImageProgress(image,LoadImageTag,(MagickOffsetType) row_index,
                image->rows);
              if (status == MagickFalse)
                break;
            }
        }
        break;
      }
      case ReadTileMethod:
      {
        /*
          Convert tile TIFF image.
        */
        uint32
          tile_width,
          tile_height;

        tiff_status=TIFFGetField(tiff,TIFFTAG_TILEWIDTH,&tile_width);
        tiff_status&=TIFFGetField(tiff,TIFFTAG_TILELENGTH,&tile_height);
        if (tiff_status == 0)
          ThrowTIFFException(CorruptImageError,"UnableToReadTiles");
        for (row_index=0; row_index < (ssize_t) image->rows; row_index+=tile_height)
        {
          register Quantum
            *magick_restrict q;

          register ssize_t
            x,
            y;

          ssize_t
            row_count;

          unsigned char
            *p;

          for (x=0; x < (ssize_t) image->columns; x+=tile_width)
          {
            tiff_status=TIFFReadTile(tiff,pixels,x,row_index,0,0);
            if (tiff_status == -1)
              break;
            p=pixels;
            row_count=(ssize_t) MagickMin((size_t) tile_height,image->rows-row_index);
            for (y=0; y < row_count; y++)
            {
              q=QueueAuthenticPixels(image,x,row_index+y,tile_width,1,exception);
              if (q == (Quantum *) NULL)
                break;
              for (i=0; i < (ssize_t) tile_width; i++)
              {
                if ((x+i) >= (ssize_t) image->columns)
                  break;
                SetPixelRed(image,ScaleCharToQuantum(*p++),q);
                SetPixelGreen(image,ScaleCharToQuantum(*p++),q);
                SetPixelBlue(image,ScaleCharToQuantum(*p++),q);
                if (image->alpha_trait != UndefinedPixelTrait)
                  SetPixelAlpha(image,ScaleCharToQuantum(*p++),q);
                q+=GetPixelChannels(image);
              }
              if (SyncAuthenticPixels(image,exception) == MagickFalse)
                break;
            }
          }
          if (image->previous == (Image *) NULL)
            {
              status=SetImageProgress(image,LoadImageTag,(MagickOffsetType) row_index,
                image->rows);
              if (status == MagickFalse)
                break;
            }
        }
        break;
      }
