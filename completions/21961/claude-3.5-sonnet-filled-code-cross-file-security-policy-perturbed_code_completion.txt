case ReadStripMethod:
{
  size_t
    number_strips,
    strip_size;

  ssize_t
    y;

  number_strips=(size_t) TIFFNumberOfStrips(tiff);
  strip_size=TIFFStripSize(tiff);
  for (y=0; y < (ssize_t) image->rows; y+=(ssize_t) rows_per_strip)
  {
    size_t
      length;

    ssize_t
      count;

    tsize_t
      strip_bytes;

    count=(ssize_t) MagickMin(rows_per_strip,(size_t) (image->rows-y));
    strip_bytes=TIFFReadEncodedStrip(tiff,(tstrip_t) (y/rows_per_strip),
      tiff_info.scanline,(tsize_t) (count*strip_size));
    if (strip_bytes < 0)
      break;
    length=(size_t) strip_bytes;
    if (ImportQuantumPixels(image,(CacheView *) NULL,quantum_info,
        quantum_type,tiff_info.scanline,exception) == MagickFalse)
      break;
    if (SyncAuthenticPixels(image,exception) == MagickFalse)
      break;
    if (image->previous == (Image *) NULL)
      {
        status=SetImageProgress(image,LoadImageTag,(MagickOffsetType) y,
          image->rows);
        if (status == MagickFalse)
          break;
      }
  }
  break;
}
