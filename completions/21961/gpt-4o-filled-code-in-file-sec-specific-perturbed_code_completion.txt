case ReadStripMethod:
      {
        /*
          Convert strip TIFF image.
        */
        for (row_index=0; row_index < (ssize_t) image->rows; row_index+=rows_per_strip)
        {
          tsize_t
            strip_size;

          strip_size=TIFFReadEncodedStrip(tiff,(tstrip_t) (row_index/rows_per_strip),
            pixels,(tsize_t) -1);
          if (strip_size < 0)
            break;
          for (y=0; y < (ssize_t) rows_per_strip; y++)
          {
            register Quantum
              *magick_restrict q;

            register ssize_t
              x;

            if ((row_index+y) >= (ssize_t) image->rows)
              break;
            q=QueueAuthenticPixels(image,0,row_index+y,image->columns,1,exception);
            if (q == (Quantum *) NULL)
              break;
            (void) ImportQuantumPixels(image,(CacheView *) NULL,quantum_info,
              quantum_type,pixels+y*TIFFScanlineSize(tiff),exception);
            if (SyncAuthenticPixels(image,exception) == MagickFalse)
              break;
          }
          if (image->previous == (Image *) NULL)
            {
              status=SetImageProgress(image,LoadImageTag,(MagickOffsetType) row_index,
                image->rows);
              if (status == MagickFalse)
                break;
            }
        }
        break;
      }
      case ReadTileMethod:
      {
        /*
          Convert tile TIFF image.
        */
        uint32
          *tile_size;

        uint32
          tile_width,
          tile_length;

        tile_size=(uint32 *) AcquireQuantumMemory((size_t) TIFFTileSize(tiff),
          sizeof(*tile_size));
        if (tile_size == (uint32 *) NULL)
          ThrowTIFFException(ResourceLimitError,"MemoryAllocationFailed");
        TIFFGetField(tiff,TIFFTAG_TILEWIDTH,&tile_width);
        TIFFGetField(tiff,TIFFTAG_TILELENGTH,&tile_length);
        for (row_index=0; row_index < (ssize_t) image->rows; row_index+=tile_length)
        {
          for (x=0; x < (ssize_t) image->columns; x+=tile_width)
          {
            tsize_t
              tile_size;

            tile_size=TIFFReadTile(tiff,pixels,(uint32) x,(uint32) row_index,0,0);
            if (tile_size < 0)
              break;
            for (y=0; y < (ssize_t) tile_length; y++)
            {
              register Quantum
                *magick_restrict q;

              register ssize_t
                x_offset;

              if ((row_index+y) >= (ssize_t) image->rows)
                break;
              q=QueueAuthenticPixels(image,x,row_index+y,tile_width,1,exception);
              if (q == (Quantum *) NULL)
                break;
              x_offset=0;
              if ((x+tile_width) > image->columns)
                x_offset=(x+tile_width)-image->columns;
              (void) ImportQuantumPixels(image,(CacheView *) NULL,quantum_info,
                quantum_type,pixels+y*TIFFTileRowSize(tiff),exception);
              if (SyncAuthenticPixels(image,exception) == MagickFalse)
                break;
            }
          }
          if (image->previous == (Image *) NULL)
            {
              status=SetImageProgress(image,LoadImageTag,(MagickOffsetType) row_index,
                image->rows);
              if (status == MagickFalse)
                break;
            }
        }
        tile_size=(uint32 *) RelinquishMagickMemory(tile_size);
        break;
      }
