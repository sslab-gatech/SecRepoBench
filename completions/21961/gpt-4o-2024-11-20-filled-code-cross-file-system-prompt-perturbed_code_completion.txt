case ReadStripMethod:
      {
        /*
          Convert strip-based TIFF image.
        */
        tsize_t
          strip_size;

        tstrip_t
          strip;

        strip_size=TIFFStripSize(tiff);
        if (strip_size <= 0)
          ThrowTIFFException(ResourceLimitError,"MemoryAllocationFailed");
        for (row_index=0; row_index < (ssize_t) image->rows; row_index += rows_per_strip)
        {
          register Quantum
            *magick_restrict q;

          register ssize_t
            x,
            y;

          unsigned char
            *p;

          strip=(tstrip_t) (row_index/rows_per_strip);
          tiff_status=TIFFReadEncodedStrip(tiff,strip,pixels,strip_size);
          if (tiff_status == -1)
            break;
          p=pixels;
          for (y=0; y < (ssize_t) rows_per_strip; y++)
          {
            if ((row_index+y) >= (ssize_t) image->rows)
              break;
            q=QueueAuthenticPixels(image,0,row_index+y,image->columns,1,exception);
            if (q == (Quantum *) NULL)
              break;
            for (x=0; x < (ssize_t) image->columns; x++)
            {
              ImportQuantumPixels(image,quantum_info,quantum_type,p,q,exception);
              p+=samples_per_pixel*((bits_per_sample+7) >> 3);
              q+=GetPixelChannels(image);
            }
            if (SyncAuthenticPixels(image,exception) == MagickFalse)
              break;
          }
          if (image->previous == (Image *) NULL)
            {
              status=SetImageProgress(image,LoadImageTag,(MagickOffsetType) row_index,
                image->rows);
              if (status == MagickFalse)
                break;
            }
        }
        break;
      }
      case ReadTileMethod:
      {
        /*
          Convert tile-based TIFF image.
        */
        uint32
          tile_width,
          tile_height;

        tsize_t
          tile_size;

        uint32
          x,
          y;

        tile_size=TIFFTileSize(tiff);
        if (tile_size <= 0)
          ThrowTIFFException(ResourceLimitError,"MemoryAllocationFailed");
        TIFFGetField(tiff,TIFFTAG_TILEWIDTH,&tile_width);
        TIFFGetField(tiff,TIFFTAG_TILELENGTH,&tile_height);
        for (y=0; y < image->rows; y += tile_height)
        {
          for (x=0; x < image->columns; x += tile_width)
          {
            tiff_status=TIFFReadTile(tiff,pixels,x,y,0,0);
            if (tiff_status == -1)
              break;
            for (row_index=0; row_index < (ssize_t) tile_height; row_index++)
            {
              register Quantum
                *magick_restrict q;

              register ssize_t
                col;

              unsigned char
                *p;

              if ((y+row_index) >= image->rows)
                break;
              q=QueueAuthenticPixels(image,x,y+row_index,tile_width,1,exception);
              if (q == (Quantum *) NULL)
                break;
              p=pixels+(row_index*tile_width*samples_per_pixel*((bits_per_sample+7) >> 3));
              for (col=0; col < (ssize_t) tile_width; col++)
              {
                if ((x+col) >= image->columns)
                  break;
                ImportQuantumPixels(image,quantum_info,quantum_type,p,q,exception);
                p+=samples_per_pixel*((bits_per_sample+7) >> 3);
                q+=GetPixelChannels(image);
              }
              if (SyncAuthenticPixels(image,exception) == MagickFalse)
                break;
            }
          }
          if (image->previous == (Image *) NULL)
            {
              status=SetImageProgress(image,LoadImageTag,(MagickOffsetType) y,
                image->rows);
              if (status == MagickFalse)
                break;
            }
        }
        break;
      }
