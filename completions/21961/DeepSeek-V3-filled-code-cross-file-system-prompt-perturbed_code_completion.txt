case ReadStripMethod:
      {
        uint32
          strip_size;

        /*
          Convert strip TIFF image.
        */
        strip_size=TIFFStripSize(tiff);
        for (row_index=0; row_index < (ssize_t) image->rows; row_index++)
        {
          register Quantum
            *magick_restrict q;

          register ssize_t
            x;

          unsigned char
            *p;

          tiff_status=TIFFReadEncodedStrip(tiff,TIFFComputeStrip(tiff,(uint32)
            row_index,0),(tdata_t) pixels,strip_size);
          if (tiff_status == -1)
            break;
          q=QueueAuthenticPixels(image,0,row_index,image->columns,1,exception);
          if (q == (Quantum *) NULL)
            break;
          p=pixels;
          for (x=0; x < (ssize_t) image->columns; x++)
          {
            ImportQuantumPixels(image,(CacheView *) NULL,quantum_info,
              quantum_type,p,q,exception);
            p+=quantum_info->depth/8;
            q+=GetPixelChannels(image);
          }
          if (SyncAuthenticPixels(image,exception) == MagickFalse)
            break;
          if (image->previous == (Image *) NULL)
            {
              status=SetImageProgress(image,LoadImageTag,(MagickOffsetType) row_index,
                image->rows);
              if (status == MagickFalse)
                break;
            }
        }
        break;
      }
      case ReadTileMethod:
      {
        uint32
          tile_height,
          tile_width;

        /*
          Convert tile TIFF image.
        */
        (void) TIFFGetField(tiff,TIFFTAG_TILEWIDTH,&tile_width);
        (void) TIFFGetField(tiff,TIFFTAG_TILELENGTH,&tile_height);
        for (row_index=0; row_index < (ssize_t) image->rows; row_index+=tile_height)
        {
          register Quantum
            *magick_restrict q;

          register ssize_t
            x,
            y;

          unsigned char
            *p;

          for (x=0; x < (ssize_t) image->columns; x+=tile_width)
          {
            tiff_status=TIFFReadTile(tiff,(tdata_t) pixels,(uint32) x,
              (uint32) row_index,0,0);
            if (tiff_status == -1)
              break;
            p=pixels;
            for (y=0; y < (ssize_t) tile_height; y++)
            {
              if ((row_index+y) >= (ssize_t) image->rows)
                break;
              q=QueueAuthenticPixels(image,x,row_index+y,
                MagickMin(tile_width,image->columns-x),1,exception);
              if (q == (Quantum *) NULL)
                break;
              ImportQuantumPixels(image,(CacheView *) NULL,quantum_info,
                quantum_type,p,q,exception);
              p+=quantum_info->depth/8*tile_width;
              if (SyncAuthenticPixels(image,exception) == MagickFalse)
                break;
            }
          }
          if (image->previous == (Image *) NULL)
            {
              status=SetImageProgress(image,LoadImageTag,(MagickOffsetType) row_index,
                image->rows);
              if (status == MagickFalse)
                break;
            }
        }
        break;
      }
