case ReadStripMethod:
{
  uint32
    strip;

  /*
    Convert strip TIFF image.
  */
  for (strip=0; strip < TIFFNumberOfStrips(tiff); strip++)
  {
    register Quantum
      *magick_restrict q;

    register ssize_t
      x;

    unsigned char
      *p;

    tiff_status=TIFFReadEncodedStrip(tiff,strip,(char *) pixels);
    if (tiff_status == -1)
      break;
    q=QueueAuthenticPixels(image,0,strip*rows_per_strip,image->columns,1,
      exception);
    if (q == (Quantum *) NULL)
      break;
    p=pixels;
    for (x=0; x < (ssize_t) image->columns; x++)
    {
      SetPixelRed(image,ScaleCharToQuantum((unsigned char) TIFFGetR(*p)),q);
      SetPixelGreen(image,ScaleCharToQuantum((unsigned char) TIFFGetG(*p)),q);
      SetPixelBlue(image,ScaleCharToQuantum((unsigned char) TIFFGetB(*p)),q);
      if (image->alpha_trait!= UndefinedPixelTrait)
        SetPixelAlpha(image,ScaleCharToQuantum((unsigned char) TIFFGetA(*p)),
          q);
      p++;
      q+=GetPixelChannels(image);
    }
    if (SyncAuthenticPixels(image,exception) == MagickFalse)
      break;
    if (image->previous == (Image *) NULL)
      {
        status=SetImageProgress(image,LoadImageTag,(MagickOffsetType) strip,
          TIFFNumberOfStrips(tiff));
        if (status == MagickFalse)
          break;
      }
  }
  break;
}
case ReadTileMethod:
{
  uint32
    tile;

  /*
    Convert tile TIFF image.
  */
  for (tile=0; tile < TIFFNumberOfTiles(tiff); tile++)
  {
    register Quantum
      *magick_restrict q;

    register ssize_t
      x,
      y;

    unsigned char
      *p;

    tiff_status=TIFFReadEncodedTile(tiff,tile,(char *) pixels);
    if (tiff_status == -1)
      break;
    q=QueueAuthenticPixels(image,0,tile*rows_per_strip,image->columns,1,
      exception);
    if (q == (Quantum *) NULL)
      break;
    p=pixels;
    for (y=0; y < (ssize_t) TIFFTileRowSize(tiff); y++)
    {
      for (x=0; x < (ssize_t) TIFFTileColSize(tiff); x++)
      {
        SetPixelRed(image,ScaleCharToQuantum((unsigned char) TIFFGetR(*p)),q);
        SetPixelGreen(image,ScaleCharToQuantum((unsigned char) TIFFGetG(*p)),q);
        SetPixelBlue(image,ScaleCharToQuantum((unsigned char) TIFFGetB(*p)),q);
        if (image->alpha_trait!= UndefinedPixelTrait)
          SetPixelAlpha(image,ScaleCharToQuantum((unsigned char) TIFFGetA(*p)),
            q);
        p++;
        q+=GetPixelChannels(image);
      }
    }
    if (SyncAuthenticPixels(image,exception) == MagickFalse)
      break;
    if (image->previous == (Image *) NULL)
      {
        status=SetImageProgress(image,LoadImageTag,(MagickOffsetType) tile,
          TIFFNumberOfTiles(tiff));
        if (status == MagickFalse)
          break;
      }
  }
  break;
}
