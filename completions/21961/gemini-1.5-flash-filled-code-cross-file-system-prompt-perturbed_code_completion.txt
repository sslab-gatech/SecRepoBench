case ReadStripMethod:
      {
        /*
          Convert TIFF image to pixel packets.
        */
        for (row_index=0; row_index < (ssize_t) image->rows; row_index+=rows_per_strip)
        {
          size_t
            length;

          register Quantum
            *magick_restrict q;

          register ssize_t
            x;

          ssize_t
            count;

          unsigned char
            *p;

          count=(ssize_t) MagickMin((size_t) rows_per_strip,(size_t)
            image->rows-row_index);
          length=TIFFReadEncodedStrip(tiff,(uint32) row_index/rows_per_strip,
            (char *) pixels);
          if (length <= 0)
            break;
          if (length > number_pixels)
            ThrowTIFFException(CorruptImageError,"InsufficientImageDataInFile");
          p=pixels;
          for (x=0; x < (ssize_t) image->columns; x++)
          {
            q=QueueAuthenticPixels(image,0,row_index+x,image->columns,1,
              exception);
            if (q == (Quantum *) NULL)
              break;
            ImportQuantumPixels(image,quantum_info,quantum_type,p,q,exception);
            p+=quantum_info->pad;
            q+=GetPixelChannels(image);
          }
          if (SyncAuthenticPixels(image,exception) == MagickFalse)
            break;
          if (image->previous == (Image *) NULL)
            {
              status=SetImageProgress(image,LoadImageTag,(MagickOffsetType)
                row_index,image->rows);
              if (status == MagickFalse)
                break;
            }
        }
        break;
      }
      case ReadTileMethod:
      {
        uint32
          tile_height,
          tile_width;

        /*
          Convert tiled TIFF image to pixel packets.
        */
        if (TIFFGetField(tiff,TIFFTAG_TILEWIDTH,&tile_width) != 1)
          ThrowTIFFException(CoderError,"UnableToReadTileWidth");
        if (TIFFGetField(tiff,TIFFTAG_TILELENGTH,&tile_height) != 1)
          ThrowTIFFException(CoderError,"UnableToReadTileLength");
        for (row_index=0; row_index < (ssize_t) image->rows; row_index+=tile_height)
        {
          size_t
            length;

          register Quantum
            *magick_restrict q;

          register ssize_t
            x;

          ssize_t
            count,
            tile_row;

          unsigned char
            *p;

          count=(ssize_t) MagickMin((size_t) tile_height,(size_t)
            image->rows-row_index);
          for (tile_row=0; tile_row < (ssize_t) count; tile_row++)
          {
            length=TIFFReadEncodedTile(tiff,(uint32) (row_index/tile_height)*
              image->columns+tile_row,(char *) pixels);
            if (length <= 0)
              break;
            if (length > number_pixels)
              ThrowTIFFException(CorruptImageError,
                "InsufficientImageDataInFile");
            p=pixels;
            for (x=0; x < (ssize_t) image->columns; x++)
            {
              q=QueueAuthenticPixels(image,0,row_index+tile_row+x,
                image->columns,1,exception);
              if (q == (Quantum *) NULL)
                break;
              ImportQuantumPixels(image,quantum_info,quantum_type,p,q,
                exception);
              p+=quantum_info->pad;
              q+=GetPixelChannels(image);
            }
            if (SyncAuthenticPixels(image,exception) == MagickFalse)
              break;
            if (image->previous == (Image *) NULL)
              {
                status=SetImageProgress(image,LoadImageTag,(MagickOffsetType)
                  row_index+tile_row,image->rows);
                if (status == MagickFalse)
                  break;
              }
          }
        }
        break;
      }
