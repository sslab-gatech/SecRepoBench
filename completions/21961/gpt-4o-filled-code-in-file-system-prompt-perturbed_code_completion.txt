case ReadStripMethod:
      {
        /*
          Convert strip TIFF image.
        */
        for (row_index=0; row_index < (ssize_t) image->rows; row_index += (ssize_t) rows_per_strip)
        {
          register Quantum
            *magick_restrict q;

          register ssize_t
            x;

          size_t
            number_rows;

          tiff_status=TIFFReadEncodedStrip(tiff,(tstrip_t) (row_index/rows_per_strip),
            pixels,(tsize_t) -1);
          if (tiff_status == -1)
            break;
          number_rows=(size_t) MagickMin((size_t) rows_per_strip,image->rows-(size_t) row_index);
          q=QueueAuthenticPixels(image,0,row_index,image->columns,number_rows,exception);
          if (q == (Quantum *) NULL)
            break;
          (void) ImportQuantumPixels(image,(CacheView *) NULL,quantum_info,quantum_type,pixels,exception);
          if (SyncAuthenticPixels(image,exception) == MagickFalse)
            break;
          if (image->previous == (Image *) NULL)
            {
              status=SetImageProgress(image,LoadImageTag,(MagickOffsetType) row_index,image->rows);
              if (status == MagickFalse)
                break;
            }
        }
        break;
      }
      case ReadTileMethod:
      {
        /*
          Convert tile TIFF image.
        */
        uint32
          *tile_width,
          *tile_length;

        tile_width=(uint32 *) AcquireQuantumMemory(1,sizeof(*tile_width));
        tile_length=(uint32 *) AcquireQuantumMemory(1,sizeof(*tile_length));
        if ((tile_width == (uint32 *) NULL) || (tile_length == (uint32 *) NULL))
          ThrowTIFFException(ResourceLimitError,"MemoryAllocationFailed");
        (void) TIFFGetField(tiff,TIFFTAG_TILEWIDTH,tile_width);
        (void) TIFFGetField(tiff,TIFFTAG_TILELENGTH,tile_length);
        for (row_index=0; row_index < (ssize_t) image->rows; row_index += (ssize_t) *tile_length)
        {
          register Quantum
            *magick_restrict q;

          register ssize_t
            x;

          size_t
            number_rows;

          for (x=0; x < (ssize_t) image->columns; x += (ssize_t) *tile_width)
          {
            tiff_status=TIFFReadTile(tiff,pixels,(uint32) x,(uint32) row_index,0,0);
            if (tiff_status == -1)
              break;
            number_rows=(size_t) MagickMin((size_t) *tile_length,image->rows-(size_t) row_index);
            q=QueueAuthenticPixels(image,x,row_index,*tile_width,number_rows,exception);
            if (q == (Quantum *) NULL)
              break;
            (void) ImportQuantumPixels(image,(CacheView *) NULL,quantum_info,quantum_type,pixels,exception);
            if (SyncAuthenticPixels(image,exception) == MagickFalse)
              break;
          }
          if (image->previous == (Image *) NULL)
            {
              status=SetImageProgress(image,LoadImageTag,(MagickOffsetType) row_index,image->rows);
              if (status == MagickFalse)
                break;
            }
        }
        tile_width=(uint32 *) RelinquishMagickMemory(tile_width);
        tile_length=(uint32 *) RelinquishMagickMemory(tile_length);
        break;
      }
