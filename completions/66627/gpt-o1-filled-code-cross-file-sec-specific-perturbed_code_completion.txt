    /* Open the attribute named "MATLAB_fields" */
    attr_id = H5Aopen(dset_id, "MATLAB_fields", H5P_DEFAULT);
    if ( attr_id < 0 ) {
        return MATIO_E_GENERIC_READ_ERROR;
    }

    /* Retrieve the dataspace of the opened attribute */
    dspace_id = H5Aget_space(attr_id);
    if ( dspace_id < 0 ) {
        H5Aclose(attr_id);
        return MATIO_E_GENERIC_READ_ERROR;
    }

    /* Verify that the dataspace has exactly one dimension */
    if ( H5Sget_simple_extent_ndims(dspace_id) != 1 ) {
        H5Sclose(dspace_id);
        H5Aclose(attr_id);
        return MATIO_E_GENERIC_READ_ERROR;
    }

    /* Obtain the number of fields */
    {
        hsize_t dims[1];
        if ( H5Sget_simple_extent_dims(dspace_id, dims, NULL) < 0 ) {
            H5Sclose(dspace_id);
            H5Aclose(attr_id);
            return MATIO_E_GENERIC_READ_ERROR;
        }
        *nfields = dims[0];
