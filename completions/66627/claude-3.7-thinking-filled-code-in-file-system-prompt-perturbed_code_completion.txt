attr_id = H5Aopen_by_name(dset_id, ".", "MATLAB_fields", H5P_DEFAULT, H5P_DEFAULT);
if (attr_id < 0) {
    return MATIO_E_ATTRIBUTE_NOT_FOUND;
}

dspace_id = H5Aget_space(attr_id);
if (dspace_id < 0) {
    H5Aclose(attr_id);
    return MATIO_E_UNKNOWN_ERROR;
}

{
    int rank;
    hsize_t dims[H5S_MAX_RANK];
    rank = H5Sget_simple_extent_ndims(dspace_id);
    if (rank < 0) {
        H5Sclose(dspace_id);
        H5Aclose(attr_id);
        return MATIO_E_UNKNOWN_ERROR;
    }
    herr = H5Sget_simple_extent_dims(dspace_id, dims, NULL);
    if (herr < 0 || rank != 1) {
        H5Sclose(dspace_id);
        H5Aclose(attr_id);
        return MATIO_E_UNKNOWN_ERROR;
    }
    *nfields = dims[0];
}

err = MATIO_E_NO_ERROR;
