herr_t herr;
int err = MATIO_E_NO_ERROR;

/* Open the attribute named "MATLAB_fields" */
attr_id = H5Aopen(dset_id, "MATLAB_fields", H5P_DEFAULT);
if ( attr_id < 0 ) {
    return MATIO_E_GENERIC_READ_ERROR;
}

/* Get the dataspace of the attribute */
dspace_id = H5Aget_space(attr_id);
if ( dspace_id < 0 ) {
    H5Aclose(attr_id);
    return MATIO_E_GENERIC_READ_ERROR;
}

/* Get the number of fields, which is the number of elements in the dataspace */
hssize_t num_elem = H5Sget_simple_extent_npoints(dspace_id);
if ( num_elem < 0 ) {
    H5Sclose(dspace_id);
    H5Aclose(attr_id);
    return MATIO_E_GENERIC_READ_ERROR;
}

*nfields = (hsize_t)num_elem;
