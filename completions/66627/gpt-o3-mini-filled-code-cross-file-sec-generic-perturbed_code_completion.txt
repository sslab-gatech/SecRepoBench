attr_id = H5Aopen(dset_id, "MATLAB_fields", H5P_DEFAULT);
    if ( attr_id < 0 ) {
        return MATIO_E_GENERIC_READ_ERROR;
    }

    dspace_id = H5Aget_space(attr_id);
    if ( dspace_id < 0 ) {
        H5Aclose(attr_id);
        return MATIO_E_GENERIC_READ_ERROR;
    }

    /* Ensure the dataspace is one-dimensional */
    int ndims = H5Sget_simple_extent_ndims(dspace_id);
    if ( ndims != 1 ) {
        H5Sclose(dspace_id);
        H5Aclose(attr_id);
        return MATIO_E_GENERIC_READ_ERROR;
    }

    hsize_t extent[1];
    if ( H5Sget_simple_extent_dims(dspace_id, extent, NULL) < 0 ) {
        H5Sclose(dspace_id);
        H5Aclose(attr_id);
        return MATIO_E_GENERIC_READ_ERROR;
    }
    *nfields = extent[0];
    err = MATIO_E_NO_ERROR;
