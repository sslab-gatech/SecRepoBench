attr_id = H5Aopen(dset_id, "MATLAB_fields", H5P_DEFAULT);
    if (attr_id < 0) {
        return MATIO_E_GENERIC_READ_ERROR;
    }

    dspace_id = H5Aget_space(attr_id);
    if (dspace_id < 0) {
        H5Aclose(attr_id);
        return MATIO_E_GENERIC_READ_ERROR;
    }

    hsize_t dims[1];
    int ndims = H5Sget_simple_extent_dims(dspace_id, dims, NULL);
    if (ndims < 0) {
        H5Sclose(dspace_id);
        H5Aclose(attr_id);
        return MATIO_E_GENERIC_READ_ERROR;
    }

    *nfields = dims[0];
    if (*nfields <= 0) {
        H5Sclose(dspace_id);
        H5Aclose(attr_id);
        return MATIO_E_NO_ERROR; // No fields to read
    }

    H5Sclose(dspace_id);
