int err = MATIO_E_NO_ERROR;
attr_id = H5Aopen_by_name(dset_id, ".", "MATLAB_fields", H5P_DEFAULT, H5P_DEFAULT);
if (attr_id < 0) {
    return MATIO_E_GENERIC_READ_ERROR;
}
dspace_id = H5Aget_space(attr_id);
if (dspace_id < 0) {
    H5Aclose(attr_id);
    return MATIO_E_GENERIC_READ_ERROR;
}
int ndims = H5Sget_simple_extent_ndims(dspace_id);
if (ndims <= 0) {
    H5Sclose(dspace_id);
    H5Aclose(attr_id);
    return MATIO_E_GENERIC_READ_ERROR;
}
hsize_t dims[H5S_MAX_RANK];
if (H5Sget_simple_extent_dims(dspace_id, dims, NULL) < 0) {
    H5Sclose(dspace_id);
    H5Aclose(attr_id);
    return MATIO_E_GENERIC_READ_ERROR;
}
if (ndims == 1) {
    *nfields = dims[0];
} else {
    hsize_t prod = 1;
    for (int j = 0; j < ndims; j++) {
        prod *= dims[j];
    }
    *nfields = prod;
}
