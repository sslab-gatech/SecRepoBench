u_int32_t pos = a + 2; // Start searching after the current CR+LF
if ((pos + 1) < packet->payload_packet_len &&
    packet->payload[pos] == 0x0d && packet->payload[pos + 1] == 0x0a) {
  // Found another CR+LF sequence, calculate remaining payload length beyond this sequence
  u_int32_t remaining_payload_len = packet->payload_packet_len - (pos + 2);
  
  // Limit the length to the size of the initial_binary_bytes buffer
  if (remaining_payload_len > sizeof(packet->initial_binary_bytes)) {
    remaining_payload_len = sizeof(packet->initial_binary_bytes);
  }

  // Copy the data into the initial_binary_bytes buffer
  memcpy(packet->initial_binary_bytes, &packet->payload[pos + 2], remaining_payload_len);

  // Update the length of the copied data in initial_binary_bytes_len
  packet->initial_binary_bytes_len = (u_int16_t)remaining_payload_len;
}
