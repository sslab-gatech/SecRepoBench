// Check for two consecutive CR+LF ("\r\n\r\n") sequences
      if ((a + 3) < packet->payload_packet_len &&
          packet->payload[a + 2] == 0x0d && packet->payload[a + 3] == 0x0a) {
        // Calculate the remaining payload length beyond the sequence
        u_int32_t remaining_payload_len = packet->payload_packet_len - (a + 4);

        // Limit the length to the size of the initial_binary_bytes buffer
        u_int32_t copy_len = (remaining_payload_len > sizeof(packet->initial_binary_bytes))
                               ? sizeof(packet->initial_binary_bytes)
                               : remaining_payload_len;

        // Copy the data into the initial_binary_bytes buffer
        memcpy(packet->initial_binary_bytes, &packet->payload[a + 4], copy_len);

        // Update the length of the copied data
        packet->initial_binary_bytes_len = copy_len;
      }
