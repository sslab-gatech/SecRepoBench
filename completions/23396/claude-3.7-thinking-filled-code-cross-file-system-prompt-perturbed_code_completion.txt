if(((a + 3) < packet->payload_packet_len) && 
   (packet->payload[a+2] == 0x0d) && 
   (packet->payload[a+3] == 0x0a)) {
  /* Found the double CRLF sequence ("\r\n\r\n") marking the end of HTTP headers */
  
  /* Calculate where the HTTP body starts and how many bytes are in it */
  u_int32_t body_start = a + 4;
  u_int32_t body_length = packet->payload_packet_len - body_start;
  
  if(body_length > 0) {
    /* There is a body after the headers */
    
    /* Limit to the size of initial_binary_bytes buffer */
    if(body_length > sizeof(packet->initial_binary_bytes))
      body_length = sizeof(packet->initial_binary_bytes);
    
    /* Copy the initial bytes of the body */
    memcpy(packet->initial_binary_bytes, &packet->payload[body_start], body_length);
    
    /* Update the length */
    packet->initial_binary_bytes_len = body_length;
  }
}
