if ((a + 3) < packet->payload_packet_len &&
    packet->payload[a+2] == 0x0d && packet->payload[a+3] == 0x0a) {
  u_int32_t binary_offset = a + 4;
  u_int32_t binary_len = packet->payload_packet_len - binary_offset;
  if (binary_len > sizeof(packet->initial_binary_bytes))
    binary_len = sizeof(packet->initial_binary_bytes);
  memcpy(packet->initial_binary_bytes, packet->payload + binary_offset, binary_len);
  packet->initial_binary_bytes_len = binary_len;
}
