if (packet->parsed_lines > 0 && packet->payload_packet_len > a + 4 &&
    packet->payload[a] == 0x0d && packet->payload[a + 1] == 0x0a && packet->payload[a + 2] == 0x0d &&
    packet->payload[a + 3] == 0x0a) {
  size_t bytes_to_copy = packet->payload_packet_len - (a + 4);
  if (bytes_to_copy > sizeof(packet->initial_binary_bytes)) {
    bytes_to_copy = sizeof(packet->initial_binary_bytes);
  }
  memcpy(packet->initial_binary_bytes, &packet->payload[a + 4], bytes_to_copy);
  packet->initial_binary_bytes_len = (u_int16_t)bytes_to_copy;
}
