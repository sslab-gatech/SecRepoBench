if ((a + 3) < packet->payload_packet_len && packet->payload[a+2] == 0x0d && packet->payload[a+3] == 0x0a) {
    u_int32_t remaining_length = packet->payload_packet_len - (a + 4);
    u_int32_t copy_length = MIN(remaining_length, NDPI_MAX_INITIAL_BINARY_BYTES);
    memcpy(initial_binary_bytes, &packet->payload[a + 4], copy_length);
    initial_binary_bytes_len = copy_length;
}
