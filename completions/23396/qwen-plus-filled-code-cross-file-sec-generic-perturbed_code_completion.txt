u_int32_t remaining_payload_len = packet->payload_packet_len - (a + 2);
if ((a + 4) < packet->payload_packet_len &&
    packet->payload[a + 2] == 0x0d && packet->payload[a + 3] == 0x0a) {
  if (remaining_payload_len > sizeof(packet->initial_binary_bytes)) {
    remaining_payload_len = sizeof(packet->initial_binary_bytes);
  }
  memcpy(packet->initial_binary_bytes, &packet->payload[a + 4], remaining_payload_len);
  packet->initial_binary_bytes_len = (u_int16_t)remaining_payload_len;
}
