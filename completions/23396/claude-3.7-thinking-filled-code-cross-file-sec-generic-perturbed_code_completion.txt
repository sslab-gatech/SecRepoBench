if((a + 3) < packet->payload_packet_len && 
   packet->payload[a+2] == 0x0d && 
   packet->payload[a+3] == 0x0a) {
  // Found "\r\n\r\n" - end of HTTP headers
  u_int32_t pos = a + 4; // Position after "\r\n\r\n"
  u_int32_t len = 0;
  
  if(pos < packet->payload_packet_len) {
    len = packet->payload_packet_len - pos; // Remaining length
    
    // Limit to buffer size
    if(len > sizeof(packet->initial_binary_bytes))
      len = sizeof(packet->initial_binary_bytes);
    
    if(len > 0) {
      memcpy(packet->initial_binary_bytes, &packet->payload[pos], len);
    }
  }
  
  packet->initial_binary_bytes_len = len;
}
