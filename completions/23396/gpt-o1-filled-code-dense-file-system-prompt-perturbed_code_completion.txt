if((a + 3 < packet->payload_packet_len) &&
   (packet->payload[a+2] == 0x0d) &&
   (packet->payload[a+3] == 0x0a)) {
  /* Found "\r\n\r\n" sequence */
  u_int32_t offset = a + 4;

  if(offset < packet->payload_packet_len) {
    u_int32_t remainder = packet->payload_packet_len - offset;

    if(remainder > sizeof(packet->initial_binary_bytes))
      remainder = sizeof(packet->initial_binary_bytes);

    if(remainder > 0) {
      memcpy(packet->initial_binary_bytes, &packet->payload[offset], remainder);
      packet->initial_binary_bytes_len = remainder;
    }
  }
}
