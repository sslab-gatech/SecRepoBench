// Check for the presence of a sequence of two consecutive CR+LF ("\r\n\r\n")
      if (a + 3 < packet->payload_packet_len && 
          packet->payload[a + 2] == 0x0d && 
          packet->payload[a + 3] == 0x0a) {
        
        // Calculate the remaining payload length beyond the sequence
        size_t remaining_length = packet->payload_packet_len - (a + 4);
        
        // Limit the length to the size of the initial_binary_bytes buffer
        size_t copy_length = remaining_length < INITIAL_BINARY_BYTES_SIZE ? remaining_length : INITIAL_BINARY_BYTES_SIZE;

        // Copy the data into the initial_binary_bytes buffer
        memcpy(initial_binary_bytes, &packet->payload[a + 4], copy_length);
        
        // Update the length of the copied data
        initial_binary_bytes_len = copy_length;
      }
