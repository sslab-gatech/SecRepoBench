if (ch >= '0' && ch <= '9') {
        ch -= '0';
      } else if (ch >= 'A' && ch <= 'F') {
        ch -= 'A' - 10;
      } else if (ch >= 'a' && ch <= 'f') {
        ch -= 'a' - 10;
      } else {
        return FP_VAL;
      }

      if (k == FP_SIZE) {
          return FP_VAL;
      }

      if (j == 0) {
          fpnumber->dp[k] = (fp_digit)ch;
      } else {
          fpnumber->dp[k] = (fpnumber->dp[k] << 4) | (fp_digit)ch;
      }

      if (++j == 2) {
          j = 0;
          k++;
      }
