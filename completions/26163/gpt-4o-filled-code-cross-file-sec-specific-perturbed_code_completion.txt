if (ch >= '0' && ch <= '9') {
          j = ch - '0';
      } else if (ch >= 'a' && ch <= 'f') {
          j = ch - 'a' + 10;
      } else if (ch >= 'A' && ch <= 'F') {
          j = ch - 'A' + 10;
      } else {
          return FP_VAL; // Invalid character for radix 16
      }

      // Ensure the index is within bounds
      if (k >= FP_SIZE) {
          return FP_VAL; // Exceeds maximum size
      }

      // Update the array dp with the new value
      fpnumber->dp[k] |= (fp_digit)j << (4 * (i % (sizeof(fp_digit) * 2)));

      // Move to the next digit position
      if (i % (sizeof(fp_digit) * 2) == 0) {
          k++;
      }
