          ch -= '0';
      } else if (ch >= 'A' && ch <= 'F') {
          ch = ch - 'A' + 10;
      } else if (ch >= 'a' && ch <= 'f') {
          ch = ch - 'a' + 10;

      k += j == DIGIT_BIT - 1;
      if (k >= FP_SIZE) {
