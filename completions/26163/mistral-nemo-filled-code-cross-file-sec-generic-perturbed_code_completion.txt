if (ch >= '0' && ch <= '9') {
  d = ch - '0';
} else if (ch >= 'A' && ch <= 'F') {
  d = ch - 'A' + 10;
} else if (ch >= 'a' && ch <= 'f') {
  d = ch - 'a' + 10;
} else {
  return FP_VAL;  // Invalid character
}

if (k + 1 > FP_SIZE) {
  return FP_OVERFLOW;  // Number too large
}

if (j + DIGIT_BIT > FP_DIGIT_BIT) {
  j = 0;
  k++;
}

dp[k] |= d << j;
j += DIGIT_BIT;
