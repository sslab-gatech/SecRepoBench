if (ch >= '0' && ch <= '9') {
  d = ch - '0';
} else if (ch >= 'A' && ch <= 'F') {
  d = ch - 'A' + 10;
} else if (ch >= 'a' && ch <= 'f') {
  d = ch - 'a' + 10;
} else {
  return FP_VAL;  // Invalid character, return error
}

if (k + 1 > FP_SIZE) {
  return FP_OVERFLOW;  // Number of digits exceeds maximum size
}

if (i >= k) {
  d = 0;
} else {
  d = d << (4 * (k - i - 1));
}

dp[k - i - 1] |= d;
