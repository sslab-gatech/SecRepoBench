// <MASK>
int ch_val;
if (ch >= '0' && ch <= '9') {
    ch_val = ch - '0';
} else if (ch >= 'A' && ch <= 'F') {
    ch_val = 10 + (ch - 'A');
} else if (ch >= 'a' && ch <= 'f') {
    ch_val = 10 + (ch - 'a');
} else {
    return FP_VAL; // Return error code if character is not valid
}

if (k >= FP_SIZE) {
    return FP_VAL; // Return error code if number of digits exceeds maximum size
}

fpnumber->dp[k] |= (fp_digit)ch_val << (j * 4);
j++;
if (j >= 8) {
    j = 0;
    k++;
}
