int digit_value;

      if (ch >= '0' && ch <= '9') {
          digit_value = ch - '0';
      } else if (ch >= 'A' && ch <= 'F') {
          digit_value = ch - 'A' + 10;
      } else if (ch >= 'a' && ch <= 'f') {
          digit_value = ch - 'a' + 10;
      } else {
          return FP_VAL;
      }

      if (k >= FP_SIZE) {
          return FP_VAL;
      }

      fpnumber->dp[k] |= (fp_digit)digit_value << j;
      j += 4;

      if (j >= DIGIT_BIT) {
          j = 0;
          k++;
      }
