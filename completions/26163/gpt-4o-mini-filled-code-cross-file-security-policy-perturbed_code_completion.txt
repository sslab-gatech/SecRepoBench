          dp[k] = (fp_digit)(ch - '0');
      } else if (ch >= 'A' && ch <= 'F') {
          dp[k] = (fp_digit)(10 + ch - 'A');
      } else if (ch >= 'a' && ch <= 'f') {
          dp[k] = (fp_digit)(10 + ch - 'a');
      } else {
          return FP_VAL; // Invalid character
      }

      } else {
