if ((ch >= '0') && (ch <= '9')) {
    j |= (ch - '0') << (k << 2);
} else if ((ch >= 'A') && (ch <= 'F')) {
    j |= (ch - 'A' + 10) << (k << 2);
} else if ((ch >= 'a') && (ch <= 'f')) {
    j |= (ch - 'a' + 10) << (k << 2);
} else {
    return FP_VAL;
}
++k;
if (k == (int)sizeof(fp_digit)) {
    k = 0;
    if (fp_grow(fpnumber, fpnumber->used + 1) != FP_OKAY) {
        return FP_MEM;
    }
    fpnumber->dp[fpnumber->used++] = j;
    j = 0;
}
