if (ch >= '0' && ch <= '9') {
    ch -= '0';
} else if (ch >= 'a' && ch <= 'f') {
    ch = ch - 'a' + 10;
} else if (ch >= 'A' && ch <= 'F') {
    ch = ch - 'A' + 10;
} else {
    return FP_VAL; // Invalid character
}

if (k >= FP_SIZE) {
    return FP_VAL; // Exceeds maximum size
}

fpnumber->dp[k] += (ch << j);
if (++j == 4) {
    j = 0;
    ++k;
}
