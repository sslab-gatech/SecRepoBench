if (ch >= '0' && ch <= '9') {
          y = ch - '0';
      } else if (ch >= 'A' && ch <= 'F') {
          y = ch - 'A' + 10;
      } else if (ch >= 'a' && ch <= 'f') {
          y = ch - 'a' + 10;
      } else {
          return FP_VAL; // Invalid character for base 16
      }

      j = (i * 4) % DIGIT_BIT;
      k = (i * 4) / DIGIT_BIT;

      if (k >= FP_SIZE) {
          return FP_VAL; // Exceeds maximum size
      }

      fpnumber->dp[k] |= ((fp_digit)y << j);
  }
