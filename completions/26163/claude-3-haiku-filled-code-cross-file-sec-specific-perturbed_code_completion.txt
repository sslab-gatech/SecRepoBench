// <MASK>
int ch_val;
if (ch >= '0' && ch <= '9') {
    ch_val = ch - '0';
} else if (ch >= 'A' && ch <= 'F') {
    ch_val = 10 + (ch - 'A');
} else if (ch >= 'a' && ch <= 'f') {
    ch_val = 10 + (ch - 'a');
} else {
    return FP_VAL; // return error code if character is not valid
}

int pos = k * 4;
if (pos >= FP_SIZE * DIGIT_BIT) {
    return FP_VAL; // return error code if number of digits exceeds maximum size
}

fpnumber->dp[j] |= (fp_digit)ch_val << pos;
k++;
j = k / DIGIT_BIT;
