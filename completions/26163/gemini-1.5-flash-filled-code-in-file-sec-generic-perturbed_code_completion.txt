          j = ch - '0';
      } else if (ch >= 'A' && ch <= 'F') {
          j = ch - 'A' + 10;
      } else if (ch >= 'a' && ch <= 'f') {
          j = ch - 'a' + 10;
      } else {
          return FP_VAL;
      }
      k = (j << (k & (DIGIT_BIT - 1))) | (k >> (DIGIT_BIT - (k & (DIGIT_BIT - 1))));
      k += (j << (k & (DIGIT_BIT - 1)));
      if (k >= DIGIT_BIT) {
          fpnumber->dp[j] = (fp_digit)k;
          k >>= DIGIT_BIT;
          j++;
      }
      if (j >= FP_SIZE) {
          return FP_VAL;
