if (ch >= '0' && ch <= '9') {
          y = ch - '0';
      } else if (ch >= 'A' && ch <= 'Z') {
          y = ch - 'A' + 10;
      } else if (ch >= 'a' && ch <= 'z') {
          y = ch - 'a' + 36;
      } else {
          return FP_VAL; // invalid character
      }

      if (y >= radix) {
          return FP_VAL; // character not valid for this radix
      }

      /* add to the number */
      a->dp[k] = (a->dp[k] * radix) + y;

      /* increment the digit count */
      if (a->dp[k] == 0) {
          k++;
      }
