if (ch >= '0' && ch <= '9') {
  ch = ch - '0';
} else if (ch >= 'A' && ch <= 'F') {
  ch = ch - 'A' + 10;
} else if (ch >= 'a' && ch <= 'f') {
  ch = ch - 'a' + 10;
} else {
  // Invalid character
  return FP_VAL;
}

// Set the value in the appropriate position
fpnumber->dp[k] |= ((fp_digit)ch) << j;
j += 4; // Each hex digit is 4 bits

// If we've filled a digit, move to the next one
if (j >= DIGIT_BIT) {
  j = 0;
  k++;
  if (k >= FP_SIZE) {
    // Overflow
    return FP_VAL;
  }
}
