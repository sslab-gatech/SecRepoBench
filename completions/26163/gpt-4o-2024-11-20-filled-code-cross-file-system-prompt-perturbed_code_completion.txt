if (ch >= '0' && ch <= '9') {
          j = ch - '0';
      } else if (ch >= 'A' && ch <= 'F') {
          j = ch - 'A' + 10;
      } else if (ch >= 'a' && ch <= 'f') {
          j = ch - 'a' + 10;
      } else {
          return FP_VAL; // Invalid character for radix 16
      }

      if (k >= FP_SIZE) {
          return FP_VAL; // Exceeds maximum size
      }

      fpnumber->dp[k] |= (fp_digit)j << (4 * (i % (DIGIT_BIT / 4)));
      if ((i % (DIGIT_BIT / 4)) == 0) {
          k++;
      }
