if (ch >= '0' && ch <= '9') {
          dp[k++] = ch - '0';
      } else if (ch >= 'A' && ch <= 'F') {
          dp[k++] = ch - 'A' + 10;
      } else if (ch >= 'a' && ch <= 'f') {
          dp[k++] = ch - 'a' + 10;
      } else {
          return FP_VAL; // Invalid character for base 16
      }

      if (k > FP_SIZE) {
          return FP_VAL; // Exceeds maximum size
      }
