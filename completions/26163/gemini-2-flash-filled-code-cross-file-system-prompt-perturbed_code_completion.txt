int digit;

      if (isdigit((unsigned char)ch)) {
          digit = ch - '0';
      } else if (ch >= 'a' && ch <= 'f') {
          digit = ch - 'a' + 10;
      } else if (ch >= 'A' && ch <= 'F') {
          digit = ch - 'A' + 10;
      } else {
          return FP_VAL;
      }

      if (j == FP_DIGIT_BIT) {
          j = 0;
          ++k;
          if (k >= FP_SIZE) {
              return FP_RANGE;
          }
      }

      fpnumber->dp[k] |= ((fp_digit)digit) << j;
      j += 4;
