if (ch >= '0' && ch <= '9') {
          ch -= '0';
      } else if (ch >= 'A' && ch <= 'F') {
          ch -= 'A' - 10;
      } else if (ch >= 'a' && ch <= 'f') {
          ch -= 'a' - 10;
      } else {
          return FP_VAL;
      }

      if (k == 0) {
          fpnumber->dp[0] = ch;
          k = 1;
      } else {
          if (k >= FP_SIZE) {
              return FP_VAL;
          }
          j = fp_mul_d(fpnumber, 16, fpnumber);
          if (j != FP_OKAY) {
              return j;
          }
          j = fp_add_d(fpnumber, ch, fpnumber);
          if (j != FP_OKAY) {
              return j;
          }
          k++;
      }
