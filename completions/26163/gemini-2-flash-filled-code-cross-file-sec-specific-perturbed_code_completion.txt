int digit_value;

      if (isdigit((unsigned char)ch)) {
          digit_value = ch - '0';
      } else if (isalpha((unsigned char)ch)) {
          ch = (char)tolower((unsigned char)ch);
          digit_value = ch - 'a' + 10;
      } else {
          return FP_VAL;
      }

      if (digit_value < 0 || digit_value >= 16) {
          return FP_VAL;
      }

      if (j == FP_DIGIT_BIT_SIZE) {
          ++k;
          if (k >= FP_SIZE) {
              return FP_VAL;
          }
          j = 0;
      }

      fpnumber->dp[k] |= ((fp_digit)digit_value) << j;
      j += 4;
