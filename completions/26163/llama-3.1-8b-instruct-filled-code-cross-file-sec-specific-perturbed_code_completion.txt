// <MASK>
if (ch >= '0' && ch <= '9') {
    d = ch - '0';
} else if (ch >= 'A' && ch <= 'F') {
    d = ch - 'A' + 10;
} else if (ch >= 'a' && ch <= 'f') {
    d = ch - 'a' + 10;
} else {
    return FP_VAL; // invalid character
}

if (k >= FP_SIZE) {
    return FP_VAL; // number of digits exceeds maximum size
}

if (j == 0 && d == 0) {
    return FP_VAL; // leading zero
}

fpnumber->dp[k] = d;
++k;
