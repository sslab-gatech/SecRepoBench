u_int16_t name_offset = body_offset + 13;
for(i=0; (i<20) && (name_offset < packet->payload_packet_len); i++) {
  if(packet->payload[name_offset] == 0x1b) break;
  name_offset++;
}

if(name_offset < packet->payload_packet_len) {
  u_int8_t realm_len;
  name_offset += 1;
  realm_len = packet->payload[name_offset];

  if((name_offset + realm_len + 1) < packet->payload_packet_len) {
    char realm_str[48];
    u_int16_t realm_str_len = realm_len;

    if(realm_str_len > sizeof(realm_str) - 1)
      realm_str_len = sizeof(realm_str) - 1;

    name_offset += 1;
    strncpy(realm_str, (char*)&packet->payload[name_offset], realm_str_len);
    realm_str[realm_str_len] = '\0';

    for(i=0; i<realm_str_len; i++)
      realm_str[i] = tolower(realm_str[i]);

    snprintf(flow->protos.kerberos.domain, sizeof(flow->protos.kerberos.domain), "%s", realm_str);
  }
}

if(flow->kerberos_buf.pktbuf != NULL) {
  ndpi_free(flow->kerberos_buf.pktbuf);
  packet->payload = original_packet_payload;
  packet->payload_packet_len = original_payload_packet_len;
  flow->kerberos_buf.pktbuf = NULL;
}
