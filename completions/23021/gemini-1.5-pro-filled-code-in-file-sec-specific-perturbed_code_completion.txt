u_int16_t name_offset = body_offset + 11;
		for(i=0; (i<20) && (name_offset < packet->payload_packet_len); i++) {
		  if(packet->payload[name_offset] != 0x1b)
		    name_offset++; /* ASN.1 */
		}

		if(name_offset < packet->payload_packet_len) {
		  u_int realm_len;

		  // Adjust offset for realm data
		  name_offset += 9;

		  if (name_offset < packet->payload_packet_len) {
		    realm_len = packet->payload[name_offset];

		    if((name_offset + realm_len) < packet->payload_packet_len) {
		      char realm_str[48];
		      size_t realm_len_safe = realm_len;

		      if (realm_len_safe > sizeof(realm_str) - 1) {
			realm_len_safe = sizeof(realm_str) - 1;
		      }

		      strncpy(realm_str, (char*)&packet->payload[name_offset + 1], realm_len_safe);
		      realm_str[realm_len_safe] = '\0';
		      for(i=0; i<realm_len_safe; i++) realm_str[i] = tolower(realm_str[i]);

#ifdef KERBEROS_DEBUG
		      printf("[TGS-REQ][Kerberos Realm][len: %zu][%s]\n", realm_len_safe, realm_str);
#endif
		      snprintf(flow->protos.kerberos.domain, sizeof(flow->protos.kerberos.domain), "%s", realm_str);
		    }
		  }
		}

		if(flow->kerberos_buf.pktbuf != NULL) {
		  ndpi_free(flow->kerberos_buf.pktbuf);
		  packet->payload = original_packet_payload;
		  packet->payload_packet_len = original_payload_packet_len;
		  flow->kerberos_buf.pktbuf = NULL;
		}
