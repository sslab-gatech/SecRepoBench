ret = parser_ctxt->freeElems;
parser_ctxt->freeElems = ret->next;
if (parser_ctxt->freeElems != NULL) {
    parser_ctxt->freeElems->prev = NULL;
}
parser_ctxt->freelistNr--;
memset(ret, 0, sizeof(xmlNode));
ret->type = XML_ELEMENT_NODE;
if (lname == NULL) {
    if (parser_ctxt->dictNames)
        ret->name = xmlDictLookup(parser_ctxt->dict, localname, -1);
    else
        ret->name = xmlStrdup(localname);
} else {
    ret->name = lname;
    lname = NULL;
}
if (ret->name == NULL) {
    xmlSAX2ErrMemory(parser_ctxt, "xmlSAX2StartElementNs");
    return;
}
