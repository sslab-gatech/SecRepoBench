if (ip_port_separator != NULL) {
  ip_len = ndpi_min(sizeof(flow->protos.softether.ip) - 1,
                    (size_t)(ip_port_separator - (char const *)packet_data->payload - NDPI_STATICSTRING_LEN("IP=")));
