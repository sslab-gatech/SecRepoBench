xmlParserInputBufferGrow(xmlParserInputBufferPtr buf, int len) {
    xmlBufPtr buffer;
    int ret;

    if (buf == NULL)
        return(-1);
    if (buf->buffer == NULL)
        return(0);
    if (buf->end - buf->cur < len) {
        buffer = buf->buffer;
        ret = xmlBufGrow(buffer, len);
        if (ret < 0) {
            xmlErrMemory(ctxt, NULL);
            return(-1);
        }
        buf->buffer = buffer;
        buf->end = xmlBufEnd(buffer);
    }
    return(0);
}
