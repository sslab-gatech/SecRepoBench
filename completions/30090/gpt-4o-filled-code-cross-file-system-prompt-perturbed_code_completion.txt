if (k && v && v->type == MSGPACK_OBJECT_STR) {
    /* Parse the time string */
    memset(&tm, '\0', sizeof(struct tm));
    ret = strptime(v->via.str.ptr, parser->time_fmt, &tm);
    if (!ret) {
        flb_warn("[parser] could not parse time string '%s'", v->via.str.ptr);
        time_lookup = 0;
    }
    else {
        time_lookup = mktime(&tm);
        if (time_lookup == -1) {
            flb_warn("[parser] mktime failed for time string '%s'", v->via.str.ptr);
            time_lookup = 0;
        }
    }
}
else {
    time_lookup = 0;
}
