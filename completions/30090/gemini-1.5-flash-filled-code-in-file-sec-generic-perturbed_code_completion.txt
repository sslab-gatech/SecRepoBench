        if (v->type == MSGPACK_OBJECT_STR) {
            len = v->via.str.size;
            if (len > 254) {
                len = 254;
            }
            strncpy(tmp, v->via.str.ptr, len);
            tmp[len] = '\0';
            time_lookup = strtol(tmp, NULL, 10);
            if (time_lookup == 0 && errno == EINVAL) {
                flb_plg_error(parser->parent, "invalid time format");
                time_lookup = 0;
            }
        }
        else if (v->type == MSGPACK_OBJECT_POSITIVE_INTEGER) {
            time_lookup = v->via.u64;
        }
        else if (v->type == MSGPACK_OBJECT_FLOAT) {
            time_lookup = (time_t) v->via.f64;
            tmfrac = v->via.f64 - time_lookup;
        }
        else {
            flb_plg_error(parser->parent, "invalid time format");
            time_lookup = 0;
        }
    }
    else {
        time_lookup = 0;
