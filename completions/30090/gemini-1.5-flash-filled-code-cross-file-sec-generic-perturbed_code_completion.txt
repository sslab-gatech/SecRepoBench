if (skip >= 0 && skip < map_size) {
        if (v->type == MSGPACK_OBJECT_STR) {
            ret = flb_utils_time_to_ts(v->via.str.ptr, v->via.str.size,
                                        &time_lookup, &tmfrac);
            if (ret == -1) {
                flb_warn("[flb_parser_json] time parsing error '%.*s'",
                         (int)v->via.str.size, v->via.str.ptr);
                time_lookup = 0;
            }
        }
        else if (v->type == MSGPACK_OBJECT_POSITIVE_INTEGER) {
            time_lookup = v->via.u64;
        }
        else if (v->type == MSGPACK_OBJECT_NEGATIVE_INTEGER) {
            time_lookup = v->via.i64;
        }
        else {
            flb_warn("[flb_parser_json] invalid time format");
            time_lookup = 0;
        }
    }
