if (!((yr_le64toh(tilde_header->Valid) >> bit_check) & 0x01))
{
  // This table isn't present (bit not set), skip it.
  continue;
}

// Check that we can safely read the row count.
if (!fits_in_pe(pe_struct, (uint8_t*)(row_offset + matched_bits), sizeof(uint32_t)))
{
  // Can't read row count, stop parsing further tables.
  break;
}

num_rows = yr_le32toh(*(row_offset + matched_bits));

// If there are zero rows, skip processing this table but still advance matched_bits.
if (num_rows == 0)
{
  matched_bits++;
  continue;
}
