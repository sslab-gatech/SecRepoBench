if (!((yr_le64toh(tilde_header->Valid) >> bit_check) & 0x01))
  continue;

num_rows = yr_le32toh(*(row_offset + matched_bits));

// Ensure that the table's row count does not exceed a reasonable limit
// to prevent potential heap overflow.
if (num_rows > MAX_ROWS_LIMIT)
{
  return;
}

// Validate that the table exists within the PE file boundaries.
if (!fits_in_pe(pe_struct, table_offset, num_rows * MIN_TABLE_ROW_SIZE))
{
  return;
}
