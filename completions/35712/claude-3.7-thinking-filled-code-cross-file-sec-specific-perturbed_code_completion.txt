CASE(OP_EXT1, Z) {
  FETCH_Z();
  instruction = READ_B();
  switch (instruction) {
    CASE(OP_LAMBDA, BB)
    c = OP_L_LAMBDA | OP_L_CAPTURE;
    goto L_MAKE_LAMBDA;

    CASE(OP_BLOCK, BB) {
      c = OP_L_BLOCK | OP_L_CAPTURE;
      goto L_MAKE_LAMBDA;
    }
    CASE(OP_METHOD, BB) {
      c = OP_L_METHOD | OP_L_CAPTURE;
      goto L_MAKE_LAMBDA;
    }
    CASE(OP_LAMBDA, BB)
    c = OP_L_LAMBDA | OP_L_STRICT;
    goto L_MAKE_LAMBDA;

    CASE(OP_BLOCK, BB) {
      c = OP_L_BLOCK | OP_L_STRICT;
      goto L_MAKE_LAMBDA;
    }
    CASE(OP_METHOD, BB) {
      c = OP_L_METHOD | OP_L_STRICT;
      goto L_MAKE_LAMBDA;
    }
    default:
      break;
  }
  pc--;
  NEXT;
}
