CASE(OP_EXT1, Z) {
      uint8_t ext_op = *pc++;
      switch (ext_op) {
        case OP_EXT1_LOADI:
          FETCH_B();
          SET_FIXNUM_VALUE(regs[a], b);
          break;
        case OP_EXT1_MOVE:
          FETCH_BB();
          regs[a] = regs[b];
          break;
      }
      pc--;
      NEXT;
    }

    CASE(OP_EXT2, Z) {
      uint16_t ext_op = (uint16_t)(*pc++) << 8 | *pc++;
      switch (ext_op) {
        case OP_EXT2_LOADL:
          FETCH_BB();
          switch (pool[b].tt) {
            case IREP_TT_INT32:
              regs[a] = mrb_int_value(mrb, (mrb_int)pool[b].u.i32);
              break;
            default:
              regs[a] = mrb_nil_value();
              break;
          }
          break;
      }
      pc -= 2;
      NEXT;
    }

    CASE(OP_EXT3, Z) {
      uint32_t ext_op = (uint32_t)(*pc++) << 16 | (uint16_t)(*pc++) << 8 | *pc++;
      switch (ext_op) {
        case OP_EXT3_ARYCAT:
          FETCH_B();
          mrb_ary_push(mrb, regs[a], regs[a+1]);
          break;
      }
      pc -= 3;
      NEXT;
    }
