CASE(OP_EXT1, Z) {
  uint8_t opcode = *pc++;
  
  switch (opcode) {
    default:
      mrb_raise(mrb, E_RUNTIME_ERROR, "Unknown OP_EXT1 instruction: %d", opcode);
  }
  pc--;  // Adjust pc because we read an extra byte
  NEXT;
}
