CASE(OP_EXT1, Z) {
  insn = READ_B();
  switch (insn) {
  case OP_ENTER:
    FETCH_W();
    goto L_ENTER;
  default:
    pc--;
    NEXT;
  }
}

CASE(OP_EXT2, Z) {
  insn = READ_B();
  switch (insn) {
  case OP_ENTER:
    FETCH_W();
    goto L_ENTER;
  default:
    pc--;
    NEXT;
  }
}

CASE(OP_EXT3, Z) {
  insn = READ_B();
  switch (insn) {
  case OP_ENTER:
    FETCH_W();
    goto L_ENTER;
  case OP_AREF:
    FETCH_BBB();
    goto L_AREF;
  case OP_ASET:
    FETCH_BBB();
    goto L_ASET;
  case OP_LAMBDA:
    FETCH_BB();
    c = OP_L_LAMBDA | OP_L_CAPTURE;
    goto L_MAKE_LAMBDA;
  case OP_BLOCK:
    FETCH_BB();
    c = OP_L_BLOCK | OP_L_CAPTURE;
    goto L_MAKE_LAMBDA;
  case OP_METHOD:
    FETCH_BB();
    c = OP_L_METHOD | OP_L_CAPTURE;
    goto L_MAKE_LAMBDA;
  case OP_SEND:
    FETCH_BBB();
    goto L_SEND;
  case OP_SUPER:
    FETCH_BB();
    goto L_SUPER;
  default:
    pc--;
    NEXT;
  }
}
