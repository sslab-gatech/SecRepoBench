CASE(OP_EXT1, B) {
  FETCH_B();
  switch (b) {
    case EXT_OP_INSTRUCTION:
      FETCH_BBB();
      pc--; // Adjust for the extra byte read
      goto L_OP_INSTRUCTION_BODY;
    default:
      mrb_raise(mrb, E_NOTIMP_ERROR, "unknown extended opcode");
      goto L_RAISE;
  }
  NEXT;
}
