CASE(OP_EXT1, B) {
  uint8_t ext_opcode;
  FETCH_B(ext_opcode);
  pc--;
  switch (ext_opcode) {
    case 0x00: // OP_EXT1_SEND
      FETCH_BB(b, c);
      mid = syms[b];
      c = 1;
      goto L_SENDB_SYM;
    case 0x01: // OP_EXT1_SENDB
      FETCH_BB(b, c);
      mid = syms[b];
      goto L_SENDB_SYM;
    case 0x02: // OP_EXT1_ENTER
      FETCH_W(a);
      goto L_ENTER;
    default:
      mrb_raise(mrb, E_SCRIPT_ERROR, "unknown extended opcode");
  }
  NEXT;
}
