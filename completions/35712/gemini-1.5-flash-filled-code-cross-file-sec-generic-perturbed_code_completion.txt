CASE(OP_EXT1, X) {
      instruction = pc[1];
      pc++;
      switch (instruction) {
        CASE(OP_EXT1_CALL, BBB) {
          c = CALL_MAXARGS;
          goto L_SEND;
        }
        CASE(OP_EXT1_SENDB, BBB) {
          c = CALL_MAXARGS;
          goto L_SENDB;
        }
        CASE(OP_EXT1_SEND, BBB) {
          goto L_SEND;
        }
        CASE(OP_EXT1_ENTER, W) {
          goto L_OP_ENTER;
        }
        CASE(OP_EXT1_BLKPUSH, BS) {
          goto L_OP_BLKPUSH;
        }
        default:
          mrb_assert(!"unhandled ext1 opcode");
          break;
      }
      pc--;
      NEXT;
    }
    CASE(OP_EXT2, X) {
      instruction = pc[1];
      pc++;
      switch (instruction) {
        CASE(OP_EXT2_GETUPVAR, BBB) {
          goto L_OP_GETUPVAR;
        }
        CASE(OP_EXT2_SETUPVAR, BBB) {
          goto L_OP_SETUPVAR;
        }
        default:
          mrb_assert(!"unhandled ext2 opcode");
          break;
      }
      pc--;
      NEXT;
    }
    CASE(OP_EXT3, X) {
      instruction = pc[1];
      pc++;
      switch (instruction) {
        CASE(OP_EXT3_JMPUW, S) {
          goto L_OP_JMPUW;
        }
        default:
          mrb_assert(!"unhandled ext3 opcode");
          break;
      }
      pc--;
      NEXT;
    }
