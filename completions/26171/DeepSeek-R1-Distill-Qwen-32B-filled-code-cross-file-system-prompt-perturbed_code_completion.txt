if (filters == NULL) {
  /* Single stream case */
  int32_t cbytes = sw32_(src);
  int32_t run = src[4];
  if (run == 0) {
    /* Not a run, decompress */
    if (cbytes == 0) {
      /* Uncompressed block */
      memcpy(dest + destinationoffset, src + 4, (unsigned int)bsize);
    } else {
      /* Decompress using the appropriate format */
      if (blosc_decompress(src + 4, dest + destinationoffset, bsize) < 0) {
        return -1;
      }
    }
  } else {
    /* It's a run, fill with the same value */
    memset(dest + destinationoffset, run, (unsigned int)bsize);
  }
  ntbytes = bsize;
}
