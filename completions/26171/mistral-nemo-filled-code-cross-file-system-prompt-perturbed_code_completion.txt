/* Iterate over each stream */
  for (int j = 0; j < context->nstreams; j++) {
    int32_t csize = sw32_(src + 12 + j * 4);   /* compressed buffer size */
    int32_t cbytes = blosc_d(thread_context, bsize, leftoverblock, src + csize, csize, 0,

    if (cbytes < 0) {
      return cbytes;         /* error in blosc_d */
    }
    destinationoffset += cbytes;
  }
