if (delta > 0) {
    // Calculate the new keyframe positions and values based on the pre-behavior type
    switch ((*it).pre) {
        case LWO::PrePostBehaviour_OffsetRepeat:
            // OffsetRepeat behavior: repeat the envelope with an offset
            for (auto& key : (*it).keys) {
                key.time += delta;
            }
            break;
        case LWO::PrePostBehaviour_Repeat:
            // Repeat behavior: repeat the envelope
            for (auto& key : (*it).keys) {
                key.time += delta;
            }
            break;
        case LWO::PrePostBehaviour_Oscillate:
            // Oscillate behavior: reverse the envelope periodically
            for (auto& key : (*it).keys) {
                key.time = firsttime + fmod(key.time - firsttime, 2.0 * delta) - delta;
            }
            break;
        default:
            // No pre-behavior: do nothing
            break;
    }
}
