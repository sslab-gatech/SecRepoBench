            switch ((*it).pre) {

            case LWO::PrePostBehaviour_Repeat:
                // Add offset keys to match LightWave pre behavior
                time = firsttime - delta;
                while (time >= first) {
                    unsigned int old_key = (unsigned int)((my_last - time) / delta * (double)(num_keys - 1u));

                    LWO::Key key = (*it).keys[old_key];

                    (*it).keys.insert((*it).keys.begin(), key);
                    time -= delta;
                }
                break;

                    if (!(old_key & 0x1)) {
                        old_key = (num_keys - 1u) - old_key;
                    } else {

                    }

                    key.time = time;
