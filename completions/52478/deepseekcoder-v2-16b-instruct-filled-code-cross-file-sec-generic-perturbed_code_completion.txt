// Handles the pre-behavior of LightWave envelopes by inserting additional keyframes
        // for pre-defined behaviors such as OffsetRepeat, Repeat, and Oscillate. This involves
        // calculating the time delta between keys and adjusting keyframe values accordingly.
        // The logic uses the start time, end time, and value delta to calculate new keyframe
        // positions and values, adapting to the pre-behavior type.
        switch ((*it).pre) {
        case LWO::PrePostBehaviour_OffsetRepeat:
        case LWO::PrePostBehaviour_Repeat:
        case LWO::PrePostBehaviour_Oscillate:
            // Calculate the time delta between keys
            double time_delta = (my_last - firsttime) / ((*it).keys.size() - 1);

            // Calculate the value delta between keys
            float value_delta = (*it).keys.back().value - (*it).keys.front().value;

            // Insert new keyframes based on the pre-behavior type
            for (size_t i = 0; i < (*it).keys.size(); ++i) {
                LWO::Key new_key;
                new_key.time = firsttime + i * time_delta;
                new_key.value = (*it).keys.front().value + (i * value_delta);
                new_key.inter = (*it).keys[i].inter;
                new_key.params[0] = (*it).keys[i].params[0];
                new_key.params[1] = (*it).keys[i].params[1];
                new_key.params[2] = (*it).keys[i].params[2];
                new_key.params[3] = (*it).keys[i].params[3];
                new_key.params[4] = (*it).keys[i].params[4];

                (*it).keys.insert((*it).keys.begin() + i, new_key);
                ++i; // Skip the inserted keyframe
            }
            break;

        default:
            // silence compiler warning
            break;
        }
