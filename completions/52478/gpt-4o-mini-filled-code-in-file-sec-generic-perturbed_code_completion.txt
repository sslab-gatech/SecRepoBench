// Calculate the new keyframe positions and values based on the pre-behavior type
        switch ((*it).pre) {
            case LWO::PrePostBehaviour_OffsetRepeat:
                // Insert keyframes for offset repeat behavior
                (*it).keys.insert((*it).keys.begin(), {firsttime - delta, (*it).keys.front().value});
                break;
            case LWO::PrePostBehaviour_Repeat:
                // Insert keyframes for repeat behavior
                (*it).keys.insert((*it).keys.begin(), {firsttime - delta, (*it).keys.front().value});
                (*it).keys.push_back({my_last + delta, (*it).keys.back().value});
                break;
            case LWO::PrePostBehaviour_Oscillate:
                // Insert keyframes for oscillate behavior
                (*it).keys.insert((*it).keys.begin(), {firsttime - delta, (*it).keys.front().value});
                (*it).keys.push_back({my_last + delta, (*it).keys.front().value});
                break;
            default:
                // Handle linear and constant behaviors
                break;
        }
