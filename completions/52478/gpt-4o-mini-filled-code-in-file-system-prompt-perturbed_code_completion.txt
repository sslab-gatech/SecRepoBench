        if ((*it).pre == LWO::PrePostBehaviour_Linear) {
            LWO::Key preKey;
            preKey.time = firsttime - delta;
            preKey.value = (*it).keys.front().value; // Use the first key's value
            (*it).keys.insert((*it).keys.begin(), preKey);
        } else if ((*it).pre == LWO::PrePostBehaviour_Reset) {
            LWO::Key preKey;
            preKey.time = firsttime - delta;
            preKey.value = 0.f; // Reset value
            (*it).keys.insert((*it).keys.begin(), preKey);
        } else if ((*it).pre == LWO::PrePostBehaviour_Constant) {
            LWO::Key preKey;
