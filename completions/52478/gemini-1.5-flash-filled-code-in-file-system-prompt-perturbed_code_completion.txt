        case LWO::PrePostBehaviour_Repeat:
        case LWO::PrePostBehaviour_Oscillate: {
            double firsttime = (*it).keys.front().time;
            double lasttime = (*it).keys.back().time;
            double val_delta = (*it).keys.back().value - (*it).keys.front().value;

            // insert a keyframe at the beginning
            LWO::Key k;
            k.time = firsttime - delta;
            k.value = (*it).keys.front().value - val_delta;
            (*it).keys.push_front(k);

            // insert a keyframe at the end
            k.time = lasttime + delta;
            k.value = (*it).keys.back().value + val_delta;
            (*it).keys.push_back(k);
        }
            break;

        case LWO::PrePostBehaviour_Linear:
        case LWO::PrePostBehaviour_Constant:
        case LWO::PrePostBehaviour_Reset:
