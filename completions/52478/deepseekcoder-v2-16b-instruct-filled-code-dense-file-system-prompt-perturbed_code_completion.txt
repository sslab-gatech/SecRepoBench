// Handles the pre-behavior of LightWave envelopes by inserting additional keyframes
        // for pre-defined behaviors such as OffsetRepeat, Repeat, and Oscillate. This involves
        // calculating the time delta between keys and adjusting keyframe values accordingly.
        // The logic uses the start time, end time, and value delta to calculate new keyframe
        // positions and values, adapting to the pre-behavior type.
        if (envl->pre == LWO::PrePostBehaviour_OffsetRepeat) {
            double offset = delta / (*it).keys.size();
            for (size_t i = 0; i < (*it).keys.size(); ++i) {
                (*it).keys[i].time = firsttime + i * offset;
            }
        } else if (envl->pre == LWO::PrePostBehaviour_Repeat) {
            double repeat_time = delta;
            for (size_t i = 0; i < (*it).keys.size(); ++i) {
                (*it).keys[i].time = firsttime + fmod((*it).keys[i].time - firsttime, repeat_time);
            }
        } else if (envl->pre == LWO::PrePostBehaviour_Oscillate) {
            double oscillate_time = delta / 2.0;
            for (size_t i = 0; i < (*it).keys.size(); ++i) {
                (*it).keys[i].time = firsttime + fmod((*it).keys[i].time - firsttime, oscillate_time * 2.0);
                if ((*it).keys[i].time > oscillate_time) {
                    (*it).keys[i].time = oscillate_time - ((*it).keys[i].time - oscillate_time);
                }
