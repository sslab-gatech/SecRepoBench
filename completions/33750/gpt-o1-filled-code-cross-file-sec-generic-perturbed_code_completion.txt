/* Free original mp_buf if it is different from tmp_out_buf */
    skip = -1;
    if (tmp_out_buf != mp_buf) {
        flb_free(mp_buf);
        mp_buf = NULL;
    }

    /* If parser has no time format, skip time resolution */
    if (!parser->time_fmt) {
        msgpack_unpacked_destroy(&result);
        return *out_size;
    }

    /* Determine the time key */
    if (parser->time_key) {
        time_key = parser->time_key;
    }
    else {
        time_key = "time";
    }
    slen = strlen(time_key);

    /* Locate the time key in the map */
    map_size = map.via.map.size;
    k = NULL;
    time_value = NULL;
    for (i = 0; i < map_size; i++) {
        msgpack_object *cur_key = &map.via.map.ptr[i].key;

        if (cur_key->type != MSGPACK_OBJECT_STR) {
            continue;
        }

        if (cur_key->via.str.size == slen &&
            strncmp(cur_key->via.str.ptr, time_key, slen) == 0) {
            time_value = &map.via.map.ptr[i].val;
            k = cur_key;
            /* Mark this index to skip if time key is not to be kept */
            if (parser->time_keep == FLB_FALSE) {
                skip = i;
            }
            break;
        }
    }
