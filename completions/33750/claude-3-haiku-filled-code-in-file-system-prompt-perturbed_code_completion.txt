/* Determine the time key to be used, defaulting to "time" if not specified. */
time_key = parser->time_key ? parser->time_key : "time";
slen = strlen(time_key);

/* Iterate over the map's key-value pairs to find the time key. */
for (i = 0, map_size = map.via.map.size; i < map_size; i++) {
    k = &map.via.map.ptr[i].key;
    if (k->type == MSGPACK_OBJECT_STR && k->via.str.size == slen &&
        strncmp(k->via.str.ptr, time_key, slen) == 0) {
        time_value = &map.via.map.ptr[i].val;
        if (parser->time_keep == FLB_FALSE) {
            skip = i;
        }
        break;
    }
    k = NULL;
    time_value = NULL;
}
