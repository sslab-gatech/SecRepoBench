/* Free the original message pack buffer if it differs from the output buffer. */
if (tmp_out_buf != mp_buf) {
    flb_free(mp_buf);
}

/* Check if time resolution is needed by examining the parser's time format. */
if (parser->time_fmt_full && strstr(parser->time_fmt_full, ".")) {
    time_key = "time";
    time_key_len = strlen(time_key);
}
else {
    time_key = NULL;
    time_key_len = 0;
}

/* Determine the time key to be used, defaulting to "time" if not specified. */
if (parser->time_key) {
    time_key = parser->time_key;
    time_key_len = strlen(time_key);
}
else {
    time_key = "time";
    time_key_len = strlen(time_key);
}

/* Iterate over the map's key-value pairs to find the time key. */
for (i = 0; i < map.via.map.size; i++) {
    k = map.via.map.ptr[i].key;
    if (k.type == MSGPACK_OBJECT_STR && k.via.str.size == time_key_len &&
        memcmp(k.via.str.ptr, time_key, time_key_len) == 0) {
        time_value = &map.via.map.ptr[i].val;
        skip = i;
        break;
    }
}

/* If no match is found, set pointers to NULL and continue the search. */
if (i >= map.via.map.size || !k || !time_value) {
    time_value = NULL;
    skip = -1;
}
