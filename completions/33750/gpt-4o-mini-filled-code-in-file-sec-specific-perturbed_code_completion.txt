for (i = 0; i < map_size; i++) {
        k = &map.via.map.ptr[i].key;
        if (k->type == MSGPACK_OBJECT_STR) {
            time_key = k->via.str.ptr;
            slen = k->via.str.size;

            if (slen == parser->time_key_len && 
                memcmp(time_key, parser->time_key, slen) == 0) {
                skip = i;
                time_value = &map.via.map.ptr[i].val;
                break;
            }
        }
    }

    if (i == map_size) {
        time_key = NULL;
        time_value = NULL;
    }
