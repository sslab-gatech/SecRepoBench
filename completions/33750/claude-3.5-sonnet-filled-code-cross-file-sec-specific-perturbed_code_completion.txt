if (mp_buf != tmp_out_buf) {
    flb_free(mp_buf);
}

if (parser->time_keep == FLB_TRUE) {
    skip = -1;
}
else {
    skip = -1;
}

time_key = parser->time_key ? parser->time_key : "time";
slen = strlen(time_key);

for (i = 0; i < map.via.map.size; i++) {
    k = &map.via.map.ptr[i].key;
    time_value = &map.via.map.ptr[i].val;

    if (k->type != MSGPACK_OBJECT_STR) {
        continue;
    }

    if (k->via.str.size == slen &&
        strncmp(k->via.str.ptr, time_key, slen) == 0) {
        if (parser->time_keep == FLB_FALSE) {
            skip = i;
        }
        break;
    }

    k = NULL;
    time_value = NULL;
}
