        time_key_len = strlen(time_key);
    }

    if (k.type!= MSGPACK_OBJECT_STR) {
        continue;

    /* Compare each key's size and content to the desired time key */
    if (k.via.str.size == len && flb_sds_cmp(k.via.str.ptr, time_key, len) == 0) {
