/* Free the original message pack buffer if it differs from the output buffer */
    if (mp_buf != tmp_out_buf) {
        flb_free(mp_buf);
    }

    /* If there's no configured time format, no time resolution is needed */
    if (!parser->time_fmt) {
        msgpack_unpacked_destroy(&result);
        return *out_size;
    }

    /* Determine the time key to be used */
    if (parser->time_key) {
        time_key = parser->time_key;
    }
    else {
        time_key = "time";
    }
    slen = strlen(time_key);

    /* Initialize variables to search for time key */
    map_size = map.via.map.size;
    skip = -1;
    k = NULL;
    time_value = NULL;

    /* Search the map for the time key */
    for (i = 0; i < map_size; i++) {
        msgpack_object *key_obj = &map.via.map.ptr[i].key;

        if (key_obj->type != MSGPACK_OBJECT_STR) {
            continue;
        }

        if (key_obj->via.str.size == slen &&
            strncmp(key_obj->via.str.ptr, time_key, slen) == 0) {
            time_value = &map.via.map.ptr[i].val;
            k = key_obj;

            if (parser->time_keep == FLB_FALSE) {
                skip = i;
            }
            break;
        }
    }
