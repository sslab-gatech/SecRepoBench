if (tmp_out_buf != mp_buf) {
    flb_free(mp_buf);
}

/* Do time resolution ? */
if (!parser->time_fmt) {
    msgpack_unpacked_destroy(&result);
    return *out_size;
}

/* Get the time key */
if (parser->time_key) {
    time_key = parser->time_key;
}
else {
    time_key = "time";
}
slen = strlen(time_key);

/* Get the map size */
map_size = map.via.map.size;

skip = -1;
time_value = NULL;
k = NULL;
for (i = 0; i < map_size; i++) {
    msgpack_object *key = &map.via.map.ptr[i].key;
    if (key->type != MSGPACK_OBJECT_STR) {
        continue;
    }

    if (key->via.str.size != slen) {
        continue;
    }

    if (strncmp(key->via.str.ptr, time_key, slen) == 0) {
        k = key;
        time_value = &map.via.map.ptr[i].val;
        if (parser->time_keep == FLB_FALSE) {
            skip = i;
        }
        break;
    }
}
