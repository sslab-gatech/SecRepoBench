    if (tmp_out_buf != mp_buf) {
        flb_free(mp_buf);
    }

    if (parser->time_key) {
        time_key = parser->time_key;
    } else {
        time_key = "time";
    }
    slen = strlen(time_key);

    // Determine the time key to be used, defaulting to "time" if not specified.
    map_size = map.via.map.size;
    for (i = 0; i < map_size; i++) {
        k = &map.via.map.ptr[i].key;
        if (k->type == MSGPACK_OBJECT_STR && k->via.str.size == slen &&
            memcmp(k->via.str.ptr, time_key, slen) == 0) {
            time_value = &map.via.map.ptr[i].val;
            skip = i; // Note the index to skip
            break;
        }
    }

    // If no match is found, set pointers to NULL and continue the search.
    if (i >= map_size) {
        k = NULL;
        time_value = NULL;
