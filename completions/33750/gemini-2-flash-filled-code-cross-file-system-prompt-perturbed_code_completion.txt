if (tmp_out_buf != mp_buf) {
        flb_free(mp_buf);
    }

    if (parser->time_fmt == FLB_TIME_FORMAT_NONE) {
        msgpack_unpacked_destroy(&result);
        return *out_size;
    }

    if (parser->time_key) {
        time_key = parser->time_key;
    }
    else {
        time_key = "time";
    }
    slen = strlen(time_key);

    /* Iterate keys to lookup time */
    map_size = map.via.map.size;
    k = NULL;
    time_value = NULL;
    skip = -1;
    for (i = 0; i < map_size; i++) {
        k = &map.via.map.ptr[i].key;

        if (k->type == MSGPACK_OBJECT_STR && k->via.str.size == slen &&
            strncmp(k->via.str.ptr, time_key, slen) == 0) {
            time_value = &map.via.map.ptr[i].val;
            if (parser->time_keep == FLB_FALSE) {
                skip = i;
            }
            break;
        }
        else {
            k = NULL;
            time_value = NULL;
        }
    }
