/* Determine time key */
    if (parser->time_key) {
        time_key = parser->time_key;
        slen = strlen(parser->time_key);
    }
    else {
        time_key = "time";
        slen = 4;
    }

    /* Prepare to locate time key/value in the map */
    map_size = map.via.map.size;
    skip = -1;
    k = NULL;
    time_value = NULL;

    /* Locate time key in the map */
    for (i = 0; i < map_size; i++) {
        /* Make sure the map key is a string */
        if (map.via.map.ptr[i].key.type != MSGPACK_OBJECT_STR) {
            continue;
        }

        /* Compare key with desired time_key */
        if (map.via.map.ptr[i].key.via.str.size == slen &&
            strncmp(map.via.map.ptr[i].key.via.str.ptr, time_key, slen) == 0) {
            k = &map.via.map.ptr[i].key;
            time_value = &map.via.map.ptr[i].val;
            if (parser->time_keep == FLB_FALSE) {
                skip = i;
            }
            break;
        }
    }
