if (checkcompoundtriple && scpd == 0 && !compound_words && !checkedstriple && (i > 1)) {
    // Detect a triple-letter sequence: if st[i] == st[i-1] and also
    // either st[i] == st[i-2] or st[i] == st[i+1], trigger a re-check.
    // This is a simplified approach to address potential triple-letter
    // combinations in compound words.
    char c_cur   = (i < st.size()) ? st[i] : '\0';
    char c_prev  = st[i - 1];
    char c_prev2 = st[i - 2];
    char c_next  = ((size_t)(i + 1) < st.size()) ? st[i + 1] : '\0';

    if ((c_prev == c_cur) && ((c_prev == c_prev2) || (c_prev == c_next))) {
        // Move the index back one position to re-check compound formation
        // after removing or adjusting the triple letter sequence.
        i--;
        striple = 1;
        checkedstriple = 1;
        break;
    }
}
