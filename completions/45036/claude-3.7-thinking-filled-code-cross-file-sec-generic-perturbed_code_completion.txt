ETH_P_MPLS_MCAST:
  network_flow->stats.mpls_count++;
  type = ETH_P_IP;
  if(header->caplen < ip_offset + 4)
    return(nproto); /* Too short for MPLS header*/
  
  /* Get the first MPLS label */
  mpls.u32 = ntohl(*((uint32_t *)&packet[ip_offset]));
  ip_offset += 4;
  
  /* Handle multiple MPLS labels (if any) */
  while((mpls.mpls.s == 0) && (header->caplen >= ip_offset + 4)) {
    mpls.u32 = ntohl(*((uint32_t *)&packet[ip_offset]));
    ip_offset += 4;
  }
  
  recheck_type = 1;
  break;
