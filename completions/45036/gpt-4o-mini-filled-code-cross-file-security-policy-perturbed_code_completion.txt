      return(nproto); /* Too short for MPLS header */

    mpls.u32 = ntohl(*(uint32_t *)&packet[ip_offset]);
    network_flow->stats.mpls_count++;
    ip_offset += sizeof(mpls.mpls);

      ip_offset += sizeof(mpls.mpls);
