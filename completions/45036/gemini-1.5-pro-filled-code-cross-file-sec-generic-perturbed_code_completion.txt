ETH_P_MPLS_MC:
    mpls.u32 = ntohl(*((uint32_t *)(packet + ip_offset)));
    network_flow->stats.mpls_count++;
    type = ETH_P_IP;
    ip_offset += 4;

    // Multiple labels
    while ((mpls.mpls.label & MPLS_LS_MASK) == 0 && (ip_offset + 4 < header->caplen)) {
      ip_offset += 4;
      mpls.u32 = ntohl(*((uint32_t *)(packet + ip_offset - 4)));
    }
    recheck_type = 1;
    break;
