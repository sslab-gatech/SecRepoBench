ETH_P_MPLS_MC:
    {
      u_int8_t mpls_labels = 0;
      do {
        if(ip_offset + sizeof(mpls) > header->caplen)
          return(nproto);

        mpls.u32 = ntohl(*((u_int32_t*)&packet[ip_offset]));
        network_flow->stats.mpls_count++;

        ip_offset += 4;
        mpls_labels++;

        if(mpls_labels > 10) {
          /* Corrupted packet */
          return nproto;
        }
      } while(!(mpls.mpls.label_s & 0x01)); /* Bottom of Label Stack */

      type = ETH_P_IP;
      recheck_type = 1;
    }
    break;
