ETH_P_MPLS_MCAST:
  /* MPLS header */
  network_flow->stats.mpls_count++;
  type = ETH_P_IP;
  
  do {
    if(header->caplen < ip_offset + 4)
      return(nproto); /* Too short for MPLS header*/
    
    mpls.u32 = *((uint32_t *) &packet[ip_offset]);
    mpls.u32 = ntohl(mpls.u32);
    ip_offset += 4;
  } while((mpls.mpls.s == 0) && (ip_offset < header->caplen));
  
  recheck_type = 1;
  break;
