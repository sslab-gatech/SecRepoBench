ETH_P_MPLS_MULTI:
  network_flow->stats.mpls_count++;
  type = ETH_P_IP;
  
  while (1) {
    if (ip_offset + 4 > header->caplen)
      return(nproto); /* Too short for MPLS header*/
      
    // Extract MPLS label from network to host byte order
    mpls.u32 = ntohl(*((uint32_t*)&packet[ip_offset]));
    
    ip_offset += 4; /* Move past this label */
    
    if(mpls.mpls.s) /* Bottom of Stack bit is set */
      break;
  }
  
  recheck_type = 1;
  break;
