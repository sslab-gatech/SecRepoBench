ETH_P_MPLS_MC:
    for (int i = 0; i < 5; i++) {
      if(ip_offset + 4 > header->caplen)
        break;

      mpls.u32 = ntohl(*((uint32_t*)&packet[ip_offset]));
      network_flow->stats.mpls_count++;

      if ((mpls.mpls.label & 0x01) != 0x0) {
        type = ETH_P_IP;
        ip_offset += 4;
        break;
      } else {
        ip_offset += 4;
      }
    }
    recheck_type = 1;
    break;
