ETH_P_MPLS_MC:
    for (int i = 0; i < 5; i++) {
      mpls.u32 = ntohl(*((uint32_t*)&packet[ip_offset]));
      network_flow->stats.mpls_count++;

      if ((mpls.mpls.label_stack_bottom & 0x01) != 0x01) {
        ip_offset += 4;
      } else {
        type = ETH_P_IP;
        ip_offset += 4;
        recheck_type = 1;
        break;
      }

      if (ip_offset + 4 > header->caplen) {
        /* Invalid packet */
        recheck_type = 0;
        break;
      }
    }
    break;
