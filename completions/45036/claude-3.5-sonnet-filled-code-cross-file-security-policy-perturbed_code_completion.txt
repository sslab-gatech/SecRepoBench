ETH_P_MPLS_MC:
    network_flow->stats.mpls_count++;
    mpls.u32 = ntohl(*((uint32_t *) &packet[ip_offset]));
    type = ETH_P_IP;
    ip_offset += 4;

    while (!mpls.mpls.s) {
      if (ip_offset + 4 >= header->caplen)
        return (nproto); // Too short for another MPLS header
      mpls.u32 = ntohl(*((uint32_t *) &packet[ip_offset]));
      ip_offset += 4;
    }
    recheck_type = 1;
    break;
