if (c->format == BIT_PLANAR || c->format == BIT_LINE)
        buf_size = (aligned_width * c->avctx->height * c->bpp + 7) >> 3;
    else if (c->format == CHUNKY && c->bpp == 24)
        buf_size = aligned_width * c->avctx->height * 3;

    if (c->video_size < buf_size) {
        av_log(avctx, AV_LOG_ERROR, "Insufficient video data, needed %d but only have %d\n",
               buf_size, c->video_size);
        return AVERROR_INVALIDDATA;
    }

    if (!compressiontype) {
        if (c->palette_size && c->bpp <= 8) {
            avctx->pix_fmt = AV_PIX_FMT_PAL8;
        } else if (((c->bpp == 6 || c->bpp == 8) && c->palette_size) ||
                   (c->bpp == 8 && !c->palette_size)) {
            avctx->pix_fmt = AV_PIX_FMT_BGR24;
        } else if (c->bpp == 24 && c->format == CHUNKY && !c->palette_size) {
            avctx->pix_fmt = AV_PIX_FMT_RGB24;
        } else {
            avpriv_request_sample(avctx, "bpp %d, format %d, palette_size %d",
                                 c->bpp, c->format, c->palette_size);
            return AVERROR_PATCHWELCOME;
        }
    } else {
        avctx->pix_fmt = AV_PIX_FMT_BGR24;
    }
