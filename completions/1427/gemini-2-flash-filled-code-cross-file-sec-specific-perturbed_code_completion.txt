if (c->format == CHUNKY) {
        if (c->video_size < avctx->height * avctx->width * c->bpp / 8)
            return AVERROR_INVALIDDATA;
    } else {
        if (c->video_size < aligned_width * avctx->height * c->bpp / 8)
            return AVERROR_INVALIDDATA;
    }

    if (!compressiontype && c->palette_size && c->bpp <= 8) {
        avctx->pix_fmt = AV_PIX_FMT_PAL8;
        p->palette_has_changed = 1;
        memcpy(p->palette, c->palette, c->palette_size);
    } else if (compressiontype && (c->bpp == 6 || c->bpp == 8) &&
               c->palette_size == 768) {
        avctx->pix_fmt = AV_PIX_FMT_BGR24;
    } else if (!compressiontype && c->bpp == 24 && c->format == CHUNKY &&
               !c->palette_size) {
        avctx->pix_fmt = AV_PIX_FMT_RGB24;
    } else {
        avpriv_request_sample(avctx, "Encoding %s with bpp=%d, palette=%d",
                              compressiontype ? "HAM" : "RAW", c->bpp,
                              c->palette_size);
        return AVERROR_PATCHWELCOME;
    }
