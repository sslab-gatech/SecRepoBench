{
    int required_bytes = (((aligned_width * c->bpp) + 7) / 8) * h;
    if (c->video_size < required_bytes)
        return AVERROR_INVALIDDATA;

    if (!compressiontype) {
        if (c->palette_size && c->bpp <= 8)
            avctx->pix_fmt = AV_PIX_FMT_PAL8;
        else if (c->bpp == 24 && c->format == CHUNKY && !c->palette_size)
            avctx->pix_fmt = AV_PIX_FMT_RGB24;
        else {
            avpriv_request_sample(avctx, "Unsupported unencoded format: bpp=%d, palette_size=%d, format=0x%x",
                                  c->bpp, c->palette_size, c->format);
            return AVERROR_PATCHWELCOME;
        }
    } else {
        if ((c->bpp == 6 || c->bpp == 8) && c->palette_size)
            avctx->pix_fmt = AV_PIX_FMT_BGR24;
        else {
            avpriv_request_sample(avctx, "Unsupported encoded format: bpp=%d, palette_size=%d, format=0x%x",
                                  c->bpp, c->palette_size, c->format);
            return AVERROR_PATCHWELCOME;
        }
    }
}
