    size = (size + 7) >> 3;
    if (size > c->video_size)
        return AVERROR_INVALIDDATA;
    if (c->palette_size == 0 && c->bpp == 24 && c->format == CHUNKY)
        avctx->pix_fmt = AV_PIX_FMT_RGB24;
    else if (c->bpp <= 8 && c->palette_size > 0)
        avctx->pix_fmt = AV_PIX_FMT_PAL8;
    else if (c->bpp == 6 || c->bpp == 8) {
        if (c->palette_size == 768)
            avctx->pix_fmt = AV_PIX_FMT_BGR24;
        else {
            avpriv_request_sample(avctx, "bpp=%d palette=%d", c->bpp, c->palette_size);
            return AVERROR_PATCHWELCOME;
        }
    } else {
        avpriv_request_sample(avctx, "bpp=%d", c->bpp);
        return AVERROR_PATCHWELCOME;
