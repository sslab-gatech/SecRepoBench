// Check for reasonable bpp values
if (c->bpp > 32) {
    return AVERROR_INVALIDDATA;
}

// Check if the video size is sufficient (with overflow protection)
if (aligned_width <= 0 || h <= 0) {
    return AVERROR_INVALIDDATA;
}

// Calculate required bytes for video data, checking for overflow
uint64_t required_bytes = (uint64_t)aligned_width * h * c->bpp / 8;
if (required_bytes > c->video_size) {
    return AVERROR_INVALIDDATA;
}

// Determine appropriate pixel format
if (!compressiontype && c->palette_size && c->bpp <= 8) {
    avctx->pix_fmt = AV_PIX_FMT_PAL8;
} else if (compressiontype && ((c->bpp == 6 && c->palette_size == 32) || 
                             (c->bpp == 8 && c->palette_size == 64))) {
    avctx->pix_fmt = AV_PIX_FMT_BGR24;
} else if (!compressiontype && c->bpp == 24 && c->format == CHUNKY && !c->palette_size) {
    avctx->pix_fmt = AV_PIX_FMT_RGB24;
} else {
    avpriv_request_sample(avctx, "Compression %d, bpp %d, palette size %d", 
                         compressiontype, c->bpp, c->palette_size);
    return AVERROR_PATCHWELCOME;
}
