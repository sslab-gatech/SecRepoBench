// Check if the video size is sufficient based on the aligned width, height,
// and bits per pixel (bpp). Return an error if it is not.
int required_size = FFALIGN(c->avctx->width, 16) * c->avctx->height * c->bpp / 8;
if (c->video_size < required_size) {
    return AVERROR_INVALIDDATA;
}

// Determine the appropriate pixel format for the AVCodecContext based on
// encoding, palette size, bits per pixel, and video format.
if (c->format == CHUNKY && c->bpp == 24 && c->palette_size == 0) {
    avctx->pix_fmt = AV_PIX_FMT_RGB24;
} else if (c->bpp <= 8 && c->palette_size > 0) {
    avctx->pix_fmt = AV_PIX_FMT_PAL8;
} else if (c->bpp == 6 || c->bpp == 8) {
    avctx->pix_fmt = AV_PIX_FMT_BGR24;
} else {
    avpriv_request_sample(avctx, "Pixel format 0x%0x", c->format);
    return AVERROR_PATCHWELCOME;
}
