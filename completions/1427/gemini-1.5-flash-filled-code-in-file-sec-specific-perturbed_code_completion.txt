int size = aligned_width * h * ((c->bpp + 7) / 8);
    if (size > c->video_size)
        return AVERROR_INVALIDDATA;
    if (!c->palette_size && c->bpp <= 8 && c->format != CHUNKY)
        avctx->pix_fmt = AV_PIX_FMT_PAL8;
    else if (c->bpp == 6 && c->palette_size == 96)
        avctx->pix_fmt = AV_PIX_FMT_BGR24;
    else if (c->bpp == 8 && c->palette_size == 512)
        avctx->pix_fmt = AV_PIX_FMT_BGR24;
    else if (!c->palette_size && c->bpp == 24 && c->format == CHUNKY)
        avctx->pix_fmt = AV_PIX_FMT_RGB24;
    else {
        avpriv_request_sample(avctx, "Pixel format 0x%0x, bpp %d, palette size %d", c->format, c->bpp, c->palette_size);
        return AVERROR_PATCHWELCOME;
