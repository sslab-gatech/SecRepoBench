if (c->video_size < (unsigned int)aligned_width * h * c->bpp / 8)
        return AVERROR_INVALIDDATA;
    if (c->palette_size && c->bpp <= 8 && !compressiontype)
        avctx->pix_fmt = AV_PIX_FMT_PAL8;
    else if (c->bpp == 6 || c->bpp == 8) {
        if (c->palette_size == 768)
            avctx->pix_fmt = AV_PIX_FMT_BGR24;
        else {
            avpriv_request_sample(avctx, "bpp=%d, palette_size=%d", c->bpp, c->palette_size);
            return AVERROR_PATCHWELCOME;
        }
    } else if (!compressiontype && c->bpp == 24 && c->format == CHUNKY && !c->palette_size)
        avctx->pix_fmt = AV_PIX_FMT_RGB24;
    else {
        avpriv_request_sample(avctx, "bpp=%d, compressiontype=%d, format=%d, palette_size=%d",
                              c->bpp, compressiontype, c->format, c->palette_size);
        return AVERROR_PATCHWELCOME;
    }
