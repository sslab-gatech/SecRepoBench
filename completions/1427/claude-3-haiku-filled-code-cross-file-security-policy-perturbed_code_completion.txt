// Determine the appropriate pixel format for the AVCodecContext based on
// encoding, palette size, bits per pixel, and video format.
if (c->format == CHUNKY && c->bpp == 24 && c->palette_size == 0) {
    avctx->pix_fmt = AV_PIX_FMT_RGB24;
} else if (c->bpp == 8 && c->palette_size == 256) {
    avctx->pix_fmt = AV_PIX_FMT_PAL8;
} else if (c->bpp == 6 || c->bpp == 8) {
    avctx->pix_fmt = AV_PIX_FMT_BGR24;
} else {
    avpriv_request_sample(avctx, "Unsupported pixel format: bpp=%d, palette_size=%d, format=0x%x",
                          c->bpp, c->palette_size, c->format);
    return AVERROR_PATCHWELCOME;
}
