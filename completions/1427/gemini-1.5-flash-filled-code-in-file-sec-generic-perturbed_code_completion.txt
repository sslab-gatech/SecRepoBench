int size = avctx->width * avctx->height * ((c->bpp + 7) >> 3);
    if (size > c->video_size)
        return AVERROR_INVALIDDATA;
    if (!c->palette_size && c->bpp <= 8 && !compressiontype)
        avctx->pix_fmt = AV_PIX_FMT_PAL8;
    else if (c->bpp == 6 && c->palette_size == 96 || c->bpp == 8 && c->palette_size == 512)
        avctx->pix_fmt = AV_PIX_FMT_BGR24;
    else if (!compressiontype && c->bpp == 24 && c->format == CHUNKY && !c->palette_size)
        avctx->pix_fmt = AV_PIX_FMT_RGB24;
    else {
        avpriv_request_sample(avctx, "Pixel format bpp:%d format:0x%0x palette_size:%d", c->bpp, c->format, c->palette_size);
        return AVERROR_PATCHWELCOME;
    }
