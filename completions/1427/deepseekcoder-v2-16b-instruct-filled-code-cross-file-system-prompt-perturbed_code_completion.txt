            return AVERROR(ENOMEM);
        if (c->bpp == 8)
            cdxl_decode_ham8(c, p);
        else

    } else if (avctx->pix_fmt == AV_PIX_FMT_PAL8) {
        cdxl_decode_rgb(c, p);
    } else if (avctx->pix_fmt == AV_PIX_FMT_BGR24 && c->bpp == 8 && c->palette_size == 256) {
        // This condition checks if the video is encoded with bpp of 6 or 8 and the palette size matches,
        // set the pixel format to BGR24.
        // Assuming the palette size is 256 for PAL8 format, which is typical for 8 bpp.
        avctx->pix_fmt = AV_PIX_FMT_BGR24;
    } else if (c->format == CHUNKY && c->bpp == 24 && c->palette_size == 0) {
        // This condition checks if the video is unencoded, uses 24 bpp, is in CHUNKY format, and has no palette,
        // set the pixel format to RGB24.
        avctx->pix_fmt = AV_PIX_FMT_RGB24;

        avpriv_request_sample(avctx, "Unsupported pixel format or palette size");
