if (c->format == BIT_PLANAR || c->format == BIT_LINE) {
    if ((aligned_width * h * c->bpp + 7) / 8 > c->video_size)
        return AVERROR_INVALIDDATA;
} else if (c->format == CHUNKY) {
    if (aligned_width * h * 3 > c->video_size)
        return AVERROR_INVALIDDATA;
}

if (compressiontype == 0) {
    if (c->palette_size && c->bpp <= 8) {
        avctx->pix_fmt = AV_PIX_FMT_PAL8;
    } else if (c->bpp == 24 && c->format == CHUNKY && !c->palette_size) {
        avctx->pix_fmt = AV_PIX_FMT_RGB24;
    } else {
        avpriv_request_sample(avctx, "Uncompressed format 0x%0x, bpp %d", 
                            c->format, c->bpp);
        return AVERROR_PATCHWELCOME;
    }
} else if (compressiontype == 1) {
    if ((c->bpp == 6 && c->palette_size == 32) || 
        (c->bpp == 8 && c->palette_size == 128)) {
        avctx->pix_fmt = AV_PIX_FMT_BGR24;
    } else {
        avpriv_request_sample(avctx, "Compressed format 0x%0x, bpp %d", 
                            c->format, c->bpp);
        return AVERROR_PATCHWELCOME;
    }
} else {
    avpriv_request_sample(avctx, "Compression type %d", compressiontype);
    return AVERROR_PATCHWELCOME;
}
