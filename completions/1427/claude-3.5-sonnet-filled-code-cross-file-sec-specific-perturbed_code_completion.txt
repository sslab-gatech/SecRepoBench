if (c->video_size < (aligned_width * c->bpp * avctx->height + 7) >> 3)
    return AVERROR_INVALIDDATA;

if (!c->palette_size && c->bpp <= 8) {
    avpriv_request_sample(avctx, "Palette-less video with %d bpp", c->bpp);
    return AVERROR_PATCHWELCOME;
}

if (c->palette_size && c->bpp <= 8) {
    avctx->pix_fmt = AV_PIX_FMT_PAL8;
} else if ((c->bpp == 6 || c->bpp == 8) && c->palette_size == (1 << c->bpp)) {
    avctx->pix_fmt = AV_PIX_FMT_BGR24;
} else if (c->bpp == 24 && c->format == CHUNKY && !c->palette_size) {
    avctx->pix_fmt = AV_PIX_FMT_RGB24;
} else {
    avpriv_request_sample(avctx, "Unsupported format %d bpp, palette size %d", 
                          c->bpp, c->palette_size);
    return AVERROR_PATCHWELCOME;
}
