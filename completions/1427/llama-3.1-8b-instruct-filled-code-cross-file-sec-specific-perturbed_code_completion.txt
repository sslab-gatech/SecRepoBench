if (c->format == CHUNKY)
    aligned_width = avctx->width;
else
    aligned_width = FFALIGN(c->avctx->width, 16);
c->padded_bits  = aligned_width - c->avctx->width;

if (compressiontype) {
    // Determine the appropriate pixel format for the AVCodecContext based on
    // encoding, palette size, bits per pixel, and video format. If the video is
    // unencoded, has a palette, and uses <= 8 bpp, set the pixel format to PAL8.
    // If the video is encoded with bpp of 6 or 8 and the palette size matches,
    // set the pixel format to BGR24. If the video is unencoded, uses 24 bpp, is
    // in CHUNKY format, and has no palette, set the pixel format to RGB24.
    // If the format is unsupported, request a sample and return a patch-welcome
    // error to indicate the need for further support.
    if (c->palette_size == 0 && c->bpp == 24 && c->format == CHUNKY) {
        avctx->pix_fmt = AV_PIX_FMT_RGB24;
    } else if (c->palette_size == c->bpp && c->bpp == 8) {
        avctx->pix_fmt = AV_PIX_FMT_BGRA;
    } else if (c->palette_size > 0 && c->bpp <= 8) {
        avctx->pix_fmt = AV_PIX_FMT_PAL8;
    } else {
        avpriv_request_sample(avctx, "Unsupported pixel format");
        return AVERROR_PATCHWELCOME;
    }
} else {
    // If the video is unencoded, determine the pixel format based on the
    // bits per pixel and video format. If the video is in CHUNKY format and
    // uses 24 bpp, set the pixel format to RGB24. If the video is in PLANAR
    // format and uses 16 bpp, set the pixel format to RGB555LE. If the video
    // is in PLANAR format and uses 32 bpp, set the pixel format to BGR0.
    if (c->format == CHUNKY && c->bpp == 24) {
        avctx->pix_fmt = AV_PIX_FMT_RGB24;
    } else if (c->format == BIT_PLANAR && c->bpp == 16) {
        avctx->pix_fmt = AV_PIX_FMT_RGB555LE;
    } else if (c->format == BIT_PLANAR && c->bpp == 32) {
        avctx->pix_fmt = AV_PIX_FMT_BGR0;
    } else {
        avpriv_request_sample(avctx, "Unsupported pixel format");
        return AVERROR_PATCHWELCOME;
    }
}
