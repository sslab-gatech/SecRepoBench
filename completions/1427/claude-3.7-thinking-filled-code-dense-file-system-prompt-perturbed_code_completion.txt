// Check if the video size is sufficient based on aligned width, height, and bits per pixel
if (c->format == CHUNKY) {
    if (c->bpp > 8 && c->bpp != 24) {
        avpriv_request_sample(avctx, "Unsupported CHUNKY video bit depth: %d", c->bpp);
        return AVERROR_PATCHWELCOME;
    }
    if (c->video_size < (aligned_width * avctx->height * c->bpp + 7) >> 3)
        return AVERROR_INVALIDDATA;
} else {
    if (c->bpp > 8) {
        avpriv_request_sample(avctx, "Unsupported planar/line video bit depth: %d", c->bpp);
        return AVERROR_PATCHWELCOME;
    }
    if (c->video_size < (aligned_width * avctx->height * c->bpp + 7) >> 3)
        return AVERROR_INVALIDDATA;
}

// Determine the appropriate pixel format based on encoding, palette, bpp, and format
if (compressiontype == 0) {
    if (c->palette_size && c->bpp <= 8) {
        avctx->pix_fmt = AV_PIX_FMT_PAL8;
    } else if (c->format == CHUNKY && c->bpp == 24 && !c->palette_size) {
        avctx->pix_fmt = AV_PIX_FMT_RGB24;
    } else {
        avpriv_request_sample(avctx, "Unsupported color format: bpp: %d, palette size: %d, format: %d", 
                             c->bpp, c->palette_size, c->format);
        return AVERROR_PATCHWELCOME;
    }
} else if ((c->bpp == 6 || c->bpp == 8) && c->palette_size) {
    avctx->pix_fmt = AV_PIX_FMT_BGR24;
} else {
    avpriv_request_sample(avctx, "Unsupported compression type: %d, bpp: %d", 
                         compressiontype, c->bpp);
    return AVERROR_PATCHWELCOME;
}
