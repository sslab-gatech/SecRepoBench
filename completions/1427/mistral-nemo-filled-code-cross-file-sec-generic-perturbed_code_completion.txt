if (c->video_size < (c->bpp * w * h + 7) / 8) {
        av_log(avctx, AV_LOG_ERROR, "Video size too small\n");
        return AVERROR_INVALIDDATA;
    }

    switch (c->format) {
    case BIT_PLANAR:
        avctx->pix_fmt = AV_PIX_FMT_PAL8;
        break;
    case BIT_LINE:
        if (c->bpp <= 8 && c->palette_size == 768)
            avctx->pix_fmt = AV_PIX_FMT_BGR24;
        else {
            avpriv_request_sample(avctx, "Unsupported bit line format");
            return AVERROR_PATCHWELCOME;
        }
        break;
    case CHUNKY:
        if (c->bpp == 24 && !c->palette_size)
            avctx->pix_fmt = AV_PIX_FMT_RGB24;
        else {
            avpriv_request_sample(avctx, "Unsupported chunky format");
            return AVERROR_PATCHWELCOME;
        }
        break;
    default:
        avpriv_request_sample(avctx, "Unsupported format");
        return AVERROR_PATCHWELCOME;
    }
