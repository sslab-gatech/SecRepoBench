if (c->format == BIT_PLANAR || c->format == BIT_LINE) {
        if (c->video_size < (aligned_width * c->avctx->height * c->bpp + 7) / 8) {
            av_log(avctx, AV_LOG_ERROR, "video data is too small\n");
            return AVERROR_INVALIDDATA;
        }
    }

    if (!compressiontype) {
        if (c->palette_size && c->bpp <= 8) {
            avctx->pix_fmt = AV_PIX_FMT_PAL8;
        } else if (((c->bpp == 6 && c->palette_size == 32) ||
                   (c->bpp == 8 && c->palette_size == 64)) && compressiontype) {
            avctx->pix_fmt = AV_PIX_FMT_BGR24;
        } else if (c->bpp == 24 && c->format == CHUNKY && !c->palette_size) {
            avctx->pix_fmt = AV_PIX_FMT_RGB24;
        } else {
            avpriv_request_sample(avctx, "bpp %d, format %d, palette_size %d, compression %d",
                                 c->bpp, c->format, c->palette_size, compressiontype);
            return AVERROR_PATCHWELCOME;
        }
    } else {
        avctx->pix_fmt = AV_PIX_FMT_BGR24;
    }
