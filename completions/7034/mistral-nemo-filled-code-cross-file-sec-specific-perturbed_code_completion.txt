// Fill the masked region with a color
static MagickBooleanType FillMaskedRegion(Image *image, const DrawInfo *draw_info, ExceptionInfo *exception)
{
  MagickBooleanType
    status;

  status=MagickTrue;
  if (draw_info->clip_mask == (char *) NULL)
    return(status);
  if (draw_info->fill_pattern != (Image *) NULL)
    {
      if (draw_info->fill_pattern->class == DirectClass)
        {
          status=DrawImage(image,draw_info,draw_info->fill_pattern,exception);
          if (status == MagickFalse)
            return(status);
        }
      else
        {
          status=DrawImage(image,draw_info,draw_info->fill_pattern,exception);
          if (status == MagickFalse)
            return(status);
        }
    }
  else
    {
      status=DrawFillToBorderImage(image,draw_info,exception);
      if (status == MagickFalse)
        return(status);
    }
  return(status);
}
