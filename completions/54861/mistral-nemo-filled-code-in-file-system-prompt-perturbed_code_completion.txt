if (codeccontext->codec_id == AV_CODEC_ID_H_264 || codeccontext->codec_id == AV_CODEC_ID_HEVC) {
    *width  = FFALIGN(*width,  16);
    *height = FFALIGN(*height, 16);
} else if (codeccontext->codec_id == AV_CODEC_ID_VP8) {
    *width  = FFALIGN(*width,  16*2);
    *height = FFALIGN(*height, 16 * 2);
} else {
    *width  = FFALIGN(*width,  32);
    *height = FFALIGN(*height, 32);
