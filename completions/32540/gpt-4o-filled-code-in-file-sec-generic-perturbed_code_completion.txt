// Check if the current set of glyphs for the given lookup index has already been processed.
    hb_set_t *covered_glyph_set = done_lookups_glyph_set->get (indexlookup);
    if (!covered_glyph_set)
    {
      // If not, update the glyph count for this lookup index and ensure the glyph set is initialized
      // and cleared as needed. Then, obtain the glyph set associated with the lookup index.
      done_lookups_glyph_count->set (indexlookup, 0);
      covered_glyph_set = hb_set_create ();
      if (unlikely (!covered_glyph_set))
        return true; // In case of error, consider the lookup done to avoid infinite loops.
      done_lookups_glyph_set->set (indexlookup, covered_glyph_set);
    }

    // Determine if the parent active glyphs are a subset of the obtained glyph set.
    if (hb_set_is_subset (parent_active_glyphs (), covered_glyph_set))
      return true; // If they are a subset, return true to indicate the lookup is done.

    // Otherwise, return false.
    return false;
