if (indexlookup >= done_lookups_glyph_count->size() || indexlookup >= done_lookups_glyph_set->size()) {
    // Index out of bounds, return true to indicate an error state
    return true;
}

if (!done_lookups_glyph_set->is_initialized(indexlookup)) {
    done_lookups_glyph_set->initialize(indexlookup);
    done_lookups_glyph_count->update_count(indexlookup);
}

hb_set_t* current_glyph_set = done_lookups_glyph_set->get_set(indexlookup);
if (hb_set_is_subset(parent_active_glyphs(), current_glyph_set)) {
    return true;
}
