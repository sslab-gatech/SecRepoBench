unsigned int glyph_count = done_lookups_glyph_count->get (indexlookup);
if (glyph_count == 0) {
  glyph_count = parent_active_glyphs ()->get_population ();
  done_lookups_glyph_count->set (indexlookup, glyph_count);
}

  done_lookups_glyph_set->add (indexlookup);
  hb_set_clear (glyph_set);
}

hb_set_union (glyph_set, parent_active_glyphs ());
if (hb_set_is_subset (parent_active_glyphs (), glyph_set))
  return true;
