{
  done_lookups_glyph_count->resize(indexlookup + 1);
  unsigned &old_count = done_lookups_glyph_count->arrayZ[indexlookup];
  unsigned new_count = hb_set_get_population(parent_active_glyphs());
  if (old_count != new_count)
  {
    old_count = new_count;
    done_lookups_glyph_set->resize(indexlookup + 1);
    if (!done_lookups_glyph_set->arrayZ[indexlookup])
      done_lookups_glyph_set->arrayZ[indexlookup] = hb_set_create();
    else
      hb_set_clear(done_lookups_glyph_set->arrayZ[indexlookup]);
  }
  hb_set_t *lookup_set = done_lookups_glyph_set->arrayZ[indexlookup];
  if (hb_set_is_subset(parent_active_glyphs(), lookup_set))
    return true;
}
