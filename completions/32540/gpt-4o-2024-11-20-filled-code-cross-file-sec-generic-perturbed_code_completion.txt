hb_set_t *glyph_set = done_lookups_glyph_set->get (indexlookup);
    if (!glyph_set)
    {
      unsigned int current_count = done_lookups_glyph_count->get (indexlookup);
      if (!done_lookups_glyph_count->set (indexlookup, current_count + 1))
        return true;

      glyph_set = hb_set_create ();
      if (!done_lookups_glyph_set->set (indexlookup, glyph_set))
      {
        hb_set_destroy (glyph_set);
        return true;
      }
    }
    else
    {
      if (hb_set_is_subset (parent_active_glyphs (), glyph_set))
        return true;
    }
