if (done_lookups_glyph_count->has (indexlookup))
      return done_lookups_glyph_count->get (indexlookup) == glyphs->len;

    unsigned int glyph_count = glyphs->len;
    unsigned int covered_count = done_lookups_glyph_count->get (indexlookup);
    if (covered_count == glyph_count)
      return true;

    done_lookups_glyph_count->set (indexlookup, glyph_count);
    return false;
