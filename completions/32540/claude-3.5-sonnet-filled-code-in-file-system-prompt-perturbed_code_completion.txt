unsigned int current_count = (*done_lookups_glyph_count)[indexlookup];
if (current_count != parent_active_glyphs ()->get_population ())
{
  (*done_lookups_glyph_count)[indexlookup] = parent_active_glyphs ()->get_population ();
  if (!done_lookups_glyph_set->has (indexlookup))
  {
    hb_set_t *covered_glyph_set = hb_set_create ();
    done_lookups_glyph_set->set (indexlookup, covered_glyph_set);
  }
  else
  {
    hb_set_clear (done_lookups_glyph_set->get (indexlookup));
  }
}

if (hb_set_is_subset (parent_active_glyphs (), covered_glyph_set))
