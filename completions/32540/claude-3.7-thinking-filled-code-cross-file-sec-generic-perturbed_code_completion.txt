  if (!done_lookups_glyph_count->resize (indexlookup + 1))
    return true;
}

{
  (*done_lookups_glyph_count)[indexlookup] = glyphs->get_population ();
  
  if (unlikely (indexlookup >= done_lookups_glyph_set->get_population ()))
  {
    if (!done_lookups_glyph_set->resize (indexlookup + 1))
      return true;
  }
  
  (*done_lookups_glyph_set)[indexlookup].clear ();
  covered_glyph_set = &(*done_lookups_glyph_set)[indexlookup];
  
  return false;
}
