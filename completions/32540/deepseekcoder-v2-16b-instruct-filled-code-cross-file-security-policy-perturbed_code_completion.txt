if (indexlookup >= done_lookups_glyph_count->size() || indexlookup >= done_lookups_glyph_set->size()) {
    return false;
}

if (!done_lookups_glyph_count->get(indexlookup) || !done_lookups_glyph_set->get(indexlookup)) {
    return false;
}

hb_set_clear(covered_glyph_set);
hb_set_add(covered_glyph_set, parent_active_glyphs());

if (hb_set_is_subset(parent_active_glyphs(), done_lookups_glyph_set->get(indexlookup))) {
    done_lookups_glyph_count->set(indexlookup, true);
    return true;
} else {
    return false;
}
