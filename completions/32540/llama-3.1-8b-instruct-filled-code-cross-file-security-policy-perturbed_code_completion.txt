if (done_lookups_glyph_count->in_error () ||
    done_lookups_glyph_set->in_error ())
  return true;

// Check if the current set of glyphs for the given lookup index has already been processed.
if (done_lookups_glyph_count->get (indexlookup)!= 0)
  return true;

// Update the glyph count for this lookup index
done_lookups_glyph_count->set (indexlookup, 1);

// Ensure the glyph set is initialized and cleared as needed
if (done_lookups_glyph_set->in_error ())
  done_lookups_glyph_set->clear ();

// Obtain the glyph set associated with the lookup index
hb_set_t* lookup_glyph_set = get_glyph_set (indexlookup);

// Determine if the parent active glyphs are a subset of the obtained glyph set
hb_set_union (covered_glyph_set, parent_active_glyphs ());
if (hb_set_is_subset (covered_glyph_set, lookup_glyph_set))
  return true;

// If they are not a subset, return false
return false;
