if (indexlookup >= done_lookups_glyph_count->length)
      return false;

    unsigned *glyph_count = done_lookups_glyph_count->arrayZ + indexlookup;
    hb_set_t *covered_glyph_set = done_lookups_glyph_set->arrayZ + indexlookup;

    if (*glyph_count != current_glyphs->get_population ())
    {
      *glyph_count = current_glyphs->get_population ();
      hb_set_clear (covered_glyph_set);
    }

    if (hb_set_is_subset (parent_active_glyphs (), covered_glyph_set))
      return true;
