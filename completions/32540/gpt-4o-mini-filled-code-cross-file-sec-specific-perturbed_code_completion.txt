if (indexlookup >= MAX_LOOKUP_INDEX) {
    return true; // Index out of bounds, treat as done
}

if (!glyph_sets_initialized[indexlookup]) {
    initialize_glyph_set(indexlookup); // Ensure the glyph set is initialized
}

if (glyph_counts[indexlookup] == 0) {
    clear_glyph_set(indexlookup); // Clear the glyph set if no glyphs are present
}

glyph_counts[indexlookup]++; // Safely increment the glyph count for the lookup index
