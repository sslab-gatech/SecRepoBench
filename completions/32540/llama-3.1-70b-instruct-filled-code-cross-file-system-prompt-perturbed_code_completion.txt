if (!done_lookups_glyph_count->get (indexlookup)) {
  done_lookups_glyph_count->set (indexlookup, glyph_count);
  if (!done_lookups_glyph_set->get (indexlookup)) {
    done_lookups_glyph_set->set (indexlookup, hb_set_create ());
  }
  hb_set_clear (done_lookups_glyph_set->get (indexlookup));
  hb_set_union (done_lookups_glyph_set->get (indexlookup), glyphs);
  if (hb_set_is_subset (parent_active_glyphs (), done_lookups_glyph_set->get (indexlookup))) {
    return true;
  }
