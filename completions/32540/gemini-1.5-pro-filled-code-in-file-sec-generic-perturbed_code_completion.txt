unsigned int lookup_glyph_count = done_lookups_glyph_count->get (indexlookup);
if (lookup_glyph_count == (unsigned)-1)
{
  done_lookups_glyph_count->set (indexlookup, hb_set_get_population (parent_active_glyphs ()));
  hb_set_t *&covered_glyph_set = done_lookups_glyph_set->get (indexlookup);
  if (!covered_glyph_set)
  {
    covered_glyph_set = hb_set_create ();
    if (unlikely (!covered_glyph_set))
      return true;
    done_lookups_glyph_set->set (indexlookup, covered_glyph_set);
  }
  hb_set_clear (covered_glyph_set);
}
else if (hb_set_get_population (parent_active_glyphs ()) <= lookup_glyph_count)
{
  hb_set_t *covered_glyph_set = done_lookups_glyph_set->get (indexlookup);
  if (hb_set_is_subset (parent_active_glyphs (), covered_glyph_set))
    return true;
}
