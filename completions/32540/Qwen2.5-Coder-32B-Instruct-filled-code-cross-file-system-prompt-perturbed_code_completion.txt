if (!done_lookups_glyph_set->has(indexlookup)) {
    done_lookups_glyph_count->set(indexlookup, done_lookups_glyph_count->get(indexlookup) + 1);
    done_lookups_glyph_set->init();
    done_lookups_glyph_set->clear();
    hb_set_t* current_glyph_set = get_glyph_set_for_lookup(indexlookup);
    if (hb_set_is_subset(parent_active_glyphs(), current_glyph_set)) {
        return true;
    }
}
