if (indexlookup >= done_lookups_glyph_count->length || indexlookup >= done_lookups_glyph_set->length) {
    return false; // Index out of bounds
}

if (done_lookups_glyph_count->arrayZ[indexlookup] > 0) {
    // Glyph count for this lookup index has already been processed
    return true;
}

if (done_lookups_glyph_set->arrayZ[indexlookup] == nullptr) {
    done_lookups_glyph_set->resize(indexlookup + 1);
}
done_lookups_glyph_set->arrayZ[indexlookup]->clear();
