if (!done_lookups_glyph_count->resize (indexlookup + 1))
  return true;

if ((*done_lookups_glyph_count)[indexlookup] == glyphs->get_population ())
  return true;

(*done_lookups_glyph_count)[indexlookup] = glyphs->get_population ();

if (!done_lookups_glyph_set->resize (indexlookup + 1))
  return true;

hb_set_t *covered_glyph_set = (*done_lookups_glyph_set)[indexlookup];
if (unlikely (!covered_glyph_set))
{
  covered_glyph_set = hb_set_create ();
  if (unlikely (!covered_glyph_set))
    return true;
  (*done_lookups_glyph_set)[indexlookup] = covered_glyph_set;
}
else
  covered_glyph_set->clear ();

if (hb_set_is_subset (parent_active_glyphs (), covered_glyph_set))
  return true;
