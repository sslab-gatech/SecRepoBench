if (indexlookup >= done_lookups_glyph_count->get_population ())
      done_lookups_glyph_count->set (indexlookup, 0);

    if (done_lookups_glyph_set->has (indexlookup))
    {
      hb_set_t *glyph_set = done_lookups_glyph_set->get (indexlookup);
      if (hb_set_is_subset (parent_active_glyphs (), glyph_set))
        return true;
    }
    else
    {
      hb_set_t *glyph_set = hb_set_create ();
      if (glyph_set->in_error ())
      {
        hb_set_destroy (glyph_set);
        return true;
      }
      done_lookups_glyph_set->set (indexlookup, glyph_set);
    }

    return false;
