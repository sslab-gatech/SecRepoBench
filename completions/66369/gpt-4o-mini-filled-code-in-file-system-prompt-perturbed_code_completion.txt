case 'R':
            if (cram_add_read_names(c, s, r, b) < 0)
                return -1;
            break;

        case 'M':
            if (cram_add_mate(c, s, r, b) < 0)
                return -1;
            break;

        case 'N':
            if (cram_add_next(c, s, r, b) < 0)
                return -1;
            break;

        case 'T':
            if (cram_add_tag(c, s, r, b) < 0)
                return -1;
            break;

        default:
            hts_log_error("Unknown auxiliary tag type '%c'", *aux);
            goto err;
    }
