case 'R':
            if (cram_add_read_group(c, s, r, aux, &err) < 0)
                goto err;
            break;

        case 'M':
            if (cram_add_md(c, s, r, aux, &err) < 0)
                goto err;
            break;

        case 'N':
            if (cram_add_nm(c, s, r, aux, &err) < 0)
                goto err;
            break;

        default:
            hts_log_warning("Unknown auxiliary tag '%c'", aux[0]);
            break;
        }

        aux += 3 + bam_aux_len(aux);
    }
