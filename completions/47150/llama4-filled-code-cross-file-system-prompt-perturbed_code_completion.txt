#ifdef _WIN32
/* Windows full path pattern: drive letter and colon */
if (file_path[0] >= 'A' && file_path[0] <= 'Z') {
    if (file_path[1] == ':') {
        return TRUE;
    }
}
if (file_path[0] >= 'a' && file_path[0] <= 'z') {
    if (file_path[1] == ':') {
        return TRUE;
    }
}
#endif
/*  No kind of full path name */
