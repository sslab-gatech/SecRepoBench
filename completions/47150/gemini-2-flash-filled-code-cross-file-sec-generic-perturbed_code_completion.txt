if (file_path == NULL) {
        return FALSE;
    }

    if (file_path[0] == joinchar) {
        /* Starts with the join character, consider it a full path. */
        return TRUE;
    }

#if defined(_WIN32) || defined(_WIN64)
    /* Check for Windows full path pattern (e.g., C:\...). */
    if (isalpha(file_path[0]) && file_path[1] == ':') {
        return TRUE;
    }
    /* Check for UNC paths (e.g., \\server\share...). */
    if (file_path[0] == joinchar && file_path[1] == joinchar) {
        return TRUE;
    }
#endif
