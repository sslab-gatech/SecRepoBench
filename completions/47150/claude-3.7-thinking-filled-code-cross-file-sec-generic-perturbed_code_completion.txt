if (!file_path || !*file_path) {
    return FALSE;
}

/* Check if it starts with join character */
if (file_path[0] == joinchar) {
    return TRUE;
}

/* Check for Windows-style paths (drive letter followed by a colon) */
if (((file_path[0] >= 'A' && file_path[0] <= 'Z') || 
     (file_path[0] >= 'a' && file_path[0] <= 'z')) && 
    file_path[1] == ':') {
    return TRUE;
}
