if (!file_path) {
    return FALSE;
}

/* Check for Unix-like absolute path */
if (file_path[0] == joinchar) {
    return TRUE;
}

/* Check for Windows-style full path (drive letter followed by colon) */
if (file_path[0] && file_path[1] &&
    ((file_path[0] >= 'A' && file_path[0] <= 'Z') || 
     (file_path[0] >= 'a' && file_path[0] <= 'z')) && 
    file_path[1] == ':') {
    return TRUE;
}
