if (!file_path || file_path[0] == '\0') {
        return FALSE; // Invalid or empty path
    }

    // Check if the path starts with the join character
    if (file_path[0] == joinchar) {
        return TRUE;
    }

    // Check for Windows-style full path (e.g., "C:\")
    if (((file_path[0] >= 'A' && file_path[0] <= 'Z') || 
         (file_path[0] >= 'a' && file_path[0] <= 'z')) &&
        file_path[1] == ':' &&
        file_path[2] == joinchar) {
        return TRUE;
    }

    // Not a full path
    return FALSE;
