if (!file_path || !*file_path) {
    return FALSE;
}

/*  If it starts with the join character, it's a full path */
if (*file_path == joinchar) {
    return TRUE;
}

#if defined (_WIN32)
/*  For Windows, check if it's a drive letter followed by a colon */
if (file_path[1] && 
    ((file_path[0] >= 'A' && file_path[0] <= 'Z') || 
     (file_path[0] >= 'a' && file_path[0] <= 'z')) && 
    file_path[1] == ':') {
    return TRUE;
}
#endif
