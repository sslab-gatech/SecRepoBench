char *full_path = NULL;
int full_path_len = 0;

if (joinchar == '/' || joinchar == '\\') {
    if (file_path[0] == joinchar) {
        return TRUE;
    }
} else if (isalpha(file_path[0]) && file_path[1] == ':') {
    return TRUE;
}

full_path_len = strlen(file_path);
full_path = malloc(full_path_len + 1);
if (!full_path) {
    return FALSE;
}
strncpy(full_path, file_path, full_path_len);
full_path[full_path_len] = '\0';
