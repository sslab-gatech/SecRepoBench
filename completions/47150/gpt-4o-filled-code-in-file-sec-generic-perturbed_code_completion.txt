if (!file_path || !*file_path) {
        return FALSE;
    }

    // Check for Unix-style absolute path
    if (file_path[0] == '/') {
        return TRUE;
    }

#if defined(_WIN32)
    // Check for Windows-style absolute path
    if (isalpha(file_path[0]) && file_path[1] == ':') {
        return TRUE;
    }
#endif

    // Check for join character at the start
    if (file_path[0] == joinchar) {
        return TRUE;
    }

    return FALSE;
