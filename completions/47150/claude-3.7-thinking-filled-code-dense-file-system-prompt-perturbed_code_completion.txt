if (firstc == joinchar) {
    return TRUE;
}

// Check for Windows-style paths with drive letters
if (((firstc >= 'a' && firstc <= 'z') || 
     (firstc >= 'A' && firstc <= 'Z')) && 
    file_path[1] == ':') {
    // Check if followed by a path separator
    if (file_path[2] == '\\' || file_path[2] == '/') {
        return TRUE;
    }
