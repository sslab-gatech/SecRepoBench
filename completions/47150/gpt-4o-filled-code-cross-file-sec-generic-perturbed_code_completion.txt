        return FALSE;
    }

    // Check for Unix-like full path
    if (file_path[0] == joinchar) {
        return TRUE;
    }

    // Check for Windows full path (e.g., "C:\")
    if (isalpha(file_path[0]) && file_path[1] == ':') {
        return TRUE;
    }

    // Not a full path
    return FALSE;
