if (!file_path || !*file_path) {
    /* Empty or NULL path can't be a full path */
    return FALSE;
}

/* Check if it starts with the join character (likely '/' on Unix systems) */
if (file_path[0] == joinchar) {
    return TRUE;
}

/* Check for Windows full path pattern (e.g., "C:") */
/* Drive letter followed by colon */
if ((file_path[0] >= 'A' && file_path[0] <= 'Z') || 
    (file_path[0] >= 'a' && file_path[0] <= 'z')) {
    if (file_path[1] == ':') {
        return TRUE;
    }
}
