// Fetch frame length
  if (frame_len != NULL) {
    from_big(frame_len, frame_pointer + FRAME_LEN, sizeof(*frame_len));
    if (*frame_len <= 0 || *frame_len > BLOSC2_MAX_FRAME_LEN) {
      BLOSC_TRACE_ERROR("Invalid frame length in header.");
      return BLOSC2_ERROR_INVALID_HEADER;
    }
  }

  // Fetch uncompressed size
  if (nbytes != NULL) {
    from_big(nbytes, frame_pointer + FRAME_NBYTES, sizeof(*nbytes));
    if (*nbytes < 0) {
      BLOSC_TRACE_ERROR("Invalid uncompressed size in header.");
      return BLOSC2_ERROR_INVALID_HEADER;
    }
  }

  // Fetch compressed size
  if (cbytes != NULL) {
    from_big(cbytes, frame_pointer + FRAME_CBYTES, sizeof(*cbytes));
    if (*cbytes < 0 || (nbytes != NULL && *cbytes > *nbytes)) {
      BLOSC_TRACE_ERROR("Invalid compressed size in header.");
      return BLOSC2_ERROR_INVALID_HEADER;
    }
  }

  // Fetch block size
  if (blocksize != NULL) {
    from_big(blocksize, frame_pointer + FRAME_BLOCKSIZE, sizeof(*blocksize));
    if (*blocksize <= 0 || *blocksize > BLOSC2_MAX_BLOCKSIZE) {
      BLOSC_TRACE_ERROR("Invalid block size in header.");
      return BLOSC2_ERROR_INVALID_HEADER;
    }
  }

  // Fetch chunk size
  if (chunksize != NULL) {
    from_big(chunksize, frame_pointer + FRAME_CHUNKSIZE, sizeof(*chunksize));
    if (*chunksize <= 0 || *chunksize > BLOSC2_MAX_CHUNKSIZE) {
      BLOSC_TRACE_ERROR("Invalid chunk size in header.");
      return BLOSC2_ERROR_INVALID_HEADER;
    }
  }

  // Fetch type size
  if (typesize != NULL) {
    from_big(typesize, frame_pointer + FRAME_TYPESIZE, sizeof(*typesize));
    if (*typesize <= 0 || *typesize > BLOSC2_MAX_TYPESIZE) {
      BLOSC_TRACE_ERROR("Invalid type size in header.");
      return BLOSC2_ERROR_INVALID_HEADER;
