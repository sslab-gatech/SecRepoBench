// Validate header length
  if (*header_len < FRAME_HEADER_MINLEN || *header_len > FRAME_HEADER_MAXLEN) {
    BLOSC_TRACE_ERROR("Header length is out of bounds.");
    return BLOSC2_ERROR_INVALID_HEADER;
  }

  // Fetch and validate frame length
  if (frame_len != NULL) {
    from_big(frame_len, frame_pointer + FRAME_LEN, sizeof(*frame_len));
    if (*frame_len <= 0) {
      BLOSC_TRACE_ERROR("Frame length is invalid.");
      return BLOSC2_ERROR_INVALID_HEADER;
    }
  }

  // Fetch and validate uncompressed size
  if (nbytes != NULL) {
    from_big(nbytes, frame_pointer + FRAME_NBYTES, sizeof(*nbytes));
    if (*nbytes < 0) {
      BLOSC_TRACE_ERROR("Uncompressed size is invalid.");
      return BLOSC2_ERROR_INVALID_HEADER;
    }
  }

  // Fetch and validate compressed size
  if (cbytes != NULL) {
    from_big(cbytes, frame_pointer + FRAME_CBYTES, sizeof(*cbytes));
    if (*cbytes < 0 || (*nbytes > 0 && *cbytes > *nbytes)) {
      BLOSC_TRACE_ERROR("Compressed size is invalid.");
      return BLOSC2_ERROR_INVALID_HEADER;
    }
  }

  // Fetch and validate block size
  if (blocksize != NULL) {
    from_big(blocksize, frame_pointer + FRAME_BLOCKSIZE, sizeof(*blocksize));
    if (*blocksize <= 0) {
      BLOSC_TRACE_ERROR("Block size is invalid.");
      return BLOSC2_ERROR_INVALID_HEADER;
    }
  }

  // Fetch and validate chunk size
  if (chunksize != NULL) {
    from_big(chunksize, frame_pointer + FRAME_CHUNKSIZE, sizeof(*chunksize));
    if (*chunksize <= 0) {
      BLOSC_TRACE_ERROR("Chunk size is invalid.");
      return BLOSC2_ERROR_INVALID_HEADER;
    }
  }

  // Fetch and validate type size
  if (typesize != NULL) {
    from_big(typesize, frame_pointer + FRAME_TYPESIZE, sizeof(*typesize));
    if (*typesize <= 0) {
      BLOSC_TRACE_ERROR("Type size is invalid.");
      return BLOSC2_ERROR_INVALID_HEADER;
