if (*header_len <= 0 || *header_len > FRAME_HEADER_MAXLEN) {
    BLOSC_TRACE_ERROR("Invalid header length in frame.");
    return BLOSC2_ERROR_INVALID_HEADER;
}
from_big(frame_len, frame_pointer + FRAME_FRAME_LEN, sizeof(*frame_len));
if (*frame_len < *header_len || *frame_len > MAX_FRAME_LENGTH) {
    BLOSC_TRACE_ERROR("Invalid frame length in frame.");
    return BLOSC2_ERROR_INVALID_HEADER;
}
from_big(nbytes, frame_pointer + FRAME_NBYTES, sizeof(*nbytes));
if (*nbytes < 0) {
    BLOSC_TRACE_ERROR("Invalid uncompressed size in frame.");
    return BLOSC2_ERROR_INVALID_HEADER;
}
from_big(cbytes, frame_pointer + FRAME_CBYTES, sizeof(*cbytes));
if (*cbytes < 0 || *cbytes > *frame_len - *header_len) {
    BLOSC_TRACE_ERROR("Invalid compressed size in frame.");
    return BLOSC2_ERROR_INVALID_HEADER;
}
from_big(blocksize, frame_pointer + FRAME_BLOCKSIZE, sizeof(*blocksize));
if (*blocksize <= 0 || *blocksize > MAX_BLOCKSIZE) {
    BLOSC_TRACE_ERROR("Invalid block size in frame.");
    return BLOSC2_ERROR_INVALID_HEADER;
}
from_big(chunksize, frame_pointer + FRAME_CHUNKSIZE, sizeof(*chunksize));
if (*chunksize <= 0 || *chunksize > MAX_CHUNKSIZE) {
    BLOSC_TRACE_ERROR("Invalid chunk size in frame.");
    return BLOSC2_ERROR_INVALID_HEADER;
}
from_big(typesize, frame_pointer + FRAME_TYPESIZE, sizeof(*typesize));
if (*typesize <= 0 || *typesize > MAX_TYPESIZE) {
    BLOSC_TRACE_ERROR("Invalid type size in frame.");
    return BLOSC2_ERROR_INVALID_HEADER;
}
