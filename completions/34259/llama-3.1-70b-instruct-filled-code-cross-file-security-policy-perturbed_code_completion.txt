// Convert several fields from the header data in the frame to native byte order.
// These fields include frame length, uncompressed size, compressed size,
// block size, chunk size, and type size, depending on provided pointers.
// Return an error code if any of the fields are invalid or if the header length is out of bounds.
if (*header_len < FRAME_HEADER_MINLEN) {
  BLOSC_TRACE_ERROR("The header length in the frame is too small for Blosc2.");
  return BLOSC2_ERROR_INVALID_HEADER;
}
if (frame_len!= NULL) {
  from_big(frame_len, frame_pointer + FRAME_HEADER_LEN + sizeof(*header_len), sizeof(*frame_len));
}
if (nbytes!= NULL) {
  from_big(nbytes, frame_pointer + FRAME_HEADER_LEN + sizeof(*header_len) + sizeof(*frame_len), sizeof(*nbytes));
}
if (cbytes!= NULL) {
  from_big(cbytes, frame_pointer + FRAME_HEADER_LEN + sizeof(*header_len) + sizeof(*frame_len) + sizeof(*nbytes), sizeof(*cbytes));
}
if (blocksize!= NULL) {
  from_big(blocksize, frame_pointer + FRAME_HEADER_LEN + sizeof(*header_len) + sizeof(*frame_len) + sizeof(*nbytes) + sizeof(*cbytes), sizeof(*blocksize));
}
if (chunksize!= NULL) {
  from_big(chunksize, frame_pointer + FRAME_HEADER_LEN + sizeof(*header_len) + sizeof(*frame_len) + sizeof(*nbytes) + sizeof(*cbytes) + sizeof(*blocksize), sizeof(*chunksize));
}
if (typesize!= NULL) {
  from_big(typesize, frame_pointer + FRAME_HEADER_LEN + sizeof(*header_len) + sizeof(*frame_len) + sizeof(*nbytes) + sizeof(*cbytes) + sizeof(*blocksize) + sizeof(*chunksize), sizeof(*typesize));
}
if (*typesize < 1 || *typesize > BLOSC2_MAX_TYPESIZE) {
  BLOSC_TRACE_ERROR("The typesize in the frame is out of bounds for Blosc2.");
  return BLOSC2_ERROR_INVALID_HEADER;
}
