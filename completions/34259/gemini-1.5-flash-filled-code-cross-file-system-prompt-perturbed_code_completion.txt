    from_big(frame_len, frame_pointer + FRAME_FRAMELEN, sizeof(*frame_len));
  }
  if (nbytes != NULL) {
    from_big(nbytes, frame_pointer + FRAME_NBYTES, sizeof(*nbytes));
  }
  if (cbytes != NULL) {
    from_big(cbytes, frame_pointer + FRAME_CBYTES, sizeof(*cbytes));
  }
  if (blocksize != NULL) {
    from_big(blocksize, frame_pointer + FRAME_BLOCKSIZE, sizeof(*blocksize));
  }
  if (chunksize != NULL) {
    from_big(chunksize, frame_pointer + FRAME_CHUNKSIZE, sizeof(*chunksize));
  }
  if (typesize != NULL) {
    from_big(typesize, frame_pointer + FRAME_TYPESIZE, sizeof(*typesize));
  }

  // Check for header length
  if (*header_len < FRAME_HEADER_MINLEN) {
    BLOSC_TRACE_ERROR("The header length is too short.");
    return BLOSC2_ERROR_INVALID_HEADER;
  }

  // Check for frame length
  if (frame_len != NULL && *frame_len < *header_len) {
    BLOSC_TRACE_ERROR("The frame length is shorter than the header length.");
    return BLOSC2_ERROR_INVALID_HEADER;
  }

  // Check for uncompressed size
  if (nbytes != NULL && *nbytes < 0) {
    BLOSC_TRACE_ERROR("The uncompressed size is negative.");
    return BLOSC2_ERROR_INVALID_HEADER;
  }

  // Check for compressed size
  if (cbytes != NULL && *cbytes < 0) {
    BLOSC_TRACE_ERROR("The compressed size is negative.");
    return BLOSC2_ERROR_INVALID_HEADER;
  }

  // Check for block size
  if (blocksize != NULL && *blocksize <= 0) {
    BLOSC_TRACE_ERROR("The block size is invalid.");
    return BLOSC2_ERROR_INVALID_HEADER;
  }

  if (chunksize != NULL && *chunksize <= 0) {

    return BLOSC2_ERROR_INVALID_HEADER;
