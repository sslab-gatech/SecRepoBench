// Convert several fields from the header data in the frame to native byte order.
// These fields include frame length, uncompressed size, compressed size,
// block size, chunk size, and type size, depending on provided pointers.
// Return an error code if any of the fields are invalid or if the header length is out of bounds.
from_big(&frame_len, header + FRAME_LEN, sizeof(frame_len));
from_big(&nbytes, header + FRAME_NBYTES, sizeof(nbytes));
from_big(&cbytes, header + FRAME_CBYTES, sizeof(cbytes));
from_big(&blocksize, header + FRAME_BLOCKSIZE, sizeof(blocksize));
from_big(&chunksize, header + FRAME_CHUNKSIZE, sizeof(chunksize));
from_big(&typesize, header + FRAME_TYPESIZE, sizeof(typesize));

if (frame_len < 0 || nbytes < 0 || cbytes < 0 || blocksize < 0 || chunksize < 0 || typesize < 0 || *header_len < FRAME_HEADER_MINLEN) {
    return BLOSC2_ERROR_INVALID_HEADER;
}
