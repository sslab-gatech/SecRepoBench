// Check if the header length is valid
if (*header_len < FRAME_HEADER_MINLEN || (frame->cframe != NULL && *header_len > frame->len)) {
  BLOSC_TRACE_ERROR("Invalid header length in frame.");
  return BLOSC2_ERROR_INVALID_HEADER;
}

// Convert and check frame length
if (frame_len != NULL) {
  from_big(frame_len, frame_pointer + FRAME_LEN, sizeof(*frame_len));
  if (frame->cframe != NULL && *frame_len != frame->len) {
    BLOSC_TRACE_ERROR("Invalid frame length in frame header.");
    return BLOSC2_ERROR_INVALID_HEADER;
  }
}

// Get uncompressed size
if (nbytes != NULL) {
  from_big(nbytes, frame_pointer + FRAME_NBYTES, sizeof(*nbytes));
  if (*nbytes < 0) {
    BLOSC_TRACE_ERROR("Invalid uncompressed size in frame header.");
    return BLOSC2_ERROR_INVALID_HEADER;
  }
}

// Get compressed size
if (cbytes != NULL) {
  from_big(cbytes, frame_pointer + FRAME_CBYTES, sizeof(*cbytes));
  if (*cbytes < 0) {
    BLOSC_TRACE_ERROR("Invalid compressed size in frame header.");
    return BLOSC2_ERROR_INVALID_HEADER;
  }
}

// Get block size
if (blocksize != NULL) {
  from_big(blocksize, frame_pointer + FRAME_BLOCKSIZE, sizeof(*blocksize));
  if (*blocksize <= 0) {
    BLOSC_TRACE_ERROR("Invalid block size in frame header.");
    return BLOSC2_ERROR_INVALID_HEADER;
  }
}

// Get chunk size
if (chunksize != NULL) {
  from_big(chunksize, frame_pointer + FRAME_CHUNKSIZE, sizeof(*chunksize));
  // A chunksize of 0 is allowed for variable-length chunks
}

// Get type size
if (typesize != NULL) {
  from_big(typesize, frame_pointer + FRAME_TYPESIZE, sizeof(*typesize));
  if (*typesize <= 0) {
    BLOSC_TRACE_ERROR("Invalid type size in frame header.");
    return BLOSC2_ERROR_INVALID_HEADER;
  }
}
