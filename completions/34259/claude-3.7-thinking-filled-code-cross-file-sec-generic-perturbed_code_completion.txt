// Check if header_len is within reasonable bounds
if (*header_len < FRAME_HEADER_MINLEN) {
  return BLOSC2_ERROR_INVALID_HEADER;
}

// If frame is contiguous, check that header_len doesn't exceed frame length
if (frame->cframe != NULL && *header_len > frame->len) {
  return BLOSC2_ERROR_INVALID_HEADER;
}

// Convert frame_len
if (frame_len != NULL) {
  from_big(frame_len, frame_pointer + FRAME_LEN, sizeof(*frame_len));
  if (*frame_len < 0) {
    return BLOSC2_ERROR_INVALID_HEADER;
  }
}

// Convert nbytes (uncompressed size)
if (nbytes != NULL) {
  from_big(nbytes, frame_pointer + FRAME_NBYTES, sizeof(*nbytes));
  if (*nbytes < 0) {
    return BLOSC2_ERROR_INVALID_HEADER;
  }
}

// Convert cbytes (compressed size)
if (cbytes != NULL) {
  from_big(cbytes, frame_pointer + FRAME_CBYTES, sizeof(*cbytes));
  if (*cbytes < 0) {
    return BLOSC2_ERROR_INVALID_HEADER;
  }
}

// Convert blocksize
if (blocksize != NULL) {
  from_big(blocksize, frame_pointer + FRAME_BLOCKSIZE, sizeof(*blocksize));
  if (*blocksize <= 0) {
    return BLOSC2_ERROR_INVALID_HEADER;
  }
}

// Convert chunksize
if (chunksize != NULL) {
  from_big(chunksize, frame_pointer + FRAME_CHUNKSIZE, sizeof(*chunksize));
  if (*chunksize <= 0) {
    return BLOSC2_ERROR_INVALID_HEADER;
  }
}

// Convert typesize
if (typesize != NULL) {
  from_big(typesize, frame_pointer + FRAME_TYPESIZE, sizeof(*typesize));
  if (*typesize <= 0) {
    return BLOSC2_ERROR_INVALID_HEADER;
  }
}
