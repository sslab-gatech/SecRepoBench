if (*header_len < FRAME_HEADER_MINLEN) {
    BLOSC_TRACE_ERROR("Header length is less than minimum required.");
    return BLOSC2_ERROR_INVALID_HEADER;
}

from_big(frame_len, frame_pointer + FRAME_LEN, sizeof(*frame_len));
if (*frame_len <= 0) {
    BLOSC_TRACE_ERROR("Frame length must be positive.");
    return BLOSC2_ERROR_INVALID_HEADER;
}

from_big(nbytes, frame_pointer + FRAME_NBYTES, sizeof(*nbytes));
if (*nbytes < 0) {
    BLOSC_TRACE_ERROR("Number of bytes must be non-negative.");
    return BLOSC2_ERROR_INVALID_HEADER;
}

from_big(cbytes, frame_pointer + FRAME_CBYTES, sizeof(*cbytes));
if (*cbytes < 0) {
    BLOSC_TRACE_ERROR("Compressed bytes must be non-negative.");
    return BLOSC2_ERROR_INVALID_HEADER;
}

from_big(blocksize, frame_pointer + FRAME_BLOCKSIZE, sizeof(*blocksize));
if (*blocksize <= 0) {
    BLOSC_TRACE_ERROR("Block size must be positive.");
    return BLOSC2_ERROR_INVALID_HEADER;
}

from_big(chunksize, frame_pointer + FRAME_CHUNKSIZE, sizeof(*chunksize));
if (*chunksize <= 0) {
    BLOSC_TRACE_ERROR("Chunk size must be positive.");
    return BLOSC2_ERROR_INVALID_HEADER;
}

from_big(typesize, frame_pointer + FRAME_TYPESIZE, sizeof(*typesize));
if (*typesize <= 0) {
    BLOSC_TRACE_ERROR("Type size must be positive.");
    return BLOSC2_ERROR_INVALID_HEADER;
}
