from_big(frame_len, frame_pointer + FRAME_FRAME_LEN, sizeof(*frame_len));
  from_big(nbytes, frame_pointer + FRAME_NBYTES, sizeof(*nbytes));
  from_big(cbytes, frame_pointer + FRAME_CBYTES, sizeof(*cbytes));
  from_big(blocksize, frame_pointer + FRAME_BLOCKSIZE, sizeof(*blocksize));
  from_big(chunksize, frame_pointer + FRAME_CHUNKSIZE, sizeof(*chunksize));
  from_big(typesize, frame_pointer + FRAME_TYPESIZE, sizeof(*typesize));

  if (*header_len <= 0 || *header_len > FRAME_HEADER_MAXLEN) {
    BLOSC_TRACE_ERROR("Invalid header length.");
    return BLOSC2_ERROR_INVALID_HEADER;
  }
  if (*frame_len <= 0) {
    BLOSC_TRACE_ERROR("Invalid frame length.");
    return BLOSC2_ERROR_INVALID_HEADER;
  }
  if (*nbytes < 0) {
    BLOSC_TRACE_ERROR("Invalid uncompressed size.");
    return BLOSC2_ERROR_INVALID_HEADER;
  }
  if (*cbytes < 0) {
    BLOSC_TRACE_ERROR("Invalid compressed size.");
    return BLOSC2_ERROR_INVALID_HEADER;
  }
  if (*blocksize <= 0) {
    BLOSC_TRACE_ERROR("Invalid block size.");
    return BLOSC2_ERROR_INVALID_HEADER;
  }
  if (*chunksize <= 0) {
    BLOSC_TRACE_ERROR("Invalid chunk size.");
    return BLOSC2_ERROR_INVALID_HEADER;

    BLOSC_TRACE_ERROR("Invalid type size.");
