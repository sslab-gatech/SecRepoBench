if (child->type == RE_NODE_STAR || child->type == RE_NODE_PLUS) {
  if (child->children_head->type == RE_NODE_ANY) {
    RE_AST* new_ast = (RE_AST*)yr_malloc(sizeof(RE_AST));
    if (new_ast == NULL) {
      return ERROR_INSUFFICIENT_MEMORY;
    }
    new_ast->root_node = child;
    new_ast->root_node->next_sibling = child->next_sibling;
    if (child->next_sibling != NULL) {
      child->next_sibling->prev_sibling = NULL;
    }
    child->next_sibling = NULL;

    *remainder_re_ast = new_ast;
    *min_gap = child->min_gap;
    *maximumgap = child->maximumgap;

    return ERROR_SUCCESS;
  }
}
