if (child->type == RE_NODE_RANGE && !child->greedy && child->max >= THRESHOLD)
    {
      *min_gap = child->min;
      *maximumgap = child->max;

      // Create a new AST for the remainder
      *remainder_re_ast = (RE_AST*) yr_malloc(sizeof(RE_AST));
      if (*remainder_re_ast == NULL)
        return ERROR_INSUFFICIENT_MEMORY;

      (*remainder_re_ast)->root_node = child->next_sibling;
      child->next_sibling = NULL;

      // Update the current AST to separate the nodes before the point of splitting
      if (child->prev_sibling != NULL)
        child->prev_sibling->next_sibling = NULL;
      else
        re_ast->root_node->children_head = NULL;

      // Remove the child node from the current AST
      yr_re_node_destroy(child);

      return ERROR_SUCCESS;
    }
