(child->type == RE_NODE_RANGE_ANY && 
     child->greedy == false && 
     (child->end - child->start) > *min_gap)
    {
      *min_gap = child->start;
      *maximumgap = child->end;

      // Create new AST for remainder
      *remainder_re_ast = (RE_AST*) yr_malloc(sizeof(RE_AST));
      if (*remainder_re_ast == NULL)
        return ERROR_INSUFFICIENT_MEMORY;

      (*remainder_re_ast)->root_node = child->next_sibling;
      (*remainder_re_ast)->flags = re_ast->flags;
      (*remainder_re_ast)->code = NULL;

      // Update current AST
      re_ast->root_node->children_tail = child;
      child->next_sibling = NULL;

      return ERROR_SUCCESS;
    }
