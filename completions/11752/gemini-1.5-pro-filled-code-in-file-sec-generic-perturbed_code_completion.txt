(child->type == RE_NODE_RANGE_ANY &&
        !child->greedy &&
        child->start == 0 &&
        child->end > RE_CHAINING_THRESHOLD)
    {
      *min_gap = child->start;
      *maximumgap = child->end;

      FAIL_ON_ERROR(yr_re_ast_create(remainder_re_ast));

      (*remainder_re_ast)->root_node = child->next_sibling;
      (*remainder_re_ast)->flags = re_ast->flags;

      if (child->prev_sibling != NULL)
      {
        concat = yr_re_node_create(RE_NODE_CONCAT);

        if (concat == NULL)
          return ERROR_INSUFFICIENT_MEMORY;

        concat->children_head = re_ast->root_node->children_head;
        concat->children_tail = child->prev_sibling;
        concat->children_tail->next_sibling = NULL;
        re_ast->root_node->children_head = concat;
      }
      else
      {
        re_ast->root_node = NULL;
      }

      child->prev_sibling = NULL;
      child->next_sibling = NULL;
      yr_re_node_destroy(child);

      return ERROR_SUCCESS;
    }
