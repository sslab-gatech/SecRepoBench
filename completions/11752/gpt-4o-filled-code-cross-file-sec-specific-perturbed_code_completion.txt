(child->type == RE_NODE_RANGE_ANY && !child->greedy &&
     (child->end - child->start) > SOME_THRESHOLD)
    {
      // Create a new AST for the remainder
      *remainder_re_ast = (RE_AST*) yr_malloc(sizeof(RE_AST));
      if (*remainder_re_ast == NULL)
        return ERROR_INSUFFICIENT_MEMORY;

      (*remainder_re_ast)->root_node = re_ast->root_node;
      re_ast->root_node = child->next_sibling;

      if (re_ast->root_node != NULL)
        re_ast->root_node->prev_sibling = NULL;

      child->next_sibling = NULL;

      // Update min and max gap
      *min_gap = child->start;
      *maximumgap = child->end;

      // Remove the child node from the original AST
      if (child->prev_sibling != NULL)
        child->prev_sibling->next_sibling = NULL;

      yr_re_node_destroy(child);

      return ERROR_SUCCESS;
    }
