if (child->type == RE_NODE_ANY && child->is_non_greedy) {
    *remainder_re_ast = yr_re_ast_create();
    if (*remainder_re_ast == NULL) {
        return ERROR_INSUFFICIENT_MEMORY;
    }
    (*remainder_re_ast)->root_node = child;
    yr_re_node_remove_child(re_ast->root_node, child);
    *min_gap = child->min;
    *maximumgap = child->max;
    result = ERROR_SUCCESS;
    break;
}
