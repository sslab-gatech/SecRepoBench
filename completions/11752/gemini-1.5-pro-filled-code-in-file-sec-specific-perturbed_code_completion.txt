(child->type == RE_NODE_RANGE_ANY &&
        !child->greedy &&
        child->start == 0 &&
        child->end > RE_CHAINING_THRESHOLD)
    {
      *min_gap = child->start;
      *maximumgap = child->end;

      result = yr_re_ast_create(remainder_re_ast);

      if (result != ERROR_SUCCESS)
        return result;

      (*remainder_re_ast)->root_node = child->next_sibling;
      (*remainder_re_ast)->flags = re_ast->flags;

      concat = re_ast->root_node;
      child = child->prev_sibling;

      if (child != NULL)
      {
        child->next_sibling = NULL;
        concat->children_tail = child;
      }
      else
      {
        concat->children_head = NULL;
        concat->children_tail = NULL;
      }

      yr_free(child->next_sibling); // Free the RE_NODE_RANGE_ANY node
      return ERROR_SUCCESS;
    }
