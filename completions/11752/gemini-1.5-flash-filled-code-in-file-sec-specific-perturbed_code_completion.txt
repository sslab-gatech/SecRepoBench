(child->type == RE_NODE_RANGE_ANY && !child->greedy &&
      child->end > RE_CHAINING_POINT_THRESHOLD)
    {
      RE_AST* remainder_ast;

      result = yr_re_ast_create(&remainder_ast);

      if (result != ERROR_SUCCESS)
        return result;

      remainder_ast->root_node = child->next_sibling;
      child->next_sibling = NULL;

      *remainder_re_ast = remainder_ast;
      *min_gap = child->start;
      *maximumgap = child->end;

      return ERROR_SUCCESS;
    }
