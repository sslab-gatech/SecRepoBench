(
      child->type == RE_NODE_ANY && 
      child->flags & RE_FLAGS_NON_GREEDY && 
      child->range_max - child->range_min > MAX_SPLIT_THRESHOLD
    )
    {
      RE_AST* new_ast = (RE_AST*) yr_malloc(sizeof(RE_AST));
      if (new_ast == NULL)
        return ERROR_INSUFFICIENT_MEMORY;

      new_ast->root_node = child->next_sibling;
      child->next_sibling = NULL;

      if (new_ast->root_node != NULL)
      {
        new_ast->root_node->prev_sibling = NULL;
      }

      *remainder_re_ast = new_ast;
      *min_gap = child->range_min;
      *maximumgap = child->range_max;

      // Remove the chaining point node from the original AST.
      if (child->prev_sibling != NULL)
        child->prev_sibling->next_sibling = NULL;

      if (re_ast->root_node->children_head == child)
        re_ast->root_node->children_head = child->prev_sibling;

      re_ast->root_node->children_tail = child->prev_sibling;

      result = ERROR_SUCCESS;
      goto exit;
    }
