if (child->type == RE_NODE_RANGE_ANY &&
   !child->greedy &&
    child->start == 0 &&
    child->end > RE_CHAINING_POINT_THRESHOLD)
{
  RE_AST* new_re_ast;
  RE_AST* remainder_re_ast;
  int32_t min_gap;
  int32_t max_gap;

  *result_re_ast = yr_re_node_create(RE_NODE_CONCAT);
  yr_re_node_append_child(*result_re_ast, re_ast->root_node->children_head);

  new_re_ast = yr_re_node_create(RE_NODE_CONCAT);
  yr_re_node_append_child(new_re_ast, child);

  remainder_re_ast = yr_re_node_create(RE_NODE_CONCAT);
  yr_re_node_append_child(remainder_re_ast, child->next_sibling);

  yr_re_node_append_child(*result_re_ast, new_re_ast);
  yr_re_node_append_child(*result_re_ast, remainder_re_ast);

  min_gap = child->end;
  max_gap = child->end;

  yr_re_node_destroy(re_ast->root_node);

  re_ast->root_node = *result_re_ast;
  *remainder_re_ast = remainder_re_ast;
  *min_gap = min_gap;
  *maximumgap = max_gap;

  return ERROR_SUCCESS;
}
