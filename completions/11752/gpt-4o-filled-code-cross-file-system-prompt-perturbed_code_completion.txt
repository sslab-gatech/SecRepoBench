(child->type == RE_NODE_RANGE_ANY && !child->greedy &&
     (child->end - child->start) > SOME_THRESHOLD)
    {
      *min_gap = child->start;
      *maximumgap = child->end;

      // Create a new AST for the remainder
      *remainder_re_ast = yr_re_ast_create();
      if (*remainder_re_ast == NULL)
        return ERROR_INSUFFICIENT_MEMORY;

      // Set the root node of the remainder AST
      (*remainder_re_ast)->root_node = yr_re_node_create(RE_NODE_CONCAT);
      if ((*remainder_re_ast)->root_node == NULL)
      {
        yr_re_ast_destroy(*remainder_re_ast);
        return ERROR_INSUFFICIENT_MEMORY;
      }

      // Append the remaining siblings to the new AST
      while (child->next_sibling != NULL)
      {
        yr_re_node_append_child((*remainder_re_ast)->root_node, child->next_sibling);
        child->next_sibling = child->next_sibling->next_sibling;
      }

      // Remove the current child from the original AST
      if (child->prev_sibling != NULL)
        child->prev_sibling->next_sibling = NULL;
      else
        re_ast->root_node->children_head = NULL;

      yr_re_node_destroy(child);

      return ERROR_SUCCESS;
    }
