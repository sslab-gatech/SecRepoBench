if (!strncmp(data+index, "esds", 4)) {
	GF_ESDBox *esds = NULL;
	if (ptr->esd) {
		gf_isom_box_del_parent(&ptr->child_boxes, (GF_Box *)ptr->esd);
		ptr->esd = NULL;
	}
	bs = gf_bs_new(data+index, size-index, GF_BITSTREAM_READ);
	gf_bs_set_cookie(bs, GF_ISOM_BS_COOKIE_NO_LOGS);
	e = esds_box_read((GF_Box*)&esds, bs);
	gf_bs_del(bs);
	if (!e) {
		if (esds->type == GF_ISOM_BOX_TYPE_ESDS) {
			ptr->esd = esds;
			gf_list_add(ptr->child_boxes, esds);
			return GF_OK;
		} else {
			gf_isom_box_del((GF_Box*)esds);
		}
	} else {
		gf_bs_del(bs);
	}
}
