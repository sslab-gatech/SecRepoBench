if (!strncmp(data+index, "esds", 4)) {
	GF_ESDBox *esds = NULL;
	GF_BitStream *temp_bs = gf_bs_new(data+index-4, size-index+4, GF_BITSTREAM_READ);
	gf_bs_set_cookie(temp_bs, GF_ISOM_BS_COOKIE_NO_LOGS);
	if (ptr->esd) {
		gf_list_del_item(ptr->child_boxes, ptr->esd);
		gf_isom_box_del((GF_Box*)ptr->esd);
		ptr->esd = NULL;
	}
	e = gf_isom_box_parse((GF_Box **)&esds, temp_bs);
	gf_bs_del(temp_bs);
	if (!e && (esds->type == GF_ISOM_BOX_TYPE_ESDS)) {
		ptr->esd = esds;
		gf_list_add(ptr->child_boxes, ptr->esd);
		e = GF_OK;
		break;
	} else if (esds) {
		gf_isom_box_del((GF_Box *)esds);
	}
}
