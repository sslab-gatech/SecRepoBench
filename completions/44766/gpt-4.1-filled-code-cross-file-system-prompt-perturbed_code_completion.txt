if ((data[index+4]=='e') && (data[index+5]=='s') && (data[index+6]=='d') && (data[index+7]=='s')) {
	GF_BitStream *esds_bs = gf_bs_new((u8*)data+index, size-index, GF_BITSTREAM_READ);
	if (esds_bs) {
		gf_bs_set_cookie(esds_bs, GF_ISOM_BS_COOKIE_SUPPRESS_LOG);
		if (ptr->esd) {
			gf_isom_box_remove_child((GF_Box*)ptr, (GF_Box*)ptr->esd);
			gf_isom_box_del((GF_Box*)ptr->esd);
			ptr->esd = NULL;
		}
		GF_Box *esds_box = NULL;
		GF_Err esds_err = gf_isom_box_parse(&esds_box, esds_bs);
		if (!esds_err && esds_box && (esds_box->type==GF_ISOM_BOX_TYPE_ESDS)) {
			ptr->esd = (GF_ESDBox*)esds_box;
			gf_isom_box_add_child((GF_Box*)ptr, esds_box);
		} else if (esds_box) {
			gf_isom_box_del(esds_box);
		}
		gf_bs_del(esds_bs);
	}
	break;
}
