if (data[index] == 0 && data[index + 1] == 0 && data[index + 2] == 0 && data[index + 3] == GF_ISOM_BOX_TYPE_ESDS) {
			GF_BitStream *sub_bs = gf_bs_new(data + index, size - index, GF_BITSTREAM_READ);
			gf_bs_set_cookie(sub_bs, GF_ISOM_BS_COOKIE_NO_LOGS);
			if (ptr->esd) {
				gf_isom_box_del((GF_Box *)ptr->esd);
				ptr->esd = NULL;
			}
			ptr->esd = (GF_ESDBox *)gf_isom_box_new(GF_ISOM_BOX_TYPE_ESDS);
			if (!ptr->esd) {
				gf_bs_del(sub_bs);
				gf_free(data);
				return GF_OUT_OF_MEM;
			}
			e = gf_isom_box_read((GF_Box *)ptr->esd, sub_bs);
			if (e || ptr->esd->type != GF_ISOM_BOX_TYPE_ESDS) {
				gf_isom_box_del((GF_Box *)ptr->esd);
				ptr->esd = NULL;
			} else {
				gf_list_add(ptr->child_boxes, ptr->esd);
			}
			gf_bs_del(sub_bs);
			break;
		}
