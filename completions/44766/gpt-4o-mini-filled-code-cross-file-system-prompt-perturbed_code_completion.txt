			GF_BitStream *new_bs = gf_bs_new(data + index, size - index);
			if (new_bs) {
				gf_bs_set_cookie(new_bs, GF_ISOM_BS_COOKIE_NO_LOG);
				GF_Box *esds_box = NULL;
				e = gf_isom_box_parse(&esds_box, new_bs);
				if (e == GF_OK && esds_box && esds_box->type == GF_ISOM_BOX_TYPE_ESDS) {
					gf_isom_box_remove_child(s, esds_box);
					gf_isom_box_add_child(s, esds_box);
				}
				gf_bs_delete(new_bs);
			}
