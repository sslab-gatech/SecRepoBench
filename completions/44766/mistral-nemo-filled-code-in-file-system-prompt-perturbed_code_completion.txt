// Masked region for FilePartitionBox
GF_Err file_partition_box_read(GF_Box *s, GF_BitStream *bs)
{
    GF_FilePartitionBox *ptr = (GF_FilePartitionBox *)s;

    // Masked region start
    ISOM_DECREASE_SIZE(ptr, 12);
    ptr->partition_id = gf_bs_read_u32(bs);
    ptr->partition_size = gf_bs_read_u64(bs);
    ptr->partition_offset = gf_bs_read_u64(bs);
    // Masked region end

    return GF_OK;
}
