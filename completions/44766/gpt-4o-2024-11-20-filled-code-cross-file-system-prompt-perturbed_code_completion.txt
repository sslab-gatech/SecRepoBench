if (data[index] == 'e' && data[index + 1] == 's' && data[index + 2] == 'd' && data[index + 3] == 's') {
			GF_BitStream *esds_bs;
			GF_ESDBox *esds_box = NULL;

			// Create a new bitstream starting at the identified location
			esds_bs = gf_bs_new((unsigned char *)(data + index), size - index, GF_BITSTREAM_READ);
			if (!esds_bs) {
				gf_free(data);
				return GF_OUT_OF_MEM;
			}

			// Set a cookie to suppress logging
			gf_bs_set_cookie(esds_bs, GF_ISOM_BS_COOKIE_SUPPRESS_LOG);

			// Check if an ESDS box already exists and remove it
			if (ptr->esd) {
				gf_list_rem(ptr->child_boxes, ptr->esd);
				gf_isom_box_del((GF_Box *)ptr->esd);
				ptr->esd = NULL;
			}

			// Parse a new ESDS box
			e = gf_isom_box_parse((GF_Box **)&esds_box, esds_bs);
			if (!e && esds_box && esds_box->type == GF_ISOM_BOX_TYPE_ESDS) {
				ptr->esd = esds_box;
				gf_list_add(ptr->child_boxes, esds_box);
			} else if (esds_box) {
				gf_isom_box_del((GF_Box *)esds_box);
			}

			// Clean up
			gf_bs_del(esds_bs);
			break;
		}
