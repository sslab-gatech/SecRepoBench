GF_BitStream *new_bs = gf_bs_new(data + index, bytesToRead - index, GF_BITSTREAM_READ);
		if (!new_bs) return GF_OUT_OF_MEM;
		gf_bs_set_cookie(new_bs, GF_ISOM_BS_COOKIE_NO_LOGS);
		if (ptr->esd) {
			gf_odf_desc_del((GF_Descriptor*)ptr->esd);
			ptr->esd = NULL;
		}
		e = gf_odf_desc_read(data + index, bytesToRead - index, (GF_Descriptor **) &ptr->esd);
		gf_bs_del(new_bs);
		if (e == GF_OK && ptr->esd && (ptr->esd->tag == GF_ODF_ESD_TAG)) {
			gf_list_add(ptr->child_boxes, (GF_Box *)ptr->esd);
		}
		break;
