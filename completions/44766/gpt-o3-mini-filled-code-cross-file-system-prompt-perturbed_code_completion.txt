if (!memcmp(&data[index], "esds", 4)) {
    GF_Box *esdsBox = NULL;
    GF_Box *oldEsds = gf_isom_box_find_child(s->child_boxes, GF_ISOM_BOX_TYPE_ESDS);
    GF_BitStream *sub_bs = gf_bs_new_memory((u8*)(data + index), size - index);
    /* Set cookie on the new bitstream to suppress logging */
    gf_bs_set_cookie(sub_bs, gf_bs_get_cookie(bs));
    if (oldEsds) {
        gf_isom_box_remove_child(s, oldEsds);
        gf_isom_box_delete(oldEsds);
    }
    e = gf_isom_box_parse(&esdsBox, sub_bs);
    if (!e && esdsBox && esdsBox->type == GF_ISOM_BOX_TYPE_ESDS) {
        gf_isom_box_add(s, esdsBox);
    } else if (esdsBox) {
        gf_isom_box_delete(esdsBox);
    }
    gf_bs_delete(sub_bs);
    break;
}
