u32 box_size;
		GF_BitStream *sub_bs;
		GF_Box *esds_box;
		u32 i;

		for (i = 0; i < size - 8; i++) {
			if (data[i] == 'E' && data[i + 1] == 'S' && data[i + 2] == 'D' && data[i + 3] == 'S') {
				box_size = gf_bs_read_u32(&bs);
				sub_bs = gf_bs_new(data + i, box_size);
				gf_bs_set_cookie(sub_bs, GF_ISOM_BS_COOKIE_NO_LOG);
				esds_box = gf_isom_box_find_child(s->child_boxes, GF_ISOM_BOX_TYPE_ESDS);
				if (esds_box) {
					gf_isom_box_remove_child(s, esds_box);
					gf_isom_box_del(esds_box);
				}
				e = gf_isom_box_parse(&esds_box, sub_bs);
				if (!e && esds_box->type == GF_ISOM_BOX_TYPE_ESDS) {
					gf_isom_box_add_child(s, esds_box);
				}
				gf_bs_del(sub_bs);
				break;
			}
