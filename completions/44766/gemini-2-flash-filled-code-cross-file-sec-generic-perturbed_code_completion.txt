			if (ptr->esds) {
				gf_isom_box_remove_child(s, ptr->esds);
				gf_isom_box_del(ptr->esds);
				ptr->esds = NULL;
			}

			e = gf_isom_box_parse(&esds_box, esds_bs);
			if (!e && esds_box && esds_box->type == GF_ISOM_BOX_TYPE_ESDS) {
				ptr->esds = (GF_ESDescriptorBox*) esds_box;
				gf_isom_box_add_child(s, esds_box);
				e = GF_OK;
			} else {
				gf_isom_box_del(esds_box);
			}
			gf_bs_close(esds_bs);
			gf_bs_seek(bs, cur_pos);
