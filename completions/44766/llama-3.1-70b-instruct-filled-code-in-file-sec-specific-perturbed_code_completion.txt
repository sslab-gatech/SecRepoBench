GF_Box *audio_sample_entry_box_new()
{
    GF_SAFEALLOC(tmp, GF_MPEGAudioSampleEntryBox);
    if (tmp == NULL) return NULL;
    gf_isom_audio_sample_entry_init((GF_AudioSampleEntryBox*)tmp);
    tmp->internal_type = GF_ISOM_SAMPLE_ENTRY_AUDIO;
    tmp->type = GF_ISOM_BOX_TYPE_MP4A;
    return (GF_Box *)tmp;
}
