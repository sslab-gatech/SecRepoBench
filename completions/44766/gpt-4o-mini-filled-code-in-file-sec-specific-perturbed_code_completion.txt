u8 iv_size = gf_bs_read_u8(bs);
            if (iv_size > 16) {
                GF_LOG(GF_LOG_ERROR, GF_LOG_CONTAINER, ("[iso file] Invalid IV size in SEIG entry\n"));
                gf_free(ptr);
                return NULL;
            }
            memcpy(ptr->key_info + 4, kid, 16);
            if (iv_size) {
                ptr->key_info[20] = iv_size;
                gf_bs_read_data(bs, ptr->key_info + 21, iv_size);
                *total_bytes += 1 + iv_size;
            }
            else {
                ptr->key_info[20] = 0;
            }
            *total_bytes += 20;
        }
    }
