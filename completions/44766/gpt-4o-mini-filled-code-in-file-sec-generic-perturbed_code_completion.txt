u8 ivsize = gf_bs_read_u8(bs);
            if (ivsize > 16) {
                GF_LOG(GF_LOG_ERROR, GF_LOG_CONTAINER, ("[iso file] Invalid IV size %u in SEIG entry\n", ivsize));
                gf_free(ptr);
                return NULL;
            }
            memcpy(ptr->iv, &tmpName[17], ivsize);
            ptr->ivsize = ivsize;
            bytes_in_box -= 17 + ivsize;
            if (bytes_in_box < 0) {
                gf_free(ptr);
                return NULL;
            }
            *total_bytes = 17 + ivsize;
        } else {
            ptr->ivsize = 0;
            *total_bytes = 17;
        }
        return ptr;
