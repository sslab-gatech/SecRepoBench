if (off_pos < BLOSC_EXTENDED_HEADER_LENGTH || off_pos > frame->len) {
        BLOSC_TRACE_ERROR("Computed offset position is out of bounds.");
        return NULL;
    }

    uint8_t* start_pointer = (uint8_t*)frame->cframe + off_pos;
    if (compressed_chunk_size != NULL) {
        *compressed_chunk_size = (int32_t)(frame->len - off_pos);
    }
    return start_pointer;
