if (off_pos > frame->len) {
      BLOSC_TRACE_ERROR("Offset position exceeds frame length.");
      return NULL;
    }
    uint8_t* coffsets_ptr = frame->cframe + off_pos;
    if (compressed_chunk_size != NULL) {
      rc = blosc2_cbuffer_sizes(coffsets_ptr, NULL, &chunk_cbytes, NULL);
      if (rc < 0) {
        return NULL;
      }
      *compressed_chunk_size = (int32_t)chunk_cbytes;
    }
    return coffsets_ptr;
