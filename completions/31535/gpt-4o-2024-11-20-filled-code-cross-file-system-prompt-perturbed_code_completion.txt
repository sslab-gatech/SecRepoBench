int32_t coffsets_cbytes = (int32_t)(frame->len - off_pos);
    if (coffsets_cbytes <= 0 || off_pos + coffsets_cbytes > frame->len) {
      BLOSC_TRACE_ERROR("Invalid offsets position or size.");
      return NULL;
    }

    if (compressed_chunk_size != NULL) {
      *compressed_chunk_size = coffsets_cbytes;
    }

    uint8_t* coffsets = frame->cframe + off_pos;
    return coffsets;
