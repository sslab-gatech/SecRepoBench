if (off_pos < 0 || off_pos + BLOSC_MIN_HEADER_LENGTH > frame->len) {
      BLOSC_TRACE_ERROR("Cannot read the offsets out of the frame.");
      return NULL;
    }
    uint8_t* coffsets = frame->cframe + off_pos;
    if (compressed_chunk_size != NULL) {
      rc = blosc2_cbuffer_sizes(coffsets, NULL, compressed_chunk_size, NULL);
      if (rc < 0) {
        BLOSC_TRACE_ERROR("Cannot get the compressed chunk size.");
        return NULL;
      }
    }
    return coffsets;
