if (off_pos + BLOSC_EXTENDED_HEADER_LENGTH > frame->len) {
      BLOSC_TRACE_ERROR("Not enough room for Blosc header in frame.");
      return NULL;
    }
    uint8_t* start = frame->cframe + off_pos;
    if (compressed_chunk_size != NULL) {
      rc = blosc2_cbuffer_sizes(start, NULL, &chunk_cbytes, NULL);
      if (rc < 0) {
        return NULL;
      }
      *compressed_chunk_size = (int32_t)chunk_cbytes;
    }
    return start;
