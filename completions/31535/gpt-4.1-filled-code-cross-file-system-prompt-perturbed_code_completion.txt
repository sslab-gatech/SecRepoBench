if (off_pos + BLOSC_MAX_OVERHEAD > frame->len) {
  BLOSC_TRACE_ERROR("Not enough room to read the offsets out of the frame.");
  return NULL;
}
uint8_t* coffsets = frame->cframe + off_pos;
if (compressed_chunk_size != NULL) {
  rc = blosc2_cbuffer_sizes(coffsets, NULL, &chunk_cbytes, NULL);
  if (rc < 0) {
    return NULL;
  }
  *compressed_chunk_size = (int32_t)chunk_cbytes;
}
frame->coffsets = coffsets;
return coffsets;
