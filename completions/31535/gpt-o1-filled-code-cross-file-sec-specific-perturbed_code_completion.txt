if (off_pos + (int64_t)BLOSC_MIN_HEADER_LENGTH > frame->len) {
  BLOSC_TRACE_ERROR("Not enough space in 'cframe' to read the offsets chunk header.");
  return NULL;
}
uint8_t* offstart = frame->cframe + off_pos;
rc = blosc2_cbuffer_sizes(offstart, NULL, &chunk_cbytes, NULL);
if (rc < 0) {
  return NULL;
}
if (off_pos + chunk_cbytes > frame->len) {
  BLOSC_TRACE_ERROR("Not enough space in 'cframe' to read the complete offsets chunk.");
  return NULL;
}
if (compressed_chunk_size != NULL) {
  *compressed_chunk_size = chunk_cbytes;
}
return offstart;
