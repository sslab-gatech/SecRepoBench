if (header_len + off_pos + BLOSC_MIN_HEADER_LENGTH > frame->len) {
      return NULL;
    }
    uint8_t *start_offset = frame->cframe + off_pos;
    if (compressed_chunk_size != NULL) {
      rc = blosc2_cbuffer_sizes(start_offset, NULL, &chunk_cbytes, NULL);
      if (rc < 0) {
        return NULL;
      }
      *compressed_chunk_size = chunk_cbytes;
    }
    return start_offset;
