uint8_t* start = frame->cframe + off_pos;
    // Check if we can safely read the Blosc header
    if (frame->len < off_pos + BLOSC_EXTENDED_HEADER_LENGTH) {
        BLOSC_TRACE_ERROR("Not enough space for Blosc header in frame.");
        return NULL;
    }
    if (compressed_chunk_size != NULL) {
        rc = blosc2_cbuffer_sizes(start, NULL, &chunk_cbytes, NULL);
        if (rc < 0) {
            return NULL;
        }
        *compressed_chunk_size = (int32_t)chunk_cbytes;
    }
    return start;
