// Calculate the start pointer for the offsets based on the current frame position.
  uint8_t* framep = frame->cframe;
  if (frame->sframe) {
    framep = sframe_open_index(frame->urlpath, "rb");
    fseek(framep, header_len, SEEK_SET);
  } else {
    fseek(framep, header_len + cbytes, SEEK_SET);
  }
  size_t rbytes = fread(coffsets, 1, (size_t)coffsets_cbytes, framep);
  if (rbytes != (size_t)coffsets_cbytes) {
    BLOSC_TRACE_ERROR("Cannot read the offsets out of the frame.");
    free(coffsets);
    return NULL;
  }
  frame->coffsets = coffsets;
  return coffsets;
