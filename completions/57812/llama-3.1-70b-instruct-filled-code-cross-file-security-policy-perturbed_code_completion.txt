if (pictureparameterset->num_ref_loc_offsets > MAX_REF_LOC_OFFSETS) {
    av_log(avctx, AV_LOG_ERROR,
           "Number of reference location offsets exceeds maximum allowed value.\n");
    return AVERROR_INVALIDDATA;
}

for (int i = 0; i < pictureparameterset->num_ref_loc_offsets; i++) {
    pictureparameterset->ref_loc_offset_layer_id[i] = get_bits(gb, 6);
    if (pictureparameterset->ref_loc_offset_layer_id[i] >= sps->num_layers) {
        av_log(avctx, AV_LOG_ERROR,
               "Reference location offset layer ID is out of range.\n");
        return AVERROR_INVALIDDATA;
    }

    pictureparameterset->scaled_ref_layer_offsets_present_flag[i] = get_bits1(gb);
    if (pictureparameterset->scaled_ref_layer_offsets_present_flag[i]) {
        pictureparameterset->scaled_ref_layer_left_offset[i] = get_ue_golomb_long(gb);
        pictureparameterset->scaled_ref_layer_top_offset[i] = get_ue_golomb_long(gb);
        pictureparameterset->scaled_ref_layer_right_offset[i] = get_ue_golomb_long(gb);
        pictureparameterset->scaled_ref_layer_bottom_offset[i] = get_ue_golomb_long(gb);
    }

    pictureparameterset->ref_region_offsets_present_flag[i] = get_bits1(gb);
    if (pictureparameterset->ref_region_offsets_present_flag[i]) {
        pictureparameterset->ref_region_left_offset[i] = get_ue_golomb_long(gb);
        pictureparameterset->ref_region_top_offset[i] = get_ue_golomb_long(gb);
        pictureparameterset->ref_region_right_offset[i] = get_ue_golomb_long(gb);
        pictureparameterset->ref_region_bottom_offset[i] = get_ue_golomb_long(gb);
    }

    pictureparameterset->resample_phase_set_present_flag[i] = get_bits1(gb);
    if (pictureparameterset->resample_phase_set_present_flag[i]) {
        pictureparameterset->resample_phase_set_luma[i] = get_bits(gb, 5);
        pictureparameterset->resample_phase_set_chroma[i] = get_bits(gb, 5);
    }
}
