#define MAX_REF_LOC_OFFSETS 32
if (pictureparameterset->num_ref_loc_offsets > MAX_REF_LOC_OFFSETS) {
    av_log(avctx, AV_LOG_ERROR,
           "num_ref_loc_offsets (%d) exceeds maximum allowed (%d).\n",
           pictureparameterset->num_ref_loc_offsets, MAX_REF_LOC_OFFSETS);
    return AVERROR_INVALIDDATA;
}
for (int i = 0; i < pictureparameterset->num_ref_loc_offsets; i++) {
    /* Read 6-bit layer ID for the current reference location */
    pictureparameterset->ref_loc_layer_id[i] = get_bits(gb, 6);

    /* Read flag indicating if scaled reference layer offsets are present */
    pictureparameterset->scaled_ref_layer_offset_flag[i] = get_bits1(gb);
    if (pictureparameterset->scaled_ref_layer_offset_flag[i])
        pictureparameterset->ref_layer_offset[i] = get_se_golomb_long(gb);
    else
        pictureparameterset->ref_layer_offset[i] = 0;

    /* Read flag indicating if reference region offsets are present */
    pictureparameterset->ref_region_offset_present_flag[i] = get_bits1(gb);
    if (pictureparameterset->ref_region_offset_present_flag[i])
        pictureparameterset->ref_region_offset[i] = get_se_golomb_long(gb);
    else
        pictureparameterset->ref_region_offset[i] = 0;

    /* Read flag indicating if resample phase set is present */
    pictureparameterset->resample_phase_set_present_flag[i] = get_bits1(gb);
    if (pictureparameterset->resample_phase_set_present_flag[i]) {
        pictureparameterset->luma_resample_phase[i]   = get_ue_golomb_long(gb);
        pictureparameterset->chroma_resample_phase[i] = get_ue_golomb_long(gb);
    } else {
        pictureparameterset->luma_resample_phase[i]   = 0;
        pictureparameterset->chroma_resample_phase[i] = 0;
    }
}
