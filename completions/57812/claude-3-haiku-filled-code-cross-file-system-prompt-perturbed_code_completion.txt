// Parse reference location offsets for multiple layers in the PPS
// Iterate over all reference location offsets specified by num_ref_loc_offsets
for (int i = 0; i < pictureparameterset->num_ref_loc_offsets; i++) {
    int layer_id = get_bits(gb, 6);
    int scaled_ref_layer_offset_present_flag = get_bits1(gb);
    int ref_region_offset_present_flag = get_bits1(gb);
    int resample_phase_set_present_flag = get_bits1(gb);

    if (scaled_ref_layer_offset_present_flag) {
        pictureparameterset->scaled_ref_layer_offset[i][0] = get_se_golomb_long(gb);
        pictureparameterset->scaled_ref_layer_offset[i][1] = get_se_golomb_long(gb);
    }

    if (ref_region_offset_present_flag) {
        pictureparameterset->ref_region_offset[i][0] = get_se_golomb_long(gb);
        pictureparameterset->ref_region_offset[i][1] = get_se_golomb_long(gb);
        pictureparameterset->ref_region_offset[i][2] = get_se_golomb_long(gb);
        pictureparameterset->ref_region_offset[i][3] = get_se_golomb_long(gb);
    }

    if (resample_phase_set_present_flag) {
        pictureparameterset->resample_phase_set[i][0][0] = get_se_golomb_long(gb);
        pictureparameterset->resample_phase_set[i][0][1] = get_se_golomb_long(gb);
        pictureparameterset->resample_phase_set[i][1][0] = get_se_golomb_long(gb);
        pictureparameterset->resample_phase_set[i][1][1] = get_se_golomb_long(gb);
    }
}
