for (int i = 0; i < pictureparameterset->num_ref_loc_offsets; i++) {
    int layer_id = get_bits(gb, 6);
    if (layer_id >= sps->max_layers) {
        av_log(avctx, AV_LOG_ERROR,
               "Reference location offset layer ID %d exceeds maximum layer ID %d.\n",
               layer_id, sps->max_layers);
        return AVERROR_INVALIDDATA;
    }

    pictureparameterset->ref_loc_offset_layer_id[i] = layer_id;

    int scaled_ref_layer_offset_present_flag = get_bits1(gb);
    if (scaled_ref_layer_offset_present_flag) {
        pictureparameterset->scaled_ref_layer_offset[i] = get_ue_golomb_long(gb);
    }

    int ref_region_offset_present_flag = get_bits1(gb);
    if (ref_region_offset_present_flag) {
        pictureparameterset->ref_region_offset[i] = get_ue_golomb_long(gb);
    }

    int resample_phase_set_present_flag = get_bits1(gb);
    if (resample_phase_set_present_flag) {
        pictureparameterset->resample_phase_set[i].luma = get_bits(gb, 5);
        pictureparameterset->resample_phase_set[i].chroma = get_bits(gb, 5);
    }
}
