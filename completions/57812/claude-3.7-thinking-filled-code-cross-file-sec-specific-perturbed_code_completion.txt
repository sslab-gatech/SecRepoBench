// Validate num_ref_loc_offsets to prevent buffer overflow
if (pictureparameterset->num_ref_loc_offsets > 64) {
    av_log(avctx, AV_LOG_ERROR,
           "num_ref_loc_offsets exceeds maximum value.\n");
    return AVERROR_INVALIDDATA;
}

// Process each reference location offset
for (int i = 0; i < pictureparameterset->num_ref_loc_offsets; i++) {
    // Read layer ID
    pictureparameterset->ref_loc_offset_layer_id[i] = get_bits(gb, 6);
    
    // Read and store scaled reference layer offset flags and values
    pictureparameterset->scaled_ref_layer_offset_present_flag[i] = get_bits1(gb);
    if (pictureparameterset->scaled_ref_layer_offset_present_flag[i]) {
        pictureparameterset->scaled_ref_layer_left_offset[i] = get_se_golomb_long(gb);
        pictureparameterset->scaled_ref_layer_top_offset[i] = get_se_golomb_long(gb);
        pictureparameterset->scaled_ref_layer_right_offset[i] = get_se_golomb_long(gb);
        pictureparameterset->scaled_ref_layer_bottom_offset[i] = get_se_golomb_long(gb);
    }
    
    // Read and store reference region offset flags and values
    pictureparameterset->ref_region_offset_present_flag[i] = get_bits1(gb);
    if (pictureparameterset->ref_region_offset_present_flag[i]) {
        pictureparameterset->ref_region_left_offset[i] = get_se_golomb_long(gb);
        pictureparameterset->ref_region_top_offset[i] = get_se_golomb_long(gb);
        pictureparameterset->ref_region_right_offset[i] = get_se_golomb_long(gb);
        pictureparameterset->ref_region_bottom_offset[i] = get_se_golomb_long(gb);
    }
    
    // Read and store resample phase set flags and values
    pictureparameterset->resample_phase_set_present_flag[i] = get_bits1(gb);
    if (pictureparameterset->resample_phase_set_present_flag[i]) {
        pictureparameterset->phase_hor_luma[i] = get_ue_golomb_long(gb);
        pictureparameterset->phase_ver_luma[i] = get_ue_golomb_long(gb);
        pictureparameterset->phase_hor_chroma[i] = get_ue_golomb_long(gb);
        pictureparameterset->phase_ver_chroma[i] = get_ue_golomb_long(gb);
    }
}
