if (pictureparameterset->num_ref_loc_offsets < 0 || pictureparameterset->num_ref_loc_offsets > MAX_REF_LOC_OFFSETS) {
        av_log(avctx, AV_LOG_ERROR, "num_ref_loc_offsets %d is out of range [0, %d]\n",
               pictureparameterset->num_ref_loc_offsets, MAX_REF_LOC_OFFSETS);
        return AVERROR_INVALIDDATA;
    }

    for (int i = 0; i < pictureparameterset->num_ref_loc_offsets; i++) {
        pictureparameterset->ref_loc_offsets[i].layer_id = get_bits(gb, 6);
        pictureparameterset->ref_loc_offsets[i].scaled_ref_layer_offset_present_flag = get_bits1(gb);
        if (pictureparameterset->ref_loc_offsets[i].scaled_ref_layer_offset_present_flag) {
            pictureparameterset->ref_loc_offsets[i].scaled_ref_layer_offset_x = get_se_golomb_long(gb);
            pictureparameterset->ref_loc_offsets[i].scaled_ref_layer_offset_y = get_se_golomb_long(gb);
            pictureparameterset->ref_loc_offsets[i].scaled_ref_layer_offset_width = get_se_golomb_long(gb);
            pictureparameterset->ref_loc_offsets[i].scaled_ref_layer_offset_height = get_se_golomb_long(gb);
        }

        pictureparameterset->ref_loc_offsets[i].ref_region_offset_present_flag = get_bits1(gb);
        if (pictureparameterset->ref_loc_offsets[i].ref_region_offset_present_flag) {
            pictureparameterset->ref_loc_offsets[i].ref_region_offset_x = get_se_golomb_long(gb);
            pictureparameterset->ref_loc_offsets[i].ref_region_offset_y = get_se_golomb_long(gb);
            pictureparameterset->ref_loc_offsets[i].ref_region_offset_width = get_se_golomb_long(gb);
            pictureparameterset->ref_loc_offsets[i].ref_region_offset_height = get_se_golomb_long(gb);
        }

        pictureparameterset->ref_loc_offsets[i].resample_phase_present_flag = get_bits1(gb);
        if (pictureparameterset->ref_loc_offsets[i].resample_phase_present_flag) {
            pictureparameterset->ref_loc_offsets[i].phase_hor_luma = get_ue_golomb_long(gb);
            pictureparameterset->ref_loc_offsets[i].phase_ver_luma = get_ue_golomb_long(gb);
            pictureparameterset->ref_loc_offsets[i].phase_hor_chroma = get_ue_golomb_long(gb);
            pictureparameterset->ref_loc_offsets[i].phase_ver_chroma = get_ue_golomb_long(gb);

            if (pictureparameterset->ref_loc_offsets[i].phase_hor_luma >= (1 << 16) ||
                pictureparameterset->ref_loc_offsets[i].phase_ver_luma >= (1 << 16) ||
                pictureparameterset->ref_loc_offsets[i].phase_hor_chroma >= (1 << 16) ||
                pictureparameterset->ref_loc_offsets[i].phase_ver_chroma >= (1 << 16)) {
                av_log(avctx, AV_LOG_ERROR, "Resample phase values exceed 16-bit limit\n");
                return AVERROR_INVALIDDATA;
            }
        }
    }
