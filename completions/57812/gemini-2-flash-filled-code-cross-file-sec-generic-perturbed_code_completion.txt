if (pictureparameterset->num_ref_loc_offsets > 0) {
        if (pictureparameterset->num_ref_loc_offsets > 255) {
            av_log(avctx, AV_LOG_ERROR,
                   "num_ref_loc_offsets shall be in the range [0, 255].\n");
            return AVERROR_INVALIDDATA;
        }

        for (int i = 0; i < pictureparameterset->num_ref_loc_offsets; i++) {
            pictureparameterset->ref_layer_id[i] = get_bits(gb, 6);
            pictureparameterset->scaled_ref_layer_offsets_present_flag[i] = get_bits1(gb);
            if (pictureparameterset->scaled_ref_layer_offsets_present_flag[i]) {
                pictureparameterset->ref_region_offsets_present_flag[i] = get_bits1(gb);
                } else
                    pictureparameterset->ref_region_offsets_present_flag[i] = 0;

            if (pictureparameterset->ref_region_offsets_present_flag[i]) {
                pictureparameterset->resample_phase_set_flag[i] = get_bits1(gb);
            } else
                pictureparameterset->resample_phase_set_flag[i] = 0;

            if (pictureparameterset->scaled_ref_layer_offsets_present_flag[i]) {
                pictureparameterset->scaled_ref_layer_left_offset[i] = get_se_golomb_long(gb);
                pictureparameterset->scaled_ref_layer_top_offset[i] = get_se_golomb_long(gb);
                pictureparameterset->scaled_ref_layer_right_offset[i] = get_se_golomb_long(gb);
                pictureparameterset->scaled_ref_layer_bottom_offset[i] = get_se_golomb_long(gb);
            }

            if (pictureparameterset->ref_region_offsets_present_flag[i]) {
                pictureparameterset->ref_region_left_offset[i] = get_se_golomb_long(gb);
                pictureparameterset->ref_region_top_offset[i] = get_se_golomb_long(gb);
                pictureparameterset->ref_region_right_offset[i] = get_se_golomb_long(gb);
                pictureparameterset->ref_region_bottom_offset[i] = get_se_golomb_long(gb);
            }

            if (pictureparameterset->resample_phase_set_flag[i]) {
                pictureparameterset->resample_phase_luma[i] = get_ue_golomb_long(gb);
                pictureparameterset->resample_phase_chroma[i] = get_ue_golomb_long(gb);
            }
        }
