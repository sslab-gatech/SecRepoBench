for (int i = 0; i < pictureparameterset->num_ref_loc_offsets; i++) {
    pictureparameterset->ref_loc_offset[i].layer_id = get_bits(gb, 6);
    pictureparameterset->ref_loc_offset[i].scaled_ref_layer_offset_flag = get_bits1(gb);
    pictureparameterset->ref_loc_offset[i].ref_reg_offset_flag = get_bits1(gb);
    pictureparameterset->ref_loc_offset[i].resample_phase_flag = get_bits1(gb);
    
    if (pictureparameterset->ref_loc_offset[i].scaled_ref_layer_offset_flag) {
        pictureparameterset->ref_loc_offset[i].delta_poc = get_se_golomb_long(gb);
        pictureparameterset->ref_loc_offset[i].delta_poc_q = get_se_golomb_long(gb);
    }
    
    if (pictureparameterset->ref_loc_offset[i].ref_reg_offset_flag) {
        pictureparameterset->ref_loc_offset[i].delta_x = get_se_golomb_long(gb);
        pictureparameterset->ref_loc_offset[i].delta_y = get_se_golomb_long(gb);
    }
    
    if (pictureparameterset->ref_loc_offset[i].resample_phase_flag) {
        pictureparameterset->ref_loc_offset[i].luma_resample_phase = get_bits(gb, 2);
        pictureparameterset->ref_loc_offset[i].cb_resample_phase = get_bits(gb, 1);
        pictureparameterset->ref_loc_offset[i].cr_resample_phase = get_bits(gb, 1);
    }
}
