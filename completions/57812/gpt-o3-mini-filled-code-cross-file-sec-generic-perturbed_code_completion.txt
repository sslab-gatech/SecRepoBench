for (int i = 0; i < pictureparameterset->num_ref_loc_offsets; i++) {
    // Read the layer identifier (6 bits)
    pictureparameterset->ref_layer_id[i] = get_bits(gb, 6);
    // Read and process the flag for scaled reference layer offset
    pictureparameterset->scaled_ref_layer_offset_flag[i] = get_bits1(gb);
    if (pictureparameterset->scaled_ref_layer_offset_flag[i])
        pictureparameterset->scaled_ref_layer_offset[i] = get_se_golomb_long(gb);
    // Read and process the flag for reference region offset
    pictureparameterset->ref_region_offset_flag[i] = get_bits1(gb);
    if (pictureparameterset->ref_region_offset_flag[i])
        pictureparameterset->ref_region_offset[i] = get_se_golomb_long(gb);
    // Read and process the flag for resample phase set
    pictureparameterset->resample_phase_set_flag[i] = get_bits1(gb);
    if (pictureparameterset->resample_phase_set_flag[i]) {
        pictureparameterset->luma_resample_phase[i]   = get_bits(gb, 6);
        pictureparameterset->chroma_resample_phase[i] = get_bits(gb, 6);
    }
}
