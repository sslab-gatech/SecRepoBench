for (int i = 0; i < pictureparameterset->num_ref_loc_offsets; i++) {
    pictureparameterset->ref_loc_offset[i].ref_loc_layer_id = get_bits(gb, 6);

    pictureparameterset->ref_loc_offset[i].scaled_ref_layer_offset_present_flag = get_bits1(gb);
    if (pictureparameterset->ref_loc_offset[i].scaled_ref_layer_offset_present_flag) {
        pictureparameterset->ref_loc_offset[i].scaled_ref_layer_left_offset   = get_se_golomb_long(gb);
        pictureparameterset->ref_loc_offset[i].scaled_ref_layer_top_offset    = get_se_golomb_long(gb);
        pictureparameterset->ref_loc_offset[i].scaled_ref_layer_right_offset  = get_se_golomb_long(gb);
        pictureparameterset->ref_loc_offset[i].scaled_ref_layer_bottom_offset = get_se_golomb_long(gb);
    }

    pictureparameterset->ref_loc_offset[i].ref_region_offset_present_flag = get_bits1(gb);
    if (pictureparameterset->ref_loc_offset[i].ref_region_offset_present_flag) {
        pictureparameterset->ref_loc_offset[i].ref_region_left_offset   = get_se_golomb_long(gb);
        pictureparameterset->ref_loc_offset[i].ref_region_top_offset    = get_se_golomb_long(gb);
        pictureparameterset->ref_loc_offset[i].ref_region_right_offset  = get_se_golomb_long(gb);
        pictureparameterset->ref_loc_offset[i].ref_region_bottom_offset = get_se_golomb_long(gb);
    }

    pictureparameterset->ref_loc_offset[i].resample_phase_set_present_flag = get_bits1(gb);
    if (pictureparameterset->ref_loc_offset[i].resample_phase_set_present_flag) {
        pictureparameterset->ref_loc_offset[i].phase_offset_luma_h = get_ue_golomb_long(gb);
        pictureparameterset->ref_loc_offset[i].phase_offset_luma_v = get_ue_golomb_long(gb);
        if (sps->chroma_format_idc) {
            pictureparameterset->ref_loc_offset[i].phase_offset_chroma_h = get_ue_golomb_long(gb);
            pictureparameterset->ref_loc_offset[i].phase_offset_chroma_v = get_ue_golomb_long(gb);
        }
    }
}
