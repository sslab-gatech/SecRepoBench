        if (pictureparameterset->ref_loc_offsets[i].ref_region_offset_present_flag) {
            pictureparameterset->ref_loc_offsets[i].ref_region_left_offset  = get_se_golomb_long(gb);
            pictureparameterset->ref_loc_offsets[i].ref_region_top_offset   = get_se_golomb_long(gb);

            pictureparameterset->ref_loc_offsets[i].ref_region_bottom_offset= get_se_golomb_long(gb);
        }

        pictureparameterset->ref_loc_offsets[i].resample_phase_set_present_flag = get_bits1(gb);
        if (pictureparameterset->ref_loc_offsets[i].resample_phase_set_present_flag) {
            pictureparameterset->ref_loc_offsets[i].phase_hor_luma = get_bits(gb, 2);
            pictureparameterset->ref_loc_offsets[i].phase_ver_luma = get_bits(gb, 2);
            pictureparameterset->ref_loc_offsets[i].phase_hor_chroma = get_bits(gb, 2);
