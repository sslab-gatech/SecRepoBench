for (int i = 0; i < pictureparameterset->num_ref_loc_offsets; i++) {
    pictureparameterset->ref_loc_layer_id[i] = get_bits(gb, 6);
    pictureparameterset->scaled_ref_layer_offset_flag[i] = get_bits1(gb);
    if (pictureparameterset->scaled_ref_layer_offset_flag[i]) {
        pictureparameterset->scaled_ref_layer_offset[i][0] = get_ue_golomb_long(gb);
        pictureparameterset->scaled_ref_layer_offset[i][1] = get_ue_golomb_long(gb);
    }
    pictureparameterset->ref_region_offset_flag[i] = get_bits1(gb);
    if (pictureparameterset->ref_region_offset_flag[i]) {
        pictureparameterset->ref_region_offset[i][0] = get_ue_golomb_long(gb);
        pictureparameterset->ref_region_offset[i][1] = get_ue_golomb_long(gb);
        pictureparameterset->ref_region_offset[i][2] = get_ue_golomb_long(gb);
        pictureparameterset->ref_region_offset[i][3] = get_ue_golomb_long(gb);
    }
    pictureparameterset->resample_phase_set_flag[i] = get_bits1(gb);
    if (pictureparameterset->resample_phase_set_flag[i]) {
        pictureparameterset->resample_phase_luma[i] = get_ue_golomb_long(gb);
        pictureparameterset->resample_phase_chroma[i][0] = get_ue_golomb_long(gb);
        pictureparameterset->resample_phase_chroma[i][1] = get_ue_golomb_long(gb);
    }
}
