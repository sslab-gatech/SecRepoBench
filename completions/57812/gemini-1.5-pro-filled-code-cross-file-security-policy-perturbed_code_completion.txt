if (pictureparameterset->num_ref_loc_offsets > HEVC_MAX_REF_LOC_OFFSETS) {
        av_log(avctx, AV_LOG_ERROR, "Too many reference location offsets: %d\n",
               pictureparameterset->num_ref_loc_offsets);
        return AVERROR_INVALIDDATA;
    }

    for (int i = 0; i < pictureparameterset->num_ref_loc_offsets; i++) {
        pictureparameterset->ref_loc_offset_layer_id[i] = get_bits(gb, 6);
        if (pictureparameterset->ref_loc_offset_layer_id[i] >= sps->max_sub_layers) {
            av_log(avctx, AV_LOG_ERROR, "Invalid ref_loc_offset_layer_id: %d\n",
                   pictureparameterset->ref_loc_offset_layer_id[i]);
            return AVERROR_INVALIDDATA;
        }

        if (get_bits1(gb)) { // scaled_ref_layer_offset_present_flag[ i ]
            pictureparameterset->scaled_ref_layer_offset[i] = get_se_golomb_long(gb);
        }

        if (get_bits1(gb)) { // ref_region_offset_present_flag[ i ]
            pictureparameterset->ref_region_offset[i] = get_se_golomb_long(gb);
        }

        if (get_bits1(gb)) { // resample_phase_set_present_flag[ i ]
            pictureparameterset->resample_phase_set_present_flag = 1;
            pictureparameterset->phase_hor_luma[i] = get_bits(gb, 2);
            if (sps->chroma_format_idc != 0) {
                pictureparameterset->phase_ver_chroma[i] = get_bits(gb, 2);
                pictureparameterset->phase_hor_chroma[i] = get_bits(gb, 2);
            } else {
                pictureparameterset->phase_ver_chroma[i] = 0;
                pictureparameterset->phase_hor_chroma[i] = 0;
            }
        }
    }
