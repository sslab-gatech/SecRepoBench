for (int i = 0; i < pictureparameterset->num_ref_loc_offsets; i++) {
    pictureparameterset->ref_loc_layer_id[i] = get_bits(gb, 6);

    pictureparameterset->scaled_ref_layer_offset_flag[i] = get_bits1(gb);
    if (pictureparameterset->scaled_ref_layer_offset_flag[i]) {
        pictureparameterset->scaled_ref_layer_hor_offset[i] = get_se_golomb_long(gb);
        pictureparameterset->scaled_ref_layer_ver_offset[i] = get_se_golomb_long(gb);
    }

    pictureparameterset->ref_region_offset_flag[i] = get_bits1(gb);
    if (pictureparameterset->ref_region_offset_flag[i]) {
        pictureparameterset->ref_region_left_offset[i] = get_se_golomb_long(gb);
        pictureparameterset->ref_region_right_offset[i] = get_se_golomb_long(gb);
        pictureparameterset->ref_region_top_offset[i] = get_se_golomb_long(gb);
        pictureparameterset->ref_region_bottom_offset[i] = get_se_golomb_long(gb);
    }

    pictureparameterset->resample_phase_flag[i] = get_bits1(gb);
    if (pictureparameterset->resample_phase_flag[i]) {
        pictureparameterset->resample_phases_minus1[i] = get_ue_golomb_long(gb);
        for (int j = 0; j <= pictureparameterset->resample_phases_minus1[i]; j++) {
            pictureparameterset->luma_phase[i][j] = get_bits(gb, 3);
            pictureparameterset->chroma_phase[i][j] = get_bits(gb, 2);
        }
    }
}
