// Parse reference location offsets for multiple layers in the PPS
// Iterate over all reference location offsets specified by num_ref_loc_offsets
// For each offset, read and store the layer ID from the bitstream
// Check and process flags for scaled reference layer offsets, reference region offsets, and resample phase set
// Retrieve and store the respective offset values using Golomb-coded data as needed
// Handle resample phases for luma and chroma components if specified
for (int i = 0; i < pictureparameterset->num_ref_loc_offsets; i++) {
    int ref_loc_offset_layer_id = get_bits(gb, 6);
    if (ref_loc_offset_layer_id >= FF_ARRAY_ELEMS(pictureparameterset->ref_loc_offset_layer_id)) {
        av_log(avctx, AV_LOG_ERROR, "ref_loc_offset_layer_id shall be in the range [0, %d].\n",
               FF_ARRAY_ELEMS(pictureparameterset->ref_loc_offset_layer_id) - 1);
        return AVERROR_INVALIDDATA;
    }
    pictureparameterset->ref_loc_offset_layer_id[ref_loc_offset_layer_id] = get_ue_golomb_long(gb);

    pictureparameterset->scaled_ref_layer_offset_enabled_flag = get_bits1(gb);
    if (pictureparameterset->scaled_ref_layer_offset_enabled_flag) {
        pictureparameterset->scaled_ref_layer_offset = get_ue_golomb_long(gb);
    }

    pictureparameterset->ref_region_offset_enabled_flag = get_bits1(gb);
    if (pictureparameterset->ref_region_offset_enabled_flag) {
        pictureparameterset->ref_region_offset = get_ue_golomb_long(gb);
    }

    pictureparameterset->resample_phase_set_enabled_flag = get_bits1(gb);
    if (pictureparameterset->resample_phase_set_enabled_flag) {
        pictureparameterset->resample_phase_set = get_ue_golomb_long(gb);
        pictureparameterset->resample_phase = get_ue_golomb_long(gb);
    }
}
