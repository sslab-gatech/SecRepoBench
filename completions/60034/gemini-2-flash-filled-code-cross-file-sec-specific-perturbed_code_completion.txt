hentry* homonym = entry_result;
        while (homonym) {
          if (compoundforbidflag && TESTAFF(homonym->astr, compoundforbidflag, homonym->alen) &&
              !hu_mov_rule) {
            if (simplifiedcpd) {
              homonym = homonym->next_homonym;
              continue;
            } else {
              entry_result = NULL;
              break;
            }
          }

          if (compoundflag && !TESTAFF(homonym->astr, compoundflag, homonym->alen) &&
              compoundbegin && !TESTAFF(homonym->astr, compoundbegin, homonym->alen) &&
              compoundmiddle && !TESTAFF(homonym->astr, compoundmiddle, homonym->alen)) {
            homonym = homonym->next_homonym;
            continue;
          }

          if (!checkcpdtable.empty() && scpd != 0 && checkcpdtable[scpd - 1].cond != FLAG_NULL &&
              !TESTAFF(homonym->astr, checkcpdtable[scpd - 1].cond, homonym->alen)) {
            homonym = homonym->next_homonym;
            continue;
          }

          entry_result = homonym;
          break;
        }
