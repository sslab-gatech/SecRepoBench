if (entry_result && entry_result->astr &&
    TESTAFF(entry_result->astr, compoundforbidflag, entry_result->alen)) {
  if (hu_mov_rule) {
    // If the Hungarian movement rule applies, skip this forbidden word
    continue;
  } else {
    // Otherwise, break out of the loop as we cannot proceed with this invalid entry
    break;
  }
}

// Iterate through homonyms to find one that satisfies compound conditions
while ((entry_result) && 
       (!TESTAFF(entry_result->astr, compoundflag, entry_result->alen) ||
        TESTAFF(entry_result->astr, compoundforbidflag, entry_result->alen))) {
  // Skip invalid entries based on compound flags or special Hungarian rules
  entry_result = entry_result->next_homonym;
  if (!entry_result) {
    break;  // No more homonyms to check
  }

  // Special Hungarian handling: Check additional flags
  if (langnum == LANG_hu && hu_mov_rule &&
      (TESTAFF(entry_result->astr, 'F', entry_result->alen) || 
       TESTAFF(entry_result->astr, 'G', entry_result->alen) ||
       TESTAFF(entry_result->astr, 'H', entry_result->alen))) {
    // Valid under Hungarian rules, so use this entry
    break;
  }
}

// Additional checks for simplified compound processing (`simplifiedcpd`)
if (simplifiedcpd && scpd > 0 && entry_result) {
  if (!TESTAFF(entry_result->astr, checkcpdtable[scpd - 1].cond, entry_result->alen)) {
    // Entry does not satisfy the current CHECKCOMPOUNDPATTERN condition
    entry_result = NULL;
  }
}
