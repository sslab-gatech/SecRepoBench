while (entry_result && compoundforbidflag &&
       TESTAFF(entry_result->astr, compoundforbidflag, entry_result->alen) &&
       !hu_mov_rule) {
  // Move to the next homonym if this one is forbidden in compound words
  // and not allowed by the Hungarian movement rule.
  entry_result = entry_result->next_homonym;
}

// If we've run out of homonyms, break from the loop
if (!entry_result) {
  break;
}
