while (entry_result) {
          if (compoundforbidflag &&
              TESTAFF(entry_result->astr, compoundforbidflag,
                      entry_result->alen) &&
              !hu_mov_rule) {
            if (simplifiedcpd) {
              entry_result = entry_result->next_homonym;
              continue;
            } else {
              entry_result = NULL;
              break;
            }
          }

          if (words && words[wnum]) {
            if (defcpd_check(&words, wnum, entry_result, NULL, 1))
              break;
            else {
              entry_result = entry_result->next_homonym;
              continue;
            }
          }

          if (!compoundflag ||
              !TESTAFF(entry_result->astr, compoundflag, entry_result->alen)) {
            if ((oldwordnum == 0 && compoundbegin &&
                 !TESTAFF(entry_result->astr, compoundbegin,
                          entry_result->alen)) ||
                (oldwordnum > 0 && compoundmiddle &&
                 !TESTAFF(entry_result->astr, compoundmiddle,
                          entry_result->alen))) {
              entry_result = entry_result->next_homonym;
              continue;
            }
          }

          if (!checkcpdtable.empty() && scpd != 0 &&
              checkcpdtable[scpd - 1].cond != FLAG_NULL &&
              !TESTAFF(entry_result->astr, checkcpdtable[scpd - 1].cond,
                       entry_result->alen)) {
            entry_result = entry_result->next_homonym;
            continue;
          }
          break;
        }
