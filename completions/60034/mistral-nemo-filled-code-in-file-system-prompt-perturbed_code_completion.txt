// Function to check if a word is a compound word
bool AffixMgr::isCompoundWord(const std::string& word) {
  struct hentry* rv = NULL;
  rv = compound_check(word, 0, word.size(), 0, NULL, 0, 0, 0);
  return rv != NULL;
}

// Function to check if a word is a compound word with a specific flag
bool AffixMgr::isCompoundWord(const std::string& word, FLAG flag) {
  struct hentry* rv = NULL;
  rv = compound_check(word, 0, word.size(), 0, NULL, 0, 0, flag);
  return rv != NULL;
}

// Function to check if a word is a compound word with a specific flag and a specific compound flag
bool AffixMgr::isCompoundWord(const std::string& word, FLAG flag, FLAG compoundFlag) {
  struct hentry* rv = NULL;
  rv = compound_check(word, 0, word.size(), 0, NULL, 0, 0, flag, compoundFlag);
  return rv != NULL;
}
