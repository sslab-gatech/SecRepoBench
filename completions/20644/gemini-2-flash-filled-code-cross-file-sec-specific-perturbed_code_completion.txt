if(recheck_type)
    goto ether_type_check;

  if(vlan_packet)
    workflow->stats.vlan_count++;

iph_check:
  if(ip_offset >= header->caplen)
    return(nproto);

  iph = (struct ndpi_iphdr *) &packet[ip_offset];

  if((cprototype == ETH_P_IP) || (cprototype == ETH_P_IPV6) || (datalink_type == DLT_RAW)) {
    frag_off = ntohs(iph->frag_off);
    proto = iph->protocol;
  } else {
    static u_int8_t eth_warning_used = 0;

    if(eth_warning_used == 0) {
      if(!workflow->prefs.quiet_mode)
	NDPI_LOG(0, workflow->ndpi_struct, NDPI_LOG_DEBUG, "\n\nWARNING: only Ethernet with IP protocol is supported in this demo (nDPI supports all protocols), all other packets will be discarded\n\n");
      eth_warning_used = 1;
    }
    workflow->stats.total_discarded_bytes +=  header->len;
    return(nproto);
  }

  if(header->caplen < header->len) {
    static u_int8_t caplen_warning_used = 0;

    if(caplen_warning_used == 0) {
      if(!workflow->prefs.quiet_mode)

      caplen_warning_used = 1;
