iph_check:
  /* Check and set IP header offset */
  if((ip_offset + sizeof(struct ndpi_iphdr)) > header->caplen)
    return(nproto); /* Too short */

  iph = (struct ndpi_iphdr *)&packet[ip_offset];

  /* Check if it's an IPv4 packet */
  if(cprototype == ETH_P_IP && ip_offset < header->caplen) {
    frag_off = ntohs(iph->frag_off);
    proto = iph->protocol;
  }

  /* Check if captured packet size is less than actual packet size */
  if(header->caplen < header->len) {
    static u_int8_t cap_warning_used = 0;
    
    if(cap_warning_used == 0) {
      if(!workflow->prefs.quiet_mode)
        NDPI_LOG(0, workflow->ndpi_struct, NDPI_LOG_DEBUG, 
                "\n\nWARNING: packet capture size is smaller than packet size, DETECTION MIGHT NOT WORK CORRECTLY\n\n");
      cap_warning_used = 1;
