case 'H': {
            // Hexadecimal string
            if (!tm->blk) {
                if (!(tm->blk = cram_new_block(EXTERNAL, key)))
                    goto err;
                codec->u.e_byte_array_len.val_codec->out = tm->blk;
            }

            aux += 3;
            char *start = aux;
            while (aux < aux_limit && *aux != '\0') {
                aux++;
            }
            if (aux >= aux_limit)
                goto err;

            BLOCK_APPEND(tm->blk, start, aux - start + 1);
            aux++;
            break;
        }

        case 'B': {
            // Byte array
            if (aux_limit - aux < 8)
                goto err;

            if (!tm->blk) {
                if (!(tm->blk = cram_new_block(EXTERNAL, key)))
                    goto err;
                codec->u.e_byte_array_len.val_codec->out = tm->blk;
            }

            aux += 3;
            int32_t count = le_int4(aux + 1);
            int type_size;

            switch (aux[0]) {
            case 'c': case 'C': type_size = 1; break;
            case 's': case 'S': type_size = 2; break;
            case 'i': case 'I': case 'f': type_size = 4; break;
            default:
                hts_log_error("Unknown sub-type '%c' for aux type 'B'", aux[0]);
                goto err;
            }

            int data_size = count * type_size;
            if (aux_limit - aux < 5 + data_size)
                goto err;

            BLOCK_APPEND(tm->blk, aux, 5 + data_size);
            aux += 5 + data_size;
            break;
        }
