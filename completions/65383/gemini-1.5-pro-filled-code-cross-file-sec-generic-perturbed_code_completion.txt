case 'H':
    if (aux_limit - aux < 4)
        goto err;

    if (!tm->blk) {
        if (!(tm->blk = cram_new_block(EXTERNAL, key)))
            goto err;
        codec->out = tm->blk;
    }

    aux+=3;
    if (aux_limit - aux < *aux + 1) // Check for sufficient data
        goto err;

    BLOCK_APPEND(tm->blk, aux, *aux + 1);
    aux += *aux + 1;
    break;

case 'B': {
    if (aux_limit - aux < 5)
        goto err;

    uint32_t sub_type = aux[3];
    uint32_t count = bam_auxB_len((uint8_t *)aux);

    if (aux_limit - aux < 5 + count) // Check for sufficient data
        goto err;

    if (!tm->blk) {
        if (!(tm->blk = cram_new_block(EXTERNAL, key)))
            goto err;
        codec->u.e_byte_array_len.val_codec->out = tm->blk;
    }

    aux+=4;
    BLOCK_APPEND(tm->blk, aux, count);
    aux += count;
    break;
}
