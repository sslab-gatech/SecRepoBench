if (aux[2] == 'H') {
                // Hex string
                if (!tm->blk) {
                    if (!(tm->blk = cram_new_block(EXTERNAL, key)))
                        goto err;
                    codec->out = tm->blk;
                }

                aux += 3;
                while (aux < aux_limit && *aux) {
                    BLOCK_APPEND_CHAR(tm->blk, *aux++);
                }
                aux++;
            } else if (aux[2] == 'B') {
                // Byte array
                int sub_type = aux[3];
                int count = le_to_u32(aux + 4);
                int len = 0;

                switch (sub_type) {
                case 'c': case 'C': len = 1; break;
                case 's': case 'S': len = 2; break;
                case 'i': case 'I': case 'f': len = 4; break;
                default:
                    hts_log_error("Unknown sub-type '%c' for aux type 'B'", sub_type);
                    goto err;
                }

                if (!tm->blk) {
                    if (!(tm->blk = cram_new_block(EXTERNAL, key)))
                        goto err;
                    codec->u.e_byte_array_len.val_codec->out = tm->blk;
                }

                aux += 8; // Skip 'B', sub_type, and count
                if (aux_limit - aux < count * len)
                    goto err;

                BLOCK_APPEND(tm->blk, aux, count * len);
                aux += count * len;
            } else {
                hts_log_error("Unknown aux type '%c'", aux[2]);
                goto err;
            }
