case 'Z':
            if (aux_limit - aux < 3 + 1)
                goto err;

            if (!tm->blk) {
                if (!(tm->blk = cram_new_block(EXTERNAL, key)))
                    goto err;
                codec->u.e_byte_array_len.val_codec->out = tm->blk;
            }

            aux += 3;
            size_t len = strlen(aux) + 1;
            //codec->encode(s, codec, aux, len);
            BLOCK_APPEND(tm->blk, aux, len);
            aux += len;
            break;
        case 'H':
            if (aux_limit - aux < 3 + 1)
                goto err;

            if (!tm->blk) {
                if (!(tm->blk = cram_new_block(EXTERNAL, key)))
                    goto err;
                codec->u.e_byte_array_len.val_codec->out = tm->blk;
            }

            aux += 3;
            len = strlen(aux) + 1;
            //codec->encode(s, codec, aux, len);
            BLOCK_APPEND(tm->blk, aux, len);
            aux += len;
            break;
        case 'B': {
            int type, count;
            if (aux_limit - aux < 3 + 2)
                goto err;
            type = aux[3];
            count = bam_aux2i(aux + 4);
            len = 4 + count;
            if (aux_limit - aux < len)
                goto err;

            if (!tm->blk) {
                if (!(tm->blk = cram_new_block(EXTERNAL, key)))
                    goto err;
                codec->u.e_byte_array_len.val_codec->out = tm->blk;
            }

            aux += 3;
            //codec->encode(s, codec, aux, len);
            BLOCK_APPEND(tm->blk, aux, len);
