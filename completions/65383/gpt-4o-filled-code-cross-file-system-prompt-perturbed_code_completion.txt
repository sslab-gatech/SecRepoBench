if (aux[2] == 'H') {
                if (!tm->blk) {
                    if (!(tm->blk = cram_new_block(EXTERNAL, key)))
                        goto err;
                    codec->u.e_byte_array_len.val_codec->out = tm->blk;
                }

                aux += 3;
                while (aux < aux_limit && *aux != '\0') {
                    BLOCK_APPEND_CHAR(tm->blk, *aux);
                    aux++;
                }
                aux++;
            } else if (aux[2] == 'B') {
                int32_t sub_type, count, len;
                if (aux_limit - aux < 8)
                    goto err;

                if (!tm->blk) {
                    if (!(tm->blk = cram_new_block(EXTERNAL, key)))
                        goto err;
                    codec->u.e_byte_array_len.val_codec->out = tm->blk;
                }

                aux += 3;
                sub_type = *aux++;
                count = le_to_i32(aux);
                aux += 4;

                switch (sub_type) {
                case 'c': case 'C': len = 1; break;
                case 's': case 'S': len = 2; break;
                case 'i': case 'I': case 'f': len = 4; break;
                default:
                    hts_log_error("Unknown sub-type '%c' for aux type 'B'", sub_type);
                    goto err;
                }

                if (aux_limit - aux < count * len)
                    goto err;

                BLOCK_APPEND_CHAR(tm->blk, sub_type);
                BLOCK_APPEND(tm->blk, (char *)&count, 4);
                BLOCK_APPEND(tm->blk, aux, count * len);
                aux += count * len;
            }
