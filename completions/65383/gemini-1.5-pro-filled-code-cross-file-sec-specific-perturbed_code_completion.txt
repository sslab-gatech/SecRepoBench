case 'H':
    if (aux_limit - aux < 4)
        goto err;

    if (!tm->blk) {
        if (!(tm->blk = cram_new_block(EXTERNAL, key)))
            goto err;
        codec->u.e_byte_array_len.val_codec->out = tm->blk;
    }

    aux+=3;
    size_t len = strlen(aux);
    if (aux + len >= aux_limit || aux[len] != 0)
        goto err;
    if (len & 1)
        goto err;
    if (cram_encode_aux_hex(tm->blk, aux, len) < 0)
        goto err;
    aux += len+1;
    break;

case 'B': {
    uint8_t sub_type;
    uint32_t count;

    if (aux_limit - aux < 7)
        goto err;

    if (!tm->blk) {
        if (!(tm->blk = cram_new_block(EXTERNAL, key)))
            goto err;
        codec->u.e_byte_array_len.val_codec->out = tm->blk;
    }

    aux+=3;
    sub_type = *aux++;
    memcpy(&count, aux, 4);
    count = le_to_host_32(count);
    aux+=4;

    size_t len = 0;
    switch (sub_type) {
    case 'c': case 'C': len = count; break;
    case 's': case 'S': len = count * 2; break;
    case 'i': case 'I': case 'f': len = count * 4; break;
    default:
        hts_log_error("Unhandled sub-type '%c' for aux type B", sub_type);
        goto err;
    }

    if (aux + len > aux_limit)
        goto err;

    BLOCK_APPEND(tm->blk, aux, len);
    aux += len;
    break;
}
