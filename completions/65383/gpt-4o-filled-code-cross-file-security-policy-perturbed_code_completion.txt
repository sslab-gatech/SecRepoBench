if (aux[2] == 'H') {
                if (aux_limit - aux < 3+1)
                    goto err;

                if (!tm->blk) {
                    if (!(tm->blk = cram_new_block(EXTERNAL, key)))
                        goto err;
                    codec->u.e_byte_array_len.val_codec->out = tm->blk;
                }

                aux += 3;
                while (aux < aux_limit && *aux != '\0') {
                    BLOCK_APPEND_CHAR(tm->blk, *aux);
                    aux++;
                }
                if (aux < aux_limit) aux++; // skip null terminator
            } else if (aux[2] == 'B') {
                if (aux_limit - aux < 3+5)
                    goto err;

                if (!tm->blk) {
                    if (!(tm->blk = cram_new_block(EXTERNAL, key)))
                        goto err;
                    codec->u.e_byte_array_len.val_codec->out = tm->blk;
                }

                int sub_type = aux[3];
                int count = le_to_u32(aux + 4);
                int data_size;

                switch (sub_type) {
                case 'c': case 'C': data_size = 1; break;
                case 's': case 'S': data_size = 2; break;
                case 'i': case 'I': case 'f': data_size = 4; break;
                default:
                    hts_log_error("Unknown sub-type '%c' for aux type 'B'", sub_type);
                    goto err;
                }

                if (aux_limit - aux < 3 + 5 + count * data_size)
                    goto err;

                aux += 3;
                BLOCK_APPEND(tm->blk, aux, 5 + count * data_size);
                aux += 5 + count * data_size;
            }
