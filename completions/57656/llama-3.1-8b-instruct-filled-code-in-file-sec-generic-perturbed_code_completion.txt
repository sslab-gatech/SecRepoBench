// The code checks for the existence of a file at the given path for 
// non-Windows platforms.
#ifdef __unix__
#    include <sys/stat.h>
#    include <unistd.h>
#endif

#ifdef __unix__ || __APPLE__
    struct stat filestat;
    if (stat(filePath, &filestat)!= 0) {
        return false;
    }
    return (filestat.st_mode & S_IFMT) == S_IFREG;
#endif
