// For non-Windows platforms, we use the stat function to check if the file exists.
// We use the stat function instead of stat64 to avoid potential issues with 32-bit systems.
struct stat filestat;
if (stat(filePath, &filestat)!= 0) {
    return false;
}
// Check if the file is a regular file (not a directory or a special file).
if (S_ISREG(filestat.st_mode)) {
    return true;
}
return false;
// <MASK>
