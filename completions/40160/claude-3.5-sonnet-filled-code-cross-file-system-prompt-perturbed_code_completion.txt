for(i=0; i<(cipher_length/2); i++) {
  u_int16_t cipher_id = ntohs(*((u_int16_t*)&packet->payload[cipher_offset+i*2]));

  if((cipher_id != 0x0A0A) && (cipher_id != 0x1A1A) && (cipher_id != 0x2A2A) && (cipher_id != 0x3A3A)) {
    if(ja3.client.num_cipher < MAX_NUM_JA3)
      ja3.client.cipher[ja3.client.num_cipher++] = cipher_id;
    else {
      invalid_ja3 = 1;
#ifdef DEBUG_TLS
      printf("Client TLS Invalid num cipher %u\n", ja3.client.num_cipher);
#endif
      break;
    }

    if(cipher_id == 0xC02B || cipher_id == 0xC02F || cipher_id == 0xCCA9 || cipher_id == 0x1301 ||
       cipher_id == 0x1302 || cipher_id == 0x1303) {
      safari_ciphers++;
    }

    if(cipher_id == 0x1301 || cipher_id == 0x1302 || cipher_id == 0x1303 ||
       cipher_id == 0xC02B || cipher_id == 0xC02F || cipher_id == 0xCCA9 ||
       cipher_id == 0xCCA8 || cipher_id == 0x009C || cipher_id == 0x009D ||
       cipher_id == 0x002F || cipher_id == 0x0035 || cipher_id == 0x000A ||
       cipher_id == 0x003C) {
      chrome_ciphers++;
    }

    if(cipher_id == 0x009C || cipher_id == 0x009D || cipher_id == 0x002F ||
       cipher_id == 0x0035 || cipher_id == 0x000A || cipher_id == 0x003C) {
      this_is_not_safari = 1;
    }

    if(cipher_id == 0x1301 || cipher_id == 0x1302 || cipher_id == 0x1303) {
      looks_like_safari_on_big_sur = 1;
    }
  }
}
