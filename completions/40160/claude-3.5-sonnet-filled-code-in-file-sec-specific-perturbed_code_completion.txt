for(i=0; i<cipher_length && cipher_offset+i+1 < packet->payload_packet_len; i+=2) {
  u_int16_t cipher = ntohs(*((u_int16_t*)&packet->payload[cipher_offset+i]));

  if((cipher != 0) && (packet->payload[cipher_offset+i] != packet->payload[cipher_offset+i+1])) {
    /* Skip GREASE */
    if(ja3.client.num_cipher < MAX_NUM_JA3)
      ja3.client.cipher[ja3.client.num_cipher++] = cipher;
    else {
      invalid_ja3 = 1;
#ifdef DEBUG_TLS
      printf("Client TLS Invalid cipher %u\n", ja3.client.num_cipher);
#endif
      break;
    }
  }

  if(cipher == 0x1303) /* TLS_AES_128_CCM_8_SHA256 */
    flow->protos.tls_quic_stun.tls_quic.browser_heuristics.is_firefox_tls = 1;

  if((cipher == 0xc02b) || (cipher == 0xc02f) || (cipher == 0xcca9) || (cipher == 0x1301) || (cipher == 0x1302) || (cipher == 0x1303))
    safari_ciphers++, chrome_ciphers++;

  if((cipher == 0xc02c) || (cipher == 0xc030))
    chrome_ciphers++;

  if((cipher == 0x009c) || (cipher == 0x002f))
    this_is_not_safari = 1;

  if((cipher == 0x1301) || (cipher == 0x1302) || (cipher == 0x1303))
    looks_like_safari_on_big_sur = 1;
}
