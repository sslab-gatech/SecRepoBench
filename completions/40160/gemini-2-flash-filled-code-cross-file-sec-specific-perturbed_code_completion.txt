for(i=0; (i+1)<cipher_length; i+=2) {
	  u_int16_t cipher = ntohs(*((u_int16_t*)&packet->payload[cipher_offset+i]));

	  if((cipher == 0x0A0A) || (packet->payload[cipher_offset+i] == packet->payload[cipher_offset+i+1])) {
	    /* Skip GREASE */
#ifdef DEBUG_CIPHERS
	    printf("[CIPHERS] [TLS Cipher Suite] Skipping GREASE 0x%04X\n", cipher);
#endif
	    continue;
	  }

	  if(ja3.client.num_cipher < MAX_NUM_JA3) {
	    ja3.client.cipher[ja3.client.num_cipher++] = cipher;
#ifdef DEBUG_CIPHERS
	    printf("[CIPHERS] [TLS Cipher Suite] 0x%04X\n", cipher);
#endif
	  } else {
	    invalid_ja3 = 1;
#ifdef DEBUG_TLS
	    printf("Client TLS Invalid ciphers %u\n", ja3.client.num_cipher);
#endif
	  }

	  switch(cipher) {
	  case TLS_AES_128_GCM_SHA256:
	  case TLS_AES_256_GCM_SHA384:
	  case TLS_CHACHA20_POLY1305_SHA256:
	    chrome_ciphers++;
	    break;

	  case TLS_ECDHE_ECDSA_WITH_AES_256_GCM_SHA384:
	  case TLS_ECDHE_RSA_WITH_AES_256_GCM_SHA384:
	  case TLS_ECDHE_ECDSA_WITH_CHACHA20_POLY1305_SHA256:
	  case TLS_ECDHE_RSA_WITH_CHACHA20_POLY1305_SHA256:
	  case TLS_ECDHE_ECDSA_WITH_AES_128_GCM_SHA256:
	  case TLS_ECDHE_RSA_WITH_AES_128_GCM_SHA256:
	  case TLS_ECDHE_ECDSA_WITH_AES_256_CBC_SHA384:
	  case TLS_ECDHE_RSA_WITH_AES_256_CBC_SHA384:
	  case TLS_ECDHE_ECDSA_WITH_AES_128_CBC_SHA256:
	  case TLS_ECDHE_RSA_WITH_AES_128_CBC_SHA256:
	  case TLS_ECDHE_RSA_WITH_AES_256_CBC_SHA:
	  case TLS_ECDHE_RSA_WITH_AES_128_CBC_SHA:
	    safari_ciphers++;
	    break;

	  case TLS_DHE_RSA_WITH_AES_256_GCM_SHA384:
	  case TLS_DHE_RSA_WITH_AES_128_GCM_SHA256:
	    this_is_not_safari = 1;
	    break;

	  case TLS_ECDHE_RSA_WITH_AES_128_CBC_SHA + 0x0001:
	    /*
	      Safari on Big Sur
	      https://github.com/salesforce/ja3/issues/269
	    */
	    looks_like_safari_on_big_sur = 1;
	    break;

	  default:
	    break;
	  }
	}
