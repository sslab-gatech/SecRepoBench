for(j=0; j<cipher_length; j+=2) {
	  u_int16_t cipher_suite = ntohs(*((u_int16_t*)&packet->payload[cipher_offset+j]));

	  if((cipher_suite & 0x0F0F) != 0x0F0F) { /* Skip GREASE */
	    if(ja3.client.num_cipher < MAX_NUM_JA3)
	      ja3.client.cipher[ja3.client.num_cipher++] = cipher_suite;
	    else {
	      invalid_ja3 = 1;
#ifdef DEBUG_TLS
	      printf("Client TLS Invalid num ciphers %u\n", ja3.client.num_cipher);
#endif
	    }

	    /* Heuristic checks */
	    if(cipher_suite == 0xc02b) /* TLS_ECDHE_ECDSA_WITH_AES_128_GCM_SHA256 */
	      safari_ciphers++;
	    else if(cipher_suite == 0xc02f) /* TLS_ECDHE_RSA_WITH_AES_128_GCM_SHA256 */
	      chrome_ciphers++;
	    else if(cipher_suite == 0xc030) /* TLS_ECDHE_RSA_WITH_AES_256_GCM_SHA384 */
	      chrome_ciphers++;
	    else if(cipher_suite == 0xc02c) /* TLS_ECDHE_ECDSA_WITH_AES_256_GCM_SHA384 */
	      safari_ciphers++;
	    else if(cipher_suite == 0x9e00) /* TLS_AES_128_GCM_SHA256 */
	      chrome_ciphers++;
	    else if(cipher_suite == 0x9c00) /* TLS_AES_256_GCM_SHA384 */
	      chrome_ciphers++;
	    else if(cipher_suite == 0x9d00) /* TLS_CHACHA20_POLY1305_SHA256 */
	      looks_like_safari_on_big_sur = 1;
	    else if(cipher_suite == 0x1301) /* TLS_RSA_WITH_AES_128_CBC_SHA */
	      this_is_not_safari = 1;
	    else if(cipher_suite == 0x1302) /* TLS_RSA_WITH_AES_256_CBC_SHA */
	      this_is_not_safari = 1;
	    else if(cipher_suite == 0x1303) /* TLS_RSA_WITH_3DES_EDE_CBC_SHA */
	      this_is_not_safari = 1;
	  }
