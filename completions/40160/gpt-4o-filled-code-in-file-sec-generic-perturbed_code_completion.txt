for(i = 0; i < cipher_length; i += 2) {
        u_int16_t cipher_id = ntohs(*((u_int16_t*)&packet->payload[cipher_offset + i]));

        if((cipher_id & 0x0F0F) == 0x0A0A) {
          // Skip GREASE ciphers
          continue;
        }

        if(ja3.client.num_cipher < MAX_NUM_JA3) {
          ja3.client.cipher[ja3.client.num_cipher++] = cipher_id;
        } else {
          invalid_ja3 = 1;
#ifdef DEBUG_TLS
          printf("Client TLS Invalid num ciphers %u\n", ja3.client.num_cipher);
#endif
        }

        switch(cipher_id) {
          case 0x1301: case 0x1302: case 0x1303: case 0x1304: case 0x1305:
            safari_ciphers++;
            break;
          case 0x1301: case 0x1302: case 0x1303: case 0x1304: case 0x1305:
          case 0xC02B: case 0xC02F: case 0xC02C: case 0xC030:
            chrome_ciphers++;
            break;
          case 0x1301: case 0x1302: case 0x1303: case 0x1304: case 0x1305:
          case 0xC02B: case 0xC02F: case 0xC02C: case 0xC030:
          case 0xCCA8: case 0xCCA9:
            looks_like_safari_on_big_sur++;
            break;
          default:
            this_is_not_safari = 1;
            break;
        }
      }
