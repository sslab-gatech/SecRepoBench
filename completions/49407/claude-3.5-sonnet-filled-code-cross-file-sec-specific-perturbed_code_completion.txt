if (version >= 2)
{
  hb_vector_t<int> transformed_coords;
  if (unlikely (!transformed_coords.resize (effectiveaxiscount)))
    return;

  for (unsigned int i = 0; i < effectiveaxiscount; i++)
  {
    unsigned int variation_index = varIdxMap.map (coords[i]);
    int delta = varStore.get_delta (variation_index, coords[i]);
    int transformed = coords[i] + hb_round (delta);
    transformed_coords[i] = hb_clamp (transformed, axisRanges[i].minimum, axisRanges[i].maximum);
  }

  varStore.destroy_cache ();

  for (unsigned int i = 0; i < effectiveaxiscount; i++)
    coords[i] = transformed_coords[i];
}
