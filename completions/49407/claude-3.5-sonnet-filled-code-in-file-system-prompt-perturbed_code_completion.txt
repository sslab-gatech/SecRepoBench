for (unsigned int i = 0; i < effectiveaxiscount; i++)
{
  coords[i] = map->map (coords[i]);

  const auto &v2 = * (const avarV2Tail *) map;
  hb_vector_t<int> output;
  if (unlikely (!output.alloc (effectiveaxiscount)))
    return;

  hb_array_t<const int> input (coords, effectiveaxiscount);
  unsigned int num_coords = effectiveaxiscount;

  const DeltaSetIndexMap &index_map = v2.varIdxMap;
  const VariationStore &var_store = v2.varStore;

  hb_vector_t<int> indices;
  index_map.get_output_indices (input, indices);

  VariationStore::cache_t *cache = var_store.create_cache ();
  for (unsigned int i = 0; i < num_coords; i++)
  {
    int delta = roundf (var_store.get_delta (indices[i], cache));
    output[i] = hb_clamp (coords[i] + delta, -16384, 16384);
  }
  var_store.destroy_cache (cache);

  hb_copy (output, coords, num_coords);
}
