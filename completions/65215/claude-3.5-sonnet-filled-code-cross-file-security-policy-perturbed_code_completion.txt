size_t len = strlen(urlString);
if (len >= 7 && strncmp(urlString, "file://", 7) == 0) {
    if (len >= 10 && urlString[7] == '/' && urlString[9] == ':') {
        // Case: "file:///C:/..." - remove "file:///"
        memmove(urlString, urlString + 8, len - 7);
    } else {
        // Case: "file://..." - remove "file://"
        memmove(urlString, urlString + 7, len - 6);
    }
}

char *src = urlString;
char *dst = urlString;
while (*src) {
    if (src[0] == '%' && src[1] == '2' && src[2] == '0') {
        *dst++ = ' ';
        src += 3;
    } else {
        *dst++ = *src++;
    }
}
*dst = '\0';
