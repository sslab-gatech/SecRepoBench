if (!urlString) return;

	// Check if the URL starts with "file://"
	if (!strnicmp(urlString, "file://", 7)) {
		// Move the pointer to skip the "file://" part
		urlString += 7;

		// On Windows, if the path starts with a drive letter, skip the additional slash
		if (urlString[0] == '/' && urlString[1] && urlString[2] == ':') {
			urlString++;
		}
	}

	// Replace "%20" with spaces
	char *ptr = urlString;
	while ((ptr = strstr(ptr, "%20")) != NULL) {
		*ptr = ' ';
		memmove(ptr + 1, ptr + 3, strlen(ptr + 3) + 1);
	}
