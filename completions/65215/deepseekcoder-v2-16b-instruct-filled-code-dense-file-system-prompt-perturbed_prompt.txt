GF_EXPORT
GF_FileIO *gf_fileio_from_url(const char *url)
{
	char szURL[100];
	GF_FileIO *ptr=NULL;
	if (!url) return NULL;
	if (strncmp(url, "gfio://", 7)) return NULL;

	sscanf(url, "gfio://%p", &ptr);
	sprintf(szURL, "gfio://%p", ptr);
	if (strcmp(url, szURL))
		return NULL;

	if (ptr && ptr->url && !strcmp(ptr->url, url) ) {
		return ptr;
	}
	return NULL;
}
