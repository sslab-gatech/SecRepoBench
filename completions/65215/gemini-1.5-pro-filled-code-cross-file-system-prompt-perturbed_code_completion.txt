if (strncmp(urlString, "file:///", 8) == 0) {
		// Handle URLs like "file:///C:/path/to/file"
		if (urlString[8] && urlString[9] == ':') {
			memmove(urlString, urlString + 7, strlen(urlString) - 7 + 1); // +1 for null terminator
		} else {
			memmove(urlString, urlString + 8, strlen(urlString) - 8 + 1);
		}

	} else if (strncmp(urlString, "file://", 7) == 0) {
		memmove(urlString, urlString + 7, strlen(urlString) - 7 + 1);
	}

	char *p = urlString;
	while ((p = strstr(p, "%20"))) {
		*p = ' ';
		memmove(p + 1, p + 3, strlen(p + 3) + 1);
		p++;
	}
