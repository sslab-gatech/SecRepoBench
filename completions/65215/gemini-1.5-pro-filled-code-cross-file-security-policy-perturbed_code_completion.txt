u32 len = (u32) strlen(urlString);
	if (len > 7 && strncmp(urlString, "file://", 7) == 0) {
		if (urlString[7] == '/') {
			// "file:///" prefix
			memmove(urlString, urlString + 8, len - 7);
		} else {
			// "file://" prefix
			memmove(urlString, urlString + 7, len - 6);
		}
	}

	// Replace "%20" with spaces
	char *p = urlString;
	while ((p = strstr(p, "%20"))) {
		size_t remaining_len = strlen(p + 3) + 1; // Include null terminator
		if (remaining_len > 0 && len >= (p - urlString) + remaining_len + 1) {
			memmove(p + 1, p + 3, remaining_len);
			*p = ' ';
			p++;
		} else {
			break; // Handle potential overflow
		}
