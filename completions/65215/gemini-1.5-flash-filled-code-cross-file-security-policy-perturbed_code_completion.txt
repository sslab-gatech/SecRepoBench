	if (urlLen > 7 && strncmp(urlString, "file:///", 8) == 0) {
		memmove(urlString, urlString + 8, urlLen - 7);
		urlString[urlLen - 8] = '\0';
		urlLen -= 8;
	} else if (urlLen > 5 && strncmp(urlString, "file://", 7) == 0) {
		memmove(urlString, urlString + 7, urlLen - 6);
		urlString[urlLen - 7] = '\0';
		urlLen -= 7;
	}
	char *pos = urlString;
	while ((pos = strstr(pos, "%20")) != NULL) {
		size_t remainingLen = strlen(pos);
		memmove(pos + 1, pos + 3, remainingLen - 2);
		pos[remainingLen - 2] = '\0';
