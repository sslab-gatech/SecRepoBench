size_t len = strlen(urlString);
	if (len > 7 && strncmp(urlString, "file:///", 7) == 0) {
		memmove(urlString, urlString + 7, len - 7 + 1);
		len -= 7;
		if (len > 2 && urlString[1] == ':' && urlString[2] == '/') {
			memmove(urlString, urlString + 2, len - 2 + 1);
			len -= 2;
		}
	}
	char *ptr = urlString;
	while ((ptr = strstr(ptr, "%20")) != NULL) {
		memmove(ptr, ptr + 3, strlen(ptr + 3) + 1);
		*ptr = ' ';
	}
}
