if (!urlString) return;

    // Check if the URL starts with "file://"
    const char *filePrefix = "file://";
    size_t prefixLen = strlen(filePrefix);
    if (strncmp(urlString, filePrefix, prefixLen) == 0) {
        // Move the pointer past the "file://" prefix
        urlString += prefixLen;

        // On Windows, if the path starts with a drive letter, it will be in the form "file:///C:/..."
        // We need to check for this and adjust accordingly
        if (urlString[0] == '/' && urlString[1] && urlString[2] == ':') {
            // Move past the leading slash for Windows drive letter paths
            urlString++;
        }
    }

    // Decode "%20" to spaces
    char *src = urlString;
    char *dst = urlString;
    while (*src) {
        if (src[0] == '%' && src[1] == '2' && src[2] == '0') {
            *dst++ = ' ';
            src += 3;
        } else {
            *dst++ = *src++;
        }
    }
    *dst = '\0';
