size_t urlLen = strlen(urlString);
	if (urlLen > 7 && strncmp(urlString, "file:///", 7) == 0) {
		if (urlLen > 8 && urlString[7] == '/') {
			memmove(urlString, urlString + 8, urlLen - 7);
			urlString[urlLen - 8] = '\0';
		} else {
			memmove(urlString, urlString + 7, urlLen - 7);
			urlString[urlLen - 7] = '\0';
		}
	}
	char *pos = urlString;
	while ((pos = strstr(pos, "%20")) != NULL) {
		memmove(pos, pos + 3, strlen(pos + 3) + 1);
		*pos = ' ';
	}
