        return;
    }

    // Check for "file://" prefix
    if (strncmp(urlString, "file://", 7) == 0) {
        urlString += 7; // Skip the "file://" part
    }

    // Handle Windows drive letters
    if (urlString[0] && urlString[1] == ':') {
        // If the URL starts with a drive letter, adjust for Windows
        urlString += 1; // Skip the drive letter
    }

    // Replace "%20" with spaces
    char *pos = urlString;
    while ((pos = strstr(pos, "%20")) != NULL) {
        memmove(pos + 1, pos + 3, strlen(pos + 3) + 1); // Shift the rest of the string
        *pos = ' '; // Replace "%20" with a space
    }
