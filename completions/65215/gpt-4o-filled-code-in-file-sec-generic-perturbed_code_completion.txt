void gf_url_to_fs_path(char *urlString)
{
    if (!urlString) return;

    // Check if the URL starts with "file://"
    if (!strnicmp(urlString, "file://", 7)) {
        // Move the pointer to skip the "file://" part
        urlString += 7;

        // On Windows, if the path starts with a drive letter, it should be in the form "file:///C:/..."
        // We need to adjust the pointer to skip the third slash
#ifdef WIN32
        if (urlString[0] == '/' && urlString[1] && urlString[2] == ':') {
            urlString++;
        }
#endif
    }

    // Replace "%20" with spaces
    char *src = urlString;
    char *dst = urlString;
    while (*src) {
        if (!strnicmp(src, "%20", 3)) {
            *dst++ = ' ';
            src += 3;
        } else {
            *dst++ = *src++;
        }
    }
    *dst = '\0';
}
