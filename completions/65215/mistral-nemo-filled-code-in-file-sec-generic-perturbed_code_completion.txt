void gf_url_to_fs_path(char *urlString)
{
    // Check if the URL starts with "file://"
    if (strncmp(urlString, "file://", 7) == 0) {
        // Remove the "file://" prefix
        memmove(urlString, urlString + 7, strlen(urlString) - 6);
    }

    // Replace occurrences of "%20" with spaces
    char *p = urlString;
    while ((p = strstr(p, "%20")) != NULL) {
        memmove(p, p + 3, strlen(p + 3) + 1);
        *(p + 0) = ' ';
        *(p + 1) = ' ';
        p += 2;
    }
}
