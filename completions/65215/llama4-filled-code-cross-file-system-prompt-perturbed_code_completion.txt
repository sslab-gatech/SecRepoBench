char *start = strstr(urlString, "file://");
if (start) {
    start += 7; // move past "file://"
    if (start[0] == '/') {
        start++;
        // handle Windows drive letter case
        if (start[0] && start[1] == ':') {
            start += 2;
        }
    }
    memmove(urlString, start, strlen(start) + 1);
}

char *pos = urlString;
while ((pos = strstr(pos, "%20"))) {
    *pos = ' ';
    memmove(pos + 1, pos + 3, strlen(pos + 3) + 1);
}
