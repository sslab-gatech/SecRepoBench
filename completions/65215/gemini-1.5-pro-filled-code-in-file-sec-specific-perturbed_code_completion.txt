if (strncmp(urlString, "file://", 7) == 0) {
		char* pathStart;
		size_t prefixLength = 7;  // Length of "file://"

		// Check for an additional slash after "file://" (e.g., "file:///")
		if (urlString[7] == '/') {
			prefixLength++;
			// Check for Windows drive letter format (e.g. "file:///C:/path")
			if (urlString[8] && urlString[9] == ':') {
				prefixLength++;
			}
		}
		pathStart = urlString + prefixLength;

		//move to the beginning
		size_t pathLength = strlen(pathStart);
		memmove(urlString, pathStart, pathLength + 1); // +1 for null terminator
	}

	// Replace "%20" with spaces
	char* p = urlString;
	while ((p = strstr(p, "%20")) != NULL) {
		*p = ' ';
		memmove(p + 1, p + 3, strlen(p + 3) + 1); // +1 for null terminator
		p++; // Move past the replaced space
	}
