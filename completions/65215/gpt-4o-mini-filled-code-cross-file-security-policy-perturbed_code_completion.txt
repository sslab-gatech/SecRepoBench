if (urlString && strlen(urlString) > 0) {
        const char *prefix = "file://";
        const char *prefix3 = "file:///";
        size_t len = strlen(urlString);

        // Check for "file:///" prefix
        if (len >= strlen(prefix3) && strncmp(urlString, prefix3, strlen(prefix3)) == 0) {
            // Adjust for drive letter (e.g., "file:///C:/path")
            if (len > strlen(prefix3) + 1 && urlString[8] == ':') {
                memmove(urlString + 1, urlString + 8, len - 7); // Move the path left
                urlString[0] = urlString[7]; // Set the drive letter
                urlString[1] = '/'; // Add a '/' after the drive letter
                urlString[len - 7] = '\0'; // Null-terminate the new string
            } else {
                memmove(urlString, urlString + strlen(prefix3) - 1, len - strlen(prefix3) + 1); // Move the path left
            }
        } else if (len >= strlen(prefix) && strncmp(urlString, prefix, strlen(prefix)) == 0) {
            // Check for "file://" prefix
            memmove(urlString, urlString + strlen(prefix), len - strlen(prefix) + 1); // Move the path left
        }

        // Replace "%20" with spaces
        char *pos = urlString;
        while ((pos = strstr(pos, "%20")) != NULL) {
            memmove(pos + 1, pos + 3, strlen(pos + 3) + 1); // Shift the rest of the string left
            *pos = ' '; // Replace "%20" with space
        }
