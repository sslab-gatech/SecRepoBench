        // Adjust for Windows drive letters
        if (urlString[7] == '/') {
            // Remove "file:///" and replace with the drive letter
            char driveLetter = urlString[8];
            memmove(urlString, urlString + 7, strlen(urlString) - 6);
            urlString[0] = driveLetter;
            urlString[1] = ':';
            urlString[2] = '\\';
            urlString[3] = '\0';
        } else {
            // Remove "file://"
            memmove(urlString, urlString + 7, strlen(urlString) - 6);
        }
    }

    char *pos = urlString;
    while ((pos = strstr(pos, "%20")) != NULL) {
        memmove(pos + 1, pos + 3, strlen(pos + 3) + 1); // Shift left
