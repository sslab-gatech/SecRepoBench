if (strnicmp(urlString, "file://", 7) == 0) {
		char *pathStart = urlString + 7;
		// Check for an additional slash after "file://" (e.g., "file:///")
		if (pathStart[0] == '/') {
			pathStart++;
		}
		size_t pathLen = strlen(pathStart);
		memmove(urlString, pathStart, pathLen + 1); // +1 for null terminator
	}

	char *p = urlString;
	while (*p) {
		if (!strnicmp(p, "%20", 3)) {
			*p = ' ';
			memmove(p + 1, p + 3, strlen(p + 3) + 1);
		}
		p++;
