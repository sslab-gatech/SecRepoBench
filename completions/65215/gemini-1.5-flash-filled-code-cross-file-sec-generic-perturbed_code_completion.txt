		if (len > 8 && urlString[7] == '/') {
			memmove(urlString, urlString + 8, len - 7);
			urlString[len - 8] = '\0';
		} else {
			memmove(urlString, urlString + 7, len - 7);
			urlString[len - 7] = '\0';
		}
	}
	char *ptr = urlString;
	while ((ptr = strstr(ptr, "%20")) != NULL) {
		memmove(ptr, ptr + 3, strlen(ptr + 3) + 1);
