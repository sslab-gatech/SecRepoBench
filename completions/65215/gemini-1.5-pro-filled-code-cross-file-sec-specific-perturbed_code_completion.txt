if (!urlString) return;

	char *url = urlString;
	if (strstr(urlString, "file:///") == urlString) {
		if (urlString[7] == '/') {
			url += 8;
		} else {
			url += strlen("file://");
		}
	} else if (strstr(urlString, "file://") == urlString) {
		url += strlen("file://");
	}

	size_t len = strlen(url);
	char *buf = (char *)malloc(len + 1);
	if (!buf) return;
	char *p = buf;
	while (*url) {
		if (url[0] == '%' && url[1] == '2' && url[2] == '0') {
			*p++ = ' ';
			url += 3;
		} else {
			*p++ = *url++;
		}
	}
	*p = 0;
	strcpy(urlString, buf);
	free(buf);
