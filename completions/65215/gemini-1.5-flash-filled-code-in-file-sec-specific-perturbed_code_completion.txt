size_t len = strlen(urlString);
	if (len > 7 && !strnicmp(urlString, "file://", 7)) {
		memmove(urlString, urlString + 7, len - 7 + 1);
		len -= 7;
		if (len > 2 && urlString[1] == ':' && (urlString[2] == '/' || urlString[2] == '\\')) {
			memmove(urlString + 1, urlString + 3, len - 2);
			len -= 2;
		}
	}
	for (size_t i = 0; i < len; ++i) {
		if (urlString[i] == '%' && i + 2 < len && !strnicmp(urlString + i, "%20", 3)) {
			urlString[i] = ' ';
			memmove(urlString + i + 1, urlString + i + 3, len - i - 3 + 1);
			len -= 2;
			i--;
		}
	}
