if (!urlString) return;
	
	// Check for "file://" prefix
	if (strncmp(urlString, "file://", 7) == 0) {
		// Handle Windows-style drive letter case (e.g., "file:///C:/path/to/file.txt")
		if (urlString[7] == '/' && 
			((urlString[8] >= 'A' && urlString[8] <= 'Z') || (urlString[8] >= 'a' && urlString[8] <= 'z')) && 
			urlString[9] == ':') {
			// Move the drive letter and the rest of the path to the beginning of the string
			memmove(urlString, urlString + 8, strlen(urlString + 8) + 1);
		} else {
			// Standard case, just remove "file://" prefix
			memmove(urlString, urlString + 7, strlen(urlString + 7) + 1);
		}
	}
	
	// Replace "%20" with spaces
	char *src = urlString;
	char *dst = urlString;
	while (*src) {
		if (src[0] == '%' && src[1] == '2' && src[2] == '0') {
			*dst++ = ' ';
			src += 3;
		} else {
			*dst++ = *src++;
		}
	}
	*dst = '\0';
