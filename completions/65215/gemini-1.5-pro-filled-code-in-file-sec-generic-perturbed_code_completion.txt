if (strncmp(urlString, "file://", 7) == 0) {
		char *newString;
		size_t newLen;

		// Check for and handle the "file:///" prefix with a drive letter
		if (strlen(urlString) > 7 && urlString[7] == '/') {
			newString = urlString + 8; // Skip "file:///"
		} else {
			newString = urlString + 7; // Skip "file://"
		}

		newLen = strlen(newString);
		memmove(urlString, newString, newLen + 1); // Shift the string to remove the prefix
	}

	// Replace "%20" with spaces
	char *p = urlString;
	while ((p = strstr(p, "%20")) != NULL) {
		*p = ' ';
		memmove(p + 1, p + 3, strlen(p + 3) + 1);
		p++; // Move past the replaced space
	}
