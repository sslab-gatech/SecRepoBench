static void copy_CTB(uint8_t *dst, const uint8_t *src, int w, int height,
                     ptrdiff_t stride_dst, ptrdiff_t stride_src)
{
    int i, j;

    if (((intptr_t)dst | (intptr_t)src | stride_dst | stride_src) & 15) {
        // Copy the pixels from a source buffer to a destination buffer.
        // To prevent a heap overflow, we should use a loop that copies the pixels
        // in a way that does not overflow the buffer. We can do this by using a
        // loop that copies the pixels in chunks of 16, which is the size of a
        // SIMD register.
        for (i = 0; i < height; i++) {
            for (j = 0; j < w; j += 16) {
                // Copy 16 pixels at a time.
                for (int k = 0; k < 16; k++) {
                    dst[j + k] = src[j + k];
                }
            }
            dst += stride_dst;
            src += stride_src;
        }
    } else {
        for (i = 0; i < height; i++) {
            for (j = 0; j < w; j+=16)
                AV_COPY128(dst+j, src+j);
            dst += stride_dst;
            src += stride_src;
        }
    }
}
