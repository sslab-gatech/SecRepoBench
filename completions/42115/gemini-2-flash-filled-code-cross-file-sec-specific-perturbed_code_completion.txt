char *tmp = NULL;
		  char *new_server_names = NULL;
		  char *cleaned_dns_name = NULL;
		  
		  memset(dNSName, 0, sizeof(dNSName));
		  strncpy(dNSName, (const char*)&packet->payload[i + 2], len);
		  dNSName[len] = '\0';

		  /* Clean the DNS name */
		  cleaned_dns_name = ndpi_cleanup_host_server_name(dNSName);

		  if(cleaned_dns_name != NULL && cleaned_dns_name[0] != '\0') {
		    if(strcasecmp(connection->host_server_name, cleaned_dns_name) == 0) {
		      matched_name = 1;

		      if(connection->server_names == NULL) {
			connection->server_names = ndpi_strdup(cleaned_dns_name);
		      } else {
			/* Append the new server name to the existing list */
			size_t current_len = strlen(connection->server_names);
			size_t new_name_len = strlen(cleaned_dns_name);

			new_server_names = (char*)ndpi_malloc(current_len + new_name_len + 2); // +1 for comma, +1 for null terminator
			if (new_server_names != NULL) {
			  snprintf(new_server_names, current_len + new_name_len + 2, "%s,%s", connection->server_names, cleaned_dns_name);
			  ndpi_free(connection->server_names);
			  connection->server_names = new_server_names;
			}
		      }

		      /* Detect subprotocol based on the DNS name */
		      tmp = ndpi_strstr(cleaned_dns_name, ".zoom.us");
		      if(tmp != NULL) {
			ndpi_set_detected_protocol(ndpi_struct, connection, NDPI_PROTOCOL_ZOOM, NDPI_PROTOCOL_TLS);
		      }
		    }
		  }
		  if(cleaned_dns_name != NULL)
		    ndpi_free(cleaned_dns_name);
