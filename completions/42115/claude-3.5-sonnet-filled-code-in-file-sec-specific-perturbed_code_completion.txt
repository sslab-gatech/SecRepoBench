int alpn_len = comma_or_nul - alpn_start;

if(!is_a_common_alpn(ndpi_struct, alpn_start, alpn_len)) {

#endif
  ndpi_set_risk(ndpi_struct, flow, NDPI_TLS_UNCOMMON_ALPN);
  break;
}
