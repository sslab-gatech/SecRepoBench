if (unicodeLength > 0) {

    if (SizeOfTag < unicodeLength * sizeof(cmsUInt16Number)) goto Done;
    SizeOfTag -= unicodeLength * sizeof(cmsUInt16Number);

    UnicodeString = (wchar_t*) _cmsCalloc(self ->ContextID, unicodeLength + 1, sizeof(wchar_t));
    if (UnicodeString == NULL) goto Error;

    if (!_cmsReadWCharArray(io, unicodeLength, UnicodeString)) goto Error;
    UnicodeString[unicodeLength] = 0;

    if (!cmsMLUsetWide(mlu, cmsV2Unicode, cmsV2Unicode, UnicodeString)) goto Error;
    _cmsFree(self ->ContextID, UnicodeString);
    UnicodeString = NULL;
}

Done:

    // Do we have ScriptCode?
    if (SizeOfTag >= 3) {

        if (!_cmsReadUInt16Number(io, &ScriptCodeCode)) goto Error;
        if (!_cmsReadUInt8Number(io, &ScriptCodeCount)) goto Error;
        SizeOfTag -= 3;

        if (SizeOfTag < ScriptCodeCount) goto Error;
        SizeOfTag -= ScriptCodeCount;

        for (i=0; i < ScriptCodeCount; i++) {
            if (!_cmsReadUInt8Number(io, NULL)) goto Error;
        }
    }

    // Do we have Apple stuff?
    if (SizeOfTag >= 2) {

        if (!_cmsReadUInt16Number(io, &Dummy)) goto Error;
        SizeOfTag -= 2;
    }

    *nItems = 1;
    return mlu;

Error:
    if (Text != NULL) _cmsFree(self ->ContextID, Text);
    if (UnicodeString != NULL) _cmsFree(self ->ContextID, UnicodeString);
    if (mlu != NULL) cmsMLUfree(mlu);
