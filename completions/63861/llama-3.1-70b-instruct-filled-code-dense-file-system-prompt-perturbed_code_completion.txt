// Allocate memory to store the Unicode string
UnicodeString = (wchar_t*) _cmsMalloc(self ->ContextID, unicodeLength * sizeof(wchar_t));
if (UnicodeString == NULL) goto Done;

// Read the Unicode string from the IO handler
if (io ->Read(io, UnicodeString, sizeof(wchar_t), unicodeLength)!= unicodeLength) goto Done;

// Check if there is enough data left to process the Unicode string
if (SizeOfTag < unicodeLength * sizeof(wchar_t)) goto Done;

// Convert the Unicode string to a wide character array
if (!cmsMLUsetWide(mlu, cmsV2Unicode, cmsV2Unicode, UnicodeString, unicodeLength)) goto Done;

// Add the Unicode string to the MLU structure
_cmsFree(self ->ContextID, UnicodeString);
UnicodeString = NULL;

// Adjust the remaining size of the tag data
SizeOfTag -= unicodeLength * sizeof(wchar_t);

// Check for the presence of optional ScriptCode elements
if (SizeOfTag >= sizeof(cmsUInt16Number) + sizeof(cmsUInt8Number)) {
    // Read and discard the ScriptCode code and count
    if (!_cmsReadUInt16Number(io, &ScriptCodeCode)) goto Done;
    if (!_cmsReadUInt8Number(io, &ScriptCodeCount)) goto Done;
    SizeOfTag -= sizeof(cmsUInt16Number) + sizeof(cmsUInt8Number);

    // Skip over the ScriptCode data
    if (SizeOfTag < ScriptCodeCount) goto Done;
    if (io ->Seek(io, ScriptCodeCount)!= ScriptCodeCount) goto Done;
    SizeOfTag -= ScriptCodeCount;
}

Done:
