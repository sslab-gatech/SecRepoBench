// <MASK>
if (SizeOfTag < sizeof(cmsUInt16Number) + sizeof(cmsUInt32Number)) {
    goto Done;
}

// Allocate memory to store the Unicode string
UnicodeString = (wchar_t*)_cmsMalloc(self->ContextID, (SizeOfTag / sizeof(cmsUInt16Number)) * sizeof(wchar_t) + sizeof(wchar_t));
if (UnicodeString == NULL) {
    goto Error;
}

// Read the Unicode string from the IO handler
if (io->Read(io, UnicodeString, sizeof(cmsUInt16Number), SizeOfTag / sizeof(cmsUInt16Number)) != SizeOfTag / sizeof(cmsUInt16Number)) {
    goto Error;
}

// Ensure the Unicode string is null-terminated
UnicodeString[SizeOfTag / sizeof(cmsUInt16Number)] = L'\0';

// Convert the Unicode string to a wide character array and add it to the MLU structure
if (!cmsMLUsetWide(mlu, cmsNoLanguage, cmsNoCountry, UnicodeString)) {
    goto Error;
}

// Adjust the remaining size of the tag data
SizeOfTag -= SizeOfTag;

// Check for the presence of optional ScriptCode elements
if (SizeOfTag >= sizeof(cmsUInt8Number) + sizeof(cmsUInt16Number)) {
    if (!_cmsReadUInt8Number(io, &ScriptCodeCount)) {
        goto Done;
    }
    SizeOfTag -= sizeof(cmsUInt8Number);

    for (i = 0; i < ScriptCodeCount; i++) {
        if (!_cmsReadUInt16Number(io, &ScriptCodeCode)) {
            goto Done;
        }
        if (!_cmsReadUInt16Number(io, &Dummy)) {
            goto Done;
        }
        SizeOfTag -= 2 * sizeof(cmsUInt16Number);
    }
}

Done:
return (void*)mlu;

Error:
if (Text) _cmsFree(self->ContextID, (void*)Text);
if (UnicodeString) _cmsFree(self->ContextID, (void*)UnicodeString);
if (mlu) cmsMLUfree(mlu);
return NULL;
