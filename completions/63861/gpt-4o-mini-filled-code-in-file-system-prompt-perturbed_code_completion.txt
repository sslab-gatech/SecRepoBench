if (SizeOfTag < sizeof(cmsUInt32Number) * 2) goto Done;

    // Read the Unicode code and length
    if (!_cmsReadUInt32Number(io, &UnicodeCode)) goto Done;
    if (!_cmsReadUInt32Number(io, &unicodeLength)) goto Done;
    SizeOfTag -= 2 * sizeof(cmsUInt32Number);

    // Read the Unicode string
    if (unicodeLength > 0) {
        wchar_t* unicodeString = (wchar_t*)_cmsMalloc(self->ContextID, (unicodeLength + 1) * sizeof(wchar_t));
        if (unicodeString == NULL) goto Done;

        if (!_cmsReadWCharArray(io, unicodeLength, unicodeString)) {
            _cmsFree(self->ContextID, unicodeString);
            goto Done;
        }
        unicodeString[unicodeLength] = 0; // Null-terminate the string
        cmsMLUsetWide(mlu, UnicodeCode, unicodeString);
        _cmsFree(self->ContextID, unicodeString);
    }

    // Handle optional ScriptCode elements
    if (SizeOfTag > 0) {
        cmsUInt32Number scriptCodeCount;
        if (!_cmsReadUInt32Number(io, &scriptCodeCount)) goto Done;
        SizeOfTag -= sizeof(cmsUInt32Number);

        for (i = 0; i < scriptCodeCount; i++) {
            // Read script code and its description
            cmsUInt16Number scriptCode;
            if (!_cmsReadUInt16Number(io, &scriptCode)) goto Done;
            SizeOfTag -= sizeof(cmsUInt16Number);

            wchar_t* scriptDescription = NULL;
            if (!ReadOneWChar(io, &a.DisplayValue, i, &scriptDescription)) goto Done;

            // Store the script code and description in the MLU
            cmsMLUsetWide(mlu, scriptCode, scriptDescription);
            if (scriptDescription) _cmsFree(self->ContextID, scriptDescription);
        }
    }

Done:
    if (mlu) cmsMLUfree(mlu);
    return NULL;
