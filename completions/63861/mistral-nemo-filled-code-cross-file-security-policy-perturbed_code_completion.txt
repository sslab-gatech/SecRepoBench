// Allocate memory for the Unicode string, including an extra element for the null terminator
    UnicodeString = (wchar_t*) _cmsMalloc(self ->ContextID, unicodeLength * sizeof(wchar_t) + sizeof(wchar_t));
    if (UnicodeString == NULL) goto Error;

    // Check if there is enough data left to process the Unicode string
    if (SizeOfTag < unicodeLength * sizeof(wchar_t)) goto Error;

    // Read the Unicode string from the IO handler
    if (io ->Read(io, UnicodeString, sizeof(wchar_t), unicodeLength) != unicodeLength) goto Error;
    SizeOfTag -= unicodeLength * sizeof(wchar_t);

    // Convert the Unicode string to a wide character array and add it to the MLU structure
    if (!cmsMLUsetUnicode(mlu, UnicodeCode, UnicodeString, unicodeLength)) goto Error;
    _cmsFree(self ->ContextID, (void*) UnicodeString);
    UnicodeString = NULL;

    // Skip over optional ScriptCode elements, if present
    while (SizeOfTag >= 2 * sizeof(cmsUInt32Number)) {
        if (!_cmsReadUInt32Number(io, &ScriptCodeCode)) goto Done;
        if (!_cmsReadUInt32Number(io, &ScriptCodeCount)) goto Done;
        SizeOfTag -= 2 * sizeof(cmsUInt32Number);
        if (SizeOfTag < ScriptCodeCount) goto Done;
        io ->Skip(io, ScriptCodeCount);
        SizeOfTag -= ScriptCodeCount;
    }

    goto Done;

Error:
    if (UnicodeString) _cmsFree(self ->ContextID, (void*) UnicodeString);
    if (mlu) cmsMLUfree(mlu);
    return NULL;

Done:
    return mlu;
