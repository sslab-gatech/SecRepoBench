if (SizeOfTag < 2 * sizeof(cmsUInt32Number)) goto Done;
    if (!_cmsReadUInt32Number(io, &UnicodeCode)) goto Done;
    if (!_cmsReadUInt32Number(io, &unicodeLength)) goto Done;
    SizeOfTag -= 2 * sizeof(cmsUInt32Number);

    // Read the Unicode string
    if (unicodeLength > 0) {
        UnicodeString = (wchar_t*) _cmsMalloc(self->ContextID, (unicodeLength + 1) * sizeof(wchar_t));
        if (UnicodeString == NULL) goto Done;

        if (!_cmsReadWCharArray(io, unicodeLength, UnicodeString)) {
            _cmsFree(self->ContextID, UnicodeString);
            goto Done;
        }
        UnicodeString[unicodeLength] = 0; // Null-terminate the string
        if (!cmsMLUsetWide(mlu, cmsNoLanguage, cmsNoCountry, UnicodeString)) goto Done;
        _cmsFree(self->ContextID, UnicodeString);
    }

    // Skip optional ScriptCode elements
    if (SizeOfTag < 2 * sizeof(cmsUInt32Number)) goto Done;
    if (!_cmsReadUInt32Number(io, NULL)) goto Done; // ScriptCode
    if (!_cmsReadUInt32Number(io, NULL)) goto Done; // ScriptCode count

    // Read the ScriptCode description (if any)
    if (SizeOfTag > 0) {
        if (!_cmsReadWCharArray(io, 67, NULL)) goto Done; // Skip ScriptCode description
    }

Done:
    if (UnicodeString) _cmsFree(self->ContextID, UnicodeString);
    if (mlu) cmsMLUfree(mlu);
    return NULL;
