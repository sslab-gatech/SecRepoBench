// Check if there is enough data left to process the Unicode string
    if (SizeOfTag < UnicodeCode + unicodeLength) goto Done;

    // Allocate memory for the Unicode string
    UnicodeString = (wchar_t*) _cmsMalloc(self ->ContextID, unicodeLength * sizeof(wchar_t));
    if (UnicodeString == NULL) goto Error;

    // Read the Unicode string from the IO handler
    if (io ->Read(io, UnicodeString, sizeof(wchar_t), unicodeLength) != unicodeLength) {
        _cmsFree(self ->ContextID, UnicodeString);
        goto Error;
    }
    SizeOfTag -= unicodeLength * sizeof(wchar_t);

    // Null-terminate the Unicode string
    UnicodeString[unicodeLength] = 0;

    // Add the Unicode string to the MLU structure
    if (!cmsMLUsetWide(mlu, cmsNoLanguage, cmsNoCountry, UnicodeString)) {
        _cmsFree(self ->ContextID, UnicodeString);
        goto Error;
    }

    _cmsFree(self ->ContextID, UnicodeString);
    UnicodeString = NULL;

Done:
    // Skip ScriptCode elements if present
    if (SizeOfTag >= sizeof(cmsUInt8Number)) {
        if (!_cmsReadUInt8Number(io, &ScriptCodeCount)) goto Error;
        SizeOfTag -= sizeof(cmsUInt8Number);

        if (SizeOfTag < ScriptCodeCount * sizeof(cmsUInt16Number)) goto Done;

        for (i = 0; i < ScriptCodeCount; i++) {
            if (!_cmsReadUInt16Number(io, &ScriptCodeCode)) goto Error;
            if (!_cmsReadUInt16Number(io, &Dummy)) goto Error;
        }
        SizeOfTag -= ScriptCodeCount * sizeof(cmsUInt16Number) * 2;
    }

Error:
    if (Text) _cmsFree(self ->ContextID, (void*) Text);
    if (UnicodeString) _cmsFree(self ->ContextID, UnicodeString);
    if (mlu) cmsMLUfree(mlu);
    return NULL;
