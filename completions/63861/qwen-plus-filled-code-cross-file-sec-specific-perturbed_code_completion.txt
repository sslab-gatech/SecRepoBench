// Check if there is enough data left for the Unicode string
    if (SizeOfTag < unicodeLength * sizeof(cmsUInt16Number)) goto Done;

    // Allocate memory for the Unicode string, ensuring it has space for a null terminator
    UnicodeString = (wchar_t*) _cmsMalloc(self ->ContextID, (unicodeLength + 1) * sizeof(wchar_t));
    if (UnicodeString == NULL) goto Error;

    // Read the Unicode string from the IO handler
    for (i = 0; i < unicodeLength; i++) {
        if (!_cmsReadUInt16Number(io, (cmsUInt16Number*)&UnicodeString[i])) {
            _cmsFree(self ->ContextID, UnicodeString);
            goto Error;
        }
    }

    // Null-terminate the Unicode string
    UnicodeString[unicodeLength] = 0;

    // Update the remaining size of the tag data
    SizeOfTag -= unicodeLength * sizeof(cmsUInt16Number);

    // Add the Unicode string to the MLU structure
    if (!cmsMLUsetWide(mlu, cmsNoLanguage, cmsNoCountry, UnicodeString)) {
        _cmsFree(self ->ContextID, UnicodeString);
        goto Error;
    }

    // Free the allocated Unicode string after adding it to the MLU
    _cmsFree(self ->ContextID, UnicodeString);
    UnicodeString = NULL;

    // Skip optional ScriptCode elements if present
    if (SizeOfTag >= sizeof(cmsUInt8Number)) {
        if (!_cmsReadUInt8Number(io, &ScriptCodeCount)) goto Done;
        SizeOfTag -= sizeof(cmsUInt8Number);

        if (SizeOfTag < ScriptCodeCount * 2 * sizeof(cmsUInt8Number)) goto Done;

        for (i = 0; i < ScriptCodeCount; i++) {
            if (!_cmsReadUInt8Number(io, &Dummy)) goto Done;
            if (!_cmsReadUInt8Number(io, &ScriptCodeCode)) goto Done;
            SizeOfTag -= 2 * sizeof(cmsUInt8Number);
        }
    }

Done:
    if (Text) _cmsFree(self ->ContextID, (void*) Text);
    return mlu;

Error:
    if (UnicodeString) _cmsFree(self ->ContextID, UnicodeString);
    if (mlu) cmsMLUfree(mlu);
    return NULL;
