// Check if there is enough data left to process the Unicode string
if (SizeOfTag < unicodeLength * sizeof(wchar_t)) goto Done;

// Allocate a buffer for the Unicode string, including a null terminator
UnicodeString = (wchar_t*) _cmsMalloc(self ->ContextID, (unicodeLength + 1) * sizeof(wchar_t));
if (UnicodeString == NULL) goto Done;

// Read the Unicode string from the IO handler
if (io ->Read(io, UnicodeString, sizeof(wchar_t), unicodeLength)!= unicodeLength) goto Error;
SizeOfTag -= unicodeLength * sizeof(wchar_t);

// Make sure there is a terminator
UnicodeString[unicodeLength] = 0;

// Add the Unicode string to the MLU structure
if (!cmsMLUsetWide(mlu, cmsNoLanguage, cmsNoCountry, UnicodeString)) goto Error;
_cmsFree(self ->ContextID, (void*) UnicodeString);
UnicodeString = NULL;

// Check for the presence of optional ScriptCode elements
if (SizeOfTag >= sizeof(cmsUInt8Number) + sizeof(cmsUInt16Number)) {
    if (!_cmsReadUInt8Number(io, &ScriptCodeCount)) goto Done;
    if (!_cmsReadUInt16Number(io, &ScriptCodeCode)) goto Done;
    SizeOfTag -= sizeof(cmsUInt8Number) + sizeof(cmsUInt16Number);

    // Skip over the ScriptCode data
    if (SizeOfTag < ScriptCodeCount * sizeof(cmsUInt16Number)) goto Done;
    for (i = 0; i < ScriptCodeCount; i++) {
        if (!_cmsReadUInt16Number(io, &Dummy)) goto Done;
    }
    SizeOfTag -= ScriptCodeCount * sizeof(cmsUInt16Number);
}

Done:
*nItems = 1;
return mlu;

Error:
if (Text) _cmsFree(self ->ContextID, (void*) Text);
if (UnicodeString) _cmsFree(self ->ContextID, (void*) UnicodeString);
if (mlu) cmsMLUfree(mlu);
return NULL;
