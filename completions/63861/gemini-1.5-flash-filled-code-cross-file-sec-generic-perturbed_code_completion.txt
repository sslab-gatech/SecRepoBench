if (unicodeLength > 0) {
        UnicodeString = (wchar_t*) _cmsMalloc(self ->ContextID, unicodeLength * sizeof(wchar_t));
        if (UnicodeString == NULL) goto Error;
        if (io ->Read(io, UnicodeString, sizeof(wchar_t), unicodeLength) != unicodeLength) {
            _cmsFree(self ->ContextID, UnicodeString);
            goto Error;
        }
        if (!cmsMLUsetUnicode(mlu, cmsNoLanguage, cmsNoCountry, UnicodeString)) {
            _cmsFree(self ->ContextID, UnicodeString);
            goto Error;
        }
        _cmsFree(self ->ContextID, (void*) UnicodeString);
        UnicodeString = NULL;
        SizeOfTag -= unicodeLength * sizeof(wchar_t);
    }

    // Check for ScriptCode
    if (SizeOfTag >= sizeof(cmsUInt8Number) + sizeof(cmsUInt16Number)) {
        if (!_cmsReadUInt8Number(io, &ScriptCodeCount)) goto Done;
        SizeOfTag -= sizeof(cmsUInt8Number);
        for (i = 0; i < ScriptCodeCount; i++) {
            if (SizeOfTag < sizeof(cmsUInt16Number)) goto Done;
            if (!_cmsReadUInt16Number(io, &ScriptCodeCode)) goto Done;
            SizeOfTag -= sizeof(cmsUInt16Number);
        }
    }

Done:
    *nItems = 1;
    return (void*) mlu;
