// Check if there is enough data left to read the Unicode string
    if (SizeOfTag < unicodeLength * sizeof(cmsUInt16Number)) goto Done;

    // Allocate memory for the Unicode string
    UnicodeString = (wchar_t*) _cmsMalloc(self ->ContextID, unicodeLength * sizeof(wchar_t));
    if (UnicodeString == NULL) goto Done;

    // Read the Unicode characters from the IO handler
    if (!_cmsReadWCharArray(io, unicodeLength, UnicodeString)) {
        _cmsFree(self ->ContextID, UnicodeString);
        goto Done;
    }

    SizeOfTag -= unicodeLength * sizeof(cmsUInt16Number);

    // Add the Unicode string to MLU structure
    if (!cmsMLUsetWide(mlu, cmsNoLanguage, cmsNoCountry, UnicodeString)) {
        _cmsFree(self ->ContextID, UnicodeString);
        goto Done;
    }

    _cmsFree(self ->ContextID, UnicodeString);
    UnicodeString = NULL;

    // Skip optional ScriptCode elements
    if (SizeOfTag < sizeof(cmsUInt8Number)) goto Done;

    if (!_cmsReadUInt8Number(io, &ScriptCodeCount)) goto Done;
    SizeOfTag -= sizeof(cmsUInt8Number);

    // Ensure that enough ScriptCode data is available
    if (SizeOfTag < ScriptCodeCount * sizeof(cmsUInt16Number)) goto Done;

    for (i = 0; i < ScriptCodeCount; i++) {
        if (!_cmsReadUInt16Number(io, &Dummy)) goto Done;
        SizeOfTag -= sizeof(cmsUInt16Number);
    }

Done:
    if (Text) _cmsFree(self ->ContextID, (void*) Text);
    if (UnicodeString) _cmsFree(self ->ContextID, (void*) UnicodeString);

    if (mlu && SizeOfTag != 0) {
        cmsMLUfree(mlu);
        mlu = NULL;
    }

    return (void*) mlu;
