// Allocate memory for the Unicode string
    UnicodeString = (wchar_t*) _cmsMalloc(self->ContextID, (unicodeLength + 1) * sizeof(wchar_t));
    if (UnicodeString == NULL) goto Error;

    // Read the Unicode string
    if (io->Read(io, UnicodeString, sizeof(wchar_t), unicodeLength) != unicodeLength) goto Error;
    UnicodeString[unicodeLength] = 0; // Null-terminate the wide string

    // Set the MLU entry for the Unicode string
    if (!cmsMLUsetUnicode(mlu, cmsNoLanguage, cmsNoCountry, UnicodeString)) goto Error;

    // Free the allocated Unicode string
    _cmsFree(self->ContextID, (void*) UnicodeString);
    UnicodeString = NULL;

    // Adjust SizeOfTag for optional ScriptCode elements
    while (SizeOfTag >= sizeof(cmsUInt8Number)) {
        if (!_cmsReadUInt8Number(io, &ScriptCodeCount)) break; // Read ScriptCodeCount
        SizeOfTag -= sizeof(cmsUInt8Number);
        
        for (i = 0; i < ScriptCodeCount; i++) {
            if (SizeOfTag < sizeof(cmsUInt16Number)) break; // Check for enough data
            if (!_cmsReadUInt16Number(io, &ScriptCodeCode)) break; // Read ScriptCode
            SizeOfTag -= sizeof(cmsUInt16Number);
        }
    }

Done:
    *nItems = 1; // Indicate that one item has been read
    return mlu;

Error:
    if (UnicodeString) _cmsFree(self->ContextID, (void*) UnicodeString);
    if (mlu) cmsMLUfree(mlu);
    return NULL;
