if (unicodeLength > (SizeOfTag / 2))
    goto Done;

UnicodeString = (wchar_t*)_cmsCalloc(self->ContextID, unicodeLength + 1, sizeof(wchar_t));
if (UnicodeString == NULL)
    goto Done;

if (!convert_utf16_to_utf32(io, (cmsInt32Number)unicodeLength, UnicodeString))
    goto Done;

// Ensure it is null-terminated
UnicodeString[unicodeLength] = 0;

// Store in MLU
if (!cmsMLUsetWide(mlu, cmsNoLanguage, cmsNoCountry, UnicodeString, unicodeLength))
    goto Done;

// Free temporary buffer
_cmsFree(self->ContextID, UnicodeString);
UnicodeString = NULL;

// Account for the characters read
SizeOfTag -= (unicodeLength * 2);

// Read and skip ScriptCode data
if (SizeOfTag < 2 * sizeof(cmsUInt16Number))
    goto Done;

if (!_cmsReadUInt16Number(io, &ScriptCodeCode))
    goto Done;
if (!_cmsReadUInt16Number(io, &Dummy))
    goto Done;

SizeOfTag -= 2 * sizeof(cmsUInt16Number);

if (SizeOfTag < 1 + 67)
    goto Done;

if (!_cmsReadUInt8Number(io, &ScriptCodeCount))
    goto Done;

SizeOfTag--;

if (ScriptCodeCount > 67)
    goto Done;

if (io->Read(io, NULL, ScriptCodeCount, 1) != 1)
    goto Done;

SizeOfTag -= ScriptCodeCount;

// All done reading extended info
return (void*)mlu;
