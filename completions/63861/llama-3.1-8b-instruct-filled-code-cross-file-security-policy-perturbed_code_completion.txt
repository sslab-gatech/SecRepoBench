// Return the populated MLU object and handle any potential errors, ensuring allocated memory is freed if an error occurs.
if (SizeOfTag >= 2 * sizeof(cmsUInt32Number)) {
    // Allocate memory to store the Unicode string
    UnicodeString = (wchar_t*) _cmsMalloc(self ->ContextID, unicodeLength + 1);
    if (UnicodeString == NULL) goto Error;

    // Check if there is enough data left to process the Unicode string
    if (SizeOfTag < unicodeLength) goto Done;

    // Read the Unicode string from the IO handler
    if (io ->Read(io, UnicodeString, sizeof(wchar_t), unicodeLength)!= unicodeLength) goto Error;
    UnicodeString[unicodeLength] = 0;

    // Add the Unicode string to the MLU structure
    if (!cmsMLUsetUnicode(mlu, UnicodeString)) goto Error;

    // Adjust the remaining size of the tag data
    SizeOfTag -= unicodeLength;

    // Check for the presence of optional ScriptCode elements
    if (SizeOfTag >= 2 * sizeof(cmsUInt16Number)) {
        // Read and discard the ScriptCode data
        if (!_cmsReadUInt16Number(io, &ScriptCodeCode)) goto Done;
        if (!_cmsReadUInt16Number(io, &ScriptCodeCount)) goto Done;
        SizeOfTag -= 2 * sizeof(cmsUInt16Number);
    }
