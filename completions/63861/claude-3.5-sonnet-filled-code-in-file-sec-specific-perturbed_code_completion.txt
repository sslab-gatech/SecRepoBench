if (SizeOfTag < 2 * sizeof(cmsUInt32Number)) goto Error;

    UnicodeString = (wchar_t*) _cmsMalloc(self->ContextID, SizeOfTag);
    if (UnicodeString == NULL) goto Error;

    if (!convert_utf16_to_utf32(io, unicodeLength, UnicodeString)) {
        _cmsFree(self->ContextID, UnicodeString);
        goto Error;
    }

    if (!cmsMLUsetWide(mlu, cmsNoLanguage, cmsNoCountry, UnicodeString)) {
        _cmsFree(self->ContextID, UnicodeString);
        goto Error;
    }

    _cmsFree(self->ContextID, UnicodeString);

    SizeOfTag -= 2 * sizeof(cmsUInt32Number) + unicodeLength * sizeof(cmsUInt16Number);

    // Optional ScriptCode Code & count 
    if (SizeOfTag > 0) {
        if (!_cmsReadUInt16Number(io, &ScriptCodeCode)) goto Error;
        if (!_cmsReadUInt8Number(io, &ScriptCodeCount)) goto Error;
        if (!io->Seek(io, io->Tell(io) + ScriptCodeCount)) goto Error;
    }
