// Allocate memory to store the Unicode string and read the characters from the IO handler.
    UnicodeString = (wchar_t*) _cmsMalloc(self ->ContextID, unicodeLength * sizeof(wchar_t));
    if (UnicodeString == NULL) goto Error;
    if (io ->Read(io, UnicodeString, sizeof(wchar_t), unicodeLength) != unicodeLength) goto Error;
    SizeOfTag -= unicodeLength * sizeof(wchar_t);

    // Convert the Unicode string to a wide character array and add it to the MLU (Multi-Language Unicode) structure.
    if (!cmsMLUsetUnicode(mlu, UnicodeString, unicodeLength)) goto Error;
    _cmsFree(self ->ContextID, UnicodeString);
    UnicodeString = NULL;

    // After processing the Unicode string, adjust the remaining size of the tag data and check for the presence of optional ScriptCode elements.
    if (SizeOfTag < 2* sizeof(cmsUInt32Number)) goto Done;

    // If present, skip over the ScriptCode data by reading and discarding the necessary bytes.
    if (!_cmsReadUInt32Number(io, &ScriptCodeCount)) goto Done;
    SizeOfTag -= sizeof(cmsUInt32Number);
    for (i = 0; i < ScriptCodeCount; i++) {
        if (!_cmsReadUInt16Number(io, &ScriptCodeCode)) goto Done;
        if (!_cmsReadUInt16Number(io, &Dummy)) goto Done;
        SizeOfTag -= 2 * sizeof(cmsUInt16Number);
    }

    goto Done;

Error:
    if (UnicodeString) _cmsFree(self ->ContextID, UnicodeString);
    if (mlu) cmsMLUfree(mlu);
    return NULL;

Done:
    *nItems = mlu ->UsedEntries;
    return (void*) mlu;
