if (unicodeLength > 0) {
        if (SizeOfTag < unicodeLength * sizeof(cmsUInt16Number)) goto SkipUnicode;

        UnicodeString = (wchar_t*) _cmsMalloc(self->ContextID, (unicodeLength + 1) * sizeof(wchar_t));
        if (UnicodeString == NULL) goto Error;

        if (io->Read(io, UnicodeString, sizeof(cmsUInt16Number), unicodeLength) != unicodeLength) {
            _cmsFree(self->ContextID, UnicodeString);
            goto Error;
        }

        UnicodeString[unicodeLength] = 0;

        if (!cmsMLUsetWide(mlu, cmsNoLanguage, cmsNoCountry, UnicodeString)) {
            _cmsFree(self->ContextID, UnicodeString);
            goto Error;
        }

        _cmsFree(self->ContextID, UnicodeString);
        UnicodeString = NULL;
        SizeOfTag -= unicodeLength * sizeof(cmsUInt16Number);
    }

SkipUnicode:
    if (SizeOfTag >= sizeof(cmsUInt8Number)) {
        if (!_cmsReadUInt8Number(io, &ScriptCodeCount)) goto Done;
        SizeOfTag -= sizeof(cmsUInt8Number);

        if (SizeOfTag >= sizeof(cmsUInt16Number) + ScriptCodeCount) {
            if (!_cmsReadUInt16Number(io, &ScriptCodeCode)) goto Done;
            SizeOfTag -= sizeof(cmsUInt16Number);

            for (i = 0; i < ScriptCodeCount; i++) {
                if (!_cmsReadUInt8Number(io, &Dummy)) goto Done;
                SizeOfTag--;
            }
        }
    }

Done:
    *nItems = 1;
    return mlu;
