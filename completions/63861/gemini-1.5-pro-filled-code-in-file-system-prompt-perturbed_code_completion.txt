if (unicodeLength > 0) {

        if (SizeOfTag < unicodeLength * sizeof(cmsUInt16Number)) goto Done;
        SizeOfTag -= unicodeLength * sizeof(cmsUInt16Number);

        UnicodeString = (wchar_t*) _cmsMalloc(self ->ContextID, unicodeLength * sizeof(wchar_t));
        if (UnicodeString == NULL) goto Error;

        if (!_cmsReadWCharArray(io, unicodeLength, UnicodeString)) goto Error;

        if (!cmsMLUsetWide(mlu, cmsV2Unicode, cmsV2Unicode, UnicodeString)) goto Error;
        _cmsFree(self ->ContextID, UnicodeString);
        UnicodeString = NULL;
    }

Done:
    // If ScriptCodeCount is not zero, read and discard the ScriptCode elements.
    if (SizeOfTag >= 2) {
        if (!_cmsReadUInt16Number(io, &ScriptCodeCode)) goto Done;
        SizeOfTag -= sizeof(cmsUInt16Number);
    }

    if (SizeOfTag >= 1) {
        if (!_cmsReadUInt8Number(io, &ScriptCodeCount)) goto Done;
        SizeOfTag -= sizeof(cmsUInt8Number);
    }

    if (SizeOfTag >= ScriptCodeCount * 67) {
        for (i = 0; i < ScriptCodeCount; i++) {
            if (!_cmsReadUInt16Number(io, &Dummy)) goto Done;
            if (SizeOfTag >= 67) {
                if (io->Read(io, NULL, 67, 1) != 1) goto Done;
                SizeOfTag -= 67;
            }
        }
    }

    *nItems = 1;
    return (void*)mlu;

Error:
    if (UnicodeString != NULL) _cmsFree(self ->ContextID, UnicodeString);
