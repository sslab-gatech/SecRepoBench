// Read and handle the Unicode string component of a text description.
if (SizeOfTag >= 2 * sizeof(cmsUInt32Number)) {
    if (_cmsReadUInt32Number(io, &UnicodeCode) && _cmsReadUInt32Number(io, &unicodeLength)) {
        SizeOfTag -= 2 * sizeof(cmsUInt32Number);
        if (SizeOfTag >= unicodeLength * sizeof(cmsUInt16Number)) {
            UnicodeString = (wchar_t*)_cmsMalloc(self->ContextID, (unicodeLength + 1) * sizeof(wchar_t));
            if (UnicodeString) {
                if (io->Read(io, UnicodeString, sizeof(cmsUInt16Number), unicodeLength) == unicodeLength) {
                    UnicodeString[unicodeLength] = 0;
                    if (!cmsMLUsetWide(mlu, cmsNoLanguage, cmsNoCountry, UnicodeString)) {
                        goto Error;
                    }
                    SizeOfTag -= unicodeLength * sizeof(cmsUInt16Number);
                    if (SizeOfTag >= 2 * sizeof(cmsUInt16Number)) {
                        if (_cmsReadUInt16Number(io, &ScriptCodeCode) && _cmsReadUInt16Number(io, &Dummy)) {
                            SizeOfTag -= 2 * sizeof(cmsUInt16Number);
                            ScriptCodeCount = 1;
                        }
                    }
                }
                _cmsFree(self->ContextID, UnicodeString);
            }
        }
    }
}
*nItems = 1;
Done:
return mlu;
Error:
if (Text) _cmsFree(self->ContextID, (void*)Text);
if (mlu) cmsMLUfree(mlu);
return NULL;
