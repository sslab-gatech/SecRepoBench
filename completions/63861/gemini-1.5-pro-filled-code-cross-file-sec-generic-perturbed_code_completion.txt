if (unicodeLength > 0 && SizeOfTag >= unicodeLength * sizeof(cmsUInt16Number)) {

        UnicodeString = (wchar_t*) _cmsMalloc(self ->ContextID, unicodeLength * sizeof(wchar_t));
        if (UnicodeString == NULL) goto Error;

        if (io ->Read(io, UnicodeString, sizeof(cmsUInt16Number), unicodeLength) != unicodeLength) goto Error;
        SizeOfTag -= unicodeLength * sizeof(cmsUInt16Number);

        if (!cmsMLUsetWide(mlu, cmsNoLanguage, cmsNoCountry, UnicodeString)) goto Error;

        _cmsFree(self ->ContextID, (void*) UnicodeString);
        UnicodeString = NULL;
    }

Done:
    // Read optional ScriptCode elements
    ScriptCodeCount = 0;
    if (SizeOfTag >= sizeof(cmsUInt8Number)) {
        if (io->Read(io, &ScriptCodeCount, sizeof(cmsUInt8Number), 1) != 1) goto Error;
        SizeOfTag -= sizeof(cmsUInt8Number);
    }

    for (i = 0; i < ScriptCodeCount; i++) {
        if (SizeOfTag >= sizeof(cmsUInt16Number) * 2) {
            if (!_cmsReadUInt16Number(io, &ScriptCodeCode)) goto Error;
            if (!_cmsReadUInt16Number(io, &Dummy)) goto Error;  // Skip the dummy value
            SizeOfTag -= sizeof(cmsUInt16Number) * 2;
        } else {
            goto Error; // Not enough data for ScriptCode element
        }
    }


    *nItems = 1;
    return mlu;
