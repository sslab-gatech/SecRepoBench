// Read the Unicode code
    if (SizeOfTag < 2 * sizeof(cmsUInt32Number)) goto Done;
    if (!_cmsReadUInt32Number(io, &UnicodeCode)) goto Done;
    if (!_cmsReadUInt32Number(io, &unicodeLength)) goto Done;
    SizeOfTag -= 2 * sizeof(cmsUInt32Number);

    // Read the Unicode string
    if (SizeOfTag < 2 * unicodeLength) goto Done;
    UnicodeString = (wchar_t*) _cmsMalloc(self->ContextID, (unicodeLength + 1) * sizeof(wchar_t));
    if (UnicodeString == NULL) goto Done;

    if (!_cmsReadWCharArray(io, unicodeLength, UnicodeString)) goto Done;

    // Set the Unicode string in the MLU
    if (!cmsMLUsetWide(mlu, cmsV2Unicode, cmsV2Unicode, UnicodeString, unicodeLength)) goto Done;

    // Free the allocated Unicode string
    _cmsFree(self->ContextID, UnicodeString);

Done:
    // Adjust the remaining size of the tag data
    SizeOfTag -= 2 * unicodeLength;
    return TRUE;
