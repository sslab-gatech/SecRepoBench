if (unicodeLength > 0 && SizeOfTag >= unicodeLength * sizeof(cmsUInt16Number)) {

        UnicodeString = (wchar_t*) _cmsMalloc(self ->ContextID, (unicodeLength + 1) * sizeof(wchar_t));
        if (UnicodeString == NULL) goto Error;

        if (io ->Read(io, UnicodeString, sizeof(cmsUInt16Number), unicodeLength) != unicodeLength) goto Error;
        SizeOfTag -= unicodeLength * sizeof(cmsUInt16Number);

        UnicodeString[unicodeLength] = 0; // Null-terminate the string

        if (!cmsMLUsetWide(mlu, cmsNoLanguage, cmsNoCountry, UnicodeString)) goto Error;
        _cmsFree(self ->ContextID, (void*) UnicodeString);
        UnicodeString = NULL;

       // Read in ScriptCode if any
        if (SizeOfTag >= 2) {
            if (!_cmsReadUInt16Number(io, &ScriptCodeCode)) goto Done;
            if (!_cmsReadUInt8Number(io, &ScriptCodeCount)) goto Done;
            if (!_cmsReadUInt8Number(io, &Dummy)) goto Done; // Padding
            SizeOfTag -= 4;

            if (SizeOfTag >= ScriptCodeCount * 2) {

                // We don't support ScriptCode yet, so just skip it.
                if (io ->Seek(io, ScriptCodeCount * 2, cmsIO_SEEK_CUR) != 0) goto Done;
                SizeOfTag -= ScriptCodeCount * 2;
            }
        }
    }

Done:
    *nItems = 1;
    return mlu;

Error:
