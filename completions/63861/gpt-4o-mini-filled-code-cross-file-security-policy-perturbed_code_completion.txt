if (unicodeLength == 0 || SizeOfTag < unicodeLength * sizeof(wchar_t)) goto Done;

UnicodeString = (wchar_t*) _cmsMalloc(self->ContextID, (unicodeLength + 1) * sizeof(wchar_t));
if (UnicodeString == NULL) goto Error;

if (io->Read(io, UnicodeString, sizeof(wchar_t), unicodeLength) != unicodeLength) goto Error;
UnicodeString[unicodeLength] = 0; // Null terminator

if (!cmsMLUsetUnicode(mlu, cmsNoLanguage, cmsNoCountry, UnicodeString)) goto Error;
_cmsFree(self->ContextID, (void*) UnicodeString);
UnicodeString = NULL;

SizeOfTag -= unicodeLength * sizeof(wchar_t);

// Optional ScriptCode handling
if (SizeOfTag >= sizeof(cmsUInt8Number)) {
    if (io->Read(io, &ScriptCodeCount, sizeof(cmsUInt8Number), 1) != 1) goto Done;
    SizeOfTag -= sizeof(cmsUInt8Number);
    
    for (i = 0; i < ScriptCodeCount; i++) {
        if (SizeOfTag < sizeof(cmsUInt16Number)) goto Done;
        if (io->Read(io, &ScriptCodeCode, sizeof(cmsUInt16Number), 1) != 1) goto Done;
        SizeOfTag -= sizeof(cmsUInt16Number);
    }
}
