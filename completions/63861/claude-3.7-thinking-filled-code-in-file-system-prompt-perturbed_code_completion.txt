// Read and handle the Unicode string component of a text description. 
// Allocate memory to store the Unicode string and read the characters from the IO handler. 
// Convert the Unicode string to a wide character array and add it to the MLU (Multi-Language Unicode) structure. 
// After processing the Unicode string, read the ScriptCode code, count, and description (67 bytes). 
// Return the populated MLU object and handle any potential errors, ensuring allocated memory is freed if an error occurs.

// Allocate memory for Unicode string
UnicodeString = (wchar_t*) _cmsMallocZero(self ->ContextID, (unicodeLength + 1) * sizeof(wchar_t));
if (UnicodeString == NULL) goto Error;

// Read Unicode string
if (!_cmsReadWCharArray(io, unicodeLength, UnicodeString)) goto Error;

// Add Unicode string to MLU
if (unicodeLength > 0) {
    if (!cmsMLUsetWide(mlu, cmsV2Unicode, cmsV2Unicode, UnicodeString)) goto Error;
}

_cmsFree(self ->ContextID, UnicodeString);
UnicodeString = NULL;

// Read ScriptCode section
cmsUInt16Number ScriptCodeCode;
cmsUInt8Number ScriptCodeCount;

if (!_cmsReadUInt16Number(io, &ScriptCodeCode)) goto Error;
if (!_cmsReadUInt8Number(io, &ScriptCodeCount)) goto Error;

// Read ScriptCode description (always 67 bytes)
char ScriptCodeDesc[67];
if (io ->Read(io, ScriptCodeDesc, 1, 67) != 67) goto Error;

*nItems = 1;
return (void*) mlu;
