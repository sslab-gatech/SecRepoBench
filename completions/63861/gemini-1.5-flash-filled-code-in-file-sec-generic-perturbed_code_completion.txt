if (unicodeLength > 0) {

        UnicodeString = (wchar_t*) _cmsCalloc(self ->ContextID, unicodeLength, sizeof(wchar_t));
        if (UnicodeString == NULL) goto Error;

        if (!_cmsReadWCharArray(io, unicodeLength, UnicodeString)) goto Error;
        SizeOfTag -= unicodeLength * sizeof(cmsUInt16Number);

        if (!cmsMLUsetWide(mlu, cmsNoLanguage, cmsNoCountry, UnicodeString)) goto Error;
        _cmsFree(self ->ContextID, (void*) UnicodeString);
        UnicodeString = NULL;
    }

Done:
    if (SizeOfTag > 0) {

        // Skip ScriptCode data if present
        if (!_cmsReadUInt16Number(io, &ScriptCodeCode)) goto Error;
        if (!_cmsReadUInt8Number(io, &ScriptCodeCount)) goto Error;
        if (SizeOfTag < ScriptCodeCount) goto Error;
        if (io ->Read(io, NULL, sizeof(cmsUInt8Number), ScriptCodeCount) != ScriptCodeCount) goto Error;
    }

    *nItems = 1;
    return (void*) mlu;
