// Allocate memory to store the Unicode string
UnicodeString = (wchar_t*) _cmsMalloc(self ->ContextID, unicodeLength * sizeof(wchar_t));
if (UnicodeString == NULL) goto Error;

// Read the Unicode string from the IO handler
if (io ->Read(io, UnicodeString, sizeof(wchar_t), unicodeLength)!= unicodeLength) goto Error;
SizeOfTag -= unicodeLength * sizeof(wchar_t);

// Convert the Unicode string to a wide character array and add it to the MLU structure
if (!cmsMLUsetWide(mlu, cmsNoLanguage, cmsNoCountry, UnicodeString)) goto Error;
_cmsFree(self ->ContextID, (void*) UnicodeString);
UnicodeString = NULL;

// Check for the presence of optional ScriptCode elements
if (SizeOfTag >= sizeof(cmsUInt8Number)) {
    if (!_cmsReadUInt8Number(io, &ScriptCodeCount)) goto Done;
    SizeOfTag -= sizeof(cmsUInt8Number);

    // Skip over the ScriptCode data by reading and discarding the necessary bytes
    for (i = 0; i < ScriptCodeCount; i++) {
        if (SizeOfTag < 2 * sizeof(cmsUInt16Number)) goto Done;
        if (!_cmsReadUInt16Number(io, &ScriptCodeCode)) goto Done;
        if (!_cmsReadUInt16Number(io, &Dummy)) goto Done;
        SizeOfTag -= 2 * sizeof(cmsUInt16Number);
    }
}

// Return the populated MLU object
Done:
if (mlu) return (void*) mlu;
Error:
if (Text) _cmsFree(self ->ContextID, (void*) Text);
if (UnicodeString) _cmsFree(self ->ContextID, (void*) UnicodeString);
if (mlu) cmsMLUfree(mlu);
return NULL;
