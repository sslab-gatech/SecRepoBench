// Read and handle the Unicode string component of a text description.
UnicodeString = (wchar_t*) _cmsMalloc(self->ContextID, (unicodeLength + 1) * sizeof(wchar_t));
if (UnicodeString == NULL) goto Error;

if (io->Read(io, UnicodeString, sizeof(wchar_t), unicodeLength) != unicodeLength) {
    _cmsFree(self->ContextID, UnicodeString);
    goto Error;
}
UnicodeString[unicodeLength] = 0;

if (!cmsMLUsetWide(mlu, cmsNoLanguage, cmsNoCountry, UnicodeString)) {
    _cmsFree(self->ContextID, UnicodeString);
    goto Error;
}

_cmsFree(self->ContextID, UnicodeString);
UnicodeString = NULL;

// Check for optional ScriptCode elements
if (SizeOfTag >= 2 * sizeof(cmsUInt16Number)) {
    if (!_cmsReadUInt16Number(io, &ScriptCodeCode)) goto Done;
    if (!_cmsReadUInt16Number(io, &Dummy)) goto Done;
    SizeOfTag -= 2 * sizeof(cmsUInt16Number);

    if (SizeOfTag >= sizeof(cmsUInt8Number)) {
        if (!_cmsReadUInt8Number(io, &ScriptCodeCount)) goto Done;
        SizeOfTag -= sizeof(cmsUInt8Number);

        // Skip over the ScriptCode data
        if (SizeOfTag < ScriptCodeCount * sizeof(cmsUInt16Number)) goto Done;
        SizeOfTag -= ScriptCodeCount * sizeof(cmsUInt16Number);
    }
}

*nItems = 1;
return mlu;

Error:
if (Text) _cmsFree(self->ContextID, (void*)Text);
if (UnicodeString) _cmsFree(self->ContextID, UnicodeString);
if (mlu) cmsMLUfree(mlu);
return NULL;

Done:
*nItems = 1;
return mlu;
