// Check if there is enough data for the Unicode string
    if (SizeOfTag < unicodeLength * sizeof(cmsUInt16Number)) goto Done;

    // Allocate memory for the Unicode string
    UnicodeString = (wchar_t*)_cmsMalloc(self->ContextID, (unicodeLength + 1) * sizeof(wchar_t));
    if (UnicodeString == NULL) goto Done;

    // Read the Unicode string
    if (io->Read(io, UnicodeString, sizeof(cmsUInt16Number), unicodeLength) != unicodeLength) goto Done;

    // Null-terminate the Unicode string
    UnicodeString[unicodeLength] = 0;

    // Add the Unicode string to the MLU
    if (!cmsMLUsetWide(mlu, cmsNoLanguage, cmsNoCountry, UnicodeString)) goto Done;

    // Free the Unicode string memory
    _cmsFree(self->ContextID, (void*)UnicodeString);
    UnicodeString = NULL;

    // Adjust the remaining size
    SizeOfTag -= unicodeLength * sizeof(cmsUInt16Number);

    // Check for optional ScriptCode elements
    if (SizeOfTag < sizeof(cmsUInt8Number)) goto Done;
    if (!io->Read(io, &ScriptCodeCount, sizeof(cmsUInt8Number), 1)) goto Done;
    SizeOfTag -= sizeof(cmsUInt8Number);

    // Skip over ScriptCode elements if present
    for (i = 0; i < ScriptCodeCount; i++) {
        if (SizeOfTag < 2 * sizeof(cmsUInt16Number)) goto Done;
        if (!io->Read(io, &ScriptCodeCode, sizeof(cmsUInt16Number), 1)) goto Done;
        if (!io->Read(io, &Dummy, sizeof(cmsUInt16Number), 1)) goto Done;
        SizeOfTag -= 2 * sizeof(cmsUInt16Number);
    }

Done:
    if (UnicodeString) _cmsFree(self->ContextID, (void*)UnicodeString);
    return (void*)mlu;

Error:
    if (Text) _cmsFree(self->ContextID, (void*)Text);
    if (UnicodeString) _cmsFree(self->ContextID, (void*)UnicodeString);
    if (mlu) cmsMLUfree(mlu);
    return NULL;
