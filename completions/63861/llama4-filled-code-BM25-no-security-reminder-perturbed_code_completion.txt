// Read and handle the Unicode string component of a text description. 
// Allocate memory to store the Unicode string and read the characters from the IO handler. 
// Check if there is enough data left to process the Unicode string and allocate a buffer for it. 
// Convert the Unicode string to a wide character array and add it to the MLU (Multi-Language Unicode) structure. 
// After processing the Unicode string, adjust the remaining size of the tag data and check for the presence of optional ScriptCode elements. 
// If present, skip over the ScriptCode data by reading and discarding the necessary bytes. 
// Return the populated MLU object and handle any potential errors, ensuring allocated memory is freed if an error occurs.

    if (unicodeLength > 0) {
        UnicodeString = (wchar_t*) _cmsMalloc(self->ContextID, (unicodeLength * sizeof(wchar_t)));
        if (UnicodeString == NULL) goto Error;

        if (io->Read(io, UnicodeString, sizeof(wchar_t), unicodeLength) != unicodeLength) goto Error;
        SizeOfTag -= unicodeLength * sizeof(wchar_t);

        if (!cmsMLUsetWCS(mlu, cmsNoLanguage, cmsNoCountry, UnicodeString)) goto Error;
        _cmsFree(self->ContextID, (void*) UnicodeString);
        UnicodeString = NULL;
    }

    // Check for ScriptCodeCount
    if (SizeOfTag < sizeof(cmsUInt8Number)) goto Done;
    if (io->Read(io, &ScriptCodeCount, sizeof(cmsUInt8Number), 1) != 1) goto Done;
    SizeOfTag -= sizeof(cmsUInt8Number);

    // If there are ScriptCodes, skip them
    if (ScriptCodeCount > 0) {
        if (SizeOfTag < (ScriptCodeCount * (sizeof(cmsUInt16Number) + sizeof(cmsUInt16Number)))) goto Done;
        for (i = 0; i < ScriptCodeCount; i++) {
            if (io->Read(io, &ScriptCodeCode, sizeof(cmsUInt16Number), 1) != 1) goto Done;
            if (io->Read(io, &Dummy, sizeof(cmsUInt16Number), 1) != 1) goto Done;
        }
        SizeOfTag -= ScriptCodeCount * (sizeof(cmsUInt16Number) + sizeof(cmsUInt16Number));
    }

Done:
    *nItems = 1;
    return (void*) mlu;

Error:
    if (Text) _cmsFree(self->ContextID, (void*) Text);
    if (UnicodeString) _cmsFree(self->ContextID, (void*) UnicodeString);
    if (mlu) cmsMLUfree(mlu);
    return NULL;
