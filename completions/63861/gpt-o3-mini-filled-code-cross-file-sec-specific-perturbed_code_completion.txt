if (unicodeLength > SizeOfTag / sizeof(cmsUInt16Number))
        goto Done;

    UnicodeString = (wchar_t*) _cmsMalloc(self->ContextID, (unicodeLength + 1) * sizeof(wchar_t));
    if (UnicodeString == NULL) goto Error;

    if (io->Read(io, UnicodeString, sizeof(cmsUInt16Number), unicodeLength) != unicodeLength)
        goto Error;

    UnicodeString[unicodeLength] = 0;

    if (!cmsMLUsetWChar(mlu, cmsNoLanguage, cmsNoCountry, UnicodeString))
        goto Error;

    _cmsFree(self->ContextID, UnicodeString);
    UnicodeString = NULL;

    SizeOfTag -= unicodeLength * sizeof(cmsUInt16Number);

    if (SizeOfTag >= (2 * sizeof(cmsUInt16Number) + sizeof(cmsUInt8Number))) {

        if (!_cmsReadUInt16Number(io, &ScriptCodeCode))
            goto Done;
        if (!_cmsReadUInt16Number(io, &Dummy))
            goto Done;
        if (io->Read(io, &ScriptCodeCount, sizeof(cmsUInt8Number), 1) != 1)
            goto Done;

        SizeOfTag -= (2 * sizeof(cmsUInt16Number) + sizeof(cmsUInt8Number));

        if (ScriptCodeCount > SizeOfTag)
            goto Done;

        {
            char* dummyBuffer = (char*) _cmsMalloc(self->ContextID, ScriptCodeCount);
            if (dummyBuffer == NULL)
                goto Error;
            if (io->Read(io, dummyBuffer, sizeof(cmsUInt8Number), ScriptCodeCount) != ScriptCodeCount) {
                _cmsFree(self->ContextID, dummyBuffer);
                goto Error;
            }
            _cmsFree(self->ContextID, dummyBuffer);
            SizeOfTag -= ScriptCodeCount;
        }
    }

    *nItems = 1;
    return mlu;
