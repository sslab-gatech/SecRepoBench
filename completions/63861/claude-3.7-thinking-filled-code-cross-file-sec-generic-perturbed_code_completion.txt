// Check if there's enough data left in the tag for the Unicode string
if (unicodeLength > (SizeOfTag / sizeof(cmsUInt16Number))) goto Done;

// Allocate memory for the Unicode string (with overflow protection)
if (unicodeLength >= (SIZE_MAX / sizeof(wchar_t)) - 1) goto Done;
UnicodeString = (wchar_t*) _cmsMalloc(self->ContextID, (unicodeLength + 1) * sizeof(wchar_t));
if (UnicodeString == NULL) goto Error;

// Read the Unicode string from the IO handler
for (i = 0; i < unicodeLength; i++) {
    cmsUInt16Number ucChar;
    if (!_cmsReadUInt16Number(io, &ucChar)) goto Error;
    UnicodeString[i] = (wchar_t) ucChar;
}
UnicodeString[unicodeLength] = 0;  // Null-terminate

// Add the Unicode string to the MLU
if (!cmsMLUsetWide(mlu, cmsNoLanguage, cmsNoCountry, UnicodeString)) goto Error;

// Update SizeOfTag
SizeOfTag -= unicodeLength * sizeof(cmsUInt16Number);

// Free the Unicode string as it's no longer needed
_cmsFree(self->ContextID, UnicodeString);
UnicodeString = NULL;

// Handle ScriptCode data if present
if (SizeOfTag < sizeof(cmsUInt8Number)) goto Done;
if (!io->Read(io, &ScriptCodeCount, sizeof(cmsUInt8Number), 1)) goto Done;
SizeOfTag -= sizeof(cmsUInt8Number);

// Read and skip over each ScriptCode entry
for (i = 0; i < ScriptCodeCount; i++) {
    if (SizeOfTag < 2 * sizeof(cmsUInt16Number)) goto Done;
    if (!_cmsReadUInt16Number(io, &ScriptCodeCode)) goto Done;
    if (!_cmsReadUInt16Number(io, &Dummy)) goto Done;
    SizeOfTag -= 2 * sizeof(cmsUInt16Number);
}

// Successfully read the text description
*nItems = 1;
return mlu;

Done:
    *nItems = 1;
    return mlu;
