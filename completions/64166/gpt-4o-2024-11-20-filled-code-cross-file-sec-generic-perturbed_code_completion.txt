if (DeviceLink == NULL) {
        cmsDeleteTransform(xform);
        cmsSignalError(m->ContextID, cmsERROR_COLORSPACE_CHECK, "Cannot duplicate DeviceLink in CRD creation");
        return 0;
    }

    // Force the use of a CLUT
    dwFlags |= cmsFLAGS_FORCE_CLUT;

    // Optimize the pipeline for color rendering
    if (!_cmsOptimizePipeline(DeviceLink, Intent, dwFlags, m->ContextID)) {
        cmsPipelineFree(DeviceLink);
        cmsDeleteTransform(xform);
        cmsSignalError(m->ContextID, cmsERROR_COLORSPACE_CHECK, "Pipeline optimization failed in CRD creation");
        return 0;
    }
