if (DeviceLink == NULL) {
        cmsDeleteTransform(xform);
        cmsSignalError(m ->ContextID, cmsERROR_COLORSPACE_CHECK, "Cannot duplicate device link");
        return 0;
    }

    // Force the use of a CLUT
    DeviceLink ->Flags |= cmsFLAGS_FORCE_CLUT;

    // Optimize the pipeline for color rendering
    if (!cmsOptimizePipeline(DeviceLink)) {
        cmsPipelineFree(DeviceLink);
        cmsDeleteTransform(xform);
        cmsSignalError(m ->ContextID, cmsERROR_COLORSPACE_CHECK, "Failed to optimize pipeline");
        return 0;
    }
