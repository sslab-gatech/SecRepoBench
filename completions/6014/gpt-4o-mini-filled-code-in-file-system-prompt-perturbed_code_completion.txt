// Iterate over the normalized string to remove leading spaces from each word.
      char *start = p;
      char *end = p;
      while (*end != '\0') {
        // Skip leading spaces
        while (*end == ' ') {
          end++;
        }
        // Copy non-space characters
        if (*end != '\0') {
          if (start != p) {
            *start++ = ' '; // Add a space before the next word
          }
          while (*end != '\0' && *end != ' ') {
            *start++ = *end++;
          }
        }
      }
      *start = '\0'; // Null-terminate the compacted string
