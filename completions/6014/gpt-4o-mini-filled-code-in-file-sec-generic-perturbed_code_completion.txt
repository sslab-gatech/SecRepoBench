// Iterate over the normalized string to remove leading spaces from each word.
      char *start = p; // Pointer to the start of the current word.
      char *end = p;   // Pointer to the end of the current word.
      while (*end != '\0') {
          // Skip leading spaces.
          while (*end == ' ') {
              end++;
          }
          // If we found a non-space character, copy it to the start pointer.
          if (*end != '\0') {
              if (start != p) {
                  *start++ = ' '; // Add a space before the new word if it's not the first word.
              }
              while (*end != '\0' && *end != ' ') {
                  *start++ = *end++; // Copy the word.
              }
          }
      }
      *start = '\0'; // Null-terminate the compacted string.
