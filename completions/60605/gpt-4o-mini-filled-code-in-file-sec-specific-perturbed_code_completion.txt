} else {
	    // Handle the case where the filename is enclosed in quotes
	    u_int8_t end_quote_index = attachment_len + attachment_filename_len - 1;
	    if (packet->content_disposition_line.ptr[end_quote_index] == '\"') {
	      attachment_filename_len -= 2; // Exclude the quotes
	      flow->http.filename = ndpi_malloc(attachment_filename_len + 1);
	      if (flow->http.filename != NULL) {
	        strncpy(flow->http.filename, (char*)packet->content_disposition_line.ptr + attachment_len + 1, attachment_filename_len);
	        flow->http.filename[attachment_filename_len] = '\0';
	      }
	    } else {
	      // Handle the case where the filename is not enclosed in quotes
	      flow->http.filename = ndpi_malloc(attachment_filename_len + 1);
	      if (flow->http.filename != NULL) {
	        strncpy(flow->http.filename, (char*)packet->content_disposition_line.ptr + attachment_len, attachment_filename_len);
	        flow->http.filename[attachment_filename_len] = '\0';
	      }
