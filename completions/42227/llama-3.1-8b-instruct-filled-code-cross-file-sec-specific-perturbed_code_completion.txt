if (it8_parser->ch == '\'' || it8_parser->ch == '\"')
{
    // Allocate a buffer to store the string
    cmsUInt32Number str_len = 0;
    cmsChar8* str_buf = NULL;

    // Read each character until we reach the closing quote
    while (it8_parser->ch!= '\'' && it8_parser->ch!= '\"')
    {
        // Check for newline or carriage return
        if (it8_parser->ch == '\n' || it8_parser->ch == '\r')
        {
            SynError(it8_parser, "Unexpected newline or carriage return in string");
            break;
        }

        // Check for buffer overflow
        if (str_len >= CMS_MAX_STRING_LENGTH)
        {
            SynError(it8_parser, "String too long");
            break;
        }

        // Append the character to the buffer
        str_buf = realloc(str_buf, (str_len + 1) * sizeof(cmsChar8));
        str_buf[str_len++] = it8_parser->ch;
        NextCh(it8_parser);
    }

    // Check if we reached the closing quote
    if (it8_parser->ch!= '\'' && it8_parser->ch!= '\"')
    {
        SynError(it8_parser, "Missing closing quote in string");
    }
    else
    {
        // Null-terminate the string
        str_buf[str_len] = '\0';

        // Set the current symbol to SSTRING
        it8_parser->sy = SSTRING;

        // Set the string value
        it8_parser->str = str_buf;

        // Move to the next character
        NextCh(it8_parser);
    }
}
