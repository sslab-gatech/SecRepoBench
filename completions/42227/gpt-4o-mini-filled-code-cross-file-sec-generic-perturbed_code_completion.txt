        it8_parser->str[0] = '\0'; // Clear the current string buffer

        while (it8_parser->ch != quote && it8_parser->ch != '\n' && it8_parser->ch != '\r' && it8_parser->ch != SEOF) {
            // Append the character to the string buffer
            strncat(it8_parser->str, &it8_parser->ch, 1);
            NextCh(it8_parser); // Move to the next character
        }

        if (it8_parser->ch == quote) {
            // Successfully parsed the quoted string
            it8_parser->sy = SSTRING; // Set the current symbol to SSTRING
            NextCh(it8_parser); // Move past the closing quote
        } else {
