        while ((it8_parser->ch != quote) &&

            if (idx < (sizeof(it8_parser->str) - 1))

                it8_parser->str[idx++] = it8_parser->ch;
                it8_parser->str[idx]   = '\0';

            NextCh(it8_parser);
        }
