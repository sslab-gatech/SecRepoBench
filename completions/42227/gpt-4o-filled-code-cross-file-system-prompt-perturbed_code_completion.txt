        it8_parser->str[0] = '\0';
        int str_index = 0;

        while (it8_parser->ch != quote && it8_parser->ch != '\n' && it8_parser->ch != '\r' && it8_parser->ch != '\0') {

                it8_parser->str[str_index++] = it8_parser->ch;
            }
            NextCh(it8_parser);
