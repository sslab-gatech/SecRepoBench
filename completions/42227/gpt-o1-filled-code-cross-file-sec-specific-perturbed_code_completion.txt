{
    char quote = it8_parser->ch;
    size_t idx = 0;

    // Clear the buffer
    it8_parser->str[0] = '\0';

    // Move past the initial quote
    NextCh(it8_parser);

    // Read until matching quote or newline/carriage return
    while (it8_parser->ch != quote)
    {
        if (it8_parser->ch == '\n' || it8_parser->ch == '\r' || it8_parser->ch == 0)
        {
            // String not properly closed, end here
            return;
        }

        // Safely append character to buffer if space remains
        if (idx < (sizeof(it8_parser->str) - 1))
        {
            it8_parser->str[idx++] = it8_parser->ch;
            it8_parser->str[idx] = '\0';
        }

        NextCh(it8_parser);
    }

    // Move past the closing quote
    NextCh(it8_parser);

    // String successfully parsed
    it8_parser->sy = SSTRING;
}
