char quote = it8_parser->ch;
NextCh(it8_parser); // Move past the opening quote
ClearStringBuffer(it8_parser); // Clear the current string buffer

while (it8_parser->ch != quote && it8_parser->ch != '\n' && it8_parser->ch != '\r' && !EndOfInput(it8_parser))
{
    AppendCharToStringBuffer(it8_parser, it8_parser->ch);
    NextCh(it8_parser);
}

if (it8_parser->ch == quote)
{
    NextCh(it8_parser); // Move past the closing quote
    it8_parser->sy = SSTRING; // Set the current symbol to SSTRING
}
else
{
    SynError(it8_parser, "Unterminated string");
}
