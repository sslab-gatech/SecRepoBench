char quote = it8_parser->ch;
NextCh(it8_parser);

ClearStringBuffer(it8_parser->str);

while (it8_parser->ch != quote && it8_parser->ch != '\n' && it8_parser->ch != '\r' && it8_parser->ch != SEOF)
{
    AppendCharToStringBuffer(it8_parser->str, it8_parser->ch);
    NextCh(it8_parser);
}

if (it8_parser->ch == quote)
{
    NextCh(it8_parser);
    it8_parser->sy = SSTRING;
}
else
{
    SynError(it8_parser, "Unterminated string");
}
