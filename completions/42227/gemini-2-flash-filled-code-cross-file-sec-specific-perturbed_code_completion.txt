char quote = it8_parser->ch;
        NextCh(it8_parser);

        // Clear the string buffer
        it8_parser->str[0] = 0;
        it8_parser->nStr = 0;

        while (it8_parser->ch != 0 && it8_parser->ch != '\n' && it8_parser->ch != '\r' && it8_parser->ch != quote)
        {
            if (it8_parser->nStr < MAX_IT8_STRING - 1)
            {
                it8_parser->str[it8_parser->nStr++] = it8_parser->ch;
                it8_parser->str[it8_parser->nStr] = 0; // Null-terminate the string
                NextCh(it8_parser);
            }
            else
            {
                SynError(it8_parser, "String too long");
                it8_parser->sy = SSYNERROR;
                return;
            }
        }

        if (it8_parser->ch == quote)
        {
            it8_parser->sy = SSTRING;
            NextCh(it8_parser);
        }
        else
        {
            SynError(it8_parser, "Unterminated string");
            it8_parser->sy = SSYNERROR;
        }
