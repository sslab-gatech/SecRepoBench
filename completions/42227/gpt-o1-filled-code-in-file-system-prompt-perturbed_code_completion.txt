{
    int quote = it8_parser->ch;
    StringClear(it8_parser->str);
    NextCh(it8_parser);

    while (it8_parser->ch != quote &&
           it8_parser->ch != 0 &&
           it8_parser->ch != '\n' &&
           it8_parser->ch != '\r')
    {
        StringAppend(it8_parser->str, (char) it8_parser->ch);
        NextCh(it8_parser);
    }

    if (it8_parser->ch != quote)
    {
        SynError(it8_parser, "Unterminated string");
        return;
    }

    NextCh(it8_parser);
    it8_parser->sy = SSTRING;
}
