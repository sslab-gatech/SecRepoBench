char* strBuffer = NULL;
    size_t strBufferSize = 0;
    char quoteChar = it8_parser->ch;

    // Allocate initial buffer for the string
    strBuffer = (char*)malloc(1);
    if (strBuffer == NULL) {
        SynError(it8_parser, "Memory allocation failed");
        return;
    }
    strBuffer[0] = '\0';

    // Read characters until the closing quote or newline
    while (it8_parser->ch != quoteChar && it8_parser->ch != '\n' && it8_parser->ch != '\r') {
        size_t newBufferSize = strBufferSize + 1;
        char* newBuffer = (char*)realloc(strBuffer, newBufferSize + 1);
        if (newBuffer == NULL) {
            free(strBuffer);
            SynError(it8_parser, "Memory allocation failed");
            return;
        }
        strBuffer = newBuffer;

        strBuffer[strBufferSize] = it8_parser->ch;
        strBufferSize++;
        strBuffer[strBufferSize] = '\0';

        NextCh(it8_parser);
    }

    // Check if the closing quote was found
    if (it8_parser->ch != quoteChar) {
        free(strBuffer);
        SynError(it8_parser, "Unterminated string");
        return;
    }

    // Move to the next character after processing
    NextCh(it8_parser);

    // Set the current symbol to SSTRING
    it8_parser->sy = SSTRING;
    it8_parser->str = strBuffer;
    cmsIT8SetSheetType(it8_parser, strBuffer);
