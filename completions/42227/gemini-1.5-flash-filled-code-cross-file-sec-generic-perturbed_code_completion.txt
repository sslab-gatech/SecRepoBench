char* p = it8_parser->str;
        int n = 0;
        char c;

        *p = 0;

        while (TRUE) {
            NextCh(it8_parser);
            c = it8_parser->ch;

            if (c == '\n' || c == '\r' || c == it8_parser->quote) break;

            if (n >= MAX_IT8_STRING - 1) {
                SynError(it8_parser, "String too long");
                break;
            }

            p[n++] = c;
        }
