char* buffer = (char*)malloc(1024);
int len = 0;

while (it8_parser->ch != '\'' && it8_parser->ch != '\"' && it8_parser->ch != '\n' && it8_parser->ch != '\r') {
    if (len >= 1023) {
        // Buffer overflow detected, handle the error
        SynError(it8_parser, "String too long");
        return;
    }
    buffer[len++] = it8_parser->ch;
    NextCh(it8_parser);
}

if (it8_parser->ch == '\n' || it8_parser->ch == '\r') {
    // Newline encountered, end of string
    SynError(it8_parser, "Unterminated string");
    free(buffer);
    return;
}

// Null-terminate the string
buffer[len] = '\0';

// Set the current symbol to SSTRING and move to the next character
it8_parser->sy = SSTRING;
it8_parser->str = buffer;
NextCh(it8_parser);
