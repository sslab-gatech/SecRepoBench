int iQuote = it8_parser->ch;
it8_parser->str[0] = '\0';  // Clear string buffer
NextCh(it8_parser);

while (it8_parser->ch != iQuote && it8_parser->ch != '\n' && it8_parser->ch != '\r' && it8_parser->ch != '\0') {
    // Append character to string buffer
    int len = strlen(it8_parser->str);
    it8_parser->str[len] = it8_parser->ch;
    it8_parser->str[len + 1] = '\0';
    NextCh(it8_parser);
}

if (it8_parser->ch == iQuote) {
    NextCh(it8_parser);
    it8_parser->sy = SSTRING;
}
