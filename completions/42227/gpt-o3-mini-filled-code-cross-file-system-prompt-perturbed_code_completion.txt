{
    char quote = it8_parser->ch;
    int pos = 0;

    /* Clear the current string buffer */
    it8_parser->str[0] = '\0';

    /* Skip the starting quote character */
    NextCh(it8_parser);

    /* Read characters until the matching quote, newline or carriage return is encountered */
    while (it8_parser->ch != quote &&
           it8_parser->ch != '\n' &&
           it8_parser->ch != '\r' &&
           it8_parser->ch != '\0') {
        it8_parser->str[pos++] = it8_parser->ch;
        it8_parser->str[pos] = '\0';
        NextCh(it8_parser);
    }

    /* If the closing quote is found, skip it */
    if (it8_parser->ch == quote)
        NextCh(it8_parser);

    /* Set the symbol to SSTRING */
    it8_parser->sy = SSTRING;
}
