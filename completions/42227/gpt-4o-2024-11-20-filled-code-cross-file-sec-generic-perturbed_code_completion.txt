char quote = it8_parser->ch; // Store the opening quote character
        ClearString(it8_parser);    // Clear the current string buffer
        NextCh(it8_parser);         // Move to the next character

        while (it8_parser->ch != quote && it8_parser->ch != '\n' && it8_parser->ch != '\r' && it8_parser->ch != '\0') {
            AppendToString(it8_parser, it8_parser->ch); // Append the character to the string buffer
            NextCh(it8_parser);                        // Move to the next character
        }

        if (it8_parser->ch == quote) {
            it8_parser->sy = SSTRING; // Successfully parsed a quoted string
            NextCh(it8_parser);       // Move past the closing quote
        } else {
            SynError(it8_parser, "Unterminated string"); // Handle unterminated string error
        }
