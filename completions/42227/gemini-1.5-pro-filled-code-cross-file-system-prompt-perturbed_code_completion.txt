char quote = it8_parser->ch;
        cmsIT8StringClear(it8_parser->str);

        NextCh(it8_parser);

        while (it8_parser->ch != 0 && it8_parser->ch != quote) {

            if (it8_parser->ch == '\n' || it8_parser->ch == '\r')
                break;  // Stop at newline

            cmsIT8StringAppendChar(it8_parser->str, it8_parser->ch);
            NextCh(it8_parser);
        }

        if (it8_parser->ch == quote) {
            it8_parser->sy = SSTRING;
            NextCh(it8_parser);
        }
