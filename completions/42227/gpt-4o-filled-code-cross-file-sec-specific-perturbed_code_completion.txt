        NextCh(it8_parser);

        it8_parser->str[0] = '\0';
        size_t str_len = 0;
        size_t max_len = sizeof(it8_parser->str) - 1; // Reserve space for null terminator

        while (it8_parser->ch != quote && it8_parser->ch != '\n' && it8_parser->ch != '\r' && it8_parser->ch != '\0') {
            if (str_len < max_len) {
                it8_parser->str[str_len++] = it8_parser->ch;
                it8_parser->str[str_len] = '\0'; // Null-terminate the string
            }
            NextCh(it8_parser);
        }

        if (it8_parser->ch == quote) {
            NextCh(it8_parser);
            it8_parser->sy = SSTRING;
        } else {
            SynError(it8_parser, "Unterminated string");
