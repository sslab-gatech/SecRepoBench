char quote = it8_parser->ch;
        cmsIT8ClearString(it8_parser); // Start with a clean string buffer

        NextCh(it8_parser); // Move to the next character after the quote

        while (it8_parser->ch != 0 && it8_parser->ch != quote) {
            if (it8_parser->ch == '\n' || it8_parser->ch == '\r') {
                return; // Stop at newline
            }
            cmsIT8AppendStringChar(it8_parser, it8_parser->ch); // Append to buffer safely
            NextCh(it8_parser);
        }

        if (it8_parser->ch == quote) {
            it8_parser->sy = SSTRING;
            NextCh(it8_parser);
        }
