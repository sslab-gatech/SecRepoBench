handler = xmlFindCharEncodingHandler(encoding_guess);
if (handler != NULL) {
    if (xmlStrcasecmp(handler->name, BAD_CAST "UTF-8") != 0)
        xmlSwitchEncoding(ctxt, handler->encoding);
} else {
    htmlParseErrInt(ctxt, XML_ERR_UNSUPPORTED_ENCODING,
                    "Encoding %s is not supported\n", encoding_guess);
    xmlSwitchEncoding(ctxt, XML_CHAR_ENCODING_8859_1);
}
xmlFree(encoding_guess);
