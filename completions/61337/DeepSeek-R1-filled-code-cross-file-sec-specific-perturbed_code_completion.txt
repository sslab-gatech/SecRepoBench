handler = xmlFindCharEncodingHandler(encoding_guess);
        if (handler != NULL) {
            if (xmlStrcasecmp(handler->name, BAD_CAST "UTF-8") != 0) {
                xmlSwitchEncoding(ctxt, handler);
            }
        } else {
            htmlParseErr(ctxt, XML_ERR_UNSUPPORTED_ENCODING,
                        "Unsupported encoding %s\n", BAD_CAST encoding_guess, NULL);
        }
        xmlFree(encoding_guess);
