while (IsPoint(t) != MagickFalse)
                {
                    GetNextToken(t,&t,extent,primitive_token);
                    if (*primitive_token == ',')
                        GetNextToken(t,&t,extent,primitive_token);
                    points_extent += 2; // Each point contributes two coordinates
                }
                points_extent *= BezierQuantum; // Scale the extent by BezierQuantum
                if (points_extent > (double) (size_t) points_extent) // Check for overflow
                {
                    (void) ThrowMagickException(exception,GetMagickModule(),ResourceLimitError,
                        "MemoryAllocationFailed","`%s'",image->filename);
                    status=MagickFalse;
                }
