void FillMaskedRegion(Image *image, DrawInfo *draw_info, PrimitiveInfo *primitive_info,
    ExceptionInfo *exception)
{
    PointInfo
        start,
        end;

    start = primitive_info[0].point;
    end.x = primitive_info[1].point.x;
    end.y = primitive_info[2].point.y;

    // Set the fill color and style
    draw_info->fill = ClonePixel(image, draw_info->fill);
    draw_info->fill_rule = EvenOddFillRule;
    draw_info->fill_opacity = OpaqueAlpha;

    // Fill the masked region with the specified color
    FillRectangle(image, draw_info, &start, &end, exception);

    // Reset the fill color and style
    draw_info->fill = ClonePixel(image, draw_info->stroke);
    draw_info->fill_rule = NonZeroFillRule;
    draw_info->fill_opacity = draw_info->stroke_alpha;
}
