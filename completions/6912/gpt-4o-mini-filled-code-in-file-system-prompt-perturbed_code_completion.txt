while (IsPoint(t) != MagickFalse)
        {
          t=GetNextToken(t,&t,MagickPathExtent,token);
          if (*token == ',')
            t=GetNextToken(t,&t,MagickPathExtent,token);
          point.x=StringToDouble(token,&next_token);
          if (token == next_token)
            ThrowPointExpectedException(token,exception);
          t=GetNextToken(t,&t,MagickPathExtent,token);
          if (*token == ',')
            t=GetNextToken(t,&t,MagickPathExtent,token);
          point.y=StringToDouble(token,&next_token);
          if (token == next_token)
            ThrowPointExpectedException(token,exception);
          TracePoint(q,point);
          q+=q->coordinates;
        }
