{
  /* Calculate the extent of points needed for a path primitive. */
  points_extent = 1;
  s = primitive_token;
  while (*s != '\0')
  {
    /* Skip any whitespace, commas, etc. */
    while (isspace((int) (unsigned char) *s) || (*s == ','))
      s++;
    /* If next chunk looks like a number, parse it and increment the count. */
    if ((*s == '+') || (*s == '-') || (*s == '.') || (isdigit((int) (unsigned char) *s) != 0))
    {
      (void) StringToDouble(s,&t);
      if (t == s)
        break;
      s = t;
      points_extent += 1.0;
    }
    else if (*s != '\0')
      s++;
  }
  /* Scale the speculative points needed by a factor related to BezierQuantum. */
  points_extent *= BezierQuantum;
}
