for (s=t; *s != '\0'; s=t)
        {
          GetNextToken(s,&t,extent,token);
          if (strchr("AaCcQqSsTt",*token) != (char *) NULL)
            {
              if (strchr("CcSs",*token) != (char *) NULL)
                points_extent+=3;
              else if (strchr("QqTt",*token) != (char *) NULL)
                points_extent+=2;
              else if (strchr("Aa",*token) != (char *) NULL)
                {
                  double
                    alpha,
                    beta,
                    radius;

                  alpha=fabs(StringToDouble(s,&next_token));
                  if (s == next_token)
                    ThrowPointExpectedException(s,exception);
                  GetNextToken(s,&s,extent,token);
                  if (*token == ',')
                    GetNextToken(s,&s,extent,token);
                  beta=fabs(StringToDouble(s,&next_token));
                  if (s == next_token)
                    ThrowPointExpectedException(s,exception);
                  radius=hypot((double) alpha,(double) beta);
                  points_extent+=ceil(MagickPI*MagickPI*radius)+
                    6*BezierQuantum+360;
                }
              else
                points_extent++;
              continue;
            }
          if (strchr("Zz",*token) != (char *) NULL)
            {
              points_extent++;
              continue;
            }
          if (strchr("HhVv",*token) != (char *) NULL)
            {
              points_extent++;
              continue;
            }
          if (strchr("Mm",*token) != (char *) NULL)
            {
              points_extent++;
              continue;
            }
          if (strchr("Ll",*token) != (char *) NULL)
            {
              points_extent++;
              continue;
            }
        }
        points_extent*=BezierQuantum;
        break;
