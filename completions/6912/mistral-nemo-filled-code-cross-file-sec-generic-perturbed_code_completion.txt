void FillRectangle(DrawInfo *draw_info, PrimitiveInfo *primitive_info)
{
    Image
        *image;

    PointInfo
        point;

    RectangleInfo
        bounds;

    image = draw_info->image;
    bounds.x = (ssize_t) ceil(primitive_info->point.x - 0.5);
    bounds.y = (ssize_t) ceil(primitive_info->point.y - 0.5);
    bounds.width = (size_t) floor(primitive_info[1].point.x + 0.5) - bounds.x;
    bounds.height = (size_t) floor(primitive_info[1].point.y + 0.5) - bounds.y;

    // Set the fill color based on the draw_info context
    SetImageFillColor(image, draw_info->fill);

    // Fill the rectangle
    FillRectangleImage(image, &bounds, draw_info->exception);
}
