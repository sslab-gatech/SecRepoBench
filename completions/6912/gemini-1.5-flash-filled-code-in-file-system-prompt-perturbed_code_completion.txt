        {
          GetNextToken(s,&s,extent,token);
          if (LocaleCompare("M",token) == 0)
            points_extent+=2;
          else
            if (LocaleCompare("C",token) == 0)
              points_extent+=4*BezierQuantum;
            else
              if (LocaleCompare("c",token) == 0)
                points_extent+=4*BezierQuantum;
              else
                if (LocaleCompare("L",token) == 0)
                  points_extent+=2;
                else
                  if (LocaleCompare("l",token) == 0)
                    points_extent+=2;
                  else
                    if (LocaleCompare("H",token) == 0)
                      points_extent++;
                    else
                      if (LocaleCompare("h",token) == 0)
                        points_extent++;
                      else
                        if (LocaleCompare("V",token) == 0)
                          points_extent++;
                        else
                          if (LocaleCompare("v",token) == 0)
                            points_extent++;
                          else
                            if (LocaleCompare("Q",token) == 0)
                              points_extent+=3*BezierQuantum;
                            else
                              if (LocaleCompare("q",token) == 0)
                                points_extent+=3*BezierQuantum;
                              else
                                if (LocaleCompare("T",token) == 0)
                                  points_extent+=2*BezierQuantum;
                                else
                                  if (LocaleCompare("t",token) == 0)
                                    points_extent+=2*BezierQuantum;
                                  else
                                    if (LocaleCompare("S",token) == 0)
                                      points_extent+=4*BezierQuantum;
                                    else
                                      if (LocaleCompare("s",token) == 0)
                                        points_extent+=4*BezierQuantum;
                                      else
                                        if (LocaleCompare("A",token) == 0)
                                          points_extent+=7*BezierQuantum;
                                        else
                                          if (LocaleCompare("a",token) == 0)
                                            points_extent+=7*BezierQuantum;
                                          else
                                            if (LocaleCompare("z",token) == 0)
                                              points_extent++;
                                            else
                                              if (LocaleCompare("Z",token) == 0)
                                                points_extent++;
        }
