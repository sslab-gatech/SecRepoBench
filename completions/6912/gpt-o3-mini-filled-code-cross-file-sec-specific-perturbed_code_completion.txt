{
  char *p = t;
  size_t count = 0;
  double value;
  char *endptr = NULL;

  while (*p != '\0') {
    /* Skip any leading whitespace or commas */
    while (isspace((unsigned char)*p) || (*p == ','))
      p++;
    if (*p == '\0')
      break;
    errno = 0;
    value = strtod(p, &endptr);
    if (p == endptr) {
      /* No valid conversion found, skip this character */
      p++;
      continue;
    }
    count++;
    p = endptr;
  }
  /* Scale the extent by a factor defined by BezierQuantum */
  points_extent = count * (double) BezierQuantum;
}
