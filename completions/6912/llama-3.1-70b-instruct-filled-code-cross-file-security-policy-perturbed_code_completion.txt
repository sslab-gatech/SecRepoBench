for (s = t; *s!= '\0'; s++)
{
  if (isalpha((int) *s))
    continue;
  if (IsPoint(s) == MagickFalse)
    break;
  GetNextToken(s, &s, extent, token);
  point.x = StringToDouble(token, &next_token);
  GetNextToken(s, &s, extent, token);
  if (*token == ',')
    GetNextToken(s, &s, extent, token);
  point.y = StringToDouble(token, &next_token);
  GetNextToken(s, (const char **) NULL, extent, token);
  if (*token == ',')
    GetNextToken(s, &s, extent, token);
  points_extent += BezierQuantum;
}
