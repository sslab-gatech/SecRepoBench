for (s=t; *s != '\0'; s++)
{
  if (isalpha((int) ((unsigned char) *s)) != 0)
  {
    switch (*s)
    {
      case 'C': case 'c': /* curveto */
        points_extent+=3.0*BezierQuantum;
        break;
        
      case 'S': case 's': /* smooth curveto */
        points_extent+=2.0*BezierQuantum;
        break;
        
      case 'Q': case 'q': /* quadratic curveto */
        points_extent+=2.0*BezierQuantum;
        break;
        
      case 'A': case 'a': /* elliptical arc */
        points_extent+=4.0*BezierQuantum;
        break;
        
      default:
        points_extent+=1.0;
        break;
    }
  }
  else if ((isdigit((int) ((unsigned char) *s)) != 0) || 
           (*s == '-') || (*s == '+') || (*s == '.'))
  {
    /* Skip numeric values */
    double value;
    char *end;
    
    value=StringToDouble(s,&end);
    if (s != end)
      s=end-1;
  }
}
points_extent*=2.0; /* ensure enough space for coordinates */
