const char *s = primitive_token;
char *next_token;
size_t count = 0;

while (*s != '\0')
{
    while ((*s != '\0') && (!isdigit((unsigned char) *s)) && (*s != '-') && (*s != '+') && (*s != '.'))
        s++;
    if (*s == '\0')
        break;
    (void) StringToDouble(s, &next_token);
    if (s == next_token)
        s++;
    else
    {
        count++;
        s = next_token;
    }
}
points_extent = (double)(count / 2) * BezierQuantum;
