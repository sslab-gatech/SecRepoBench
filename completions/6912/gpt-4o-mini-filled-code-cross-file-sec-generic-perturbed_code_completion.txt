while (IsPoint(t) != MagickFalse)
        {
          GetNextToken(t,&t,extent,primitive_token);
          if (*primitive_token == ',')
            GetNextToken(t,&t,extent,primitive_token);
          point.x=StringToDouble(primitive_token,&next_token);
          if (primitive_token == next_token)
            ThrowPointExpectedException(primitive_token,exception);
          GetNextToken(t,&t,extent,primitive_token);
          if (*primitive_token == ',')
            GetNextToken(t,&t,extent,primitive_token);
          point.y=StringToDouble(primitive_token,&next_token);
          if (primitive_token == next_token)
            ThrowPointExpectedException(primitive_token,exception);
          primitive_info[i].point=point;
          i++;
          if (i >= (ssize_t) number_points)
          {
            number_points<<=1;
            primitive_info=(PrimitiveInfo *) ResizeQuantumMemory(primitive_info,
              (size_t) number_points,sizeof(*primitive_info));
            if ((primitive_info == (PrimitiveInfo *) NULL) ||
                (number_points != (MagickSizeType) ((size_t) number_points)))
              ThrowBinaryException(ResourceLimitError,"MemoryAllocationFailed",
                image->filename);
          }
        }
