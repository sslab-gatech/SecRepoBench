for (s=t; *s != '\0'; s=t)
{
  GetNextToken(s,&t,extent,token);
  if (*token == '\0')
    break;
  if (strchr("AaCcQqSsTt",*token) != (char *) NULL)
    {
      if (strchr("CcSs",*token) != (char *) NULL)
        points_extent+=3;
      else
        points_extent+=2;
      continue;
    }
  if (strchr("HhVv",*token) != (char *) NULL)
    {
      points_extent++;
      continue;
    }
  if (strchr("Mm",*token) != (char *) NULL)
    continue;
  if (strchr("LlZz",*token) != (char *) NULL)
    {
      points_extent++;
      continue;
    }
  if (LocaleCompare(token,"pop") == 0)
    break;
  if (LocaleCompare(token,"push") == 0)
    break;
}
points_extent=points_extent*BezierQuantum/4+360;
if (points_extent > 21438)
  {
    (void) ThrowMagickException(exception,GetMagickModule(),DrawError,
      "TooManyBezierCoordinates","`%s'",path);
    status=MagickFalse;
    break;
  }
break;
