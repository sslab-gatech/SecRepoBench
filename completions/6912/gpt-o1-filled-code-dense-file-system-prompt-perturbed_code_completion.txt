for (s = t; *s != '\0'; )
{
  char segment[MagickPathExtent];

  GetNextToken(s, &s, MagickPathExtent, segment);
  if (*segment == '\0')
    break;
  if (isalpha((int) (unsigned char) *segment) != 0)
    continue;
  if (StringToDouble(segment, &next_token) != 0.0 || segment != next_token)
    points_extent++;
}
