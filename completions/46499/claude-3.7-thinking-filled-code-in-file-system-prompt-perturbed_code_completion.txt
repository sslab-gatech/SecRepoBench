while(i < payload_len && packet_payload[i] != ' ' && ndpi_isdigit(packet_payload[i])) i++;
if(i < payload_len && packet_payload[i] == ' ') {
  i++;
  numbers_left--;
  while(numbers_left > 0 && i < payload_len) {
    u_int16_t old_i = i;
    while(i < payload_len && packet_payload[i] != ' ' && ndpi_isdigit(packet_payload[i])) i++;
    if(i < payload_len && packet_payload[i] == ' ' && i > old_i) {
      i++;
      numbers_left--;
    } else {
      break;
    }
  }
  
  if(numbers_left == 0) {
    while(i < payload_len && packet_payload[i] != '\n' && 
          (ndpi_isdigit(packet_payload[i]) || ndpi_isalpha(packet_payload[i]) || 
           packet_payload[i] == '/' || packet_payload[i] == '+' || 
           packet_payload[i] == '=' || packet_payload[i] == '-')) {
      i++;
    }
    
    if(i < payload_len && packet_payload[i] == '\n') {
      connection_flow->tinc_state++;
      
      if(connection_flow->tinc_state > 3) {
        if(ndpi_struct->tinc_cache == NULL) {
          ndpi_struct->tinc_cache = cache_new(512);
        }
        
        if(ndpi_struct->tinc_cache != NULL) {
          cache_add(ndpi_struct->tinc_cache, &connection_flow->tinc_cache_entry, 
                    sizeof(connection_flow->tinc_cache_entry), (uint32_t) IPPROTO_UDP);
        }
        
        NDPI_LOG_INFO(ndpi_struct, "found tinc tcp connection\n");
        ndpi_set_detected_protocol(ndpi_struct, connection_flow, NDPI_PROTOCOL_TINC, 
                                  NDPI_PROTOCOL_UNKNOWN, NDPI_CONFIDENCE_DPI);
      }
      
      return;
    }
  }
