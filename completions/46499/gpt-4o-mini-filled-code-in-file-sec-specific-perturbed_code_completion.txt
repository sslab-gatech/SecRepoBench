while(i < payload_len && numbers_left > 0) {
        if(packet_payload[i] >= '0' && packet_payload[i] <= '9') {
          while(i < payload_len && packet_payload[i] != ' ' && packet_payload[i] != '\n') {
            i++;
          }
          numbers_left--;
        }
        i++;
      }
      if(numbers_left == 0 && i < payload_len && packet_payload[i] == '\n') {
        connection_flow->tinc_state++;
        if (ndpi_struct->tinc_cache != NULL) {
          cache_add(ndpi_struct->tinc_cache, &connection_flow->tinc_cache_entry, sizeof(connection_flow->tinc_cache_entry));
        }
        NDPI_LOG_INFO(ndpi_struct, "detected tinc TCP connection\n");
      }
