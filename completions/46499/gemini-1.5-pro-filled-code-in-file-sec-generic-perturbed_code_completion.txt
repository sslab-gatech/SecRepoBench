while(numbers_left > 0 && i < payload_len) {
	  while(i < payload_len && packet_payload[i] != ' ') {
	    if(packet_payload[i] < '0' || packet_payload[i] > '9') {
	      return;
	    }
	    i++;
	  }
	  i++;
	  numbers_left--;
	}

	if(i < payload_len) {
	  while(i < payload_len && packet_payload[i] != '\n') {
	    if((packet_payload[i] < '0' || packet_payload[i] > '9') &&
	       (packet_payload[i] < 'a' || packet_payload[i] > 'z') &&
	       (packet_payload[i] < 'A' || packet_payload[i] > 'Z')) {
	      return;
	    }
	    i++;
	  }

	  if(i+1 == payload_len && packet_payload[i] == '\n') {
	    connection_flow->tinc_state++;

	    if(connection_flow->tinc_state > 3) {
	      if(ndpi_struct->tinc_cache == NULL) {
		ndpi_struct->tinc_cache = cache_init(1024, sizeof(struct tinc_cache_entry));
	      }

	      if(ndpi_struct->tinc_cache != NULL) {
		cache_add(ndpi_struct->tinc_cache, &connection_flow->tinc_cache_entry, sizeof(connection_flow->tinc_cache_entry));
	      }

	      NDPI_LOG_INFO(ndpi_struct, "found tinc tcp connection\n");
	      ndpi_set_detected_protocol(ndpi_struct, connection_flow, NDPI_PROTOCOL_TINC, NDPI_PROTOCOL_UNKNOWN, NDPI_CONFIDENCE_DPI_PROTOCOL);
	    }
	    return;
	  }
	}
