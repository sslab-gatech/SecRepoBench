while (i < payload_len && numbers_left > 0) {
	  if (isdigit(packet_payload[i])) {
	    while (i < payload_len && isdigit(packet_payload[i])) i++;
	    if (i < payload_len && packet_payload[i] == ' ') {
	      i++;
	      numbers_left--;
	    } else {
	      return;
	    }
	  } else {
	    return;
	  }
	}

	while (i < payload_len && isalnum(packet_payload[i])) i++;

	if (i < payload_len && packet_payload[i] == '\n') {
	  connection_flow->tinc_state++;
	  if (connection_flow->tinc_state > 3) {
	    if (ndpi_struct->tinc_cache == NULL) {
	      ndpi_struct->tinc_cache = cache_init("tinc", 1, sizeof(struct tinc_cache_entry), 0, NULL);
	    }
	    if (ndpi_struct->tinc_cache != NULL) {
	      cache_add(ndpi_struct->tinc_cache, &connection_flow->tinc_cache_entry, sizeof(connection_flow->tinc_cache_entry), 1);
	    }
	    NDPI_LOG_INFO(ndpi_struct, "found tinc tcp connection\n");
	    ndpi_set_detected_protocol(ndpi_struct, connection_flow, NDPI_PROTOCOL_TINC, NDPI_PROTOCOL_UNKNOWN, NDPI_CONFIDENCE_HIGH);
	  }
	}
