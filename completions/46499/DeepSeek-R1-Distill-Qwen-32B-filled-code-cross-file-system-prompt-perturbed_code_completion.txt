while (i < payload_len && isdigit(packet_payload[i])) {
        i++;
      }
      
      if (i < payload_len && packet_payload[i] == ' ') {
        numbers_left--;
        i++;
        
        if (numbers_left == 0) {
          break;
        }
      }
    }

    if (numbers_left == 0 && i < payload_len && packet_payload[i] == '\n') {
      NDPI_LOG_INFO(ndpi_struct, "found tinc tcp connection\n");
      ndpi_set_detected_protocol(ndpi_struct, connection_flow, NDPI_PROTOCOL_TINC, NDPI_PROTOCOL_UNKNOWN, NDPI_CONFIDENCE_DPI_CACHE);
      
      if (ndpi_struct->tinc_cache == NULL) {
        ndpi_struct->tinc_cache = cache_create(100, 10, 0, NULL);
      }
      
      struct tinc_cache_entry *cache_entry = malloc(sizeof(struct tinc_cache_entry));
      cache_entry->src_address = packet->iph->saddr;
      cache_entry->dst_address = packet->iph->daddr;
      cache_entry->dst_port = packet->tcp->dest;
      
      cache_add(ndpi_struct->tinc_cache, cache_entry, sizeof(struct tinc_cache_entry));
    }
