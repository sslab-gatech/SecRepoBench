if (nbchar > HTML_PARSER_BIG_BUFFER_SIZE - 10) {
    /*
     * Flush the buffer, the 10 is there to allow some
     * characters to be stored in the buffer while we
     * check for the end of the current section.
     */
    if ((ctxt->sax != NULL) && (!ctxt->disableSAX)) {
        buf[nbchar] = 0;  /* Safe because buf size is HTML_PARSER_BIG_BUFFER_SIZE + 5 */
        if (ctxt->sax->cdataBlock != NULL)
            ctxt->sax->cdataBlock(ctxt->userData, buf, nbchar);
        else if (ctxt->sax->characters != NULL)
            ctxt->sax->characters(ctxt->userData, buf, nbchar);
    }
    nbchar = 0;
}
