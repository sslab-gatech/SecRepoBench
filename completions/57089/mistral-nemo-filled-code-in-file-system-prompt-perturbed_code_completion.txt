def htmlInitParserCtxt(htmlParserCtxtPtr ctxt, htmlSAXHandlerPtr sax, void *userData):
    # Initialize the parser context
    if ctxt is None:
        return None
    # Reset the parser context
    htmlCtxtReset(ctxt)
    # Set the SAX handler
    ctxt.sax = sax
    ctxt.userData = userData
    return ctxt
