int64_t off_start = header_len + cbytes;
if (cbytes < INT64_MAX - header_len) {
    uint8_t* off_start_ptr = container->cframe + off_start;
    if (off_cbytes != NULL) {
        *off_cbytes = *(int32_t*)(off_start_ptr + BLOSC2_CHUNK_CBYTES);
    }
    return off_start_ptr;
} else {
    BLOSC_TRACE_ERROR("Integer overflow in offset calculation.");
    return NULL;
}
