uint8_t* off_start = NULL;
  if (container->cframe != NULL) {
    off_start = container->cframe + header_len;
  } else {
    FILE* fp = fopen(container->urlpath, "rb");
    fseek(fp, header_len, SEEK_SET);
    size_t rbytes = fread(off_start, 1, cbytes, fp);
    fclose(fp);
    if (rbytes != cbytes) {
      BLOSC_TRACE_ERROR("Cannot read the offsets out of the frame.");
      return NULL;
    }
  }
  return off_start;
