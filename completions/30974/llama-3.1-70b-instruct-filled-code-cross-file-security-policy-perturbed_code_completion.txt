if (cbytes < INT64_MAX - header_len) {
  off_start = container->cframe + header_len + cbytes;
  if (off_cbytes!= NULL) {
    *off_cbytes = *(int32_t *) (off_start + BLOSC2_CHUNK_CBYTES);
  }
  return off_start;
} else {
  BLOSC_TRACE_ERROR("Integer overflow detected when calculating offset.");
  return NULL;
}
