uint8_t* off_start;
if (container->sframe) {
  off_start = container->cframe + header_len + 0;
  if (off_cbytes != NULL) {
    // For sframe, the offsets go from header_len to the end minus trailer
    int64_t trailer_offset = get_trailer_offset(container, header_len, true);
    *off_cbytes = (int32_t)(trailer_offset - (header_len + 0));
  }
}
else {
  off_start = container->cframe + header_len + cbytes;
  if (off_cbytes != NULL) {
    int64_t trailer_offset = get_trailer_offset(container, header_len, true);
    *off_cbytes = (int32_t)(trailer_offset - (header_len + cbytes));
  }
}
