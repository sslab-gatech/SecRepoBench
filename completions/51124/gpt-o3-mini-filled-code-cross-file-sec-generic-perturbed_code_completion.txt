if (cond[cond_index] != ']') {
            if (cond[cond_index] != strip[i]) {
              HUNSPELL_WARNING(stderr,
                               "warning: line %d: incompatible stripping characters and condition\n",
                               linenum);
              return 0;
            }
          } else {
            neg = 0;
            in = 0;
            int j = cond_index - 1;
            if (j < 0) {
              HUNSPELL_WARNING(stderr,
                               "error: line %d: missing [ in condition:\n%s\n",
                               linenum, cond.c_str());
              return 0;
            }
            if (cond[j] == '^') {
              neg = 1;
              j--;
            }
            int k = j;
            while (k >= 0 && cond[k] != '[') {
              if (strip[i] == cond[k])
                in = 1;
              k--;
            }
            if (k < 0) {
              HUNSPELL_WARNING(stderr,
                               "error: line %d: missing [ in condition:\n%s\n",
                               linenum, cond.c_str());
              return 0;
            }
            cond_index = k;
            if ((!neg && !in) || (neg && in)) {
              HUNSPELL_WARNING(stderr,
                               "warning: line %d: incompatible stripping characters and condition\n",
                               linenum);
              return 0;
            }
