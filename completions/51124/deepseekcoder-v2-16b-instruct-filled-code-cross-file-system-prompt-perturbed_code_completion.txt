if (cond[cond_index] == ']') {
  if (cond_index == (condl - 1) || cond[cond_index + 1] != '[') {
    HUNSPELL_WARNING(stderr, "error: line %d: missing [ in condition:\n%s\n", linenum, cond.c_str());
    return 0;
  }
  cond_index++;
  if (strip[i] == cond[cond_index]) {
    in = 1;
  }
} else {
  if (cond[cond_index] != strip[i]) {
    HUNSPELL_WARNING(stderr, "warning: line %d: incompatible stripping characters and condition\n", linenum);
    return 0;
  }
}
