if (cond[cond_index] != ']') {
    if (cond[cond_index] != strip[i]) {
        HUNSPELL_WARNING(stderr,
                         "warning: line %d: incompatible stripping "
                         "characters and condition\n",
                         linenum);
        return 0;
    }
} else {
    int in = 0;
    int neg = 0;
    int j = cond_index - 1;
    while (j >= 0 && cond[j] != '[') {
        if (strip[i] == cond[j])
            in = 1;
        j--;
    }
    if (j < 0) {
        HUNSPELL_WARNING(stderr,
                         "error: line %d: missing [ in condition:\n%s\n",
                         linenum, cond.c_str());
        return 0;
    }
    if ((j + 1 < cond_index) && (cond[j + 1] == '^'))
        neg = 1;
    if ((!neg && !in) || (neg && in)) {
        HUNSPELL_WARNING(stderr,
                         "warning: line %d: incompatible stripping "
                         "characters and condition\n",
                         linenum);
        return 0;
    }
    cond_index = j;
}
