if (cond[cond_index] != ']') {
  if (cond[cond_index] != strip[i]) {
    HUNSPELL_WARNING(stderr,
                     "warning: line %d: incompatible stripping "
                     "characters and condition\n",
                     linenum);
    return 0;
  }
} else {
  // Find the matching '[' for this ']'
  int set_end = cond_index;
  int set_start = cond_index;
  while (set_start >= 0 && cond[set_start] != '[') {
    set_start--;
  }
  if (set_start < 0) {
    HUNSPELL_WARNING(stderr,
                     "error: line %d: missing [ in condition:\n%s\n",
                     linenum, cond.c_str());
    return 0;
  }
  neg = (cond[set_start + 1] == '^') ? 1 : 0;
  in = 0;
  for (int j = set_start + 1 + neg; j < set_end; j++) {
    if (strip[i] == cond[j])
      in = 1;
  }
  if ((!neg && !in) || (neg && in)) {
    HUNSPELL_WARNING(stderr,
                     "warning: line %d: incompatible stripping "
                     "characters and condition\n",
                     linenum);
    return 0;
  }
  cond_index = set_start;
}
