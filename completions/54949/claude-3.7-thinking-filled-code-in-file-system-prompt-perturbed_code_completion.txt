{
    // Create a new dictionary node
    plist_t new_dict = plist_new_dict();
    plist_t key = *plist;
    *plist = NULL;  // Clear the original plist to avoid double-free

    if (!PLIST_IS_STRING(key)) {
        // Dictionary keys must be strings
        plist_free(key);
        plist_free(new_dict);
        return PLIST_ERR_PARSE;
    }

    ctx.pos++; // Skip the '=' sign
    plist_t value = NULL;
    err = node_from_openstep(&ctx, &value);
    if (err != 0 || !value) {
        // Failed to parse the value
        plist_free(key);
        plist_free(new_dict);
        return PLIST_ERR_PARSE;
    }

    // Add the key-value pair to the dictionary
    plist_dict_set_item(new_dict, plist_get_string_ptr(key, NULL), value);
    plist_free(key);

    // Check for a terminating semicolon
    parse_skip_ws(&ctx);
    if (ctx.pos < ctx.end && *ctx.pos == ';') {
        ctx.pos++; // Skip the ';'
        // Parse more dictionary items
        parse_dict_data(&ctx, new_dict);
        if (ctx.err) {
            plist_free(new_dict);
            return PLIST_ERR_PARSE;
        }
    } else {
        // Missing semicolon
        plist_free(new_dict);
        return PLIST_ERR_PARSE;
    }

    *plist = new_dict;
}
