{
    // Try to parse as a dictionary of strings (legacy OpenStep format)
    plist_t dict = plist_new_dict();
    struct _parse_ctx dict_ctx = { ctx.pos, ctx.pos, ctx.end, 0 };
    int dict_err = node_from_openstep(&dict_ctx, &dict);
    if (dict_err != 0 || !dict) {
        plist_free(dict);
        plist_free(*plist);
        *plist = NULL;
        err = PLIST_ERR_PARSE;
    } else {
        plist_free(*plist);
        *plist = dict;
        ctx.pos = dict_ctx.pos;
    }
}
