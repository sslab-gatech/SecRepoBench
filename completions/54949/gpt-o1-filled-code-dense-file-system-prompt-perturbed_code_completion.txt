{
    ctx.pos++; /* skip '=' */
    parse_skip_ws(&ctx);

    plist_t dict = NULL;
    struct _parse_ctx subctx = { ctx.pos, ctx.pos, ctx.end, 0 };
    int res = node_from_openstep(&subctx, &dict);
    if (res != 0 || !dict || plist_get_node_type(dict) != PLIST_DICT) {
        /* parsing failed, free the allocated plist and set an error code */
        plist_free(*plist);
        *plist = NULL;
        err = PLIST_ERR_PARSE;
    } else {
        /* parsing succeeded, assign the newly created dictionary to the plist */
        plist_free(*plist);
        *plist = dict;
        ctx.pos = subctx.pos;
        ctx.err = subctx.err;
    }
}
