{
    plist_t dict = plist_new_dict();
    if (!dict) {
        plist_free(*plist);
        *plist = NULL;
        return PLIST_ERR_NO_MEM;
    }

    ctx.pos++;  // Skip the '='
    while (ctx.pos < ctx.end) {
        plist_t key = NULL;
        plist_t value = NULL;

        int err = node_from_openstep(&ctx, &key);
        if (err != 0 || !key) {
            plist_free(dict);
            plist_free(*plist);
            *plist = NULL;
            return PLIST_ERR_PARSE;
        }

        if (ctx.pos < ctx.end && *ctx.pos == '=') {
            ctx.pos++;
            err = node_from_openstep(&ctx, &value);
            if (err != 0 || !value) {
                plist_free(key);
                plist_free(dict);
                plist_free(*plist);
                *plist = NULL;
                return PLIST_ERR_PARSE;
            }
            plist_dict_set_item(dict, plist_get_string_ptr(key), value);
        }

        plist_free(key);

        if (ctx.pos < ctx.end && *ctx.pos == ';') {
            ctx.pos++;
        } else {
            break;
        }
    }

    plist_free(*plist);
    *plist = dict;
}
