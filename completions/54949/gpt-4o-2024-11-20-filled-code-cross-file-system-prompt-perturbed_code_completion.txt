{
            plist_t dict = plist_new_dict();
            plist_t key = NULL;
            plist_t value = NULL;

            while (ctx.pos < ctx.end && *ctx.pos != ';') {
                parse_skip_ws(&ctx);

                if (ctx.pos >= ctx.end || *ctx.pos == ';') {
                    break;
                }

                // Parse key
                err = node_from_openstep(&ctx, &key);
                if (err != 0 || !key || plist_get_node_type(key) != PLIST_STRING) {
                    plist_free(dict);
                    plist_free(key);
                    plist_free(value);
                    *plist = NULL;
                    return PLIST_ERR_PARSE;
                }

                parse_skip_ws(&ctx);

                if (ctx.pos >= ctx.end || *ctx.pos != '=') {
                    plist_free(dict);
                    plist_free(key);
                    plist_free(value);
                    *plist = NULL;
                    return PLIST_ERR_PARSE;
                }

                ctx.pos++; // Skip '='
                parse_skip_ws(&ctx);

                // Parse value
                err = node_from_openstep(&ctx, &value);
                if (err != 0 || !value) {
                    plist_free(dict);
                    plist_free(key);
                    plist_free(value);
                    *plist = NULL;
                    return PLIST_ERR_PARSE;
                }

                plist_dict_set_item(dict, plist_get_string_val(key), value);
                plist_free(key);
                key = NULL;
                value = NULL;

                parse_skip_ws(&ctx);

                if (ctx.pos < ctx.end && *ctx.pos == ',') {
                    ctx.pos++; // Skip ','
                }
            }

            if (ctx.pos >= ctx.end || *ctx.pos != ';') {
                plist_free(dict);
                *plist = NULL;
                return PLIST_ERR_PARSE;
            }

            ctx.pos++; // Skip ';'
            plist_free(*plist);
            *plist = dict;
        }
