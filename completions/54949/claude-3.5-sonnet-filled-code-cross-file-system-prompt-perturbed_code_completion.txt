{
    plist_t dict = plist_new_dict();
    plist_t key = NULL;
    plist_t value = NULL;
    while (ctx.pos < ctx.end && !ctx.err) {
        parse_skip_ws(&ctx);
        if (ctx.pos >= ctx.end) {
            break;
        }
        ctx.err = node_from_openstep(&ctx, &key);
        if (ctx.err != 0 || !key) {
            break;
        }
        parse_skip_ws(&ctx);
        if (*ctx.pos != '=') {
            plist_free(key);
            break;
        }
        ctx.pos++;
        parse_skip_ws(&ctx);
        ctx.err = node_from_openstep(&ctx, &value);
        if (ctx.err != 0 || !value) {
            plist_free(key);
            break;
        }
        plist_dict_set_item(dict, key, value);
        key = NULL;
        value = NULL;
        parse_skip_ws(&ctx);
        if (*ctx.pos != ';') {
            break;
        }
        ctx.pos++;
    }
    if (ctx.err) {
        plist_free(dict);
    } else {
        plist_free(*plist);
        *plist = dict;
    }
}
