{
    plist_free(*plist);
    *plist = NULL;
    
    plist_t dict = plist_new_dict();
    if (!dict) {
        return PLIST_ERR_NO_MEM;
    }
    
    // Skip the '=' character
    ctx.pos++;
    
    // Parse as key-value pairs
    while (ctx.pos < ctx.end && !ctx.err) {
        parse_skip_ws(&ctx);
        if (ctx.pos >= ctx.end) {
            break;
        }
        
        // Parse key
        plist_t key = NULL;
        err = node_from_openstep(&ctx, &key);
        if (err != 0 || !key) {
            plist_free(dict);
            return (err != 0) ? err : PLIST_ERR_PARSE;
        }
        
        // Check for '=' after key
        parse_skip_ws(&ctx);
        if (ctx.pos >= ctx.end || *ctx.pos != '=') {
            plist_free(key);
            plist_free(dict);
            return PLIST_ERR_PARSE;
        }
        ctx.pos++; // Skip '='
        
        // Parse value
        plist_t value = NULL;
        err = node_from_openstep(&ctx, &value);
        if (err != 0 || !value) {
            plist_free(key);
            plist_free(dict);
            return (err != 0) ? err : PLIST_ERR_PARSE;
        }
        
        // Add to dictionary
        plist_dict_set_item(dict, key, value);
        
        // Check for semicolon separator
        parse_skip_ws(&ctx);
        if (ctx.pos < ctx.end && *ctx.pos == ';') {
            ctx.pos++; // Skip ';'
        }
    }
    
    if (ctx.err) {
        plist_free(dict);
        return PLIST_ERR_PARSE;
    }
    
    *plist = dict;
}
