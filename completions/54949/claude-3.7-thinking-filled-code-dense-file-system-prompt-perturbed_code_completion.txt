{
    // Create a new dictionary
    plist_t dict = plist_new_dict();
    
    // Get the key string from the current plist
    char* key_str = NULL;
    if (plist_get_node_type(*plist) == PLIST_STRING) {
        plist_get_string_val(*plist, &key_str);
    }
    
    if (!key_str) {
        // Invalid key
        plist_free(dict);
        plist_free(*plist);
        *plist = NULL;
        return PLIST_ERR_PARSE;
    }
    
    // Skip the '=' character
    ctx.pos++;
    
    // Parse the value
    plist_t value_node = NULL;
    int parse_result = node_from_openstep(&ctx, &value_node);
    
    if (parse_result != 0 || !value_node) {
        // Parsing failed
        free(key_str);
        plist_free(dict);
        plist_free(*plist);
        *plist = NULL;
        return parse_result ? parse_result : PLIST_ERR_PARSE;
    }
    
    // Add the key-value pair to the dictionary
    plist_dict_set_item(dict, key_str, value_node);
    
    // Clean up and set the dictionary as the result
    free(key_str);
    plist_free(*plist);
    *plist = dict;
}
