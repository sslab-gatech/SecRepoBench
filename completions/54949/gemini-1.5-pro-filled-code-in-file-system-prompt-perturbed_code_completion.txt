{
            plist_t dict = plist_new_dict();
            plist_t key = *plist;
            ctx.pos++;
            plist_t val = NULL;
            ctx.err = node_from_openstep(&ctx, &val);
            if (!ctx.err) {
                plist_dict_set_item(dict, plist_get_string_ptr(key, NULL), val);
                plist_free(key);
                *plist = dict;
                while (ctx.pos < ctx.end && *ctx.pos == ';') {
                    ctx.pos++;
                    if (ctx.pos >= ctx.end)
                        break;
                    key = NULL;
                    ctx.err = node_from_openstep(&ctx, &key);
                    if (ctx.err)
                        break;
                    if (!PLIST_IS_STRING(key)) {
                        PLIST_OSTEP_ERR("Invalid type for dictionary key at offset %ld\n", ctx.pos - ctx.start);
                        ctx.err++;
                        break;
                    }
                    if (ctx.pos < ctx.end && *ctx.pos == '=') {
                        ctx.pos++;
                        val = NULL;
                        ctx.err = node_from_openstep(&ctx, &val);
                        if (ctx.err) {
                            plist_free(key);
                            break;
                        }
                        plist_dict_set_item(dict, plist_get_string_ptr(key, NULL), val);
                        plist_free(key);
                    } else {
                        ctx.err++;
                        break;
                    }
                }
            }
            if (ctx.err) {
                plist_free(dict);
                *plist = NULL;
                return PLIST_ERR_PARSE;
            }

        }
