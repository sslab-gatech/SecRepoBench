ctx.pos++;
parse_skip_ws(&ctx);
if (ctx.pos >= ctx.end) {
    PLIST_OSTEP_ERR("Unexpected end of data while parsing dictionary\n");
    ctx.err++;
    return PLIST_ERR_PARSE;
}
plist_t dict = plist_new_dict();
err = node_from_openstep(&ctx, &dict);
if (err != 0) {
    plist_free(dict);
    return err;
}
*plist = dict;
