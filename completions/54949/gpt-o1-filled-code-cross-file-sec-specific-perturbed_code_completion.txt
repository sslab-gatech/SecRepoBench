{
    /* Skip the '=' character */
    ctx.pos++;
    parse_skip_ws(&ctx);

    plist_t new_plist = NULL;
    int parse_err = node_from_openstep(&ctx, &new_plist);
    if (parse_err != 0 || !new_plist || plist_get_node_type(new_plist) != PLIST_DICT) {
        /* Parsing failed, free existing plist and set error code */
        plist_free(*plist);
        *plist = NULL;
        plist_free(new_plist);
        err = parse_err ? parse_err : PLIST_ERR_PARSE;
    } else {
        /* Parsing succeeded, replace current plist with the new dictionary */
        plist_free(*plist);
        *plist = new_plist;
    }
}
