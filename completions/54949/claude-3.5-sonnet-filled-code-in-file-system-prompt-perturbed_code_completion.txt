{
    // Create a new dictionary to hold the parsed data
    plist_t dict = plist_new_dict();
    
    // Parse the remaining data as key-value pairs
    while (ctx.pos < ctx.end && !ctx.err) {
        plist_t key = NULL;
        plist_t value = NULL;
        
        // Parse the key
        err = node_from_openstep(&ctx, &key);
        if (err != 0 || !PLIST_IS_STRING(key)) {
            plist_free(dict);
            plist_free(key);
            *plist = NULL;
            return PLIST_ERR_PARSE;
        }
        
        // Check for the '=' separator
        parse_skip_ws(&ctx);
        if (ctx.pos >= ctx.end || *ctx.pos != '=') {
            plist_free(dict);
            plist_free(key);
            *plist = NULL;
            return PLIST_ERR_PARSE;
        }
        ctx.pos++;
        
        // Parse the value
        err = node_from_openstep(&ctx, &value);
        if (err != 0) {
            plist_free(dict);
            plist_free(key);
            plist_free(value);
            *plist = NULL;
            return PLIST_ERR_PARSE;
        }
        
        // Add the key-value pair to the dictionary
        plist_dict_set_item(dict, plist_get_string_ptr(key, NULL), value);
        plist_free(key);
        
        // Check for the ';' separator between pairs
        parse_skip_ws(&ctx);
        if (ctx.pos < ctx.end && *ctx.pos == ';') {
            ctx.pos++;
        }
    }
    
    // Free the original plist and assign the new dictionary
    plist_free(*plist);
    *plist = dict;
}
