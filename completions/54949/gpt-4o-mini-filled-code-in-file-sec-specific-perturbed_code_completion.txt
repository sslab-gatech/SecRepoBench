{
            ctx.pos++;
            parse_skip_ws(&ctx);
            if (ctx.pos < ctx.end && *ctx.pos == '{') {
                plist_t dict = plist_new_dict();
                parse_dict_data(&ctx, dict);
                if (ctx.err) {
                    plist_free(dict);
                    return PLIST_ERR_PARSE;
                }
                *plist = dict;
            } else {
                PLIST_OSTEP_ERR("Expected '{' for dictionary at offset %ld\n", ctx.pos - ctx.start);
                return PLIST_ERR_PARSE;
            }
