{
    plist_t dict = plist_new_dict();
    plist_t key = NULL;
    plist_t value = NULL;
    ctx.pos++; // skip '='
    parse_skip_ws(&ctx);
    while (ctx.pos < ctx.end && *ctx.pos != ';') {
        parse_skip_ws(&ctx);
        if (*ctx.pos == ';') {
            break;
        }
        err = node_from_openstep(&ctx, &key);
        if (err != 0 || !key) {
            plist_free(dict);
            return PLIST_ERR_PARSE;
        }
        parse_skip_ws(&ctx);
        if (*ctx.pos != '=') {
            plist_free(dict);
            plist_free(key);
            return PLIST_ERR_PARSE;
        }
        ctx.pos++; // skip '='
        parse_skip_ws(&ctx);
        err = node_from_openstep(&ctx, &value);
        if (err != 0 || !value) {
            plist_free(dict);
            plist_free(key);
            return PLIST_ERR_PARSE;
        }
        plist_dict_set_item(dict, plist_get_string_val(key), value);
        plist_free(key);
        key = NULL;
        value = NULL;
        parse_skip_ws(&ctx);
        if (*ctx.pos == ';') {
            ctx.pos++; // skip ';'
        }
    }
    if (*ctx.pos != ';') {
        plist_free(dict);
        return PLIST_ERR_PARSE;
    }
    ctx.pos++; // skip ';'
    *plist = dict;
}
