{
    // Attempt to parse as a dictionary of 'strings'
    plist_t dict = plist_new_dict();
    plist_t key = *plist;
    *plist = NULL;
    ctx.pos++;
    parse_skip_ws(&ctx);
    plist_t value = NULL;
    err = node_from_openstep(&ctx, &value);
    if (err == 0 && value) {
        plist_dict_set_item(dict, plist_get_string_ptr(key, NULL), value);
        plist_free(key);
        *plist = dict;
        parse_skip_ws(&ctx);
        if (ctx.pos < ctx.end && *ctx.pos == ';') {
            ctx.pos++;
            parse_dict_data(&ctx, dict);
        }
    } else {
        plist_free(dict);
        plist_free(key);
        err = PLIST_ERR_PARSE;
    }
}
