{
    plist_t tmp_plist = *plist;
    plist_free(tmp_plist);
    *plist = NULL;

    plist_t dict = plist_new_dict();
    if (!dict) {
        return PLIST_ERR_OUT_OF_MEMORY;
    }

    const char *start = ctx.pos + 1;
    const char *end = ctx.end;
    size_t len = end - start;

    char *str = strndup(start, len);
    if (!str) {
        plist_free(dict);
        return PLIST_ERR_OUT_OF_MEMORY;
    }

    int res = unescape_entities(str, &len);
    if (res != 0) {
        free(str);
        plist_free(dict);
        return PLIST_ERR_PARSE;
    }

    plist_t key = plist_new_string(str);
    free(str);
    if (!key) {
        plist_free(dict);
        return PLIST_ERR_OUT_OF_MEMORY;
    }

    plist_dict_set_item(dict, "value", key);
    *plist = dict;
}
