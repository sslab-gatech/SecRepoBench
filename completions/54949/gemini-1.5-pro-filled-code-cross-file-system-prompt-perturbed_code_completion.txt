{
            plist_t dict = plist_new_dict();
            char *key = NULL;
            char *val = NULL;
            while (ctx.pos < ctx.end) {
                parse_skip_ws(&ctx);
                if (*ctx.pos == '=') {
                    break;
                }
                if (key) {
                    free(key);
                }
                key = parse_unquoted_string(&ctx);
                if (!key || ctx.err) {
                    goto err_out;
                }
                parse_skip_ws(&ctx);
                if (*ctx.pos != '=') {
                    PLIST_OSTEP_ERR("Expected '=' but found '%c' at offset %ld\n", *ctx.pos, ctx.pos - ctx.start);
                    ctx.err++;
                    goto err_out;
                }
                ctx.pos++;
                parse_skip_ws(&ctx);
                if (val) {
                    free(val);
                }
                val = parse_unquoted_string(&ctx);
                if (!val || ctx.err) {
                    goto err_out;
                }
                plist_dict_set_item(dict, key, plist_new_string(val, strlen(val)));
                parse_skip_ws(&ctx);
            }
            if (key) {
                free(key);
            }
            if (val) {
                free(val);
            }
            plist_free(*plist);
            *plist = dict;
        }
