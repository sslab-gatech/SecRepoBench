if (ctx.pos < ctx.end && *ctx.pos == '=') {
    ctx.pos++;
    plist_t dict = plist_new_dict();
    plist_t key = *plist;
    plist_t value = NULL;
    int err = node_from_openstep(&ctx, &value);
    if (err!= 0) {
        plist_free(dict);
        plist_free(key);
        plist_free(value);
        *plist = NULL;
        return PLIST_ERR_PARSE;
    }
    plist_dict_set_item(dict, key, value);
    *plist = dict;
}
