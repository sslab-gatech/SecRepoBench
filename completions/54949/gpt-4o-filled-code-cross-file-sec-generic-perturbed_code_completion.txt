{
    plist_t dict = plist_new_dict();
    plist_t key = NULL;
    plist_t value = NULL;
    ctx.pos++; // skip the '='
    parse_skip_ws(&ctx);
    while (ctx.pos < ctx.end && !ctx.err) {
        parse_skip_ws(&ctx);
        if (*ctx.pos == ';') {
            ctx.pos++;
            parse_skip_ws(&ctx);
            continue;
        }
        if (*ctx.pos == '}') {
            ctx.pos++;
            break;
        }
        ctx.err = node_from_openstep(&ctx, &key);
        if (ctx.err != 0 || !key) {
            plist_free(dict);
            plist_free(key);
            return PLIST_ERR_PARSE;
        }
        parse_skip_ws(&ctx);
        if (*ctx.pos != '=') {
            plist_free(dict);
            plist_free(key);
            return PLIST_ERR_PARSE;
        }
        ctx.pos++;
        parse_skip_ws(&ctx);
        ctx.err = node_from_openstep(&ctx, &value);
        if (ctx.err != 0 || !value) {
            plist_free(dict);
            plist_free(key);
            plist_free(value);
            return PLIST_ERR_PARSE;
        }
        plist_dict_set_item(dict, plist_get_string_val(key), value);
        plist_free(key);
        key = NULL;
        value = NULL;
        parse_skip_ws(&ctx);
        if (*ctx.pos != ';') {
            plist_free(dict);
            return PLIST_ERR_PARSE;
        }
        ctx.pos++;
    }
    if (ctx.err) {
        plist_free(dict);
        return PLIST_ERR_PARSE;
    }
    *plist = dict;
}
