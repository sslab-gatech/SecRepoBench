{
            plist_t dict = plist_new_dict();
            plist_t key = *plist;
            *plist = dict;

            parse_skip_ws(&ctx);
            ctx.pos++; // skip '='

            plist_t val = NULL;
            err = node_from_openstep(&ctx, &val);

            if (err != 0) {
                plist_free(dict);
                plist_free(key);
                *plist = NULL;
                return PLIST_ERR_PARSE;
            }
            if (!val) {
                plist_free(dict);
                plist_free(key);
                *plist = NULL;
                PLIST_OSTEP_ERR("Missing value for dictionary item at offset %ld\n", ctx.pos - ctx.start);
                return PLIST_ERR_PARSE;
            }
            parse_skip_ws(&ctx);
            if (*ctx.pos != ';') {
                plist_free(val);
                plist_free(key);
                plist_free(dict);
                *plist = NULL;
                PLIST_OSTEP_ERR("Missing terminating ';' while parsing dictionary item at offset %ld\n", ctx.pos - ctx.start);
                return PLIST_ERR_PARSE;
            }
            plist_dict_set_item(dict, plist_get_string_ptr(key, NULL), val);
            plist_free(key);
            ctx.pos++;
        }
