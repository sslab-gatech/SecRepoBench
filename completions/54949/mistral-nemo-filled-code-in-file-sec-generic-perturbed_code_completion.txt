plist_t dict = plist_new_dict();
    ctx.pos++;
    parse_dict_data(&ctx, dict);
    if (ctx.err) {
        plist_free(dict);
        *plist = NULL;
        return PLIST_ERR_PARSE;
    }
    if (*ctx.pos != '}') {
        PLIST_OSTEP_ERR("Missing terminating '}' at offset %ld\n", ctx.pos - ctx.start);
        ctx.err++;
        plist_free(dict);
        *plist = NULL;
        return PLIST_ERR_PARSE;
    }
    ctx.pos++;
    *plist = dict;
    parse_skip_ws(&ctx);
    break;
