{
    // Attempt to parse as a dictionary of 'strings'
    plist_t dict = plist_new_dict();
    plist_t key = *plist;
    *plist = NULL;
    ctx.pos++;
    parse_skip_ws(&ctx);
    
    while (ctx.pos < ctx.end && !ctx.err) {
        plist_t value = NULL;
        err = node_from_openstep(&ctx, &value);
        if (err != 0 || !value) {
            plist_free(dict);
            plist_free(key);
            ctx.err++;
            break;
        }
        
        plist_dict_set_item(dict, plist_get_string_ptr(key, NULL), value);
        plist_free(key);
        
        parse_skip_ws(&ctx);
        if (ctx.pos >= ctx.end || *ctx.pos != ';') {
            plist_free(dict);
            ctx.err++;
            break;
        }
        ctx.pos++;
        
        parse_skip_ws(&ctx);
        if (ctx.pos >= ctx.end) {
            break;
        }
        
        err = node_from_openstep(&ctx, &key);
        if (err != 0 || !key || !PLIST_IS_STRING(key)) {
            plist_free(dict);
            plist_free(key);
            ctx.err++;
            break;
        }
        
        parse_skip_ws(&ctx);
        if (ctx.pos >= ctx.end || *ctx.pos != '=') {
            plist_free(dict);
            plist_free(key);
            ctx.err++;
            break;
        }
        ctx.pos++;
    }
    
    if (!ctx.err) {
        *plist = dict;
    } else {
        plist_free(dict);
        err = PLIST_ERR_PARSE;
    }
}
