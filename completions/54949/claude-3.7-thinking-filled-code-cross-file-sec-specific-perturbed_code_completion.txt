{
    // Save the key
    plist_t key = *plist;
    *plist = NULL;  // Prevent double free if error occurs
    
    // Skip the equals sign
    ctx.pos++;
    
    // Parse the value
    plist_t value = NULL;
    err = node_from_openstep(&ctx, &value);
    if (err != 0) {
        plist_free(key);
        return err;
    }
    
    // Create a new dictionary
    plist_t dict = plist_new_dict();
    if (!dict) {
        plist_free(key);
        if (value) plist_free(value);
        return PLIST_ERR_NO_MEM;
    }
    
    // Add key-value pair to dictionary
    if (value) {
        plist_dict_set_item(dict, key, value);
        *plist = dict;
    } else {
        plist_free(key);
        plist_free(dict);
        return PLIST_ERR_PARSE;
    }
}
