{
    // Free the current plist
    plist_free(*plist);
    *plist = NULL;
    
    // Create a new dictionary
    plist_t dict = plist_new_dict();
    if (!dict) {
        return PLIST_ERR_MEMORY;
    }
    
    // Try to parse the data as a dictionary of strings
    err = node_from_openstep(&ctx, &dict);
    
    if (err != 0 || !dict) {
        // Parsing failed
        plist_free(dict);
        return PLIST_ERR_PARSE;
    }
    
    // Set the dictionary as the result
    *plist = dict;
}
