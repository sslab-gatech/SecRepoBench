int err = node_from_openstep(&ctx, plist);
if (err!= 0) {
    plist_free(*plist);
    *plist = NULL;
    return err;
}
plist_t dict = plist_new_dict();
plist_t key = plist_new_string("key");
plist_t value = plist_new_string("value");
plist_dict_set_item(dict, key, *plist);
plist_free(key);
plist_free(value);
plist_free(*plist);
*plist = dict;
