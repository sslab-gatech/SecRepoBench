{
            plist_t dict = plist_new_dict();
            plist_t key = NULL;
            plist_t value = NULL;

            while (ctx.pos < ctx.end && *ctx.pos != ';') {
                parse_skip_ws(&ctx);

                // Parse key
                err = node_from_openstep(&ctx, &key);
                if (err != 0 || !key) {
                    plist_free(dict);
                    plist_free(*plist);
                    *plist = NULL;
                    return PLIST_ERR_PARSE;
                }

                parse_skip_ws(&ctx);

                // Expect '='
                if (ctx.pos >= ctx.end || *ctx.pos != '=') {
                    plist_free(key);
                    plist_free(dict);
                    plist_free(*plist);
                    *plist = NULL;
                    return PLIST_ERR_PARSE;
                }
                ctx.pos++;

                parse_skip_ws(&ctx);

                // Parse value
                err = node_from_openstep(&ctx, &value);
                if (err != 0 || !value) {
                    plist_free(key);
                    plist_free(dict);
                    plist_free(*plist);
                    *plist = NULL;
                    return PLIST_ERR_PARSE;
                }

                // Add key-value pair to dictionary
                plist_dict_set_item(dict, plist_get_string_val(key), value);
                plist_free(key);
                key = NULL;
                value = NULL;

                parse_skip_ws(&ctx);

                // Expect ',' or ';'
                if (ctx.pos < ctx.end && *ctx.pos == ',') {
                    ctx.pos++;
                } else if (ctx.pos < ctx.end && *ctx.pos == ';') {
                    break;
                } else {
                    plist_free(dict);
                    plist_free(*plist);
                    *plist = NULL;
                    return PLIST_ERR_PARSE;
                }
            }

            if (ctx.pos >= ctx.end || *ctx.pos != ';') {
                plist_free(dict);
                plist_free(*plist);
                *plist = NULL;
                return PLIST_ERR_PARSE;
            }

            ctx.pos++; // Skip the ';'
            plist_free(*plist);
            *plist = dict;
        }
