{
    plist_t dict = plist_new_dict();
    if (!dict) {
        plist_free(*plist);
        *plist = NULL;
        return PLIST_ERR_NO_MEM;
    }

    while (ctx.pos < ctx.end) {
        plist_t key = NULL;
        plist_t value = NULL;

        parse_skip_ws(&ctx);
        if (ctx.pos >= ctx.end) break;

        if (node_from_openstep(&ctx, &key) != PLIST_ERR_SUCCESS) {
            plist_free(dict);
            plist_free(*plist);
            *plist = NULL;
            return PLIST_ERR_PARSE;
        }

        parse_skip_ws(&ctx);
        if (ctx.pos >= ctx.end || *ctx.pos != '=') {
            plist_free(key);
            plist_free(dict);
            plist_free(*plist);
            *plist = NULL;
            return PLIST_ERR_PARSE;
        }
        ctx.pos++;

        parse_skip_ws(&ctx);
        if (ctx.pos >= ctx.end) {
            plist_free(key);
            plist_free(dict);
            plist_free(*plist);
            *plist = NULL;
            return PLIST_ERR_PARSE;
        }

        if (node_from_openstep(&ctx, &value) != PLIST_ERR_SUCCESS) {
            plist_free(key);
            plist_free(dict);
            plist_free(*plist);
            *plist = NULL;
            return PLIST_ERR_PARSE;
        }

        plist_dict_set_item(dict, plist_get_key_ptr(key), value);
        plist_free(key);

        parse_skip_ws(&ctx);
        if (ctx.pos < ctx.end && *ctx.pos == ';') {
            ctx.pos++;
        }
    }

    plist_free(*plist);
    *plist = dict;
}
