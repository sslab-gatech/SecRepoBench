hb_set_t *work_set = hb_set_create ();
  hb_set_union (work_set, initial_gids_to_retain);

  // Collect all needed glyphs including their composite references.
  while (work_set->next (&gid))
  {
    work_set->del (gid);
    if (hb_set_has (full_glyphs_to_retain, gid))
      continue;

    full_glyphs_to_retain->add (gid);

    hb_set_t *component_glyphs = hb_set_create ();
    glyf.get_component_glyphs (gid, component_glyphs);
    hb_set_union (work_set, component_glyphs);
    hb_set_destroy (component_glyphs);
  }

  hb_set_destroy (work_set);
  hb_set_destroy (initial_gids_to_retain);

  gid = HB_SET_VALUE_INVALID;
  while (full_glyphs_to_retain->next (&gid))
    glyphs->push (gid);

  cmap.fini ();
  glyf.fini ();

  return full_glyphs_to_retain;
