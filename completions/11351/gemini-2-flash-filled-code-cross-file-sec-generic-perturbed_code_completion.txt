hb_set_t *current_glyphs = hb_set_copy (initial_gids_to_retain);
  hb_codepoint_t current_gid = HB_SET_VALUE_INVALID;
  while (current_glyphs->next (&current_gid))
  {
    if (full_glyphs_to_retain->has (current_gid))
      continue;

    full_glyphs_to_retain->add (current_gid);

    unsigned int num_components = glyf.get_num_components (current_gid);
    for (unsigned int j = 0; j < num_components; ++j)
    {
      hb_codepoint_t component_gid = glyf.get_component_glyph (current_gid, j);
      if (component_gid == HB_INVALID_GLYPH_ID)
        continue;
      hb_set_add (initial_gids_to_retain, component_gid);
      hb_set_add (current_glyphs, component_gid);
    }
  }
  hb_set_destroy (current_glyphs);
  hb_set_destroy (initial_gids_to_retain);

  glyphs->alloc (full_glyphs_to_retain->get_population ());
  gid = HB_SET_VALUE_INVALID;
  while (full_glyphs_to_retain->next (&gid))
    glyphs->push (gid);

  glyf.fini ();
  cmap.fini ();

  return full_glyphs_to_retain;
