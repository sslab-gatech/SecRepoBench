while (initial_gids_to_retain->next (&gid))
  _add_gid_and_children (glyf, gid, full_glyphs_to_retain);

hb_set_destroy (initial_gids_to_retain);

_remove_invalid_gids (full_glyphs_to_retain, face->num_glyphs);

glyphs->resize (full_glyphs_to_retain->get_population ());
unsigned int i = 0;
while (full_glyphs_to_retain->next (&gid))
  glyphs->elem (i++) = gid;

glyf.fini ();
cmap.fini ();

return full_glyphs_to_retain;
