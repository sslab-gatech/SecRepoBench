while (initial_gids_to_retain->next (&gid))
  {
    full_glyphs_to_retain->add (gid);
    glyf.add_gid_and_children (gid, full_glyphs_to_retain);
  }
  hb_set_destroy (initial_gids_to_retain);

  glyphs->alloc (full_glyphs_to_retain->get_population () + 1);
  hb_codepoint_t g = HB_SET_VALUE_INVALID;
  while (full_glyphs_to_retain->next (&g))
    glyphs->push (g);

  glyf.fini ();
  cmap.fini ();
  hb_set_destroy (full_glyphs_to_retain);

  return full_glyphs_to_retain;
