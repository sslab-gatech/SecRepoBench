hb_codepoint_t gid;
while (initial_gids_to_retain->next (&gid))
{
  if (!glyf.get_glyph_data (gid, nullptr))
    continue;

  glyphs->push (gid);
  full_glyphs_to_retain->add (gid);

  hb_codepoint_t component_gid;
  unsigned int start_offset, end_offset;
  if (glyf.get_components (gid, &component_gid, &start_offset, &end_offset))
  {
    do
    {
      if (!full_glyphs_to_retain->has (component_gid))
      {
        glyphs->push (component_gid);
        full_glyphs_to_retain->add (component_gid);
      }
    } while (glyf.get_components (gid, &component_gid, &start_offset, &end_offset));
  }
}

cmap.fini ();
glyf.fini ();

return full_glyphs_to_retain;
