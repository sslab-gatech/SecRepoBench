while (initial_gids_to_retain->next (&gid))
{
  hb_set_add (full_glyphs_to_retain, gid);
  if (!glyf.get_composite_members (face, gid, full_glyphs_to_retain))
  {
    DEBUG_MSG(SUBSET, nullptr, "Glyph %u has no components.", gid);
  }
}

hb_set_destroy (initial_gids_to_retain);

if (!glyphs->alloc (full_glyphs_to_retain->get_population (), true))
{
  DEBUG_MSG(SUBSET, nullptr, "Failed to allocate memory for glyphs vector.");
  hb_set_destroy (full_glyphs_to_retain);
  cmap.fini();
  glyf.fini();
  return nullptr;
}

hb_codepoint_t current_gid = HB_SET_VALUE_INVALID;
while (full_glyphs_to_retain->next (&current_gid))
{
  glyphs->push (current_gid);
}

cmap.fini();
glyf.fini();

return full_glyphs_to_retain;
