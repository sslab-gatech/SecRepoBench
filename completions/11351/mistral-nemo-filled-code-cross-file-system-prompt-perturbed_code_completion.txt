// Populate a full set of glyphs to retain by adding all referenced
  // composite glyphs.
  hb_codepoint_t gid = HB_SET_VALUE_INVALID;
  hb_set_t *full_glyphs_to_retain = hb_set_create ();
  full_glyphs_to_retain->add_all (initial_gids_to_retain);
  hb_vector_t<hb_codepoint_t> components;
  while (initial_gids_to_retain->next (&gid))
  {
    if (glyf.get_components (gid, &components))
    {
      for (unsigned int i = 0; i < components.len; i++)
      {
        full_glyphs_to_retain->add (components[i]);
      }
    }
  }
  hb_set_destroy (initial_gids_to_retain);
  glyphs->resize (full_glyphs_to_retain->len());
  unsigned int i = 0;
  full_glyphs_to_retain->next (&gid);
  while (gid != HB_SET_VALUE_INVALID)
  {
    glyphs->set (i, gid);
    i++;
    full_glyphs_to_retain->next (&gid);
  }
  cmap.fini ();
  glyf.fini ();
  return full_glyphs_to_retain;
