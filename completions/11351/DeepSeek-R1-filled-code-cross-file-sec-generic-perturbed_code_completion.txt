hb_set_t *current_batch = hb_set_create ();
  hb_codepoint_t gid = HB_SET_VALUE_INVALID;
  while (initial_gids_to_retain->next (&gid))
    current_batch->add (gid);

  while (!hb_set_is_empty (current_batch))
  {
    hb_codepoint_t gid = HB_SET_VALUE_INVALID;
    while (current_batch->next (&gid))
    {
      current_batch->del (gid);
      if (full_glyphs_to_retain->has (gid)) continue;

      full_glyphs_to_retain->add (gid);

      if (glyf.glyph_is_composite (gid))
      {
        hb_vector_t<hb_codepoint_t> components;
        if (glyf.get_components (gid, &components))
        {
          for (unsigned i = 0; i < components.len; i++)
          {
            hb_codepoint_t component_gid = components[i];
            if (!full_glyphs_to_retain->has (component_gid))
              current_batch->add (component_gid);
          }
        }
      }
    }
  }

  hb_set_destroy (current_batch);
  hb_set_destroy (initial_gids_to_retain);

  gid = HB_SET_VALUE_INVALID;
  while (full_glyphs_to_retain->next (&gid))
    glyphs->push (gid);

  glyf.fini ();
  cmap.fini ();

  return full_glyphs_to_retain;
