hb_set_t *pending = hb_set_create ();
  pending->union_ (initial_gids_to_retain);

  while (!pending->is_empty ())
  {
    hb_codepoint_t gid = HB_SET_VALUE_INVALID;
    hb_set_t *components = hb_set_create ();
    while (pending->next (&gid))
    {
      if (full_glyphs_to_retain->has (gid)) continue;
      full_glyphs_to_retain->add (gid);
      glyf.add_component_glyphs (gid, components);
    }
    hb_set_clear (pending);
    pending->union_ (components);
    hb_set_destroy (components);
  }

  hb_set_destroy (pending);
  hb_set_destroy (initial_gids_to_retain);

  glyphs->alloc (full_glyphs_to_retain->get_population ());
  hb_codepoint_t gid = HB_SET_VALUE_INVALID;
  while (full_glyphs_to_retain->next (&gid))
    glyphs->push (gid);
  glyphs->qsort ();

  glyf.fini ();
  cmap.fini ();

  return full_glyphs_to_retain;
