while (initial_gids_to_retain->next (&gid))
  {
    full_glyphs_to_retain->add (gid);
    hb_ot_glyf_collect_glyphs (face, gid, full_glyphs_to_retain);
  }

  hb_set_destroy (initial_gids_to_retain);

  glyphs->reserve (full_glyphs_to_retain->get_population ());
  gid = HB_SET_VALUE_INVALID;
  while (full_glyphs_to_retain->next (&gid))
    glyphs->push (gid);

  hb_set_destroy (full_glyphs_to_retain);
  cmap.fini ();
  glyf.fini ();

  return hb_set_create_from_vector (*glyphs);
