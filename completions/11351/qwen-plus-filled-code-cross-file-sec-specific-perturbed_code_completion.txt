while (initial_gids_to_retain->next (&gid))
{
  hb_set_add (full_glyphs_to_retain, gid);
  if (!glyf.get_composite_components (gid, full_glyphs_to_retain))
  {
    DEBUG_MSG(SUBSET, nullptr, "Glyph %u has no components.", gid);
  }
}

hb_set_destroy (initial_gids_to_retain);

if (!glyphs->alloc (full_glyphs_to_retain->get_population (), true))
{
  DEBUG_MSG(SUBSET, nullptr, "Memory allocation failed for glyphs vector.");
}

hb_codepoint_t gid_iter = HB_SET_VALUE_INVALID;
while (full_glyphs_to_retain->next (&gid_iter))
{
  glyphs->push (gid_iter);
}

cmap.fini ();
glyf.fini ();

return full_glyphs_to_retain;
