while (initial_gids_to_retain->next (&gid))
  {
    if (unlikely (gid >= face->get_num_glyphs ()))
      continue;
    full_glyphs_to_retain->add (gid);
    glyf.add_gid_and_components (gid, full_glyphs_to_retain);
    glyphs->push (gid);
  }

  hb_set_destroy (initial_gids_to_retain);

  _create_old_gid_to_new_gid_map (*glyphs, codepoint_to_glyph);

  cmap.fini ();
  glyf.fini ();

  return full_glyphs_to_retain;
