hb_set_t *gids_to_retain = hb_set_create ();
hb_codepoint_t gid = HB_SET_VALUE_INVALID;
while (initial_gids_to_retain->next (&gid))
{
  if (gid > face->num_glyphs)
  {
    DEBUG_MSG(SUBSET, nullptr, "Drop gid %d; out of range", gid);
    continue;
  }
  gids_to_retain->add (gid);
  glyf.get_components (gid, glyphs);
  for (unsigned int i = 0; i < glyphs->len; i++)
  {
    hb_codepoint_t component_gid = (*glyphs)[i];
    if (component_gid > face->num_glyphs)
    {
      DEBUG_MSG(SUBSET, nullptr, "Drop gid %d; out of range", component_gid);
      continue;
    }
    gids_to_retain->add (component_gid);
  }
}
hb_set_destroy (initial_gids_to_retain);
hb_vector_t<hb_codepoint_t> *glyphs_to_retain = hb_vector_create ();
for (unsigned int i = 0; i < gids_to_retain->get_population (); i++)
{
  hb_codepoint_t gid = gids_to_retain->get_value (i);
  glyphs_to_retain->push (gid);
}
hb_set_destroy (gids_to_retain);
cmap.fini ();
glyf.fini ();
return glyphs_to_retain;
