while (initial_gids_to_retain->next (&gid))
{
  hb_set_add (full_glyphs_to_retain, gid);
  if (!glyf.get_composite_glyph_components (face, gid, full_glyphs_to_retain))
    DEBUG_MSG(SUBSET, nullptr, "Glyph %u has no components", gid);
}

hb_set_destroy (initial_gids_to_retain);

glyphs->alloc (full_glyphs_to_retain->get_population ());
hb_codepoint_t new_gid = 0;
hb_codepoint_t old_gid = HB_SET_VALUE_INVALID;
while (full_glyphs_to_retain->next (&old_gid))
{
  glyphs->push (old_gid);
  codepoint_to_glyph->set (old_gid, new_gid++);
}

cmap.fini ();
glyf.fini ();

return full_glyphs_to_retain;
