// Transfer the initial glyphs to a work set and perform a closure over any composite glyphs.
  hb_set_t *work_list = hb_set_create ();
  hb_set_union (work_list, initial_gids_to_retain);

  while (work_list->pop (&gid))
  {
    // If we've already added this glyph to the full set, skip it.
    if (full_glyphs_to_retain->has (gid))
      continue;

    // Mark this glyph as needed.
    full_glyphs_to_retain->add (gid);

    // Look up any composite components and add them to the work list.
    unsigned int component_count = 0;
    const hb_codepoint_t *components = glyf.get_components (gid, &component_count);
    for (unsigned int i = 0; i < component_count; ++i)
      work_list->add (components[i]);
  }

  // Release the initial set.
  hb_set_destroy (initial_gids_to_retain);
  hb_set_destroy (work_list);

  // Populate the vector of final glyph IDs.
  gid = HB_SET_VALUE_INVALID;
  while (full_glyphs_to_retain->next (&gid))
    glyphs->push (gid);

  // Finalize and clean up.
  glyf.fini ();
  cmap.fini ();

  return full_glyphs_to_retain;
