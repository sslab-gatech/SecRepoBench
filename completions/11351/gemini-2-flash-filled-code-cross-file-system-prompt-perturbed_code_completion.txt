hb_set_t *current_glyphs = hb_set_copy (initial_gids_to_retain);
  while (current_glyphs->next (&gid))
  {
    full_glyphs_to_retain->add (gid);

    unsigned int num_components = glyf.get_num_components (gid);
    if (num_components > 0)
    {
      for (unsigned int j = 0; j < num_components; j++)
      {
        hb_codepoint_t component_gid = glyf.get_component_glyph (gid, j);
        if (!full_glyphs_to_retain->has (component_gid))
        {
          initial_gids_to_retain->add (component_gid);
          current_glyphs->add (component_gid);
        }
      }
    }
  }
  hb_set_destroy (current_glyphs);
  hb_set_destroy (initial_gids_to_retain);

  glyphs->alloc (full_glyphs_to_retain->get_population ());
  gid = HB_SET_VALUE_INVALID;
  while (full_glyphs_to_retain->next (&gid))
    glyphs->push (gid);
  glyphs->sort ();

  glyf.fini ();
  cmap.fini ();

  return full_glyphs_to_retain;
