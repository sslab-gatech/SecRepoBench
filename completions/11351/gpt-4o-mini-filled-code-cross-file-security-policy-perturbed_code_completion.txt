while (initial_gids_to_retain->next (&gid))
  {
    if (gid < face->get_max_glyph_id()) // Ensure gid is within valid range
    {
      full_glyphs_to_retain->add (gid);
      // Add component glyphs if gid is a composite glyph
      hb_codepoint_t component_gid;
      if (glyf.get_component_glyphs (gid, &component_gid))
      {
        full_glyphs_to_retain->add (component_gid);
      }
    }
  }

  hb_set_destroy (initial_gids_to_retain);
  glyphs->clear();
  full_glyphs_to_retain->next (nullptr, glyphs);
  
  cmap.fini();
  glyf.fini();
  
  return full_glyphs_to_retain;
