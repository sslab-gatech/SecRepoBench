while (ctx->pos < ctx->end) {
    if (*ctx->pos == quote_char) {
        if (ctx->pos > p && *(ctx->pos - 1) != '\\') {
            break;
        }
    } else if (*ctx->pos == '\\') {
        num_escapes++;
    }
    ctx->pos++;
}
if (ctx->pos >= ctx->end) {
    PLIST_OSTEP_ERR("Missing closing quote at offset %ld\n", ctx->pos - ctx->start);
    ctx->err++;
    plist_free_data(data);
    goto err_out;
}
size_t slen = ctx->pos - p;
char* strbuf = (char*)malloc(slen + 1);
if (!strbuf) {
    PLIST_OSTEP_ERR("Out of memory when parsing string at offset %ld\n", p - ctx->start);
    ctx->err++;
    plist_free_data(data);
    goto err_out;
}
ctx->pos++;
