size_t slen = 0;
            while (ctx->pos < ctx->end) {
                if (*ctx->pos == quote_char) {
                    ctx->pos++;
                    break;
                }
                if (*ctx->pos == '\\') {
                    num_escapes++;
                    ctx->pos++;
                }
                ctx->pos++;
            }
            slen = ctx->pos - p - num_escapes;
            char *strbuf = (char *)malloc(slen + 1);
            if (!strbuf) {
                ctx->err++;
                break;
            }
            ctx->pos = p;
            size_t i = 0;
            while (i < slen) {
                if (p[i] == '\\') {
                    i++;
                    switch (p[i]) {
                        case '0': strbuf[i++] = '\0'; break;
                        case 'a': strbuf[i++] = '\a'; break;
                        case 'b': strbuf[i++] = '\b'; break;
                        case 'f': strbuf[i++] = '\f'; break;
                        case 'n': strbuf[i++] = '\n'; break;
                        case 'r': strbuf[i++] = '\r'; break;
                        case 't': strbuf[i++] = '\t'; break;
                        case 'v': strbuf[i++] = '\v'; break;
                        case '"': strbuf[i++] = '"'; break;
                        case '\'': strbuf[i++] = '\''; break;
                        default: strbuf[i++] = p[i]; break;
                    }
                } else {
                    strbuf[i++] = p[i];
                }
            }
            strbuf[i] = '\0';
            data->type = PLIST_STRING;
            data->strval = strbuf;
            data->length = slen;
            *plist = plist_new_node(data);
