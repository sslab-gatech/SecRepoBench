/* traverse until we find the matching quote_char, counting escapes */
            const char *startp = ctx->pos;
            while (ctx->pos < ctx->end) {
                if (*ctx->pos == '\\') {
                    ctx->pos++;
                    if (ctx->pos < ctx->end) {
                        ctx->pos++;
                    }
                    num_escapes++;
                } else if (*ctx->pos == quote_char) {
                    break;
                } else {
                    ctx->pos++;
                }
            }
            if (ctx->pos >= ctx->end || *ctx->pos != quote_char) {
                PLIST_OSTEP_ERR("Missing terminating quote at offset %ld\n", ctx->pos - ctx->start);
                ctx->err++;
                plist_free_data(data);
                goto err_out;
            }
            size_t slen = ctx->pos - startp;
            ctx->pos++; /* skip closing quote */
            char *strbuf = (char *)malloc(slen + 1);
            if (!strbuf) {
                PLIST_OSTEP_ERR("Memory allocation error parsing quoted string\n");
                ctx->err++;
                plist_free_data(data);
                goto err_out;
            }
