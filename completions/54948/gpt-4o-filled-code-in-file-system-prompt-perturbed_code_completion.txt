while (ctx->pos < ctx->end) {
                if (*ctx->pos == quote_char) {
                    break;
                }
                if (*ctx->pos == '\\') {
                    num_escapes++;
                    ctx->pos++;
                }
                ctx->pos++;
            }
            if (ctx->pos >= ctx->end) {
                PLIST_OSTEP_ERR("EOF while parsing quoted string at offset %ld\n", ctx->pos - ctx->start);
                ctx->err++;
                plist_free_data(data);
                break;
            }
            size_t slen = ctx->pos - p;
            char *strbuf = (char *)malloc(slen + 1);
            if (!strbuf) {
                PLIST_OSTEP_ERR("Memory allocation failed while parsing quoted string\n");
                ctx->err++;
                plist_free_data(data);
                break;
            }
            ctx->pos++;
