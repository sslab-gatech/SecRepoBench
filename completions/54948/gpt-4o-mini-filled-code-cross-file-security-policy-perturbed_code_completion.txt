size_t slen = 0;
            char *strbuf = malloc(256); // Allocate initial buffer for the string
            if (!strbuf) {
                ctx->err++;
                PLIST_OSTEP_ERR("Memory allocation failed for string buffer at offset %ld\n", ctx->pos - ctx->start);
                break;
            }
            while (ctx->pos < ctx->end) {
                if (*ctx->pos == quote_char) {
                    ctx->pos++;
                    break;
                }
                if (*ctx->pos == '\\') {
                    num_escapes++;
                    ctx->pos++;
                    if (ctx->pos >= ctx->end) {
                        PLIST_OSTEP_ERR("Unexpected end of string at offset %ld\n", ctx->pos - ctx->start);
                        ctx->err++;
                        free(strbuf);
                        break;
                    }
                }
                strbuf[slen++] = *ctx->pos++;
                if (slen >= 256) { // Resize if necessary
                    strbuf = realloc(strbuf, slen + 256);
                    if (!strbuf) {
                        ctx->err++;
                        PLIST_OSTEP_ERR("Memory reallocation failed for string buffer at offset %ld\n", ctx->pos - ctx->start);
                        break;
                    }
                }
            }
            if (ctx->err) {
                free(strbuf);
                break;
            }
            strbuf[slen] = '\0';
            data->type = PLIST_STRING;
            data->strval = strbuf;
            data->length = slen;
