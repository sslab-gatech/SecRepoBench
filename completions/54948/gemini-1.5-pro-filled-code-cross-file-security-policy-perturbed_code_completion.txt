while (ctx->pos < ctx->end) {
                if (*ctx->pos == quote_char) {
                    if (ctx->pos > p && *(ctx->pos - 1) == '\\') {
                        num_escapes++;
                    } else {
                        break;
                    }
                } else if (*ctx->pos == '\\') {
                    num_escapes++;
                }
                ctx->pos++;
            }
            if (ctx->pos >= ctx->end) {
                PLIST_OSTEP_ERR("Missing terminating '%c' at offset %ld\n", quote_char, ctx->pos - ctx->start);
                ctx->err++;
                goto err_out;
            }
            size_t slen = ctx->pos - p;
            char *strbuf = (char *)malloc(slen + 1);
            if (!strbuf) {
                ctx->err++;
                goto err_out;
            }
            ctx->pos++;
