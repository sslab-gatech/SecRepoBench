size_t slen = 0;
            while (ctx->pos < ctx->end) {
                if (*ctx->pos == '\\') {
                    num_escapes++;
                    ctx->pos++;
                    if (ctx->pos >= ctx->end) {
                        PLIST_OSTEP_ERR("Unexpected end of string after escape character at offset %ld\n", ctx->pos - ctx->start);
                        ctx->err++;
                        goto err_out;
                    }
                    ctx->pos++;
                } else if (*ctx->pos == quote_char) {
                    break;
                } else {
                    ctx->pos++;
                }
            }
            if (ctx->pos >= ctx->end) {
                PLIST_OSTEP_ERR("Missing terminating quote at offset %ld\n", ctx->pos - ctx->start);
                ctx->err++;
                goto err_out;
            }
            slen = ctx->pos - p;
            ctx->pos++;
            char *strbuf = (char *)malloc(slen + 1);
            if (strbuf == NULL) {
                PLIST_OSTEP_ERR("Out of memory at offset %ld\n", ctx->pos - ctx->start);
                ctx->err++;
                goto err_out;
            }
