while (ctx->pos < ctx->end) {
                if (*ctx->pos == quote_char) {
                    break;
                }
                if (*ctx->pos == '\\') {
                    num_escapes++;
                    ctx->pos++;
                    if (ctx->pos >= ctx->end) {
                        PLIST_OSTEP_ERR("Unexpected end of input in escape sequence at offset %ld\n", ctx->pos - ctx->start);
                        ctx->err++;
                        break;
                    }
                }
                ctx->pos++;
            }
            if (ctx->pos >= ctx->end || *ctx->pos != quote_char) {
                PLIST_OSTEP_ERR("Missing terminating quote at offset %ld\n", ctx->pos - ctx->start);
                ctx->err++;
                plist_free_data(data);
                goto err_out;
            }
            size_t slen = ctx->pos - p;
            char *strbuf = (char *)malloc(slen + 1);
            if (!strbuf) {
                PLIST_OSTEP_ERR("Memory allocation failed for string buffer\n");
                ctx->err++;
                plist_free_data(data);
                goto err_out;
            }
            ctx->pos++;
