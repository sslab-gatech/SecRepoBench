size_t slen = 0;
            while (ctx->pos < ctx->end && *ctx->pos != quote_char) {
                if (*ctx->pos == '\\') {
                    num_escapes++;
                    ctx->pos++;
                }
                ctx->pos++;
            }
            if (ctx->pos >= ctx->end) {
                PLIST_OSTEP_ERR("Missing terminating quote at offset %ld\n", ctx->pos - ctx->start);
                ctx->err++;
                goto err_out;
            }
            slen = ctx->pos - p;
            char *strbuf = (char *)malloc(slen + 1 + num_escapes * 2); // +1 for null terminator, +2 for each escape sequence
            if (!strbuf) {
                PLIST_OSTEP_ERR("Memory allocation failed\n");
                ctx->err++;
                goto err_out;
            }
            ctx->pos++;
