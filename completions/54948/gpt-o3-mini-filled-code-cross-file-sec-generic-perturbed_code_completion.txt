size_t slen = 0;
char *strbuf = NULL;
const char *q = ctx->pos;
while (ctx->pos < ctx->end) {
    if (*ctx->pos == '\\') {
        num_escapes++;
        ctx->pos++;
        if (ctx->pos >= ctx->end) {
            PLIST_OSTEP_ERR("Unterminated escape sequence in quoted string at offset %ld\n", ctx->pos - ctx->start);
            ctx->err++;
            break;
        }
    } else if (*ctx->pos == quote_char) {
        break;
    }
    ctx->pos++;
    slen++;
}
if (ctx->err) {
    goto err_out;
}
if (ctx->pos >= ctx->end || *ctx->pos != quote_char) {
    PLIST_OSTEP_ERR("Missing terminating quote '%c' at offset %ld\n", quote_char, ctx->pos - ctx->start);
    ctx->err++;
    goto err_out;
}
ctx->pos++;  /* Skip the closing quote */
strbuf = (char *)malloc(slen + 1);
if (!strbuf) {
    PLIST_OSTEP_ERR("Memory allocation failed at offset %ld\n", ctx->pos - ctx->start);
    ctx->err++;
    goto err_out;
}
