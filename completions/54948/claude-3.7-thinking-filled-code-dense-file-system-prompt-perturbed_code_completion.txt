while (ctx->pos < ctx->end) {
    if (*ctx->pos == '\\') {
        num_escapes++;
        ctx->pos++; // Skip the backslash
        if (ctx->pos >= ctx->end) {
            PLIST_OSTEP_ERR("Unexpected end of string at offset %ld\n", ctx->pos - ctx->start);
            ctx->err++;
            goto err_out;
        }
    } else if (*ctx->pos == quote_char) {
        break; // Found the closing quote
    }
    ctx->pos++;
}

if (ctx->pos >= ctx->end) {
    PLIST_OSTEP_ERR("Missing closing %c quote at offset %ld\n", quote_char, p - ctx->start);
    ctx->err++;
    goto err_out;
}

size_t slen = ctx->pos - p;
char *strbuf = (char *)malloc(slen + 1);
if (!strbuf) {
    PLIST_OSTEP_ERR("Out of memory when allocating string buffer\n");
    ctx->err++;
    goto err_out;
}
ctx->pos++; // Move past the closing quote
