case PLIST_REAL:
    val = (char*)malloc(64);
    val_len = dtostr(val, 64, node_data->realval);
    str_buf_append(*outbuf, val, val_len);
    free(val);
    break;

case PLIST_UINT:
    val = (char*)malloc(64);
    if (node_data->length == 16) {
        val_len = snprintf(val, 64, "%"PRIu64, node_data->intval);
    } else {
        val_len = snprintf(val, 64, "%"PRIi64, node_data->intval);
    }
    str_buf_append(*outbuf, val, val_len);
    free(val);
    break;
