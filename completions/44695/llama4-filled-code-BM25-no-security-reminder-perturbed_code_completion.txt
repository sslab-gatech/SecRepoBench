if (endp == str_end) {
    primitivevalue = plist_new_uint64((uint64_t)intpart);
} else {
    double dblval = parse_double(str_val, str_end, &endp);
    if (endp != str_end) {
        PLIST_JSON_ERR("%s: invalid numeric value '%.*s' encountered\n", __func__, (int)str_len, str_val);
    } else {
        plist_data_t data = plist_new_plist_data();
        data->type = PLIST_REAL;
        data->realval = dblval;
        primitivevalue = plist_new_node(data);
    }
}
