if (str_val[str_len - 1] == '.') {

    char* endp = (char*)str_val;
    int64_t intpart = parse_decimal(str_val, str_end, &endp);

        PLIST_JSON_ERR("%s: invalid floating point value '%.*s' encountered\n", __func__, (int)str_len, str_val);
        return NULL;
    }
    // Check for exponent part
    if (*endp == 'e' || *endp == 'E') {
        // Exponent part found, create a plist node for a real number
        plist_data_t data = plist_new_plist_data();
        data->type = PLIST_REAL;
        data->real = strtod(str_val, NULL);
        primitivevalue = plist_new_node(data);
    } else {
        // No exponent part, create a plist node for a real number
        plist_data_t data = plist_new_plist_data();
        data->type = PLIST_REAL;
        data->real = strtod(str_val, NULL);
        primitivevalue = plist_new_node(data);
    }
} else if (str_val[str_len - 1] == 'f' || str_val[str_len - 1] == 'F') {

    char* endp = (char*)str_val;
    int64_t intpart = parse_decimal(str_val, str_end, &endp);

        return NULL;
    }
    // Check for exponent part
    if (*endp == 'e' || *endp == 'E') {
        // Exponent part found, create a plist node for a real number
        plist_data_t data = plist_new_plist_data();
        data->type = PLIST_REAL;
        data->real = strtod(str_val, NULL);
        primitivevalue = plist_new_node(data);

        data->type = PLIST_REAL;
        data->real = strtod(str_val, NULL);
        primitivevalue = plist_new_node(data);
