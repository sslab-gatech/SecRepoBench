if (endp < str_end && *endp == '.') {
            // Parse fractional part
            double fraction = 0.0;
            double divisor = 10.0;
            endp++;
            while (endp < str_end && isdigit(*endp)) {
                fraction += (*endp - '0') / divisor;
                divisor *= 10.0;
                endp++;
            }
            intpart += (intpart < 0 ? -fraction : fraction);
        }

        if (endp < str_end && (*endp == 'e' || *endp == 'E')) {
            // Parse exponent part
            endp++;
            int exp_sign = 1;
            if (endp < str_end && (*endp == '-' || *endp == '+')) {
                exp_sign = (*endp == '-') ? -1 : 1;
                endp++;
            }
            int exponent = 0;
            while (endp < str_end && isdigit(*endp)) {
                exponent = exponent * 10 + (*endp - '0');
                endp++;
            }
            intpart *= pow(10, exp_sign * exponent);
        }

        if (endp == str_end) {
            if (floor(intpart) == intpart) {
                // It's an integer
                if (intpart >= 0) {
                    primitivevalue = plist_new_uint((uint64_t)intpart);
                } else {
                    primitivevalue = plist_new_int((int64_t)intpart);
                }
            } else {
                // It's a floating point number
                primitivevalue = plist_new_real(intpart);
            }
        } else {
            PLIST_JSON_ERR("%s: invalid numeric value '%.*s' encountered\n", __func__, (int)str_len, str_val);
