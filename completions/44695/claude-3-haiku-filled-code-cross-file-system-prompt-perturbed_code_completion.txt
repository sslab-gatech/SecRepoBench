// Determine if the parsed numeric value is an integer or floating point.
// If it's an integer, create a plist node for an unsigned integer.
// If it's a floating point, convert and accumulate the fractional and exponent parts.
// Create a plist node for a real number if the value is a valid floating point number.
// Handle errors for invalid characters or unrepresentable floating point values.
int64_t intpart = parse_decimal(str_val, str_end, &endp);
if (endp == str_end) {
    // Integer value
    primitivevalue = plist_new_uint(intpart);
} else {
    // Floating point value
    double floatpart = 0.0;
    if (*endp == '.') {
        endp++;
        double divisor = 1.0;
        while (endp < str_end && isdigit(*endp)) {
            floatpart = floatpart * 10.0 + (*endp - '0');
            divisor *= 10.0;
            endp++;
        }
        floatpart /= divisor;
    }
    if (*endp == 'e' || *endp == 'E') {
        endp++;
        int exponent = 0;
        int sign = 1;
        if (*endp == '+') {
            endp++;
        } else if (*endp == '-') {
            sign = -1;
            endp++;
        }
        while (endp < str_end && isdigit(*endp)) {
            exponent = exponent * 10 + (*endp - '0');
            endp++;
        }
        floatpart *= pow(10.0, sign * exponent);
    }
    if (endp != str_end) {
        PLIST_JSON_ERR("%s: invalid character '%c' in numeric value\n", __func__, *endp);
        return NULL;
    }
    double value = intpart + floatpart;
    if (value >= 0.0 && value <= UINT64_MAX) {
        primitivevalue = plist_new_uint((uint64_t)value);
    } else if (value >= INT64_MIN && value <= INT64_MAX) {
        primitivevalue = plist_new_int((int64_t)value);
    } else {
        PLIST_JSON_ERR("%s: numeric value out of range\n", __func__);
        return NULL;
    }
}
