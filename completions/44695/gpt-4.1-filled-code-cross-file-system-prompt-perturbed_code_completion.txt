// Check if the number is an integer (no '.' or 'e'/'E' in the string)
        const char* p = endp;
        int is_float = 0;
        while (p < str_end) {
            if (*p == '.' || *p == 'e' || *p == 'E') {
                is_float = 1;
                break;
            }
            p++;
        }
        if (!is_float && endp == str_end) {
            // Integer value
            if (intpart >= 0) {
                primitivevalue = plist_new_uint((uint64_t)intpart);
            } else {
                primitivevalue = plist_new_int(intpart);
            }
        } else {
            // Floating point value
            char numbuf[64];
            size_t numlen = str_len < sizeof(numbuf)-1 ? str_len : sizeof(numbuf)-1;
            memcpy(numbuf, str_val, numlen);
            numbuf[numlen] = '\0';
            char* endptr = NULL;
            double dval = strtod(numbuf, &endptr);
            if (endptr == numbuf + numlen) {
                primitivevalue = plist_new_real(dval);
            } else {
                PLIST_JSON_ERR("%s: invalid number '%.*s'\n", __func__, (int)str_len, str_val);
            }
        }
