if (endp < str_end) {
            // Floating point value
            if (*endp == '.') {
                endp++;
                int64_t fracpart = parse_decimal(endp, str_end, &endp);
                int64_t exp = 0;
                if (*endp == 'e' || *endp == 'E') {
                    endp++;
                    exp = parse_decimal(endp, str_end, &endp);
                }

                if (endp == str_end) {
                    double realValue = intpart + (double)fracpart / pow(10, endp - (str_val + str_len - (endp - str_val) + 1) - (str_val+1));
                    if (exp != 0) {
                        realValue *= pow(10, exp);
                    }
                    if (isfinite(realValue)) {
                        primitivevalue = plist_new_real(realValue);
                    } else {
                        PLIST_JSON_ERR("%s: invalid floating point value '%.*s'\n", __func__, (int)str_len, str_val);
                    }
                } else {
                    PLIST_JSON_ERR("%s: invalid floating point value '%.*s'\n", __func__, (int)str_len, str_val);
                }
            } else {
                PLIST_JSON_ERR("%s: invalid numeric value '%.*s'\n", __func__, (int)str_len, str_val);
            }
        } else {
            // Integer value
            primitivevalue = plist_new_uint(intpart);
        }
