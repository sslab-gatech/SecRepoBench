if (intpart == INT64_MIN) {
            PLIST_JSON_ERR("%s: integer overflow\n", __func__);
            return NULL;
        }
        if (endp > str_end) {
            PLIST_JSON_ERR("%s: invalid number format\n", __func__);
            return NULL;
        }
        if (endp != str_end) {
            // Check for a decimal point to determine if it's a real number
            if (*endp == '.') {
                endp++;
                double frac = 0.0;
                double divisor = 10.0;
                while (endp < str_end && isdigit(*endp)) {
                    frac += (*endp - '0') / divisor;
                    divisor *= 10.0;
                    endp++;
                }
                // Create a plist node for a real number
                primitivevalue = plist_new_real(intpart + frac);
            } else {
                // Create a plist node for an unsigned integer
                primitivevalue = plist_new_uint(intpart);
            }
        } else {
            // Handle the case where the number is not valid
            PLIST_JSON_ERR("%s: invalid number format\n", __func__);
            return NULL;
        }
