if (tlvDataBasePtr == nullptr || tlvDataLength == 0) {
				TLVRecordType nullRecord;
				nullRecord.assign(nullptr);
				return nullRecord;
			}

			if (tlvDataBasePtr[0] >> 6 == 0x02) { // Tag Type Length
				uint16_t tag = (tlvDataBasePtr[0] << 8) | tlvDataBasePtr[1];
				uint16_t length = (tlvDataBasePtr[2] << 8) | tlvDataBasePtr[3];
				TLVRecordType record;
				record.assign(tlvDataBasePtr, tag, length);
				return record;
			}

			// If the first byte does not indicate a valid TLV record, return a null record
			TLVRecordType nullRecord;
			nullRecord.assign(nullptr);
