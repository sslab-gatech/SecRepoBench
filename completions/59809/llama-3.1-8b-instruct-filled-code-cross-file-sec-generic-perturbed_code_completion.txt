TLVRecordType getFirstTLVRecord(uint8_t* tlvDataBasePtr, size_t tlvDataLength) const
{
    if (tlvDataBasePtr == nullptr || tlvDataLength == 0)
    {
        TLVRecordType record;
        record.assign(NULL);
        return record;
    }

    // Check if the TLV record is present in the byte stream
    if (tlvDataLength < sizeof(TLVRecordType))
    {
        TLVRecordType record;
        record.assign(NULL);
        return record;
    }

    // Extract the TLV record from the byte stream
    uint8_t* recordPtr = tlvDataBasePtr;
    TLVRecordType record(recordPtr, tlvDataLength);

    // Check if the TLV record is valid
    if (!record.isValid())
    {
        record.assign(NULL);
        return record;
    }

    return record;
}
