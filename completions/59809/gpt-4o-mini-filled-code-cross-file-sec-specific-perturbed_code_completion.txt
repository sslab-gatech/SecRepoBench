if (tlvDataBasePtr == NULL || tlvDataLength == 0) {
    return TLVRecordType(); // Return a logically NULL instance
}

size_t offset = 0;
if (offset + sizeof(TLVRecordType) > tlvDataLength) {
    return TLVRecordType(); // Prevent reading past the end of the buffer
}

TLVRecordType firstRecord;
memcpy(&firstRecord, tlvDataBasePtr + offset, sizeof(TLVRecordType));
return firstRecord;
