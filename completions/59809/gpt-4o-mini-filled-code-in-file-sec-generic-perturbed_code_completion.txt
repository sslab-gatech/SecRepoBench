if (tlvDataBasePtr == nullptr || tlvDataLength == 0)
			{
				return TLVRecordType(NULL); // Return a logically NULL instance
			}

			TLVRawData* rawData = reinterpret_cast<TLVRawData*>(tlvDataBasePtr);
			if (tlvDataLength < sizeof(TLVRawData))
			{
				return TLVRecordType(NULL); // Not enough data for a valid TLV record
			}

			size_t totalSize = sizeof(TLVRawData) + rawData->recordLen;
			if (totalSize > tlvDataLength)
			{
				return TLVRecordType(NULL); // Not enough data for the specified length
			}

			return TLVRecordType(tlvDataBasePtr); // Return a valid TLVRecordType instance
