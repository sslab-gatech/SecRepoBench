if (tlvDataBasePtr == nullptr || tlvDataLength == 0) {
    TLVRecordType nullRecord;
    nullRecord.assign(nullptr);
    return nullRecord;
}

if (offset + sizeof(TLVRecordType) > tlvDataLength) {
    TLVRecordType nullRecord;
    nullRecord.assign(nullptr);
    return nullRecord;
}

TLVRecordType firstRecord;
firstRecord.assign(tlvDataBasePtr + offset);
return firstRecord;
