def xsltSwapTopCompMatch(comp):
    i = 0
    j = comp.nbStep - 1

    if j > 0:
        cur = comp.steps[i].value
        next = comp.steps[j].value
        comp.steps[i].value = next
        comp.steps[j].value = cur

        cur = comp.steps[i].value2
        next = comp.steps[j].value2
        comp.steps[i].value2 = next
        comp.steps[j].value2 = cur

        cur = comp.steps[i].value3
        next = comp.steps[j].value3
        comp.steps[i].value3 = next
        comp.steps[j].value3 = cur

        op = comp.steps[i].op
        next = comp.steps[j].op
        comp.steps[i].op = next
        comp.steps[j].op = op

        expr = comp.steps[i].comp
        next = comp.steps[j].comp
        comp.steps[i].comp = next
        comp.steps[j].comp = expr

        t = comp.steps[i].previousExtra
        next = comp.steps[j].previousExtra
        comp.steps[i].previousExtra = next
        comp.steps[j].previousExtra = t

        t = comp.steps[i].indexExtra
        next = comp.steps[j].indexExtra
        comp.steps[i].indexExtra = next
        comp.steps[j].indexExtra = t

        t = comp.steps[i].lenExtra
        next = comp.steps[j].lenExtra
        comp.steps[i].lenExtra = next
        comp.steps[j].lenExtra = t
