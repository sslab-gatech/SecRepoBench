if (tmp == NULL) {
	    xsltTransformError(NULL, parserCtx->style, parserCtx->elem,
		"xsltCompMatchAdd: memory reallocation failed\n");
	    if (parserCtx->style != NULL)
		parserCtx->style->errors++;
	    return -1;
	}
	comp->steps = tmp;
	comp->maxStep *= 2;
	// Initialize the newly allocated memory to zero
	memset(comp->steps + comp->nbStep, 0, (comp->maxStep - comp->nbStep) * sizeof(xsltStepOp));
