case DW_OP_const_type:           /* DWARF5 */
    case DW_OP_GNU_const_type:     /* DWARF5 */
    case DW_OP_regval_type:        /* DWARF5 */
    case DW_OP_GNU_regval_type:    /* DWARF5 */
    case DW_OP_convert:            /* DWARF5 */
    case DW_OP_reinterpret:        /* DWARF5 */
    case DW_OP_GNU_parameter_ref:  /* DWARF5 */
        {
            Dwarf_Small iplen = offset_size;
            if (version_stamp == DW_CU_VERSION2 /* 2 */) {
                iplen = address_size;
            }
            READ_UNALIGNED_CK(dbg, operand1, Dwarf_Unsigned, loc_ptr, iplen, error_ptr, section_end);
            loc_ptr = loc_ptr + iplen;
            if (loc_ptr > section_end) {
                _dwarf_error(dbg, error_ptr, DW_DLE_LOCEXPR_OFF_SECTION_END);
                return DW_DLV_ERROR;
            }
            offset = offset + iplen;

            if (atom == DW_OP_const_type || atom == DW_OP_GNU_const_type) {
                /*  The value is a CU-relative DIE offset. */
                DECODE_LEB128_SWORD_LEN_CK(loc_ptr, operand2, leb128_length, dbg, error_ptr, section_end);
                offset = offset + leb128_length;
            } else if (atom == DW_OP_regval_type || atom == DW_OP_GNU_regval_type) {
                /*  The value is a CU-relative DIE offset and a register number. */
                DECODE_LEB128_SWORD_LEN_CK(loc_ptr, operand2, leb128_length, dbg, error_ptr, section_end);
                offset = offset + leb128_length;
                DECODE_LEB128_UWORD_LEN_CK(loc_ptr, operand3, leb128_length, dbg, error_ptr, section_end);
                offset = offset + leb128_length;
            } else if (atom == DW_OP_convert || atom == DW_OP_reinterpret) {
                /*  The value is a CU-relative DIE offset or zero. */
                DECODE_LEB128_SWORD_LEN_CK(loc_ptr, operand2, leb128_length, dbg, error_ptr, section_end);
                offset = offset + leb128_length;
            } else if (atom == DW_OP_GNU_parameter_ref) {
                /*  The value is a 4-byte unsigned integer. */
                READ_UNALIGNED_CK(dbg, operand2, Dwarf_Unsigned, loc_ptr, 4, error_ptr, section_end);
                loc_ptr = loc_ptr + 4;
                if (loc_ptr > section_end) {
                    _dwarf_error(dbg, error_ptr, DW_DLE_LOCEXPR_OFF_SECTION_END);
                    return DW_DLV_ERROR;
                }
                offset = offset + 4;
            }
        }
        break;
    case DW_OP_addrx :           /* DWARF5 */
    case DW_OP_GNU_addr_index :  /* DWARF5 */
        {
            DECODE_LEB128_UWORD_LEN_CK(loc_ptr, operand1, leb128_length, dbg, error_ptr, section_end);
            offset = offset + leb128_length;
        }
        break;
    case DW_OP_constx :          /* DWARF5 */
    case DW_OP_GNU_const_index : /* DWARF5 */
        {
            DECODE_LEB128_UWORD_LEN_CK(loc_ptr, operand1, leb128_length, dbg, error_ptr, section_end);
            offset = offset + leb128_length;
        }
        break;
