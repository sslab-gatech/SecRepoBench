case DW_OP_const_type:
    case DW_OP_GNU_const_type: {
        DECODE_LEB128_UWORD_LEN_CK(loc_ptr, operand1, leb128_length,
            dbg, error_ptr, section_end);
        offset += leb128_length;
        /* operand1 is the CU-relative DIE offset */
        DECODE_LEB128_UWORD_LEN_CK(loc_ptr, operand2, leb128_length,
            dbg, error_ptr, section_end);
        offset += leb128_length;
        /* operand2 is the size of the following data block */
        /* Save pointer to data block in operand3 */
        operand3 = (Dwarf_Unsigned)(uintptr_t)loc_ptr;
        /* Advance past the data block */
        offset += operand2;
        loc_ptr += operand2;
        if (loc_ptr > section_end) {
            _dwarf_error(dbg, error_ptr, DW_DLE_LOCEXPR_OFF_SECTION_END);
            return DW_DLV_ERROR;
        }
        break;
    }
    case DW_OP_regval_type:
    case DW_OP_GNU_regval_type: {
        DECODE_LEB128_UWORD_LEN_CK(loc_ptr, operand1, leb128_length,
            dbg, error_ptr, section_end);
        offset += leb128_length;
        /* operand1 is the register number */
        DECODE_LEB128_UWORD_LEN_CK(loc_ptr, operand2, leb128_length,
            dbg, error_ptr, section_end);
        offset += leb128_length;
        /* operand2 is the CU-relative DIE offset */
        break;
    }
    case DW_OP_convert:
    case DW_OP_GNU_convert:
    case DW_OP_reinterpret:
    case DW_OP_GNU_reinterpret: {
        DECODE_LEB128_UWORD_LEN_CK(loc_ptr, operand1, leb128_length,
            dbg, error_ptr, section_end);
        offset += leb128_length;
        /* operand1 is the DIE offset (or zero) */
        break;
    }
    case DW_OP_GNU_parameter_ref: {
        READ_UNALIGNED_CK(dbg, operand1, Dwarf_Unsigned,
            loc_ptr, 4, error_ptr, section_end);
        loc_ptr += 4;
        offset += 4;
        break;
