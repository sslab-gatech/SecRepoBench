DECODE_LEB128_UWORD_LEN_CK(loc_ptr, operand1, leb128_length,
            dbg, error_ptr, section_end);
        if (operand1 > UINT32_MAX) {
            _dwarf_error(dbg, error_ptr, DW_DLE_LOCEXPR_BAD_OPERAND);
            return DW_DLV_ERROR;
        }
        offset += leb128_length;

        DECODE_LEB128_UWORD_LEN_CK(loc_ptr, operand2, leb128_length,
            dbg, error_ptr, section_end);
        offset += leb128_length;
        break;

    case DW_OP_convert:
    case DW_OP_reinterpret:
        DECODE_LEB128_UWORD_LEN_CK(loc_ptr, operand1, leb128_length,
            dbg, error_ptr, section_end);
        if (operand1 > UINT32_MAX) {
            _dwarf_error(dbg, error_ptr, DW_DLE_LOCEXPR_BAD_OPERAND);
            return DW_DLV_ERROR;
        }
        offset += leb128_length;
        break;

    case DW_OP_GNU_parameter_ref:
        if (loc_ptr + sizeof(Dwarf_Unsigned) > section_end) {
            _dwarf_error(dbg, error_ptr, DW_DLE_LOCEXPR_OFF_SECTION_END);
            return DW_DLV_ERROR;
        }
        READ_UNALIGNED_CK(dbg, operand1, Dwarf_Unsigned, loc_ptr,
            sizeof(Dwarf_Unsigned), error_ptr, section_end);
        loc_ptr += sizeof(Dwarf_Unsigned);
        offset += sizeof(Dwarf_Unsigned);
        break;
