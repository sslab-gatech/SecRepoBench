case DW_OP_const_type:           /* DWARF5 */
    case DW_OP_GNU_const_type:     /* DWARF5 */
    case DW_OP_regval_type:        /* DWARF5 */
    case DW_OP_GNU_regval_type:    /* DWARF5 */
    case DW_OP_convert:            /* DWARF5 */
    case DW_OP_reinterpret:        /* DWARF5 */
    case DW_OP_GNU_parameter_ref:  /* DWARF5 */
    {
        Dwarf_Small iplen = offset_size;
        if (version_stamp == DW_CU_VERSION2 /* 2 */ ) {
            iplen = address_size;
        }
        READ_UNALIGNED_CK(dbg, operand1, Dwarf_Unsigned, loc_ptr, iplen,
            error_ptr,section_end);
        loc_ptr = loc_ptr + iplen;
        if (loc_ptr > section_end) {
            _dwarf_error(dbg,error_ptr,DW_DLE_LOCEXPR_OFF_SECTION_END);
            return DW_DLV_ERROR;
        }
        offset = offset + iplen;

        if (atom == DW_OP_const_type || atom == DW_OP_GNU_const_type) {
            /*  The value is a CU-relative DIE offset
                (a simple unsigned integer) in DWARF3,4
                followed by a data block size, then obtain
                the data block based on the size, adjusting
                the location pointer and offset accordingly. */
            DECODE_LEB128_UWORD_LEN_CK(loc_ptr, operand2,leb128_length,
                dbg,error_ptr,section_end);
            offset = offset + leb128_length;

            /*  Second operand is block of 'operand2' bytes of stuff. */
            /*  This using the second operand as a pointer
                is quite ugly. */
            /*  This gets an ugly compiler warning. Sorry. */
            operand3 = (Dwarf_Unsigned)(uintptr_t)loc_ptr;
            offset = offset + operand2;
            loc_ptr = loc_ptr + operand2;
            if (loc_ptr > section_end) {
                _dwarf_error(dbg,error_ptr,DW_DLE_LOCEXPR_OFF_SECTION_END);
                return DW_DLV_ERROR;
            }
        } else {
            /*  The value is a CU-relative DIE offset
                (a simple unsigned integer) in DWARF3,4
                followed by a signed leb128 offset. */
            DECODE_LEB128_SWORD_LEN_CK(loc_ptr, operand2,leb128_length,
                dbg,error_ptr,section_end);
            offset = offset + leb128_length;
        }
        break;
    }
