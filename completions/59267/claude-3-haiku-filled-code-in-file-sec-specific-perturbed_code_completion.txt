{
    memcpy(outDataPtr, compressedUnknownBuf, unknownCompressedSize);
    outDataPtr += unknownCompressedSize;
}

if (acCompressedSize > 0)
{
    memcpy(outDataPtr, compressedAcBuf, acCompressedSize);
    outDataPtr += acCompressedSize;
}

if (dcCompressedBytes > 0)
{
    memcpy(outDataPtr, compressedDcBuf, dcCompressedBytes);
    outDataPtr += dcCompressedBytes;
}

if (rleCompressedSize > 0)
{
    memcpy(outDataPtr, compressedRleBuf, rleCompressedSize);
    outDataPtr += rleCompressedSize;
}
