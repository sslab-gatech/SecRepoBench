// Allocate the output buffer for uncompressed data
    if (uncompressed_size > SIZE_MAX) {
        return EXR_ERR_OUT_OF_MEMORY;
    }

    me->_outBuffer = me->alloc_fn(uncompressed_size);
    if (!me->_outBuffer) {
        return EXR_ERR_OUT_OF_MEMORY;
    }
    memset(me->_outBuffer, 0, uncompressed_size);

    outBufferEnd = (uint8_t*)me->_outBuffer;
    packedAcBufferEnd = me->_packedAcBuffer;
    packedDcBufferEnd = me->_packedDcBuffer;

    // Initialize pointers for compressed data sections
    dataPtr = inPtr + headerSize;
    dataLeft = iSize - headerSize;

    // Ensure that the sizes of compressed data do not exceed the available data
    if (unknownCompressedSize > dataLeft || 
        acCompressedSize > dataLeft || 
        dcCompressedBytes > dataLeft || 
        rleCompressedSize > dataLeft) {
        return EXR_ERR_CORRUPT_CHUNK;
    }
