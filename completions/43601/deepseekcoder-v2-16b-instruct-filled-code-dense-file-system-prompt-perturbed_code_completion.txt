// Search for the specified tag in the profile directory. 
// If the tag is not found, handle the error.
TagDescriptor = _cmsGetTagDescriptor(Icc-> ContextID, tagSignature);
if (TagDescriptor == NULL) {
    char String[5];
    _cmsTagSignature2String(String, tagSignature);
    cmsSignalError(Icc ->ContextID, cmsERROR_UNKNOWN_EXTENSION, "Unknown tag type '%s' found.", String);
    goto Error;     // Unsupported.
}
