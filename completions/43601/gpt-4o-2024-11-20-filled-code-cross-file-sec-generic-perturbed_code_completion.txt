// Search for the specified tag in the profile directory.
    n = _cmsSearchTag(Icc, tagSignature, TRUE);
    if (n < 0) {
        // Tag not found, signal an error and unlock the mutex
        char String[5];
        _cmsTagSignature2String(String, tagSignature);
        cmsSignalError(Icc->ContextID, cmsERROR_UNKNOWN_EXTENSION, "Tag '%s' not found.", String);
        _cmsUnlockMutex(Icc->ContextID, Icc->UsrMutex);
        return NULL;
    }

    // Check if the tag has already been read
    if (Icc->TagPtrs[n] != NULL) {
        // If already read, unlock the mutex and return the cached pointer
        _cmsUnlockMutex(Icc->ContextID, Icc->UsrMutex);
        return Icc->TagPtrs[n];
    }
