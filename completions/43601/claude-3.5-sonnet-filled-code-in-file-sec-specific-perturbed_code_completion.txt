n = _cmsSearchTag(Icc, tagSignature, TRUE);
if (n < 0) {
    cmsSignalError(Icc->ContextID, cmsERROR_UNKNOWN_EXTENSION, "Tag '%x' not found", tagSignature);
    goto Error;
}

// Check if the tag has already been loaded
if (Icc->TagPtrs[n] != NULL) {
    _cmsUnlockMutex(Icc->ContextID, Icc->UsrMutex);
    return Icc->TagPtrs[n];
}
