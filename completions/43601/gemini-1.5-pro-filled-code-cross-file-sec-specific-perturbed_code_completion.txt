n = _cmsSearchTag(Icc, tagSignature, FALSE);
    if (n < 0) {

        // Not found - Create a fake tag descriptor to let user to read the tag anyway.
        // If the tag is not an lcms-supported tag, ElemCount would be zero.
        memset(&LocalTypeHandler, 0, sizeof(LocalTypeHandler));
        LocalTypeHandler.Signature = tagSignature;

        Icc -> TagPtrs[Icc ->TagCount] = NULL;
        Icc -> TagOffsets[Icc ->TagCount] = 0;
        Icc -> TagSizes[Icc ->TagCount] = 0;
        Icc -> TagNames[Icc ->TagCount] = tagSignature;
        Icc -> TagTypeHandlers[Icc ->TagCount] = &LocalTypeHandler;

        _cmsUnlockMutex(Icc->ContextID, Icc ->UsrMutex);
        return NULL;
    }

    if (n >= MAX_TABLE_TAG) {
        _cmsUnlockMutex(Icc->ContextID, Icc->UsrMutex);
        return NULL;
    }
