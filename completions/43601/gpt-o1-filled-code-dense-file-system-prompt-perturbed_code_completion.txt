n = _cmsSearchTag(Icc, tagSignature, TRUE);
if (n < 0) {
    char String[5];
    _cmsTagSignature2String(String, tagSignature);
    cmsSignalError(Icc->ContextID, cmsERROR_UNKNOWN_EXTENSION, "Couldn't find tag '%s' in the profile.", String);
    goto Error;
}

// If the tag is already read, just return it
if (Icc->TagPtrs[n] != NULL) {
    _cmsUnlockMutex(Icc->ContextID, Icc->UsrMutex);
    return Icc->TagPtrs[n];
}
