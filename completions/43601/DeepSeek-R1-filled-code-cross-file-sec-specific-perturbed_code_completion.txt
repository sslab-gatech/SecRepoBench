n = _cmsSearchTag(Icc, tagSignature, TRUE);
    if (n < 0 || n >= Icc->TagCount) {
        char String[5];
        _cmsTagSignature2String(String, tagSignature);
        cmsSignalError(Icc->ContextID, n < 0 ? cmsERROR_TAG_NOT_FOUND : cmsERROR_CORRUPTION_DETECTED,
                       n < 0 ? "Tag '%s' not found" : "Invalid tag index for '%s'", String);
        goto Error;
    }
