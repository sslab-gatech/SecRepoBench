if (*pfx->pex == '?') {
  if (ptern->addrQuery != NULL_ADDRESS || ptern->addrColon != NULL_ADDRESS) {
    return MagickFalse;
  }
  pfx->pex++;
  ptern->addrQuery = pfx->usedElements;
  (void) AddAddressingElement(pfx, rCopyFrom, pfx->usedElements);
  pfx->usedElements++;
} else if (*pfx->pex == ':') {
  if (ptern->addrQuery == NULL_ADDRESS) {
    return MagickFalse;
  }
  if (ptern->addrColon != NULL_ADDRESS) {
    return MagickFalse;
  }
  pfx->pex++;
  ptern->addrColon = pfx->usedElements;
  (void) AddAddressingElement(pfx, rCopyFrom, pfx->usedElements);
  pfx->usedElements++;
} else {
  return MagickFalse;
}
