while (*src) {
        char* endptr;
        long num = strtol(src, &endptr, 10);
        if (endptr == src || num < 0 || num >= DEFAULTFLAGS) {
          if (af != NULL) {
            HUNSPELL_WARNING(stderr,
                             "error: line %d: invalid or out-of-range flag id %ld (max: %d)\n",
                             af->getlinenum(), num, DEFAULTFLAGS - 1);
          }
          num = 0;
        }
        *dest++ = (unsigned short)num;
        if (*endptr == ',') {
          src = endptr + 1;
        } else {
          src = endptr;
          break;
        }
      }
      break;
