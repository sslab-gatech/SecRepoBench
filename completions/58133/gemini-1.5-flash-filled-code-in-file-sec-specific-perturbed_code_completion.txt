while (src && *src) {
        char* end;
        long i = strtol(src, &end, 10);
        if (i <= 0 || i >= DEFAULTFLAGS) {
          if (af) {
            HUNSPELL_WARNING(stderr,
                             "error: line %d: flag id %ld is wrong (min: 1, max: %d)\n",
                             af->getlinenum(), i, DEFAULTFLAGS - 1);
          }
          i = 0;
        }
        *dest = (unsigned short)i;
        dest++;
        src = (*end == ',') ? end + 1 : NULL;
      }
      break;
    }
