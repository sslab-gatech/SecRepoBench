for (int i = 0; i < len; i++) {
  char* endptr;
  unsigned long flag = strtoul(src, &endptr, 10);
  if (flag >= DEFAULTFLAGS && af != NULL) {
    HUNSPELL_WARNING(stderr,
                     "error: line %d: flag id %d is too large (max: %d)\n",
                     af->getlinenum(), flag, DEFAULTFLAGS - 1);
    flag = 0;
  }
  if (flag == 0 && af != NULL) {
    HUNSPELL_WARNING(stderr, "error: line %d: 0 is not a valid flag id\n",
                     af->getlinenum());
  }
  *dest = (unsigned short)flag;
  dest++;
  if (*endptr == ',')
    src = endptr + 1;
  else if (i < len - 1) {
    HUNSPELL_WARNING(stderr, "error: line %d: parse error\n",
                     af->getlinenum());
    break;
  }
}
