while (true) {
        const char* start = src;
        while (*src && *src != ',') src++;
        std::string number(start, src);
        if (!number.empty()) {
          int i = atoi(number.c_str());
          if (i < 0) {
            HUNSPELL_WARNING(stderr, "error: line %d: flag id %d is negative\n", af->getlinenum(), i);
            i = 0;
          } else if (i >= DEFAULTFLAGS) {
            HUNSPELL_WARNING(stderr, "error: line %d: flag id %d is too large (max: %d)\n", af->getlinenum(), i, DEFAULTFLAGS - 1);
            i = 0;
          }
          *dest++ = (unsigned short)i;
        }
        if (*src == '\0') break;
        src++; // skip the comma
      }
