while (*src) {
        char* endptr;
        long num = strtol(src, &endptr, 10);
        if (endptr == src) {
          HUNSPELL_WARNING(stderr, "error: line %d: bad flagvector\n", af->getlinenum());
          break;
        }
        if (num >= DEFAULTFLAGS) {
          HUNSPELL_WARNING(stderr, "error: line %d: flag id %ld is too large (max: %d)\n",
                           af->getlinenum(), num, DEFAULTFLAGS - 1);
          num = 0;
        }
        if (num == 0) {
          HUNSPELL_WARNING(stderr, "error: line %d: 0 is wrong flag id\n", af->getlinenum());
        }
        *dest++ = (unsigned short)num;
        src = (*endptr == ',') ? endptr + 1 : endptr;
      }
      break;
