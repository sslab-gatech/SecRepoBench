int index = 0;
while (*src) {
  char* endptr;
  unsigned long flag = strtoul(src, &endptr, 10);
  if (endptr == src) {
    if (af != NULL) {
      HUNSPELL_WARNING(stderr, "error: line %d: invalid flag number\n", af->getlinenum());
    }
    break;
  }
  if (flag >= DEFAULTFLAGS) {
    if (af != NULL) {
      HUNSPELL_WARNING(stderr, "error: line %d: flag id %lu is too large (max: %d)\n",
                       af->getlinenum(), flag, DEFAULTFLAGS - 1);
    }
    flag = 0;
  }
  if (flag == 0 && af != NULL) {
    HUNSPELL_WARNING(stderr, "error: line %d: 0 is not a valid flag id\n", af->getlinenum());
  }
  dest[index++] = (unsigned short)flag;
  src = (*endptr == ',') ? endptr + 1 : endptr;
}
len = index;
