int index = 0;
while (*src) {
  char* endptr;
  unsigned long flag = strtoul(src, &endptr, 10);
  if (endptr == src) {
    // No valid number found
    break;
  }
  if (flag >= DEFAULTFLAGS && af != NULL) {
    HUNSPELL_WARNING(stderr,
                     "error: line %d: flag id %lu is too large (max: %d)\n",
                     af->getlinenum(), flag, DEFAULTFLAGS - 1);
    flag = 0;
  }
  dest[index++] = (unsigned short)flag;
  src = (*endptr == ',') ? endptr + 1 : endptr;
}
len = index;
