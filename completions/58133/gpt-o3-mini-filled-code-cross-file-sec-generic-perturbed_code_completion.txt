for (int i = 0; i < len; ++i) {
  char* endptr = nullptr;
  long value = std::strtol(src, &endptr, 10);
  if (src == endptr) {
    if (af != NULL) {
      HUNSPELL_WARNING(stderr, "error: line %d: invalid flag number\n", af->getlinenum());
    }
    value = 0;
  }
  if (value >= DEFAULTFLAGS) {
    if (af != NULL) {
      HUNSPELL_WARNING(stderr, "error: line %d: flag id %ld is too large (max: %d)\n",
                       af->getlinenum(), value, DEFAULTFLAGS - 1);
    }
    value = 0;
  }
  if (value == 0) {
    if (af != NULL) {
      HUNSPELL_WARNING(stderr, "error: line %d: 0 is wrong flag id\n", af->getlinenum());
    }
  }
  dest[i] = static_cast<unsigned short>(value);
  src = endptr;
  if (*src == ',') {
    ++src;
  }
}
