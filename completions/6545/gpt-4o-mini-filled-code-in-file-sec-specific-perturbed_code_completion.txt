// Initialize the number of stops to zero and set the stops pointer to NULL.
  ssize_t number_stops = 0;
  StopInfo *stops = NULL;

  // Allocate memory for graphic context, which is an array of DrawInfo pointers.
  DrawInfo **graphic_context = (DrawInfo **) AcquireQuantumMemory(2 * sizeof(*graphic_context));
  if (graphic_context == (DrawInfo **) NULL)
  {
    ThrowBinaryException(ResourceLimitError, "MemoryAllocationFailed", image->filename);
  }

  // Set the initial number of points for primitives and allocate memory for primitive information storage.
  size_t number_points = 16;
  PrimitiveInfo *primitive_info = (PrimitiveInfo *) AcquireQuantumMemory(number_points * sizeof(*primitive_info));
  if (primitive_info == (PrimitiveInfo *) NULL)
  {
    graphic_context = DestroyDrawInfo(graphic_context);
    ThrowBinaryException(ResourceLimitError, "MemoryAllocationFailed", image->filename);
  }
