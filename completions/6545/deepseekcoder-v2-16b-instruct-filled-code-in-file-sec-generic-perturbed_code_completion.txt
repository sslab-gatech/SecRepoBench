/*
  Allocate memory for the polygon edges.
*/
polygon_info->edges = (EdgeInfo *) AcquireQuantumMemory(number_edges, sizeof(*polygon_info->edges));
if (polygon_info->edges == (EdgeInfo *) NULL) {
  return ((PolygonInfo *) NULL);
}

/*
  Initialize the edges array.
*/
for (i = 0; i < number_edges; i++) {
  polygon_info->edges[i].points = (PointInfo *) AcquireQuantumMemory(number_points, sizeof(PointInfo));
  if (polygon_info->edges[i].points == (PointInfo *) NULL) {
    for (j = 0; j < i; j++) {
      RelinquishMagickMemory(polygon_info->edges[j].points);
    }
    RelinquishMagickMemory(polygon_info->edges);
    return ((PolygonInfo *) NULL);
  }
}

/*
  Fill the edges array with the points.
*/
for (i = 0; i < number_edges; i++) {
  for (j = 0; j < number_points; j++) {
    polygon_info->edges[i].points[j] = points[i * number_points + j];
  }
}

/*
  Set the bounds for each edge.
*/
for (i = 0; i < number_edges; i++) {
  polygon_info->edges[i].bounds.x1 = bounds.x1;
  polygon_info->edges[i].bounds.y1 = bounds.y1;
  polygon_info->edges[i].bounds.x2 = bounds.x2;
  polygon_info->edges[i].bounds.y2 = bounds.y2;
}

/*
  Set the number of edges.
*/
polygon_info->number_edges = number_edges;

/*
  Return the initialized polygon info.
*/
return polygon_info;
