// Ensure the compound pattern table is not empty
if (!checkcpdtable.empty()) {
    // Iterate through the compound pattern table
    for (size_t j = 0; j < checkcpdtable.size(); ++j) {
        const auto& pattern = checkcpdtable[j];

        // Check if the current pattern matches the string at position `i`
        if (pattern.pattern && isRevSubset(pattern.pattern, st.c_str() + i - 1, pattern.len)) {
            // Replace the matched part of the string with the replacement pattern
            st.replace(i - pattern.len, pattern.len, pattern.replacement);

            // Update the indices and length parameters
            i = i - pattern.len + pattern.replacement.size();
            len = st.size();

            // Adjust character minimum and maximum values for further processing
            setcminmax(&cmin, &cmax, st.c_str(), len);

            // Break out of the loop as a match has been found and processed
            break;
        }
    }
