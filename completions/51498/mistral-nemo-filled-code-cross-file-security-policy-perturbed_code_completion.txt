// Check if there is an active compound pattern processing.
// Find a matching pattern in the compound pattern table.
// If a match is found, replace parts of the string 'st' with new patterns
// as specified in the compound pattern table.
// Update the indices and length parameters to reflect the changes made
// to the string 'st'.
// Adjust character minimum and maximum values for further processing.
if (scpd != 0 && !checkcpdtable.empty() && scpd < checkcpdtable.size()) {
  const CompoundPattern& cpd = checkcpdtable[scpd];
  if (cpd.cond != FLAG_NULL && !TESTAFF(rv->astr, cpd.cond, rv->alen)) {
    return NULL;
  }
  if (cpd.cond2 != FLAG_NULL && !TESTAFF(rv->astr, cpd.cond2, rv->alen)) {
    return NULL;
  }
  if (!cpd.pattern.empty()) {
    size_t pos = compound_word.find(cpd.pattern, i);
    if (pos != std::string::npos) {
      st.replace(pos, cpd.pattern.size(), cpd.replacement);
      i += cpd.replacement.size() - cpd.pattern.size();
      len = compound_word.size() - pos;
      cmin = std::max(0, int(pos + cpd.replacement.size() - cpd.pattern.size()));
      cmax = std::min(int(len), cmax);
    }
  }
}
