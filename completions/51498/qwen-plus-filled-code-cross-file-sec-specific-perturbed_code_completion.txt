// named `compound_patterns`, we can iterate through it to find a matching pattern.
// Once a match is found, we replace parts of the string 'st' with new patterns as specified.

for (const auto& pattern : compound_patterns) {
    if (pattern.matches(st.substr(0, i))) {  // Check if the current substring matches the pattern
        std::string replacement = pattern.get_replacement();  // Get the replacement string from the pattern
        st.replace(0, i, replacement);  // Replace the matched part with the replacement string
        
        // Update indices and length parameters
        len = st.size();
        cmin = calculate_new_cmin(st.c_str(), len);  // Function to recalculate minimum character index
        cmax = calculate_new_cmax(st.c_str(), len);  // Function to recalculate maximum character index

        // Adjust word number and syllable count based on the pattern's rules
        wordnum += pattern.get_word_increment();
        numsyllable += pattern.get_syllable_increment();

        // If the pattern specifies any additional conditions, check them here
        if (!pattern.check_conditions(st)) {
            return NULL;  // If conditions are not met, exit early
        }

        break;  // Exit the loop after applying the first matching pattern
    }
}
