// Check for compound patterns
        if (checkcpdtable.empty() || scpd >= checkcpdtable.size()) {
            return NULL; // No valid compound patterns to check
        }

        // Check if the current character is part of a valid compound pattern
        if (TESTAFF(st.c_str(), checkcpdtable[scpd].cond, len)) {
            // Update the word number and syllable count based on the compound pattern
            wordnum++;
            numsyllable += get_syllable(st.substr(0, i));
            // Store the current state for backtracking
            oldnumsyllable = numsyllable;
            oldwordnum = wordnum;

            // Proceed to check the next part of the compound word
            rv = compound_check(st.substr(i), wordnum, numsyllable, maxwordnum, wnum + 1, words, rwords, hu_mov_rule, is_sug, info);
            if (rv) {
                return rv; // Return if a valid compound word is found
            }

            // Restore the previous state if no valid compound was found
            numsyllable = oldnumsyllable;
            wordnum = oldwordnum;
        }
