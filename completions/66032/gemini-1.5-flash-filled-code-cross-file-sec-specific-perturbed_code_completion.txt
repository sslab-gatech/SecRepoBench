while (probelength && (probe_data[0] == ' ' || probe_data[0] == '\n' || probe_data[0] == '\r' || probe_data[0] == '\t')) {
		probe_data++;
		probelength--;
	}
	if (!probelength) {
		gf_free(dst);
		return NULL;
	}

	u32 len = 0;
	while (probelength && probe_data[len] && probe_data[len] != '<') {
		len++;
	}
	if (len) {
		char *tmp = (char *)gf_malloc(sizeof(char) * (len + 1));
		if (!tmp) {
			gf_free(dst);
			return NULL;
		}
		memcpy(tmp, probe_data, len);
		tmp[len] = 0;
		gf_free(dst);
		dst = tmp;
		probe_data += len;
		probelength -= len;
	}
	
	if (!probelength) {
		gf_free(dst);
		return NULL;
	}
