		probelength--;

		const char *next = my_strstr(probe_data, "?>", probelength);
		if (!next) goto exit;
		probelength -= (u32) (next - (char*)probe_data + 2);
		probe_data = next+2;
	}
	while (probelength && (probe_data[0]=='<') && (probe_data[1]=='!') ) {
		const char *next = my_strstr(probe_data, ">", probelength);
		if (!next) goto exit;
		probelength -= (u32) (next - (char*)probe_data + 1);
		probe_data = next+1;
