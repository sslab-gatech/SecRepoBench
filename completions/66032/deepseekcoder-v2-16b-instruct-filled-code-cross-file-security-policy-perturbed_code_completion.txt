//probe_data is now the first element of the document, if XML
	//we should refine by getting the xmlns attribute value rather than searching for its value...
	if (my_strstr(probe_data, "http://www.w3.org/1999/XSL/Transform", probelength)) {
		const char *xmlns_attr = "http://www.w3.org/1999/XSL/Transform";
		char *xmlns_value = NULL;
		u32 xmlns_length = probelength;
		gf_utf_get_attribute_value(probe_data, probelength, xmlns_attr, &xmlns_value, &xmlns_length);
		if (xmlns_value) {
			if (my_strstr(xmlns_value, "application/x-xmt", xmlns_length)) {
				mime_type = "application/x-xmt";
			} else if (my_strstr(xmlns_value, "model/x3d+xml", xmlns_length)) {
				mime_type = "model/x3d+xml";
			} else if (my_strstr(xmlns_value, "application/x-LASeR+xml", xmlns_length)) {
				mime_type = "application/x-LASeR+xml";
			}
			gf_free(xmlns_value);
		}
