while (probelength && (probe_data[0] == 0x20 || probe_data[0] == 0x0A || probe_data[0] == 0x0D || probe_data[0] == 0x09)) {
		probe_data++;
		probelength--;
	}
	if (!probelength) goto exit;

	while (probelength && (probe_data[0] != '<')) {
		const char *next = my_strstr(probe_data, "<", probelength);
		if (!next) goto exit;
		probelength -= (u32)(next - (char*)probe_data);
		probe_data = next;
	}
	if (!probelength) goto exit;
