//probe_data is now the first element of the document, if XML
	//we should refine by getting the xmlns attribute value rather than searching for its value...
	if (my_strstr(probe_data, "http://www.w3.org/1999/XSL/Transform", probelength)) {
		const char *xmlns_attr = "http://www.w3.org/1999/XSL/Transform";
		const char *xmlns_value = NULL;
		const char *probe_start = probe_data;
		u32 probe_length = probelength;

		// Skip leading whitespace and newline characters
		while (probe_length && *probe_start && strchr("\n\r\t ", *probe_start)) {
			probe_start++;
			probe_length--;
		}

		// Check for xmlns attribute
		if (!strncmp(probe_start, "xmlns", strlen("xmlns"))) {
			xmlns_value = probe_start + strlen("xmlns");
			// Skip any spaces after xmlns
			while (*xmlns_value && strchr(" \t", *xmlns_value)) {
				xmlns_value++;
			}
			// Check if the value matches the desired xmlns
			if (my_strstr(xmlns_value, xmlns_attr, strlen(xmlns_attr))) {
				mime_type = "application/x-xslt";
				goto exit;
			}
		}
