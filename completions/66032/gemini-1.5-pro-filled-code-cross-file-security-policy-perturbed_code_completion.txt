		probe_data ++;
		probelength--;

	while (probelength>strlen("<?xml") && !strncmp(probe_data, "<?xml", strlen("<?xml"))) {
		const char *next = my_strstr(probe_data, "?>", probelength);
		if (!next)
			break;

		probelength -= (u32) (next + 2 - (char*)probe_data);
		probe_data = (const u8*)next + 2;

		while (probelength && probe_data[0] && strchr("\n\r\t ", (char) probe_data[0])) {
			probe_data ++;
			probelength--;
		}
	}
	//skip doctype
	if (probelength>strlen("<!DOCTYPE") && !strncmp(probe_data, "<!DOCTYPE", strlen("<!DOCTYPE"))) {
		const char *next = my_strstr(probe_data, ">", probelength);
		if (!next)
			break;
		probelength -= (u32) (next + 1 - (char*)probe_data);
		probe_data = (const u8*)next + 1;
		while (probelength && probe_data[0] && strchr("\n\r\t ", (char) probe_data[0])) {
			probe_data ++;
			probelength--;
		}
	}
	//skip comments
	while (probelength>strlen("<!--") && !strncmp(probe_data, "<!--", strlen("<!--"))) {
		const char *next = my_strstr(probe_data, "-->", probelength);
		if (!next)
			break;

		probelength -= (u32) (next + 3 - (char*)probe_data);
		probe_data = (const u8*)next + 3;
