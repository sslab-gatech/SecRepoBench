// Strip leading whitespace and newlines
	while (probelength && probe_data[0] && strchr("\n\r\t ", (char) probe_data[0])) {
		probe_data++;
		probelength--;
	}

	// Remove XML comments, DOCTYPE, and XML headers
	while (probelength && probe_data[0] == '<') {
		if (!strncmp((const char *)probe_data, "<?xml", 5)) {
			// skip XML header
			const char *end = my_strstr(probe_data, "?>", probelength);
			if (!end) break;
			u32 skip = (u32)(end - (char*)probe_data + 2);
			probe_data += skip;
			probelength -= skip;
			while (probelength && probe_data[0] && strchr("\n\r\t ", (char) probe_data[0])) {
				probe_data++;
				probelength--;
			}
			continue;
		}
		if (!strncmp((const char *)probe_data, "<!--", 4)) {
			// skip XML comment
			const char *end = my_strstr(probe_data, "-->", probelength);
			if (!end) break;
			u32 skip = (u32)(end - (char*)probe_data + 3);
			probe_data += skip;
			probelength -= skip;
			while (probelength && probe_data[0] && strchr("\n\r\t ", (char) probe_data[0])) {
				probe_data++;
				probelength--;
			}
			continue;
		}
		if (!strncmp((const char *)probe_data, "<!DOCTYPE", 9)) {
			// skip DOCTYPE
			const char *end = my_strstr(probe_data, ">", probelength);
			if (!end) break;
			u32 skip = (u32)(end - (char*)probe_data + 1);
			probe_data += skip;
			probelength -= skip;
			while (probelength && probe_data[0] && strchr("\n\r\t ", (char) probe_data[0])) {
				probe_data++;
				probelength--;
			}
			continue;
		}
		break;
