//we should refine by getting the xmlns attribute value rather than searching for its value...
//first, we need to find the start of the root element
const char *root_start = my_strstr(probe_data, "<", probelength);
if (!root_start) goto exit;

//then, we need to find the end of the root element
const char *root_end = my_strstr(root_start, ">", probelength);
if (!root_end) goto exit;

//now, we can extract the xmlns attribute value
const char *xmlns_start = my_strstr(root_start, "xmlns=\"", probelength);
if (xmlns_start) {
    const char *xmlns_end = my_strstr(xmlns_start, "\"", probelength);
    if (xmlns_end) {
        //we need to remove the "xmlns=\"" prefix
        const char *xmlns_value = xmlns_start + strlen("xmlns=\"");
        //we need to remove the "\" suffix
        const char *xmlns_value_end = my_strstr(xmlns_value, "\"", probelength);
        if (xmlns_value_end) {
            //now, we can check if the xmlns attribute value matches any of the known values
            if (my_strstr(xmlns_value, "http://www.w3.org/1999/XSL/Transform", probelength)) {
                mime_type = "application/xslt+xml";
            } else if (my_strstr(xmlns_value, "http://www.web3d.org/specifications/x3d-3.0.xsd", probelength)) {
                mime_type = "model/x3d+xml";
            } else if (my_strstr(xmlns_value, "http://www.w3.org/2000/svg", probelength)) {
                mime_type = "image/svg+xml";
            }
        }
    }
}
