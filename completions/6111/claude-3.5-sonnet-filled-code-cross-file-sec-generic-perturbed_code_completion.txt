image=ReadImage(image_info,exception);
if (image == (Image *) NULL)
  return((Image *) NULL);
(void) CopyMagickString(image->filename,image_info->filename,MagickPathExtent);
if (LocaleCompare(image_info->magick,"EPS") == 0)
  (void) CopyMagickString(image->magick,"EPS",MagickPathExtent);
else
  (void) CopyMagickString(image->magick,"PS",MagickPathExtent);
(void) CloseBlob(image);
image->page=page;
if (image_info->number_scenes != 0)
  {
    Image
      *clone_image;

    clone_image=CloneImage(image,0,0,MagickTrue,exception);
    if (clone_image != (Image *) NULL)
      {
        image=DestroyImageList(image);
        image=clone_image;
      }
  }
if (LocaleCompare(image_info->magick,"EPS") == 0)
  {
    Image
      *next;

    next=image;
    while (next != (Image *) NULL)
    {
      next->colorspace=CMYKColorspace;
      next=GetNextImageInList(next);
    }
  }
