// Check if the image list is not empty
if (postscript_image!= (Image *) NULL)
{
  // Initialize the image list
  image = NewImageList();
  // Append the postscript image to the image list
  AppendImageToList(&image, postscript_image);
  // Return the image list
  return image;
}
else
{
  // If the image list is empty, return NULL
  return (Image *) NULL;
}
