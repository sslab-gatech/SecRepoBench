// Initialize necessary variables and structures for processing Postscript images.
image=(Image *) NULL;
postscript_image=(Image *) NULL;
read_info=CloneImageInfo(image_info);
if (read_info == (ImageInfo *) NULL)
  {
    InheritException(exception,&image_info->exception);
    return((Image *) NULL);
  }
// Open the input image file and create a symbolic link to it for processing.
filedescriptor=OpenBlob(read_info->filename,read_info->mode,exception);
if (filedescriptor == -1)
  {
    InheritException(exception,&read_info->exception);
    return((Image *) NULL);
  }
// Initialize an array to map hex characters to their integer values.
for (c=0; c < (int) MagickMax(256,sizeof(hex_digit)); c++)
  hex_digit[c]=0;
for (c=0; c < (int) MagickMax(16,sizeof(hex_digit)); c++)
  hex_digit[(int) ((unsigned char) "0123456789ABCDEF"[c])]=c;
for (c=0; c < (int) MagickMax(16,sizeof(hex_digit)); c++)
  hex_digit[(int) ((unsigned char) "0123456789abcdef"[c])]=c;
// Set the page density for the image based on the provided resolution or default values.
density=GetImageOption(read_info,"density");
if (density == (char *) NULL)
  density=GetImageOption(read_info,"resolution");
if (density == (char *) NULL)
  density=DefaultPSDensity;
// Determine the page geometry using the Postscript bounding box comments.
(void) GetImageOption(read_info,"page","%gx%g%+g%+g",columns,rows,x_offset,
  y_offset);
if ((columns == 0) || (rows == 0))
  {
    (void) CopyMagickString(geometry,read_info->geometry,MagickPathExtent);
    (void) ParseAbsoluteGeometry(geometry,&geometry_info);
    columns=geometry_info.width;
    rows=geometry_info.height;
    x_offset=geometry_info.x;
    y_offset=geometry_info.y;
  }
// Initialize various variables including color space, resolution, and page dimensions.
options=GetImageOption(read_info,"ps:options");
if (options == (char *) NULL)
  options=DefaultPSOptions;
// Create a control file for Ghostscript to render the Postscript image.
(void) FormatPSDelegateInfo(read_info,exception);
// Use the Ghostscript delegate to render the Postscript into a raster format.
postscript_image=ReadPSDelegateImage(read_info,exception);
if (postscript_image == (Image *) NULL)
  {
    InheritException(exception,&read_info->exception);
    return((Image *) NULL);
  }
