/*
    Open image file.
  */
  assert(image_info != (ImageInfo *) NULL);
  assert(exception != (ExceptionInfo *) NULL);
  assert(image_info->signature == MagickCoreSignature);
  if (image_info->debug != MagickFalse)
    (void) LogMagickEvent(TraceEvent,GetMagickModule(),"%s",
      image_info->filename);
  image=NewImageList();
  read_info=CloneImageInfo(image_info);
  (void) SetImageInfo(read_info,1,exception);
  if (LocaleCompare(read_info->magick,"PS") != 0)
    (void) CopyMagickString(read_info->magick,"PS",MagickPathExtent);
  filedescriptor=AcquireUniqueFileResource(postscript_filename);
  if (filedescriptor == -1)
    {
      read_info=DestroyImageInfo(read_info);
      ThrowReaderException(FileOpenError,"UnableToCreateTemporaryFile");
    }
  (void) WriteImage(read_info,postscript_image);
  (void) CloseUniqueFileResource(postscript_filename);
  (void) FormatLocaleString(command,MagickPathExtent,
    "gs -q -dSAFER -dBATCH -dNOPAUSE -sDEVICE=ppmraw -dTextAlphaBits=4 "
    "-dGraphicsAlphaBits=4 -r%gx%g -g%gx%g -sOutputFile=%s %s",
    (size_t) floor(0.5*read_info->density.x+0.5),
    (size_t) floor(0.5*read_info->density.y+0.5),
    (size_t) floor(0.5*geometry_info.rho+0.5),
    (size_t) floor(0.5*geometry_info.sigma+0.5),
    postscript_filename,read_info->filename);
  status=InvokePostscriptDelegate(read_info->verbose,command,exception);
  if (status == MagickFalse)
    {
      read_info=DestroyImageInfo(read_info);
      ThrowReaderException(DelegateError,"PostscriptDelegateFailed");
    }
  (void) CopyMagickString(image->filename,postscript_filename,
    MagickPathExtent);
  image=ReadImages(read_info,postscript_filename,exception);
  (void) RelinquishUniqueFileResource(postscript_filename);
  if (image == (Image *) NULL)
    {
      read_info=DestroyImageInfo(read_info);
      return((Image *) NULL);
    }
  (void) CopyMagickString(image->filename,image_info->filename,
    MagickPathExtent);
  read_info=DestroyImageInfo(read_info);
