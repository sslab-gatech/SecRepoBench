/*
    Validate input parameters and initialize variables.
  */
  assert(image_info != (ImageInfo *) NULL);
  assert(image_info->signature == MagickCoreSignature);
  assert(exception != (ExceptionInfo *) NULL);

  image=NewImageList();
  postscript_image=(Image *) NULL;
  read_info=CloneImageInfo(image_info);
  if (read_info == (ImageInfo *) NULL)
    ThrowReaderException(ResourceLimitError,"MemoryAllocationFailed");

  /*
    Open the input file.
  */
  filedescriptor=OpenBlob(read_info,ReadBinaryBlobMode,exception);
  if (filedescriptor == -1)
    {
      read_info=DestroyImageInfo(read_info);
      ThrowReaderException(FileOpenError,"UnableToOpenFile");
    }

  /*
    Create a temporary file for processing.
  */
  if (AcquireUniqueFilename(input_filename) == MagickFalse)
    {
      read_info=DestroyImageInfo(read_info);
      ThrowReaderException(FileOpenError,"UnableToCreateTemporaryFile");
    }
  (void) CopyMagickString(filename,input_filename,MagickPathExtent);

  /*
    Write the input file to the temporary file.
  */
  if (WriteBlobFile(filedescriptor,filename) == MagickFalse)
    {
      read_info=DestroyImageInfo(read_info);
      ThrowReaderException(FileWriteError,"UnableToWriteTemporaryFile");
    }
  (void) CloseBlob(read_info);

  /*
    Use Ghostscript to render the Postscript file.
  */
  delegate_info=GetDelegateInfo("ps:decode",NULL,exception);
  if (delegate_info == (const DelegateInfo *) NULL)
    {
      read_info=DestroyImageInfo(read_info);
      ThrowReaderException(DelegateError,"PostscriptDelegateFailed");
    }
  (void) FormatLocaleString(command,MagickPathExtent,"%s \"%s\" \"%s\"",
    delegate_info->commands,filename,postscript_filename);
  status=SystemCommand(MagickFalse,command);
  if (status != 0)
    {
      read_info=DestroyImageInfo(read_info);
      ThrowReaderException(DelegateError,"PostscriptDelegateFailed");
    }

  /*
    Read the rendered image.
  */
  postscript_image=ReadImages(read_info,postscript_filename,exception);
  if (postscript_image == (Image *) NULL)
    {
      read_info=DestroyImageInfo(read_info);
      ThrowReaderException(CoderError,"UnableToReadRenderedImage");
    }

  /*
    Clean up temporary files.
  */
  (void) RelinquishUniqueFileResource(input_filename);
  (void) RelinquishUniqueFileResource(postscript_filename);
  read_info=DestroyImageInfo(read_info);
