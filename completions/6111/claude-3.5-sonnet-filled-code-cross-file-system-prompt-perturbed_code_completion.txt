MagickBooleanType
  status;

Image
  *images;

unsigned long
  columns,
  rows,
  scene;

RectangleInfo
  page;

/*
  Read Postscript image.
*/
image=AcquireImage(image_info,exception);
status=OpenBlob(image_info,image,ReadBinaryBlobMode,exception);
if (status == MagickFalse)
  {
    image=DestroyImageList(image);
    return((Image *) NULL);
  }
  
/*
  Initialize image structure.
*/
(void) ResetImagePage(image,"0x0+0+0");
/*
  Render Postscript image.
*/
images=ReadImages(image_info,filename,exception);
if (images == (Image *) NULL)
  ThrowReaderException(DelegateError,"PostscriptDelegateFailed");
if (LocaleCompare(images->magick,"BMP") == 0)
  {
    (void) CopyMagickString(images->filename,filename,MagickPathExtent);
    (void) CopyMagickString(images->magick,image->magick,MagickPathExtent);
  }
columns=images->columns;
rows=images->rows;
images=DestroyImageList(images);
/*
  Read Ghostscript output.
*/
(void) CloseBlob(image);
image->columns=columns;
image->rows=rows;
(void) CopyMagickString(image->filename,filename,MagickPathExtent);
(void) CopyMagickString(image->magick_filename,filename,MagickPathExtent);
image->page=page;
(void) CloseBlob(image);
if (image_info->number_scenes != 0)
  {
    Image
      *clone_image;

    /*
      Add subsequent Postscript pages.
    */
    clone_image=CloneImage(image,0,0,MagickTrue,exception);
    if (clone_image != (Image *) NULL)
      {
        Image
          *previous_image;

        previous_image=image;
        image=clone_image;
        for (scene=1; scene < (unsigned long) image_info->scene; scene++)
        {
          (void) CopyMagickString(image->filename,filename,MagickPathExtent);
          (void) CopyMagickString(image->magick_filename,filename,
            MagickPathExtent);
          image->scene=scene;
          next=ReadImage(image_info,exception);
          if (next == (Image *) NULL)
            break;
          AppendImageToList(&image,next);
          if (image->progress_monitor != (MagickProgressMonitor) NULL)
            {
              MagickBooleanType
                proceed;

              proceed=SetImageProgress(image,LoadImagesTag,scene,
                image_info->number_scenes);
              if (proceed == MagickFalse)
                break;
            }
        }
        AppendImageToList(&previous_image,image);
        image=previous_image;
      }
  }
postscript_image=GetFirstImageInList(image);
