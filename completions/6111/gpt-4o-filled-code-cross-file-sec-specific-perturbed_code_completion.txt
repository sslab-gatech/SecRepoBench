// Initialize the image list and read the Postscript image
  image=NewImageList();
  read_info=CloneImageInfo(image_info);
  (void) CopyMagickString(read_info->filename,filename,MagickPathExtent);
  postscript_image=ReadImage(read_info,exception);
  read_info=DestroyImageInfo(read_info);
  if (postscript_image == (Image *) NULL)
    return((Image *) NULL);

  // Process each image in the list
  for (next=GetFirstImageInList(postscript_image); next != (Image *) NULL; )
  {
    // Ensure the image is valid and append it to the image list
    if (next->columns == 0 || next->rows == 0)
    {
      postscript_image=DestroyImageList(postscript_image);
      return((Image *) NULL);
    }
    AppendImageToList(&image,CloneImage(next,0,0,MagickTrue,exception));
    next=GetNextImageInList(next);
  }
  postscript_image=DestroyImageList(postscript_image);
