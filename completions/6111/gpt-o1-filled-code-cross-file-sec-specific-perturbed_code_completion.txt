{
  ImageInfo
    *read_info;

  MagickBooleanType
    status;

  /* 1) Clone the incoming ImageInfo for our local processing. */
  read_info=CloneImageInfo(image_info);

  /* 2) Prepare a unique temporary filename and create a symbolic link. */
  (void) CopyMagickString(input_filename,image_info->filename,MagickPathExtent);
  if (AcquireUniqueFilename(filename) == MagickFalse)
    ThrowReaderException(FileOpenError,"UnableToCreateTemporaryFile");
  status=Symlink(input_filename,filename);
  if (status != 0)
  {
    (void) RelinquishUniqueFileResource(filename);
    ThrowReaderException(FileOpenError,"UnableToSymlinkFile");
  }

  /* 3) Update the read_info structure for Postscript processing. */
  (void) CopyMagickString(read_info->filename,filename,MagickPathExtent);
  if ((image_info->density != (char *) NULL) && (*image_info->density != '\0'))
    (void) CloneString(&read_info->density,image_info->density);

  /* 4) Invoke Ghostscript (or similar delegate) by reading from the symbolic link. */
  postscript_image=ReadImages(read_info,filename,exception);

  /* 5) Clean up and handle errors. */
  read_info=DestroyImageInfo(read_info);
  (void) RelinquishUniqueFileResource(filename);
  if (postscript_image == (Image *) NULL)
    ThrowReaderException(DelegateError,"PostscriptDelegateFailed");
}
