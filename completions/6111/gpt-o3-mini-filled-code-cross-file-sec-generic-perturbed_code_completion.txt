{
  int status;
  ImageInfo *read_info_temp = CloneImageInfo(image_info);

  /* Acquire a unique temporary filename for Ghostscript output */
  if (AcquireUniqueFilename(postscript_filename) == MagickFalse)
  {
    read_info_temp = DestroyImageInfo(read_info_temp);
    ThrowReaderException(FileOpenError, "UnableToGenerateTemporaryFileName");
  }

  /* Determine the resolution using the provided density or use default (72 dpi). */
  long resolution = 72;
  if (image_info->density != (char *) NULL)
    resolution = strtol(image_info->density, (char **) NULL, 10);

  /* Construct the Ghostscript command securely.
     Note: Ensure that image_info->filename and postscript_filename are trusted or sanitized. */
  (void) FormatString(command, MagickPathExtent,
    "gs -q -dNOPAUSE -dBATCH -sDEVICE=pngalpha -r%ld -sOutputFile=%s %s",
    resolution, postscript_filename, image_info->filename);

  /* Execute the Ghostscript command and check for errors. */
  status = SystemCommand(command);
  if (status != 0)
  {
    LiberateUniqueFile(postscript_filename);
    read_info_temp = DestroyImageInfo(read_info_temp);
    ThrowReaderException(CoderError, "GhostscriptDelegateFailed");
  }

  /* Read the rendered image from the temporary Ghostscript output. */
  (void) CopyMagickString(read_info_temp->filename, postscript_filename, MagickPathExtent);
  postscript_image = ReadImage(read_info_temp, exception);
  read_info_temp = DestroyImageInfo(read_info_temp);
  LiberateUniqueFile(postscript_filename);
  if (postscript_image == (Image *) NULL)
    ThrowReaderException(CoderError, "UnableToReadGhostscriptOutput");
}
