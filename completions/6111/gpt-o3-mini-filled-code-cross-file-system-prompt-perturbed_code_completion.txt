{
  ImageInfo
    *read_info;

  char
    command[MagickPathExtent];

  /*
    Set the temporary filename for the Ghostscript rendered output.
  */
  (void) FormatLocaleString(postscript_filename,MagickPathExtent,
    "%s.miff",image_info->filename);

  /*
    Acquire the Ghostscript delegate information.
  */
  delegate_info=GetDelegateInfo("ps:decode",exception);
  if (delegate_info == (const DelegateInfo *) NULL)
    ThrowReaderException(DelegateError,"DelegateFailed");

  /*
    Build the Ghostscript command to render the Postscript image.
    The command invokes Ghostscript in safe, non-interactive mode,
    sets the output device to MIFF, and writes the output to the temporary file.
  */
  (void) FormatLocaleString(command,MagickPathExtent,
    "%s -dSAFER -dBATCH -dNOPAUSE -sDEVICE=miff -sOutputFile=%s %s",
    delegate_info->command,postscript_filename,image_info->filename);

  /*
    Execute the Ghostscript delegate command.
  */
  if (InvokeDelegate(command,image_info->filename,exception) == MagickFalse)
    ThrowReaderException(DelegateError,"PostscriptRenderingFailed");

  /*
    Read the rendered image from the temporary file.
  */
  read_info=CloneImageInfo(image_info);
  (void) CopyMagickString(read_info->filename,postscript_filename,MagickPathExtent);
  postscript_image=ReadImage(read_info,exception);
  read_info=DestroyImageInfo(read_info);
  if (postscript_image == (Image *) NULL)
    ThrowReaderException(DelegateError,"PostscriptImageNotReturned");
}
