/*
    Render the Postscript image using Ghostscript delegate.
  */
  assert(image_info != (ImageInfo *) NULL);
  assert(image_info->signature == MagickCoreSignature);
  if (image_info->debug != MagickFalse)
    (void) LogMagickEvent(TraceEvent,GetMagickModule(),"%s",
      image_info->filename);
  assert(exception != (ExceptionInfo *) NULL);

  read_info=CloneImageInfo(image_info);
  (void) CopyMagickString(read_info->filename,image_info->filename,
    MagickPathExtent);
  (void) SetImageOption(read_info,"filename",image_info->filename);

  /*
    Create a temporary file for the Postscript input.
  */
  filedescriptor=AcquireUniqueFileResource(input_filename);
  if (filedescriptor == -1)
    ThrowReaderException(FileOpenError,"UnableToCreateTemporaryFile");
  (void) WriteBlobFile(filedescriptor,image_info->filename);
  (void) CloseBlob(filedescriptor);

  /*
    Invoke Ghostscript delegate to render Postscript to raster.
  */
  delegate_info=GetDelegateInfo("ps:decode",NULL,exception);
  if (delegate_info == (const DelegateInfo *) NULL)
    ThrowReaderException(DelegateError,"PostscriptDelegateFailed");
  (void) FormatLocaleString(command,MagickPathExtent,"%s \"%s\" \"%s\"",
    delegate_info->commands,input_filename,filename);
  status=InvokeDelegate(read_info,command,exception);
  if (status == MagickFalse)
    ThrowReaderException(DelegateError,"PostscriptDelegateFailed");

  /*
    Read the rendered image.
  */
  postscript_image=ReadImages(read_info,filename,exception);
  if (postscript_image == (Image *) NULL)
    ThrowReaderException(DelegateError,"UnableToReadImage");
  image=DestroyImageInfo(read_info);
