ssize_t
    scene = 0,
    columns = 0,
    rows = 0;

  RectangleInfo
    page;

  postscript_image = (Image *) NULL;
  image = (Image *) NULL;
  read_info = CloneImageInfo(image_info);
  (void) CopyMagickString(input_filename, image_info->filename, MagickPathExtent);
  (void) CopyMagickString(filename, image_info->filename, MagickPathExtent);
  (void) CopyMagickString(postscript_filename, image_info->filename, MagickPathExtent);

  // Set up the delegate command to render the Postscript file
  delegate_info = GetDelegateInfo("ps:decode", image_info->magick, exception);
  if (delegate_info == (const DelegateInfo *) NULL)
    {
      (void) ThrowMagickException(exception, GetMagickModule(), DelegateError,
        "NoDecodeDelegateForThisImageFormat", "`%s'", image_info->filename);
      return((Image *) NULL);
    }

  // Set up density and geometry
  density = (char *) GetImageOption(image_info, "density");
  if (density == (char *) NULL)
    density = (char *) "72x72";
  (void) CopyMagickString(geometry, "", MagickPathExtent);

  // Set up page geometry
  page.width = 612;
  page.height = 792;
  page.x = 0;
  page.y = 0;

  // Render the Postscript file to a raster image using the delegate
  (void) FormatLocaleString(command, MagickPathExtent, "%s \"%s\" \"%s\"",
    delegate_info->command, input_filename, postscript_filename);

  if (system(command) != 0)
    {
      (void) ThrowMagickException(exception, GetMagickModule(), DelegateError,
        "PostscriptDelegateFailed", "`%s'", command);
      return((Image *) NULL);
    }

  // Read the rendered image(s)
  (void) CopyMagickString(read_info->filename, postscript_filename, MagickPathExtent);
  postscript_image = ReadImages(read_info, postscript_filename, exception);
  read_info = DestroyImageInfo(read_info);
  if (postscript_image == (Image *) NULL)
    {
      (void) ThrowMagickException(exception, GetMagickModule(), CorruptImageError,
        "NoImagesWereLoaded", "`%s'", postscript_filename);
      return((Image *) NULL);
    }
  // Set up page and geometry info for each image
  page = postscript_image->page;
  columns = postscript_image->magick_columns;
  rows = postscript_image->magick_rows;
