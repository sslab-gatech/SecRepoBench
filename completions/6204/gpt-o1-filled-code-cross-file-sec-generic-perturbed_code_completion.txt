{
  char colorspace[MagickPathExtent];
  register ssize_t i = 0;

  (void) ResetMagickMemory(colorspace, 0, MagickPathExtent);
  /*
    Copy characters from color_name into colorspace until we reach
    '(' or the end of the string, ensuring we do not exceed the buffer.
  */
  while ((color_name[i] != '\0') && (color_name[i] != '(') &&
         (i < (MagickPathExtent - 1)))
  {
    colorspace[i] = color_name[i];
    i++;
  }
  colorspace[i] = '\0';
}
