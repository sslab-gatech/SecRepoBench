for (i=0; i < (ssize_t) number_vertices; i++)
{
  if (primitive_info[i].point.y > y)
    continue;
  if ((primitive_info[i].point.y < y) && 
      (primitive_info[i+1].point.y < y))
    continue;
  if (primitive_info[i+1].point.y < primitive_info[i].point.y)
    {
      winding_number--;
      continue;
    }
  if (primitive_info[i].point.x > x)
    continue;
  if (primitive_info[i+1].point.x > x)
    {
      winding_number++;
      continue;
    }
  dx=primitive_info[i+1].point.x-primitive_info[i].point.x;
  dy=primitive_info[i+1].point.y-primitive_info[i].point.y;
  if ((dy*(x-primitive_info[i].point.x)) > 
      (dx*(y-primitive_info[i].point.y)))
    winding_number++;
}
