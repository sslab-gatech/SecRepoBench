for (i=0; i < (ssize_t) number_vertices; i++)
{
  if (primitive_info[i].point.y <= (double) y)
    continue;
  if (primitive_info[i-1].point.y > (double) y)
    continue;
  if (primitive_info[i].point.x <= (double) x)
    continue;
  if ((primitive_info[i].point.x > (double) x) &&
      (primitive_info[i-1].point.x <= (double) x))
    winding_number++;
  else
    {
      dx=primitive_info[i].point.x-primitive_info[i-1].point.x;
      dy=primitive_info[i].point.y-primitive_info[i-1].point.y;
      if ((dy*(x-primitive_info[i-1].point.x)) >
          (dx*(y-primitive_info[i-1].point.y)))
        winding_number++;
    }
}
