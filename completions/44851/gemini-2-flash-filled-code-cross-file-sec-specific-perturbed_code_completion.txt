if ((double) ycoordinate <= (p->points[0].y-mid-0.5))
      continue;
    if ((double) ycoordinate > (p->points[p->number_points-1].y+mid+0.5))
      continue;
    /*
      Determine if point is left or right of edge.
    */
    if (p->sign >= 0)
      {
        if ((((p->points[1].x-p->points[0].x)*
             ((double) ycoordinate-p->points[0].y))-
             ((p->points[1].y-p->points[0].y)*((double) x-p->points[0].x))) >
            0.0)
          winding_number++;
        continue;
      }
    if ((((p->points[1].x-p->points[0].x)*((double) ycoordinate-p->points[0].y))-
         ((p->points[1].y-p->points[0].y)*((double) x-p->points[0].x))) <
        0.0)
      winding_number--;
