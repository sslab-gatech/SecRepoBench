winding_number=0;
p=polygon_info->edges;
for (j=0; j < (ssize_t) polygon_info->number_edges; j++, p++)
{
  if ((double) ycoordinate > (p->bounds.y2+mid+0.5))
    continue;
  if ((double) ycoordinate <= (p->bounds.y1-mid-0.5))
    break;
  if (((double) x > (p->bounds.x2+mid+0.5)) || ((double) x <= (p->bounds.x1-mid-0.5)))
    continue;
  if (p->scanline != (double) ycoordinate)
    {
      p->scanline=(double) ycoordinate;
      p->highwater=(size_t) i;
    }
  if (p->direction == LeftToRightDirection)
    winding_number++;
  else
    winding_number--;
}
