mrb_int base = 10;
  if (state->c->ci->argc > 0) {
    mrb_value base_arg = state->c->ci->stack[1];
    if (!mrb_integer_p(base_arg)) {
      mrb_raisef(state, E_TYPE_ERROR, "can't convert %Y into Integer", base_arg);
    }
    base = mrb_integer(base_arg);
  }
