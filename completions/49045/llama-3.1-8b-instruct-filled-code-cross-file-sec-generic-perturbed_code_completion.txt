static mrb_value
int_to_s(mrb_state *mrb, mrb_value obj)
{
  mrb_int num;
  mrb_int base = 10;
  mrb_get_args(mrb, "i|i", &num, &base);

  if (base < 2 || base > 36) {
    mrb_raise(mrb, E_ARGUMENT_ERROR, "base must be between 2 and 36");
  }

  char buf[64];
  mrb_snprintf(buf, sizeof(buf), "%d", num);
  return mrb_str_new_cstr(mrb, buf);
}
