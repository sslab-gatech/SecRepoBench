static mrb_value
int_to_s(mrb_state *state, mrb_value self)
{
  mrb_int num;
  mrb_int base = 10;
  mrb_get_args(state, "i|i", &num, &base);

  if (num == 0) {
    return mrb_str_new_cstr(state, "0");
  }

  mrb_value str = mrb_str_new(state, 0);
  mrb_value sign = mrb_str_new_cstr(state, "");

  if (num < 0) {
    mrb_str_set(state, str, mrb_str_new_cstr(state, "-"));
    mrb_int_to_str(state, -num, base, str);
  } else {
    mrb_int_to_str(state, num, base, str);
  }

  return str;
}
