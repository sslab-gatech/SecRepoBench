mrb_int num;
  mrb_get_args(mrb, "i", &num);
  mrb_int base = 10; // default base is 10
  mrb_get_args(mrb, "|i", &base); // get optional base argument
  if (base < 2 || base > 36) { // validate base
    mrb_raise(mrb, E_ARGUMENT_ERROR, "base must be between 2 and 36");
  }
  char *str = mrb_int_to_s(mrb, num, base); // convert integer to string
  return mrb_str_new_cstr(mrb, str); // return string
