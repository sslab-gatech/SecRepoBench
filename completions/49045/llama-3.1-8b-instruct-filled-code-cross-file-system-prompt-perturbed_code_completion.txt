static mrb_value
int_to_s(mrb_state *state, mrb_value self)
{
  mrb_int num;
  mrb_get_args(state, "i", &num);
  if (mrb_nil_p(num)) {
    return mrb_str_new_cstr(state, "nil");
  }
  if (mrb_fixnum_p(num)) {
    return mrb_str_new_cstr(state, mrb_fixnum_to_s(num));
  }
  return mrb_str_new_cstr(state, mrb_int_to_s(num));
}
