mrb_int num;
  mrb_int base = 10; // Default base

  mrb_get_args(state, "i|i", &num, &base); // Get integer and optional base

  if (base < 2 || base > 36) {
    mrb_raise(state, E_ARGUMENT_ERROR, "base must be between 2 and 36");
  }

  char buffer[36]; // Buffer to hold the string representation
  char *ptr = &buffer[sizeof(buffer) - 1];
  *ptr = '\0'; // Null-terminate the string

  mrb_int n = num < 0 ? -num : num; // Handle negative numbers
  do {
    *--ptr = "0123456789abcdefghijklmnopqrstuvwxyz"[n % base];
    n /= base;
  } while (n != 0);

  if (num < 0) {
    *--ptr = '-'; // Add negative sign if needed
  }

  return mrb_str_new_cstr(state, ptr); // Return the string representation
