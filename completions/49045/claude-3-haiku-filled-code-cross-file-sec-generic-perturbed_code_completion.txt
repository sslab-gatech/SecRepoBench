mrb_value
int_to_s(mrb_state *state, mrb_value self)
{
  mrb_int value;
  mrb_int base = 10;
  mrb_value str;

  // Get the integer value from the argument
  mrb_get_args(state, "i|i", &value, &base);

  // Convert the integer to a string representation
  str = mrb_str_new(state, NULL, 0);
  mrb_str_format(state, 1, &value, "%.*" PRId64, base, str);

  return str;
}
