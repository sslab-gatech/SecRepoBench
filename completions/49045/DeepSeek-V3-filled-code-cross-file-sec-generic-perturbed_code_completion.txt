mrb_int base = 10;
  mrb_int num = mrb_integer(obj);
  mrb_get_args(mrb, "|i", &base);

  if (base < 2 || base > 36) {
    mrb_raise(mrb, E_ARGUMENT_ERROR, "invalid radix");
  }

  char buf[MRB_INT_BIT+2]; // enough for binary representation + sign + null
  char *p = buf + sizeof(buf) - 1;
  *p = '\0';

  int neg = 0;
  if (num < 0) {
    neg = 1;
    num = -num;
  }

  do {
    mrb_int digit = num % base;
    *--p = (digit < 10) ? '0' + digit : 'a' + digit - 10;
    num /= base;
  } while (num > 0);

  if (neg) {
    *--p = '-';
  }

  return mrb_str_new_cstr(mrb, p);
