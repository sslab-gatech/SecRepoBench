mrb_int num;
mrb_int base = 10; // default base

mrb_get_args(state, "i|i", &num, &base);

if (base < 2 || base > 36) {
  mrb_raise(state, E_ARGUMENT_ERROR, "base must be between 2 and 36");
}

char buffer[36]; // buffer to hold the string representation
char *ptr = &buffer[sizeof(buffer) - 1];
*ptr = '\0';

mrb_int n = num;
do {
  mrb_int digit = n % base;
  *(--ptr) = (digit < 10) ? (digit + '0') : (digit - 10 + 'a');
  n /= base;
} while (n != 0);

return mrb_str_new_cstr(state, ptr);
