/* Check if the image is NULL */
if (image == (Image *) NULL)
{
  if (logging != MagickFalse)
    (void) LogMagickEvent(CoderEvent,GetMagickModule(),
      "  image is NULL, returning NULL");
  return((Image *) NULL);
}

/* Check if image dimensions are zero */
if (image->columns == 0 || image->rows == 0)
{
  if (logging != MagickFalse)
    (void) LogMagickEvent(CoderEvent,GetMagickModule(),
      "  image dimensions are zero, closing blob and returning destroyed image list");
  (void) CloseBlob(image);
  return(DestroyImageList(image));
}

/* Update mng_info to reference this image */
mng_info->image = image;
