if (image == NULL) {
  (void) LogMagickEvent(CoderEvent,GetMagickModule(),
    "  image is NULL, returning NULL.");
  return(NULL);
}

if (image->columns == 0 || image->rows == 0) {
  (void) LogMagickEvent(CoderEvent,GetMagickModule(),
    "  image dimensions are 0, returning NULL.");
  return(DestroyImageList(image));
}

mng_info->image = image;
return(image);
