if (image == (Image *) NULL)
{
    if (logging != MagickFalse)
        (void) LogMagickEvent(CoderEvent, GetMagickModule(), "  Image is NULL, returning NULL.");
    return (Image *) NULL;
}

if (image->columns == 0 || image->rows == 0)
{
    (void) CloseBlob(image);
    image = DestroyImageList(image);
    return (Image *) NULL;
}

mng_data->image = image;
