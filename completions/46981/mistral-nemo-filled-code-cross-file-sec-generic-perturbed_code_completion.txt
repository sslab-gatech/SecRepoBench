                Image
                  *cropped_image;

                ssize_t
                  y;

                Quantum
                  *q;

                /*
                  Allocate next image structure.
                */
                AcquireNextImage(image_info,image,exception);

                if (GetNextImageInList(image) == (Image *) NULL)
                  return(DestroyImageList(image));

                cropped_image=SyncNextImageInList(image);

                cropped_image->columns=crop_box.right-crop_box.left;
                cropped_image->rows=crop_box.bottom-crop_box.top;

                cropped_image->page.width=crop_box.right-crop_box.left;
                cropped_image->page.height=crop_box.bottom-crop_box.top;
                cropped_image->page.x=crop_box.left;
                cropped_image->page.y=crop_box.top;

                if (logging != MagickFalse)
                  (void) LogMagickEvent(CoderEvent,GetMagickModule(),
                    "    Cropped image, W=%.20g, H=%.20g",
                    (double) cropped_image->columns,
                    (double) cropped_image->rows);

                if (image->alpha_trait != UndefinedPixelTrait)
                  (void) SetImageBackgroundColor(cropped_image,exception);

                else
                  {
                    cropped_image->background_color=image->background_color;
                    (void) SetImageBackgroundColor(cropped_image,exception);
                  }

                q=GetAuthenticPixels(image,0,0,image->columns,1,exception);
                if (q == (Quantum *) NULL)
                  {
                    cropped_image=DestroyImage(cropped_image);
                    break;
                  }

                for (y=0; y < (ssize_t) image->rows; y++)
                {
                  Quantum
                    *p;

                  p=q+(crop_box.top-y)*image->columns+crop_box.left;
                  (void) memcpy(cropped_image->pixels+y*cropped_image->columns,
                    p,cropped_image->columns*GetPixelChannels(cropped_image));
                  q+=image->columns*GetPixelChannels(image);
                }

                if (SyncAuthenticPixels(cropped_image,exception) == MagickFalse)
                  {
                    cropped_image=DestroyImage(cropped_image);
                    break;
                  }

                DeleteImageFromList(&image);

                image=cropped_image;
                mng_data->image=image;
              }
