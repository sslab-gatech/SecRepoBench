if (image == NULL)
{
  if (logging != MagickFalse)
    (void) LogMagickEvent(CoderEvent,GetMagickModule(),
      "  ReadOneMNGImage() failed to read image");
  return NULL;
}

if (image->columns == 0 || image->rows == 0)
{
  (void) CloseBlob(image);
  return DestroyImageList(image);
}

mng_data->image = image;

if (mng_data->image != NULL)
{
  if (mng_data->image->columns == 0 || mng_data->image->rows == 0)
  {
    mng_data->image = DestroyImageList(mng_data->image);
    mng_data->image = NULL;
  }
}

if (mng_data->image == NULL)
{
  if (logging != MagickFalse)
    (void) LogMagickEvent(CoderEvent,GetMagickModule(),
      "  ReadOneMNGImage() failed to read valid image");
  return NULL;
}
