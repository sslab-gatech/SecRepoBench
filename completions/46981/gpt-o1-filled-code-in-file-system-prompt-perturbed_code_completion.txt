if (image == (Image *) NULL)
  {
    /* Image pointer is null, log an error if logging is enabled, then return. */
    if (logging != MagickFalse)
      (void) LogMagickEvent(CoderEvent,GetMagickModule(),
        "ReadOnePNGImage() returned a NULL image, returning NULL");
    return((Image *) NULL);
  }

  /* If image dimensions are zero, close the image blob and destroy the image list. */
  if (image->columns == 0 || image->rows == 0)
  {
    if (logging != MagickFalse)
      (void) LogMagickEvent(CoderEvent,GetMagickModule(),
        "exit ReadOnePNGImage() with error.");
    (void) CloseBlob(image);
    ThrowReaderException(CorruptImageError,"CorruptImage");
  }

  /* The image is valid with non-zero dimensions; update mng_data to reference it. */
  mng_info->image = image;
