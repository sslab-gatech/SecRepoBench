if(extension_id == 43 /* supported_versions */) {
  u_int16_t s_offset = offset+extension_offset;
  u_int16_t tot_supported_versions_len = packet->payload[s_offset];
  
  s_offset++;
  tot_supported_versions_len += s_offset;
  
  while(s_offset < tot_supported_versions_len && s_offset < total_len) {
    u_int16_t version = ntohs(*((u_int16_t*)&packet->payload[s_offset]));
    
    if(ja3.client.num_supported_versions < MAX_NUM_TLS_SUPPORTED_VERSIONS)
      ja3.client.supported_versions[ja3.client.num_supported_versions++] = version;
    
    s_offset += 2;
  }
  
  if(ja3.client.num_supported_versions > 0) {
    int rc = snprintf(ja3.client.supported_versions, sizeof(ja3.client.supported_versions), "%u", ja3.client.supported_versions[0]);
    for(i=1; i<ja3.client.num_supported_versions; i++) {
      rc += snprintf(ja3.client.supported_versions + rc, sizeof(ja3.client.supported_versions) - rc, "-%u", ja3.client.supported_versions[i]);
    }
  }
} else if(extension_id == 65486 /* encrypted_server_name */) {
  flow->protos.tls_quic.encrypted_sni.esni = 1;
} else if(extension_id == 0x39 /* QUIC transport parameters */) {
  processQuicTransportParameters(ndpi_struct, flow, extension_len, offset+extension_offset);
