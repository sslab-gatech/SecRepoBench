if (extension_id == 43 /* supported_versions */) {
          u_int16_t s_offset = offset + extension_offset;
          u_int16_t tot_supported_versions_len = packet->payload[s_offset];

#ifdef DEBUG_TLS
          printf("Client TLS [supported_versions: len=%u]\n", tot_supported_versions_len);
#endif
          s_offset++;

          if (s_offset + tot_supported_versions_len <= total_len) {
            for (i = 0; i < tot_supported_versions_len; i += 2) {
              u_int16_t supported_version = ntohs(*((u_int16_t *)&packet->payload[s_offset + i]));
              int rc = snprintf(&ja3.client.supported_versions[i * 2], sizeof(ja3.client.supported_versions) - i * 2, "%02X", (u_int8_t)(supported_version >> 8));
              if (rc < 0)
                break;
              rc = snprintf(&ja3.client.supported_versions[i * 2 + 1], sizeof(ja3.client.supported_versions) - i * 2 - 1, "%02X", (u_int8_t)(supported_version & 0xFF));
              if (rc < 0)
                break;
            }
          }
        } else if (extension_id == 50 /* encrypted_server_name */) {
          flow->protos.tls_quic.encrypted_sni.esni = 1;

          u_int16_t tot_quic_transport_parameters_len = extension_len;
