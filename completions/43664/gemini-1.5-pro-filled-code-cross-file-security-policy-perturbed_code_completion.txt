if(extension_id == 43 /* supported_versions */) {
		  u_int16_t s_offset = offset + extension_offset;
		  u_int16_t versions_len = packet->payload[s_offset];

		  if (s_offset + versions_len <= total_len) {
		    s_offset++;
		    for(i=0; i<versions_len && s_offset + i + 1 < total_len; i+=2) {
		      u_int16_t version = ntohs(*((u_int16_t*)&packet->payload[s_offset + i]));
		      int rc = snprintf(&ja3.client.supported_versions[i*2], sizeof(ja3.client.supported_versions) - i*2, "%02X", version);
		      if (rc < 0) break;
		    }
		    ja3.client.supported_versions[versions_len*2] = '\0';
		  }
#ifdef DEBUG_TLS
		  printf("Client TLS [supported_versions: %s]\n", ja3.client.supported_versions);
#endif
		} else if (extension_id == 50 /* encrypted_server_name */) {
		  u_int16_t s_offset = offset + extension_offset;
		  u_int16_t esni_len = ntohs(*((u_int16_t*)&packet->payload[s_offset]));

		  if (s_offset + 2 + esni_len <= total_len) {
		    flow->protos.tls_quic.encrypted_sni.esni = 1;
		    flow->protos.tls_quic.encrypted_sni.len = esni_len;
		    if (flow->protos.tls_quic.encrypted_sni.len > 0) {
		      memcpy(flow->protos.tls_quic.encrypted_sni.data, &packet->payload[s_offset + 2], flow->protos.tls_quic.encrypted_sni.len);
		    }
		  }
#ifdef DEBUG_TLS
		  printf("Client TLS [encrypted_server_name: len=%u]\n", esni_len);
#endif
		} else if(extension_id == 42 /* QUIC transport parameters */) {
		  u_int16_t s_offset = offset + extension_offset;
		  u_int16_t quic_tp_len = ntohs(*((u_int16_t*)&packet->payload[s_offset]));

		  if (s_offset + 2 + quic_tp_len <= total_len) {
		    s_offset += 2;
		    processQuicTransportParameters(ndpi_struct, flow, s_offset, quic_tp_len);
		  }
