char * next_comma = strchr(alpn_start, ',');
    if (next_comma != NULL) {
      *next_comma = '\0';
      comma_or_nul = next_comma;
    } else {
      comma_or_nul = alpn_start + strlen(alpn_start);
    }

    if (!ndpi_is_alpn_common(alpn_start)) {
      char alpn_string[256];
      snprintf(alpn_string, sizeof(alpn_string), "Uncommon ALPN: %s", alpn_start);

      NDPI_LOG_DBG(ndpi_struct, alpn_string);
      ndpi_set_risk(ndpi_struct, connectionflow, NDPI_RISK_UNCOMMON_ALPN, alpn_string);
    }
