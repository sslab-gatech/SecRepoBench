comma_or_nul = strchr(alpn_start, ',');
    if (comma_or_nul == NULL) {
      comma_or_nul = strchr(alpn_start, '\0');
    }

    if (ndpi_is_common_alpn(alpn_start, comma_or_nul - alpn_start) == 0) {
      char alpn_str[256];

      snprintf(alpn_str, sizeof(alpn_str), "Uncommon ALPN: %.*s",
               (int)(comma_or_nul - alpn_start), alpn_start);

      NDPI_LOG_DBG(ndpi_struct, "%s", alpn_str);
      ndpi_set_risk(ndpi_struct, connectionflow, NDPI_UNCOMMON_ALPN, alpn_str);
    }
