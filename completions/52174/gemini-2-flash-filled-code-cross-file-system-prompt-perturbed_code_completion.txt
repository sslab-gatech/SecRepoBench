	  u_int32_t username_len = 0;
	  const char *username_ptr = ndpi_get_next_word(packet->payload + 5, packet->payload_packet_len - 5, &username_len);

	  if (username_ptr != NULL && username_len > 0 && username_len < sizeof(flow->l4.tcp.irc_username) - 1) {
	    memcpy(flow->l4.tcp.irc_username, username_ptr, username_len);
	    flow->l4.tcp.irc_username[username_len] = '\0';
	    NDPI_LOG_DBG(detection_module, "Found IRC username: %s", flow->l4.tcp.irc_username);
	    NDPI_SET_FLOW_RISK(flow, NDPI_RISK_CLEAR_TEXT_CREDENTIALS);
	  }
	}
