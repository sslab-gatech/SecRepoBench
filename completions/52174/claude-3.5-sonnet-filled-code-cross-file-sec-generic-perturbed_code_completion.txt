if (memcmp(packet->payload, "USER ", 5) == 0) {
  u_int16_t username_len = 0;
  const u_int16_t max_username_len = 32;
  char username[max_username_len + 1];

  // Extract the username
  for (i = 5; i < packet->payload_packet_len && i < max_username_len + 5; i++) {
    if (packet->payload[i] == ' ' || packet->payload[i] == '\r' || packet->payload[i] == '\n') {
      break;
    }
    username[username_len++] = packet->payload[i];
  }
  username[username_len] = '\0';

  NDPI_LOG_INFO(detection_module, "Found IRC username: %s", username);
  flow->risk |= NDPI_CLEAR_TEXT_CREDENTIALS;
}

NDPI_LOG_DBG2(detection_module, "IRC-related keywords detected");
flow->l4.tcp.irc_stage = 1;
