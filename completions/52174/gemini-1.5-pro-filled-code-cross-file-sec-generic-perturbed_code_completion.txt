if (memcmp(packet->payload, "USER ", 5) == 0) {
          const u_int8_t *username_ptr = packet->payload + 5;
          size_t username_len = 0;
          while (username_ptr + username_len < packet->payload + packet->payload_packet_len &&
                 username_ptr[username_len] != ' ' && username_ptr[username_len] != '\r' &&
                 username_ptr[username_len] != '\n') {
            username_len++;
          }

          if (username_len > 0) {
            char username[username_len + 1];
            memcpy(username, username_ptr, username_len);
            username[username_len] = '\0';
            NDPI_LOG_INFO(detection_module, "Found IRC username: %s", username);

            flow->risk_level = NDPI_RISK_LEVEL_CLEARTEXT_CREDENTIALS;
          }
	}
