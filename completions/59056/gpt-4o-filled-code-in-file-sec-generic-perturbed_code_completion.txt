            ParseError("unexpected end of file", parser.LastToken());
        }

        // Insert the element into the map
        elements.insert(std::make_pair(element->KeyToken().StringContents(), element));

        if (!isTopLevel) {
            n = parser.AdvanceToNextToken();
            if (n == nullptr) {
                delete_Element(element);
