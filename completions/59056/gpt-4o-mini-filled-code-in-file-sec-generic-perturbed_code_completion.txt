if (n->Type() == TokenType_CLOSE_BRACKET) {
            ParseError("unexpected closing bracket", n);
        }

        // Insert the element into the map
        if (elements.find(str) != elements.end()) {
            delete_Element(element);
            ParseError("duplicate key found: " + str, n);
        }
        elements[str] = element;

        // Advance to the next token
        n = parser.AdvanceToNextToken();
    }
