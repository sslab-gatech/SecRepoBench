if (n->Type() == TokenType_CLOSE_BRACKET) {
            // If we are at the top level, insert the element into the map
            if (isTopLevel) {
                elements.emplace(str, element);
                return;
            } else {
                // If not at the top level, delete the element and throw an error
                delete_Element(element);
                ParseError("unexpected end of file, expected closing bracket", parser.LastToken());
            }
        }

        // If we have a valid token, insert the element into the map
        if (n->Type() == TokenType_KEY) {
            elements.emplace(n->StringContents(), element);
        }
