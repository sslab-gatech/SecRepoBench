size_t num_digits = 0;
if (mpz_sgn(*number) == 0) {
  num_digits = 1; // Special case for zero
} else {
  mpz_t temp;
  mpz_init(temp);
  mpz_set(temp, *number);
  while (mpz_cmp_ui(temp, 0) != 0) {
    mpz_divmod_ui(temp, NULL, temp, 10);
    num_digits++;
  }
  mpz_clear(temp);
}
return num_digits;
