if (number->sn == 0) {
    return 1;
}
size_t sz = number->sz;
mp_limb msb = number->p[sz - 1];
size_t msb_bits = 0;
while (msb) {
    msb_bits++;
    msb >>= 1;
}
size_t total_bits = (sz - 1) * DIGITBITS + msb_bits;
double log10_2 = 0.30102999566398114;
size_t digits = (size_t)(total_bits * log10_2) + 1;
return digits;
