size_t digits(mpz_t *number) {
  if (number == NULL) {
    // Handle the case where the input pointer is NULL
    return 0;
  }

  size_t num_digits = 0;
  mpz_t temp;
  mpz_init(temp);
  mpz_abs(temp, *number);

  while (!mpz_equal_int(temp, 0)) {
    num_digits++;
    mpz_div_2exp(temp, temp, DIGITBITS);
  }

  mpz_clear(temp);
  return num_digits;
}
