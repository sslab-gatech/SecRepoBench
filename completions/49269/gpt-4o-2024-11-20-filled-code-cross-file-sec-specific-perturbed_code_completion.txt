if (number == NULL || number->p == NULL) {
    return 0; // Return 0 if the input pointer is NULL or uninitialized
  }

  size_t count = 0;
  mpz_t temp;
  mpz_init(NULL, &temp); // Initialize a temporary mpz_t variable

  mpz_set(NULL, &temp, number); // Copy the input number to the temporary variable
  while (temp.sz > 0) {
    count++;
    mpz_div_2exp(NULL, &temp, &temp, DIGITBITS); // Divide the number by 2^DIGITBITS
  }

  mpz_clear(NULL, &temp); // Clear the temporary variable
  return count;
