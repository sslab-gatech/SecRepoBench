if (number->sz == 0) {
    return 0; // Return 0 if the number is empty
  }
  
  size_t count = 0;
  for (size_t i = number->sz; i > 0; i--) {
    if (number->p[i - 1] != 0) {
      count += (i - 1) * DIGITBITS; // Count the bits of the non-zero limb
      break;
    }
  }
  
  // Count the bits in the last non-zero limb
  count += __builtin_clz(number->p[number->sz - 1]);
  
  return count;
