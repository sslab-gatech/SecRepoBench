static size_t
digits(mpz_t *number)
{
  // This function returns the number of significant digits in a multi-precision integer.
  size_t count = 0;
  mpz_t temp;
  mpz_init(temp);
  mpz_set(temp, number);

  while (mpz_cmp_ui(temp, 0) != 0) {
    mpz_div_ui(temp, temp, 10);
    count++;
  }

  mpz_clear(temp);
  return count;
}
