size_t count = 0;
mpz_t temp;
mpz_init_set(temp, *number);

if (temp.sn == 0) {
  // If the number is zero, it has exactly one digit
  count = 1;
} else {
  // Count the number of digits in base 10
  while (temp.sz > 0) {
    mpz_div_2exp(NULL, &temp, &temp, DIGITBITS); // Divide by base (10^DIGITBITS)
    count++;
  }
}

mpz_clear(NULL, &temp);
return count;
