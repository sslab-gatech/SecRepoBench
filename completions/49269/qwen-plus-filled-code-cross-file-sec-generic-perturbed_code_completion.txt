size_t count = 0;
if (mpz_sgn(*number) == 0) {
  return 1; // Special case: number is zero, which has 1 digit.
}
mpz_t temp;
mpz_init(temp);
mpz_set(temp, *number);
mpz_abs(temp, temp);

while (mpz_cmp_ui(temp, 0) > 0) {
    mpz_fdiv_q_ui(temp, temp, 10); // Divide by 10
    count++;
}

mpz_clear(temp);
return count;
