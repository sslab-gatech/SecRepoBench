size_t count = 0;
if (number->sz == 0) {
  return 1; // If the number is zero, it has one significant digit.
}
for (size_t i = 0; i < number->sz; i++) {
  mp_limb limb = number->p[i];
  if (limb != 0) {
    count += sizeof(mp_limb) * CHAR_BIT - __builtin_clzl(limb);
  } else {
    count += sizeof(mp_limb) * CHAR_BIT;
  }
}
return count;
