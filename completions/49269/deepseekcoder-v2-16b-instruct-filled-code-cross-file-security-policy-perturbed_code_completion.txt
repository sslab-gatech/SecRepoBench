size_t digits_count = 0;
  mpz_t temp;
  mpz_init(temp);
  mpz_set(temp, *number);

  if (mpz_sgn(temp) == 0) {
    mpz_clear(temp);
    return 1; // A single digit for zero
  }

  while (mpz_sgn(temp) != 0) {
    mpz_div_int(temp, temp, 10);
    digits_count++;
  }

  mpz_clear(temp);
  return digits_count;
