size_t count = 0;
mpz_t temp;
mpz_init(NULL, &temp);
mpz_set(NULL, &temp, number);

if (temp.sn == 0) {
    mpz_clear(NULL, &temp);
    return 1;  // Zero has one significant digit
}

// Count digits by dividing by 10 until the number becomes zero
while (temp.sn != 0) {
    mpz_div_int(NULL, &temp, &temp, 10);
    count++;
}

mpz_clear(NULL, &temp);
return count;
