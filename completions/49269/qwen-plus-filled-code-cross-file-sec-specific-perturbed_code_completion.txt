if (number == NULL) {
    return 0;
}

size_t num_digits = 0;
if (mpz_cmp_ui(*number, 0) == 0) {
    return 1; // Special case: 0 has one digit
}

mpz_t temp;
mpz_init(temp);
mpz_set(temp, *number);
mpz_abs(temp, temp);

while (mpz_cmp_ui(temp, 0) > 0) {
    num_digits++;
    mpz_tdiv_q_ui(temp, temp, 10); // Divide by 10 to reduce the number of digits
}

mpz_clear(temp);
return num_digits;
