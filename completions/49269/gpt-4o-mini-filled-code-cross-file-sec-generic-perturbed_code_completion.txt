size_t count = 0;
  mp_limb *digits_ptr = number->p;

  for (size_t i = number->sz; i > 0; i--) {
    if (digits_ptr[i - 1] != 0) {
      count += (i - 1) * DIGITBITS; // Count the full limbs
      // Count the significant bits in the last limb
      count += __builtin_clz(digits_ptr[i - 1]) ^ (DIGITBITS - 1);
      break;
    }
  }

  return count > 0 ? count : 1; // Return at least 1 for zero
