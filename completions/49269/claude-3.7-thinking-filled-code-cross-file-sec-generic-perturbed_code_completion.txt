if (number == NULL) {
  return 0;  // Null has no significant digits
}

if (number->sn == 0 || number->sz == 0) {
  return 0;  // Zero or empty has no significant digits
}

// Find the highest non-zero limb
size_t i;
for (i = number->sz; i > 0; i--) {
  if (number->p[i - 1] != 0) {
    break;
  }
}

return i;  // Number of significant limbs
