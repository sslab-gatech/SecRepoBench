struct softether_value first_value;
  struct softether_value second_value;
  size_t processed_len = 0;
  int ret = 0;

  if(payload_len < 4)
    return 1;

  tuple_count = ntohl(get_u_int32_t(payload, 0));
  payload += 4;
  payload_len -= 4;

  if(tuple_count == 0)
    return 1;

  for(u_int32_t i = 0; i < tuple_count; i++) {
    processed_len += dissect_softether_tuples(payload + processed_len, payload_len - processed_len,
                                              &first_value, &second_value);
    if(processed_len == 0)
      return 1;

    if(first_value.type == VALUE_STR && second_value.type == VALUE_STR) {
      if(strncmp((char const *)first_value.value.ptr.raw, "host_name", first_value.value_size) == 0) {
        size_t len = ndpi_min(sizeof(ndpiflow->protos.softether.hostname) - 1, second_value.value_size);
        strncpy(ndpiflow->protos.softether.hostname, (char const *)second_value.value.ptr.raw, len);
        ndpiflow->protos.softether.hostname[len] = '\0';
      } else if(strncmp((char const *)first_value.value.ptr.raw, "ddns_fqdn", first_value.value_size) == 0) {
        size_t len = ndpi_min(sizeof(ndpiflow->protos.softether.fqdn) - 1, second_value.value_size);
        strncpy(ndpiflow->protos.softether.fqdn, (char const *)second_value.value.ptr.raw, len);
        ndpiflow->protos.softether.fqdn[len] = '\0';
      }
    }
  }

  if(processed_len != payload_len)
    ret = 1;

  return ret;
