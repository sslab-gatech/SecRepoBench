if (sourcedata == NULL || srclen <= 0) {
    mrb_raise(mrb, E_ARGUMENT_ERROR, "invalid source data");
  }

  for (i = 0; i < srclen; i++) {
    unsigned char byte = sourcedata[i];
    if (n > (MRB_INT_MAX >> 7)) { // Prevent integer overflow
      mrb_raise(mrb, E_ARGUMENT_ERROR, "BER integer too large");
    }
    n = (n << 7) | (byte & 0x7F);
    if ((byte & 0x80) == 0) { // Check if this is the last byte
      i++; // Move to the next byte after the last one
      break;
    }
  }

  if (i == srclen && (sourcedata[srclen - 1] & 0x80) != 0) {
    mrb_raise(mrb, E_ARGUMENT_ERROR, "incomplete BER integer");
  }
