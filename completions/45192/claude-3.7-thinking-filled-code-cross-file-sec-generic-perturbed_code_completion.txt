i = 0;
while (i < srclen) {
  unsigned char c = sourcedata[i];
  
  /* Check for potential overflow */
  if (n > MRB_INT_MAX / 128) {
    mrb_raise(mrb, E_RANGE_ERROR, "BER integer too large");
  }
  
  n = (n << 7) | (c & 0x7f);
  
  i++; // Move to the next byte
  
  if (!(c & 0x80)) { // If the 8th bit is clear, this is the last byte
    break;
  }
}

if (i == srclen && i > 0 && (sourcedata[i-1] & 0x80)) {
  mrb_raise(mrb, E_ARGUMENT_ERROR, "incomplete BER format");
}
