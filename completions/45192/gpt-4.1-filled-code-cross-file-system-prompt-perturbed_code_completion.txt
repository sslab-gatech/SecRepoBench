for (i = 0; i < srclen; i++) {
  unsigned char c = sourcedata[i];
  if (n > (MRB_INT_MAX >> 7)) {
    mrb_raise(mrb, E_ARGUMENT_ERROR, "BER integer too large");
  }
  n = (n << 7) | (c & 0x7f);
  if ((c & 0x80) == 0) {
    i++;
    break;
  }
}
if (i == srclen && (sourcedata[srclen-1] & 0x80)) {
  mrb_raise(mrb, E_ARGUMENT_ERROR, "malformed BER integer");
}
