for (i = 0; i < srclen; i++) {
    n = (n << 7) | (sourcedata[i] & 0x7f);
    if (!(sourcedata[i] & 0x80)) {
      i++;  // Include the last byte in the count
      break;
    }
    // Check for overflow
    if (n > (MRB_INT_MAX >> 7)) {
      mrb_raise(mrb, E_RANGE_ERROR, "BER compressed integer too large");
    }
  }
  
  // Check if we ran out of data with an incomplete number
  if (i == srclen && srclen > 0 && (sourcedata[srclen-1] & 0x80)) {
    mrb_raise(mrb, E_ARGUMENT_ERROR, "incomplete BER encoded integer");
  }
