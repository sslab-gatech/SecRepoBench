for (i = 0; i < srclen; i++) {
    mrb_int byte = sourcedata[i];
    n = (n << 7) | (byte & 0x7F);
    if ((byte & 0x80) == 0) {
      break; // If the 8th bit is clear, the value is complete
    }
  }
  
  if (i == srclen) {
    mrb_raise(mrb, E_ARGUMENT_ERROR, "invalid BER integer");
  }
  
  return i + 1; // Return the number of bytes read
