if (srclen == 0) {
  mrb_raise(mrb, E_ARGUMENT_ERROR, "empty input for BER unpacking");
}

for (i = 0; i < srclen; i++) {
  n = (n << 7) | (sourcedata[i] & 0x7f);
  if (!(sourcedata[i] & 0x80)) {
    break;
  }
  if (i == srclen - 1) {
    mrb_raise(mrb, E_ARGUMENT_ERROR, "invalid BER encoding");
  }
}

if (i == srclen) {
  mrb_raise(mrb, E_ARGUMENT_ERROR, "incomplete BER encoding");
}
