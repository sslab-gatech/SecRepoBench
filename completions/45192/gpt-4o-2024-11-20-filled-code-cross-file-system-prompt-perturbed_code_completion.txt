for (i = 0; i < srclen; i++) {
    unsigned char byte = sourcedata[i];
    n = (n << 7) | (byte & 0x7F);
    if ((byte & 0x80) == 0) {
      // If the 8th bit is clear, the value is complete
      break;
    }
    if (i == srclen - 1) {
      // If we reach the end of the source data without completing the value, raise an exception
      mrb_raise(mrb, E_ARGUMENT_ERROR, "incomplete BER integer");
    }
  }
  if (i == srclen) {
