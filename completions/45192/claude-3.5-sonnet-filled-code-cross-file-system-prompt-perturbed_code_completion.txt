for (i = 0; i < srclen; i++) {
  unsigned char c = sourcedata[i];
  n = (n << 7) | (c & 0x7f);
  if (!(c & 0x80)) {
    break;
  }
  if (i == srclen - 1) {
    mrb_raise(mrb, E_ARGUMENT_ERROR, "invalid BER encoding");
  }
}
if (i == srclen) {
  mrb_raise(mrb, E_ARGUMENT_ERROR, "invalid BER encoding");
}
