if (srclen <= 0) {
    mrb_raise(mrb, E_ARGUMENT_ERROR, "source data is empty");
  }

  for (i = 0; i < srclen; i++) {
    unsigned char byte = sourcedata[i];
    if (n > (MRB_INT_MAX >> 7)) { // Check for overflow
      mrb_raise(mrb, E_ARGUMENT_ERROR, "BER integer overflow");
    }
    n = (n << 7) | (byte & 0x7F); // Accumulate 7 bits from the current byte
    if ((byte & 0x80) == 0) { // If the 8th bit is clear, the integer is complete
      i++; // Move to the next byte after the last processed one
      break;
    }
  }

  if (i == srclen && (sourcedata[srclen - 1] & 0x80) != 0) {
    mrb_raise(mrb, E_ARGUMENT_ERROR, "incomplete BER integer");
  }
