if (srclen == 0) {
  mrb_raise(mrb, E_ARGUMENT_ERROR, "empty BER data");
}
for (i = 0; i < srclen; i++) {
  unsigned char byte = sourcedata[i];
  if (i >= (int)(sizeof(mrb_int) + 1)) {
    mrb_raise(mrb, E_ARGUMENT_ERROR, "BER encoding too long");
  }
  n = (n << 7) | (byte & 0x7F);
  if ((byte & 0x80) == 0) {
    i++;  /* consume the final byte */
    break;
  }
}
if (i == srclen && (sourcedata[i - 1] & 0x80)) {
  mrb_raise(mrb, E_ARGUMENT_ERROR, "incomplete BER encoding");
}
