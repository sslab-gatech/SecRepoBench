n = 0;
  i = 0;
  while (i < srclen) {
    unsigned char byte = sourcedata[i++];
    n = (n << 7) | (byte & 0x7F);
    if (!(byte & 0x80)) {
      break; // Last byte
    }
  }
  
  if (i == srclen && (sourcedata[i - 1] & 0x80)) {
    mrb_raise(mrb, E_ARGUMENT_ERROR, "invalid BER integer");
  }
  
  return i;
