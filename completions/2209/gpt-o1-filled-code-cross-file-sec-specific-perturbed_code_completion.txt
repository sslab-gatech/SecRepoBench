{
    HEVCFrame *ref = find_ref_idx(s, pictureordercount);
    if (ref) {
        // Check for duplicate reference
        for (int i = 0; i < list->nb_refs; i++) {
            if (list->ref[i] == ref) {
                av_log(s->avctx, AV_LOG_ERROR, "Duplicate reference with POC %d\n", pictureordercount);
                return AVERROR_INVALIDDATA;
            }
        }
    } else {
        // Attempt to generate a missing reference
        ref = ff_hevc_new_missing_reference(s, pictureordercount);
        if (!ref) {
            av_log(s->avctx, AV_LOG_ERROR, "Failed to create missing reference for POC %d\n", pictureordercount);
            return AVERROR_INVALIDDATA;
        }
    }

    // Ensure we don't exceed the reference list boundary
    if (list->nb_refs >= FF_ARRAY_ELEMS(list->ref)) {
        av_log(s->avctx, AV_LOG_ERROR, "Reference list is full\n");
        return AVERROR_INVALIDDATA;
    }
}
