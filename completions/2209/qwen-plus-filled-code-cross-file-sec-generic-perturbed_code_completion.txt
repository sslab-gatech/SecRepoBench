HEVCFrame *ref = find_ref_idx(s, pictureordercount);
if (!ref) {
    if (list->nb_refs >= MAX_REFS) {
        av_log(s->avctx, AV_LOG_ERROR, "Reference list overflow\n");
        return AVERROR_INVALIDDATA;
    }
    ref = generate_missing_ref(s, pictureordercount);
    if (!ref) {
        av_log(s->avctx, AV_LOG_ERROR, "Failed to generate missing reference frame for POC %d\n", pictureordercount);
        return AVERROR_UNKNOWN;
    }
} else {
    for (int i = 0; i < list->nb_refs; i++) {
        if (list->list[i] == ref->poc) {
            av_log(s->avctx, AV_LOG_ERROR, "Duplicate reference with POC %d\n", ref->poc);
            return AVERROR_INVALIDDATA;
        }
    }
}
