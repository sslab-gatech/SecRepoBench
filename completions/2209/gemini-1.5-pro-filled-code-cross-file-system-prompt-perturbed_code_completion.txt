    if (list->nb_refs >= MAX_REFS) {
        av_log(s->avctx, AV_LOG_ERROR, "Reference overflow (poc:%d)\n",
               pictureordercount);
        return AVERROR_INVALIDDATA;
    }

       return AVERROR(ENOMEM);
    }


    if (ref->poc == pictureordercount) {
