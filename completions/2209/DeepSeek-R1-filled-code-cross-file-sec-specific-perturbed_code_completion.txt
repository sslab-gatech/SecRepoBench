HEVCFrame *ref = find_ref_idx(s, pictureordercount);
    if (!ref) {
        av_log(s->avctx, AV_LOG_ERROR, "Reference for POC %d not found\n", pictureordercount);
        return AVERROR_INVALIDDATA;
    }

    for (int i = 0; i < list->nb_refs; i++) {
        if (list->ref[i] == ref) {
            av_log(s->avctx, AV_LOG_ERROR, "Duplicate reference POC %d\n", ref->poc);
            return AVERROR_INVALIDDATA;
        }
    }

    if (list->nb_refs >= HEVC_MAX_REFS) {
        av_log(s->avctx, AV_LOG_ERROR, "Reference list overflow\n");
        return AVERROR_INVALIDDATA;
    }
