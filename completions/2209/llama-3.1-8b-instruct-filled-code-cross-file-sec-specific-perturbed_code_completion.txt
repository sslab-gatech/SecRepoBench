// <MASK>
HEVCFrame *ref = find_ref_idx(s, pictureordercount);
if (ref && ref->poc == pictureordercount) {
    // If the found reference is the same as the current reference, 
    // return an error to prevent adding a duplicate or overflow.
    return AVERROR_INVALIDDATA;
}
if (ref) {
    // If a reference is found, attempt to add it to the reference list.
    list->list[list->nb_refs] = ref->poc;
    list->ref[list->nb_refs]  = ref;
    list->nb_refs++;
    mark_ref(ref, ref_flag);
    return 0;
} else {
    // If no reference is found, attempt to generate a missing reference frame for the given POC.
    // This may involve decoding a new frame or reusing an existing one.
    // For simplicity, let's assume we're reusing an existing frame with a similar POC.
    HEVCFrame *closest_ref = find_closest_ref(s, pictureordercount);
    if (closest_ref) {
        // Reuse the closest reference frame.
        list->list[list->nb_refs] = closest_ref->poc;
        list->ref[list->nb_refs]  = closest_ref;
        list->nb_refs++;
        mark_ref(closest_ref, ref_flag);
        return 0;
    } else {
        // If no suitable reference frame is found, return an error.
        return AVERROR_INVALIDDATA;
    }
}
