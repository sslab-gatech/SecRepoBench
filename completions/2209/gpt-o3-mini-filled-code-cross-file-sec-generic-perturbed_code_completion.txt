    int i;
    HEVCFrame *ref = find_ref_idx(s, pictureordercount);
    if (ref) {
        for (i = 0; i < list->nb_refs; i++) {
            if (list->ref[i] == ref) {
                av_log(s->avctx, AV_LOG_ERROR, "Reference frame with POC %d already added\n", pictureordercount);
                return AVERROR_INVALIDDATA;
            }
        }
    } else {
        ref = generate_missing_ref_frame(s, pictureordercount);
        if (!ref) {
            av_log(s->avctx, AV_LOG_ERROR, "Failed to generate missing reference frame for POC %d\n", pictureordercount);
            return AVERROR_INVALIDDATA;
        }
    }
