HEVCFrame *ref = find_ref_idx(s, pictureordercount);
if (ref) {
    for (int i = 0; i < list->nb_refs; i++) {
        if (list->ref[i] == ref) {
            av_log(s->avctx, AV_LOG_ERROR, "Duplicate reference POC %d\n", pictureordercount);
            return AVERROR_INVALIDDATA;
        }
    }
} else {
    av_log(s->avctx, AV_LOG_ERROR, "No reference found for POC %d\n", pictureordercount);
    return AVERROR_INVALIDDATA;
}

if (list->nb_refs >= FF_ARRAY_ELEMS(list->list)) {
    av_log(s->avctx, AV_LOG_ERROR, "Reference list overflow\n");
    return AVERROR_INVALIDDATA;
}
