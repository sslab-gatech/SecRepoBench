HEVCFrame *ref = find_ref_idx(s, poc);
if (ref == s->ref) {
    av_log(s->avctx, AV_LOG_ERROR, "Duplicate POC in a sequence: %d.\n", poc);
    return AVERROR_INVALIDDATA;
}
if (!ref) {
    ref = generate_missing_ref(s, poc);
    if (!ref)
        return AVERROR(ENOMEM);
}
