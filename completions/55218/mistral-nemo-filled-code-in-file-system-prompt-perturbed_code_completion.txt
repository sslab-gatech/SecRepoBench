if(is_utpv1_pkt(packet->payload, packet->payload_packet_len)) {
  struct ndpi_utp_hdr *h = (struct ndpi_utp_hdr*)packet->payload;
  if(h->h_type == 1) { // BitTorrent protocol
    NDPI_LOG_INFO(detectionmodule, "found BT: uTP\n");
