Dwarf_Unsigned subprog_format_count = 0;
Dwarf_Unsigned subprogs_count = 0;

// Read the subprogram format count from the current position of the data pointer.
int dres = read_uword_de(&line_ptr, &subprog_format_count, dbg, err, line_ptr_end);
if (dres != DW_DLV_OK) {
    _dwarf_error_string(dbg, err, DW_DLE_LINE_NUMBER_HEADER_ERROR,
        "Failed to read subprogram format count");
    return dres;
}

// Check if the current pointer position is within the valid range of the data section.
if (line_ptr + sizeof(Dwarf_Unsigned) > line_ptr_end) {
    _dwarf_error_string(dbg, err, DW_DLE_LINE_NUMBER_HEADER_ERROR,
        "Subprogram format count exceeds data section");
    return DW_DLV_ERROR;
}

// Increment the data pointer to move past the retrieved format count.
line_ptr += sizeof(Dwarf_Unsigned);
