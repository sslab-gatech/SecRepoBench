char
            buffer[MagickPathExtent];

          const char
            *value;

          if (p > (exif_data+length))
            break;
          if (components == 1)
            {
              switch (format)
              {
                case EXIF_FMT_BYTE:
                case EXIF_FMT_SBYTE:
                case EXIF_FMT_UNDEFINED:
                case EXIF_FMT_STRING:
                  value=(char *) p;
                  break;
                case EXIF_FMT_USHORT:
                  (void) FormatLocaleString(buffer,MagickPathExtent,"%hu",
                    ReadPropertyUnsignedShort(endian,p));
                  value=buffer;
                  break;
                case EXIF_FMT_ULONG:
                  (void) FormatLocaleString(buffer,MagickPathExtent,"%.20g",
                    (double) ReadPropertyUnsignedLong(endian,p));
                  value=buffer;
                  break;
                case EXIF_FMT_SSHORT:
                  (void) FormatLocaleString(buffer,MagickPathExtent,"%hd",
                    ReadPropertySignedShort(endian,p));
                  value=buffer;
                  break;
                case EXIF_FMT_SLONG:
                  (void) FormatLocaleString(buffer,MagickPathExtent,"%.20g",
                    (double) ReadPropertySignedLong(endian,p));
                  value=buffer;
                  break;
                case EXIF_FMT_SINGLE:
                  (void) FormatLocaleString(buffer,MagickPathExtent,"%g",
                    (double) ReadPropertySignedLong(endian,p));
                  value=buffer;
                  break;
                case EXIF_FMT_DOUBLE:
                  (void) FormatLocaleString(buffer,MagickPathExtent,"%g",
                    (double) ReadPropertySignedLong(endian,p));
                  value=buffer;
                  break;
                case EXIF_FMT_URATIONAL:
                case EXIF_FMT_SRATIONAL:
                default:
                  value="unknown";
                  break;
              }
            }
          else
            {
              value="unknown";
            }
