if (exception == (ExceptionInfo *) NULL)
  return((const MagicInfo *) NULL);
if ((magic_list == (LinkedListInfo *) NULL) ||
    (IsMagicCacheInstantiated() == MagickFalse))
  return((const MagicInfo *) NULL);

p = (const MagicInfo *) NULL;
if (magic != (const unsigned char *) NULL)
  {
    LockSemaphoreInfo(magic_cache_semaphore);
    ResetLinkedListIterator(magic_cache);
    const MagicInfo *cache_info;
    while ((cache_info = (const MagicInfo *) GetNextValueInLinkedList(magic_cache)) != (const MagicInfo *) NULL)
      {
        if ((cache_info->length == size) &&
            (memcmp(cache_info->magic, magic, size) == 0))
          {
            p = cache_info;
            break;
          }
      }
    UnlockSemaphoreInfo(magic_cache_semaphore);
    if (p != (const MagicInfo *) NULL)
      return(p);
  }

LockSemaphoreInfo(magic_list_semaphore);
ResetLinkedListIterator(magic_list);
while ((p = (const MagicInfo *) GetNextValueInLinkedList(magic_list)) != (const MagicInfo *) NULL)
  {
    if ((magic != (const unsigned char *) NULL) &&
        (p->length == size) &&
        (memcmp(p->magic, magic, size) == 0))
      break;
  }
