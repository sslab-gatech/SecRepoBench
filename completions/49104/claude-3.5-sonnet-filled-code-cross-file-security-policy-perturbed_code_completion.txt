if (exception == (ExceptionInfo *) NULL)
    return((const MagicInfo *) NULL);
if ((magic_list == (LinkedListInfo *) NULL) ||
    (IsMagicCacheInstantiated() == MagickFalse))
  return((const MagicInfo *) NULL);
if (magic != (const unsigned char *) NULL)
  {
    LockSemaphoreInfo(magic_cache_semaphore);
    p=(const MagicInfo *) GetHeadElementInLinkedList(magic_cache);
    while (p != (const MagicInfo *) NULL)
    {
      if (ValidateMagicBuffer(magic, size, p))
      {
        UnlockSemaphoreInfo(magic_cache_semaphore);
        return(p);
      }
      p=(const MagicInfo *) GetNextElementInLinkedList(magic_cache);
    }
    UnlockSemaphoreInfo(magic_cache_semaphore);
  }
LockSemaphoreInfo(magic_list_semaphore);
p=(const MagicInfo *) GetHeadElementInLinkedList(magic_list);
while (p != (const MagicInfo *) NULL)
{
  if ((magic == (const unsigned char *) NULL) ||
      ValidateMagicBuffer(magic, size, p))
    break;
  p=(const MagicInfo *) GetNextElementInLinkedList(magic_list);
}
