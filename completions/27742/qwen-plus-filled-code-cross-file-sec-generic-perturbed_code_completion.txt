parsed_object = result.data;
if (parsed_object.type != MSGPACK_OBJECT_ARRAY || parsed_object.via.array.size != 2) {
    continue;
}

msgpack_object root_array = parsed_object.via.array.ptr[0];
msgpack_object record_map = parsed_object.via.array.ptr[1];

if (root_array.type != MSGPACK_OBJECT_POSITIVE_INTEGER &&
    root_array.type != MSGPACK_OBJECT_NEGATIVE_INTEGER &&
    root_array.type != MSGPACK_OBJECT_FLOAT) {
    continue;
}

if (record_map.type != MSGPACK_OBJECT_MAP) {
    continue;
}

flb_time_pop_from_msgpack(&tms, &root_array);
map = record_map;
map_size = map.via.map.size;

if (date_key != NULL) {
    map_size += 1;
}
msgpack_pack_map(&tmp_pck, map_size);
