parsed_object = result.data;
if (parsed_object.type != MSGPACK_OBJECT_ARRAY || parsed_object.via.array.size != 2) {
    continue; // Skip if the root object is not an array with exactly two elements
}

// Unpack the timestamp and assign the second element to a map object
flb_time_pop_from_msgpack(&tms, &parsed_object.via.array.ptr[0], NULL);
map = parsed_object.via.array.ptr[1];

if (map.type != MSGPACK_OBJECT_MAP) {
    continue; // Skip if the second element is not a map
}

map_size = map.via.map.size;

// Adjust map size if a date key is provided
if (date_key != NULL) {
    map_size += 1;
}
