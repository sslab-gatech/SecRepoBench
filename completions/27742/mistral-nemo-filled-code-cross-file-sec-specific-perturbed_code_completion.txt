/* Unpack the timestamp and the map object */
    if (msgpack_unpack_next(&result, data, bytes, &off) == MSGPACK_UNPACK_SUCCESS) {
        if (result.data.type != MSGPACK_OBJECT_ARRAY ||
            result.data.via.array.size != 2) {
            continue;
        }

        /* Unpack the timestamp */
        if (msgpack_unpack_next(&result, data, bytes, &off) != MSGPACK_UNPACK_SUCCESS) {
            continue;
        }
        if (result.data.type != MSGPACK_OBJECT_POSITIVE_INTEGER) {
            continue;
        }
        time_lookup = result.data.via.u64;

        /* Unpack the map object */
        if (msgpack_unpack_next(&result, data, bytes, &off) != MSGPACK_UNPACK_SUCCESS) {
            continue;
        }
        if (result.data.type != MSGPACK_OBJECT_MAP) {
            continue;
        }
        map = result.data;

        /* Determine the size of the map */
        map_size = map.via.map.size;
        if (date_key != NULL) {
            /* If a date key is provided, increment the map size by one */
            map_size++;
        }
    }
    else {
        continue;
    }
