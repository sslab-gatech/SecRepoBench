parsed_object = result.data;
if (parsed_object.type != MSGPACK_OBJECT_ARRAY || parsed_object.via.array.size != 2) {
    continue;
}

msgpack_object timestamp = parsed_object.via.array.ptr[0];
if (timestamp.type != MSGPACK_OBJECT_POSITIVE_INTEGER && 
    timestamp.type != MSGPACK_OBJECT_NEGATIVE_INTEGER &&
    timestamp.type != MSGPACK_OBJECT_FLOAT &&
    timestamp.type != MSGPACK_OBJECT_DOUBLE) {
    continue;
}

map = parsed_object.via.array.ptr[1];
if (map.type != MSGPACK_OBJECT_MAP) {
    continue;
}

map_size = map.via.map.size;

if (date_key != NULL) {
    map_size++;
}

msgpack_pack_map(&tmp_pck, map_size);
