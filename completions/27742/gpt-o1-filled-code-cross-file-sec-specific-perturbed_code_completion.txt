parsed_object = result.data;
        /* ensure the root object is an array with exactly two elements */
        if (parsed_object.type != MSGPACK_OBJECT_ARRAY) {
            continue;
        }
        if (parsed_object.via.array.size != 2) {
            continue;
        }

        /* first element: timestamp */
        if (flb_time_pop_from_msgpack(&tms, &parsed_object, &i) != 0) {
            continue;
        }

        /* second element: must be a map */
        map = parsed_object.via.array.ptr[1];
        if (map.type != MSGPACK_OBJECT_MAP) {
            continue;
        }
        map_size = map.via.map.size;

        /* if a date key is provided, increment the map size */
        if (date_key != NULL) {
            map_size++;
        }

        /* pack a map in the temporary buffer with the computed size */
        msgpack_pack_map(&tmp_pck, map_size);
