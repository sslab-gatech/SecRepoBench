// Ensure the root object is an array with two elements
        parsed_object = result.data;
        if (parsed_object.type != MSGPACK_OBJECT_ARRAY || parsed_object.via.array.size != 2) {
            continue;
        }

        // Unpack the timestamp
        obj = &parsed_object.via.array.ptr[0];
        flb_time_pop_from_msgpack(&tms, obj, &tm);

        // Assign the second element to the map object and validate
        map = parsed_object.via.array.ptr[1];
        if (map.type != MSGPACK_OBJECT_MAP) {
            continue;
        }

        // Determine the size of the map
        map_size = map.via.map.size;
        if (date_key != NULL) {
            map_size++;
        }

        // Pack the map size into the temporary msgpack buffer
        msgpack_pack_map(&tmp_pck, map_size);
