/* Unpack the timestamp and map object */
    parsed_object = result.data;
    if (parsed_object.type != MSGPACK_OBJECT_ARRAY ||
        parsed_object.via.array.size != 2) {
        continue;
    }

    /* Unpack the timestamp */
    msgpack_unpack_next(&result, data, bytes, &off);
    if (result.data.type != MSGPACK_OBJECT_POSITIVE_INTEGER) {
        continue;
    }

    /* Unpack the map object */
    msgpack_unpack_next(&result, data, bytes, &off);
    if (result.data.type != MSGPACK_OBJECT_MAP) {
        continue;
    }

    map = result.data;
    map_size = map.via.map.size;

    /* If a date key is provided, increment the map size by one */
    if (date_key != NULL) {
        map_size++;
    }

    /* Pack the timestamp and map into the temporary message pack buffer */
    msgpack_pack_object(&tmp_pck, result.data);
    msgpack_pack_object(&tmp_pck, map);
