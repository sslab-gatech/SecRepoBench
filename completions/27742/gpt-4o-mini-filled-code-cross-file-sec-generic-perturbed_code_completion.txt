obj = &result.data;
        if (obj->type != MSGPACK_OBJECT_ARRAY || obj->via.array.size != 2) {
            continue; // Skip if not an array of size 2
        }

        // Unpack the timestamp
        if (msgpack_unpack_time(&obj->via.array.ptr[0], &tms) != 0) {
            continue; // Skip if timestamp unpacking fails
        }

        // Assign the second element to the map object
        map = obj->via.array.ptr[1];
        if (map.type != MSGPACK_OBJECT_MAP) {
            continue; // Skip if the second element is not a map
        }
        map_size = map.via.map.size;

        // If a date key is provided, increment the map size
        if (date_key != NULL) {
