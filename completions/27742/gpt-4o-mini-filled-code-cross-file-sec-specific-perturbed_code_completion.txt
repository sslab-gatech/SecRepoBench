            continue; // Skip if not an array of two elements
        }

        // Unpack the timestamp
        if (msgpack_unpack_next(&result, data, bytes, &off) == ok) {
            // Assuming the first element is the timestamp
            if (result.data.type == MSGPACK_OBJECT_POSITIVE_INTEGER) {
                tms.tm.tv_sec = result.data.via.u64;
                tms.tm.tv_nsec = 0; // Initialize nanoseconds to zero
            } else {
                continue; // Skip if timestamp is not a valid integer
            }
        }

        // The second element is the map
        if (msgpack_unpack_next(&result, data, bytes, &off) == ok) {
            map = result.data;
            if (map.type != MSGPACK_OBJECT_MAP) {
                continue; // Skip if not a map
            }
            map_size = map.via.map.size;

            // If a date key is provided, increment the map size
            if (date_key != NULL) {
                map_size++;
            }
