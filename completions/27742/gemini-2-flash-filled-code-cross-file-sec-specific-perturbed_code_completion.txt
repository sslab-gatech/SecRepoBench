parsed_object = result.data;

        /* Must be an array of two elements */
        if (parsed_object.type != MSGPACK_OBJECT_ARRAY ||
            parsed_object.via.array.size != 2) {
            continue;
        }

        /*
         * obj[0] = timestamp
         * obj[1] = map
         */
        obj = parsed_object.via.array.ptr;

        /* Unpack timestamp */
        flb_time_pop_from_object(&tms, &obj[0]);

        /* Get map */
        map = obj[1];
        if (map.type != MSGPACK_OBJECT_MAP) {
            continue;
        }
        map_size = map.via.map.size;

        /*
         * Pack the map into a temporary msgpack_sbuffer.
         *
         * If a date_key is set, we need to append it to the map. The
         * final map would be:
         *
         * {key1: val1, key2: val2, ..., date_key: date_value}
         */
        if (date_key != NULL) {
            msgpack_pack_map(&tmp_pck, map_size + 1);
        }
        else {
            msgpack_pack_map(&tmp_pck, map_size);
        }
