/* Unpack the timestamp and the map object */
    parsed_object = result.data;
    if (parsed_object.type != MSGPACK_OBJECT_ARRAY ||
        parsed_object.via.array.size != 2) {
        continue;
    }

    /* Unpack the timestamp */
    if (parsed_object.via.array.ptr[0].type != MSGPACK_OBJECT_POSITIVE_INTEGER) {
        continue;
    }
    time_lookup = parsed_object.via.array.ptr[0].via.u64;

    /* Unpack the map object */
    map = parsed_object.via.array.ptr[1];
    if (map.type != MSGPACK_OBJECT_MAP) {
        continue;
    }
    map_size = map.via.map.size;

    /* Pack the timestamp and the map object into the temporary message pack buffer */
    msgpack_pack_array(&tmp_pck, 2);
    msgpack_pack_uint64(&tmp_pck, time_lookup);
    msgpack_pack_map(&tmp_pck, map_size);

    /* Pack the remaining keys/values into the temporary message pack buffer */
    for (i = 0; i < map_size; i++) {
        k = &map.via.map.ptr[i].key;
        v = &map.via.map.ptr[i].val;
        msgpack_pack_object(&tmp_pck, *k);
        msgpack_pack_object(&tmp_pck, *v);
    }
