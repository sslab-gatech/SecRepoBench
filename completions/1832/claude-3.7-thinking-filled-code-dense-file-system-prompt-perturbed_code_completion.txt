if (s->num_primitive_matrices > max_primitive_matrices) {
    av_log(m->avctx, AV_LOG_ERROR, "Too many primitive matrices: %d, maximum is %d\n", 
           s->num_primitive_matrices, max_primitive_matrices);
    s->num_primitive_matrices = 0;
    return AVERROR_INVALIDDATA;
}

memset(s->matrix_coeff, 0, sizeof(s->matrix_coeff));
memset(s->matrix_noise_shift, 0, sizeof(s->matrix_noise_shift));

s->max_matrix_channel = 0;

for (mat = 0; mat < s->num_primitive_matrices; mat++) {
    unsigned int out_ch = get_bits(gbp, 4);
    unsigned int fbits = get_bits(gbp, 4);
    unsigned int lsb_bypass = get_bits1(gbp);
    
    if (out_ch > s->max_channel) {
        av_log(m->avctx, AV_LOG_ERROR, "Invalid matrix output channel: %d\n", out_ch);
        s->num_primitive_matrices = 0;
        return AVERROR_INVALIDDATA;
    }
    
    if (fbits > 14) {
        av_log(m->avctx, AV_LOG_ERROR, "Too many fractional bits: %d\n", fbits);
        s->num_primitive_matrices = 0;
        return AVERROR_INVALIDDATA;
    }
    
    s->max_matrix_channel = FFMAX(s->max_matrix_channel, out_ch);
    
    for (channel = 0; channel <= s->max_channel; channel++) {
        if (get_bits1(gbp)) {
            int coeff = get_sbits(gbp, fbits + 2);
            s->matrix_coeff[out_ch][channel] = coeff << (14 - fbits);
        }
    }
    
    if (s->noise_type) {
        s->matrix_noise_shift[out_ch] = get_bits(gbp, 4);
    } else {
        s->matrix_noise_shift[out_ch] = 0;
    }
}

// If noise_type is not set, adjust max_matrix_channel
if (!s->noise_type) {
    s->max_matrix_channel = FFMAX(s->max_matrix_channel, s->max_channel);
}

return 0;
