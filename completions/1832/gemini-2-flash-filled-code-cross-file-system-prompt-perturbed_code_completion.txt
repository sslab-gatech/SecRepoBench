if (s->num_primitive_matrices > max_primitive_matrices) {
        av_log(m->avctx, AV_LOG_ERROR, "Too many primitive matrices: %d.\n",
               s->num_primitive_matrices);
        s->num_primitive_matrices = 0;
        return AVERROR_INVALIDDATA;
    }

    for (mat = 0; mat < s->num_primitive_matrices; mat++) {
        unsigned int output_channel = get_bits(gbp, 6);
        unsigned int fractional_bits = get_bits(gbp, 4);
        s->matrix_lsb_bypass[mat] = get_bits1(gbp);

        if (output_channel > s->max_matrix_channel || fractional_bits > 14) {
            av_log(m->avctx, AV_LOG_ERROR, "Matrix parameter out of range.\n");
            s->num_primitive_matrices = 0;
            return AVERROR_INVALIDDATA;
        }

        s->matrix_output_channel[mat] = output_channel;
        s->matrix_exponent[mat]      = fractional_bits;
    }

    for (channel = 0; channel <= s->max_matrix_channel; channel++) {
        for (mat = 0; mat < s->num_primitive_matrices; mat++) {
            int coeff = get_sbits(gbp, 9);
            s->matrix_coeffs[channel][mat] = coeff << s->matrix_exponent[mat];
        }
    }

    if (s->noise_type) {
        for (mat = 0; mat < s->num_primitive_matrices; mat++)
            s->matrix_noise_shift[mat] = get_bits(gbp, 4);
    } else {
        for (mat = 0; mat < s->num_primitive_matrices; mat++)
            s->matrix_noise_shift[mat] = 0;
    }

    return 0;
