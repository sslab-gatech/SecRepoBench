// Validate the number of primitive matrices against the maximum allowed.
    if (s->num_primitive_matrices > max_primitive_matrices) {
        av_log(m->avctx, AV_LOG_ERROR, "Too many primitive matrices.\n");
        return AVERROR_INVALIDDATA;
    }

    for (mat = 0; mat < s->num_primitive_matrices; mat++) {
        unsigned int output_channel = get_bits(gbp, 4);
        unsigned int frac_bits = get_bits(gbp, 4);
        unsigned int bypass_lsbs = get_bits1(gbp);

        if (output_channel >= m->avctx->channels) {
            av_log(m->avctx, AV_LOG_ERROR, "Output channel %u exceeds max matrix channel %u.\n",
                   output_channel, m->avctx->channels - 1);
            return AVERROR_INVALIDDATA;
        }
        if (frac_bits > 14) {
            av_log(m->avctx, AV_LOG_ERROR, "Fractional bits %u exceeds maximum of 14.\n", frac_bits);
            return AVERROR_INVALIDDATA;
        }

        for (channel = 0; channel <= output_channel; channel++) {
            int32_t coeff = get_sbits(gbp, 16) * (1 << frac_bits);
            s->matrix[mat][channel] = coeff;
        }

        if (m->noise_type != NOISE_NONE) {
            s->matrix_noise_shift[mat] = get_bits(gbp, 4);
        } else {
            s->matrix_noise_shift[mat] = 0;
            s->max_matrix_channel = FFMAX(s->max_matrix_channel, output_channel);
        }
    }

    return 0;
