if (s->num_primitive_matrices > max_primitive_matrices) {
    av_log(m->avctx, AV_LOG_ERROR, "Too many primitive matrices: %d\n", s->num_primitive_matrices);
    return AVERROR_INVALIDDATA;
}

// Reset matrix parameters
memset(s->matrix, 0, sizeof(s->matrix));
memset(s->matrix_noise, 0, sizeof(s->matrix_noise));
s->max_matrix_channel = 0;

for (mat = 0; mat < s->num_primitive_matrices; mat++) {
    unsigned int outch = get_bits(gbp, 4);
    unsigned int fbits = get_bits(gbp, 4);
    unsigned int bypass_lsbs = get_bits1(gbp);
    
    // Validate output channel
    if (outch >= MAX_CHANNELS) {
        av_log(m->avctx, AV_LOG_ERROR, "Invalid matrix output channel: %d\n", outch);
        goto error;
    }
    
    // Validate fractional bits
    if (fbits > 14) {
        av_log(m->avctx, AV_LOG_ERROR, "Too many fractional bits: %d\n", fbits);
        goto error;
    }
    
    s->max_matrix_channel = FFMAX(s->max_matrix_channel, outch);
    
    // Adjust max channel based on noise type
    if (!s->noise_type) {
        unsigned int maxchan = MAX_CHANNELS - 2;
        
        if (s->max_matrix_channel >= maxchan) {
            av_log(m->avctx, AV_LOG_ERROR, "Invalid channel %d\n", s->max_matrix_channel);
            goto error;
        }
    }
    
    // Read coefficients for each channel
    for (channel = 0; channel <= s->max_matrix_channel; channel++) {
        s->matrix[mat][channel][outch] = get_sbits(gbp, 14) * (1 << fbits);
    }
    
    // Handle matrix noise
    if (s->noise_type) {
        s->matrix_noise[mat][outch] = get_bits(gbp, 5);
    } else {
        s->matrix_noise[mat][outch] = 0;
    }
    
    s->lsb_bypass[mat] = bypass_lsbs;
}

return 0;

error:
// Clean up the SubStream state
memset(s->matrix, 0, sizeof(s->matrix));
memset(s->matrix_noise, 0, sizeof(s->matrix_noise));
s->max_matrix_channel = 0;
return AVERROR_INVALIDDATA;
