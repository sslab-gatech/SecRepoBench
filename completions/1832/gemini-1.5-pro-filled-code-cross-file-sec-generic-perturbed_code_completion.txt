if (s->num_primitive_matrices > max_primitive_matrices) {
        av_log(m->avctx, AV_LOG_ERROR, "Too many primitive matrices: %d > %d.\n",
               s->num_primitive_matrices, max_primitive_matrices);
        return AVERROR_INVALIDDATA;
    }

    for (mat = 0; mat < s->num_primitive_matrices; mat++) {
        s->matrix_out_ch[mat]    = get_bits(gbp, 4);
        s->matrix_frac_bits[mat] = get_bits(gbp, 4);
        s->matrix_bypass[mat]   = get_bits1(gbp);

        if (s->matrix_out_ch[mat] >= m->max_matrix_channel) {
            av_log(m->avctx, AV_LOG_ERROR, "Invalid matrix output channel %d.\n",
                   s->matrix_out_ch[mat]);
            return AVERROR_INVALIDDATA;
        }
        if (s->matrix_frac_bits[mat] > 14) {
            av_log(m->avctx, AV_LOG_ERROR, "Invalid matrix fractional bits %d.\n",
                   s->matrix_frac_bits[mat]);
            return AVERROR_INVALIDDATA;
        }

        for (channel = 0; channel <= m->max_matrix_channel; channel++) {
            int coeff = get_sbits(gbp, 14);
            s->matrix_coeff[mat][channel] = coeff * (1 << s->matrix_frac_bits[mat]);
        }
    }

    if (m->noise_type) {
        s->noise_shift = get_bits(gbp, 4);
    } else {
        s->noise_shift = 0;
        if (m->max_matrix_channel < 7)
            m->max_matrix_channel += 2; // Implicit +2 noise channels
    }

    if (s->num_primitive_matrices == 0 && m->matrix_changed) {
        av_log(m->avctx, AV_LOG_ERROR, "No matrices, but matrix_changed is set.\n");
        return AVERROR_INVALIDDATA;
    }


    return 0;
