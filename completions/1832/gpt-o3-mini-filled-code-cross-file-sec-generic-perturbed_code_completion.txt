if (s->num_primitive_matrices > (unsigned int)max_primitive_matrices) {
    av_log(m->avctx, AV_LOG_ERROR,
           "Too many primitive matrices: %u (max %d).\n",
           s->num_primitive_matrices, max_primitive_matrices);
    memset(s->matrix_params, 0, sizeof(s->matrix_params));
    return AVERROR_INVALIDDATA;
}

/* If noise is not enabled, adjust the maximum allowed output channel
   so that noise channels (which would be added later) are not targetable. */
if (!s->noise_type) {
    if (s->max_matrix_channel < 2) {
        av_log(m->avctx, AV_LOG_ERROR,
               "Invalid max matrix channel (%u) for noise adjustment.\n",
               s->max_matrix_channel);
        memset(s->matrix_params, 0, sizeof(s->matrix_params));
        return AVERROR_INVALIDDATA;
    }
    s->max_matrix_channel -= 2;
}

for (mat = 0; mat < s->num_primitive_matrices; mat++) {
    unsigned int out_channel = get_bits(gbp, 6);
    if (out_channel > s->max_matrix_channel) {
        av_log(m->avctx, AV_LOG_ERROR,
               "Matrix %u: output channel %u exceeds maximum %u.\n",
               mat, out_channel, s->max_matrix_channel);
        memset(s->matrix_params, 0, sizeof(s->matrix_params));
        return AVERROR_INVALIDDATA;
    }

    unsigned int frac_bits = get_bits(gbp, 4);
    if (frac_bits > 14) {
        av_log(m->avctx, AV_LOG_ERROR,
               "Matrix %u: invalid fractional bits %u (must be <= 14).\n",
               mat, frac_bits);
        memset(s->matrix_params, 0, sizeof(s->matrix_params));
        return AVERROR_INVALIDDATA;
    }

    int lsb_bypass = get_bits1(gbp);
    /* Read a raw signed coefficient (assumed 8 bits) and scale it by 2^(frac_bits) */
    int coefficient = get_sbits(gbp, 8) * (1 << frac_bits);

    int noise_shift = 0;
    if (s->noise_type) {
        noise_shift = get_bits(gbp, 4);
    }

    s->matrix_params[mat].out_channel  = out_channel;
    s->matrix_params[mat].frac_bits    = frac_bits;
    s->matrix_params[mat].lsb_bypass   = lsb_bypass;
    s->matrix_params[mat].coefficient  = coefficient;
    s->matrix_params[mat].noise_shift  = noise_shift;
}
