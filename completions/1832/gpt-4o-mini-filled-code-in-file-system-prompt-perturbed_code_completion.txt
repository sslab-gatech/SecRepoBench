if (s->num_primitive_matrices > max_primitive_matrices) {
        av_log(m->avctx, AV_LOG_ERROR,
               "Number of primitive matrices %d exceeds maximum %d.\n",
               s->num_primitive_matrices, max_primitive_matrices);
        return AVERROR_INVALIDDATA;
    }

    for (mat = 0; mat < s->num_primitive_matrices; mat++) {
        s->matrix_out_ch[mat] = get_bits(gbp, 6);
        if (s->matrix_out_ch[mat] > s->max_matrix_channel) {
            av_log(m->avctx, AV_LOG_ERROR,
                   "Output channel %d for matrix %d exceeds max matrix channel %d.\n",
                   s->matrix_out_ch[mat], mat, s->max_matrix_channel);
            return AVERROR_INVALIDDATA;
        }

        s->matrix_coeff[mat][0] = get_sbits(gbp, 14);
        s->matrix_coeff[mat][1] = get_sbits(gbp, 14);
        if (s->matrix_coeff[mat][0] < -8192 || s->matrix_coeff[mat][0] > 8191 ||
            s->matrix_coeff[mat][1] < -8192 || s->matrix_coeff[mat][1] > 8191) {
            av_log(m->avctx, AV_LOG_ERROR,
                   "Matrix coefficients out of range for matrix %d.\n", mat);
            return AVERROR_INVALIDDATA;
        }

        s->lsb_bypass[mat] = get_bits1(gbp);
    }

    if (s->noise_type) {
        s->matrix_noise_shift[mat - 1] = get_bits(gbp, 4);
    } else {
        s->matrix_noise_shift[mat - 1] = 0;
    }
