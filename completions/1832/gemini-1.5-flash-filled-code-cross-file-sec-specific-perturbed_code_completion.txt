if (s->num_primitive_matrices > max_primitive_matrices) {
        av_log(m->avctx, AV_LOG_ERROR, "Too many primitive matrices.\n");
        memset(s, 0, sizeof(SubStream));
        return AVERROR_INVALIDDATA;
    }
    s->max_matrix_channel = get_bits(gbp, 5);
    if (s->max_matrix_channel >= MAX_CHANNELS) {
        av_log(m->avctx, AV_LOG_ERROR, "Too many matrix channels.\n");
        memset(s, 0, sizeof(SubStream));
        return AVERROR_INVALIDDATA;
    }
    for (mat = 0; mat < s->num_primitive_matrices; mat++) {
        channel = get_bits(gbp, 5);
        if (channel > s->max_matrix_channel) {
            av_log(m->avctx, AV_LOG_ERROR, "Output channel exceeds max matrix channel.\n");
            memset(s, 0, sizeof(SubStream));
            return AVERROR_INVALIDDATA;
        }
        s->fractional_bits[mat] = get_bits(gbp, 4);
        if (s->fractional_bits[mat] > 14) {
            av_log(m->avctx, AV_LOG_ERROR, "Fractional bits exceed 14.\n");
            memset(s, 0, sizeof(SubStream));
            return AVERROR_INVALIDDATA;
        }
        s->bypass_lsbs[mat] = get_bits1(gbp);
        if (s->noise_type == NOISE_NONE) {
            unsigned int ch;
            for (ch = 0; ch <= s->max_matrix_channel; ch++) {
                int64_t coeff = get_sbits(gbp, 16);
                s->matrix_coeff[mat][ch] = coeff * (1LL << s->fractional_bits[mat]);
            }
        } else {
            s->matrix_noise_shift[mat] = get_bits(gbp, 4);
            if (s->matrix_noise_shift[mat] > 14) {
                av_log(m->avctx, AV_LOG_ERROR, "Matrix noise shift exceeds 14.\n");
                memset(s, 0, sizeof(SubStream));
                return AVERROR_INVALIDDATA;
            }
        }
    }
    if (s->noise_type != NOISE_NONE) {
        s->max_matrix_channel += 2;
    }

    return 0;
