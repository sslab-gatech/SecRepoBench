if (s->num_primitive_matrices > max_primitive_matrices) {
        av_log(m->avctx, AV_LOG_ERROR, "Number of primitive matrices %d exceeds maximum %d.\n",
               s->num_primitive_matrices, max_primitive_matrices);
        return AVERROR_INVALIDDATA;
    }

    for (mat = 0; mat < s->num_primitive_matrices; mat++) {
        unsigned int outch = get_bits(gbp, 4);
        unsigned int coeff_bits, coeff_shift;
        int lsb_bypass = get_bits1(gbp);

        if (outch >= s->max_matrix_channel) {
            av_log(m->avctx, AV_LOG_ERROR, "Invalid channel %d specified as output from matrix.\n", outch);
            return AVERROR_INVALIDDATA;
        }

        coeff_bits = get_bits(gbp, 4);
        coeff_shift = get_bits(gbp, 4);

        if (coeff_bits + coeff_shift > 14) {
            av_log(m->avctx, AV_LOG_ERROR, "Total coefficient bits %d+%d exceeds 14.\n",
                   coeff_bits, coeff_shift);
            return AVERROR_INVALIDDATA;
        }

        s->matrix_out_ch[mat] = outch;
        s->lsb_bypass   [mat] = lsb_bypass;

        for (channel = 0; channel <= s->max_matrix_channel; channel++) {
            int coeff_val = get_sbits(gbp, coeff_bits);
            s->matrix_coeff[mat][channel] = coeff_val * (1 << coeff_shift);
        }

        if (s->noise_type) {
            s->matrix_noise_shift[mat] = get_bits(gbp, 4);
        } else {
            s->matrix_noise_shift[mat] = 0;
        }
    }

    return 0;
