if (s->num_primitive_matrices > max_primitive_matrices) {
    av_log(m->avctx, AV_LOG_ERROR, "Number of primitive matrices exceeds maximum allowed.\n");
    return AVERROR_INVALIDDATA;
}

for (mat = 0; mat < s->num_primitive_matrices; mat++) {
    channel = get_bits(gbp, 6);
    if (channel > s->max_matrix_channel || channel == 0) {
        av_log(m->avctx, AV_LOG_ERROR, "Output channel is invalid or out of range.\n");
        return AVERROR_INVALIDDATA;
    }

    int fractional_bits = get_bits(gbp, 4);
    if (fractional_bits > 14) {
        av_log(m->avctx, AV_LOG_ERROR, "Fractional bits exceed the maximum allowed value.\n");
        return AVERROR_INVALIDDATA;
    }

    int bypass_lsbs = get_bits1(gbp);

    for (int in_chan = 0; in_chan <= s->max_channel; in_chan++) {
        int32_t coeff = get_sbits(gbp, 16 - fractional_bits);
        s->matrix_coeff[mat][in_chan] = coeff << fractional_bits;
    }

    if (!m->noise_type_set) {
        s->max_matrix_channel += 2; // Adjust for additional noise channels
    }

    if (m->noise_type_set) {
        s->matrix_noise_shift[mat] = get_bits(gbp, 5);
    } else {
        s->matrix_noise_shift[mat] = 0;
    }
}

// Clear SubStream state if validation fails
if (channel > s->max_matrix_channel || fractional_bits > 14) {
    memset(s, 0, sizeof(SubStream));
    return AVERROR_INVALIDDATA;
}
