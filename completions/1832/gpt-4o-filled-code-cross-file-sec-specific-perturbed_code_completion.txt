if (s->num_primitive_matrices > max_primitive_matrices) {
        av_log(m->avctx, AV_LOG_ERROR, "Too many primitive matrices: %d.\n", s->num_primitive_matrices);
        return AVERROR_INVALIDDATA;
    }

    for (mat = 0; mat < s->num_primitive_matrices; mat++) {
        unsigned int output_channel = get_bits(gbp, 4);
        unsigned int fractional_bits = get_bits(gbp, 4);
        unsigned int lsb_bypass = get_bits1(gbp);

        if (output_channel > s->max_matrix_channel) {
            av_log(m->avctx, AV_LOG_ERROR, "Output channel %d is greater than max matrix channel %d.\n", output_channel, s->max_matrix_channel);
            return AVERROR_INVALIDDATA;
        }

        if (fractional_bits > 14) {
            av_log(m->avctx, AV_LOG_ERROR, "Fractional bits %d exceed maximum of 14.\n", fractional_bits);
            return AVERROR_INVALIDDATA;
        }

        if (!m->noise_type) {
            s->max_matrix_channel += 2;
        }

        for (channel = 0; channel <= s->max_matrix_channel; channel++) {
            int32_t coeff = get_sbits(gbp, 16) * (1 << fractional_bits);
            s->matrix_coeff[mat][channel] = coeff;
        }

        if (m->noise_type) {
            s->matrix_noise_shift[mat] = get_bits(gbp, 4);
        } else {
            s->matrix_noise_shift[mat] = 0;
        }
    }

    return 0;
