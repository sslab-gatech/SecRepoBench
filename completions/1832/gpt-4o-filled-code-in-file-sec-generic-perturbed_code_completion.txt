if (s->num_primitive_matrices > max_primitive_matrices) {
        av_log(m->avctx, AV_LOG_ERROR,
               "Number of primitive matrices %d exceeds maximum %d.\n",
               s->num_primitive_matrices, max_primitive_matrices);
        return AVERROR_INVALIDDATA;
    }

    for (mat = 0; mat < s->num_primitive_matrices; mat++) {
        unsigned int out_ch = get_bits(gbp, 4);
        unsigned int frac_bits = get_bits(gbp, 4);
        s->lsb_bypass[mat] = get_bits1(gbp);

        if (out_ch > s->max_matrix_channel) {
            av_log(m->avctx, AV_LOG_ERROR,
                   "Matrix output channel %d exceeds max matrix channel %d.\n",
                   out_ch, s->max_matrix_channel);
            return AVERROR_INVALIDDATA;
        }

        if (frac_bits > 14) {
            av_log(m->avctx, AV_LOG_ERROR,
                   "Matrix fractional bits %d exceeds maximum 14.\n",
                   frac_bits);
            return AVERROR_INVALIDDATA;
        }

        s->matrix_out_ch[mat] = out_ch;

        for (channel = 0; channel <= s->max_matrix_channel; channel++) {
            s->matrix_coeff[mat][channel] = get_sbits(gbp, 9) * (1 << frac_bits);
        }

        if (s->noise_type) {
            s->matrix_noise_shift[mat] = get_bits(gbp, 4);
        } else {
            s->matrix_noise_shift[mat] = 0;
        }
    }

    return 0;
