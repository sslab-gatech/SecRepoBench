// Validate the number of primitive matrices.
    if (s->num_primitive_matrices > max_primitive_matrices) {
        av_log(m->avctx, AV_LOG_ERROR,
               "Too many primitive matrices: %u (max %d).\n",
               s->num_primitive_matrices, max_primitive_matrices);
        goto error;
    }

    // If noise type is not set, adjust max channel to account for extra noise channels.
    if (!s->noise_type)
        s->max_matrix_channel += 2;

    // Read each primitive matrix.
    for (mat = 0; mat < s->num_primitive_matrices; mat++) {
        unsigned int out_ch      = get_bits(gbp, 5);
        unsigned int frac_bits   = get_bits(gbp, 4);
        unsigned int lsb_bypass  = get_bits1(gbp);
        unsigned int c;

        if (out_ch > s->max_matrix_channel) {
            av_log(m->avctx, AV_LOG_ERROR,
                   "Output channel out of range (%u > %u).\n",
                   out_ch, s->max_matrix_channel);
            goto error;
        }
        if (frac_bits > 14) {
            av_log(m->avctx, AV_LOG_ERROR,
                   "Fractional bits out of range (%u > 14).\n", frac_bits);
            goto error;
        }

        // Store and read the matrix coefficients.
        s->matrix_params[mat].out_ch     = out_ch;
        s->matrix_params[mat].frac_bits  = frac_bits;
        s->matrix_params[mat].lsb_bypass = lsb_bypass;

        for (channel = 0; channel <= out_ch; channel++) {
            int coeff = get_sbits(gbp, 20); // Example bit width for coefficient
            s->matrix_params[mat].coeff[channel] = coeff << frac_bits;
        }
    }

    // If noise type is set, read and store noise shift; otherwise set to zero.
    if (s->noise_type) {
        s->noise_shift = get_bits(gbp, 4);
    } else {
        s->noise_shift = 0;
    }

    return 0;

error:
    av_log(m->avctx, AV_LOG_ERROR, "Invalid matrix parameters.\n");
    s->num_primitive_matrices = 0;
    return AVERROR_INVALIDDATA;
