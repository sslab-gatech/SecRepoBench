if (s->num_primitive_matrices > max_primitive_matrices) {
        av_log(m->avctx, AV_LOG_ERROR, "Too many primitive matrices: %d > %d\n",
               s->num_primitive_matrices, max_primitive_matrices);
        s->num_primitive_matrices = 0;
        return AVERROR_INVALIDDATA;
    }

    for (mat = 0; mat < s->num_primitive_matrices; mat++) {
        s->matrix_out_ch[mat] = get_bits(gbp, 4);
        s->matrix_coeff_precision[mat] = get_bits(gbp, 4);
        s->matrix_out_shift[mat] = get_bits1(gbp);

        if (s->matrix_out_ch[mat] > s->max_matrix_channel) {
            av_log(m->avctx, AV_LOG_ERROR, "Matrix output channel %d > max %d\n",
                   s->matrix_out_ch[mat], s->max_matrix_channel);
            s->num_primitive_matrices = 0;
            return AVERROR_INVALIDDATA;
        }
        if (s->matrix_coeff_precision[mat] > 14) {
            av_log(m->avctx, AV_LOG_ERROR, "Matrix fractional bits too high: %d\n",
                   s->matrix_coeff_precision[mat]);
            s->num_primitive_matrices = 0;
            return AVERROR_INVALIDDATA;
        }
    }

    // If noise type is set, allow for extra noise channels
    unsigned int maxchan = s->max_matrix_channel;
    if (s->noise_type)
        maxchan += 2;

    for (mat = 0; mat < s->num_primitive_matrices; mat++) {
        for (channel = 0; channel <= maxchan; channel++) {
            int coeff = get_sbits(gbp, 9);
            s->matrix_coeff[mat][channel] = coeff * (1 << s->matrix_coeff_precision[mat]);
        }
    }

    for (mat = 0; mat < s->num_primitive_matrices; mat++) {
        if (s->noise_type)
            s->matrix_noise_shift[mat] = get_bits(gbp, 4);
        else
            s->matrix_noise_shift[mat] = 0;
    }

    return 0;
