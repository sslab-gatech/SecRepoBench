if (s->num_primitive_matrices > max_primitive_matrices) {
        av_log(m->avctx, AV_LOG_ERROR,
               "Too many primitive matrices: %d > %d.\n",
               s->num_primitive_matrices, max_primitive_matrices);

        s->num_primitive_matrices = 0; // Clean the SubStream state

        return AVERROR_INVALIDDATA;
    }


    for (mat = 0; mat < s->num_primitive_matrices; mat++) {
        s->matrix_out_ch[mat] = get_bits(gbp, 4);
        if (s->matrix_out_ch[mat] > s->max_matrix_channel) {
            av_log(m->avctx, AV_LOG_ERROR,
                   "Matrix output channel %d can't be greater than %d.\n",
                   s->matrix_out_ch[mat], s->max_matrix_channel);

            s->num_primitive_matrices = 0; // Clean the SubStream state

            return AVERROR_INVALIDDATA;
        }

        s->lsb_bypass[mat] = get_bits1(gbp);

        if (get_bits(gbp, 2) != 0) { // fractional_bits must be 0
            av_log(m->avctx, AV_LOG_ERROR, "fractional_bits must be 0.\n");
            s->num_primitive_matrices = 0; // Clean the SubStream state
            return AVERROR_INVALIDDATA;
        }

        if (!s->noise_type)
            s->max_channel = s->max_matrix_channel + 2;
    }

    for (channel = 0; channel <= s->max_channel; channel++) {
        for (mat = 0; mat < s->num_primitive_matrices; mat++) {
            s->matrix_coeff[mat][channel] = get_sbits(gbp, 14);
        }
    }

    if (s->noise_type) {
        for (mat = 0; mat < s->num_primitive_matrices; mat++) {
            s->matrix_noise_shift[mat] = get_bits(gbp, 4);
        }
    } else {
        memset(s->matrix_noise_shift, 0, sizeof(s->matrix_noise_shift));
    }
