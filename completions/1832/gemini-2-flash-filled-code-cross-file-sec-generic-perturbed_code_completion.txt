if (s->num_primitive_matrices > max_primitive_matrices) {
        av_log(m->avctx, AV_LOG_ERROR, "Too many primitive matrices.\n");
        s->num_primitive_matrices = 0;
        return AVERROR_INVALIDDATA;
    }

    for (mat = 0; mat < s->num_primitive_matrices; mat++) {
        s->matrix_out_channel[mat] = get_bits(gbp, 5);
        s->matrix_fractional_bits[mat] = get_bits(gbp, 4);
        s->matrix_lsb_bypass[mat] = get_bits1(gbp);

        if (s->matrix_out_channel[mat] > m->max_matrix_channel ||
            s->matrix_fractional_bits[mat] > 14) {
            av_log(m->avctx, AV_LOG_ERROR, "Invalid matrix parameters.\n");
            s->num_primitive_matrices = 0;
            return AVERROR_INVALIDDATA;
        }
    }

    if (m->noise_type == NOISE_NONE) {
        if (m->max_matrix_channel + 3 > m->avctx->channels)
            m->max_matrix_channel = m->avctx->channels - 3;
    }

    for (channel = 0; channel <= m->max_matrix_channel; channel++) {
        for (mat = 0; mat < s->num_primitive_matrices; mat++) {
            int coeff = get_sbits(gbp, 12);
            s->matrix_coeffs[channel][mat] = coeff * (1 << s->matrix_fractional_bits[mat]);
        }
    }

    if (m->noise_type != NOISE_NONE) {
        s->matrix_noise_shift = get_bits(gbp, 4);
    } else {
        s->matrix_noise_shift = 0;
    }

    return 0;
