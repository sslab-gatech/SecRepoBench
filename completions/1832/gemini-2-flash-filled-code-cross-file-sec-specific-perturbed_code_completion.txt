{
    if (s->num_primitive_matrices > max_primitive_matrices) {
        av_log(m->avctx, AV_LOG_ERROR, "Too many primitive matrices.\n");
        s->num_primitive_matrices = 0;
        return AVERROR_INVALIDDATA;
    }

    for (mat = 0; mat < s->num_primitive_matrices; mat++) {
        unsigned int output_channel = get_bits(gbp, 6);
        unsigned int frac_bits      = get_bits(gbp, 4);
        s->matrix_bypass_lsbs[mat]  = get_bits1(gbp);

        if (output_channel > s->max_matrix_channel || frac_bits > 14) {
            av_log(m->avctx, AV_LOG_ERROR, "Matrix parameter out of range.\n");
            s->num_primitive_matrices = 0;
            return AVERROR_INVALIDDATA;
        }

        s->matrix_out_channel[mat] = output_channel;
        s->matrix_frac_bits[mat]  = frac_bits;
    }

    for (channel = 0; channel <= s->max_matrix_channel; channel++) {
        for (mat = 0; mat < s->num_primitive_matrices; mat++) {
            int coeff = get_sbits(gbp, 12);
            s->matrix_coeffs[channel][mat] = coeff * (1 << s->matrix_frac_bits[mat]);
        }
    }

    if (s->noise_type) {
        for (mat = 0; mat < s->num_primitive_matrices; mat++)
            s->matrix_noise_shift[mat] = get_bits(gbp, 4);
    } else {
        for (mat = 0; mat < s->num_primitive_matrices; mat++)
            s->matrix_noise_shift[mat] = 0;
    }

    return 0;
}
