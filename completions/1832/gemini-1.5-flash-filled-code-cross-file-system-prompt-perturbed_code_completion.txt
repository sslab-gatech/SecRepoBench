if (s->num_primitive_matrices > max_primitive_matrices) {
        av_log(m->avctx, AV_LOG_ERROR, "Too many primitive matrices.\n");
        memset(s, 0, sizeof(*s));
        return AVERROR_INVALIDDATA;
    }
    for (mat = 0; mat < s->num_primitive_matrices; mat++) {
        channel = get_bits(gbp, 5);
        if (channel > m->max_matrix_channel) {
            av_log(m->avctx, AV_LOG_ERROR, "Output channel exceeds maximum.\n");
            memset(s, 0, sizeof(*s));
            return AVERROR_INVALIDDATA;
        }
        s->primitive_matrix[mat].channel = channel;
        s->primitive_matrix[mat].frac_bits = get_bits(gbp, 4);
        if (s->primitive_matrix[mat].frac_bits > 14) {
            av_log(m->avctx, AV_LOG_ERROR, "Fractional bits exceed maximum.\n");
            memset(s, 0, sizeof(*s));
            return AVERROR_INVALIDDATA;
        }
        s->primitive_matrix[mat].bypass_lsbs = get_bits1(gbp);
        for (unsigned int ch = 0; ch <= channel; ch++) {
            int coeff = get_sbits(gbp, 16);
            s->primitive_matrix[mat].coeff[ch] = coeff * (1 << s->primitive_matrix[mat].frac_bits);
        }
    }
    if (m->noise_type) {
        s->noise_shift = get_bits(gbp, 5);
    } else {
        s->noise_shift = 0;
    }
    s->max_matrix_channel = channel;
    return 0;
