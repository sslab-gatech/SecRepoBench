if (s->num_primitive_matrices > max_primitive_matrices) {
        av_log(m->avctx, AV_LOG_ERROR, "Number of primitive matrices %d exceeds maximum %d.\n",
               s->num_primitive_matrices, max_primitive_matrices);
        return AVERROR_INVALIDDATA;
    }

    s->primitive_matrix = av_malloc_array(s->num_primitive_matrices, sizeof(*s->primitive_matrix));
    if (!s->primitive_matrix) {
        av_log(m->avctx, AV_LOG_ERROR, "Failed to allocate memory for primitive matrices.\n");
        return AVERROR(ENOMEM);
    }

    for (mat = 0; mat < s->num_primitive_matrices; mat++) {
        s->primitive_matrix[mat].output_channel = get_bits(gbp, 4);
        if (s->primitive_matrix[mat].output_channel >= m->avctx->channels) {
            av_log(m->avctx, AV_LOG_ERROR, "Output channel %d exceeds maximum channel count %d.\n",
                   s->primitive_matrix[mat].output_channel, m->avctx->channels);
            return AVERROR_INVALIDDATA;
        }

        s->primitive_matrix[mat].fractional_bits = get_bits(gbp, 4);
        if (s->primitive_matrix[mat].fractional_bits > 14) {
            av_log(m->avctx, AV_LOG_ERROR, "Fractional bits %d exceed maximum 14.\n",
                   s->primitive_matrix[mat].fractional_bits);
            return AVERROR_INVALIDDATA;
        }

        s->primitive_matrix[mat].lsb_bypass = get_bits1(gbp);

        for (channel = 0; channel < s->primitive_matrix[mat].output_channel; channel++) {
            int32_t coeff_value;
            coeff_value = get_sbits(gbp, s->primitive_matrix[mat].fractional_bits + 5);
            s->primitive_matrix[mat].coeff[channel] = coeff_value << s->primitive_matrix[mat].fractional_bits;
        }

        if (get_bits1(gbp)) {
            s->primitive_matrix[mat].matrix_noise_shift = get_bits(gbp, 4);
        } else {
            s->primitive_matrix[mat].matrix_noise_shift = 0;
        }
    }

    for (channel = 0; channel < m->avctx->channels; channel++) {
        if (s->param_presence_flags & PARAM_CHANNEL) {
            if ((ret = read_channel_params(m, substr, gbp, channel)) < 0) {
                return ret;
            }
        }
    }

    if (s->param_presence_flags & PARAM_NOISE) {
        s->noise_type = get_bits(gbp, 2);
        if (s->noise_type >= 2) {
            av_log(m->avctx, AV_LOG_ERROR, "Invalid noise type %d.\n", s->noise_type);
            return AVERROR_INVALIDDATA;
        }

        if (s->noise_type) {
            s->noise_shift = get_bits(gbp, 4);
        } else {
            s->noise_shift = 0;
        }
    } else {
        s->noise_type = 0;
        s->noise_shift = 0;
    }

    if (s->param_presence_flags & PARAM_MATRIX_NOISE) {
        if (get_bits1(gbp)) {
            s->matrix_noise_shift = get_bits(gbp, 4);
        } else {
            s->matrix_noise_shift = 0;
        }
    } else {
        s->matrix_noise_shift = 0;
    }

    if (s->param_presence_flags & PARAM_MATRIX_NOISE) {
        if (s->matrix_noise_shift && s->noise_type) {
            av_log(m->avctx, AV_LOG_ERROR, "Both matrix noise and channel noise are set.\n");
            return AVERROR_INVALIDDATA;
        }
    }

    if (s->param_presence_flags & PARAM_MATRIX_NOISE) {
        if (s->matrix_noise_shift && !s->noise_type) {
            s->noise_type = 1;
        }
    }

    if (s->param_presence_flags & PARAM_MATRIX_NOISE) {
        if (!s->matrix_noise_shift && s->noise_type) {
            av_log(m->avctx, AV_LOG_ERROR, "Matrix noise shift is not set while noise type is set.\n");
            return AVERROR_INVALIDDATA;
        }
    }

    if (s->param_presence_flags & PARAM_MATRIX_NOISE) {
        if (s->matrix_noise_shift && s->noise_type) {
            generate_2_noise_channels(m, substr);
        }
    }

    return 0;
}
