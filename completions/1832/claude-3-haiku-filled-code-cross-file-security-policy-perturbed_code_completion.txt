// Validate the number of primitive matrices against the maximum allowed.
if (s->num_primitive_matrices > max_primitive_matrices) {
    av_log(m->avctx, AV_LOG_ERROR, "Number of primitive matrices %d exceeds maximum %d.\n",
           s->num_primitive_matrices, max_primitive_matrices);
    s->num_primitive_matrices = 0;
    return AVERROR_INVALIDDATA;
}

for (mat = 0; mat < s->num_primitive_matrices; mat++) {
    channel = get_bits(gbp, 5);
    if (channel >= MAX_MATRIX_CHANNELS) {
        av_log(m->avctx, AV_LOG_ERROR, "Output channel %d exceeds maximum %d.\n",
               channel, MAX_MATRIX_CHANNELS - 1);
        s->num_primitive_matrices = 0;
        return AVERROR_INVALIDDATA;
    }

    s->matrix_coeff_precision[mat] = get_bits(gbp, 4);
    if (s->matrix_coeff_precision[mat] > 14) {
        av_log(m->avctx, AV_LOG_ERROR, "Matrix coefficient precision %d exceeds maximum 14.\n",
               s->matrix_coeff_precision[mat]);
        s->num_primitive_matrices = 0;
        return AVERROR_INVALIDDATA;
    }

    s->matrix_noise_type[mat] = get_bits1(gbp);
    if (!s->matrix_noise_type[mat]) {
        s->max_matrix_channel = FFMAX(s->max_matrix_channel, channel);
    }

    for (channel = 0; channel <= s->max_matrix_channel; channel++) {
        s->matrix_coeff[mat][channel] = get_sbits(gbp, s->matrix_coeff_precision[mat]) *
                                        (1 << (14 - s->matrix_coeff_precision[mat]));
    }

    if (s->matrix_noise_type[mat]) {
        s->matrix_noise_shift[mat] = get_bits(gbp, 4);
    } else {
        s->matrix_noise_shift[mat] = 0;
    }
}
return 0;
