// Validate the number of primitive matrices.
    if (s->num_primitive_matrices > max_primitive_matrices) {
        av_log(m->avctx, AV_LOG_ERROR, "Too many primitive matrices.\n");
        // Clean state before returning.
        s->num_primitive_matrices = 0;
        return AVERROR_INVALIDDATA;
    }

    // Determine maximum channel during matrix parsing.
    // If the noise type is not set, adjust for additional noise channels.
    {
        unsigned int max_ch = s->max_matrix_channel;
        if (!m->noise_type)
            max_ch += 2;

        for (mat = 0; mat < s->num_primitive_matrices; mat++) {
            unsigned int out_ch      = get_bits(gbp, 5);
            unsigned int frac_bits   = get_bits(gbp, 4);
            unsigned int bypass_lsb  = get_bits1(gbp);

            // Validate output channel and fractional bits.
            if (out_ch > max_ch) {
                av_log(m->avctx, AV_LOG_ERROR,
                       "Matrix output channel (%u) out of range (max %u).\n",
                       out_ch, max_ch);
                s->num_primitive_matrices = 0;
                return AVERROR_INVALIDDATA;
            }
            if (frac_bits > 14) {
                av_log(m->avctx, AV_LOG_ERROR, "Fractional bits (%u) out of range.\n",
                       frac_bits);
                s->num_primitive_matrices = 0;
                return AVERROR_INVALIDDATA;
            }

            // For each channel up to out_ch, read a coefficient, then shift.
            for (channel = 0; channel <= out_ch; channel++) {
                if (channel >= MAX_CHANNELS) {
                    av_log(m->avctx, AV_LOG_ERROR,
                           "Matrix channel index (%u) exceeds compile-time limit.\n",
                           channel);
                    s->num_primitive_matrices = 0;
                    return AVERROR_INVALIDDATA;
                }
                {
                    int coeff = get_sbits(gbp, 8);
                    s->matrix_coeff[mat][channel] = coeff << frac_bits;
                }
            }

            // Store parsed data for this matrix.
            s->matrix_out_ch[mat]     = out_ch;
            s->matrix_fbits[mat]      = frac_bits;
            s->matrix_bypass[mat]     = bypass_lsb;
        }
    }

    // If noise is set, read noise shift. Otherwise, clear it.
    if (m->noise_type) {
        s->noise_shift = get_bits(gbp, 4);
        if (s->noise_shift > 14) {
            av_log(m->avctx, AV_LOG_ERROR, "Noise shift (%u) out of range.\n",
                   s->noise_shift);
            s->num_primitive_matrices = 0;
            return AVERROR_INVALIDDATA;
        }
    } else {
        s->noise_shift = 0;
    }

    return 0;
