if (s->num_primitive_matrices > max_primitive_matrices) {
        av_log(m->avctx, AV_LOG_ERROR, "Number of primitive matrices exceeds maximum allowed.\n");
        return AVERROR_INVALIDDATA;
    }

    for (mat = 0; mat < s->num_primitive_matrices; mat++) {
        channel = get_bits(gbp, 4);
        if (channel >= max_primitive_matrices) {
            av_log(m->avctx, AV_LOG_ERROR, "Output channel %d exceeds maximum %d.\n", channel, max_primitive_matrices);
            return AVERROR_INVALIDDATA;
        }

        int fractional_bits = get_bits(gbp, 4);
        if (fractional_bits > 14) {
            av_log(m->avctx, AV_LOG_ERROR, "Fractional bits %d exceed maximum of 14.\n", fractional_bits);
            return AVERROR_INVALIDDATA;
        }

        // Adjust max channel if noise type is not set
        if (!(s->param_presence_flags & PARAM_NOISE_TYPE)) {
            s->max_matrix_channel = channel + 2; // Assuming 2 additional noise channels
        }

        for (unsigned int ch = 0; ch < s->max_matrix_channel; ch++) {
            int32_t coefficient = get_sbits(gbp, 16) * (1 << fractional_bits);
            s->channel_params[ch].coeff[mat] = coefficient; // Store the coefficient
        }

        // Read noise shift value if noise type is set
        if (s->param_presence_flags & PARAM_NOISE_TYPE) {
            s->noise_shift = get_bits(gbp, 4);
        } else {
            s->noise_shift = 0;
        }
    }

    return 0;
