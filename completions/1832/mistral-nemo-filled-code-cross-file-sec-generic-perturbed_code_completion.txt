for (mat = 0; mat < s->num_primitive_matrices; mat++) {
        channel = get_bits(gbp, 4);
        if (channel > max_primitive_matrices) {
            av_log(m->avctx, AV_LOG_ERROR, "Output channel %d exceeds max matrix channel %d.\n", channel, max_primitive_matrices);
            return AVERROR_INVALIDDATA;
        }

        s->primitive_matrix[mat].fractional_bits = get_bits(gbp, 4);
        if (s->primitive_matrix[mat].fractional_bits > 14) {
            av_log(m->avctx, AV_LOG_ERROR, "Fractional bits %d exceed maximum 14.\n", s->primitive_matrix[mat].fractional_bits);
            return AVERROR_INVALIDDATA;
        }

        s->primitive_matrix[mat].bypass_lsbs = get_bits1(gbp);

        if (m->noise_type == NOISE_TYPE_MLP) {
            max_primitive_matrices += 2;
        }

        for (int ch = 0; ch < max_primitive_matrices; ch++) {
            s->primitive_matrix[mat].coeff[ch] = get_sbits(gbp, 16) * (1 << s->primitive_matrix[mat].fractional_bits);
        }

        if (m->noise_type == NOISE_TYPE_MLP) {
            s->primitive_matrix[mat].noise_shift = get_bits(gbp, 4);
        } else {
            s->primitive_matrix[mat].noise_shift = 0;
        }
    }

    if (s->num_primitive_matrices > max_primitive_matrices) {
        av_log(m->avctx, AV_LOG_ERROR, "Number of primitive matrices exceeds max allowed.\n");
        return AVERROR_INVALIDDATA;
    }

    return 0;
