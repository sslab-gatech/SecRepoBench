if (s->num_primitive_matrices > max_primitive_matrices) {
        av_log(m->avctx, AV_LOG_ERROR, "Too many primitive matrices: %d > %d.\n",
               s->num_primitive_matrices, max_primitive_matrices);
        s->num_primitive_matrices = 0; // Sanitize the invalid value
        return AVERROR_INVALIDDATA;
    }

    for (mat = 0; mat < s->num_primitive_matrices; mat++) {
        s->matrix[mat].out_ch    = get_bits(gbp, 4);
        s->matrix[mat].frac_bits = get_bits(gbp, 4);
        s->matrix[mat].lsb_bypass = get_bits1(gbp);

        if (s->matrix[mat].out_ch >= s->max_matrix_channel) {
            av_log(m->avctx, AV_LOG_ERROR, "Invalid output channel.\n");
            memset(&s->matrix[0], 0, sizeof(s->matrix)); // Clean up matrix data
            return AVERROR_INVALIDDATA;
        }
        if (s->matrix[mat].frac_bits > 14) {
            av_log(m->avctx, AV_LOG_ERROR, "Invalid fractional bits.\n");
            memset(&s->matrix[0], 0, sizeof(s->matrix)); // Clean up matrix data
            return AVERROR_INVALIDDATA;
        }

        for (channel = 0; channel < s->num_output_channels; channel++) {
            int32_t coeff = get_sbits(gbp, 16);
            s->matrix_coeff[mat][channel] = coeff * (1 << s->matrix[mat].frac_bits);
        }
    }

    if (s->noise_type) {
        s->noise_shift = get_bits(gbp, 4);
    } else {
        s->noise_shift = 0;
        if (s->max_matrix_channel < s->num_output_channels + 2)
            s->max_matrix_channel = s->num_output_channels + 2;
    }
