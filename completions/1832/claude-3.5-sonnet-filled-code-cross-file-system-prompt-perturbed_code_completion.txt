if (s->num_primitive_matrices > max_primitive_matrices) {
    av_log(m->avctx, AV_LOG_ERROR, "Too many primitive matrices.\n");
    return AVERROR_INVALIDDATA;
}

for (mat = 0; mat < s->num_primitive_matrices; mat++) {
    s->matrix_out_ch[mat] = get_bits(gbp, 4);
    s->matrix_frac_bits[mat] = get_bits(gbp, 4);
    s->lsb_bypass[mat] = get_bits1(gbp);

    if (s->matrix_out_ch[mat] > s->max_matrix_channel) {
        av_log(m->avctx, AV_LOG_ERROR, "Invalid matrix output channel.\n");
        goto error;
    }

    if (s->matrix_frac_bits[mat] > 14) {
        av_log(m->avctx, AV_LOG_ERROR, "Invalid matrix fractional bits.\n");
        goto error;
    }
}

if (!s->noise_type)
    s->max_matrix_channel += 2;

for (mat = 0; mat < s->num_primitive_matrices; mat++) {
    for (channel = 0; channel <= s->max_matrix_channel; channel++) {
        int coeff = get_sbits(gbp, 16);
        s->matrix_coeff[mat][channel] = coeff * (1 << s->matrix_frac_bits[mat]);
    }
}

if (s->noise_type) {
    s->matrix_noise_shift = get_bits(gbp, 4);
} else {
    s->matrix_noise_shift = 0;
}

return 0;

error:
    memset(s->matrix_out_ch, 0, sizeof(s->matrix_out_ch));
    memset(s->matrix_frac_bits, 0, sizeof(s->matrix_frac_bits));
    memset(s->lsb_bypass, 0, sizeof(s->lsb_bypass));
    memset(s->matrix_coeff, 0, sizeof(s->matrix_coeff));
    s->matrix_noise_shift = 0;
    return AVERROR_INVALIDDATA;
