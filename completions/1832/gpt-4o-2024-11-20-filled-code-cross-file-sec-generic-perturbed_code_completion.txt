if (s->num_primitive_matrices > max_primitive_matrices) {
        av_log(m->avctx, AV_LOG_ERROR, "Too many primitive matrices: %d.\n", s->num_primitive_matrices);
        return AVERROR_INVALIDDATA;
    }

    for (mat = 0; mat < s->num_primitive_matrices; mat++) {
        unsigned int fractional_bits, bypass_lsbs;

        channel = get_bits(gbp, 4);
        fractional_bits = get_bits(gbp, 4);
        bypass_lsbs = get_bits1(gbp);

        if (channel > s->max_matrix_channel) {
            av_log(m->avctx, AV_LOG_ERROR, "Matrix channel %d exceeds max allowed channel %d.\n", channel, s->max_matrix_channel);
            return AVERROR_INVALIDDATA;
        }

        if (fractional_bits > 14) {
            av_log(m->avctx, AV_LOG_ERROR, "Fractional bits %d exceed maximum allowed value of 14.\n", fractional_bits);
            return AVERROR_INVALIDDATA;
        }

        s->matrix_out_ch[mat] = channel;
        s->matrix_out_fb[mat] = fractional_bits;
        s->matrix_out_lsbs[mat] = bypass_lsbs;
    }

    if (!s->noise_type) {
        s->max_matrix_channel += s->noise_channels;
    }

    for (channel = 0; channel <= s->max_matrix_channel; channel++) {
        for (mat = 0; mat < s->num_primitive_matrices; mat++) {
            int32_t coeff = get_sbits(gbp, 16);
            s->matrix_coeff[mat][channel] = coeff * (1 << s->matrix_out_fb[mat]);
        }
    }

    if (s->noise_type) {
        s->noise_shift = get_bits(gbp, 4);
    } else {
        s->noise_shift = 0;
    }

    if (get_bits_left(gbp) < 0) {
        av_log(m->avctx, AV_LOG_ERROR, "Insufficient bits left in bitstream.\n");
        memset(s, 0, sizeof(*s));
        return AVERROR_INVALIDDATA;
    }

    return 0;
