if (s->num_primitive_matrices > max_primitive_matrices) {
        av_log(m->avctx, AV_LOG_ERROR, "Too many primitive matrices: %d.\n", s->num_primitive_matrices);
        return AVERROR_INVALIDDATA;
    }

    for (mat = 0; mat < s->num_primitive_matrices; mat++) {
        unsigned int fractional_bits, bypass_lsbs;

        s->matrix_out_ch[mat] = get_bits(gbp, 4);
        fractional_bits = get_bits(gbp, 4);
        bypass_lsbs = get_bits1(gbp);

        if (s->matrix_out_ch[mat] > s->max_matrix_channel) {
            av_log(m->avctx, AV_LOG_ERROR, "Matrix output channel %d is invalid.\n", s->matrix_out_ch[mat]);
            return AVERROR_INVALIDDATA;
        }

        if (fractional_bits > 14) {
            av_log(m->avctx, AV_LOG_ERROR, "Fractional bits %d exceed maximum allowed.\n", fractional_bits);
            return AVERROR_INVALIDDATA;
        }

        s->matrix_coeff[mat] = av_malloc_array(s->max_matrix_channel + 1, sizeof(int32_t));
        if (!s->matrix_coeff[mat])
            return AVERROR(ENOMEM);

        for (channel = 0; channel <= s->max_matrix_channel; channel++) {
            s->matrix_coeff[mat][channel] = get_sbits(gbp, 16) * (1 << fractional_bits);
        }

        if (s->noise_type) {
            s->matrix_noise_shift[mat] = get_bits(gbp, 4);
        } else {
            s->matrix_noise_shift[mat] = 0;
        }
    }

    if (s->noise_type) {
        s->max_matrix_channel += 2;
    }

    return 0;
