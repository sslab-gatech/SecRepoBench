if (s->num_primitive_matrices > max_primitive_matrices) {
        av_log(m->avctx, AV_LOG_ERROR, "Too many primitive matrices: %u.\n", s->num_primitive_matrices);
        return AVERROR_INVALIDDATA;
    }

    for (mat = 0; mat < s->num_primitive_matrices; mat++) {
        unsigned int fractional_bits, noise_shift;
        int bypass_lsbs;

        channel = get_bits(gbp, 4);
        if (channel > s->max_matrix_channel) {
            av_log(m->avctx, AV_LOG_ERROR, "Invalid output channel: %u.\n", channel);
            return AVERROR_INVALIDDATA;
        }

        fractional_bits = get_bits(gbp, 4);
        if (fractional_bits > 14) {
            av_log(m->avctx, AV_LOG_ERROR, "Fractional bits too large: %u.\n", fractional_bits);
            return AVERROR_INVALIDDATA;
        }

        bypass_lsbs = get_bits1(gbp);

        if (s->noise_type) {
            noise_shift = get_bits(gbp, 4);
            if (noise_shift > 14) {
                av_log(m->avctx, AV_LOG_ERROR, "Noise shift too large: %u.\n", noise_shift);
                return AVERROR_INVALIDDATA;
            }
            s->matrix_noise_shift[mat] = noise_shift;
        } else {
            s->matrix_noise_shift[mat] = 0;
        }

        for (unsigned int ch = 0; ch <= s->max_matrix_channel; ch++) {
            int32_t coeff = get_sbits(gbp, 16);
            s->matrix_coeff[mat][ch] = coeff * (1 << fractional_bits);
        }
    }

    return 0;
