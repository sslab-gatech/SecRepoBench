if (s->num_primitive_matrices > max_primitive_matrices) {
        av_log(m->avctx, AV_LOG_ERROR,
               "Number of primitive matrices %d exceeds maximum %d.\n",
               s->num_primitive_matrices, max_primitive_matrices);
        memset(s, 0, sizeof(SubStream));
        return AVERROR_INVALIDDATA;
    }

    for (mat = 0; mat < s->num_primitive_matrices; mat++) {
        int output_ch, fractional_bits;

        output_ch       = get_bits(gbp, 4);
        fractional_bits = get_bits(gbp, 4);
        s->lsb_bypass[mat] = get_bits1(gbp);
        s->matrix_out_ch[mat] = output_ch;

        if (output_ch > s->max_matrix_channel) {
            av_log(m->avctx, AV_LOG_ERROR,
                   "Matrix output channel %d exceeds max matrix channel %d.\n",
                   output_ch, s->max_matrix_channel);
            memset(s, 0, sizeof(SubStream));
            return AVERROR_INVALIDDATA;
        }
        if (fractional_bits > 14) {
            av_log(m->avctx, AV_LOG_ERROR,
                   "Fractional bits %d exceeds maximum 14.\n",
                   fractional_bits);
            memset(s, 0, sizeof(SubStream));
            return AVERROR_INVALIDDATA;
        }

        for (channel = 0; channel <= s->max_channel; channel++) {
            s->matrix_coeff[mat][channel] = get_sbits(gbp, fractional_bits + 2) * (1 << (14 - fractional_bits));
        }
        if (!s->noise_type)
            s->matrix_noise_shift[mat] = 0;
        else
            s->matrix_noise_shift[mat] = get_bits(gbp, 4);
    }
