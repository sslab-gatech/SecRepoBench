if (!s->noise_type)
    s->max_matrix_channel += 2;

for (mat = 0; mat < s->num_primitive_matrices; mat++) {
    unsigned int out_channel = get_bits(gbp, 5);
    unsigned int frac_bits   = get_bits(gbp, 4);
    unsigned int lsb_bypass  = get_bits1(gbp);

    if (out_channel > s->max_matrix_channel) {
        av_log(m->avctx, AV_LOG_ERROR,
               "Matrix output channel %u exceeds maximum %u.\n",
               out_channel, s->max_matrix_channel);
        memset(s->primitive_matrix, 0, sizeof(s->primitive_matrix));
        return AVERROR_INVALIDDATA;
    }
    if (frac_bits > 14) {
        av_log(m->avctx, AV_LOG_ERROR,
               "Matrix fractional bits %u exceeds maximum 14.\n",
               frac_bits);
        memset(s->primitive_matrix, 0, sizeof(s->primitive_matrix));
        return AVERROR_INVALIDDATA;
    }

    s->primitive_matrix[mat].output_channel = out_channel;
    s->primitive_matrix[mat].frac_bits      = frac_bits;
    s->primitive_matrix[mat].lsb_bypass     = lsb_bypass;

    /* For each channel, read a coefficient value,
       multiplying it by 2^(frac_bits) */
    for (channel = 0; channel < s->max_matrix_channel; channel++) {
        int coeff = get_sbits(gbp, 16);
        s->primitive_matrix[mat].coeff[channel] = coeff * (1 << frac_bits);
    }

    /* Read the matrix noise shift if noise type is set,
       otherwise default it to 0 */
    if (s->noise_type)
        s->primitive_matrix[mat].noise_shift = get_bits(gbp, 4);
    else
        s->primitive_matrix[mat].noise_shift = 0;
}
