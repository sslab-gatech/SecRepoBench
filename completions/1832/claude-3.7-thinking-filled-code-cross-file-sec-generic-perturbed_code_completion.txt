if (s->num_primitive_matrices > max_primitive_matrices) {
    av_log(m->avctx, AV_LOG_ERROR, "Too many primitive matrices: %d (max is %d).\n",
           s->num_primitive_matrices, max_primitive_matrices);
    return AVERROR_INVALIDDATA;
}

/* Clear existing matrix data */
memset(s->matrix_out_ch, 0, sizeof(s->matrix_out_ch));
memset(s->matrix_frac_bits, 0, sizeof(s->matrix_frac_bits));
memset(s->matrix_bypass_lsbs, 0, sizeof(s->matrix_bypass_lsbs));
memset(s->matrix_coeff, 0, sizeof(s->matrix_coeff));

for (mat = 0; mat < s->num_primitive_matrices; mat++) {
    s->matrix_out_ch[mat] = get_bits(gbp, 4);
    s->matrix_frac_bits[mat] = get_bits(gbp, 4);
    s->matrix_bypass_lsbs[mat] = get_bits1(gbp);

    if (s->matrix_out_ch[mat] > s->max_matrix_channel) {
        av_log(m->avctx, AV_LOG_ERROR, "Invalid matrix output channel: %d.\n", 
               s->matrix_out_ch[mat]);
        goto error;
    }

    if (s->matrix_frac_bits[mat] > 14) {
        av_log(m->avctx, AV_LOG_ERROR, "Invalid matrix fractional bits: %d.\n", 
               s->matrix_frac_bits[mat]);
        goto error;
    }
}

/* Determine the maximum channel index */
unsigned int max_channel = s->max_matrix_channel;
if (s->noise_type)
    max_channel += 2; /* Account for noise channels if noise type is set */

/* Read coefficients for each channel */
for (mat = 0; mat < s->num_primitive_matrices; mat++) {
    for (channel = 0; channel <= max_channel; channel++) {
        int coeff_val = get_sbits(gbp, 8);
        s->matrix_coeff[mat][channel] = coeff_val * (1 << s->matrix_frac_bits[mat]);
    }
}

/* Read matrix noise shift if needed */
if (s->noise_type) {
    s->matrix_noise_shift = get_bits(gbp, 4);
} else {
    s->matrix_noise_shift = 0;
}

return 0;

error:
    /* Clean SubStream state in case of error */
    s->num_primitive_matrices = 0;
    memset(s->matrix_out_ch, 0, sizeof(s->matrix_out_ch));
    memset(s->matrix_frac_bits, 0, sizeof(s->matrix_frac_bits));
    memset(s->matrix_bypass_lsbs, 0, sizeof(s->matrix_bypass_lsbs));
    memset(s->matrix_coeff, 0, sizeof(s->matrix_coeff));
    return AVERROR_INVALIDDATA;
