if (s->num_primitive_matrices > max_primitive_matrices) {
        av_log(m->avctx, AV_LOG_ERROR, "Number of primitive matrices %d exceeds maximum allowed %d.\n",
               s->num_primitive_matrices, max_primitive_matrices);
        return AVERROR_INVALIDDATA;
    }

    s->primitive_matrix = av_malloc_array(s->num_primitive_matrices, sizeof(*s->primitive_matrix));
    if (!s->primitive_matrix) {
        av_log(m->avctx, AV_LOG_ERROR, "Failed to allocate memory for primitive matrices.\n");
        return AVERROR(ENOMEM);
    }

    for (mat = 0; mat < s->num_primitive_matrices; mat++) {
        s->primitive_matrix[mat].output_channel = get_bits(gbp, 4);
        if (s->primitive_matrix[mat].output_channel > m->max_matrix_channel) {
            av_log(m->avctx, AV_LOG_ERROR, "Output channel %d exceeds maximum matrix channel %d.\n",
                   s->primitive_matrix[mat].output_channel, m->max_matrix_channel);
            av_freep(&s->primitive_matrix);
            return AVERROR_INVALIDDATA;
        }

        s->primitive_matrix[mat].fractional_bits = get_bits(gbp, 4);
        if (s->primitive_matrix[mat].fractional_bits > 14) {
            av_log(m->avctx, AV_LOG_ERROR, "Fractional bits %d exceed maximum allowed 14.\n",
                   s->primitive_matrix[mat].fractional_bits);
            av_freep(&s->primitive_matrix);
            return AVERROR_INVALIDDATA;
        }

        s->primitive_matrix[mat].lsb_bypass = get_bits1(gbp);

        for (channel = 0; channel < s->primitive_matrix[mat].output_channel; channel++) {
            s->primitive_matrix[mat].coeff[channel] = get_bits(gbp, 16);
            s->primitive_matrix[mat].coeff[channel] *= (1 << s->primitive_matrix[mat].fractional_bits);
        }

        if (get_bits1(gbp)) {
            s->primitive_matrix[mat].matrix_noise_shift = get_bits(gbp, 4);
        } else {
            s->primitive_matrix[mat].matrix_noise_shift = 0;
        }
    }

    for (mat = 0; mat < s->num_primitive_matrices; mat++) {
        for (channel = 0; channel < s->primitive_matrix[mat].output_channel; channel++) {
            if ((ret = read_channel_params(m, substr, gbp, mat * s->max_matrix_channel + channel)) < 0) {
                av_freep(&s->primitive_matrix);
                return ret;
            }
        }
    }

    if (m->noise_type_set) {
        s->max_matrix_channel++;
        s->max_matrix_channel++;
    }

    for (mat = 0; mat < s->num_primitive_matrices; mat++) {
        for (channel = 0; channel < s->primitive_matrix[mat].output_channel; channel++) {
            if ((ret = read_channel_params(m, substr, gbp, mat * s->max_matrix_channel + channel)) < 0) {
                av_freep(&s->primitive_matrix);
                return ret;
            }
        }
    }

    if (m->noise_type_set) {
        generate_2_noise_channels(m, substr);
    }
