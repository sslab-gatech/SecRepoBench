if (s->num_primitive_matrices > max_primitive_matrices) {
        av_log(m->avctx, AV_LOG_ERROR, "Number of primitive matrices exceeds maximum allowed.\n");
        return AVERROR_INVALIDDATA;
    }

    s->primitive_matrix = av_malloc_array(s->num_primitive_matrices, sizeof(*s->primitive_matrix));
    if (!s->primitive_matrix) {
        av_log(m->avctx, AV_LOG_ERROR, "Failed to allocate memory for primitive matrices.\n");
        return AVERROR(ENOMEM);
    }

    for (mat = 0; mat < s->num_primitive_matrices; mat++) {
        s->primitive_matrix[mat].output_channel = get_bits(gbp, 4);
        if (s->primitive_matrix[mat].output_channel > m->max_matrix_channel) {
            av_log(m->avctx, AV_LOG_ERROR, "Output channel exceeds maximum matrix channel.\n");
            av_freep(&s->primitive_matrix);
            return AVERROR_INVALIDDATA;
        }

        s->primitive_matrix[mat].fractional_bits = get_bits(gbp, 4);
        if (s->primitive_matrix[mat].fractional_bits > 14) {
            av_log(m->avctx, AV_LOG_ERROR, "Fractional bits exceed maximum allowed.\n");
            av_freep(&s->primitive_matrix);
            return AVERROR_INVALIDDATA;
        }

        s->primitive_matrix[mat].bypass_lsb = get_bits1(gbp);

        for (channel = 0; channel < s->primitive_matrix[mat].output_channel; channel++) {
            int32_t coeff_value = get_bits(gbp, 16);
            int32_t coeff_shift = s->primitive_matrix[mat].fractional_bits;
            s->primitive_matrix[mat].coeff_values[channel] = coeff_value << coeff_shift;
        }

        if (get_bits1(gbp)) {
            s->primitive_matrix[mat].matrix_noise_shift = get_bits(gbp, 4);
        } else {
            s->primitive_matrix[mat].matrix_noise_shift = 0;
        }
    }

    for (mat = 0; mat < s->num_primitive_matrices; mat++) {
        for (channel = 0; channel < s->primitive_matrix[mat].output_channel; channel++) {
            if ((ret = read_channel_params(m, substr, gbp, channel)) < 0) {
                av_freep(&s->primitive_matrix);
                return ret;
            }
        }
    }
