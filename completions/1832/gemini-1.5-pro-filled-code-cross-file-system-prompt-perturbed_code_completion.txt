if (s->num_primitive_matrices == 0 || s->num_primitive_matrices > max_primitive_matrices) {
        av_log(m->avctx, AV_LOG_ERROR, "Invalid number of primitive matrices: %d\n",
               s->num_primitive_matrices);
        return AVERROR_INVALIDDATA;
    }

    for (mat = 0; mat < s->num_primitive_matrices; mat++) {
        s->matrix_out_ch[mat]    = get_bits(gbp, 4);
        s->matrix_frac_bits[mat] = get_bits(gbp, 4);
        s->matrix_bypass[mat] = get_bits1(gbp);

        if (s->matrix_out_ch[mat] >= m->max_matrix_channel) {
            av_log(m->avctx, AV_LOG_ERROR, "Invalid matrix output channel.\n");
            goto cleanup;
        }
        if (s->matrix_frac_bits[mat] > 14) {
            av_log(m->avctx, AV_LOG_ERROR, "Invalid matrix fractional bits.\n");
            goto cleanup;
        }
    }

    if (!m->noise_type)
        m->max_matrix_channel += 2; // Account for generated noise channels

    for (channel = 0; channel < m->max_matrix_channel; channel++) {
        for (mat = 0; mat < s->num_primitive_matrices; mat++) {
            int32_t coeff = get_sbits(gbp, 14);
            s->matrix_coeff[channel][mat] = coeff * (1 << s->matrix_frac_bits[mat]);
        }
    }

    if (m->noise_type) {
        s->noise_shift = get_bits(gbp, 4);
    } else {
        s->noise_shift = 0;
    }

    return 0;

cleanup:
    memset(s, 0, sizeof(*s)); // Clean SubStream state
    return AVERROR_INVALIDDATA;
