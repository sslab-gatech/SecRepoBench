if (s->num_primitive_matrices > max_primitive_matrices) {
        av_log(m->avctx, AV_LOG_ERROR, "Number of primitive matrices cannot exceed %d.\n",
               max_primitive_matrices);
        s->num_primitive_matrices = 0;
        return AVERROR_INVALIDDATA;
    }

    for (mat = 0; mat < s->num_primitive_matrices; mat++) {
        unsigned int outch = get_bits(gbp, 4);
        unsigned int fracbits = get_bits(gbp, 4);
        unsigned int lsb_bypass = get_bits1(gbp);
        unsigned int max_channel = s->max_matrix_channel;

        if (!s->noise_type)
            max_channel += 2;

        if (outch > max_channel) {
            av_log(m->avctx, AV_LOG_ERROR, "Invalid matrix output channel %u.\n", outch);
            s->num_primitive_matrices = 0;
            return AVERROR_INVALIDDATA;
        }

        if (fracbits > 14) {
            av_log(m->avctx, AV_LOG_ERROR, "Too many fractional bits (%u).\n", fracbits);
            s->num_primitive_matrices = 0;
            return AVERROR_INVALIDDATA;
        }

        for (channel = 0; channel <= s->max_matrix_channel; channel++) {
            int coeff = get_sbits(gbp, 8);
            s->matrix_out_ch[mat][channel] = coeff * (1 << fracbits);
        }

        if (s->noise_type)
            s->matrix_noise_shift[mat] = get_bits(gbp, 4);
        else
            s->matrix_noise_shift[mat] = 0;
    }

    return 0;
