if (s->num_primitive_matrices > max_primitive_matrices) {
        av_log(m->avctx, AV_LOG_ERROR, "Too many primitive matrices: %d\n", s->num_primitive_matrices);
        return AVERROR_INVALIDDATA;
    }

    for (mat = 0; mat < s->num_primitive_matrices; mat++) {
        s->matrix_out_ch[mat] = get_bits(gbp, 4);
        s->matrix_out_ch_fraction[mat] = get_bits(gbp, 4);
        s->matrix_out_ch_lsbs[mat] = get_bits1(gbp);

        if (s->matrix_out_ch[mat] > s->max_matrix_channel) {
            av_log(m->avctx, AV_LOG_ERROR, "Output channel number too high: %d\n", s->matrix_out_ch[mat]);
            return AVERROR_INVALIDDATA;
        }

        if (s->matrix_out_ch_fraction[mat] > 14) {
            av_log(m->avctx, AV_LOG_ERROR, "Fractional bits too high: %d\n", s->matrix_out_ch_fraction[mat]);
            return AVERROR_INVALIDDATA;
        }
    }

    if (!m->noise_type) {
        s->max_matrix_channel += m->noise_channels;
    }

    for (channel = 0; channel <= s->max_matrix_channel; channel++) {
        for (mat = 0; mat < s->num_primitive_matrices; mat++) {
            int32_t coeff = get_sbits(gbp, 9);
            s->matrix_coeff[mat][channel] = coeff * (1 << s->matrix_out_ch_fraction[mat]);
        }
    }

    if (m->noise_type) {
        s->noise_shift = get_bits(gbp, 4);
    } else {
        s->noise_shift = 0;
    }

    return 0;
