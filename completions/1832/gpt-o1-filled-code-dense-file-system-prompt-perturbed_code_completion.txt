if (s->num_primitive_matrices > max_primitive_matrices) {
        av_log(m->avctx, AV_LOG_ERROR,
               "Number of primitive matrices %u exceeds maximum %d.\n",
               s->num_primitive_matrices, max_primitive_matrices);
        goto fail;
    }

    /* If the noise type is not set, adjust the max channel to account for noise. */
    if (!m->noise_type)
        s->max_matrix_channel++;

    for (mat = 0; mat < s->num_primitive_matrices; mat++) {
        unsigned int out_ch     = get_bits(gbp, 4);
        unsigned int frac_bits  = get_bits(gbp, 4);
        unsigned int lsb_bypass = get_bits1(gbp);

        if (out_ch > s->max_matrix_channel) {
            av_log(m->avctx, AV_LOG_ERROR,
                   "Matrix output channel %u invalid (max %u).\n",
                   out_ch, s->max_matrix_channel);
            goto fail;
        }
        if (frac_bits > 14) {
            av_log(m->avctx, AV_LOG_ERROR,
                   "Fractional bits %u exceed allowed maximum of 14.\n",
                   frac_bits);
            goto fail;
        }

        for (channel = 0; channel <= s->max_matrix_channel; channel++) {
            if (get_bits1(gbp)) {
                int coeff = get_sbits(gbp, frac_bits + 2);
                coeff <<= (14 - frac_bits);
                s->matrix_coeff[mat][channel] = coeff;
            } else {
                s->matrix_coeff[mat][channel] = 0;
            }
        }

        /* If noise type is set, read noise shift; otherwise set it to zero. */
        s->matrix_noise_shift[mat] = m->noise_type ? get_bits(gbp, 4) : 0;
    }

    return 0;
fail:
    memset(s->matrix_coeff, 0, sizeof(s->matrix_coeff));
    s->num_primitive_matrices = 0;
    return AVERROR_INVALIDDATA;
