if (s->num_primitive_matrices > max_primitive_matrices) {
        av_log(m->avctx, AV_LOG_ERROR,
               "Too many primitive matrices: %d > %d.\n",
               s->num_primitive_matrices, max_primitive_matrices);
        s->num_primitive_matrices = 0;
        return AVERROR_INVALIDDATA;
    }

    for (mat = 0; mat < s->num_primitive_matrices; mat++) {
        s->matrix_out_ch[mat] = get_bits(gbp, 4);
        if (s->matrix_out_ch[mat] > s->max_matrix_channel) {
            av_log(m->avctx, AV_LOG_ERROR,
                   "Matrix output channel is greater than max matrix channel.\n");
            goto cleanup;
        }

        s->lsb_bypass[mat] = get_bits1(gbp);

        skip_bits(gbp, 3); // fractional_bits
    }

    if (!s->noise_type)
        max_channel += 2; // add two noise channels

    for (channel = s->min_channel; channel <= max_channel; channel++) {
        for (mat = 0; mat < s->num_primitive_matrices; mat++) {
            int coeff_val = get_sbits(gbp, 14); // matrix_coeff_bits = 14
            s->matrix_coeff[mat][channel] = coeff_val;
        }
    }

    if (s->noise_type) {
        for (mat = 0; mat < s->num_primitive_matrices; mat++) {
            s->matrix_noise_shift[mat] = get_bits(gbp, 4);
        }
    } else {
        memset(s->matrix_noise_shift, 0, sizeof(s->matrix_noise_shift));
    }

    return 0;

cleanup:
    s->num_primitive_matrices = 0;
    return AVERROR_INVALIDDATA;
