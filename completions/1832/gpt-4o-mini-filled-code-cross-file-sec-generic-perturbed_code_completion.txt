if (s->num_primitive_matrices > max_primitive_matrices) {
        av_log(m->avctx, AV_LOG_ERROR, "Number of primitive matrices exceeds maximum allowed.\n");
        return AVERROR_INVALIDDATA;
    }

    for (mat = 0; mat < s->num_primitive_matrices; mat++) {
        channel = get_bits(gbp, 5); // Read output channel
        if (channel >= MAX_CHANNELS) {
            av_log(m->avctx, AV_LOG_ERROR, "Output channel %d exceeds maximum channels %d.\n", channel, MAX_CHANNELS);
            return AVERROR_INVALIDDATA;
        }

        int fractional_bits = get_bits(gbp, 4); // Read fractional bits
        if (fractional_bits > 14) {
            av_log(m->avctx, AV_LOG_ERROR, "Fractional bits %d exceed maximum of 14.\n", fractional_bits);
            return AVERROR_INVALIDDATA;
        }

        // Check if LSBs should be bypassed
        int bypass_lsb = get_bits1(gbp);
        
        // Adjust max channel if noise type is not set
        if (!(s->param_presence_flags & PARAM_NOISE_TYPE)) {
            s->max_matrix_channel = channel + 1; // Adjust for additional noise channels
        }

        // Read coefficient value
        int32_t coefficient = get_sbits(gbp, fractional_bits);
        s->channel_params[channel].coeff[0] = coefficient; // Store coefficient

        // Read noise shift value
        if (s->param_presence_flags & PARAM_NOISE_TYPE) {
            s->channel_params[channel].noise_shift = get_bits(gbp, 4);
        } else {
            s->channel_params[channel].noise_shift = 0; // Set to zero if noise type is not set
        }
    }

    return 0;
