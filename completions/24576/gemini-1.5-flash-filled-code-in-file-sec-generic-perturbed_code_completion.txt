case DLT_PPP_SERIAL:
    ethertype = ETH_P_IP;
    ip_offset = 2 + eth_offset;
    break;
  case DLT_PPP_HDLC:
    chdlc = (struct ndpi_chdlc *)&packet[eth_offset];
    if(chdlc->address == 0xFF && chdlc->control == 0x03) {
      ethertype = ntohs(chdlc->protocol);
      ip_offset = 4 + eth_offset;
    } else {
      goto v4_warning;
    }
    break;
